1. read a as string  
2. replace all 'dot' in a with '.'  
3. replace all 'at' in a with '@'  
4. if a[0] = '.': a ← 'dot' + a[1:]  
5. set co ← 0  
6. let c be empty list  
7. if a[0] = '@': a ← 'at' + a[1:]  
8. for each char i in a:  
 if i = '@':  
  if co > 0: append 'at' to c; co ← 1  
  else: append '@' to c; co ← 1  
 else: append i to c  
9. set c ← join elements of c into string  
10. if c[-1] = '.': c ← c[:-1] + 'dot'  
11. print c
