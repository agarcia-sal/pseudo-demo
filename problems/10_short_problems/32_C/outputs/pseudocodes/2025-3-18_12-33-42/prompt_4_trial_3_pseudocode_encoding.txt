# Define a function to calculate the adjusted number based on inputs
FUNCTION calculate_adjusted_value(total, divisor):
    # Divide total by divisor and get quotient and remainder
    quotient, remainder = DIVIDE(total, divisor)
    
    # If there is a remainder, return remainder multiplied by (quotient + 1)
    IF remainder > 0 THEN
        RETURN remainder * (quotient + 1)
    ELSE
        RETURN total

# Read input values for n, m, and s
INPUT n, m, s

# Calculate the final result by multiplying the adjusted values of n and m
RESULT = calculate_adjusted_value(n, s) * calculate_adjusted_value(m, s)

# Output the result
PRINT RESULT
