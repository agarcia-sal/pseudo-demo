FUNCTION calculate_value(n, s):
    // Determine how many complete groups fit into n and the remainder
    q, r = DIVIDE n BY s
    // If there is a remainder, return adjusted value; otherwise return n
    IF r > 0 THEN
        RETURN r * (q + 1)
    ELSE
        RETURN n

// Read input values for n, m, s from user
INPUT n, m, s

// Calculate the results for n and m using the calculate_value function and multiply them
OUTPUT calculate_value(n, s) * calculate_value(m, s)
