```
BEGIN

    // Read a line of input and remove any surrounding whitespace
    Read inputLine from standard input
    Remove spaces from the inputLine

    // Replace occurrences of 'dot' with '.' and 'at' with '@'
    Replace 'dot' with '.' in inputLine
    Replace 'at' with '@' in inputLine

    // If the modified input starts with a '.', prepend 'dot' to it
    IF inputLine starts with '.' THEN
        Prepend 'dot' to inputLine

    // Initialize a count for consecutive '@' symbols
    Initialize consecutiveAtCount to 0
    Initialize an empty list for characters named outputCharacters

    // If the modified input starts with '@', prepend 'at' to it
    IF inputLine starts with '@' THEN
        Prepend 'at' to inputLine

    // Iterate through each character in the modified input
    FOR each character in inputLine DO
        
        // Check if the character is '@'
        IF character is '@' THEN
            // If it is the second consecutive '@', append 'at' to the list
            IF consecutiveAtCount > 0 THEN
                Append 'at' to outputCharacters
                Set consecutiveAtCount to 1   // Count consecutive '@'
            ELSE
                Append '@' to outputCharacters
                Set consecutiveAtCount to 1   // Count first '@'
            END IF
        ELSE
            // If it is not '@', append the character to the list
            Append character to outputCharacters
        END IF
    END FOR

    // Join all characters in the outputCharacters list into a string
    Join characters in outputCharacters into modifiedText

    // If the last character in the modified text is '.', replace it with 'dot'
    IF modifiedText ends with '.' THEN
        Replace the last character with 'dot' in modifiedText
    END IF

    // Print the final modified text
    Output modifiedText

END
```
