BEGIN
    // Read input from the user
    SET inputString TO read input line from standard input
    REMOVE surrounding whitespace from inputString

    // Replace specified words with symbols
    REPLACE all occurrences of "dot" with "."
    REPLACE all occurrences of "at" with "@"

    // Ensure that the string starts with "dot" if it begins with '.'
    IF the first character of inputString is '.' THEN
        SET inputString TO "dot" concatenated with the substring of inputString from index 1

    // Initialize variables for processing
    SET consecutiveAtCount TO 0
    CREATE empty list called resultCharacters
    SET lengthCounter TO 0 (not used in this pseudocode)

    // If the string starts with '@', prefix it with "at"
    IF the first character of inputString is '@' THEN
        SET inputString TO "at" concatenated with the substring of inputString from index 1

    // Iterate through each character in the inputString
    FOR each character in inputString DO
        IF character is '@' THEN
            IF consecutiveAtCount is greater than 0 THEN
                APPEND "at" to resultCharacters
                SET consecutiveAtCount TO 1
            ELSE
                APPEND "@" to resultCharacters
                SET consecutiveAtCount TO 1
            END IF
        ELSE
            APPEND character to resultCharacters
        END IF
    END FOR

    // Join the characters back into a single string
    SET outputString TO join all characters in resultCharacters into a single string

    // If the last character of outputString is '.', replace it with 'dot'
    IF the last character of outputString is '.' THEN
        SET outputString TO the substring of outputString up to the last character concatenated with "dot"
    END IF

    // Display the final output
    PRINT outputString
END
