```pseudocode
BEGIN

    // Read input from standard input
    INPUT userInput
    userInput = REMOVE whitespace from userInput
    
    // Replace words with symbols
    userInput = REPLACE 'dot' with '.' in userInput
    userInput = REPLACE 'at' with '@' in userInput
    
    // Ensure input does not start with a '.'
    IF first character of userInput is '.' THEN
        userInput = 'dot' + remaining characters of userInput
    ENDIF
    
    // Initialize variables
    occurrenceCounter = 0
    modifiedCharacters = []
    
    // Ensure input does not start with an '@'
    IF first character of userInput is '@' THEN
        userInput = 'at' + remaining characters of userInput
    ENDIF
    
    // Process each character in the userInput
    FOR each character in userInput DO
        IF character is '@' THEN
            IF occurrenceCounter > 0 THEN
                ADD 'at' to modifiedCharacters
                occurrenceCounter = 1
            ELSE
                ADD '@' to modifiedCharacters
                occurrenceCounter = 1
            ENDIF
        ELSE
            ADD character to modifiedCharacters
        ENDIF
    ENDFOR

    // Join modified characters into a single string
    modifiedOutput = JOIN modifiedCharacters into a string
    
    // Ensure output does not end with a '.'
    IF last character of modifiedOutput is '.' THEN
        modifiedOutput = REMOVE last character from modifiedOutput + 'dot'
    ENDIF

    // Print the final modified output
    PRINT modifiedOutput

END
```
