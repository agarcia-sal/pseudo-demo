FUNCTION processEmail(inputString):
    // Read the input string and remove any surrounding whitespace
    cleanedInput = TRIM(inputString)
    
    // Replace instances of 'dot' with '.' and 'at' with '@'
    cleanedInput = REPLACE(cleanedInput, "dot", ".")
    cleanedInput = REPLACE(cleanedInput, "at", "@")
    
    // Check if the first character is a '.' and prepend 'dot' if it is
    IF cleanedInput[0] IS '.':
        cleanedInput = "dot" + cleanedInput[1:]

    // Initialize a counter and an empty list for the final characters
    countAtSymbols = 0
    finalCharacters = []
    
    // Check if the first character is an '@' and prepend 'at' if it is
    IF cleanedInput[0] IS '@':
        cleanedInput = "at" + cleanedInput[1:]

    // Iterate over each character in the cleaned input
    FOR EACH character IN cleanedInput:
        IF character IS '@':
            // If we've already encountered an '@', append 'at' instead of '@'
            IF countAtSymbols > 0:
                APPEND(finalCharacters, "at")
                countAtSymbols = 1
            ELSE:
                APPEND(finalCharacters, '@')
                countAtSymbols = 1
        ELSE:
            APPEND(finalCharacters, character)

    // Join the characters into a single string
    finalString = JOIN(finalCharacters)

    // If the last character is a '.', replace it with 'dot'
    IF finalString[-1] IS '.':
        finalString = REMOVE_LAST_CHARACTER(finalString) + "dot"

    // Output the final processed string
    PRINT(finalString)
