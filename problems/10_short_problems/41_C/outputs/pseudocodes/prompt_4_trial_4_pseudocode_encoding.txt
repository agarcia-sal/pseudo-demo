```plaintext
BEGIN

// Read input string from the user
INPUT string a

// Replace occurrences in the string for better readability
REPLACE 'dot' in a with '.'
REPLACE 'at' in a with '@'

// Ensure the string starts with 'dot' if it begins with a dot
IF a starts with '.' THEN
    a = 'dot' + substring of a from index 1 onwards
END IF

// Initialize variables
SET count to 0
SET resultList to empty list
SET length to 0

// Ensure the string starts with 'at' if it begins with an '@'
IF a starts with '@' THEN
    a = 'at' + substring of a from index 1 onwards
END IF

// Iterate through each character in the modified string
FOR each character i in a DO
    IF i is '@' THEN
        // If the previous character was also an '@', add 'at'
        IF count > 0 THEN
            APPEND 'at' to resultList
            SET count to 1 
        ELSE
            APPEND '@' to resultList
            SET count to 1
        END IF
    ELSE
        APPEND i to resultList
    END IF
END FOR

// Join the result list into a single string
SET result to join resultList into a string

// Replace trailing dot with 'dot'
IF result ends with '.' THEN
    result = substring of result from index 0 to length-1 + 'dot'
END IF

// Output the final modified string
OUTPUT result

END
```
