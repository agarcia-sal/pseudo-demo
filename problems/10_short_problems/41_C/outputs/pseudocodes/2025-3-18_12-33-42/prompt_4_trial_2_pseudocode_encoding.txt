```
BEGIN 

    // Read input from user
    READ inputLine
    
    // Replace occurrences of 'dot' with '.' and 'at' with '@' in the input
    REPLACE 'dot' with '.' IN inputLine
    REPLACE 'at' with '@' IN inputLine

    // If the first character is '.', prepend 'dot'
    IF inputLine[0] EQUALS '.' THEN
        SET inputLine EQUALS 'dot' + inputLine[1:]
    ENDIF

    // Initialize counter and list for output characters
    SET counter EQUALS 0
    SET outputList EQUALS empty list

    // If the first character is '@', prepend 'at'
    IF inputLine[0] EQUALS '@' THEN
        SET inputLine EQUALS 'at' + inputLine[1:]
    ENDIF

    // Process each character in the modified input
    FOR EACH character IN inputLine DO
        IF character EQUALS '@' THEN
            // If '@' has already been added, append 'at' to output list
            IF counter GREATER THAN 0 THEN
                APPEND 'at' TO outputList
                SET counter EQUALS 1
            ELSE
                APPEND '@' TO outputList
                SET counter EQUALS 1
            ENDIF
        ELSE 
            // Add the character to the output list
            APPEND character TO outputList
        ENDIF
    ENDFOR

    // Join the output list into a single string
    SET outputString EQUALS JOIN outputList

    // If the last character is '.', replace it with 'dot'
    IF outputString[-1] EQUALS '.' THEN
        SET outputString EQUALS REMOVE LAST CHARACTER from outputString + 'dot'
    ENDIF

    // Print the final processed string
    PRINT outputString

END
```
