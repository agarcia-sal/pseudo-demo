// Read input from the user and strip any surrounding whitespace
Read input line as 'userInput'

// Replace 'dot' with '.' in the input
Replace 'dot' with '.' in userInput

// Replace 'at' with '@' in the input
Replace 'at' with '@' in userInput

// If the first character is '.', add 'dot' at the beginning
If the first character of userInput is '.' then
    Prepend 'dot' to userInput (remove the first character)

// Initialize a variable to count occurrences of '@'
Set occurrenceCount to 0

// Initialize a list to store processed characters
Initialize an empty list 'processedCharacters'

// Process the input string character by character
For each character 'currentChar' in userInput do
    If currentChar is '@' then
        If occurrenceCount is greater than 0 then
            Append 'at' to processedCharacters
            Set occurrenceCount to 1
        Else
            Append '@' to processedCharacters
            Set occurrenceCount to 1
    Else // For characters other than '@'
        Append currentChar to processedCharacters

// Join the list of processed characters into a string
Join processedCharacters into a string 'finalOutput'

// If the last character is '.', replace it with 'dot'
If the last character of finalOutput is '.' then
    Remove the last character and append 'dot'

// Print the final transformed output
Print finalOutput
