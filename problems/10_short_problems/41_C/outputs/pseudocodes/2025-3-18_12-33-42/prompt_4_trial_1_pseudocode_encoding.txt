    // Read input from the standard input
    input_string = READ line from standard input
    input_string = REMOVE whitespace from both ends of input_string

    input_string = REPLACE 'dot' with '.' in input_string
    input_string = REPLACE 'at' with '@' in input_string

    // If the first character is a '.', prepend 'dot'
    IF input_string[0] == '.' THEN
        input_string = 'dot' + input_string FROM index 1 TO END
    END IF

    // Initialize variables for processing
    count_at = 0
    result_array = empty list
    length = 0

    // If the first character is an '@', prepend 'at'
    IF input_string[0] == '@' THEN
        input_string = 'at' + input_string FROM index 1 TO END
    END IF

    // Iterate through each character in the modified input string
    FOR each character in input_string DO
        IF character == '@' THEN
            // Check if we have already added an '@'
            IF count_at > 0 THEN
                ADD 'at' to result_array
                count_at = 1
            ELSE
                ADD '@' to result_array
                count_at = 1
            END IF
        ELSE
            // Add non-@ character to result_array
            ADD character to result_array
        END IF
    END FOR

    // Join the result_array into a single string
    final_string = JOIN result_array into a string

    // If the last character is a '.', replace it with 'dot'
    IF final_string[-1] == '.' THEN
        final_string = REMOVE last character from final_string
        final_string = final_string + 'dot'
    END IF

    // Output the final processed string
    PRINT final_string

END
