BEGIN

    // Input two strings while removing spaces
    INITIALIZE firstString AS input string without spaces
    INITIALIZE secondString AS input string without spaces

    // Create a list to store frequency differences of character counts
    INITIALIZE frequencyDifferences AS empty list

    // For each character in the ASCII range from 'A' to 'z'
    FOR EACH characterCode FROM ASCII value of 'A' TO ASCII value of 'z' DO
        // Count occurrences of the character in both strings
        INITIALIZE countInFirstString AS number of occurrences of character in firstString

        // Calculate the difference in frequencies and store it
        APPEND (countInFirstString - countInSecondString) TO frequencyDifferences
    END FOR

    // Check if any frequency differences are negative
    INITIALIZE negativeCount AS count of elements in frequencyDifferences that are less than 0

    // Print "YES" if there are no negative differences, otherwise print "NO"
    IF negativeCount EQUALS 0 THEN
        PRINT "YES"
    ELSE
        PRINT "NO"
    END IF

END
