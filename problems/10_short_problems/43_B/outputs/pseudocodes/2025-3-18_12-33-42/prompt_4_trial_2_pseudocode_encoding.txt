```plaintext
// Read two strings from user input, remove spaces, and store them as s1 and s2
s1 = remove spaces from input string 1
s2 = remove spaces from input string 2

// Initialize a list to hold the frequency differences
frequencyDifferences = []

// For each character from 'A' to 'z'
FOR each ASCII value from 'A' (65) to 'z' (122) DO
    // Calculate the frequency of the character in both strings
    frequencyInS1 = count occurrences of character in s1
    frequencyInS2 = count occurrences of character in s2
    
    // Calculate the difference in frequencies
    difference = frequencyInS1 - frequencyInS2
    
    // Store the difference in the list
    APPEND difference to frequencyDifferences
END FOR

// Check if all frequency differences are non-negative
IF there are no negative differences in frequencyDifferences THEN
    PRINT "YES"
ELSE
    PRINT "NO"
END IF
```
