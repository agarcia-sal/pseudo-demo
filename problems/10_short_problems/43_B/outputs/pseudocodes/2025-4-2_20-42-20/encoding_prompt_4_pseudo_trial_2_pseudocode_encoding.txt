FUNCTION check_character_frequencies
    // Read two lines of input
    LINE_1 = read input
    LINE_2 = read input

    // Remove spaces and store characters in lists
    CHAR_LIST_1 = list of characters from LINE_1 excluding spaces
    CHAR_LIST_2 = list of characters from LINE_2 excluding spaces

    // Initialize a list to store frequency differences
    FREQUENCY_DIFFERENCES = empty list

    // Calculate frequency difference for each character from 'A' to 'z'
    FOR each character code from 'A' (65) to 'z' (122) DO
        // Get character from code
        CHARACTER = character corresponding to current character code
        
        // Count occurrences in both character lists
        COUNT_IN_LIST_1 = count of CHARACTER in CHAR_LIST_1
        COUNT_IN_LIST_2 = count of CHARACTER in CHAR_LIST_2
        
        // Append the difference to the frequency list
        APPEND (COUNT_IN_LIST_1 - COUNT_IN_LIST_2) to FREQUENCY_DIFFERENCES

    // Check if all frequency differences are non-negative
    IF there are no negative values in FREQUENCY_DIFFERENCES THEN
        PRINT "YES"
    ELSE
        PRINT "NO"
        
END FUNCTION
