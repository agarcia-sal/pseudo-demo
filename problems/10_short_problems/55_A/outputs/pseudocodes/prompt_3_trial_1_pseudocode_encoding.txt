BEGIN

    // Receive input for the number of elements
    SET numberOfElements TO integer value from user input

    // Initialize a list of boolean values indicating the status of each element
    CREATE list statusList WITH size numberOfElements initialized to True

    // Initialize counters for the loop
    SET currentIndex TO 0
    SET iterationCount TO 1

    // Continue looping while the iterationCount is less than or equal to 500000
    WHILE iterationCount <= 500000 DO

        // If the current element in the statusList is True
        IF statusList[currentIndex] IS True THEN
            
            // Mark the current element as False
            SET statusList[currentIndex] TO False
        
        ENDIF
        
        // Move to the next iteration count
        INCREMENT iterationCount BY 1
        
        // Calculate the next index to modify in a circular manner
        SET currentIndex TO (currentIndex + iterationCount) MOD numberOfElements

    ENDWHILE

    // Create a new list containing only the True values from the statusList
    CREATE list remainingTrueElements AS list of elements from statusList WHERE element IS True

    // Check if there are any True values left in the remainingTrueElements list
    IF length of remainingTrueElements IS 0 THEN
        
        // If no True values are found, output 'YES'
        PRINT 'YES'
    
    ELSE
        
        // If True values are found, output 'NO'
        PRINT 'NO'
    
    ENDIF

END
