While (i <= 500000):
    // If the current element in 'b' at index 'j' is True
    If (b[j] is True):
        // Set the element to False
        b[j] = False
    
    // Increment 'i' by 1
    i = i + 1
    
    // Update 'j' to the next index, wrapping around with modulo 'n'
    j = (j + i) Mod n

// Create a new list 'x' with elements from 'b' that are still True
x = FilterList(b, Condition=True)

// If the length of 'x' is 0

    // Print 'YES' indicating all elements were set to False
    Print('YES')
