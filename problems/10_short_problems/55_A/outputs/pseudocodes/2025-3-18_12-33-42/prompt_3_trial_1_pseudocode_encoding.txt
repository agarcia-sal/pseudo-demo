```plaintext
// Read an integer input for the size of the list
SET numberOfElements to READ integer input

// Create a boolean list initialized to True for each index from 0 to numberOfElements-1
INITIALIZE booleanList as a list of size numberOfElements filled with True

// Initialize an index variable for processing
SET index to 0
SET increment to 1

// Loop while increment is less than or equal to 500000
WHILE increment is less than or equal to 500000 DO
    // If the current boolean value at index is True
    IF booleanList[index] is True THEN
        // Set that boolean value to False
        SET booleanList[index] to False
    END IF

    // Increment the counter
    INCREMENT increment by 1

    // Calculate the next index using modular arithmetic
    SET index to (index + increment) MOD numberOfElements
END WHILE

// Create a new list that contains only the True values from the booleanList
SET remainingTrueValues to a new list containing items from booleanList where item is True

// Check if there are no remaining True values
IF the length of remainingTrueValues is 0 THEN
    // Output "YES" if there are no True values
    PRINT 'YES'
ELSE
    // Output "NO" if there are still True values
    PRINT 'NO'
END IF
```
