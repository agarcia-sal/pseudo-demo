BEGIN

    // Read input value for n
    SET n TO input as an integer

    // Initialize a list of boolean values, all set to True
    CREATE list b of size n, initialized to True

    SET j TO 0
    SET i TO 1

    // Loop until i exceeds 500000
    WHILE (i <= 500000) DO
        // If the current position in the list is True
        IF (b[j] is True) THEN
            // Mark the position as False
            SET b[j] TO False
        END IF
        
        // Increment i
        INCREMENT i BY 1
        
        // Update the index j in a circular manner
        SET j TO (j + i) MOD n
    END WHILE

    // Create a list x of True values from b
    CREATE list x containing all elements from b that are True

    // Check if there are any True values remaining
    IF (length of x is 0) THEN
        PRINT 'YES'
    ELSE
        PRINT 'NO'
    END IF

END
