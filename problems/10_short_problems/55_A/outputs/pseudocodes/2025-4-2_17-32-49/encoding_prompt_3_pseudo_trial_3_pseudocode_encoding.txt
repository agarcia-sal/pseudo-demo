BEGIN

    // Step 1: Read input value for `n`
    SET n TO integer value from user input

    // Step 2: Initialize a list `b` with `n` True values
    CREATE list b of size n, initialized to True

    // Step 3: Set initial indices for processing
    SET j TO 0
    SET i TO 1
    
    // Step 4: Loop until `i` exceeds 500,000
    WHILE i is less than or equal to 500,000 DO

        // Step 5: Check if the current position in list `b` is True
        IF b[j] is True THEN
            // Step 6: Mark the current position in list `b` as False
            SET b[j] TO False
        END IF

        // Step 7: Increment the counter `i`
        INCREMENT i by 1

        // Step 8: Update the index `j` for the next iteration
        SET j TO (j + i) MOD n // Update index circularly

    END WHILE

    // Step 9: Create a new list `x` containing only True values from `b`
    CREATE list x containing all elements from b that are True

    // Step 10: Check if `x` is empty or not
    IF length of x is equal to 0 THEN
        OUTPUT 'YES' // Indicating all positions are False
    ELSE
        OUTPUT 'NO' // Indicating there are still True values present
    END IF

END
