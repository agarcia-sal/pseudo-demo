```plaintext
BEGIN

    // Read an integer input for the size of the array
    SET n TO integer input

    // Initialize an array of boolean values all set to True
    CREATE boolean array b of size n, and set all values to True

    // Initialize two counters
    SET j TO 0
    SET i TO 1

    // Loop until i exceeds 500000
    WHILE i <= 500000 DO
        // If the current position j in array b is True
        IF b[j] IS True THEN
            // Set the value at position j in array b to False
            SET b[j] TO False
        ENDIF
        
        // Increment i
        INCREMENT i
        
        // Update j to the new position
        SET j TO (j + i) MOD n
    ENDWHILE

    // Create a new list x with all values in b that are still True
    CREATE list x FROM b WHERE each value is True

    // Check if list x is empty
    IF LENGTH of x IS 0 THEN
        PRINT 'YES'
    ELSE
        PRINT 'NO'
    ENDIF

END
```
