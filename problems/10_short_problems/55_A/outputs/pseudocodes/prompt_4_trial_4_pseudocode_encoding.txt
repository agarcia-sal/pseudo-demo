```
// Initialize a variable n to store user input as an integer
n = get integer input from the user

// Create a list b of size n, filled with True values
b = list of size n filled with True

// Initialize indices j and i
j = 0
i = 1

// Loop while i is less than or equal to 500000
while i <= 500000:
    
    // If the current index j in list b is True
    if b[j] is True:
        // Set the value at index j in list b to False
        b[j] = False
    
    // Increment i by 1
    i = i + 1
    
    // Update j to the next position based on the new value of i, wrapping around with modulo
    j = (j + i) % n

// Create a list x containing all True values from list b
x = list of elements in b that are True

// If list x is empty, print 'YES', otherwise print 'NO'
if length of x is 0:
    print 'YES'
else:
    print 'NO'
```
