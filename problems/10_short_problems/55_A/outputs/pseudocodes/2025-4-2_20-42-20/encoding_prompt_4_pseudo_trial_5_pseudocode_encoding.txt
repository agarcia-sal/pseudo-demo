```plaintext
// INPUT: An integer n from user
n = get user input as integer

// Initialize a list of boolean values to True for the range 0 to n
Create a boolean list 'isActive' of size n, filled with True

// Initialize indices
j = 0  // Current index
i = 1  // Step increment

// Loop until i exceeds 500,000
While i is less than or equal to 500,000 do:
    // If the current position is active
    If isActive[j] is True then:
        // Deactivate current position
        Set isActive[j] to False
    EndIf
    
    // Increment step by 1
    i = i + 1
    
    // Update index j in circular manner
    j = (j + i) mod n
EndWhile

// Create a list of active positions
activePositions = list of indices in isActive where value is True

// Check if there are any active positions
If length of activePositions is 0 then:
    // If no active positions remain
    Output 'YES'
Else:
    // If there are still active positions
    Output 'NO'
EndIf
```
