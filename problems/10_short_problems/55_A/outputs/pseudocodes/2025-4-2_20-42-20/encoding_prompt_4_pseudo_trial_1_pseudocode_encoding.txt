```plaintext
SET n TO integer input from user
CREATE a list b of size n initialized with True values
SET j TO 0
SET i TO 1

WHILE i is less than or equal to 500000 DO
    IF b[j] is True THEN
        SET b[j] to False  // Mark current index as False
    ENDIF
    INCREMENT i by 1
    SET j to (j + i) MOD n  // Update index for the next iteration
ENDWHILE

CREATE a list x containing all values from b that are True

IF length of x is equal to 0 THEN
    PRINT 'YES'  // All elements were marked False
ELSE
    PRINT 'NO'   // There are still True elements in the list
ENDIF
```
