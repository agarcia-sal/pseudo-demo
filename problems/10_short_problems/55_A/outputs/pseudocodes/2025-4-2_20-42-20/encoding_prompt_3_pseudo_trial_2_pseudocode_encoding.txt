BEGIN

    // Read an integer input which defines the size of an array
    SET arraySize TO READ integer input
    
    // Initialize a list of boolean values, indicating active status
    CREATE booleanList WITH all values set to True, of size arraySize
    
    // Initialize counters
    SET indexCounter TO 0
    SET stepIncrement TO 1
    
    // Loop until stepIncrement exceeds 500000
    WHILE (stepIncrement <= 500000) DO
        
        // If the current position in the boolean list is True
        IF (booleanList[indexCounter] IS True) THEN
            
            // Mark this position as False, indicating it's been processed
            SET booleanList[indexCounter] TO False
        
        ENDIF
        
        // Increment stepIncrement after processing
        INCREMENT stepIncrement BY 1
        
        // Update the indexCounter using modulo to stay within bounds of boolean list length
        SET indexCounter TO (indexCounter + stepIncrement) MOD arraySize
    
    ENDWHILE
    
    // Create a new list containing only True values remaining in the boolean list
    SET activeItems TO list containing elements from booleanList where value is True
    
    // Check if there are no True values remaining
    IF (length of activeItems EQUALS 0) THEN
        PRINT "YES"
    ELSE
        PRINT "NO"
    ENDIF

END
