```plaintext
// Read an integer from input representing the number of elements
SET numberOfElements TO integer input()

// Create a boolean array initialized to True
INITIALIZE isAvailable AS array of boolean with size numberOfElements, ALL elements set to True

// Initialize index variables
SET indexDifference TO 0
SET index TO 1

// Loop until the index surpasses 500,000
WHILE (index <= 500000) DO
    // If the current index of isAvailable is True
    IF (isAvailable[indexDifference] IS True) THEN
        // Mark this index as False (not available)
        SET isAvailable[indexDifference] TO False
    END IF
    
    // Increment index
    INCREMENT index BY 1
    
    // Update indexDifference to wrap around using modulus
    SET indexDifference TO (indexDifference + index) MOD numberOfElements
END WHILE

// Create a list of True values from isAvailable
SET availableElements AS list of all elements in isAvailable that are True

// Check if any elements are available
IF (length of availableElements IS 0) THEN
    PRINT 'YES' // All elements marked as not available
ELSE
    PRINT 'NO'
END IF
```
