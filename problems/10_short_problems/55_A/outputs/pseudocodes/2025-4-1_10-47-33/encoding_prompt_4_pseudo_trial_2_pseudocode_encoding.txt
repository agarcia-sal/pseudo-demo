```plaintext
// Read an integer input representing the number of elements
SET n to integer input

// Create a list 'b' initialized with 'True' values to track state
INITIALIZE b as a list of 'True' with length n

// Initialize counters
SET j to 0  // Index for the list
SET i to 1  // Counter for iterations

// Loop while i is less than or equal to 500000
WHILE (i <= 500000) DO:
    // If the current index j in list b is True
    IF (b[j] is True) THEN:
        // Set that index to False
        SET b[j] to False

    // Increment i by 1
    INCREMENT i by 1
    
    // Update index j with wrapping around using modulus operator
    SET j to (j + i) mod n

// Create a list 'x' containing all True values from list 'b'
SET x to a new list with elements from b that are True

// Check if the list 'x' is empty
IF (length of x is 0) THEN:
    OUTPUT 'YES' // Output if there are no True values
ELSE:
    OUTPUT 'NO'  // Output if there are any True values
```
