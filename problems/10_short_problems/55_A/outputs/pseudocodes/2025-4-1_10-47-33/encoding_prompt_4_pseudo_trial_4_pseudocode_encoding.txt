```plaintext
// Input: An integer n
n = Read an integer from input

// Initialize a list to track boolean values
booleanList = Create a list of size n initialized to True

// Initialize indices
currentIndex = 0
counter = 1

// Loop until counter exceeds 500000
While counter <= 500000 do
    // If the current index in the boolean list is True
    If booleanList[currentIndex] is True then
        // Set the current index to False
        booleanList[currentIndex] = False
    End If

    // Increment counter
    counter = counter + 1
    
    // Update current index using modular arithmetic
    currentIndex = (currentIndex + counter) mod n
End While

// Filter out the True values from the boolean list
filteredList = Create a new list with elements from booleanList that are True

// Check the length of the filtered list
If the number of elements in filteredList is 0 then
    Print "YES"  // All positions were marked False
Else
    Print "NO"   // There are still True positions left
End If
```
