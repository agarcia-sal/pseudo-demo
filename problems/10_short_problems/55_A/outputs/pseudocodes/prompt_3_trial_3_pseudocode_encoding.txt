// Start of the program
Input the integer value 'n'

// Create a list 'b' of size 'n' and initialize all elements to True
Create list 'b' of size 'n' and set all elements to True

Initialize variable 'j' to 0  // This will track the current index in the list
Initialize variable 'i' to 1   // This will be used to determine how many steps to skip

// Loop until 'i' exceeds 500000
While i is less than or equal to 500000 do:
    // Check if the current index j in list b is still marked True
    If b[j] is True then:
        Set b[j] to False  // Mark the current index as False

    Increment i by 1      // Move to the next step
    Update j to (j + i) modulo n  // Move the index j forward by i steps and wrap around using modulo n

// Create a new list 'x' containing only the True values from 'b'
Create list 'x' with elements from 'b' that are True

// Check the length of the list 'x'
If the length of list 'x' is 0 then:
    Output 'YES'  // All elements have been marked False
Else:
    Output 'NO'   // There are still True elements in the list
// End of the program
