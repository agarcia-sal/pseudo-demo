def main():
    inp = [0 for i in range(eval(input()))]
    p = 0
    for i in range(0, 10001):
        p = (p+i) % len(inp)
        inp[p] += 1
        if 2 in inp:
            print("NO")
            break
        elif 0 in inp:
            continue
        else:
            print("YES")
            break
main()
