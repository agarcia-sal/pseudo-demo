```
START

// Read and convert input to absolute integer
number = absolute_value_of(int_input()) 

// Initialize counter
index = 0 

// Loop indefinitely
WHILE true DO
    // Calculate the sum of the first 'index' integers
    sum = (index * (index + 1)) / 2 
    
    // Calculate the difference between the sum and the input number
    difference = sum - number 

    // Check if the sum equals the input number
    IF sum equals number THEN
        // Print the current index
        PRINT index 
        BREAK the loop 

    // Check if the sum exceeds the input number
    ELSE IF sum greater than number THEN
        // Check if the difference is even
        IF difference is even THEN
            // Print the current index
            PRINT index 
            BREAK the loop 

    // Increment the index for the next iteration
    index = index + 1 

END
```
