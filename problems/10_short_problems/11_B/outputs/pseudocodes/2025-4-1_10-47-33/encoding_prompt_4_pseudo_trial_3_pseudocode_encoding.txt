BEGIN

    // Read and convert input to an absolute integer
    SET number = absolute value of integer input

    // Initialize index variable
    SET index = 0

    // Infinite loop to find the solution
    WHILE true DO
        
        // Calculate the sum of first 'index' natural numbers
        SET sum = (index * (index + 1)) / 2
        
        // Calculate the difference between sum and number
        SET difference = sum - number
        
        // Check if the sum equals the number
        IF sum equals number THEN
            PRINT index
            BREAK
            
        // Check if the sum is greater than the number
        ELSE IF sum greater than number THEN
            // Check if the difference is even
            IF difference is even THEN
                PRINT index
                BREAK
        
        // Increment the index for the next iteration
        index = index + 1

END
