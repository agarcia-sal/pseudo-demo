BEGIN

// Read an integer input and store its absolute value
SET number TO absolute value of integer(input())
SET index TO 0

// Infinite loop to find the desired index
WHILE true DO
    
    // Calculate the sum of the first 'index' natural numbers
    SET sum TO (index * (index + 1)) / 2
    
    // Calculate the difference between the sum and the target number
    SET difference TO sum - number
    
    // Check if the sum equals the target number
    IF sum EQUALS number THEN
        PRINT index
        BREAK
    
    // Check if the sum exceeds the target number
    ELSE IF sum GREATER THAN number THEN
        // Check if the difference is even
        IF difference MOD 2 EQUALS 0 THEN
            PRINT index
            BREAK
    
    // Increment index for the next iteration
    SET index TO index + 1

END
