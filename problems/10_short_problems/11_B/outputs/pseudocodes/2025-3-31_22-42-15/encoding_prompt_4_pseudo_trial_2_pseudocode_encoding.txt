BEGIN 

    // Get the absolute value of an integer from user input
    SET number AS absolute value of input integer
    
    SET index AS 0
    
    // Infinite loop until a condition is met
    WHILE TRUE DO
        
        // Calculate the sum of the first 'index' natural numbers
        SET sum AS (index * (index + 1)) / 2
        
        // Calculate the difference between sum and number
        SET difference AS sum - number
        
        // Check if the sum equals the number
        IF sum IS EQUAL TO number THEN
            OUTPUT index
            BREAK
            
        // Check if the sum is greater than the number
        ELSE IF sum IS GREATER THAN number THEN
            
            // Check if the difference is even
            IF difference IS EVEN THEN
                OUTPUT index
                BREAK
        
        // Increment index for the next iteration
        INCREMENT index BY 1
        
    END WHILE

END
