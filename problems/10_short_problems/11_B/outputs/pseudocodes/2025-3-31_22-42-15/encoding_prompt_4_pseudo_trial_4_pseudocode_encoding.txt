BEGIN

    // Prompt user for input and store the absolute value of the integer
    SET number = ABSOLUTE_VALUE(ENTER_INTEGER())
    
    // Initialize a counter variable
    SET index = 0
    
    // Infinite loop to find the solution
    WHILE TRUE DO
        
        // Calculate the sum of the first 'index' integers
        SET sum = (index * (index + 1)) / 2
        
        // Calculate the difference between the sum and the input number
        SET difference = sum - number
        
        // Check if the sum equals the input number
        IF sum EQUALS number THEN
            PRINT(index) // Output the index
            EXIT WHILE // Exit the loop
            
        // Check if the sum exceeds the input number
        ELSE IF sum GREATER THAN number THEN
            // Check if the difference is even
            IF difference MODULO 2 EQUALS 0 THEN
                PRINT(index) // Output the index
                EXIT WHILE // Exit the loop
            END IF
        END IF
        
        // Increment the counter variable
        INCREMENT index
        
    END WHILE

END
