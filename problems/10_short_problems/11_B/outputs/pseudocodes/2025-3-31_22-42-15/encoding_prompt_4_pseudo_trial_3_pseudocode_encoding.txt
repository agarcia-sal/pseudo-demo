FUNCTION find_triangle_number
    INPUT: Read a number n from user
    n = absolute value of n

    SET i = 0
    WHILE TRUE:
        s = triangular number of i (i * (i + 1)) / 2
        m = s - n
        
        IF s equals n THEN
            PRINT i
            BREAK
        
        ELSE IF s greater than n THEN
            IF m is even THEN
                PRINT i
                BREAK
        
        INCREMENT i by 1
END FUNCTION
