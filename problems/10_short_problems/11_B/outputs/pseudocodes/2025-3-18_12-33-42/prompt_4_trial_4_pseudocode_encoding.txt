BEGIN

    // Read an integer input and get its absolute value
    SET number TO absolute_value_of(input_integer)

    // Initialize counter
    SET index TO 0

    // Infinite loop to find the desired value
    WHILE TRUE DO
        // Calculate the triangular number for the current index
        SET triangular_number TO (index * (index + 1)) / 2

        // Calculate the difference between the triangular number and the input number
        SET difference TO triangular_number - number

        // Check if the triangular number equals the input number
        IF triangular_number EQUALS number THEN
            PRINT index
            BREAK

        // Check if the triangular number exceeds the input number
        ELSE IF triangular_number GREATER THAN number THEN
            // Check if the difference is even
            IF difference MODULO 2 EQUALS 0 THEN
                PRINT index
                BREAK

        // Increment the index for the next iteration
        SET index TO index + 1

END
