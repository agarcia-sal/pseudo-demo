```
BEGIN

    // Read an integer input, convert to absolute value
    SET n TO absolute value of integer input

    // Initialize a counter variable
    SET i TO 0

    // Infinite loop to increment counter until a condition is met
    WHILE True DO
        
        // Calculate the sum of the first i natural numbers
        SET s TO (i * (i + 1)) / 2
        
        // Calculate the difference from n
        SET m TO (s - n)
        
        // Check if the sum equals n
        IF s EQUALS n THEN
            // Print the current value of i as result
            PRINT i
            BREAK
            
        // Check if the sum exceeds n
        ELSE IF s GREATER THAN n THEN
            // If the difference is even
            IF m MOD 2 EQUALS 0 THEN
                // Print the current value of i as result
                PRINT i
                BREAK
            
        // Increment the counter for the next iteration
        INCREMENT i BY 1

END
```
