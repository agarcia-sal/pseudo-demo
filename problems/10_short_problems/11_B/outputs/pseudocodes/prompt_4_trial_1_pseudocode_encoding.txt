BEGIN

    // Read an integer input and store its absolute value
    SET inputNumber TO ABSOLUTE_VALUE(READ_INTEGER())
    SET index TO 0

    // Infinite loop until a break condition is met
    WHILE TRUE DO
        
        // Calculate the sum of the first 'index' integers
        SET sum TO (index * (index + 1)) / 2
        
        // Calculate difference between sum and inputNumber
        SET difference TO sum - inputNumber

        // Check if the sum equals the inputNumber
        IF sum EQUALS inputNumber THEN
            PRINT index
            BREAK

        // Check if the sum exceeds the inputNumber
        ELSE IF sum GREATER THAN inputNumber THEN
            
            // Check if the difference is even
            IF difference MOD 2 EQUALS 0 THEN
                PRINT index
                BREAK
            END IF
            
        END IF

        // Increment index for the next iteration
        SET index TO index + 1

    END WHILE

END
