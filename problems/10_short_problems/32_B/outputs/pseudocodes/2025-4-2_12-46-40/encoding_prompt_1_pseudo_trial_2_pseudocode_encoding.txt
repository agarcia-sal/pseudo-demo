**Pseudocode:**

START

// Read input from the user
READ a line of input and remove any leading or trailing whitespace, store in variable 'inputString'

// Initialize an index variable to track the current position within the string
SET index to 0

// Initialize an empty string to build the resulting output
SET outputString to an empty string

// Loop while the current index is less than the length of the input string
WHILE index is less than the length of inputString DO

    // Check if the current character is a dot
    IF the character at inputString[index] is equal to '.' THEN
        // Append '0' to the output string
        APPEND '0' to outputString
        // Move to the next character
        INCREMENT index by 1

    // Check if the next character is a dot (only if the current character was not a dot)
    ELSE IF the character at inputString[index + 1] is equal to '.' THEN
        // Append '1' to the output string
        APPEND '1' to outputString
        // Skip the next character
        INCREMENT index by 2

    // If neither of the above conditions are met, it must be a '00'
    ELSE
        // Append '2' to the output string
        APPEND '2' to outputString
        // Skip the next character
        INCREMENT index by 2

END WHILE

// Output the final result
PRINT outputString

END
