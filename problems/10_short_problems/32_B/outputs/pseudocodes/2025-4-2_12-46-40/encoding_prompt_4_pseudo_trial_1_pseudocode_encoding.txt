```
BEGIN PROGRAM

// Read the input string from the user
READ input_string

// Initialize variables
SET index TO 0
SET output_string TO an empty string

// Loop through the input string until we reach the end
WHILE index is less than the length of input_string DO
    IF the character at input_string[index] is '.' THEN
        // If current character is a '.', add '0' to output_string
        APPEND '0' to output_string
        INCREASE index by 1
    ELSE IF the character at input_string[index + 1] is '.' THEN
        // If next character is also '.', add '1' to output_string
        APPEND '1' to output_string
        INCREASE index by 2
    ELSE
        // Otherwise, add '2' to output_string
        APPEND '2' to output_string
        INCREASE index by 2
    END IF
END WHILE

// Output the result
PRINT output_string

END PROGRAM
```
