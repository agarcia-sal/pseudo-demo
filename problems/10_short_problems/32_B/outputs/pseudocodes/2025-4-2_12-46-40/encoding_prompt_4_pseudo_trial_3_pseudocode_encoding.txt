BEGIN

    // Read a line of input and remove any whitespace
    READ inputString

    // Initialize index and result string
    SET index TO 0
    SET result TO an empty string

    // Loop until the end of the input string is reached
    WHILE index is less than the length of inputString DO

        // Check if the current character is a single dot
        IF inputString[index] is '.' THEN
            APPEND '0' to result
            INCREMENT index by 1
        ELSE
            // Check the next character to determine the value to append
            IF the next character (inputString[index + 1]) is '.' THEN
                APPEND '1' to result
                INCREMENT index by 2
            ELSE
                APPEND '2' to result
                INCREMENT index by 2
            ENDIF
        ENDIF

    ENDWHILE

    // Output the final result
    PRINT result

END
