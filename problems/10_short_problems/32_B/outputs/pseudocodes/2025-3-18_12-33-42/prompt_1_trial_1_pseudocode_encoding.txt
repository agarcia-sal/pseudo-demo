**Pseudocode:**

BEGIN

    // Read input string from standard input
    INPUT string s
    
    // Initialize index and output answer variable
    SET index to 0
    SET answer to an empty string
    
    // Continue processing as long as the index is less than the length of the string
    WHILE index is less than the length of s DO
        
        // Check if the current character is a dot
        IF the character at position index in s is '.' THEN
            
            // Append '0' to the answer
            APPEND '0' to answer
            // Move to the next character
            INCREMENT index by 1
            
        // Check the next character for a dot
        ELSE IF the character at position index + 1 in s is '.' THEN
            
            // Append '1' to the answer
            APPEND '1' to answer
            // Move the index forward by 2 characters
            INCREMENT index by 2
            
        ELSE
            
            // If the previous conditions are not met, append '2' to the answer
            APPEND '2' to answer
            // Move the index forward by 2 characters
            INCREMENT index by 2
            
        END IF

    END WHILE

    // Output the final answer
    PRINT answer
    
END
