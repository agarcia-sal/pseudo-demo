START

WHILE index is less than length of inputString DO
    IF current character at index is '.' THEN
        // Append '0' to the outputString
        APPEND '0' to outputString
        INCREMENT index by 1
    ELSE IF next character (index + 1) is '.' THEN
        // Append '1' to the outputString
        APPEND '1' to outputString
        INCREMENT index by 2
    ELSE
        // Append '2' to the outputString
        APPEND '2' to outputString
        INCREMENT index by 2
    END IF
END WHILE

// Print the final output string
PRINT outputString

END
