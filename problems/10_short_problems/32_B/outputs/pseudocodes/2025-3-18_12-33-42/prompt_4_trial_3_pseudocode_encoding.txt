```plaintext
BEGIN

// Read input string from the standard input
STRING inputString = read input from stdin

// Initialize index and result variable
INTEGER index = 0
STRING result = ""

// Loop through the input string until the end is reached
WHILE index < length of inputString DO
    // Check for the first character
    IF character at inputString[index] is '.' THEN
        // If current character is '.', append '0' to result
        result = result + "0"
        index = index + 1  // Move to the next character
    ELSE IF character at inputString[index + 1] is '.' THEN
        // If the next character is also '.', append '1' to result
        result = result + "1"
        index = index + 2  // Move past the current and next character
    ELSE
        // If neither case fits, append '2' to result
        result = result + "2"
        index = index + 2  // Move past the current and next character
    END IF
END WHILE

// Print the final result
output result

END
```
