```pseudocode
BEGIN

    // Read input string from standard input
    input_string = READ_INPUT()
    trimmed_string = TRIM(input_string)

    // Initialize index and result string
    index = 0
    result = ""

    // Loop through the characters in the input string
    WHILE index < LENGTH(trimmed_string) DO

        // Check for the first character
        IF trimmed_string[index] == '.' THEN
            // Add '0' to the result for a single dot
            result = result + '0'
            index = index + 1

        ELSE IF trimmed_string[index + 1] == '.' THEN
            // Add '1' to the result for a 'dots' pattern
            result = result + '1'
            index = index + 2

        ELSE 
            // Default case for replacement when neither condition is met
            result = result + '2'
            index = index + 2

        END IF

    END WHILE

    // Output the final transformed result
    PRINT(result)

END
```
