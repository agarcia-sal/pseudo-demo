BEGIN

  // Read input from standard input
  Read inputString from standard input, removing any leading or trailing whitespace
  
  Initialize index i to 0
  Initialize resultString as an empty string
  
  // Loop through the input string until the end is reached
  WHILE i is less than the length of inputString DO

    // Check the current character
    IF current character at inputString[i] is '.' THEN
      Append '0' to resultString
      Increment index i by 1
     
    ELSE IF the next character inputString[i+1] is '.' THEN
      Append '1' to resultString
      Increment index i by 2
      
    ELSE
      Append '2' to resultString
      Increment index i by 2
    
  END WHILE

  // Output the final result
  Print resultString

END
