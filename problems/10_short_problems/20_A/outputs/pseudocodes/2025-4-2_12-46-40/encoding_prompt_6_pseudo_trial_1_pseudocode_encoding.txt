BEGIN PROGRAM

    FUNCTION NormalizeFilePath(INPUT rawFilePath)

        // Trim whitespace from the input
        SET trimmedFilePath TO REMOVE_WHITESPACE(rawFilePath)
        
        // Normalize the file path by removing unnecessary segments
        SET normalizedFilePath TO REMOVE_REDUNDANT_PATH_SEGMENTS(trimmedFilePath)
        
        // Ensure the normalized path starts with a single forward slash
        SET finalFilePath TO ENSURE_SINGLE_LEADING_SLASH(normalizedFilePath)
        
        RETURN finalFilePath

    END FUNCTION

    FUNCTION REMOVE_WHITESPACE(INPUT filePath)
        // Remove leading and trailing whitespace from the filePath
        RETURN filePath WITHOUT WHITESPACE_AT_START_OR_END
    END FUNCTION

    FUNCTION REMOVE_REDUNDANT_PATH_SEGMENTS(INPUT filePath)
        // Replace multiple sequential slashes (//) with a single slash (/)
        RETURN filePath WITH_REPLACEMENTS('//' TO '/') 
    END FUNCTION

    FUNCTION ENSURE_SINGLE_LEADING_SLASH(INPUT filePath)
        IF filePath STARTS WITH '/' THEN
            RETURN filePath // Already has a leading slash
        ELSE
            RETURN '/' + filePath // Add a leading slash
        END IF
    END FUNCTION

    // Read the input from the standard input
    SET userInput TO READ_FROM_STANDARD_INPUT()
    
    // Normalize the file path
    SET resultPath TO NormalizeFilePath(userInput)

    // Print the final normalized file path
    PRINT resultPath

END PROGRAM
