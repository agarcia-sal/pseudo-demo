BEGIN

    FUNCTION NormalizeAndFormatPath():
        // Read input from standard input and remove extra whitespace
        inputPath = READ input from standard input
        
        // Normalize the file path by collapsing redundant separators
        normalizedPath = NORMALIZE the inputPath
        
        // Replace any leading slashes with a single slash
        formattedPath = SUBSTITUTE leading slashes in normalizedPath with a single slash
        
        // Output the formatted path
        PRINT formattedPath
    END FUNCTION

END
