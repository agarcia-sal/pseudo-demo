def readint(): return int(input())
def readfloat(): return float(input())
def readarray(N, foo):
        res = []
        for _ in range(N):
                res.append(foo())
        return res
def readlinearray(foo): return list(map(foo, input().split()))

from string import letters
letters = letters[:26]

def get_npath(path):
    dirs = [d for d in path.split('/') if d != '']
    return '/' + '/'.join(dirs)

if __name__ == '__main__':
    path = input()
    print(get_npath(path))
