```
BEGIN

    // Read a line of input from standard input and remove the last character
    INPUT line FROM standard input
    REMOVE last character FROM line
    SET n TO LENGTH of line
    INITIALIZE returnValue TO 0

    // Loop over substring lengths from 0 to n-1
    FOR each length l FROM 0 TO n - 1 DO
        
        // Loop over each starting index for the substrings
        FOR each index i FROM 0 TO n - 1 DO
            
            // Check if the substring of length l starting at index i occurs again later in the string
            IF substring line[i:i+l] is found in line starting from index i+1 THEN
                SET returnValue TO l
                BREAK out of the inner loop
            
        // If a match was found with length l, exit the outer loop
        IF returnValue is not equal to 0 THEN
            BREAK out of the outer loop

    // Output the longest found length of repeated substring
    PRINT returnValue

END
```
