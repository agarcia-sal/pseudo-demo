FUNCTION find_longest_repeated_substring(INPUT string line)
    SET n TO LENGTH of line
    SET longest_repetition_length TO 0

    // Loop through possible substring lengths from 0 to n-1
    FOR length FROM 0 TO n-1 DO
        // Loop through the starting position of the substring
        FOR start_position FROM 0 TO n-1 DO
            // Extract the substring of the current length
            SET substring TO substring from line starting at start_position with length
            // Check if this substring occurs again in the line
            IF substring exists in line starting from start_position + 1 THEN
                SET longest_repetition_length TO length
                BREAK OUT of the inner loop
            END IF
        END FOR
    END FOR

    // Output the length of the longest repeated substring found
    PRINT longest_repetition_length
END FUNCTION
