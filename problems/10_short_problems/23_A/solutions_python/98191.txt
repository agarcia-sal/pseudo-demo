#import sys
a=input()
l=len(a)
if l<2:
  print(0)
  exit()
   
b=list(range(len(a)))
b.pop(0)
b.reverse()
#print 'b=',b
#exit()
#print 'l=',l
for s in b:
  #print 's=',s
  for i in range(l-s):
   #print 'i=',i,'a[',i,':',s,']=',a[i:s]
   c=list(range(i+1,l-s+1))
   #print 'c=',c
   for j in c:
    #print 'j=',j,'a[',j,':',j+s,']=',a[j:j+s]
    if a[i:i+s]==a[j:j+s]:
      print(s)
      exit()
print(0)
