#!/usr/bin/env python
# -*- coding: utf-8 -*-

def main():
    s = input()
    for l in range(len(s)-1, 0, -1):
        for b in range(0, len(s)-l+1):
            if s[b:b+l] in s[b+1:]:
                print(l)
                return 0
    print(0)
    return 0

if __name__ == '__main__':
    main()
