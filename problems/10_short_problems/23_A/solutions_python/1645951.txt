from copy import copy

s = input()
L = len(s)
for l in reversed(range(1, L)):
    for i in range(L - l + 1):
        t = copy(s)
        tt = t[i:i + l]
        id = t.find(tt)
        if id == -1: break
        if t[id + 1:].find(tt) != -1:
            print(l)
            exit(0)
print(0)
