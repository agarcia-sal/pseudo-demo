def f():
    s = input()
    d = {}
    t = ""
    for length in range(len(s) - 1, -1, -1):
        for index in range(len(s) - length + 1):
            t = s[index:index+length]
            if t in d:
                print(length)
                return
            d[t] = 0
    print("0")
    return
f()
