BEGIN
    // Read an integer input representing the upper limit for checking
    t ← INPUT_INTEGER()
    
    // Initialize a variable to count the special numbers
    specialCount ← 0
    
    // Loop through each number from 1 to t
    FOR i FROM 1 TO t DO
        // Initialize a counter for prime factors
        primeFactorCount ← 0
        num ← i
        
        // Check for prime factors from 2 up to i
        FOR j FROM 2 TO i - 1 DO
            // If j is a factor of num
            IF (num MOD j EQUALS 0) THEN
                // Increment the prime factor counter
                primeFactorCount ← primeFactorCount + 1
                
                // Divide num by j until j is no longer a factor
                WHILE (num MOD j EQUALS 0) DO
                    num ← num / j
                END WHILE
            END IF
        END FOR
        
        // Check if the number has exactly two distinct prime factors
        IF (primeFactorCount EQUALS 2) THEN
            // Increment the special number count
            specialCount ← specialCount + 1
        END IF
    END FOR

    // Output the total count of special numbers found
    PRINT(specialCount)
END
