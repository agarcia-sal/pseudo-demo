BEGIN

    // Step 1: Read an integer input from the user representing the limit
    SET inputLimit TO Integer Input

    // Step 2: Initialize a variable to count the number of special numbers
    SET specialNumberCount TO 0

    // Step 3: Loop through each integer from 1 to the input limit (inclusive)
    FOR each integer currentNumber FROM 1 TO inputLimit DO
        
        // Initialize a counter for the distinct prime factors of the current number
        SET distinctPrimeFactorCount TO 0
        SET remainingNumber TO currentNumber

        // Step 4: Check for prime factors starting from 2 up to one less than the current number
        FOR each integer potentialFactor FROM 2 TO (currentNumber - 1) DO
            
            // Step 5: Determine if potentialFactor is a factor of remainingNumber
            IF (remainingNumber MOD potentialFactor = 0) THEN
                // Increment the count of distinct prime factors
                INCREMENT distinctPrimeFactorCount

                // Step 6: Divide remainingNumber by potentialFactor until it is no longer divisible
                WHILE (remainingNumber MOD potentialFactor = 0) DO
                    SET remainingNumber TO remainingNumber / potentialFactor
                END WHILE
            END IF
        END FOR

        // Step 7: Check if there were exactly two distinct prime factors
        IF (distinctPrimeFactorCount = 2) THEN
            // Increment the special number count
            INCREMENT specialNumberCount
        END IF
    END FOR

    // Step 8: Output the total count of special numbers found
    PRINT specialNumberCount

END
