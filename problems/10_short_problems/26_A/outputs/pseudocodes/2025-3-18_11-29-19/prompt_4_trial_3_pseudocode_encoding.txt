```plaintext
// Read an integer input representing the upper limit
READ t

// Initialize a counter for the number of semiprime numbers
SET res TO 0

// Loop through each number from 1 to t
FOR i FROM 1 TO t DO
    // Initialize a count for distinct prime factors
    SET cnt TO 0
    SET num TO i

    // Loop through possible factors from 2 to i-1
    FOR j FROM 2 TO i-1 DO
        // Check if j is a factor of num
        IF num MOD j EQUALS 0 THEN
            // Increment the distinct prime factor count
            INCREMENT cnt
            
            // Divide num by j as long as it is a factor
            WHILE num MOD j EQUALS 0 DO
                num SET TO num / j
            END WHILE
        END IF
    END FOR
    
    // Check if the number has exactly two distinct prime factors
    IF cnt EQUALS 2 THEN
        // Increment the semiprime counter
        INCREMENT res
    END IF
END FOR

// Output the count of semiprime numbers
PRINT res
```
