BEGIN

  // Step 1: Read the total number of test cases
  Input totalTestCases

  // Step 2: Initialize a variable to keep track of the count of numbers with exactly two prime factors
  Initialize primeFactorCount to 0

  // Step 3: Iterate through each number from 1 to totalTestCases
  FOR each number from 1 to totalTestCases DO

    // Step 4: Initialize a counter for prime factors and set the current number
    Initialize primeFactorCounter to 0
    Set currentNumber to current number

    // Step 5: Check for factors starting from 2 up to one less than the current number
    FOR each potentialFactor from 2 to currentNumber - 1 DO

      // Step 6: Determine if potentialFactor is a divisor of currentNumber
      IF currentNumber is divisible by potentialFactor THEN
        
        // Step 7: If it is a divisor, increment the prime factors counter
        Increment primeFactorCounter

        // Step 8: Divide current number by potentialFactor as long as it is still divisible
        WHILE currentNumber is divisible by potentialFactor DO
          Divide currentNumber by potentialFactor
        END WHILE
        
      END IF
      
    END FOR
    
    // Step 9: Check if the number has exactly two prime factors
    IF primeFactorCounter equals 2 THEN
      // Step 10: If yes, increment the overall prime factor count
      Increment primeFactorCount
    END IF

  END FOR

  // Step 11: Output the count of numbers with exactly two prime factors
  Output primeFactorCount

END
