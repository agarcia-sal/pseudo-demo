```
START

// Read the number of test cases
INPUT testCases AS INTEGER

// Initialize result counter for the number of semiprime numbers
SET resultCount TO 0

// Loop from 1 to the number of test cases
FOR each number FROM 1 TO testCases DO

    // Initialize a counter for prime factors
    SET primeFactorCount TO 0
    SET currentNumber TO number
    
    // Check for divisibility from 2 to the current number
    FOR each potentialFactor FROM 2 TO (number - 1) DO
    
        // If current number is divisible by potential factor
        IF currentNumber MOD potentialFactor EQUALS 0 THEN
            // Increment prime factor count
            INCREMENT primeFactorCount
            
            // Divide current number by potential factor until it's no longer divisible
            WHILE currentNumber MOD potentialFactor EQUALS 0 DO
                SET currentNumber TO currentNumber / potentialFactor
            END WHILE
        END IF
    END FOR
    
    // Check if there are exactly two distinct prime factors
    IF primeFactorCount EQUALS 2 THEN
        // Increment the count of semiprime numbers
        INCREMENT resultCount
    END IF

END FOR

// Output the total count of semiprime numbers
PRINT resultCount

END
```
