BEGIN

    // Read the integer input representing the upper limit
    SET upperLimit TO INTEGER(INPUT("Enter a number: "))

    // Initialize a counter for the results
    SET primeCount TO 0

    // Iterate from 1 to the upper limit
    FOR number FROM 1 TO upperLimit DO
        SET factorCount TO 0
        SET currentNumber TO number
        
        // Check for factors from 2 up to the current number
        FOR potentialFactor FROM 2 TO (currentNumber - 1) DO
            // If currentNumber is divisible by potentialFactor
            IF (currentNumber MOD potentialFactor EQUALS 0) THEN
                INCREMENT factorCount BY 1
                
                // Divide currentNumber by potentialFactor until it can no longer be done
                WHILE (currentNumber MOD potentialFactor EQUALS 0) DO
                    SET currentNumber TO currentNumber / potentialFactor
                END WHILE
            END IF
        END FOR
        
        // Check if the number has exactly two unique prime factors
        IF (factorCount EQUALS 2) THEN
            INCREMENT primeCount BY 1
        END IF
    END FOR

    // Print the total count of numbers with exactly two unique prime factors
    PRINT(primeCount)

END
