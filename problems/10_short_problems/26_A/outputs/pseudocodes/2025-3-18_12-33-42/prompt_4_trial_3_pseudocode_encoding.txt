BEGIN

    // Read the input value for 't'
    SET t TO integer input from user

    // Initialize result counter for prime numbers
    SET resultCount TO 0

    // Iterate over each number from 1 to t
    FOR each number 'i' FROM 1 TO t DO
        
        // Initialize counter for prime factors
        SET factorCount TO 0
        SET currentNumber TO i

        // Check for prime factors from 2 up to (i-1)
        FOR each potentialFactor 'j' FROM 2 TO (i - 1) DO
            
            // Check if 'j' is a factor of 'currentNumber'
            IF (currentNumber mod j == 0) THEN
                INCREMENT factorCount
            
                // Divide 'currentNumber' by 'j' while it is divisible
                WHILE (currentNumber mod j == 0) DO
                    SET currentNumber TO currentNumber / j
                
                END WHILE
            END IF
        
        // If exactly two prime factors are found, count it as a valid case
        IF (factorCount == 2) THEN
            INCREMENT resultCount
        END IF

    END FOR

    // Output the total count of numbers with exactly two prime factors
    PRINT resultCount

END
