BEGIN

    // Read an integer input representing the number of tests
    SET testCount TO READ integer input

    // Initialize a result counter for prime numbers
    SET primeCount TO 0

    // Loop through each number from 1 to testCount
    FOR number FROM 1 TO testCount DO

        // Initialize a divisor count and set the current number
        SET divisorCount TO 0
        SET currentNumber TO number

        // Check for divisors starting from 2 up to but not including the current number
        FOR divisor FROM 2 TO currentNumber - 1 DO
            // If the divisor divides the current number
            IF currentNumber MOD divisor EQUALS 0 THEN
                // Increment the divisor count
                INCREMENT divisorCount
                // Divide the current number by the divisor until it is no longer divisible
                WHILE currentNumber MOD divisor EQUALS 0 DO
                    SET currentNumber TO currentNumber / divisor
                END WHILE
            END IF
        END FOR

        // Check if exactly two distinct divisors were found (indicating a prime number)
        IF divisorCount EQUALS 2 THEN
            INCREMENT primeCount
        END IF

    END FOR

    // Output the total count of prime numbers found
    PRINT primeCount

END
