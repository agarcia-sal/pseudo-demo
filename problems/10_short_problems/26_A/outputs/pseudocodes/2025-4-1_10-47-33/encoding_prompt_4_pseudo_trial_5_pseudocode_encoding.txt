BEGIN

    // Read the number of test cases
    SET testCaseCount TO input as integer

    // Initialize result for counting special numbers
    SET result TO 0

    // Loop through each number from 1 to testCaseCount
    FOR each i FROM 1 TO testCaseCount DO
        // Initialize counter for divisors
        SET divisorCount TO 0
        SET currentNumber TO i

        // Check for divisors from 2 to currentNumber - 1
        FOR each j FROM 2 TO currentNumber - 1 DO
            // If j is a divisor of currentNumber
            IF (currentNumber MOD j EQUALS 0) THEN
                INCREMENT divisorCount by 1
                
                // Divide currentNumber by j until it no longer divides evenly
                WHILE (currentNumber MOD j EQUALS 0) DO
                    SET currentNumber TO currentNumber DIVIDED BY j
                END WHILE
            END IF
        END FOR

        // If exactly two distinct prime factors were found
        IF (divisorCount EQUALS 2) THEN
            INCREMENT result by 1
        END IF
    END FOR

    // Output the result
    PRINT result

END
