from math import sqrt
n = eval(input())

p = list(range(2,n))
for i in range(2,int(sqrt(n))+1):
	p = [x for x in p if x == i or x % i]

def isp2(k):
	j = x = 0
	while k>1 and j<len(p):
		if k%p[j]==0 :
			x += 1
			if x > 2 : return 0
			while k>1 and k%p[j]==0 : k/=p[j]
		j+=1
		
	return 1 if x==2 else 0		

print(sum(map(isp2, list(range(6,n+1)))))
