primes = [2]
n = int(input())
prime = [True]*(n+1)
for i in range(3,n+1,2):
    if not prime[i]:continue
    primes.append(i)
    for j in range(i*i,n+1,i):
        prime[j]=False
ans = 0
for i in range(6,n+1):
    c = 0
    for p in primes:
        if i%p==0: c+=1
        if p>i or c>2:break
    if c==2: ans+=1
print(ans)
