from math import sqrt
def cal(x):
    ret = set()
    while x != 1:
        for i in range(2, x + 1):
            if x % i == 0:
                x /= i;
                ret.add(i)
                break
    return len(ret)

n = int(input())
print(sum(1 for x in range(4, n+1) if cal(x) == 2))
