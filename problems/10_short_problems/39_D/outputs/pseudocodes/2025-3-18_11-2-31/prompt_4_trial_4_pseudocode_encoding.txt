FUNCTION doMain
    // Read two lines of input, each containing three numbers
    INPUT firstLine
    INPUT secondLine
    
    // Split the input lines into lists of strings
    SPLIT firstLine INTO firstNumbers
    SPLIT secondLine INTO secondNumbers
    
    // Initialize a count for differences
    SET differenceCount TO 0 
    
    // Compare corresponding numbers from both lists
    FOR index FROM 0 TO 2 DO
        CONVERT firstNumbers[index] TO integer AS a
        CONVERT secondNumbers[index] TO integer AS b
        
        // If the numbers are different, increment the difference count
        IF a IS NOT EQUAL TO b THEN
            INCREMENT differenceCount
        END IF
    END FOR
    
    // Determine if the number of differences is less than 3
    IF differenceCount IS LESS THAN 3 THEN
        PRINT "YES"
    ELSE
        PRINT "NO"
    END IF
END FUNCTION

// Main execution block
IF this is the main module THEN
    CALL doMain()
END IF
