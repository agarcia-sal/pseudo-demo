    SPLIT secondLine INTO secondValues
    
    // Initialize a counter for discrepancies
    DISCREPANCY_COUNT = 0
    
    // Loop through the first three elements of both lists
    FOR index FROM 0 TO 2 DO
        // Convert the current elements to integers
        FIRST_VALUE = CONVERT firstValues[index] TO INTEGER

        // Check if the values are different
        IF FIRST_VALUE IS NOT EQUAL TO SECOND_VALUE THEN
            INCREMENT DISCREPANCY_COUNT BY 1
        END IF
    END FOR
    
    // Determine the output based on the count of discrepancies
    IF DISCREPANCY_COUNT IS LESS THAN 3 THEN
        PRINT "YES"
    ELSE
        PRINT "NO"
    END IF

END FUNCTION
