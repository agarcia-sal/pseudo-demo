Function doMain
    // Read two lines of input
    Set firstInput to user input
    Set secondInput to user input

    // Split the input strings into lists of strings
    Set firstNumberList to split firstInput by spaces
    Set secondNumberList to split secondInput by spaces

    // Initialize a counter for differences
    Set differenceCount to 0 

    // Compare the first three elements of both lists
    For each index from 0 to 2 (inclusive)
        // Convert the current elements to integers
        Set firstNumber to integer value of firstNumberList at index
        Set secondNumber to integer value of secondNumberList at index
        
        // Check if the current elements are different
        If firstNumber is not equal to secondNumber
            // Increment the difference count
            Increase differenceCount by 1 

    // Determine if there are fewer than 3 differences
    If differenceCount is less than 3
        Output "YES" // Fewer than 3 differences
    Else
        Output "NO" // 3 or more differences
      
// Start the program by calling doMain
Call doMain
