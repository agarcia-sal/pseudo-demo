FUNCTION doMain

    SET tt1 TO split t1 into a list
    SET tt2 TO split t2 into a list

    // Initialize a counter for differences
    SET res TO 0 

    // Iterate over the first three elements in the lists
    FOR each index from 0 to 2 DO
        SET a TO convert tt1[index] to an integer
        SET b TO convert tt2[index] to an integer
        
        // Check if the values are different
        IF a is not equal to b THEN
            INCREMENT res by 1 

    // Check the number of differences
    IF res is less than 3 THEN
        PRINT "YES"

        PRINT "NO"

END FUNCTION
