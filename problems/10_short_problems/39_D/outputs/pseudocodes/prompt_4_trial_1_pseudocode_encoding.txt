```plaintext
BEGIN FUNCTION doMain

    // Read input from the user
    INPUT firstString
    INPUT secondString

    // Split the input strings into lists of individual elements
    SET firstList = SPLIT firstString into separate elements
    SET secondList = SPLIT secondString into separate elements

    // Initialize a counter for discrepancies
    SET discrepancyCount = 0 

    // Iterate through the first three elements of both lists
    FOR i FROM 0 TO 2 DO
        // Convert the current elements to integers
        SET firstValue = CONVERT firstList[i] TO INTEGER
        SET secondValue = CONVERT secondList[i] TO INTEGER

        // Count discrepancies between the two values
        IF firstValue is NOT EQUAL to secondValue THEN
            INCREMENT discrepancyCount
        ENDIF
    ENDFOR

    // Check if the number of discrepancies is less than 3
    IF discrepancyCount < 3 THEN
        PRINT "YES"
    ELSE
        PRINT "NO"
    ENDIF

END FUNCTION

// Main execution point
CALL doMain
```
