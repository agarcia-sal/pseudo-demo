    SPLIT firstLine INTO list firstNumbers
    SPLIT secondLine INTO list secondNumbers
    
    // Initialize a counter to track differences
    SET differenceCount TO 0 

    // Iterate over the first three elements of the lists
    FOR index FROM 0 TO 2 DO
        // Convert the string numbers to integers
        SET numberFromFirstLine TO CONVERT firstNumbers[index] TO INTEGER
        SET numberFromSecondLine TO CONVERT secondNumbers[index] TO INTEGER

        // Check if the numbers are different
        IF numberFromFirstLine IS NOT EQUAL TO numberFromSecondLine THEN
            // Increment the difference counter if they are not equal
            INCREASE differenceCount BY 1 
        END IF
    END FOR

    // If the count of differences is less than 3, print "YES"
    IF differenceCount IS LESS THAN 3 THEN
        PRINT "YES"
    ELSE
        // Otherwise, print "NO"
        PRINT "NO"
    END IF
END FUNCTION

// Entry point of the program
IF __name__ IS "main" THEN
    CALL main()
