FUNCTION main()
    // Read two input strings
    STRING input1 = READ_INPUT()
    STRING input2 = READ_INPUT()
    
    // Split the input strings into lists of integers
    LIST of INTEGER numbers1 = SPLIT_AND_CONVERT_TO_INTEGERS(input1)
    LIST of INTEGER numbers2 = SPLIT_AND_CONVERT_TO_INTEGERS(input2)
    
    // Initialize a variable to count differences
    INTEGER differenceCount = 0 

    // Compare the first three integers from both lists
    FOR EACH position FROM 0 TO 2 DO
        INTEGER value1 = numbers1[position]
        INTEGER value2 = numbers2[position]
        
        // If the integers are different, increment the difference counter
        IF value1 IS NOT EQUAL TO value2 THEN
            INCREMENT differenceCount
        ENDIF
    ENDFOR

    // Determine output based on the count of differences
    IF differenceCount IS LESS THAN 3 THEN
        OUTPUT "YES"
    ELSE
        OUTPUT "NO"
    ENDIF
END FUNCTION

// Helper function to split string and convert to integers
FUNCTION SPLIT_AND_CONVERT_TO_INTEGERS(inputString)
    RETURN SPLIT(inputString) AND CONVERT EACH ELEMENT TO INTEGER
END FUNCTION
