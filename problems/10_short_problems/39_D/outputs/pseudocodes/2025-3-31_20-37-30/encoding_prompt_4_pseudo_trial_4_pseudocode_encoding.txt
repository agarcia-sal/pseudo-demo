FUNCTION doMain
    // Read two lines of input
    DECLARE t1 AS INPUT()
    DECLARE t2 AS INPUT()
    
    // Split the input strings into lists
    DECLARE tt1 AS SPLIT(t1)
    DECLARE tt2 AS SPLIT(t2)
    
    // Initialize a result counter
    DECLARE discrepancyCount AS 0
    
    // Compare corresponding elements in the two lists
    FOR each index from 0 to 2 DO
        // Convert the current element to an integer
        DECLARE a AS INTEGER(tt1[index])
        DECLARE b AS INTEGER(tt2[index])
        
        // Check if the elements are different
        IF a is NOT equal to b THEN
            INCREMENT discrepancyCount by 1
        ENDIF
    ENDFOR
    
    // Determine if discrepancies are less than 3
    IF discrepancyCount is LESS than 3 THEN
        PRINT "YES"
    ELSE
        PRINT "NO"
    ENDIF
END FUNCTION

// Execute the main function
IF script is RUNNING AS MAIN THEN
    CALL doMain()
ENDIF
