```
DEFINE FUNCTION doMain
    // Read two lines of input
    INPUT firstLine
    INPUT secondLine
    
    // Split the input lines into lists of words
    SPLIT firstLine into firstList
    SPLIT secondLine into secondList
    
    // Initialize a counter for differences
    SET differenceCount to 0
    
    // Loop through the first three elements of both lists
    FOR index FROM 0 TO 2 DO
        // Convert elements to integers for comparison
        SET valueA to INTEGER of firstList[index]
        SET valueB to INTEGER of secondList[index]
        
        // Check if the values are different
        IF valueA IS NOT EQUAL TO valueB THEN
            INCREMENT differenceCount by 1
            
    // Check the number of differences and print the result
    IF differenceCount IS LESS THAN 3 THEN
        PRINT "YES"
    ELSE
        PRINT "NO"
        
END FUNCTION

// Start execution of the program
CALL doMain()
```
