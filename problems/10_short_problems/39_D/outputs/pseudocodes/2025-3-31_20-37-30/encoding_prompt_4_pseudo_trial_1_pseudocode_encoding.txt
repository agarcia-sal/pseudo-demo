FUNCTION doMain()

    // Read two lines of input
    INPUT firstLine
    INPUT secondLine
    
    // Split the input lines into lists of string elements
    SPLIT firstLine INTO list firstList
    SPLIT secondLine INTO list secondList
    
    // Initialize a counter for differences
    SET differenceCount TO 0
    
    // Iterate over the first three elements of both lists
    FOR index FROM 0 TO 2 DO
        // Convert strings to integers for comparison
        SET firstValue TO INTEGER(firstList[index])
        SET secondValue TO INTEGER(secondList[index])
        
        // Check if the elements are different
        IF firstValue IS NOT EQUAL TO secondValue THEN
            INCREMENT differenceCount
        END IF
    END FOR
    
    // Determine the output based on the count of differences
    IF differenceCount IS LESS THAN 3 THEN
        PRINT "YES"
    ELSE
        PRINT "NO"
    END IF

END FUNCTION

// Main execution starts here
IF program IS executed AS MAIN THEN
    CALL doMain()
END IF
