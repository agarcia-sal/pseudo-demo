FUNCTION doMain
    // Step 1: Get input from the user
    DECLARE firstInput AS STRING
    DECLARE secondInput AS STRING
    firstInput = GET user input
    secondInput = GET user input

    // Step 2: Split the input strings into lists of numbers
    DECLARE firstNumbers AS LIST OF STRING
    DECLARE secondNumbers AS LIST OF STRING
    firstNumbers = SPLIT firstInput INTO LIST
    secondNumbers = SPLIT secondInput INTO LIST

    // Step 3: Initialize a counter for differences
    DECLARE differenceCount AS INTEGER
    differenceCount = 0 

    // Step 4: Compare corresponding elements of the two number lists
    FOR index FROM 0 TO 2 DO
        // Convert string to integer for comparison
        DECLARE firstNumber AS INTEGER
        DECLARE secondNumber AS INTEGER
        firstNumber = CONVERT firstNumbers[index] TO INTEGER
        secondNumber = CONVERT secondNumbers[index] TO INTEGER

        // Check if the numbers are different
        IF firstNumber IS NOT EQUAL TO secondNumber THEN
            INCREMENT differenceCount BY 1 
        END IF
    END FOR

    // Step 5: Output result based on the number of differences
    IF differenceCount IS LESS THAN 3 THEN
        PRINT "YES"
    ELSE
        PRINT "NO"
    END IF
END FUNCTION

// Main execution starts here
CALL doMain
