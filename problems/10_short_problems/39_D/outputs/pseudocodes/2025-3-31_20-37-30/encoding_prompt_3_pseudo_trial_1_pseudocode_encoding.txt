```
FUNCTION main
    // Prompt user for two inputs
    SET firstInput = user input
    SET secondInput = user input
    
    // Split the inputs into lists of values
    SET firstList = split firstInput into a list
    SET secondList = split secondInput into a list
    
    // Initialize a counter for differences
    SET differenceCount = 0 
    
    // Iterate over the first three elements of both lists
    FOR each index from 0 to 2 DO
        SET firstValue = convert firstList[index] to integer
        SET secondValue = convert secondList[index] to integer
        
        // Check if the current values are different
        IF firstValue is NOT equal to secondValue THEN
            INCREMENT differenceCount by 1 
        END IF
    END FOR
    
    // Decide the output based on the number of differences
    IF differenceCount is LESS THAN 3 THEN
        PRINT "YES"
    ELSE
        PRINT "NO"
    END IF
END FUNCTION

// Start program execution
CALL main()
```
