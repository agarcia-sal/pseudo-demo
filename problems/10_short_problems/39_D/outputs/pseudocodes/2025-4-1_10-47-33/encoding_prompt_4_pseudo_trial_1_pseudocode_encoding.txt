FUNCTION main()
    // Read input values
    t1 = READ user input
    t2 = READ user input

    // Split the input strings into lists
    tt1 = SPLIT t1 into words
    tt2 = SPLIT t2 into words

    // Initialize difference counter
    differenceCount = 0 

    // Loop through the first three elements of both lists
    FOR index FROM 0 TO 2 DO
        // Convert string values to integers
        a = CONVERT tt1[index] to integer
        b = CONVERT tt2[index] to integer
        
        // Count differences
        IF a IS NOT EQUAL TO b THEN
            INCREMENT differenceCount by 1
        END IF
    END FOR

    // Determine output based on the number of differences
    IF differenceCount < 3 THEN
        PRINT "YES"
    ELSE
        PRINT "NO"
    END IF
END FUNCTION

// Execute main function
CALL main()
