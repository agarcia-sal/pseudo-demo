FUNCTION doMain()

    // Read two lines of input
    INPUT firstLine
    INPUT secondLine
    
    // Split the input lines into lists of strings
    SPLIT firstLine INTO list1
    SPLIT secondLine INTO list2
    
    // Initialize a counter for different values
    SET differenceCount TO 0
    
    // Compare the first three elements in both lists
    FOR index FROM 0 TO 2 DO
        // Convert the current elements to integers
        SET valueA TO INTEGER(list1[index])
        SET valueB TO INTEGER(list2[index])
        
        // If the values are not equal, increment the difference counter
        IF valueA IS NOT EQUAL TO valueB THEN
            INCREMENT differenceCount
            
    // Check if the number of differences is less than 3
    IF differenceCount < 3 THEN
        PRINT "YES"
    ELSE 
        PRINT "NO"

END FUNCTION

// Main program execution starts here
CALL doMain()
