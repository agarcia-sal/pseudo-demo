FUNCTION main()
    // Read two lines of input representing two sets of numbers
    INPUT firstSet
    INPUT secondSet
    
    // Split the input strings into lists of numbers
    SPLIT firstSet INTO list1
    SPLIT secondSet INTO list2
    
    // Initialize a counter for differences
    SET differenceCount TO 0
    
    // Compare corresponding elements from both lists
    FOR index FROM 0 TO 2 DO
        SET number1 TO integer value of list1[index]
        SET number2 TO integer value of list2[index]
        
        // Check if the numbers are different
        IF number1 is NOT equal to number2 THEN
            INCREMENT differenceCount by 1
            
    // Determine if the difference count is less than 3
    IF differenceCount < 3 THEN
        PRINT "YES"
    ELSE 
        PRINT "NO"

END FUNCTION

// Execute the main function
CALL main()
