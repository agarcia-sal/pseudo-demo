FUNCTION doMain
    // Read two lines of input from the user
    INPUT firstLine
    INPUT secondLine
    
    // Split the input lines into lists of strings
    SPLIT firstLine INTO firstList
    SPLIT secondLine INTO secondList

    // Initialize a counter for the differences
    INITIALIZE differenceCount AS 0 
    
    // Loop through the first 3 elements of both lists
    FOR index FROM 0 TO 2 DO
        // Convert string elements to integers
        SET firstValue TO INTEGER of firstList[index]
        SET secondValue TO INTEGER of secondList[index]
        
        // Compare the two values
        IF firstValue IS NOT EQUAL TO secondValue THEN
            // Increment the difference count if they are not the same
            INCREMENT differenceCount BY 1
        END IF
    END FOR

    // Check the number of differences and print the result
    IF differenceCount IS LESS THAN 3 THEN
        PRINT "YES"
    ELSE
        PRINT "NO"
    END IF
END FUNCTION

// Main execution starts here
IF this script is executed as the main program THEN
    CALL doMain()
END IF
