FUNCTION doMain:

    // Read two lines of input
    DECLARE inputLine1 AS STRING
    DECLARE inputLine2 AS STRING
    inputLine1 = READ INPUT
    inputLine2 = READ INPUT

    // Split the input lines into lists of strings
    DECLARE list1 AS LIST OF STRINGS
    DECLARE list2 AS LIST OF STRINGS
    list1 = SPLIT inputLine1 INTO SPACE SEPARATED VALUES
    list2 = SPLIT inputLine2 INTO SPACE SEPARATED VALUES

    // Initialize a counter for differences
    DECLARE differenceCount AS INTEGER
    differenceCount = 0 

    // Compare elements of both lists
    FOR index FROM 0 TO 2 DO:
        DECLARE value1 AS INTEGER
        DECLARE value2 AS INTEGER
        value1 = CONVERT list1[index] TO INTEGER
        value2 = CONVERT list2[index] TO INTEGER

        // Increment the difference count if values are not equal
        IF value1 IS NOT EQUAL TO value2 THEN:
            differenceCount = differenceCount + 1 

    // Check the number of differences and print result
    IF differenceCount < 3 THEN:
        PRINT "YES"
    ELSE:
        PRINT "NO"

END FUNCTION

// Start the program
IF this script is the main module THEN:
    CALL doMain()
