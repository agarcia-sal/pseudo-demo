Procedure StartMain
    // Read two lines of input
    Declare string input1
    Declare string input2
    input1 = ReadInput()
    input2 = ReadInput()

    // Split the input strings into lists of values
    Declare list values1
    Declare list values2
    values1 = SplitString(input1)
    values2 = SplitString(input2)

    // Initialize a counter for discrepancies
    Declare integer countDiscrepancies
    countDiscrepancies = 0 

    // Compare the elements of both lists up to the first three elements
    For index from 0 to 2 do
        // Convert the current elements to integers
        Declare integer valueA
        Declare integer valueB
        valueA = ConvertToInteger(values1[index])
        valueB = ConvertToInteger(values2[index])
        
        // If the values are not equal, increment the discrepancy counter
        If valueA is not equal to valueB then
            countDiscrepancies = countDiscrepancies + 1 
        End If
    End For

    // If there are fewer than three discrepancies, print "YES"; otherwise print "NO"
    If countDiscrepancies is less than 3 then
        Print "YES"
    Else
        Print "NO"
    End If
End Procedure

// Start the main process
StartMain()
