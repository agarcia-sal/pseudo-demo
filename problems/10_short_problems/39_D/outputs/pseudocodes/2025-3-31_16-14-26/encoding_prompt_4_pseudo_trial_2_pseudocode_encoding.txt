```
FUNCTION main
    // Read two strings of input
    SET input1 TO user input
    SET input2 TO user input

    // Split the input strings into lists of components
    SET components1 TO split input1 into a list
    SET components2 TO split input2 into a list

    // Initialize a counter for differences
    SET differenceCount TO 0 

    // Iterate over the first three components of both lists
    FOR index FROM 0 TO 2 DO
        // Convert each component to an integer
        SET value1 TO convert components1[index] to integer
        SET value2 TO convert components2[index] to integer

        // Check if the values are different
        IF value1 IS NOT equal to value2 THEN
            INCREMENT differenceCount by 1 
        END IF
    END FOR

    // Evaluate the number of differences
    IF differenceCount IS less than 3 THEN
        PRINT "YES"
    ELSE
        PRINT "NO"
    END IF
END FUNCTION

// Start program execution
CALL main
```
