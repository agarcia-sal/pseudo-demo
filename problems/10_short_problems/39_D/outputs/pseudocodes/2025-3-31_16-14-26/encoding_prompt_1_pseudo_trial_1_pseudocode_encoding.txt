```
FUNCTION main()
    // Read two lines of input from the user
    SET firstInput TO input("Enter the first set of values:")
    SET secondInput TO input("Enter the second set of values:")

    // Split the input strings into lists of values
    SET firstValues TO split firstInput into a list
    SET secondValues TO split secondInput into a list
    
    // Initialize a counter for mismatches
    SET mismatchCount TO 0 
    
    // Loop through the first three values of both lists
    FOR index FROM 0 TO 2 DO
        // Convert the string values to integers for comparison
        SET valueA TO convert firstValues[index] to integer
        SET valueB TO convert secondValues[index] to integer
        
        // Check if the values are not equal
        IF valueA is not equal to valueB THEN
            // Increment mismatch counter
            INCREMENT mismatchCount by 1 
        END IF
    END FOR
    
    // Check the number of mismatches
    IF mismatchCount is less than 3 THEN
        OUTPUT "YES"
    ELSE
        OUTPUT "NO"
    END IF
END FUNCTION

// Execute the main function when the program runs
CALL main()
``` 
