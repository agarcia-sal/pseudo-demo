FUNCTION Main
    // Read two lines of input
    SET firstLine TO userInput()
    SET secondLine TO userInput()

    // Split the input lines into lists of strings
    SET firstList TO split firstLine into words
    SET secondList TO split secondLine into words

    // Initialize a counter for differences
    SET differenceCount TO 0 

    // Loop through the first three elements of both lists
    FOR index FROM 0 TO 2 DO
        // Convert the current elements to integers
        SET firstValue TO convert firstList[index] to integer
        SET secondValue TO convert secondList[index] to integer

        // If the current elements are not equal, increment the difference counter
        IF firstValue IS NOT EQUAL TO secondValue THEN
            INCREMENT differenceCount BY 1
        ENDIF
    END FOR

    // Check the number of differences
    IF differenceCount IS LESS THAN 3 THEN
        PRINT "YES"
    ELSE
        PRINT "NO"
    ENDIF
END FUNCTION

// Start the program
CALL Main
