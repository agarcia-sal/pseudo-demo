FUNCTION Main
    // Read input values
    DECLARE t1 AS STRING
    DECLARE t2 AS STRING
    t1 = READ input from user
    t2 = READ input from user
    
    // Split input strings into lists of strings
    DECLARE tt1 AS LIST OF STRING
    DECLARE tt2 AS LIST OF STRING
    tt1 = SPLIT t1 into words
    tt2 = SPLIT t2 into words
    
    // Initialize a count of differences
    DECLARE differenceCount AS INTEGER
    differenceCount = 0 
    
    // Compare corresponding elements of the two lists
    FOR i FROM 0 TO 2 DO
        DECLARE a AS INTEGER
        DECLARE b AS INTEGER
        a = CONVERT tt1[i] to INTEGER
        b = CONVERT tt2[i] to INTEGER
        
        IF a IS NOT EQUAL TO b THEN
            INCREMENT differenceCount by 1 
        END IF
    END FOR
    
    // Determine if the differences are fewer than 3
    IF differenceCount < 3 THEN
        PRINT "YES"
    ELSE
        PRINT "NO"
    END IF
END FUNCTION

// Execute the main function
IF this is the main module THEN
    CALL Main
END IF
