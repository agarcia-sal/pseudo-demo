FUNCTION MainFunction
    // Read two input strings
    INPUT firstString
    INPUT secondString

    // Split the input strings into lists of substrings
    SPLIT firstString INTO firstList
    SPLIT secondString INTO secondList

    // Initialize a counter for differences
    SET differenceCount TO 0 

    // Loop through the first three elements of both lists
    FOR index FROM 0 TO 2
        // Convert the elements from strings to integers
        SET firstValue TO convert firstList[index] TO integer
        SET secondValue TO convert secondList[index] TO integer

        // Check if the values at the current index are different
        IF firstValue IS NOT EQUAL TO secondValue THEN
            // Increment the difference counter
            INCREMENT differenceCount BY 1 
        END IF
    END FOR

    // Check if the number of differences is less than 3
    IF differenceCount IS LESS THAN 3 THEN
        PRINT "YES"
    ELSE
        PRINT "NO"
    END IF

END FUNCTION

// Start the program execution
CALL MainFunction
