FUNCTION doMain()
    // Read two lines of input
    INPUT string t1
    INPUT string t2

    // Split the input strings into lists of strings
    LIST string tt1 = SPLIT t1 BY spaces
    LIST string tt2 = SPLIT t2 BY spaces
    
    // Initialize a counter for differences
    INTEGER res = 0 
    
    // Compare corresponding elements of the two lists
    FOR each index x from 0 to 2 DO
        INTEGER a = CONVERT tt1[x] TO integer
        INTEGER b = CONVERT tt2[x] TO integer
        
        // Count how many elements are different
        IF a is NOT equal to b THEN
            INCREMENT res by 1 
        END IF
    END FOR

    // Determine if differences are less than 3
    IF res is LESS than 3 THEN
        PRINT "YES"
    ELSE
        PRINT "NO"
    END IF
END FUNCTION

// Main execution starts here
CALL doMain()
