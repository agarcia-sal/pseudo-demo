FUNCTION doMain
    // Read two lines of input
    t1 = READ input from user
    t2 = READ input from user
    
    // Split the input strings into lists
    tt1 = SPLIT t1 by spaces
    tt2 = SPLIT t2 by spaces
    
    // Initialize a counter for differences
    differenceCount = 0 
    
    // Loop through the first three elements of the lists
    FOR index FROM 0 TO 2 DO
        // Convert string elements to integers
        a = CONVERT tt1[index] to integer
        b = CONVERT tt2[index] to integer
        
        // Check if the values are different
        IF a is NOT equal to b THEN
            differenceCount = differenceCount + 1 
    
    // Determine and print the result based on the count of differences
    IF differenceCount is LESS THAN 3 THEN
        PRINT "YES"
    ELSE 
        PRINT "NO"
END FUNCTION

// Entry point of the program
IF this file is being executed THEN
    CALL doMain
END IF
