```pseudocode
FUNCTION main()
    // Read input values
    INPUT firstTime
    INPUT secondTime
    
    // Split the input strings into lists
    SPLIT firstTime INTO components1
    SPLIT secondTime INTO components2
    
    // Initialize a counter for differences
    SET differenceCount = 0 
    
    // Compare corresponding components of both times
    FOR index FROM 0 TO 2 DO
        // Convert components to integers
        SET value1 = CONVERT components1[index] TO INTEGER
        SET value2 = CONVERT components2[index] TO INTEGER
        
        // Check if the values are different
        IF value1 IS NOT EQUAL TO value2 THEN
            INCREMENT differenceCount
        END IF
    END FOR
    
    // Determine result based on the number of differences
    IF differenceCount IS LESS THAN 3 THEN
        PRINT "YES"
    ELSE
        PRINT "NO"
    END IF
END FUNCTION

// Start the main function when the script runs
CALL main()
```
