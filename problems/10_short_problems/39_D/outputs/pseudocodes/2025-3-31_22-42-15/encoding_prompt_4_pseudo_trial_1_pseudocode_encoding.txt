FUNCTION doMain
    // Read two lines of input
    INPUT firstLine
    INPUT secondLine
    
    // Split input lines into lists of strings
    SPLIT firstLine INTO list firstList
    SPLIT secondLine INTO list secondList
    
    // Initialize the result counter
    SET mismatchCount TO 0
    
    // Iterate over the first three elements of both lists
    FOR index FROM 0 TO 2 DO
        // Convert string elements to integers
        SET firstValue TO INTEGER(firstList[index])
        SET secondValue TO INTEGER(secondList[index])
        
        // Check for mismatches
        IF firstValue NOT EQUAL TO secondValue THEN
            INCREMENT mismatchCount BY 1
        END IF
    END FOR
    
    // Determine the output based on mismatch count
    IF mismatchCount LESS THAN 3 THEN
        PRINT "YES"
    ELSE
        PRINT "NO"
    END IF
END FUNCTION

// Main execution block
IF this script is the main program THEN
    CALL doMain()
END IF
