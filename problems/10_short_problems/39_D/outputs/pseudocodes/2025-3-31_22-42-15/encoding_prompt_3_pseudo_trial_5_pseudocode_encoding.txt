FUNCTION Main
    // Step 1: Gather input data
    PROMPT user for first line of input and store it in variable firstInput
    PROMPT user for second line of input and store it in variable secondInput
    
    // Step 2: Split the input strings into lists of strings
    SPLIT firstInput by spaces and store it in variable firstList
    SPLIT secondInput by spaces and store it in variable secondList
    
    // Step 3: Initialize a counter for differences
    SET differenceCount to 0 
    
    // Step 4: Iterate through the first three elements of both lists
    FOR index FROM 0 TO 2 DO
        // Convert the corresponding elements to integers
        SET firstValue to integer value of firstList[index]
        SET secondValue to integer value of secondList[index]
        
        // Check if the two values are different
        IF firstValue is NOT equal to secondValue THEN
            INCREMENT differenceCount by 1 
        END IF
    END FOR
    
    // Step 5: Determine the output based on the count of differences
    IF differenceCount is LESS THAN 3 THEN
        PRINT "YES"
    ELSE
        PRINT "NO"
    END IF
END FUNCTION

// Execute the Main function when the program starts
IF program is run directly THEN
    CALL Main
END IF
