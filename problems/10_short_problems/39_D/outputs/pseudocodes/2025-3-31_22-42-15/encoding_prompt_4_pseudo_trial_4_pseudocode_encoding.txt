FUNCTION Main
    // Read input values
    INPUT firstString
    INPUT secondString
    
    // Split the input strings into lists
    SET firstList TO firstString SPLIT BY SPACE
    SET secondList TO secondString SPLIT BY SPACE
    
    // Initialize a mismatch counter
    SET mismatchCount TO 0 
    
    // Compare the first three elements of both lists
    FOR index FROM 0 TO 2 DO
        SET firstValue TO INTEGER firstList[index]
        SET secondValue TO INTEGER secondList[index]
        
        // If the values do not match, increment the mismatch counter
        IF firstValue NOT EQUAL TO secondValue THEN
            INCREMENT mismatchCount
        END IF
    END FOR
    
    // Check the number of mismatches and print the result
    IF mismatchCount LESS THAN 3 THEN
        OUTPUT "YES"
    ELSE 
        OUTPUT "NO"
    END IF
END FUNCTION

// Entry point of the program
IF program is executed THEN
    CALL Main
END IF
