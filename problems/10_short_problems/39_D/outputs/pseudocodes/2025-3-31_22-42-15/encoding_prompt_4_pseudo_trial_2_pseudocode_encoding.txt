```
FUNCTION doMain()
    // Read input values
    INPUT firstLine
    INPUT secondLine
    
    // Split input lines into lists of strings
    SPLIT firstLine INTO list1
    SPLIT secondLine INTO list2
    
    // Initialize a counter for differences
    SET differenceCount TO 0 
    
    // Loop through the first three elements of both lists
    FOR index FROM 0 TO 2 DO
        // Convert the string elements to integers
        SET valueA TO CONVERT list1[index] TO INTEGER
        SET valueB TO CONVERT list2[index] TO INTEGER
        
        // Check if values are different
        IF valueA IS NOT EQUAL TO valueB THEN
            // Increment the difference counter
            INCREMENT differenceCount
        ENDIF
    ENDFOR
    
    // Check the number of differences
    IF differenceCount IS LESS THAN 3 THEN
        OUTPUT "YES"
    ELSE
        OUTPUT "NO"
    ENDIF
END FUNCTION

// Main execution starts here
CALL doMain()
```
