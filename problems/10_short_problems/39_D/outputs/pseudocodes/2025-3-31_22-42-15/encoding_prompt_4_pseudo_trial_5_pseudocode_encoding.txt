FUNCTION doMain
    // Read input strings
    INPUT string t1
    INPUT string t2

    // Split the input strings into lists of values
    SPLIT t1 into list tt1
    SPLIT t2 into list tt2

    // Initialize a variable to count differences
    SET differenceCount to 0

    // Iterate over the first three elements of both lists
    FOR each index from 0 to 2 DO
        SET valueA to INTEGER of tt1[index]
        SET valueB to INTEGER of tt2[index]

        // Check if the values are different
        IF valueA is NOT equal to valueB THEN
            INCREMENT differenceCount by 1

    // Check the number of differences
    IF differenceCount is less than 3 THEN
        PRINT "YES"
    ELSE
        PRINT "NO"
END FUNCTION

// Main execution
CALL doMain
