FUNCTION doMain()
    // Get two sets of input from the user
    INPUT firstInput
    INPUT secondInput
    
    // Split the inputs into lists of strings
    SPLIT firstInput INTO firstList
    SPLIT secondInput INTO secondList
    
    // Initialize a variable to count differences
    INITIALIZE differenceCount AS 0
    
    // Iterate through the first three elements of both lists
    FOR index FROM 0 TO 2 DO
        // Convert the string values to integers
        SET firstValue TO integer value of firstList[index]
        SET secondValue TO integer value of secondList[index]
        
        // Check if the values are different
        IF firstValue IS NOT EQUAL TO secondValue THEN
            INCREMENT differenceCount BY 1
        END IF
    END FOR
    
    // Determine if the number of differences is less than 3
    IF differenceCount IS LESS THAN 3 THEN
        PRINT "YES"
    ELSE 
        PRINT "NO"
    END IF
END FUNCTION

// Main execution block
IF this script is the main module THEN
    CALL doMain()
END IF
