// Function to compare two sets of three integers and return a verdict
FUNCTION doMain()
    // Read two lines of input
    INPUT firstLine
    INPUT secondLine
    
    // Split the input strings into lists of integers
    SPLIT firstLine INTO firstList
    SPLIT secondLine INTO secondList
    
    // Initialize a counter for differences
    SET differenceCount TO 0
    
    // Compare corresponding elements in the two lists
    FOR index FROM 0 TO 2 DO
        // Convert string to integer for comparison
        SET firstNumber TO integer of firstList[index]
        SET secondNumber TO integer of secondList[index]
        
        // If the numbers are not equal, increment the difference counter
        IF firstNumber IS NOT EQUAL TO secondNumber THEN
            INCREMENT differenceCount BY 1
        ENDIF
    ENDFOR
    
    // If differences are less than 3, print "YES", otherwise print "NO"
    IF differenceCount < 3 THEN
        PRINT "YES"
    ELSE
        PRINT "NO"
    ENDIF
END FUNCTION

// Entry point of the program
IF this script is executed THEN
    CALL doMain()
ENDIF
