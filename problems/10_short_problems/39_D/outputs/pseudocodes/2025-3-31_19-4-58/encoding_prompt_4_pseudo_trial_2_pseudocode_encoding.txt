FUNCTION doMain
    // Read inputs for two sets of three integers
    INPUT firstInput FROM USER
    INPUT secondInput FROM USER

    // Split inputs into lists of strings
    SPLIT firstInput INTO firstList
    SPLIT secondInput INTO secondList

    // Initialize a counter for mismatches
    SET mismatchCount TO 0 

    // Loop through the first three elements of both lists
    FOR index FROM 0 TO 2 DO
        // Convert the string elements to integers
        SET firstValue TO INTEGER(firstList[index])
        SET secondValue TO INTEGER(secondList[index])
        
        // Check if the corresponding values are different
        IF firstValue IS NOT EQUAL TO secondValue THEN
            // Increment the mismatch count
            INCREMENT mismatchCount

    // Determine if the number of mismatches is less than 3
    IF mismatchCount IS LESS THAN 3 THEN
        PRINT "YES"
    ELSE
        PRINT "NO"

END FUNCTION

// Main execution
CALL doMain()
