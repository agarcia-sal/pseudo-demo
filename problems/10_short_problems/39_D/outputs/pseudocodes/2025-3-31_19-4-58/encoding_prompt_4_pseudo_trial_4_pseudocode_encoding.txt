FUNCTION main
    // Read input values
    t1 = GET user input for first string
    t2 = GET user input for second string

    // Split the input strings into lists
    tt1 = SPLIT t1 into a list
    tt2 = SPLIT t2 into a list
    
    // Initialize the result counter
    differenceCount = 0 

    // Compare the first three elements of both lists
    FOR each index FROM 0 TO 2 DO
        a = CONVERT tt1[index] to integer
        b = CONVERT tt2[index] to integer
        IF a is NOT equal to b THEN
            INCREMENT differenceCount by 1
        END IF
    END FOR

    // Check if the number of differences is less than 3
    IF differenceCount is less than 3 THEN
        PRINT "YES"
    ELSE
        PRINT "NO"
    END IF
END FUNCTION

// Execute the main function when the script runs
CALL main()
