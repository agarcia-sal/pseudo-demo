    // PROMPT user to input two strings
    DECLARE string t1

    SET t1 = INPUT("Enter first string: ")
    SET t2 = INPUT("Enter second string: ")

    // SPLIT the input strings into lists
    DECLARE list tt1
    DECLARE list tt2
    SET tt1 = SPLIT(t1)
    SET tt2 = SPLIT(t2)

    // INITIALIZE a counter for differences
    DECLARE integer res
    SET res = 0 

    // LOOP through the first three elements of both lists
    FOR x FROM 0 TO 2 DO
        // CONVERT current elements to integers
        DECLARE integer a
        DECLARE integer b
        SET a = CONVERT_TO_INTEGER(tt1[x])
        SET b = CONVERT_TO_INTEGER(tt2[x])

        // CHECK if the elements differ
        IF a IS NOT EQUAL TO b THEN
            INCREMENT res BY 1 
        END IF
    END FOR

    // CHECK the number of differences
    IF res IS LESS THAN 3 THEN
        OUTPUT "YES"
    ELSE
        OUTPUT "NO"
    END IF
END FUNCTION

IF THIS_MODULE_IS_MAIN THEN
