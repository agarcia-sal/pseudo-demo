**Pseudocode:**

BEGIN FUNCTION doMain

    // Get two lines of input from the user
    READ firstInput
    READ secondInput
    
    // Split the inputs into separate components
    splitFirstInput = SPLIT firstInput into individual elements
    splitSecondInput = SPLIT secondInput into individual elements
    
    // Initialize a variable to count the number of differences
    differenceCount = 0 

    // Iterate through the first three elements of both inputs
    FOR index FROM 0 TO 2 DO
        // Convert the current elements to integers
        firstValue = CONVERT splitFirstInput[index] TO INTEGER
        secondValue = CONVERT splitSecondInput[index] TO INTEGER
        
        // Check if the two values are different
        IF firstValue IS NOT EQUAL TO secondValue THEN
            // Increment the count of differences
            differenceCount = differenceCount + 1 
        END IF
    END FOR

    // If the number of differences is less than 3, print "YES", otherwise print "NO"
    IF differenceCount < 3 THEN
        PRINT "YES"
    ELSE
        PRINT "NO"
    END IF

END FUNCTION

// Main execution flow
CALL doMain()
