FUNCTION main
    // Read input values
    INPUT t1
    INPUT t2
    
    // Split input strings into lists of strings
    SPLIT t1 INTO list tt1
    SPLIT t2 INTO list tt2
    
    // Initialize a counter for differences
    SET differenceCount TO 0 
    
    // Loop through the first three elements of the lists
    FOR i FROM 0 TO 2 DO
        // Convert string elements to integers
        SET a TO INTEGER OF tt1[i]
        SET b TO INTEGER OF tt2[i]
        
        // Check if the integers are different
        IF a IS NOT EQUAL TO b THEN
            INCREMENT differenceCount BY 1 
        END IF
    END FOR
    
    // Determine the output based on the number of differences
    IF differenceCount IS LESS THAN 3 THEN
        OUTPUT "YES"
    ELSE
        OUTPUT "NO"
    END IF
END FUNCTION

// Entry point for the program
CALL main
