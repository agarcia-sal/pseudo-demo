FUNCTION doMain
    // Read input for two sets of values
    DECLARE t1 AS STRING
    DECLARE t2 AS STRING
    INPUT t1
    INPUT t2

    // Split the input strings into lists of values
    DECLARE tt1 AS LIST OF STRING
    DECLARE tt2 AS LIST OF STRING
    tt1 = SPLIT(t1)
    tt2 = SPLIT(t2)

    // Initialize a counter for differences
    DECLARE differenceCount AS INTEGER
    differenceCount = 0 

    // Compare the corresponding elements of the two lists
    FOR i FROM 0 TO 2 DO
        DECLARE valueA AS INTEGER
        DECLARE valueB AS INTEGER
        valueA = CONVERT tt1[i] TO INTEGER
        valueB = CONVERT tt2[i] TO INTEGER

        // If values are not equal, increase the difference counter
        IF valueA IS NOT EQUAL TO valueB THEN
            differenceCount = differenceCount + 1 
        END IF
    END FOR

    // Determine if the count of differences is less than 3
    IF differenceCount < 3 THEN
        PRINT "YES"
    ELSE
        PRINT "NO"
    END IF
END FUNCTION

// Main execution point
IF this script is run as the main program THEN
    CALL doMain
END IF
