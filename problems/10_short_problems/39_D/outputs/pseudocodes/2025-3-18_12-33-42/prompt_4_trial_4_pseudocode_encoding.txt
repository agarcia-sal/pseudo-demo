FUNCTION doMain
    // Read two lines of input from the user
    INPUT firstLine
    INPUT secondLine
    
    // Split the input lines into lists of strings
    SPLIT firstLine INTO list firstTokens
    SPLIT secondLine INTO list secondTokens
    
    // Initialize a variable to count differences
    SET differenceCount TO 0 
    
    // Iterate over the first three elements of the token lists
    FOR index FROM 0 TO 2 DO
        // Convert string tokens to integers
        SET firstValue TO INTEGER(firstTokens[index])
        SET secondValue TO INTEGER(secondTokens[index])
        
        // Check for differences between the two values
        IF firstValue IS NOT EQUAL TO secondValue THEN
            INCREMENT differenceCount BY 1 
        END IF
    END FOR
    
    // Assess the number of differences and print the result
    IF differenceCount IS LESS THAN 3 THEN
        PRINT "YES"
    ELSE
        PRINT "NO"
    END IF
END FUNCTION

// Main execution starts here
CALL doMain
