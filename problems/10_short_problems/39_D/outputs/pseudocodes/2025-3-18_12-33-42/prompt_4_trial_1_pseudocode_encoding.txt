FUNCTION doMain
    // Read two lines of input
    INPUT t1
    INPUT t2

    // Split the input strings into lists
    SPLIT tt1 FROM t1
    SPLIT tt2 FROM t2

    // Initialize a variable to count differences
    SET differenceCount TO 0 

    // Iterate through the three corresponding elements of both lists
    FOR index FROM 0 TO 2 DO
        // Convert the current elements to integers
        SET a TO integer value of tt1[index]
        SET b TO integer value of tt2[index]

        // If elements are different, increment the difference count
        IF a NOT EQUAL TO b THEN
            INCREMENT differenceCount
        END IF
    END FOR

    // Determine if the number of differences is less than 3
    IF differenceCount < 3 THEN
        PRINT "YES"
    ELSE
        PRINT "NO"
    END IF
END FUNCTION

// Main execution
CALL doMain
