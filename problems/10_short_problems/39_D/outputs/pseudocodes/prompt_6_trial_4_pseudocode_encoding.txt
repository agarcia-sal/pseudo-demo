```plaintext
FUNCTION main:
    // Read two lines of input
    INPUT firstString
    INPUT secondString
    
    // Split the input strings into lists of substrings
    SPLIT firstString INTO list firstList
    SPLIT secondString INTO list secondList
    
    // Initialize a variable to count differences
    SET differenceCount TO 0 
    
    // Loop through the first three elements of both lists
    FOR index FROM 0 TO 2:
        // Convert the current elements of both lists to integers
        SET firstValue TO CONVERT firstList[index] TO INTEGER
        SET secondValue TO CONVERT secondList[index] TO INTEGER
        
        // Check if the values are different
        IF firstValue IS NOT EQUAL TO secondValue:
            // Increment the count of differences
            INCREMENT differenceCount BY 1
            
    // Check the number of differences counted
    IF differenceCount IS LESS THAN 3:
        // If differences are less than 3, print "YES"
        PRINT "YES"
    ELSE:
        // Otherwise, print "NO"
        PRINT "NO"

// Call the main function if this script is executed
IF this script is executed:
    CALL main
```
