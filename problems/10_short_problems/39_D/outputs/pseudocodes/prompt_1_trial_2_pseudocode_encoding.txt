FUNCTION main
    // Read two lines of input, each containing three integers
    INPUT firstLine
    INPUT secondLine

    // Split the input lines into lists of strings
    SPLIT firstLine INTO firstList
    SPLIT secondLine INTO secondList

    // Initialize a counter for mismatches
    SET mismatchCount TO 0 

    // Loop through the first three elements of both lists
    FOR index FROM 0 TO 2 DO
        // Convert the string values to integers
        SET firstValue TO CONVERT firstList[index] TO INTEGER
        SET secondValue TO CONVERT secondList[index] TO INTEGER

        // Check if the values are different
        IF firstValue IS NOT EQUAL TO secondValue THEN
            // If they are different, increment the mismatch counter
            INCREMENT mismatchCount BY 1 
        END IF
    END FOR

    // Determine if the number of mismatches is less than three
    IF mismatchCount IS LESS THAN 3 THEN
        PRINT "YES" // If fewer than three mismatches, output "YES"
    ELSE
        PRINT "NO"  // If three or more mismatches, output "NO"
    END IF
END FUNCTION

CALL main() // Execute the main function
