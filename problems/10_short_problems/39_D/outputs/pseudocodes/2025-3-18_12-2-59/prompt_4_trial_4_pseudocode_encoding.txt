FUNCTION doMain
    // Read two lines of input from the user
    SET firstInput TO user input
    SET secondInput TO user input

    // Split the inputs into lists of strings
    SET firstList TO split(firstInput)
    SET secondList TO split(secondInput)

    // Initialize a counter for differences
    SET differenceCount TO 0 

    // Iterate over the first three elements of the lists
    FOR index FROM 0 TO 2 DO
        // Convert the elements to integers
        SET valueA TO integer of firstList[index]
        SET valueB TO integer of secondList[index]

        // Compare the two values
        IF valueA IS NOT EQUAL TO valueB THEN
            INCREMENT differenceCount by 1 
        ENDIF
    END FOR

    // Check the number of differences and print the result
    IF differenceCount < 3 THEN
        PRINT "YES"
    ELSE
        PRINT "NO"
    ENDIF
END FUNCTION

// Execute the main function when the script is run
IF this script is executed THEN
    CALL doMain()
ENDIF
