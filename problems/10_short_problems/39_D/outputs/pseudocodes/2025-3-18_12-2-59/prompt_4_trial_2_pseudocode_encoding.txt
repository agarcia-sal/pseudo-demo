FUNCTION main()
    // Read input strings
    INPUT firstString
    INPUT secondString
    
    // Split the input strings into lists of strings
    SPLIT firstList INTO firstString.split()
    SPLIT secondList INTO secondString.split()
    
    // Initialize a counter for differences
    SET differenceCount TO 0 
    
    // Loop through the first three elements of both lists
    FOR index FROM 0 TO 2 DO
        // Convert each element to an integer
        SET firstValue TO CONVERT firstList[index] TO INTEGER
        SET secondValue TO CONVERT secondList[index] TO INTEGER
        
        // If the values are not equal, increment the difference counter
        IF firstValue IS NOT EQUAL TO secondValue THEN
            INCREMENT differenceCount BY 1 
        END IF
    END FOR
    
    // Determine if the number of differences is less than 3
    IF differenceCount IS LESS THAN 3 THEN
        PRINT "YES"
    ELSE
        PRINT "NO"
    END IF
END FUNCTION

// Start execution
CALL main()
