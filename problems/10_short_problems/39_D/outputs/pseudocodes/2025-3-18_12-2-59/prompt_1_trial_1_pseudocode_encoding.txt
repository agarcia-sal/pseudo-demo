FUNCTION main()
    // Read two lines of input representing two sets of values
    INPUT firstSet
    INPUT secondSet
    
    // Split the input strings into lists of values
    SPLIT firstSet INTO values1
    SPLIT secondSet INTO values2
    
    // Initialize a counter for the number of differing values
    SET differingCount TO 0 
    
    // Loop through the first three elements in both lists
    FOR each index FROM 0 TO 2 DO
        // Convert the current elements to integers
        SET valueFromFirstSet TO integer of values1[index]
        SET valueFromSecondSet TO integer of values2[index]
        
        // Check if the current elements are different
        IF valueFromFirstSet IS NOT equal to valueFromSecondSet THEN
            // Increment the counter for differing values
            INCREMENT differingCount
        END IF
    END FOR
    
    // Evaluate the count of differing values
    IF differingCount IS LESS THAN 3 THEN
        // If less than 3 differences, output "YES"
        OUTPUT "YES"
    ELSE
        // If 3 or more differences, output "NO"
        OUTPUT "NO"
    END IF
END FUNCTION

// Start the program execution
CALL main()
