### Pseudocode

BEGIN

    FUNCTION doMain()
        // Read two lines of input from the user
        INPUT firstLine
        INPUT secondLine
        
        // Split the input lines into lists of strings
        SPLIT firstLine into list firstValues
        SPLIT secondLine into list secondValues
        
        // Initialize a counter for differences
        SET differenceCount TO 0
        
        // Loop through the first three elements of both lists
        FOR index FROM 0 TO 2 DO
            // Convert the string values to integers
            SET firstValue TO CONVERT firstValues[index] TO INTEGER
            SET secondValue TO CONVERT secondValues[index] TO INTEGER
            
            // Check for equality
            IF firstValue IS NOT EQUAL TO secondValue THEN
                // Increment the difference counter
                INCREMENT differenceCount
            END IF
        END FOR
        
        // Check how many differences were counted
        IF differenceCount IS LESS THAN 3 THEN
            OUTPUT "YES"
        ELSE
            OUTPUT "NO"
        END IF
    END FUNCTION
    
    // Entry point of the program
    CALL doMain()

END
