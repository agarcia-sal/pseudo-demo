**Pseudocode:**

BEGIN

    // Read two lines of input from the user
    SET firstInputString TO user input
    SET secondInputString TO user input
    
    // Split the input strings into lists of elements
    SET firstInputList TO split firstInputString into individual elements
    SET secondInputList TO split secondInputString into individual elements
    
    // Initialize a counter for differences
    SET differenceCounter TO 0 
    
    // Compare elements in the two lists
    FOR each index from 0 to 2 DO
        SET firstValue TO convert firstInputList[index] to an integer
        SET secondValue TO convert secondInputList[index] to an integer
        
        // Check if the corresponding values are different
        IF firstValue is NOT equal to secondValue THEN
            INCREMENT differenceCounter by 1 
        END IF
    END FOR
    
    // Evaluate the number of differences and provide output
    IF differenceCounter is less than 3 THEN
        PRINT "YES"
    ELSE 
        PRINT "NO"
    END IF

END
