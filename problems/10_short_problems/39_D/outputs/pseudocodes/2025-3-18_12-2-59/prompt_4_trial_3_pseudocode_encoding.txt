```
FUNCTION main
    // Read two lines of input
    INPUT line1
    INPUT line2
    
    // Split each input line into a list of strings
    SPLIT line1 INTO list1
    SPLIT line2 INTO list2
    
    // Initialize discrepancy counter
    SET mismatchCount TO 0 
    
    // Iterate through first three elements of both lists
    FOR index FROM 0 TO 2 DO
        // Convert strings to integers
        SET value1 TO INTEGER(list1[index])
        SET value2 TO INTEGER(list2[index])
        
        // Check if the values are different
        IF value1 IS NOT EQUAL TO value2 THEN
            INCREMENT mismatchCount
        END IF
    END FOR
    
    // Determine and output the result based on mismatch count
    IF mismatchCount LESS THAN 3 THEN
        OUTPUT "YES"
    ELSE
        OUTPUT "NO"
    END IF
END FUNCTION

// Start the program
CALL main
```
