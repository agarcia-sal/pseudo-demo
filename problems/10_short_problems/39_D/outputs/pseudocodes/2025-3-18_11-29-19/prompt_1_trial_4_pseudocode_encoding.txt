FUNCTION main:
    // Read two lines of input
    INPUT firstString
    INPUT secondString

    // Split the input strings into lists of words
    SPLIT firstString INTO list1
    SPLIT secondString INTO list2

    // Initialize a counter for differences
    differencesCount = 0 

    // Iterate over the first three elements of both lists
    FOR index FROM 0 TO 2:
        // Convert the elements to integers
        integer1 = CONVERT(list1[index] TO INTEGER)
        integer2 = CONVERT(list2[index] TO INTEGER)

        // Check if the integers are different
        IF integer1 IS NOT EQUAL TO integer2 THEN:
            // Increment the differences counter
            differencesCount = differencesCount + 1 

    // Check if the number of differences is less than 3
    IF differencesCount < 3 THEN:
        OUTPUT "YES"
    ELSE:
        OUTPUT "NO"

// Entry point of the program
CALL main
