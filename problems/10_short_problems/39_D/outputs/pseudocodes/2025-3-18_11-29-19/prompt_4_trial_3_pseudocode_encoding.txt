```pseudocode
FUNCTION main
    // Read two lines of input
    INPUT string1
    INPUT string2

    // Split the input strings into lists of string numbers
    LIST numbers1 = SPLIT string1 INTO LIST
    LIST numbers2 = SPLIT string2 INTO LIST

    // Initialize a counter for differences
    DIFFERENCE_COUNT = 0 

    // Loop through the first three elements of both lists
    FOR index FROM 0 TO 2 DO
        // Convert string numbers to integers
        INTEGER num1 = CONVERT numbers1[index] TO INTEGER
        INTEGER num2 = CONVERT numbers2[index] TO INTEGER
        
        // Check if the numbers are different
        IF num1 IS NOT EQUAL TO num2 THEN
            INCREMENT DIFFERENCE_COUNT
        ENDIF
    ENDFOR

    // Evaluate the number of differences to determine the output
    IF DIFFERENCE_COUNT IS LESS THAN 3 THEN
        PRINT "YES"
    ELSE
        PRINT "NO"
    ENDIF
END FUNCTION

// Start the program
CALL main
```
