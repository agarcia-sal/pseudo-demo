FUNCTION doMain

    // Prompt the user for the first time input and store it
    PROMPT user for first time input (t1)
    
    // Prompt the user for the second time input and store it
    PROMPT user for second time input (t2)

    // Split the inputs into lists of time components
    SPLIT t1 into a list called timeComponents1 (tt1)
    SPLIT t2 into a list called timeComponents2 (tt2)

    // Initialize a counter for differences
    SET differenceCount to 0 

    // Loop through the first three elements of each input
    FOR each index from 0 to 2 DO
        // Convert the corresponding time components to integers
        SET componentA to INTEGER value of tt1[index]
        SET componentB to INTEGER value of tt2[index]
        
        // If the components from the two times are not equal
        IF componentA is NOT equal to componentB THEN
            // Increment the difference counter
            INCREMENT differenceCount by 1 
        END IF
    END FOR

    // Check the number of differences
    IF differenceCount is LESS than 3 THEN
        // Output "YES" if differences are fewer than 3
        PRINT "YES"
    ELSE 
        // Output "NO" if differences are 3 or more
        PRINT "NO"
    END IF

END FUNCTION

// Main program execution
