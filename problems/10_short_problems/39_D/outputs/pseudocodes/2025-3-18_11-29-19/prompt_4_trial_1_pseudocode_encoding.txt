```pseudocode
FUNCTION Main
    // Read two lines of input from the user
    INPUT firstLine
    INPUT secondLine
    
    // Split the input lines into lists of strings
    SET firstList TO SPLIT firstLine INTO WORDS
    SET secondList TO SPLIT secondLine INTO WORDS
    
    // Initialize a counter for differences
    SET differenceCount TO 0 
    
    // Loop through the first three elements of both lists
    FOR index FROM 0 TO 2 DO
        // Convert the string elements to integers
        SET numberFromFirstList TO CONVERT firstList[index] TO INTEGER
        SET numberFromSecondList TO CONVERT secondList[index] TO INTEGER
        
        // Check if the two numbers are different
        IF numberFromFirstList IS NOT EQUAL TO numberFromSecondList THEN
            INCREMENT differenceCount BY 1
        END IF
    END FOR
    
    // If less than 3 differences are found, output "YES", otherwise output "NO"
    IF differenceCount < 3 THEN
        PRINT "YES"
    ELSE
        PRINT "NO"
    END IF
END FUNCTION

// Execution starts here
CALL Main
```
