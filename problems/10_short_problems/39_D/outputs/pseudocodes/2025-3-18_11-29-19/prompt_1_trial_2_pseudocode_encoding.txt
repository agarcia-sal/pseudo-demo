FUNCTION MainProcedure
    // Read two lines of input
    STRING firstInput = Get user input
    STRING secondInput = Get user input
    
    // Split the input strings into lists of values
    LIST firstList = Split firstInput by spaces
    LIST secondList = Split secondInput by spaces
    
    // Initialize a counter to track the number of differences
    INTEGER differenceCount = 0 
    
    // Iterate through the first three elements of both lists
    FOR index FROM 0 TO 2 DO
        // Convert the current element from both lists to integers
        INTEGER firstValue = Convert firstList[index] to integer
        INTEGER secondValue = Convert secondList[index] to integer
        
        // Check if the two values are different
        IF firstValue is NOT equal to secondValue THEN
            // Increment the difference counter
            differenceCount = differenceCount + 1 
        ENDIF
    ENDFOR
    
    // Check the count of differences
    IF differenceCount is LESS than 3 THEN
        // Print "YES" if there are less than 3 differences
        Print "YES"
    ELSE 
        // Print "NO" if there are 3 or more differences
        Print "NO"
    ENDIF
END FUNCTION

// Main execution starts here
CALL MainProcedure
