FUNCTION doMain
    // Read two lines of input
    SET input1 AS user input
    SET input2 AS user input

    // Split input lines into lists of strings
    SET list1 AS split input1
    SET list2 AS split input2

    // Initialize a counter for differences
    SET differenceCount TO 0 

    // Compare each element of the two lists
    FOR index FROM 0 TO 2 DO
        SET value1 AS integer of list1[index]
        SET value2 AS integer of list2[index]

        // If elements are not equal, increment the counter
        IF value1 IS NOT EQUAL TO value2 THEN
            INCREMENT differenceCount

    // If differences are less than 3, print "YES", else print "NO"
    IF differenceCount IS LESS THAN 3 THEN
        PRINT "YES"
    ELSE
        PRINT "NO"
END FUNCTION

// Main execution starting point
IF script is the main program THEN
    CALL doMain
END IF
