FUNCTION main()
    // Read two input strings
    READ firstInputString
    READ secondInputString
    
    // Split the input strings into lists of substrings
    SPLIT firstInputString INTO list1
    SPLIT secondInputString INTO list2
    
    // Initialize a counter for differences
    SET differenceCounter = 0 
    
    // Loop through the first three elements of the lists
    FOR index FROM 0 TO 2 DO
        // Convert the current elements to integers
        SET valueFromList1 = CONVERT list1[index] TO INTEGER
        SET valueFromList2 = CONVERT list2[index] TO INTEGER
        
        // Check if the values are different
        IF valueFromList1 IS NOT EQUAL TO valueFromList2 THEN
            INCREMENT differenceCounter BY 1 
        END IF
    END FOR
    
    // Check the number of differences and print the result
    IF differenceCounter < 3 THEN
        PRINT "YES"
    ELSE 
        PRINT "NO"
    END IF
END FUNCTION

// Start the program execution
IF this file is the main module THEN
    CALL main()
END IF
