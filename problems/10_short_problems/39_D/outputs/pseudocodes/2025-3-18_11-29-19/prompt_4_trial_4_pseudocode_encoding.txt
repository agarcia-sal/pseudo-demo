FUNCTION doMain()
    // Read two lines of input
    INPUT t1
    INPUT t2
    
    // Split the input strings into lists
    SET tt1 = SPLIT t1 INTO LIST
    SET tt2 = SPLIT t2 INTO LIST
    
    // Initialize a counter for differences
    SET res = 0 

    // Iterate through the first three elements of the lists
    FOR each index x from 0 to 2 DO
        // Convert elements to integers for comparison
        SET a = CONVERT tt1[x] TO INTEGER
        SET b = CONVERT tt2[x] TO INTEGER
        
        // Increment the difference counter if the elements are not equal
        IF a is NOT equal to b THEN
            INCREMENT res by 1 
        END IF
    END FOR
    
    // Check the number of differences found
    IF res is less than 3 THEN
        PRINT "YES"
    ELSE
        PRINT "NO"
    END IF
END FUNCTION

// Main entry point for the program
IF this file is the main module THEN
    CALL doMain()
END IF
