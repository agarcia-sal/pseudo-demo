**Pseudocode**:

```
FUNCTION main
    SET firstInput TO user input
    SET secondInput TO user input
    
    SET firstList TO split firstInput into words
    SET secondList TO split secondInput into words
    
    SET differenceCount TO 0 
    
    FOR index FROM 0 TO 2 DO
        SET numberFromFirst TO convert firstList[index] to integer
        SET numberFromSecond TO convert secondList[index] to integer
        
        IF numberFromFirst is not equal to numberFromSecond THEN
            INCREMENT differenceCount by 1 
        ENDIF
    ENDFOR
    
    IF differenceCount is less than 3 THEN
        OUTPUT "YES" 
    ELSE
        OUTPUT "NO" 
    ENDIF
    
END FUNCTION

CALL main
```
