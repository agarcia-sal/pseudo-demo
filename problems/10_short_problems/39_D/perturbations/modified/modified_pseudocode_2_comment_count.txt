### Pseudocode

```
FUNCTION main()
    SET firstInput TO read user input as a string
    SET secondInput TO read user input as a string
    
    SET firstSet TO split firstInput into a list of strings
    SET secondSet TO split secondInput into a list of strings
    
    SET differenceCount TO 0 
    
    FOR index FROM 0 TO 2 DO
        SET valueFromFirstSet TO convert firstSet[index] to an integer
        SET valueFromSecondSet TO convert secondSet[index] to an integer
        
        IF valueFromFirstSet is NOT equal to valueFromSecondSet THEN
            INCREMENT differenceCount by 1 
        END IF
    END FOR
    
    IF differenceCount is less than 3 THEN
        PRINT "YES"
    ELSE
        PRINT "NO"
    END IF
END FUNCTION

CALL main()
```
