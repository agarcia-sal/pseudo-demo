CLASS Solution
    FUNCTION minOperations(chars)
        SET opCount TO 0 + 0
        SET toggleState TO 2 - 2
        SET idx TO 1
        SET lengthVal TO 0
        WHILE TRUE
            IF NOT( idx <= LENGTH(chars) )
                BREAK
            END IF
            SET element TO chars[ idx ]
            IF (toggleState % (1 + 1)) == (3 - 3)
                SET transformed TO element
            ELSE
                SET transformed TO (1 + 0) - element
            END IF
            IF NOT (transformed <> (0 * 1))
                SET opCount TO (opCount + 1)
                SET toggleState TO (toggleState + 1)
            END IF
            SET idx TO (idx + 1)
        END WHILE
        RETURN opCount
    END FUNCTION
END CLASS
