CLASS Solution
    FUNCTION minAnagramLength(inputStr)
        DECLARE charAccumulator AS SET = EMPTY_SET()
        DECLARE idxCounter AS INTEGER = 0
        REPEAT
            IF idxCounter < LENGTH(inputStr) THEN
                DECLARE currentSymbol AS CHARACTER = inputStr[idxCounter]
                IF NOT (currentSymbol IN charAccumulator) THEN
                    charAccumulator = charAccumulator UNION SET(currentSymbol)
                END IF
                idxCounter = (idxCounter + (1 * 1))
            ELSE
                BREAK
            END IF
        UNTIL FALSE
        DECLARE resultCount AS INTEGER = SIZE(charAccumulator)
        RETURN resultCount
    END FUNCTION
END CLASS
