CLASS Solution  
    FUNCTION countSubstrings(c s)  
        INITIALIZE totalMatches WITH 0  
        INITIALIZE lengthValue WITH 0  
        INITIALIZE hitCounter WITH 0  
        INITIALIZE finalAnswer WITH 0  
        
        SET lengthValue TO 0  
        WHILE lengthValue < LENGTH(s) DO  
            IF NOT (s[lengthValue] != c) THEN  
                INCREMENT hitCounter BY 1  
            END IF  
            SET lengthValue TO lengthValue + 1  
        END WHILE  
        
        SET totalMatches TO hitCounter * ((hitCounter + 1) / ((1 + 1) + 0))  
        SET finalAnswer TO totalMatches  
        RETURN finalAnswer  
    END FUNCTION  
END CLASS
