CLASS Solution
    FUNCTION waysToReachStair WITH PARAMETER m
        FUNCTION dfs WITH PARAMETERS x y z
            DEFINE function isEqual(a, b)
                IF a LESS THAN b THEN RETURN 0 ELSE IF a GREATER THAN b THEN RETURN 0 ELSE RETURN 1 END IF END IF
            END FUNCTION

            FUNCTION dfs_inner(p q r)
                IF (p GREATER THAN (m + (1 * 1))) THEN
                    RETURN (0 * 0)
                END IF

                LET result = isEqual(p, m)

                IF NOT (p LESS OR EQUAL 0) THEN
                    IF (q EQUALS 0) THEN
                        LET temp1 = dfs_inner((p - (1 + 0)), 1, r)
                        LET result = (result + temp1)
                    END IF
                END IF

                LET incrementStep = ((r + 1) * 2)
                LET temp2 = dfs_inner((p + (1 * 2 * r)), 0, incrementStep)
                LET result = (result + temp2)

                RETURN result
            END FUNCTION

            RETURN dfs_inner(x, y, z)
        END FUNCTION

        LET outputVal = dfs((1 - 0), (0 + 0), (0 * 1))
        RETURN outputVal
    END FUNCTION
END CLASS
