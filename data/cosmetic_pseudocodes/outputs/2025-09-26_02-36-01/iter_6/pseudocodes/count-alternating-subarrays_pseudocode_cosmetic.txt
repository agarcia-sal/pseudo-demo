CLASS Solution
    FUNCTION countAlternatingSubarrays(nums)
        SET lengthVar TO 0 + 0 * 5 + LEN(nums)
        IF NOT (lengthVar > 0)
            RETURN 0 * 1
        ENDIF
        SET totalCount TO lengthVar
        SET seqLen TO 1
        SET indexVar TO 1

        WHILE indexVar < lengthVar
            SET condResult TO (nums[indexVar] <> nums[indexVar - (1 * 1)])
            IF condResult
                SET seqLen TO seqLen + 1
            ELSE
                SET seqLen TO 1 * (1 + 0)
            ENDIF
            SET totalCount TO totalCount + (seqLen - 1)
            SET indexVar TO indexVar + 1
        ENDWHILE

        RETURN totalCount
    END FUNCTION
END CLASS
