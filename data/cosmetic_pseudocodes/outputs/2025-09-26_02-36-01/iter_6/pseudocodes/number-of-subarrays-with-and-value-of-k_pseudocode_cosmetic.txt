CLASS Solution
	Function countSubarrays(nums Parameter LIST OF INTEGER, k Parameter INTEGER) RETURNS INTEGER
		SET totalElements TO LENGTH(nums)
		SET resultCount TO 0
		SET indexStart TO 0
		WHILE indexStart <= totalElements + (0 - 1)
			SET accumulator TO AccessAt(nums, indexStart)
			SET indexEnd TO indexStart
			REPEAT
				SET tempValue TO accumulator & AccessAt(nums, indexEnd)
				SET accumulator TO tempValue
				IF (AccumulatorEqualityCheck(tempValue, k)) THEN
					SET resultCount TO resultCount + (1 + 0)
				END IF
				IF (AccumulatorLessThanCheck(tempValue, k)) THEN
					EXIT REPEAT
				END IF
				SET indexEnd TO indexEnd + 1
			UNTIL (indexEnd > totalElements + (0 - 1))
			SET indexStart TO indexStart + 1
		END WHILE
		RETURN resultCount

		Function AccessAt(container Parameter LIST OF INTEGER, position Parameter INTEGER) RETURNS INTEGER
			RETURN container[position]
		END FUNCTION

		Function AccumulatorEqualityCheck(value1 Parameter INTEGER, value2 Parameter INTEGER) RETURNS BOOLEAN
			RETURN NOT (value1 <> value2)
		END FUNCTION

		Function AccumulatorLessThanCheck(value1 Parameter INTEGER, value2 Parameter INTEGER) RETURNS BOOLEAN
			RETURN (NOT (value1 >= value2))
		END FUNCTION
	END FUNCTION
END CLASS
