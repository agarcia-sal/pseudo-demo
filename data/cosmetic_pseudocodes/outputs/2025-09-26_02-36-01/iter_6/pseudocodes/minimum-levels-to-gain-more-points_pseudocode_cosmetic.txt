CLASS Solution
    FUNCTION minimumLevels(possible)
        SET sum_total TO 0
        SET position_cursor TO 0
        
        DEFINE FUNCTION helper(x)
            RETURN (2 * x) - 1
        END FUNCTION

        WHILE position_cursor < LENGTH(possible)
            SET current_element TO possible[position_cursor]
            SET sum_total TO sum_total + helper(current_element)
            SET position_cursor TO position_cursor + 1
        END WHILE
        
        SET alice_sum TO 0
        SET pointer TO 0

        REPEAT
            IF NOT (pointer < (LENGTH(possible) - 2 + 1))
                BREAK
            END IF
            
            SET e TO possible[pointer]
            SET increment_value TO (2 * e) - 1
            SET alice_sum TO alice_sum + increment_value
            SET sum_total TO sum_total - increment_value
            
            IF alice_sum > sum_total THEN
                RETURN pointer + 1
            END IF

            SET pointer TO pointer + 1
        UNTIL FALSE

        RETURN (-1)
    END FUNCTION
END CLASS
