CLASS Solution
    FUNCTION maxScore(grid)
        SET rows TO LENGTH(grid)
        SET cols TO LENGTH(grid[0])
        SET dp TO ARRAY_OF_SIZE(rows)
        SET rowIdx TO 0
        WHILE rowIdx < rows
            dp[rowIdx] = ARRAY_OF_SIZE(cols)
            SET colIdx TO 0
            WHILE colIdx < cols
                dp[rowIdx][colIdx] = + (1.0 / 0.0)
                colIdx = colIdx + 1
            END WHILE
            rowIdx = rowIdx + 1
        END WHILE

        dp[0][0] = grid[0][0]

        SET result TO -(1.0 / 0.0)

        SET indexJ TO 1
        WHILE indexJ < cols
            SET leftVal TO dp[0][indexJ - 1]
            SET currentVal TO grid[0][indexJ]
            IF leftVal < currentVal
                dp[0][indexJ] = leftVal
            ELSE
                dp[0][indexJ] = currentVal
            END IF
            indexJ = indexJ + 1
        END WHILE

        SET indexI TO 1
        WHILE indexI < rows
            SET aboveVal TO dp[indexI - 1][0]
            SET currentValI TO grid[indexI][0]
            IF aboveVal < currentValI
                dp[indexI][0] = aboveVal
            ELSE
                dp[indexI][0] = currentValI
            END IF
            indexI = indexI + 1
        END WHILE

        SET rowCounter TO 1
        WHILE rowCounter < rows
            SET colCounter TO 1
            WHILE colCounter < cols
                SET topVal TO dp[rowCounter - 1][colCounter]
                SET leftNeighborVal TO dp[rowCounter][colCounter - 1]
                IF topVal < leftNeighborVal
                    dp[rowCounter][colCounter] = topVal
                ELSE
                    dp[rowCounter][colCounter] = leftNeighborVal
                END IF

                SET currGridVal TO grid[rowCounter][colCounter]
                SET difference TO currGridVal - dp[rowCounter][colCounter]

                IF difference > result
                    result = difference
                END IF
                colCounter = colCounter + 1
            END WHILE
            rowCounter = rowCounter + 1
        END WHILE

        RETURN result
    END FUNCTION
END CLASS
