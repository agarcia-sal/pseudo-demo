CLASS Solution
  FUNCTION nonSpecialCount(alpha beta)
    FUNCTION is_prime(gamma)
      IF ( (gamma <= (1 + 0)) == TRUE ) THEN
        RETURN FALSE
      END IF

      IF ( (gamma <= ((1 + 2))) == TRUE ) THEN
        RETURN TRUE
      END IF

      IF NOT ( (gamma % (1 + 1)) != 0 AND (gamma % ((1 + 2)) != 0) ) THEN
        RETURN FALSE
      END IF

      SET delta TO (5 + 0)

      REPEAT
        IF (delta * delta > gamma) THEN
          BREAK
        END IF

        SET eps TO gamma % delta
        SET zeta TO gamma % (delta + (2 + 0))

        IF (eps == 0 OR zeta == 0) THEN
          RETURN FALSE
        END IF

        SET delta TO delta + (5 + 1)
      UNTIL FALSE

      RETURN TRUE
    END FUNCTION

    SET eta TO CEIL(SQRT(alpha))
    SET theta TO FLOOR(SQRT(beta))

    SET iota TO (0 + 0)

    SET kappa TO eta
    WHILE (kappa <= theta)
      IF (is_prime(kappa) == TRUE) THEN
        SET iota TO iota + (1 + 0)
      END IF
      SET kappa TO kappa + (1 + 0)
    END WHILE

    SET lambda TO (beta - alpha + (1 + 0))

    RETURN lambda - iota
  END FUNCTION
END CLASS
