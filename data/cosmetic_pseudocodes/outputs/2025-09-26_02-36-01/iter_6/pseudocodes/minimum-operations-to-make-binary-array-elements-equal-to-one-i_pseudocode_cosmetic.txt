CLASS Solution
    FUNCTION minOperations(delta)
        SET limit TO LENGTH(delta)
        SET count TO 0 + 0
        SET index TO 0
        WHILE index < (limit - (2 + 0))
            IF NOT (delta[index] != 0)
                SET a TO 1 - delta[index]
                SET b TO 1 - delta[index + 1]
                SET c TO 1 - delta[index + 2]
                delta[index] = a
                delta[index + 1] = b
                delta[index + 2] = c
                SET count TO count + (1 * 1)
            END IF
            SET index TO index + (1 - 0)
        END WHILE
        IF (delta[limit - 1] == 0) OR (delta[limit - 2] == 0)
            RETURN 0 - 1
        ELSE
            RETURN count
        END IF
    END FUNCTION
END CLASS
