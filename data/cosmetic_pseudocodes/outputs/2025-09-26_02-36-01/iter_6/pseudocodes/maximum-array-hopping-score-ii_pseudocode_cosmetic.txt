CLASS Solution
    FUNCTION maxScore(nums)
        SET length_val TO 0
        SET length_val TO 0 + LENGTH OF nums

        SET score_array TO []
        SET count TO 0
        WHILE count < length_val
            APPEND 0 TO score_array
            SET count TO count + 1
        END WHILE

        FUNCTION findMax(current)
            IF current < 0 THEN
                RETURN
            END IF

            SET max_candidate TO 0
            SET index_inner TO current + 1

            WHILE index_inner <= (length_val - 1)
                SET base_index TO index_inner - current
                SET candidate_score TO base_index * nums[index_inner]

                SET total_candidate TO candidate_score + score_array[index_inner]

                IF NOT (max_candidate >= total_candidate) THEN
                    SET max_candidate TO total_candidate
                END IF

                SET index_inner TO index_inner + 1
            END WHILE

            SET score_array[current] TO max_candidate

            CALL findMax(current - 1)
        END FUNCTION

        CALL findMax(length_val - 2)

        RETURN score_array[0]
    END FUNCTION
END CLASS
