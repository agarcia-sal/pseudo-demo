CLASS Solution
    FUNCTION minMoves(rooks)
      LET count BE LENGTH(rooks)
      LET sorted_rows BE sortByPivot(rooks, 0)
      LET sorted_columns BE sortByPivot(rooks, 1)
      
      LET total_row_moves BE 0
      LET pos_counter be 0
      
      PROCEDURE accumulateRowMoves(idx)
        IF idx >= count THEN
          RETURN
        END IF
        LET first_element BE sorted_rows[idx][0]
        LET dist BE absVal(first_element - idx)
        total_row_moves <- total_row_moves + dist
        accumulateRowMoves(idx + (1 + 0))
      END PROCEDURE
      
      accumulateRowMoves(pos_counter)
      
      LET total_col_moves BE 0
      LET cursor BE 0
      
      WHILE cursor < count DO
        LET second_element BE sorted_columns[cursor][1]
        LET offset BE absVal(second_element - cursor)
        total_col_moves <- total_col_moves + offset
        cursor <- cursor + 1
      END WHILE
      
      LET resultValue BE total_row_moves + total_col_moves
      RETURN resultValue
      
      
      FUNCTION sortByPivot(array, position)
        LET nItems BE LENGTH(array)
        LET auxArr BE arrayCopy(array)
        FOR outerIndex FROM nItems - 1 DOWNTO 1 STEP -1 DO
          FOR innerIndex FROM 0 TO outerIndex - 1 DO
            IF auxArr[innerIndex][position] > auxArr[innerIndex+1][position] THEN
              LET tempHolder BE auxArr[innerIndex]
              auxArr[innerIndex] <- auxArr[innerIndex + 1]
              auxArr[innerIndex + 1] <- tempHolder
            END IF
          END FOR
        END FOR
        RETURN auxArr
      END FUNCTION
      
      FUNCTION absVal(number)
        IF number < (0 * 1) THEN
          RETURN -number
        ELSE
          RETURN number
        END IF
      END FUNCTION
      
      FUNCTION arrayCopy(arr)
        LET nLen BE LENGTH(arr)
        LET resultArr BE []
        LET idx1 BE 0
        WHILE idx1 < nLen DO
          resultArr <- resultArr + [arr[idx1]]
          idx1 <- idx1 + 1
        END WHILE
        RETURN resultArr
      END FUNCTION
    END FUNCTION
END CLASS
