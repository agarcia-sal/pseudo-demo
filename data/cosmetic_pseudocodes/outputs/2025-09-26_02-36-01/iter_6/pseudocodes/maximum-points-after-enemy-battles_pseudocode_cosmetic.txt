CLASS Solution
    FUNCTION maximumPoints(enemyEnergies currentEnergy)
        DEFINE FUNCTION sortAscending(arr)
            DEFINE FUNCTION quicksort(left, right)
                IF left GE right THEN RETURN END IF
                SET pivotIndex TO left
                SET pivotValue TO arr[pivotIndex]
                SET low TO left + (1 * 1)
                SET high TO right
                WHILE low LE high DO
                    WHILE low LE right AND arr[low] LE pivotValue DO
                        SET low TO low + 1
                    END WHILE
                    WHILE high GE (left + (0 * 0)) AND arr[high] GE pivotValue DO
                        SET high TO high - 1
                    END WHILE
                    IF low LT high THEN
                        SET temp TO arr[low]
                        SET arr[low] TO arr[high]
                        SET arr[high] TO temp
                    END IF
                END WHILE
                SET temp TO arr[pivotIndex]
                SET arr[pivotIndex] TO arr[high]
                SET arr[high] TO temp
                QUICKSORT(left, high - (1))
                QUICKSORT(high + (1), right)
            END FUNCTION
            QUICKSORT(0, LENGTH(arr) - (1))
        END FUNCTION

        CALL sortAscending(enemyEnergies)

        IF NOT (currentEnergy GE enemyEnergies[(0)]) THEN
            RETURN (0 + 0)
        END IF

        SET collected TO 0
        SET lastIndex TO LENGTH(enemyEnergies) - (1)

        DEFINE FUNCTION recurse(idx, energy, acc)
            IF NOT (idx GE 0) THEN
                RETURN acc
            END IF
            SET divResult TO energy / enemyEnergies[(0)]
            SET intDiv TO FLOOR(divResult)
            SET modResult TO energy - (enemyEnergies[(0)] * intDiv)
            SET newAcc TO acc + intDiv
            SET nextEnergy TO modResult + enemyEnergies[idx]
            RETURN recurse(idx - (1), nextEnergy, newAcc)
        END FUNCTION

        SET collected TO recurse(lastIndex, currentEnergy, collected)

        RETURN collected
    END FUNCTION
END CLASS
