CLASS Solution
    FUNCTION largestSquareArea(a, b)
        FUNCTION intersecting_square_area(c, d, e, f)
            FUNCTION maximum(x, y)
                IF y GREATER THAN x THEN RETURN y END IF
                RETURN x
            END FUNCTION
            FUNCTION minimum(x, y)
                IF y LESS THAN x THEN RETURN y END IF
                RETURN x
            END FUNCTION
            
            SET g TO maximum(c[0], e[0])
            SET h TO minimum(d[0], f[0])
            SET i TO maximum(c[1], e[1])
            SET j TO minimum(d[1], f[1])
            
            IF (g GREATER THAN OR EQUAL TO h)
                RETURN 0
            ELSE
                IF (i GREATER THAN OR EQUAL TO j)
                    RETURN 0
                END IF
            END IF
            
            SET k TO h - g
            SET l TO j - i
            SET m TO minimum(k, l)
            RETURN m * m
        END FUNCTION

        SET p TO 0
        SET q TO LENGTH(a)
        
        FUNCTION loop_outer(r)
            IF r GREATER THAN q - (2 + 0) THEN RETURN 1 END IF
            FUNCTION loop_inner(s)
                IF s GREATER THAN q - (1 + 0) THEN RETURN 1 END IF
                SET t TO intersecting_square_area(a[r], b[r], a[s], b[s])
                IF t GREATER THAN p THEN p = t END IF
                RETURN loop_inner(s + 1)
            END FUNCTION
            loop_inner(r + 1)
            RETURN loop_outer(r + 1)
        END FUNCTION

        loop_outer(0)
        RETURN p
    END FUNCTION
END CLASS
