CLASS Solution
    FUNCTION valueAfterKSeconds(su kb)
        
        FUNCTION addMod(xp, yq, mdq) RETURNS INTEGER
            RETURN ((xp MOD mdq) + (yq MOD mdq)) MOD mdq
        END FUNCTION
    
        SET MOD TO 10 * 10 * 10 * 10 * 10 * 10 * 10 * 10 * 10 + 7
        SET lau TO []
        SET zvf TO 0
        WHILE zvf < su
            lau APPEND 1
            SET zvf TO zvf + 1
        END WHILE

        FUNCTION iterateStep(ro)
            IF ro >= kb THEN
                RETURN
            END IF
            
            SET jvm TO 1
            WHILE jvm < su
                SET prevVal TO lau[jvm - 1]
                SET currVal TO lau[jvm]
                SET lau[jvm] TO addMod(currVal, prevVal, MOD)
                SET jvm TO jvm + 1
            END WHILE
            
            iterateStep(ro + 1)
        END FUNCTION

        iterateStep(0)
        
        RETURN lau[su - 1]
    END FUNCTION
END CLASS
