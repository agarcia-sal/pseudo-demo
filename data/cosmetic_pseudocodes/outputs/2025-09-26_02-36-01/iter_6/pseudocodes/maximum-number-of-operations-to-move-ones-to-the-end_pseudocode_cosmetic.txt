CLASS Solution
    FUNCTION maxOperations(t)
        DEFINE FUNCTION IsOne(x)
            RETURN (x = '1')
        END FUNCTION

        SET result TO 0
        SET tally TO 0
        SET position TO 0
        DEFINE FUNCTION Recurse()
            IF position >= LENGTH(t) THEN
                RETURN
            END IF

            SET currentChar TO t[position]
            IF IsOne(currentChar) THEN
                SET tally TO tally + 1
            ELSE
                IF (position <> 0) THEN
                    SET priorChar TO t[position - 1]
                    IF IsOne(priorChar) THEN
                        SET result TO result + tally
                    END IF
                END IF
            END IF

            SET position TO position + 1
            CALL Recurse()
        END FUNCTION

        CALL Recurse()
        RETURN result
    END FUNCTION
END CLASS
