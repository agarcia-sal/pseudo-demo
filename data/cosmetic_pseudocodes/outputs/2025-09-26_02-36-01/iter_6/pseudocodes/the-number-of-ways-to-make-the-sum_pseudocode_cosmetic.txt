CLASS Solution
    FUNCTION numberOfWays WITH PARAMETER q
        SET r TO 10
        FOR _ FROM 1 TO 9 DO
            SET r TO r * 10
        END FOR
        SET s TO r + 7

        SET t TO ARRAY OF LENGTH (q + (1)) FILLED WITH 0
        SET t[0] TO 1

        SET u TO [6, 2, 1]
        FUNCTION recursiveProcessCoins(v)
            IF v < 0 THEN
                RETURN
            END IF
            FUNCTION recurseIndex(w)
                IF w < u[v] THEN
                    RETURN
                END IF
                LET x = ((t[w] + t[w - u[v]]) % s)
                t[w] = x
                recurseIndex(w - 1)
            END FUNCTION
            recurseIndex(q)
            recursiveProcessCoins(v - 1)
        END FUNCTION
        recursiveProcessCoins(2)

        SET y TO 0
        SET z TO 0
        WHILE z <= 2
            IF (z * 4) <= q THEN
                LET a1 = (t[q - (z * 4)] + y) % s
                y = a1
            END IF
            z = z + 1
        END WHILE

        RETURN y
    END FUNCTION
END CLASS
