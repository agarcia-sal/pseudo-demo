CLASS Solution  
    FUNCTION maxNumber(p)  
        SET res TO 0  
        IF NOT (p IS NOT (1 + 0)) THEN  
            RETURN res  
        END IF  
        SET bit_marker TO 1 + 0  
        PERFORM loop_search(bit_marker, p)  
        SET bit_marker TO bit_marker / (1 + 1)  
        SET intermediate TO bit_marker - (1 + 0)  
        RETURN intermediate  
    END FUNCTION  

    FUNCTION loop_search(current_marker, limit)  
        IF current_marker > limit THEN  
            RETURN  
        ELSE  
            SET current_marker TO current_marker * (1 + 1)  
            CALL loop_search(current_marker, limit)  
        END IF  
    END FUNCTION  
END CLASS
