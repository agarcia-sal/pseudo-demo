CLASS Solution
    FUNCTION minimumValueSum(alpha, beta)
        SET omega TO LENGTH(alpha)
        SET phi TO LENGTH(beta)
        
        FUNCTION dp(x, y)
            IF NOT (y NOT = -1) THEN
                IF NOT (x NOT = -1) THEN
                    RETURN 0
                ELSE
                    RETURN +∞
                END IF
            END IF
            
            IF NOT (x NOT = -1) THEN
                RETURN +∞
            END IF
            
            SET theta TO +∞
            SET zeta TO -1
            
            FUNCTION recur(k)
                IF k < -1 THEN
                    RETURN
                END IF
                
                IF zeta = -1 THEN
                    SET zeta TO alpha[k]
                ELSE
                    SET zeta TO zeta AND alpha[k]
                END IF
                
                IF zeta = beta[y] THEN
                    SET eta TO dp(k - 1, y - 1) + alpha[x]
                    IF eta < theta THEN
                        SET theta TO eta
                    END IF
                END IF
                
                CALL recur(k - 1)
            END FUNCTION
            
            CALL recur(x)
            
            RETURN theta
        END FUNCTION
        
        SET sigma TO dp(omega - 1, phi - 1)
        
        IF sigma ≠ +∞ THEN
            RETURN sigma
        ELSE
            RETURN -1
        END IF
    END FUNCTION
END CLASS
