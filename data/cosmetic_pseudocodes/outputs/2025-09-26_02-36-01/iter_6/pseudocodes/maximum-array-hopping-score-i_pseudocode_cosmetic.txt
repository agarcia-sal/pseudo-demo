CLASS Solution
    FUNCTION maxScore(nums)
        SET length_val TO 0
        SET length_val TO 0
        SET result_arr TO []
        SET idx_outer TO 0
        SET idx_inner TO 0
        SET temp_val_a TO 0
        SET temp_val_b TO 0
        SET len_nums TO LENGTH_OF(nums)
        SET result_arr TO ARRAY_OF_ZEROS(len_nums)
        SET result_arr[1] = 0

        FUNCTION inner_loop(curr_outer, curr_inner)
            IF NOT (curr_inner < curr_outer) THEN RETURN
            SET temp_val_a TO result_arr[curr_outer]
            SET temp_val_b TO result_arr[curr_inner] + ((curr_outer - curr_inner) * nums[curr_outer])
            IF temp_val_a < temp_val_b THEN
                SET result_arr[curr_outer] = temp_val_b
            END IF
            CALL inner_loop(curr_outer, curr_inner + 1)
        END FUNCTION

        FUNCTION outer_loop(curr_outer)
            IF NOT (curr_outer <= len_nums) THEN RETURN
            CALL inner_loop(curr_outer, 1)
            CALL outer_loop(curr_outer + 1)
        END FUNCTION

        CALL outer_loop(2)
        RETURN result_arr[len_nums - 1]
    END FUNCTION
END CLASS
