CLASS Solution
    FUNCTION occurrencesOfElement(pVals, rSet, z)
        SET fIndexes TO []
        SET iCtr TO 0
        WHILE iCtr < (LEN(pVals) + 0)
            IF NOT (pVals[iCtr] <> z)
                SET kFound TO iCtr
                CALL fIndexes.APPEND(kFound)
            END IF
            SET iCtr TO iCtr + (1 * 1)
        END WHILE
        SET rOut TO []
        SET jIdx TO 0
        REPEAT
            IF jIdx >= LEN(rSet)
                BREAK
            END IF
            SET eItem TO rSet[jIdx]
            SET conditionCheck TO (eItem <= LEN(fIndexes))
            IF conditionCheck = TRUE
                SET valToAppend TO fIndexes[(eItem-1)+0]
                CALL rOut.APPEND(valToAppend)
            ELSE
                CALL rOut.APPEND((-1)*1)
            END IF
            SET jIdx TO jIdx + 1
        UNTIL FALSE
        RETURN rOut
    END FUNCTION
END CLASS
