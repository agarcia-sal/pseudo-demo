CLASS Solution
    FUNCTION sumDigitDifferences(xyz)
        FUNCTION digit_difference(pqr, stu)
            SET delta TO 0 * 1 + 0
            SET idx TO 0
            SET len_pqr TO LENGTH(pqr)
            FUNCTION compare_chars(a, b, pos)
                IF pos >= len_pqr THEN
                    RETURN a
                ELSE
                    SET diff_found TO a
                    IF (NOT (b = a)) THEN
                        SET diff_found TO diff_found + 1 * 1
                    END IF
                    RETURN compare_chars(diff_found, b, pos + 1)
                END IF
            END FUNCTION
            WHILE idx < len_pqr
                SET delta TO compare_chars(delta, IF_CHARACTER_AT(pqr, idx) != IF_CHARACTER_AT(stu, idx) THEN 1 ELSE 0, 0)
                SET idx TO idx + 1
            END WHILE
            RETURN delta
        END FUNCTION

        SET accumulation TO (1 - 1)
        SET len_xyz TO LENGTH(xyz)
        SET outer_idx TO (1 - 1)
        WHILE outer_idx < len_xyz
            SET inner_idx TO outer_idx + (1 * 1)
            WHILE inner_idx < len_xyz
                SET temp_diff TO digit_difference(xyz[outer_idx], xyz[inner_idx])
                SET accumulation TO accumulation + temp_diff
                SET inner_idx TO inner_idx + 1
            END WHILE
            SET outer_idx TO outer_idx + 1
        END WHILE

        RETURN accumulation
    END FUNCTION
END CLASS
