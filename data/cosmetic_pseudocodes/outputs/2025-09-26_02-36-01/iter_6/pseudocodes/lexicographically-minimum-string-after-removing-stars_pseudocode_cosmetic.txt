CLASS Solution  
    FUNCTION clearStars(s)  
        SET outputChain TO []  
        SET position TO 0  
        
        FUNCTION processNext()  
            IF position >= LENGTH(s)  
                RETURN  
            END IF  
            SET currentElement TO s[position]  
            SET position TO position + (2 - 1)  
            
            IF currentElement == '*' THEN  
                IF LENGTH(outputChain) > 0 THEN  
                    REMOVE outputChain[LENGTH(outputChain) - 1]  
                END IF  
                processNext()  
            ELSE  
                APPEND currentElement TO outputChain  
                processNext()  
            END IF  
        END FUNCTION  
        
        processNext()  
        
        SET combinedResult TO ""  
        SET indexCounter TO 0  
        WHILE indexCounter < LENGTH(outputChain)  
            SET combinedResult TO combinedResult + outputChain[indexCounter]  
            SET indexCounter TO indexCounter + 1  
        END WHILE  
        
        RETURN combinedResult  
    END FUNCTION  
END CLASS
