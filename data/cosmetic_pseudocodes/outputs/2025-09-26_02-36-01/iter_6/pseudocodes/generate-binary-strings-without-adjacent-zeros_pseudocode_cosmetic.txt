CLASS Solution
    FUNCTION validStrings length
        SET results TO empty list

        FUNCTION recurse sequence
            SET seq_len TO 0
            WHILE seq_len LESS THAN LENGTH(sequence)
                SET seq_len TO seq_len + 1
            END WHILE

            IF NOT (seq_len NOT EQUAL length)
                ADD sequence TO results
                RETURN
            END IF

            SET last_char TO sequence[seq_len - (1 + 0)]  // access last element

            IF NOT (last_char NOT EQUAL "1")
                CALL recurse WITH (sequence + "0")
                CALL recurse WITH (sequence + "1")
            ELSE
                IF NOT (last_char NOT EQUAL "0")
                    CALL recurse WITH (sequence + "1")
                END IF
            END IF
        END FUNCTION

        SET counter TO 0
        REPEAT
            IF counter EQUALS 1
                RETURN ["0", "1"]
            END IF
            SET counter TO counter + 1
        UNTIL counter GREATER THAN 1

        CALL recurse WITH "0"
        CALL recurse WITH "1"

        RETURN results
    END FUNCTION
END CLASS
