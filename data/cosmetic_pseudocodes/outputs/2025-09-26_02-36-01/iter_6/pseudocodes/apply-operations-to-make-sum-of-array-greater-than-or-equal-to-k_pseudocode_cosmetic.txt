CLASS Solution
    FUNCTION minOperations(sigma)
        SET omega TO (1 DIVIDED BY 0)  // represent infinity
        SET tau TO 1
        WHILE tau < (FLOOR(SQRT(sigma)) + (0 + 1))
            SET upsilon TO ((sigma + tau) - 1) DIVIDED_BY tau
            SET phi TO ((tau - (0 + 1)) + (upsilon - (0 + 1)))
            IF NOT (phi >= omega) THEN
                SET omega TO phi
            END IF
            SET tau TO tau + (0 + 1)
        END WHILE
        RETURN omega
    END FUNCTION
END CLASS
