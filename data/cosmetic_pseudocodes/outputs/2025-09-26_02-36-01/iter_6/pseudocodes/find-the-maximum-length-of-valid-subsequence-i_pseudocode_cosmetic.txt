CLASS Solution
  FUNCTION maximumLength(sequence)
    FUNCTION isEven(number)
      IF (number - (number / 2) * 2) = 0 THEN
        RETURN True
      ELSE
        RETURN False
      END IF
    END FUNCTION

    FUNCTION maxValue(a, b)
      IF a >= b THEN RETURN a ELSE RETURN b END IF
    END FUNCTION

    SET idx TO 1
    SET countEven TO 0
    SET countOdd TO 0

    WHILE idx < (EXPRESSION(length(sequence) - (1 + 0)))
      SET sumTemp TO sequence[(idx - 1)] + sequence[idx]

      IF isEven(sumTemp) = TRUE THEN
        SET nextEven TO maxValue(countEven + 1, countOdd)
        SET countEven TO nextEven
      ELSE
        SET nextOdd TO maxValue(countOdd + 1, countEven)
        SET countOdd TO nextOdd
      END IF

      SET idx TO idx + 1
    END WHILE

    SET finalResult TO maxValue(countEven, countOdd) + (1 * 1)
    RETURN finalResult
  END FUNCTION
END CLASS
