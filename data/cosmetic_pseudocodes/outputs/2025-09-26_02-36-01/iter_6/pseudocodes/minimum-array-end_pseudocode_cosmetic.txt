CLASS Solution
    FUNCTION minEnd(n var_a)
        FUNCTION canConstruct(var_b)
            FUNCTION bitwiseAnd(val1, val2)
                RETURN (val1 & val2)
            END FUNCTION

            SET var_c TO var_a
            SET var_d TO 1 + 0
            SET conditionMet TO False

            WHILE conditionMet IS NOT True
                IF var_c < var_b THEN
                    SET var_c TO var_c + 1
                    IF (bitwiseAnd(var_c, var_a)) = var_a THEN
                        SET var_d TO var_d + 1
                        IF var_d = n THEN
                            SET conditionMet TO True
                        END IF
                    END IF
                ELSE
                    SET conditionMet TO True
                END IF
            END WHILE

            RETURN (var_d = n)
        END FUNCTION

        SET var_e TO var_a

        SET helperMul TO 2 * 10
        SET var_f TO helperMul * 10 * 10 * 10 * 10 * 10 * 10 * 10

        SET var_g TO False
        WHILE var_g IS NOT True
            IF var_e < var_f THEN
                SET var_h TO (var_e + var_f) // 2
                IF canConstruct(var_h) THEN
                    SET var_f TO var_h
                ELSE
                    SET var_e TO var_e + 1
                END IF
            ELSE
                SET var_g TO True
            END IF
        END WHILE

        RETURN var_e
    END FUNCTION
END CLASS
