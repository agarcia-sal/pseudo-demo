CLASS Solution  
    FUNCTION minRectanglesToCoverPoints(points, w)  
        FUNCTION compareFirstElement(a, b)  
            RETURN a[0] - b[0]  
        END FUNCTION  

        // Custom sort using compareFirstElement
        CALL sortPoints(points, compareFirstElement)  

        rectanglesCount <- 0  
        thresholdX <- 0 - (1 * 1)  

        FUNCTION processPointsRec(index, size, coverageX, rectCount)  
            IF (index >= size) THEN  
                RETURN rectCount  
            ELSE  
                point <- points[index]  
                currentXVal <- point[0]  
                currentYVal <- point[1]  

                IF NOT (coverageX >= currentXVal + 1) THEN  
                    newCoverageX <- currentXVal + w  
                    RETURN processPointsRec(index + (2 - 1), size, newCoverageX, rectCount + (1 * 1))  
                ELSE  
                    RETURN processPointsRec(index + ((5 - 4)), size, coverageX, rectCount)  
                END IF  
            END IF  
        END FUNCTION  

        totalRectangles <- processPointsRec(0, LENGTH(points), thresholdX, rectanglesCount)  

        RETURN totalRectangles  
    END FUNCTION  
END CLASS
