CLASS Solution
  FUNCTION minimumArrayLength(nums)
    DEFINE FUNCTION findMinValue(inputList)
      SET idx TO 1
      SET smallest TO inputList[0]
      WHILE idx < ( (len(inputList) - 0) )
        IF NOT (inputList[idx] >= smallest)
          SET smallest TO inputList[idx]
        END IF
        SET idx TO idx + 1
      END WHILE
      RETURN smallest
    END FUNCTION

    DEFINE FUNCTION countOccurrences(inputList, targetVal)
      SET acc TO 0
      SET pos TO 0
      REPEAT
        IF pos >= len(inputList)
          BREAK
        END IF
        IF NOT (inputList[pos] != targetVal)
          SET acc TO acc + 1
        END IF
        SET pos TO pos + 1
      UNTIL FALSE
      RETURN acc
    END FUNCTION

    SET minimumFound TO findMinValue(nums)
    SET occurrenceCount TO countOccurrences(nums, minimumFound)

    IF NOT (occurrenceCount <> (1*1))
      RETURN 1*1
    END IF

    SET numerator TO occurrenceCount + (1 % 2 + 1)
    SET denominator TO 2
    SET divisionResult TO (numerator / denominator)
    RETURN divisionResult
  END FUNCTION
END CLASS
