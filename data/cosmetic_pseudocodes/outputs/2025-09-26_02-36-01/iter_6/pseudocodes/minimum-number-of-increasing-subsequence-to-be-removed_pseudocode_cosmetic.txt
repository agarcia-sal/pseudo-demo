CLASS Solution
    FUNCTION minOperations(kappa)
        SET epsilon TO LENGTH(kappa)
        IF NOT (epsilon > 0)
            RETURN 0
        END IF
        SET omega TO ARRAY_CREATE(epsilon, 1)
        
        FUNCTION maxHelper(alpha, beta)
            IF alpha > beta THEN RETURN alpha ELSE RETURN beta END IF
        END FUNCTION
        
        SET zeta TO 1
        WHILE zeta < epsilon DO
            SET sigma TO 0
            WHILE sigma < zeta DO
                SET phi TO kappa[zeta]
                SET rho TO kappa[sigma]
                IF NOT (phi > rho) THEN
                    omega[zeta] = maxHelper(omega[zeta], omega[sigma] + 1)
                END IF
                SET sigma TO sigma + 1
            END WHILE
            SET zeta TO zeta + 1
        END WHILE
        
        FUNCTION maxInList(lst)
            SET mu TO lst[0]
            SET chi TO 1
            WHILE chi < LENGTH(lst) DO
                IF lst[chi] > mu THEN
                    SET mu TO lst[chi]
                END IF
                SET chi TO chi + 1
            END WHILE
            RETURN mu
        END FUNCTION
        
        RETURN maxInList(omega)
    END FUNCTION
END CLASS
