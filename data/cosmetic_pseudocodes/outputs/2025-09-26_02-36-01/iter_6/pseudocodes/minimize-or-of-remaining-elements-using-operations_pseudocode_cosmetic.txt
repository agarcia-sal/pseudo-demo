CLASS Solution
    FUNCTION minOrAfterOperations(nums o,k)
        
        FUNCTION canAchieve(p,q)
            VARIABLE r = -1
            VARIABLE s = 0
            
            FUNCTION process_index(u)
                IF u >= LENGTH(o) THEN
                    RETURN true
                ELSE
                    VARIABLE v = o[u]
                    IF r = -1 THEN
                        r = v
                    ELSE
                        r = r & v
                    END IF
                    IF (r & p) = 0 THEN
                        r = -1
                    ELSE
                        s = s+1
                        IF s > q THEN
                            RETURN false
                        END IF
                    END IF
                    RETURN process_index(u+1)
                END IF
            END FUNCTION
            
            RETURN process_index(0)
        END FUNCTION
        
        VARIABLE w = (1 * 2) ^ 30 - 1
        VARIABLE x = w
        VARIABLE y = 0
        
        WHILE y <= 29 DO
            VARIABLE z = (1 * 2) ^ y

            IF (x & z) != 0 THEN
                VARIABLE aa = canAchieve(NOT x XOR z, k)
                IF aa = true THEN
                    x = x & (NOT z)
                END IF
            END IF
            
            y = y+1
        END WHILE
        
        RETURN x
    END FUNCTION
END CLASS
