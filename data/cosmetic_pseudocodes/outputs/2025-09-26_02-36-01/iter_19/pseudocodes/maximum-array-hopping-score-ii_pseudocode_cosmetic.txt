CLASS Solution
    FUNCTION maxScore(nums)
        SET length_var TO LENGTH OF nums
        SET memo_arr TO list of zeros with LENGTH length_var

        FUNCTION computeMax(index, result_acc)
            IF index LESS THAN 0 THEN
                RETURN result_acc
            END IF

            SET current_max TO 0
            SET pos_var TO index PLUS 1

            WHILE pos_var LESS THAN length_var DO
                SET temp_val TO (pos_var MINUS index) MULTIPLIED BY nums[pos_var]
                SET combined_val TO temp_val PLUS memo_arr[pos_var]

                IF current_max LESS THAN combined_val THEN
                    SET current_max TO combined_val
                END IF

                SET pos_var TO pos_var PLUS 1
            END WHILE

            SET memo_arr[index] TO current_max

            RETURN computeMax(index MINUS 1, result_acc)
        END FUNCTION

        RETURN computeMax(length_var MINUS 2, 0)
    END FUNCTION
END CLASS
