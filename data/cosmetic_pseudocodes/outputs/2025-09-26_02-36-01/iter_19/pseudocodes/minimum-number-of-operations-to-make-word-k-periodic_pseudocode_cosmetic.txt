CLASS Solution
    FUNCTION minimumOperationsToMakeKPeriodic(word param k)
      SET alpha TO 0
      SET omega TO LENGTH OF param
      SET lambda TO an empty LIST

      WHILE alpha < omega
            SET beta TO SUBSTRING(param, alpha, k)
            APPEND beta TO lambda
            SET alpha TO alpha + k
      END WHILE

      FUNCTION customCountDuplicates(arr)
          SET tally TO an empty MAP
          FOR EACH theta IN arr
              IF theta IN tally THEN
                  SET tally[theta] TO tally[theta] + 1
              ELSE
                  SET tally[theta] TO 1
              END IF
          END FOR
          RETURN tally
      END FUNCTION

      SET hash_map TO customCountDuplicates(lambda)

      SET delta TO 0
      FOR EACH gamma, sigma IN hash_map
          IF sigma > delta THEN
              SET delta TO sigma
          END IF
      END FOR

      SET epsilon TO LENGTH OF lambda - delta
      RETURN epsilon
    END FUNCTION
END CLASS
