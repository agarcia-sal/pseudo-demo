CLASS Solution
	FUNCTION validStrings count
		SET resultCollection TO empty list

		FUNCTION explore sequence
			IF LENGTH sequence EQUALS count
				APPEND sequence TO resultCollection
				RETURN
			END IF

			SET lastChar TO element at position from the END one of sequence
			IF lastChar EQUALS string one
				CALL explore WITH argument sequence PLUS string zero
				CALL explore WITH argument sequence PLUS string one
			ELSE
				IF lastChar EQUALS string zero
					CALL explore WITH argument sequence PLUS string one
				END IF
			END IF
		END FUNCTION

		IF count EQUALS one
			RETURN list containing string zero AND string one
		END IF

		CALL explore WITH argument string zero
		CALL explore WITH argument string one

		RETURN resultCollection
	END FUNCTION
END CLASS
