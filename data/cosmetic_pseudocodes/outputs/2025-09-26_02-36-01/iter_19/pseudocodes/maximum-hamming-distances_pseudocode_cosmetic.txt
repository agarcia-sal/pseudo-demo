CLASS Solution
    FUNCTION maxHammingDistances(nums LIST OF INTEGER m INTEGER) RETURNS LIST OF INTEGER
        SET delta TO 0
        SET zymk TO 0
        SET gorka TO 0
        SET uxip TO 0
        SET vgip TO 0
        SET miraly TO 0
        SET nyfra TO 0
        SET fjep TO ""
        SET uver TO []
        SET equiv1 TO []
        SET equiv2 TO 0
        SET resarr TO []
        SET ix TO 0

        FUNCTION hamming_distance(bin1 STRING bin2 STRING) RETURNS INTEGER
            SET acc_dist TO 0
            SET pos TO 0
            WHILE pos < LENGTH(bin1)
                IF NOT (bin1[pos] = bin2[pos])
                    SET acc_dist = acc_dist + 1
                END IF
                SET pos = pos + 1
            END WHILE
            RETURN acc_dist
        END FUNCTION

        WHILE delta < LENGTH(nums)
            SET tvi TO nums[delta]
            SET tempbin TO ""
            SET tmp_idx TO 0
            WHILE tmp_idx < m
                SET bitmask TO 1 << (m - 1 - tmp_idx)
                IF (tvi & bitmask) = 0
                    SET tempbin = tempbin + "0"
                ELSE
                    SET tempbin = tempbin + "1"
                END IF
                SET tmp_idx = tmp_idx + 1
            END WHILE
            APPEND tempbin TO uver
            SET delta = delta + 1
        END WHILE

        SET ix = 0
        WHILE ix < LENGTH(nums)
            SET maxval TO 0
            SET jx TO 0
            WHILE jx < LENGTH(nums)
                IF NOT (ix = jx)
                    SET tmpdist TO hamming_distance(uver[ix], uver[jx])
                    IF tmpdist > maxval
                        SET maxval = tmpdist
                    END IF
                END IF
                SET jx = jx + 1
            END WHILE
            APPEND maxval TO resarr
            SET ix = ix + 1
        END WHILE

        RETURN resarr
    END FUNCTION
END CLASS
