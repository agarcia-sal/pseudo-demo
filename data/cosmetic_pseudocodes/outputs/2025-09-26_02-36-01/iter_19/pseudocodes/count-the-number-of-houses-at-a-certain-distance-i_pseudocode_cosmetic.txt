CLASS Solution
    FUNCTION countOfPairs(n parameter, x parameter, y parameter) RETURNS list of integer
        SET alpha TO a list filled with zeros of length n

        FUNCTION bfs(gamma parameter)
            SET sigma TO a boolean list of length (n + 1), all False
            SET omega TO a zero-filled list of length (n + 1)
            SET delta TO a deque initialized with gamma inside
            SET sigma[gamma] TO True

            REPEAT WHILE length of delta GREATER THAN zero
                SET kappa TO the front element popped from delta

                FOR EACH eta IN [kappa - 1, kappa + 1]
                    IF (eta >= 1) AND (eta <= n) AND (NOT sigma[eta])
                        SET sigma[eta] TO True
                        SET omega[eta] TO omega[kappa] + 1
                        APPEND eta TO delta
                    END IF
                END FOR

                IF (kappa = x) AND (NOT sigma[y])
                    SET sigma[y] TO True
                    SET omega[y] TO omega[kappa] + 1
                    APPEND y TO delta
                END IF

                IF (kappa = y) AND (NOT sigma[x])
                    SET sigma[x] TO True
                    SET omega[x] TO omega[kappa] + 1
                    APPEND x TO delta
                END IF
            END REPEAT

            FOR z FROM 1 TO n
                IF omega[z] > 0
                    INCREMENT alpha[omega[z] - 1] BY 1
                END IF
            END FOR
        END FUNCTION

        FOR beta FROM 1 TO n
            CALL bfs WITH beta
        END FOR

        RETURN alpha
    END FUNCTION
END CLASS
