CLASS Solution
    FUNCTION maxOperations(inputString)
        SET totalResults TO 0
        SET currentCount TO 0
        SET position TO 0
        WHILE position < LENGTH(inputString)
            IF inputString[position] == '1' THEN
                SET marker TO currentCount + 1
                SET currentCount TO marker
            ELSE
                IF position > 0 THEN
                    SET precedingChar TO inputString[position - 1]
                    IF precedingChar == '1' THEN
                        SET newTotal TO totalResults + currentCount
                        SET totalResults TO newTotal
                    END IF
                END IF
            END IF
            SET position TO position + 1
        END WHILE
        RETURN totalResults
    END FUNCTION
END CLASS
