CLASS Solution
    FUNCTION kthCharacter(k x INTEGER, operations y LIST OF INTEGER) RETURNS STRING
        SET n TO 1
        SET strategy TO []
        
        LET idx TO 0
        WHILE idx < LENGTH OF y DO
            LET elem TO y[idx]
            APPEND elem TO strategy
            IF elem = 0 THEN
                LET n TO n * 2
            ELSE
                LET n TO 2 * n
            END IF
            LET idx TO idx + 1
        END WHILE
        
        LET letter TO "a"
        LET pos TO LENGTH OF strategy - 1
        REPEAT
            IF x <= n / 2 THEN
                LET n TO n / 2
            ELSE
                LET x TO x - (n / 2)
                LET n TO n / 2
                IF strategy[pos] = 1 THEN
                    LET s TO CHAR_CODE(letter)
                    IF s = CHAR_CODE("z") THEN
                        LET letter TO "a"
                    ELSE
                        LET letter TO CHAR_FROM_CODE(s + 1)
                    END IF
                END IF
            END IF
            LET pos TO pos - 1
        UNTIL pos < 0
        
        RETURN letter
    END FUNCTION
END CLASS
