CLASS Solution
    FUNCTION minimumTimeToInitialState(word PARAMETER k)
        SET alpha TO LENGTH OF word
        SET omega TO 1
        REPEAT
            SET phi TO omega * k
            IF phi >= alpha OR (SUBSTRING word FROM phi TO alpha) = (SUBSTRING word FROM 1 TO alpha - phi) THEN
                RETURN omega
            END IF
            SET omega TO omega + 1
        UNTIL FALSE
    END FUNCTION
END CLASS
