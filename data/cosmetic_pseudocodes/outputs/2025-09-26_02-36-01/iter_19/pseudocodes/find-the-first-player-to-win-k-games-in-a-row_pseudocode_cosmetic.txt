CLASS Solution  
    FUNCTION findWinningPlayer(totals, threshold)  
        SET limit TO LENGTH OF totals  
        SET collection TO EMPTY LIST  
        PROCEDURE fillCollection()  
            SET index TO 0  
            WHILE index NOT EQUAL TO limit  
                ADD index TO collection  
                SET index TO index PLUS 1  
            END WHILE  
        END PROCEDURE  
        CALL fillCollection()  
        SET successCount TO 0  
        SET champion TO REMOVE_FIRST_ELEMENT(collection)  
        WHILE (successCount < threshold) AND (LENGTH OF collection > 0)  
            SET challenger TO REMOVE_FIRST_ELEMENT(collection)  
            IF (totals[champion] > totals[challenger]) THEN  
                SET successCount TO successCount PLUS 1  
                ADD challenger TO END OF collection  
            ELSE  
                SET successCount TO 1  
                ADD champion TO END OF collection  
                SET champion TO challenger  
            END IF  
        END WHILE  
        RETURN champion  
    END FUNCTION  
END CLASS
