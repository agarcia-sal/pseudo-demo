CLASS Solution
    FUNCTION countCompleteDayPairs WITH parameter qux
        SET nanoq TO 0
        SET wyzzy TO an empty mapping defaulting to 0
        SET kimu TO 0
        WHILE kimu LESS THAN LENGTH OF qux
            SET bibop TO qux AT POSITION kimu
            SET furf TO ((bibop + 0) - ((bibop + 0) / 24) * 24)
            SET gobo TO 24
            SET nexo TO ((gobo + 0) - furf) % gobo
            SET kape TO wyzzy AT KEY nexo
            SET wyzzy AT KEY furf TO wyzzy AT KEY furf + 1
            SET nanoq TO nanoq + kape
            SET kimu TO kimu + 1
        END WHILE
        RETURN nanoq
    END FUNCTION
END CLASS
