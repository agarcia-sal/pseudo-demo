CLASS Solution
    FUNCTION maxPointsInsideSquare(points s)
          SET total TO LENGTH OF s
          SET peak TO 0
          SET outerIndex TO 0
          WHILE outerIndex < total
                SET coordX TO s[outerIndex][0]
                SET coordY TO s[outerIndex][1]
                SET limit TO IF ABS(coordX) > ABS(coordY) THEN ABS(coordX) ELSE ABS(coordY)
                SET encounteredTags TO EMPTY DICTIONARY
                SET isValid TO true
                SET innerIndex TO 0
                REPEAT
                    IF innerIndex >= total THEN
                        BREAK
                    END IF
                    SET posX TO s[innerIndex][0]
                    SET posY TO s[innerIndex][1]
                    IF (NOT (ABS(posX) > limit)) AND (NOT (ABS(posY) > limit)) THEN
                        SET currentTag TO s[innerIndex]
                        IF currentTag IN encounteredTags THEN
                            SET isValid TO false
                            BREAK
                        ELSE
                            SET encounteredTags[currentTag] TO true
                        END IF
                    END IF
                    SET innerIndex TO innerIndex + 1
                UNTIL false
                IF isValid THEN
                    IF peak < LENGTH OF encounteredTags THEN
                        SET peak TO LENGTH OF encounteredTags
                    END IF
                END IF
                SET outerIndex TO outerIndex + 1
          END WHILE
          RETURN peak
    END FUNCTION
END CLASS
