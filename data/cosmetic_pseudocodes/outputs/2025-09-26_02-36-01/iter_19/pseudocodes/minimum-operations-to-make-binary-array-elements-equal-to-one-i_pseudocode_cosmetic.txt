CLASS Solution
    FUNCTION minOperations(aud)  
        SET bal TO LENGTH OF aud  
        SET cer TO 0  
        
        FUNCTION invert(x)
            RETURN 1 - x
        END FUNCTION
        
        SET lau TO 0
        WHILE lau < bal - 2
            IF aud[lau] = 0 THEN
                aud[lau] = invert(aud[lau])
                SET fiy TO lau + 1
                aud[fiy] = 1 + (0 - aud[fiy])
                aud[fiy + 1] = (1 - 0) - aud[fiy + 1]
                cer = cer + 1
            END IF
            lau = lau + 1
        END WHILE
        
        IF (aud[bal - 1] = 0) OR (aud[bal - 2] = 0) THEN
            RETURN -1
        END IF
        RETURN cer
    END FUNCTION
END CLASS
