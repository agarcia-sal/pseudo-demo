CLASS Solution  
    FUNCTION maximumSetSize(nums1 PARAMETER List OF integers nums2 PARAMETER List OF integers) RETURNS integer
        SET pqr TO 0  
        SET wxy TO LENGTH OF nums1  
        SET abc TO wxy / 2  

        FUNCTION extractUnique(lst PARAMETER List OF integers) RETURNS Set OF integers
            SET hli TO EMPTY SET  
            SET idx TO 0  
            WHILE idx < LENGTH OF lst DO
                IF NOT (lst[idx] IN hli) THEN
                    ADD lst[idx] TO hli  
                END IF  
                SET idx TO idx + 1  
            END WHILE  
            RETURN hli  
        END FUNCTION

        FUNCTION intersectSets(s1 PARAMETER Set OF integers s2 PARAMETER Set OF integers) RETURNS Set OF integers
            SET res TO EMPTY SET  
            FOR EACH val IN s1 DO
                IF val IN s2 THEN
                    ADD val TO res  
                END IF  
            END FOR  
            RETURN res  
        END FUNCTION

        SET vsp TO extractUnique(nums1)  
        SET tnm TO extractUnique(nums2)  
        SET klm TO intersectSets(vsp, tnm)  

        FUNCTION excludeSet(original PARAMETER Set OF integers excl PARAMETER Set OF integers) RETURNS Set OF integers
            SET retSet TO EMPTY SET  
            FOR EACH item IN original DO
                IF NOT (item IN excl) THEN
                    ADD item TO retSet  
                END IF  
            END FOR  
            RETURN retSet  
        END FUNCTION

        SET qrm TO excludeSet(vsp, klm)  
        SET jnk TO excludeSet(tnm, klm)  

        FUNCTION minVal(a PARAMETER integer b PARAMETER integer) RETURNS integer
            IF a < b THEN RETURN a ELSE RETURN b END IF  
        END FUNCTION

        FUNCTION maxVal(a PARAMETER integer b PARAMETER integer) RETURNS integer
            IF a > b THEN RETURN a ELSE RETURN b END IF  
        END FUNCTION

        SET hgf TO minVal(abc, LENGTH OF qrm)  
        SET zbc TO minVal(abc, LENGTH OF jnk)  

        SET mno TO maxVal(0, abc - hgf)  
        SET lkp TO maxVal(0, abc - zbc)  

        SET rtv TO mno + lkp  
        SET sda TO minVal(rtv, LENGTH OF klm)  

        SET retv TO hgf + zbc + sda  

        RETURN retv  
    END FUNCTION  
END CLASS
