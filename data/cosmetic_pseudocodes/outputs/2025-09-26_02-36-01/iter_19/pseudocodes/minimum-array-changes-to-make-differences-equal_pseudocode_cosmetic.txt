CLASS Solution
    FUNCTION minChanges(alfa bravo)
        SET charlie TO ARRAY_INITIALIZED_WITH_ZEROES(k + 2)
        SET delta TO LENGTH OF alfa
        FUNCTION echo(foxtrot, golf)
            IF foxtrot <= golf THEN RETURN foxtrot ELSE RETURN golf END IF
        END FUNCTION
        SET hotel TO 0
        WHILE hotel < (delta / 2) DO
            SET india TO alfa[hotel]
            SET juliett TO alfa[(delta - 1) - hotel]
            IF india <= juliett THEN
                SET kilo TO india
                SET lima TO juliett
            ELSE
                SET kilo TO juliett
                SET lima TO india
            END IF
            charlie[0] = charlie[0] + 1
            charlie[lima - kilo] = charlie[lima - kilo] - 1
            charlie[(lima - kilo) + 1] = charlie[(lima - kilo) + 1] + 1
            SET mike TO echo(lima, (k - kilo)) + 1
            charlie[mike] = charlie[mike] - 1
            charlie[mike + 1] = charlie[mike + 1] + 1
            hotel = hotel + 1
        END WHILE
        SET november TO charlie[0]
        SET oscar TO november
        SET papa TO 0
        REPEAT
            SET papa = papa + 1
            SET november = november + charlie[papa]
            IF november < oscar THEN
                oscar = november
            END IF
        UNTIL papa = (k + 1)
        RETURN oscar
    END FUNCTION
END CLASS
