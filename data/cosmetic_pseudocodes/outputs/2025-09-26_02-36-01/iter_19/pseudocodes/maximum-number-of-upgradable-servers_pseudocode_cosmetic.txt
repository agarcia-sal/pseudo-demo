```pseudocode
CLASS Solution
    FUNCTION maxUpgrades(count LIST OF INTEGER upgrade LIST OF INTEGER sell LIST OF INTEGER money LIST OF INTEGER) RETURNS LIST OF INTEGER
        SET delta TO 0
        SET omega TO LENGTH OF count
        SET genome TO []
        WHILE delta < omega
            SET phi TO 0
            SET alpha TO 0
            SET psi TO 0
            SET chi TO 0
            SET beta TO 0
            SET gamma TO 0
            SET zeta TO 0
            SET deltaVal TO 0
            SET eta TO 0
            SET theta TO 0
            SET lambda TO 0
            SET kappa TO 0
            SET rho TO 0
            SET sigma TO 0
            SET tau TO 0

            SET beta TO 0
            SET rho TO count[delta]
            SET sigma TO upgrade[delta]
            SET tau TO sell[delta]
            SET theta TO money[delta]

            SET deltaVal TO 0
            SET eta TO theta
            FOR lambda IN RANGE(rho + 1)
                SET kappa TO rho - lambda
                SET gamma TO lambda * tau
                SET zeta TO eta + gamma
                SET phi TO zeta DIV sigma
                IF phi > kappa
                    SET phi TO kappa
                END IF
                IF phi > deltaVal
                    SET deltaVal TO phi
                END IF
            END FOR
            CALL genome.APPEND(deltaVal)
            SET delta TO delta + 1
        END WHILE
        RETURN genome
    END FUNCTION
END CLASS
```
