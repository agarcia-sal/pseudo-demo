CLASS Solution
    FUNCTION distanceSum WITH PARAMETERS alpha beta gamma
        SET theta TO 1000000000 PLUS 7
        SET epsilon TO (((beta TIMES beta) TIMES (alpha TIMES alpha) TIMES ((alpha TIMES alpha TIMES alpha) MINUS alpha)) DIVIDED BY 6)
        SET zeta TO (((alpha TIMES alpha) TIMES (beta TIMES beta) TIMES ((beta TIMES beta TIMES beta) MINUS beta)) DIVIDED BY 6)
        FUNCTION binomialComb WITH PARAMETERS phi psi
            SET res TO 1
            SET i TO 0
            WHILE i LESS THAN psi
                SET res TO (res TIMES (phi MINUS i)) DIVIDED BY (i PLUS 1)
                INCREMENT i BY 1
            END WHILE
            RETURN res
        END FUNCTION
        SET varsigma TO binomialComb((alpha TIMES beta) MINUS 2, gamma MINUS 2)
        SET omicron TO (epsilon PLUS zeta) TIMES varsigma
        RETURN omicron MOD theta
    END FUNCTION
END CLASS
