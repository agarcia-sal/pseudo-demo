CLASS Solution
    FUNCTION clearStars(s)
        SET beta TO an empty list
        SET gamma TO 0
        WHILE gamma < LENGTH(s)
            SET omega TO s[gamma]
            IF NOT omega NOT EQUAL "*" THEN
                IF LENGTH(beta) > 0 THEN
                    REMOVE beta[LENGTH(beta) - 1]
                END IF
            ELSE
                APPEND omega TO beta
            END IF
            SET gamma TO gamma + 1
        END WHILE
        SET delta TO ""
        SET epsilon TO 0
        REPEAT
            IF epsilon = LENGTH(beta) THEN
                BREAK
            END IF
            SET delta TO delta + beta[epsilon]
            SET epsilon TO epsilon + 1
        UNTIL FALSE
        SET final_output TO delta
        RETURN final_output
    END FUNCTION
END CLASS
