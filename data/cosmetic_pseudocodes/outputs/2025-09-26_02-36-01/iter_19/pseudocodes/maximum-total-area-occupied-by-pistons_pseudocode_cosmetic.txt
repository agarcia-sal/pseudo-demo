CLASS Solution
    FUNCTION maxArea(aNmgkl Ukdjqlz Ygqixugj)
        SET mwnxje TO LENGTH(Ukdjqlz)
        SET Wmwdvt TO 0
        FOR Xtdtjc FROM 0 TO mwnxje - 1
            SET Wmwdvt TO Wmwdvt + Ukdjqlz[Xtdtjc]
        END FOR
        SET jrsnpc TO 0
        WHILE jrsnpc < (aNmgkl * 2)
            SET epukwq TO 0
            WHILE epukwq < mwnxje
                IF Ukdjqlz[epukwq] = 0 AND Ygqixugj[epukwq] = "D"
                    SET Ygqixugj TO CONCAT(SUBSTRING(Ygqixugj, 0, epukwq), "U", SUBSTRING(Ygqixugj, epukwq + 1, LENGTH(Ygqixugj)))
                ELSE IF Ukdjqlz[epukwq] = aNmgkl AND Ygqixugj[epukwq] = "U"
                    SET Ygqixugj TO CONCAT(SUBSTRING(Ygqixugj, 0, epukwq), "D", SUBSTRING(Ygqixugj, epukwq + 1, LENGTH(Ygqixugj)))
                END IF
                IF Ygqixugj[epukwq] = "U"
                    SET Ukdjqlz[epukwq] TO Ukdjqlz[epukwq] + 1
                ELSE
                    SET Ukdjqlz[epukwq] TO Ukdjqlz[epukwq] - 1
                END IF
                SET epukwq TO epukwq + 1
            END WHILE
            SET Lruwty TO 0
            FOR Hxqvwl FROM 0 TO mwnxje - 1
                SET Lruwty TO Lruwty + Ukdjqlz[Hxqvwl]
            END FOR
            IF Wmwdvt < Lruwty
                SET Wmwdvt TO Lruwty
            END IF
            SET jrsnpc TO jrsnpc + 1
        END WHILE
        RETURN Wmwdvt
    END FUNCTION
END CLASS
