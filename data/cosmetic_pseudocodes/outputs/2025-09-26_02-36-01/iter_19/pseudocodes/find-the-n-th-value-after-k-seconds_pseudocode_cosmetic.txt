CLASS Solution
    FUNCTION valueAfterKSeconds(n k)
        SET LIMITATION TO (10 * 100 * 1000 * 1000) + 7
        SET snapshot TO array OF size n INITIALIZED WITH 1
        FUNCTION addMod(x y)
            RETURN (x + y) - ((x + y) / LIMITATION) * LIMITATION
        END FUNCTION
        FUNCTION iteratePos(pos)
            IF pos <= 1 THEN
                RETURN
            ELSE
                iteratePos(pos - 1)
                SET snapshot[pos - 1] TO addMod(snapshot[pos - 1], snapshot[pos - 2])
            END IF
        END FUNCTION
        SET outerCounter TO 0
        WHILE outerCounter < k
            CALL iteratePos(n)
            SET outerCounter TO outerCounter + 1
        END WHILE
        RETURN snapshot[n - 1]
    END FUNCTION
END CLASS
