CLASS Solution  
    FUNCTION minOperations(Sigma)  
        SET Delta TO 0  
        SET Omega TO 0  
        DEFINE FUNCTION remainder_modulus(x, y)  
            RETURN x - y * (x DIV y)  
        END FUNCTION  
        SET index TO 0  
        WHILE index < LENGTH(Sigma)  
            SET element TO Sigma[index]  
            SET temp_var TO 0  
            IF remainder_modulus(Omega, 2) = 0 THEN  
                SET temp_var TO element  
            ELSE  
                SET temp_var TO 1 - element  
            END IF  
            IF temp_var = 0 THEN  
                SET Delta TO Delta + 1  
                SET Omega TO Omega + 1  
            END IF  
            SET index TO index + 1  
        END WHILE  
        RETURN Delta  
    END FUNCTION  
END CLASS
