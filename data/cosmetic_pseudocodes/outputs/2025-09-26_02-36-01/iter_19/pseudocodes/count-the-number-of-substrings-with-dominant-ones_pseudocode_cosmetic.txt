CLASS Solution  
    FUNCTION numberOfSubstrings(s)  
      SET lengthVar TO 0  
      SET accumulator TO 0  
      
      lengthVar = (1 + (s LENGTH) + (-1))  
      
      SET alpha TO 0  
      SET beta TO 0  
      SET pointerA TO 0  
      SET pointerB TO 0  
      SET tempFlag TO false  
      
      SET returnVal TO 0  
      
      pointerA = 0  
      WHILE pointerA <= (lengthVar - 1)  
        alpha = 0  
        beta = 0  
        pointerB = pointerA  
        REPEAT  
          tempFlag = s[pointerB] = '1'  
          IF tempFlag = true THEN  
            alpha = alpha + 1  
          ELSE  
            beta = beta + 1  
          END IF  
          
          IF alpha >= (beta * beta) THEN  
            returnVal = returnVal + 1  
          END IF  
          
          pointerB = pointerB + 1  
        UNTIL pointerB > (lengthVar - 1)  
        pointerA = pointerA + 1  
      END WHILE  
      RETURN returnVal  
    END FUNCTION  
END CLASS
