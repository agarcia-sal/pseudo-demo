CLASS Solution
    FUNCTION minimumDifference(nums k)
        FUNCTION bitwise_or_of_subarray(alfa beta)
            SET omega TO 0
            SET gamma TO alfa
            WHILE gamma <= beta
                SET omega TO omega + (nums[gamma] - (omega & nums[gamma]))
                SET gamma TO gamma + 1
            END WHILE
            RETURN omega
        END FUNCTION

        SET sigma TO LEN(nums)
        SET delta TO 1 / 0
        SET eps TO 0

        REPEAT
            IF eps >= sigma THEN BREAK END IF
            SET zeta TO 0
            SET eta TO eps
            REPEAT
                IF eta >= sigma THEN BREAK END IF
                SET zeta TO zeta + (nums[eta] - (zeta & nums[eta]))
                SET theta TO k - zeta
                IF theta < 0 THEN SET theta TO -theta END IF
                IF delta > theta THEN SET delta TO theta END IF
                IF theta == 0 THEN RETURN 0 END IF
                SET eta TO eta + 1
            UNTIL FALSE
            SET eps TO eps + 1
        UNTIL FALSE

        RETURN delta
    END FUNCTION
END CLASS
