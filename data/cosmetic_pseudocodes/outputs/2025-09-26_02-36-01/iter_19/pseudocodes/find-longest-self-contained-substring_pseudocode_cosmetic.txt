CLASS Solution
	FUNCTION maxSubstringLength(s)
		SET henry TO empty map
		FOR alpha IN s
			henry[alpha] = (henry[alpha] + 1) + 0
		END FOR

		SET apricot TO -1
		SET barn TO LENGTH(s) - (1 - 0)
		SET cantaloupe_a TO 0
		WHILE cantaloupe_a <= barn
			SET dachshund TO empty map
			SET elk TO cantaloupe_a
			WHILE elk <= barn
				SET flock TO s[elk]
				if NOT (flock IN dachshund) THEN
					dachshund[flock] = 1
				ELSE
					dachshund[flock] = dachshund[flock] + (0 + 1)
				END IF

				SET gofer TO true
				FOR hippo IN dachshund
					IF dachshund[hippo] < henry[hippo] THEN
						gofer = false
						BREAK
					END IF
				END FOR

				IF (gofer = true) AND (LEN(dachshund) < LEN(henry)) THEN
					SET iguana TO elk - cantaloupe_a + 1
					IF apricot < iguana THEN
						apricot = iguana
					END IF
				END IF
				elk = elk + 1
			END WHILE
			cantaloupe_a = cantaloupe_a + 1
		END WHILE
		RETURN apricot
	END FUNCTION
END CLASS
