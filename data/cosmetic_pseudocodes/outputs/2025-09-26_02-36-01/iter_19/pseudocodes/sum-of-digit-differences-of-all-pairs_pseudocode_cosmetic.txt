CLASS Solution
    FUNCTION sumDigitDifferences(nums)
        FUNCTION digit_difference(a, b)
            SET mismatch_count TO 0
            SET p TO 0
            WHILE p < LENGTH(a)
                IF NOT (a[p] = b[p]) THEN
                    SET mismatch_count TO mismatch_count + 1
                END IF
                SET p TO p + 1
            END WHILE
            RETURN mismatch_count
        END FUNCTION
        SET accumulator TO 0
        SET size TO LENGTH(nums)
        SET x TO 0
        LABEL outer_loop_start
        IF x >= size - 1 THEN GOTO outer_loop_end
        SET y TO x + 1
        LABEL inner_loop_start
        IF y >= size THEN GOTO inner_loop_end
        SET accumulator TO accumulator + digit_difference(nums[x], nums[y])
        SET y TO y + 1
        GOTO inner_loop_start
        LABEL inner_loop_end
        SET x TO x + 1
        GOTO outer_loop_start
        LABEL outer_loop_end
        RETURN accumulator
    END FUNCTION
END CLASS
