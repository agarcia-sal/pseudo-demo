CLASS Solution  
    FUNCTION maximumStrength(nums k)  
    SET lenZ TO LENGTH OF nums  
    DECLARE tabA AS LIST OF LISTS WITH (lenZ + 1) ELEMENTS EACH CONTAINING (k + 1) ELEMENTS INITIALIZED TO NEGATIVE_INFINITY  
    ASSIGN tabA[0][0] TO 0  
    SET w TO 1  
  
    WHILE w <= lenZ DO  
      SET x TO 1  
      DO  
        SET tempSum TO 0  
        SET v TO w  
        WHILE v >= 1 DO  
          SET tempSum TO tempSum + nums[v - 1]  
          IF (x % 2) == 1 THEN  
            SET dir TO (k - x - 1) + 1  
          ELSE  
            SET dir TO -((k - x - 1) + 1)  
          END IF  
          SET tabA[w][x] TO MAXIMUM(tabA[w][x], tabA[v - 1][x - 1] + dir * tempSum)  
          SET v TO v - 1  
        END WHILE  
        SET tabA[w][x] TO MAXIMUM(tabA[w][x], tabA[w - 1][x])  
        SET x TO x + 1  
      LOOP WHILE x <= k  
      SET w TO w + 1  
    END WHILE  
  
    RETURN tabA[lenZ][k]  
    END FUNCTION  
END CLASS
