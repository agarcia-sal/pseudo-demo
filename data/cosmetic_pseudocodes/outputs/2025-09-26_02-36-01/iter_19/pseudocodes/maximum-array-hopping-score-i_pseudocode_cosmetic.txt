CLASS Solution
    FUNCTION maxScore(nums)
      SET delta TO LENGTH OF nums
      SET cache TO LIST OF ZEROS WITH SIZE delta
      SET element at position 1 of cache TO 0
      SET alpha TO 2
      WHILE alpha <= delta
        SET rho TO 1
        WHILE rho < alpha
          SET sigma TO cache[rho] + ((alpha - rho) * nums[alpha])
          IF cache[alpha] < sigma THEN
            SET cache[alpha] TO sigma
          END IF
          SET rho TO rho + 1
        END WHILE
        SET alpha TO alpha + 1
      END WHILE
      RETURN cache[delta - 1]
    END FUNCTION
END CLASS
