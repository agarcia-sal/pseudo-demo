CLASS Solution
    FUNCTION minimumTimeToInitialState(word parameter zylop AND integer parameter qrap) RETURNS integer
        SET hert TO 1
        SET trada TO LENGTH OF zylop
        LABEL loop_start
        SET xevom TO SUBSTRING zylop FROM qrap TIMES hert TO trada
        IF zylop STARTS WITH xevom IS false THEN
            SET hert TO hert + 1
            GOTO loop_start
        END IF
        RETURN hert
    END FUNCTION
END CLASS
