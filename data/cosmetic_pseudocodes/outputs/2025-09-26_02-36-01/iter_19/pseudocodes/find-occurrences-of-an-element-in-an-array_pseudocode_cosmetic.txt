CLASS Solution
    FUNCTION occurrencesOfElement(nums, queries, x)
        SET alpha TO []
        SET beta TO LENGTH OF nums
        SET gamma TO 0
        WHILE gamma < beta
            SET delta TO nums[gamma]
            IF NOT (delta NOT EQUAL TO x)
                APPEND gamma TO alpha
            END IF
            SET gamma TO gamma + 1
        END WHILE
        SET epsilon TO []
        SET zeta TO LENGTH OF queries
        SET eta TO 0
        REPEAT
            IF eta >= zeta THEN
                BREAK
            END IF
            SET theta TO queries[eta]
            IF NOT (theta > LENGTH OF alpha)
                SET iota TO alpha[theta - 1]
                APPEND iota TO epsilon
            ELSE
                APPEND -1 TO epsilon
            END IF
            SET eta TO eta + 1
        UNTIL FALSE
        RETURN epsilon
    END FUNCTION
END CLASS
