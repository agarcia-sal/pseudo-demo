CLASS Solution
	FUNCTION minAnagramLength(k)
		DECLARE countSet AS SET
		LET countSet = CreateEmptySet()
		LET index = 0
		WHILE index < LengthOf(k)
			CALL InsertIntoSet(countSet, GetCharAt(k, index))
			index = index + 1
		END WHILE
		RETURN GetSetSize(countSet)
	END FUNCTION

	FUNCTION CreateEmptySet()
		RETURN new Set()
	END FUNCTION

	FUNCTION InsertIntoSet(ctSet, val)
		ctSet.Add(val)
	END FUNCTION

	FUNCTION GetCharAt(str, pos)
		RETURN str[pos]
	END FUNCTION

	FUNCTION LengthOf(x)
		RETURN x.Length
	END FUNCTION

	FUNCTION GetSetSize(xSet)
		RETURN xSet.Size()
	END FUNCTION
END CLASS
