CLASS Solution  
    FUNCTION mostFrequentIDs(nums PARAMETER, freq PARAMETER)  
        SET alpha TO an empty map with default 0  
        SET omega TO an empty heap container  
        SET sigma TO an empty sequence  
        FOR idx FROM 0 TO LENGTH(nums) - 1  
            SET phi TO nums[idx]  
            SET chi TO freq[idx]  
            SET tempVal TO alpha[phi] + chi  
            SET alpha[phi] TO tempVal  
            CALL omega.INSERT( (-tempVal, phi) )  
            WHILE LENGTH(omega) > 0 AND -omega[0][0] != alpha[omega[0][1]]  
                CALL omega.REMOVE_TOP()  
            END WHILE  
            IF LENGTH(omega) == 0  
                APPEND 0 TO sigma  
            ELSE  
                SET topElem TO omega[0]  
                APPEND (-topElem[0]) TO sigma  
            END IF  
        END FOR  
        RETURN sigma  
    END FUNCTION  
END CLASS
