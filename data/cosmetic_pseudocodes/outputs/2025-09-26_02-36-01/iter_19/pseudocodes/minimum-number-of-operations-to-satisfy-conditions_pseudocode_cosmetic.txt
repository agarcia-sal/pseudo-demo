CLASS Solution
    FUNCTION minimumOperations(grid)
        SET count TO 0
        SET height TO LENGTH OF grid
        SET width TO LENGTH OF element at position 0 of grid

        SET colIdx TO 0
        WHILE colIdx < width
            SET rowIdx TO 0
            WHILE rowIdx < height - 1
                IF NOT (element at position rowIdx of grid AT position colIdx = element at position rowIdx + 1 of grid AT position colIdx)
                    SET count TO count + 1
                    SET element at position rowIdx + 1 of grid AT position colIdx TO element at position rowIdx of grid AT position colIdx
                END IF
                SET rowIdx TO rowIdx + 1
            END WHILE

            SET idx TO 0
            REPEAT
                IF colIdx < width - 1 AND (element at position idx of grid AT position colIdx = element at position idx of grid AT position colIdx + 1)
                    SET count TO count + 1
                    SET replacementNumber TO 0
                    SET done TO FALSE
                    WHILE replacementNumber <= 9 AND NOT done
                        IF replacementNumber â‰  element at position idx of grid AT position colIdx
                            SET element at position idx of grid AT position colIdx + 1 TO replacementNumber
                            SET done TO TRUE
                        END IF
                        SET replacementNumber TO replacementNumber + 1
                    END WHILE
                END IF
                SET idx TO idx + 1
            UNTIL idx = height

            SET colIdx TO colIdx + 1
        END WHILE

        RETURN count
    END FUNCTION
END CLASS
