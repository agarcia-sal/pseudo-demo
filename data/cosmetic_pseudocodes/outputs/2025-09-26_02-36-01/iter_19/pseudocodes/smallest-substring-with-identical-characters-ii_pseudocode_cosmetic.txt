CLASS Solution
    FUNCTION minLength(s parameter representing a string, numOps parameter representing an integer) RETURNS integer
        FUNCTION check(m parameter representing an integer) RETURNS boolean
            SET alpha TO 0
            SET delta TO 0
            SET i TO 0
            LABEL loop_start:
            IF NOT (i < LENGTH(s)) GOTO loop_end

            SET delta TO delta + 1

            IF (i = LENGTH(s) - 1) OR (s[i] != s[i+1]) THEN
                SET alpha TO alpha + ((delta - 1) DIV m) + 1
                IF alpha > numOps THEN
                    RETURN false
                END IF
                SET delta TO 0
            END IF

            SET i TO i + 1
            GOTO loop_start
            LABEL loop_end:

            RETURN (alpha <= numOps)
        END FUNCTION

        SET total TO LENGTH(s)
        SET low TO 1
        SET high TO total

        WHILE low <> high
            SET pivot TO low + ((high - low) DIV 2)
            IF check(pivot) = true THEN
                SET high TO pivot
            ELSE
                SET low TO pivot + 1
            END IF
        END WHILE

        RETURN low
    END FUNCTION
END CLASS
