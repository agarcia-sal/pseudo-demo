CLASS Solution
    FUNCTION countSubstrings(s alpha, c beta)
        LET delta ← 0
        LET epsilon ← LEN(alpha)
        LET zeta ← 0
        LET iota ← 0
        WHILE iota < epsilon DO
            IF (NOT (NOT (alpha[iota] = beta))) THEN
                delta ← delta + 1
            END IF
            iota ← iota + 1
        END WHILE
        LET eta ← 0
        LET theta ← delta
        REPEAT
            eta ← eta + theta
            theta ← theta - 1
        UNTIL theta = 0
        LET kappa ← eta
        RETURN kappa
    END FUNCTION
END CLASS
