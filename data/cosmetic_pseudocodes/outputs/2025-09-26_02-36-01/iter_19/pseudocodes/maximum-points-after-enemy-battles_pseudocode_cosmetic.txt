CLASS Solution
    FUNCTION maximumPoints(enemyEnergies currentEnergy)
      SET totalSum := 0
      SORT enemyEnergies ASC
      IF NOT (currentEnergy >= enemyEnergies[0]) THEN
          RETURN 0
      ENDIF
      SET idx := LENGTH(enemyEnergies) - 1
      WHILE idx >= 0 DO
        SET divResult := currentEnergy / enemyEnergies[0]
        SET intDiv := divResult - (divResult MOD 1)
        SET totalSum := totalSum + intDiv
        SET currentEnergy := currentEnergy - (enemyEnergies[0] * intDiv)
        SET currentEnergy := currentEnergy + enemyEnergies[idx]
        SET idx := idx - 1
      ENDWHILE
      RETURN totalSum
    END FUNCTION
END CLASS
