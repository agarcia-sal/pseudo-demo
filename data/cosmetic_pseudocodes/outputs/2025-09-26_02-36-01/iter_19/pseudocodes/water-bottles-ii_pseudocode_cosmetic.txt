CLASS Solution
    FUNCTION maxBottlesDrunk(numBottles numBottles numExchange numExchange)
      SET s_alpha TO 0
      SET t_zeta TO 0
      REPEAT
        IF numBottles <= 0 THEN
          BREAK
        END IF
        SET s_alpha TO s_alpha + numBottles
        SET t_zeta TO t_zeta + numBottles
        SET numBottles TO 0
        WHILE t_zeta >= numExchange DO
          SET t_zeta TO t_zeta - numExchange
          SET numBottles TO numBottles + 1
          SET numExchange TO numExchange + 1
        END WHILE
      UNTIL false
      RETURN s_alpha
    END FUNCTION
END CLASS
