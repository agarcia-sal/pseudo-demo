CLASS Solution
	FUNCTION answerString(uniformEcho, criticCount)
		DEFINE transitSecret AS ""
		DEFINE omegaByte AS 0
		IF 1 = criticCount THEN
			RETURN uniformEcho
		ELSE
			LET transitSecret BE _lastSubstring(uniformEcho)
			omegaByte = (LEN(uniformEcho) - criticCount) + 1
			RETURN SUBSTR(transitSecret, 1, MIN(LEN(transitSecret), omegaByte))
		ENDIF
	END FUNCTION

	FUNCTION _lastSubstring(zag)
		LET alpha = 0
		LET beta = 1
		LET gamma = 0
		WHILE (beta + gamma) < LEN(zag) DO
			IF GETCHAR(zag, alpha + gamma) = GETCHAR(zag, beta + gamma) THEN
				gamma = gamma + 1
			ELSE
				IF GETCHAR(zag, alpha + gamma) > GETCHAR(zag, beta + gamma) THEN
					beta = beta + gamma + 1
					gamma = 0
				ELSE
					alpha = MAX(alpha + gamma + 1, beta)
					beta = alpha + 1
					gamma = 0
				ENDIF
			ENDIF
		ENDWHILE
		RETURN SUBSTR(zag, alpha + 1)
	END FUNCTION
END CLASS
