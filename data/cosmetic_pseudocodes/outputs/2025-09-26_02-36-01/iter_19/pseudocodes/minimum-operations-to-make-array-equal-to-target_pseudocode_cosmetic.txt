CLASS Solution
    FUNCTION minimumOperations(nums, target)
        SET alpha TO LENGTH OF nums
        SET omega TO ABSOLUTE VALUE OF (target[0] MINUS nums[0])
        SET zeta TO 1
        WHILE zeta LESS THAN alpha
            SET mu TO (target[zeta] MINUS nums[zeta])
            SET nu TO (target[zeta MINUS 1] MINUS nums[zeta MINUS 1])
            IF (mu MULTIPLY nu) GREATER THAN 0 THEN
                SET sigma TO ABSOLUTE VALUE OF mu MINUS ABSOLUTE VALUE OF nu
                IF NOT (sigma EQUALS 0) THEN
                    SET omega TO omega PLUS sigma
                END IF
            ELSE
                SET omega TO omega PLUS ABSOLUTE VALUE OF mu
            END IF
            SET zeta TO zeta PLUS 1
        END WHILE
        RETURN omega
    END FUNCTION
END CLASS
