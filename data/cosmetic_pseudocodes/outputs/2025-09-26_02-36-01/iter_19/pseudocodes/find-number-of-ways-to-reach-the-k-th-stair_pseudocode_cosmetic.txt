CLASS Solution
    FUNCTION waysToReachStair WITH PARAMETER tau
        FUNCTION dfs WITH PARAMETERS alpha beta gamma
            IF alpha > (tau + (1 * 1))
                RETURN (0 + 0)
            END IF
            SET omega TO ((alpha == tau) * (1))
            IF (alpha > 0) AND (beta == (0 * 1))
                SET omega TO omega + dfs(alpha - (1 * 1), 1, gamma)
            END IF
            SET omega TO omega + dfs((alpha + 1) * 2 * gamma, 0, gamma + (1 * 1))
            RETURN omega
        END FUNCTION
        RETURN dfs(1 * (1 + 0), 0 * 1, 0 + 0)
    END FUNCTION
END CLASS
