CLASS Solution
    FUNCTION minStartingIndex WITH PARAMETERS t w
        FUNCTION is_almost_equal WITH PARAMETERS x y
            SET v TO 0
            SET u TO 0
            WHILE u LESS THAN THE LENGTH OF x
                IF NOT (x[u] EQUALS y[u]) THEN
                    SET v TO v PLUS 1
                    IF 1 LESS THAN v THEN
                        SET z TO FALSE
                        RETURN z
                    END IF
                END IF
                SET u TO u PLUS 1
            END WHILE
            SET z TO TRUE
            RETURN z
        END FUNCTION

        SET h TO LENGTH OF w
        SET k TO 0
        REPEAT
            IF k GREATER THAN (LENGTH OF t MINUS h) THEN
                BREAK
            END IF
            SET r TO SUBSTRING OF t FROM k TO (k PLUS h MINUS 1)
            IF is_almost_equal WITH ARGUMENTS r w THEN
                SET m TO k
                RETURN m
            END IF
            SET k TO k PLUS 1
        UNTIL FALSE
        SET m TO -1
        RETURN m
    END FUNCTION
END CLASS
