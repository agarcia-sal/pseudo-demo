CLASS Solution
    FUNCTION doesAliceWin(jump)
              SET constants TO {p: "a", q: "e", r: "i", s: "o", t: "u"}
              SET counter_1 TO 0
              SET temp_2 TO 0

          INDEX idx = 0
          WHILE idx < LENGTH(jump)
              SET ch_3 TO jump[idx]
              IF ch_3 = constants.p OR ch_3 = constants.q OR ch_3 = constants.r OR ch_3 = constants.s OR ch_3 = constants.t
                  ADD 1 TO temp_2
              END IF
              ADD 1 TO idx
          END WHILE

          SET flagX TO FALSE

          INDEX zeta = 0
          WHILE zeta < LENGTH(jump)
                  SET sym_4 TO jump[zeta]
              IF sym_4 = constants.p OR sym_4 = constants.q OR sym_4 = constants.r OR sym_4 = constants.s OR sym_4 = constants.t
                  SET flagX TO NOT flagX
              END IF

                  IF flagX = FALSE AND (temp_2 MOD 2) = 1
                          SET counter_1 TO counter_1 + 1
                          SET temp_2 TO 0
                  ELSE IF flagX = TRUE
                          SET temp_2 TO temp_2 + 1
                  END IF

              ADD 1 TO zeta
          END WHILE

          IF flagX = TRUE AND (temp_2 MOD 2) = 1
              SET counter_1 TO counter_1 + 1
          END IF

          RETURN (counter_1 MOD 2) = 1
    END FUNCTION
END CLASS
