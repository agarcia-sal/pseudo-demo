CLASS Solution
	FUNCTION maxSubarraySum(nums)
		FUNCTION kadane(arr)
			SET alpha TO arr[1]
			SET beta TO arr[1]
			SET gamma TO 2
			WHILE gamma LESS THAN OR EQUAL TO LENGTH(arr)
				SET delta TO arr[gamma]
				IF NOT (delta LESS OR EQUAL TO alpha PLUS delta)
					SET alpha TO delta
				ELSE
					SET alpha TO alpha PLUS delta
				END IF
				IF beta LESS THAN alpha
					SET beta TO alpha
				END IF
				SET gamma TO gamma PLUS 1
			END WHILE
			RETURN beta
		END FUNCTION

		SET omega TO kadane(nums)
		SET epsilon TO EMPTY SET
		SET zeta TO 1
		WHILE zeta LESS OR EQUAL TO LENGTH(nums)
			ADD nums[zeta] TO epsilon
			SET zeta TO zeta PLUS 1
		END WHILE

		SET theta TO FIRST ELEMENT IN epsilon
		SET iota TO 1
		WHILE iota LESS OR EQUAL TO LENGTH(epsilon)
			SET rho TO EMPTY LIST

			SET kappa TO 1
			WHILE kappa LESS OR EQUAL TO LENGTH(nums)
				SET psi TO nums[kappa]
				IF NOT (psi EQUALS epsilon[iota])
					APPEND psi TO rho
				END IF
				SET kappa TO kappa PLUS 1
			END WHILE

			IF LENGTH(rho) GREATER THAN 0
				SET sigma TO kadane(rho)
				IF omega LESS THAN sigma
					SET omega TO sigma
				END IF
			END IF

			SET iota TO iota PLUS 1
		END WHILE

		RETURN omega
	END FUNCTION
END CLASS
