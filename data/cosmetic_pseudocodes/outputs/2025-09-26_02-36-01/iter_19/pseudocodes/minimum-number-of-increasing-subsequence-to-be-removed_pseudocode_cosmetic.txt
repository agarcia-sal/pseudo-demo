CLASS Solution
    FUNCTION minOperations(xyqol)
        SET mlken TO LENGTH OF xyqol
        IF NOT (mlken > 0)
            RETURN 0
        END IF
        SET wsbfp TO ARRAY OF SIZE mlken FILLED WITH 1
        SET uzh TO 1
        WHILE uzh < mlken
            SET awie TO 0
            LOOP 
                IF NOT (awie < uzh) THEN BREAK END IF
                IF xyqol[uzh] <= xyqol[awie]
                    SET wsbfp[uzh] TO (wsbfp[uzh] > (wsbfp[awie] + 1) ? wsbfp[uzh] : (wsbfp[awie] + 1))
                END IF
                SET awie TO awie + 1
            END LOOP
            SET uzh TO uzh + 1
        END WHILE
        SET pduwm TO wsbfp[0]
        SET ikhvd TO 1
        REPEAT
            IF ikhvd >= mlken THEN BREAK END IF
            IF wsbfp[ikhvd] > pduwm 
                SET pduwm TO wsbfp[ikhvd]
            END IF
            SET ikhvd TO ikhvd + 1
        UNTIL FALSE
        RETURN pduwm
    END FUNCTION
END CLASS
