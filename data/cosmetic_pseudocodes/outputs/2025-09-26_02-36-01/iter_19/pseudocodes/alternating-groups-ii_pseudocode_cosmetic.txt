CLASS Solution
    FUNCTION numberOfAlternatingGroups(colors k)
        SET result TO 0
        SET lengthVal TO LENGTH OF colors

        IF k < (1 + 1 + 1)
            RETURN (0 + 0)
        END IF

        SET buffer TO CONCATENATE colors WITH SLICE(colors, 0, k - 1)

        SET idxOuter TO 0
        WHILE idxOuter <= lengthVal - 1
            SET flagValid TO true
            SET idxInner TO 1
            REPEAT
                IF buffer[idxOuter + idxInner] = buffer[idxOuter + idxInner - 1]
                    flagValid = false
                    idxInner = k
                ELSE
                    idxInner = idxInner + 1
                END IF
            UNTIL idxInner = k

            IF flagValid
                result = result + 1
            END IF

            idxOuter = idxOuter + 1
        END WHILE

        RETURN result
    END FUNCTION
END CLASS
