CLASS Solution  
    FUNCTION minMovesToCaptureTheQueen(a b c d e f)  
        DECLARE x = 2  
        DECLARE y = 1  
        DECLARE z = (x - 1)  
        DECLARE p = (b < d)  
        DECLARE q = (d < f)  
        DECLARE r = (f < d)  
        DECLARE s = (b > d)  
        DECLARE t = (c < a)  
        DECLARE u = (a < e)  
        DECLARE v = (e < a)  
        DECLARE w = (d > b)  
        DECLARE m = (b > f)  
        DECLARE n = (f > b)  
        DECLARE alpha = (a + b)  
        DECLARE beta = (c + d)  
        DECLARE gamma = (e + f)  
        DECLARE delta = (c - d)  
        DECLARE epsilon = (e - f)  
        DECLARE eta = (a - b)  
        DECLARE theta = (c - d)  
        DECLARE i = FALSE  
        DECLARE shouldReturn = FALSE  
        DECLARE result = x

        IF (a = e) THEN  
            IF NOT (c = a) THEN  
                result = y  
                shouldReturn = TRUE  
            END IF  
            IF ((b < d AND d < f) OR (f < d AND d < b)) AND (NOT shouldReturn) THEN  
                result = x  
                shouldReturn = TRUE  
            END IF  
            IF NOT shouldReturn THEN  
                result = y  
                shouldReturn = TRUE  
            END IF  
        ELSE IF (b = f) THEN  
            IF NOT (d = b) THEN  
                result = y  
                shouldReturn = TRUE  
            END IF  
            IF ((a < c AND c < e) OR (e < c AND c < a)) AND (NOT shouldReturn) THEN  
                result = x  
                shouldReturn = TRUE  
            END IF  
            IF NOT shouldReturn THEN  
                result = y  
                shouldReturn = TRUE  
            END IF  
        ELSE IF ((c + d) = (e + f)) THEN  
            IF NOT ((a + b) = (c + d)) THEN  
                result = y  
                shouldReturn = TRUE  
            END IF  
            IF (((c < a AND a < e) AND (d < b AND b < f)) OR ((e < a AND a < c) AND (f < b AND b < d))) AND (NOT shouldReturn) THEN  
                result = x  
                shouldReturn = TRUE  
            END IF  
            IF NOT shouldReturn THEN  
                result = y  
                shouldReturn = TRUE  
            END IF  
        ELSE IF ((c - d) = (e - f)) THEN  
            IF NOT ((a - b) = (c - d)) THEN  
                result = y  
                shouldReturn = TRUE  
            END IF  
            IF (((c < a AND a < e) AND (d > b AND b > f)) OR ((e < a AND a < c) AND (f > b AND b > d))) AND (NOT shouldReturn) THEN  
                result = x  
                shouldReturn = TRUE  
            END IF  
            IF NOT shouldReturn THEN  
                result = y  
                shouldReturn = TRUE  
            END IF  
        ELSE  
            result = x  
            shouldReturn = TRUE  
        END IF

        RETURN result  
    END FUNCTION  
END CLASS
