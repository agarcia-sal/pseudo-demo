CLASS Solution
    FUNCTION minEnd(n q)
        FUNCTION canConstruct(z)
            SET alfa TO q
            SET bravo TO 1
            DEFINE recurse():
                IF alfa >= z THEN RETURN bravo == n END
                SET alfa TO alfa + 1
                IF ((alfa & q) == q) THEN
                    SET bravo TO bravo + 1
                    IF bravo == n THEN RETURN True END
                END
                RETURN recurse()
            END
            RETURN recurse()
        END FUNCTION

        SET charlie TO q
        SET delta TO (2 * 100000000)
        WHILE charlie < delta
            SET echo TO (charlie + delta) // 2
            IF canConstruct(echo) THEN
                SET delta TO echo
            ELSE
                SET charlie TO charlie + 1
            END
        END WHILE

        RETURN charlie
    END FUNCTION
END CLASS
