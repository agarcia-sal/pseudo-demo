```pseudocode
CLASS Solution
    FUNCTION minRectanglesToCoverPoints(points, w)
        DEFINE function sortPointsByX(inputList)
            IF LENGTH(inputList) <= 1 THEN
                RETURN inputList
            END IF
            SET pivotIndex TO LENGTH(inputList) DIV 2
            SET pivotValue TO inputList[pivotIndex][0]
            SET leftPartition TO []
            SET rightPartition TO []
            FOR index FROM 0 TO LENGTH(inputList) - 1
                IF index IS NOT pivotIndex THEN
                    IF inputList[index][0] <= pivotValue THEN
                        APPEND inputList[index] TO leftPartition
                    ELSE
                        APPEND inputList[index] TO rightPartition
                    END IF
                END IF
            END FOR
            RETURN CONCATENATE(sortPointsByX(leftPartition), [inputList[pivotIndex]], sortPointsByX(rightPartition))
        END FUNCTION

        SET sortedList TO sortPointsByX(points)
        SET rectCount TO 0
        SET boundaryLimit TO -1
        SET iterator TO 0
        WHILE iterator < LENGTH(sortedList)
            SET pointX TO sortedList[iterator][0]
            SET pointY TO sortedList[iterator][1]
            IF boundaryLimit < pointX THEN
                SET boundaryLimit TO pointX + w
                SET rectCount TO rectCount + 1
            END IF
            SET iterator TO iterator + 1
        END WHILE
        RETURN rectCount
    END FUNCTION
END CLASS
```
