CLASS Solution
    FUNCTION countAlternatingSubarrays(nums)
        SET p TO 0
        SET q TO LENGTH OF nums
        IF q EQUALS 0 THEN
            SET r TO 0
        ELSE
            SET s TO q
            SET t TO 1
            SET u TO 1
            WHILE u < q DO
                IF NOT (nums[u] EQUALS nums[u - 1]) THEN
                    SET t TO t + 1
                ELSE
                    SET t TO 1
                END IF
                SET s TO s + (t - 1)
                SET u TO u + 1
            END WHILE
            SET r TO s
        END IF
        RETURN r
    END FUNCTION
END CLASS
