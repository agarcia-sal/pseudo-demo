CLASS Solution  
    FUNCTION flowerGame(tau, sigma)  
        SET delta TO 1  
        SET gamma TO 2  
        SET kappa TO (tau + delta) / gamma  
        SET lambda TO tau / gamma  
        SET mu TO (sigma + delta) / gamma  
        SET nu TO sigma / gamma  
        FUNCTION computeProduct(alpha, beta)  
            RETURN alpha * beta  
        END FUNCTION  
        SET partOne TO computeProduct(kappa, nu)  
        SET partTwo TO computeProduct(lambda, mu)  
        SET omega TO partOne + partTwo  
        RETURN omega  
    END FUNCTION  
END CLASS
