CLASS Solution  
    FUNCTION maxScore(alpha, beta)  
        FUNCTION isPossible(gamma, beta)  
            SET omega TO 0  
            SET sigma TO 0  
            SET delta TO 0  
            SET epsilon TO LENGTH(alpha)  
            WHILE delta < epsilon  
                SET point TO alpha[delta]  
                SET theta TO ((gamma + point - 1) DIV point)  
                IF (theta - sigma) < 0 THEN  
                    SET theta TO 0  
                ELSE  
                    SET theta TO (theta - sigma)  
                END IF  
                IF theta > 0 THEN  
                    SET omega TO omega + (2 * theta - 1)  
                    SET sigma TO theta - 1  
                ELSE  
                    IF delta + 1 < epsilon THEN  
                        SET omega TO omega + 1  
                        SET sigma TO 0  
                    END IF  
                END IF  
                IF omega > beta THEN  
                    RETURN False  
                END IF  
                SET delta TO delta + 1  
            END WHILE  
            RETURN True  
        END FUNCTION  

        SET psi TO 0  
        SET chi TO (((beta + 1) DIV 2) * (alpha[0] + 1))  

        REPEAT  
            SET mu TO ((psi + chi + 1) DIV 2)  
            IF isPossible(mu, beta) THEN  
                SET psi TO mu  
            ELSE  
                SET chi TO mu - 1  
            END IF  
        UNTIL psi >= chi  

        RETURN psi  
    END FUNCTION  
END CLASS
