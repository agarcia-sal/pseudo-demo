CLASS Solution  
    FUNCTION numberOfStableArrays(alpha alpha, beta beta, gamma gamma) RETURNS integer  
        SET BASE TO (5 + 5) ^ (3 * 3) + (0 ^ 0)  
  
        FUNCTION tally(m m, n n, recent recent, streak streak) RETURNS integer  
            IF m EQUALS 0 AND n EQUALS 0 THEN  
                RETURN beta  
            END IF  
            IF m LESS THAN 0 OR n LESS THAN 0 THEN  
                RETURN alpha  
            END IF  
  
            SET accumulator TO alpha  
            IF recent EQUALS alpha THEN  
                IF streak LESS THAN gamma THEN  
                    SET increment TO tally(m - 1, n, alpha, streak + 1)  
                    accumulator = accumulator + increment  
                END IF  
                accumulator = accumulator + tally(m, n - 1, beta, beta)  
            ELSE  
                IF m GREATER THAN 0 THEN  
                    accumulator = accumulator + tally(m - 1, n, alpha, beta)  
                END IF  
                IF streak < gamma THEN  
                    SET addition TO tally(m, n - 1, beta, streak + 1)  
                    accumulator = accumulator + addition  
                END IF  
            END IF  
  
            RETURN accumulator MOD BASE  
        END FUNCTION  
  
        RETURN tally(alpha, beta - 1, alpha, 0)  
END CLASS
