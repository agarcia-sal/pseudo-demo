CLASS Solution
    FUNCTION numberOfRightTriangles(grid)
        SET alpha TO 0
        SET beta TO LENGTH OF grid
        SET gamma TO LENGTH OF element at position 0 of grid
        SET delta TO 0

        SET epsilon TO 0
        WHILE epsilon < beta
            SET zeta TO 0
            WHILE zeta < gamma
                IF element at position epsilon of grid THEN element at position zeta of element at position epsilon of grid EQUALS 1 THEN
                    SET eta TO 0
                    SET theta TO 0
                    SET iota TO 0
                    WHILE iota < gamma
                        IF iota NOT EQUAL zeta THEN
                            SET eta TO eta PLUS element at position iota of element at position epsilon of grid
                        END IF
                        SET iota TO iota PLUS 1
                    END WHILE
                    SET kappa TO 0
                    SET lambda TO 0
                    SET mu TO 0
                    WHILE mu < beta
                        IF mu NOT EQUAL epsilon THEN
                            SET kappa TO kappa PLUS element at position zeta of element at position mu of grid
                        END IF
                        SET mu TO mu PLUS 1
                    END WHILE
                    SET delta TO delta PLUS (eta TIMES kappa)
                END IF
                SET zeta TO zeta PLUS 1
            END WHILE
            SET epsilon TO epsilon PLUS 1
        END WHILE
        
        RETURN delta
    END FUNCTION
END CLASS
