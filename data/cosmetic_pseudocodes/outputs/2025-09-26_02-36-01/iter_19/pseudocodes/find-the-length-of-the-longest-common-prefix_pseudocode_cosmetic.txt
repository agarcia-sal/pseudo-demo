CLASS Solution
    FUNCTION longestCommonPrefix(arr1 arr1 OF integers arr2 arr2 OF integers) RETURNS integer
        SET omega TO new empty set
        FOR delta IN arr1 DO
            SET phi TO toString(delta)
            SET kappa TO 1
            WHILE kappa <= length(phi) DO
                SET psi TO substring(phi, 1, kappa)
                ADD psi TO omega
                SET kappa TO kappa + 1
            END WHILE
        END FOR

        SET sigma TO set()
        SET theta TO 0
        FOR beta IN arr2 DO
            SET gamma TO toString(beta)
            FOR lambda FROM 1 TO length(gamma) DO
                SET mu TO substring(gamma, 1, lambda)
                ADD mu TO sigma
            END FOR
        END FOR

        FOR alpha IN omega DO
            IF alpha IN sigma THEN
                IF length(alpha) > theta THEN
                    SET theta TO length(alpha)
                END IF
            END IF
        END FOR

        RETURN theta
    END FUNCTION
END CLASS
