CLASS Solution
    FUNCTION maximumLength(nums k)
        SET omega TO LENGTH(nums)
        IF 1 = omega
            RETURN 1
        END IF

        SET beta TO []
        FOR gamma FROM 0 TO omega - 1
            APPEND {} TO beta
        END FOR

        SET sigma TO 1

        FUNCTION lookupAlpha(delta, epsilon)
            RETURN beta[delta][epsilon]
        END FUNCTION

        FOR theta FROM 0 TO omega - 1
            FOR zeta FROM 0 TO theta - 1
                SET lambda TO (nums[theta] + nums[zeta]) MOD k
                IF lambda IN beta[zeta]
                    SET beta[theta][lambda] = beta[zeta][lambda] + 1
                ELSE
                    SET beta[theta][lambda] = 2
                END IF

                IF beta[theta][lambda] > sigma
                    SET sigma = beta[theta][lambda]
                END IF
            END FOR
        END FOR

        RETURN sigma
    END FUNCTION
END CLASS
