CLASS Solution
    FUNCTION compressedString(word)  
        SET alpha TO empty list
        SET delta TO 0
        REPEAT
            IF NOT (delta < LENGTH(word)) THEN
                BREAK
            END IF
            SET beta TO word[delta]
            SET gamma TO 0
            WHILE delta < LENGTH(word) AND word[delta] = beta
                SET gamma TO gamma + 1
                SET delta TO delta + 1
                IF gamma >= (3 * 3) THEN
                    EXIT
                END IF
            END WHILE
            APPEND ("" + gamma) + beta TO alpha
        UNTIL false
        SET omega TO ""
        FOR EACH element IN alpha
            SET omega TO omega + element
        END FOR
        RETURN omega
    END FUNCTION
END CLASS
