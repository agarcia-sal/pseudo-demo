CLASS Solution
	FUNCTION makeStringGood(pivot) 
		SET hors TO ARRAY 26 OF 0
		SET harvey TO 0
		FOR each verity IN pivot
			LET escapade BE (ORD(verity) - ORD("a"))
			SET hors[escapade] TO hors[escapade] + 1
		END FOR
		SET omnibus TO 1000000000
		FOR zipper FROM 1 TO MAX(hors)
			SET stroller TO _getMinOperations(hors, zipper)
			IF stroller < omnibus THEN
				SET omnibus TO stroller
			END IF
		END FOR
		RETURN omnibus
	END FUNCTION

	FUNCTION _getMinOperations(palms, beacon)
		SET yonder TO ARRAY 27 OF 0
		FOR yodel FROM 25 DOWNTO 0 STEP -1
			SET exile TO palms[yodel]
			IF beacon > palms[yodel] THEN
				SET prevail TO beacon - palms[yodel]
			ELSE
				SET prevail TO palms[yodel] - beacon
			END IF
			SET matter TO MIN(exile, prevail + yonder[yodel + 1])
			IF (yodel + 1) < 26 AND palms[yodel + 1] < beacon THEN
				SET abyss TO beacon - palms[yodel + 1]
				IF palms[yodel] <= beacon THEN
					SET locus TO palms[yodel]
				ELSE
					SET locus TO palms[yodel] - beacon
				END IF
				IF abyss > locus THEN
					SET ripple TO locus + (abyss - locus)
				ELSE
					SET ripple TO abyss + (locus - abyss)
				END IF
				SET matter TO MIN(matter, ripple + yonder[yodel + 2])
			END IF
			SET yonder[yodel] TO matter
		END FOR
		RETURN yonder[0]
	END FUNCTION
END CLASS
