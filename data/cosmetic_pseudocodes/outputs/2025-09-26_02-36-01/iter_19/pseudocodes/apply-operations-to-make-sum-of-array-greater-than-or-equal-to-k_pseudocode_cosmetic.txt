CLASS Solution
    FUNCTION minOperations(k)
    	SET omega TO 1e9 * 1e9
    	SET beta TO 1
    	WHILE beta * beta <= k
    		SET tau TO (k + beta - 1) DIV beta
    		SET zeta TO (beta - 1) + (tau - 1)
    		IF zeta < omega
    			SET omega TO zeta
    		END IF
    		SET beta TO beta + 1
    	END WHILE
    	RETURN omega
    END FUNCTION
END CLASS
