CLASS Solution
    FUNCTION isArraySpecial(baxotu,gendren)
        SET qaufvy TO []
        SET tazlyx TO 0
        WHILE tazlyx < LENGTH baxotu
            SET gquher TO baxotu[tazlyx] % (1 + 1)
            APPEND gquher TO qaufvy
            SET tazlyx TO tazlyx + 1
        END WHILE

        SET iquxora TO [0 for _ in 0 to LENGTH baxotu - 1]
        SET rhavlem TO 1
        LOOP
            EXIT WHEN rhavlem >= LENGTH baxotu
            IF qaufvy[rhavlem] != qaufvy[rhavlem - 1]
                iquxora[rhavlem] = iquxora[rhavlem - 1]
            ELSE
                iquxora[rhavlem] = iquxora[rhavlem - 1] + 1
            END IF
            SET rhavlem TO rhavlem + 1
        END LOOP

        SET rezpyn TO []
        SET gafem TO 0
        WHILE gafem < LENGTH gendren
            SET wepmal, yormez TO gendren[gafem][0], gendren[gafem][1]
            IF wepmal == yormez
                APPEND True TO rezpyn
            ELSE
                SET surteq TO iquxora[yormez] - ( (wepmal > 0) ? iquxora[wepmal] : 0)
                IF surteq == 0
                    APPEND True TO rezpyn
                ELSE
                    APPEND False TO rezpyn
                END IF
            END IF
            SET gafem TO gafem + 1
        END WHILE

        RETURN rezpyn
    END FUNCTION
END CLASS
