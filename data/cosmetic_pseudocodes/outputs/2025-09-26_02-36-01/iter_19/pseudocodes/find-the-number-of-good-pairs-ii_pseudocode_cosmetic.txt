CLASS Solution
    FUNCTION numberOfPairs(alpha, beta, gamma)
        SET delta TO EMPTY_MAP
        FOR omega IN beta
            IF omega IN delta THEN
                SET delta[omega] TO delta[omega] + 1
            ELSE
                SET delta[omega] TO 1
            END IF
        END FOR

        SET epsilon TO 0
        SET zeta TO 0
        WHILE zeta < LENGTH(alpha)
            SET theta TO alpha[zeta]
            SET iota TO 0
            VALUES = LIST_OF_KEYS(delta)
            COUNTS = LIST_OF_VALUES(delta)
            SET lamda TO LENGTH(VALUES)
            WHILE iota < lamda
                SET mu TO VALUES[iota]
                SET nu TO COUNTS[iota]
                IF ((theta MOD (mu * gamma)) == 0)
                    SET epsilon TO epsilon + nu
                END IF
                SET iota TO iota + 1
            END WHILE
            SET zeta TO zeta + 1
        END WHILE

        RETURN epsilon
    END FUNCTION
END CLASS
