CLASS Solution  
  FUNCTION minOrAfterOperations(nums k)  
    FUNCTION canAchieve(beta gamma)  
      SET omega TO -1  
      SET phi TO 0  
      FOR xi IN nums  
        IF omega == -1 THEN  
          SET omega = xi  
        ELSE  
          SET omega = omega & xi  
        END IF  
        IF (omega & beta) == 0 THEN  
          SET omega = -1  
        ELSE  
          SET phi = phi + 1  
          IF phi > k THEN  
            RETURN false  
          END IF  
        END IF  
      END FOR  
      RETURN true  
    END FUNCTION

    SET delta TO (2 ^ 30) - 1  
    SET sigma TO delta  
    FOR tau FROM 0 TO 29  
      SET epsilon TO 2 ^ tau  
      IF (sigma & epsilon) == 0 THEN  
        CONTINUE  
      END IF  
      IF canAchieve((~sigma) ^ epsilon k) THEN  
        SET sigma = sigma & (~epsilon)  
      END IF  
    END FOR  
    RETURN sigma  
  END FUNCTION  
END CLASS
