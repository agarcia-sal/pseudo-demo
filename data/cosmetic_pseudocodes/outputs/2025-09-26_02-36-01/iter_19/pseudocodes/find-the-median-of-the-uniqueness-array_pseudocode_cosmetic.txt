CLASS Solution
    FUNCTION medianOfUniquenessArray(nums)
        FUNCTION countLessOrEqual(target)
            SET omega TO 0
            SET sigma TO 0
            ASSIGN tango TO a fresh empty map from element to integer count
            SET phi TO 0

            SET p TO 0
            WHILE p LESS THAN length(nums)
                SET zulu TO nums[p]
                IF tango[zulu] EQUALS 0 THEN
                    SET phi TO phi PLUS 1
                END IF
                SET tango[zulu] TO tango[zulu] PLUS 1

                WHILE phi GREATER THAN target
                    SET mike TO nums[sigma]
                    SET tango[mike] TO tango[mike] MINUS 1
                    IF tango[mike] EQUALS 0 THEN
                        SET phi TO phi MINUS 1
                    END IF
                    SET sigma TO sigma PLUS 1
                END WHILE

                SET omega TO omega PLUS (p MINUS sigma) PLUS 1
                SET p TO p PLUS 1
            END WHILE

            RETURN omega
        END FUNCTION

        SET alpha TO length(nums)
        SET beta TO (alpha MULTIPLIED BY (alpha PLUS 1)) DIVIDED BY 2
        SET lambda TO (beta PLUS 1) DIVIDED BY 2
        SET delta TO 1
        SET epsilon TO alpha

        WHILE delta LESS THAN epsilon
            SET kappa TO delta PLUS (epsilon MINUS delta) DIVIDED BY 2
            IF countLessOrEqual(kappa) LESS THAN lambda THEN
                SET delta TO kappa PLUS 1
            ELSE
                SET epsilon TO kappa
            END IF
        END WHILE

        RETURN delta
    END FUNCTION
END CLASS
