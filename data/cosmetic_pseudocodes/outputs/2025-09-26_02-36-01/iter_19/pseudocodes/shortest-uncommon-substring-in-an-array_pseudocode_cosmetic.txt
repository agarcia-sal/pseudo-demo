CLASS Solution
    FUNCTION shortestSubstrings(arr)
        SET uzyvka TO new dictionary defaulting to zero
        FOR macsuq IN arr DO
            SET awuicr TO length of macsuq
            SET yxril TO 0
            WHILE yxril <= awuicr - 1 DO
                SET rnazk TO yxril + 1
                REPEAT
                    IF rnazk > awuicr THEN
                        BREAK
                    END IF
                    SET pdtwov TO substring of macsuq from index yxril TO rnazk - 1
                    uzyvka[pdtwov] = uzyvka[pdtwov] + 1
                    INCREMENT rnazk BY 1
                UNTIL FALSE
                INCREMENT yxril BY 1
            END WHILE
        END FOR

        SET pmgto TO empty list
        FUNCTION zraqod(tiejplq)
            FUNCTION isEmpty(strz)
                RETURN strz = ""
            END FUNCTION
            SET tnqlz TO length of tiejplq
            SET nykmew TO ""
            PROCEDURE iterate(i)
                IF i > tnqlz - 1 THEN RETURN END IF
                PROCEDURE iterateInner(j)
                    IF j > tnqlz THEN RETURN END IF
                    SET sctagy TO substring of tiejplq from i TO j - 1
                    IF uzyvka[sctagy] = 1 THEN
                        IF isEmpty(nykmew) OR
                           length of sctagy < length of nykmew OR
                           (length of sctagy = length of nykmew AND sctagy < nykmew)
                        THEN
                            nykmew = sctagy
                        END IF
                    END IF
                    iterateInner(j + 1)
                END PROCEDURE
                iterateInner(i + 1)
                iterate(i + 1)
            END PROCEDURE
            iterate(0)
            RETURN nykmew
        END FUNCTION

        FOR eqpvn IN arr DO
            SET resultSubstr TO zraqod(eqpvn)
            APPEND resultSubstr TO pmgto
        END FOR
        RETURN pmgto
    END FUNCTION
END CLASS
