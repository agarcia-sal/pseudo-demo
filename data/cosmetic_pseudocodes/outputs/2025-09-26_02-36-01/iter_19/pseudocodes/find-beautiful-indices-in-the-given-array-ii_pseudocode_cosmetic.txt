CLASS Solution
    FUNCTION beautifulIndices(s parameter a parameter b parameter k)
      SET omega TO empty list
      SET gamma TO LENGTH OF s MINUS LENGTH OF a
      SET delta TO 0
      WHILE delta <= gamma
          SET epsilon TO ""
          SET zeta TO 0
          WHILE zeta < LENGTH OF a
              SET epsilon TO epsilon + character at position delta + zeta of s
              SET zeta TO zeta + 1
          END WHILE
          IF epsilon EQUALS a
              APPEND delta TO omega
          END IF
          SET delta TO delta + 1
      END WHILE

      SET eta TO empty list
      SET theta TO LENGTH OF s MINUS LENGTH OF b
      SET iota TO 0
      REPEAT
          IF iota > theta
              BREAK
          END IF
          SET kappa TO ""
          SET lambda TO 0
          WHILE lambda < LENGTH OF b
              SET kappa TO kappa + character at position iota + lambda of s
              SET lambda TO lambda + 1
          END WHILE
          IF kappa EQUALS b THEN
              APPEND iota TO eta
          END IF
          SET iota TO iota + 1
      UNTIL FALSE

      SET mu TO empty list
      SET alpha TO 0
      SET beta TO 0
      REPEAT
          IF NOT (alpha < LENGTH OF omega AND beta < LENGTH OF eta)
              BREAK
          END IF
          SET tau TO omega at position alpha MINUS eta at position beta
          IF tau < 0
              SET tau TO 0 - tau
          END IF
          IF tau <= k
              APPEND omega at position alpha TO mu
              SET alpha TO alpha + 1
          ELSE
              IF omega at position alpha < eta at position beta
                  SET alpha TO alpha + 1
              ELSE
                  SET beta TO beta + 1
              END IF
          END IF
      UNTIL FALSE

      RETURN mu
    END FUNCTION
END CLASS
