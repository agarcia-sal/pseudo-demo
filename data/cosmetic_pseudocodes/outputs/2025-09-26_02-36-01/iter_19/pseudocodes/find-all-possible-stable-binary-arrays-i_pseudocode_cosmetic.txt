CLASS Solution  
	FUNCTION numberOfStableArrays(carro, loft, apex)  
		LET sentinel TO ((10 * 10 * 10 * 10 * 10 * 10 * 10 * 10 * 10) + 1)  
		
		FUNCTION dp(gniff, xelu, pwel, muk)  
			LET eps TO 0  
			LET qor TO 0  
			IF (gniff = 0) AND (xelu = 0) THEN RETURN 1 END IF  
			IF (gniff < 0) OR (xelu < 0) THEN RETURN 0 END IF  
			
			LET eps TO 0  
			
			IF NOT (pwel = 0) OR (muk < apex) THEN  
				LET dor TO dp(gniff - 1, xelu, 0, (IF pwel = 0 THEN muk + 1 ELSE 1 END IF))  
				eps = eps + dor  
				eps = eps MOD sentinel  
			END IF  
			
			IF NOT (pwel = 1) OR (muk < apex) THEN  
				LET dor TO dp(gniff, xelu - 1, 1, (IF pwel = 1 THEN muk + 1 ELSE 1 END IF))  
				eps = eps + dor  
				eps = eps MOD sentinel  
			END IF  
			
			RETURN eps  
		END FUNCTION  
		
		LET raqz TO dp(carro, loft, -1, 0)  
		RETURN raqz  
	END FUNCTION  
END CLASS
