CLASS Solution  
    FUNCTION queryResults(limit parameter, queries parameter)  
        SET woxpefla TO an empty dictionary mapping identifiers to colors  
        SET ruqzfa TO an empty set containing distinct colors  
        SET qijzopi TO an empty list for storing counts of unique colors  
        
        SET depxju TO 0  
        WHILE depxju < LENGTH(queries parameter) DO  
            SET wjveht TO queries parameter[depxju][0]  
            SET yitarz TO queries parameter[depxju][1]  
            
            IF HAS_KEY(woxpefla, wjveht) THEN  
                SET bkulqfu TO woxpefla[wjveht]  
                IF CONTAINS(ruqzfa, bkulqfu) THEN  
                    CALL REMOVE_ELEMENT(ruqzfa, bkulqfu)  
                END IF  
            END IF  
            
            SET woxpefla[wjveht] TO yitarz  
            ADD_ELEMENT(ruqzfa, yitarz)  
            APPEND_TO(qijzopi, SIZE(ruqzfa))  
            
            SET depxju TO depxju + (1 * 1)  
        END WHILE  
        
        RETURN qijzopi  
    END FUNCTION  
END CLASS
