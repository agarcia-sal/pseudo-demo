CLASS Solution  
    FUNCTION maxNumber(n)  
        DECLARE flag = false  
        DECLARE curr = 1  
        DECLARE resultVar = 0  
        DECLARE divider = 2  
        DECLARE tempVar = 0  

        WHILE TRUE DO  
            IF (n = 1) THEN  
                flag = true  
                BREAK  
            END IF  
            BREAK  
        END WHILE  

        IF flag THEN  
            RETURN resultVar  
        END IF  

        PROCEDURE multiplyByTwo(a)  
            RETURN a + a  
        END PROCEDURE  

        PROCEDURE divideByTwo(a)  
            RETURN a / 2  
        END PROCEDURE  

        WHILE curr <= n DO  
            curr = multiplyByTwo(curr)  
        END WHILE  

        tempVar = divideByTwo(curr)  
        resultVar = tempVar + (-1)  

        RETURN resultVar  
    END FUNCTION  
END CLASS
