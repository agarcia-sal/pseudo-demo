CLASS Solution  
  FUNCTION minimumArea(grid)  
    SET p TO FALSE  
    SET u TO FALSE  
    SET l TO LENGTH(grid)  
    SET m TO LENGTH(grid[0])  
    SET x TO +∞  
    SET y TO -∞  
    SET z TO +∞  
    SET w TO -∞  
    SET s TO 0  
    SET t TO 0  
    SET r TO 0  
    SET q TO 0  
    WHILE s < l  
      SET t TO 0  
      WHILE t < m  
        IF grid[s][t] = 1 THEN  
          IF x > s THEN  
            SET x TO s  
          END IF  
          IF y < s THEN  
            SET y TO s  
          END IF  
          IF z > t THEN  
            SET z TO t  
          END IF  
          IF w < t THEN  
            SET w TO t  
          END IF  
        END IF  
        SET t TO t + 1  
      END WHILE  
      SET s TO s + 1  
    END WHILE  
    IF l = 0 OR m = 0 THEN  
      SET r TO 0  
    ELSE  
      SET r TO (y - x + 1) * (w - z + 1)  
    END IF  
    RETURN r  
  END FUNCTION  
END CLASS
