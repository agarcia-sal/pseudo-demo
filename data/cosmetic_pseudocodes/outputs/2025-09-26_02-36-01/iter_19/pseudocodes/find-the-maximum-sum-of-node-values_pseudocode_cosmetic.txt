CLASS Solution
    FUNCTION maximumValueSum(nums k edges)
        SET alpha TO 0

        SET phi TO 1

        CALL processLoop(0)

        IF (delta % 2) != 0 THEN
            SET alpha TO alpha - omega
        END IF

        RETURN alpha

        FUNCTION processLoop(beta)
            IF beta == LENGTH(nums) THEN
                RETURN
            END IF

            SET rho TO (nums[beta] XOR k)
            SET sigma TO (rho - nums[beta])
            IF sigma > 0 THEN
                SET delta TO delta + phi
            END IF
            SET alpha TO alpha + MAX(nums[beta], rho)
            SET omega TO MIN(omega, ABS(sigma))

            CALL processLoop(beta + 1)
        END FUNCTION
    END FUNCTION
END CLASS
