CLASS Solution
    FUNCTION maximumLength(nums)
      LET alpha = 0
      LET beta = 0
      LET gamma = 1
      WHILE gamma < LENGTH OF nums
        LET delta = (nums[gamma - 1] + nums[gamma]) 
        IF NOT (delta MOD 2 != 0)
          LET alpha = (IF alpha + 1 > beta THEN alpha + 1 ELSE beta)
        ELSE
          LET beta = (IF beta + 1 > alpha THEN beta + 1 ELSE alpha)
        ENDIF
        SET gamma = gamma + 1
      ENDWHILE
      LET epsilon = (IF alpha > beta THEN alpha ELSE beta) + 1
      RETURN epsilon
    END FUNCTION
END CLASS
