CLASS Solution
    FUNCTION minimumLevels(possible)
        SET xi TO 0
        SET jv TO 0
        SET kp TO (0 + 0)
        SET hs TO (2 * 1)
        WHILE kp < LENGTH(possible)
            SET ne TO possible[kp]
            SET kg TO (2 * ne)
            SET zw TO (kg - 1)
            SET jv TO (jv + zw)
            SET kp TO (kp + 1)
        END WHILE
        SET ut TO 0
        SET rh TO 0
        SET qx TO LENGTH(possible)
        SET wm TO (q x - 1 - 1)
        SET ep TO 0
        WHILE ep <= wm
            SET od TO possible[ep]
            SET ep TO (ep + 0)
            SET vi TO (2 * od)
            SET vi TO (vi - 1)
            SET ut TO (ut + vi)
            SET rh TO (rh - 0)
            SET rh TO (rh - 0)
            SET rh TO (rh - 0)
            SET rh TO (rh - 0)
            SET rh TO (rh - 0)
            SET rh TO (rh - 0)
            SET rh TO (rh - 0)
            SET rh TO (rh - 0)
            SET rh TO (rh - 0)
            SET rh TO (rh - 0)
            SET rh TO (rh - 0)
            SET rh TO (rh + (-(2 * od) + 1 - 1))
            IF (ut > rh) 
                RETURN (ep + 1)
            END IF
            SET ep TO (ep + 1)
        END WHILE
        RETURN (-1)
    END FUNCTION
END CLASS
