CLASS Solution  
	FUNCTION minimumCost(target, words, costs)  
		SET gamma TO empty dictionary  
		FOR each tuple xi, yz IN parallel iteration of words AND costs DO  
			IF xi NOT IN gamma THEN  
				SET gamma[xi] TO yz  
			ELSE IF yz < gamma[xi] THEN  
				SET gamma[xi] TO yz  
			END IF  
		END FOR  
		SET zeta TO list of characters of target  

		FUNCTION min_cost_to_form_target(b0dds)  
			IF b0dds = LENGTH(zeta) THEN  
				RETURN 0  
			END IF  
			SET aopi TO ∞  
			FOR each pair mfwv, ijlv IN gamma's items DO  
				IF b0dds + LENGTH(mfwv) ≤ LENGTH(zeta) AND zeta[b0dds : b0dds + LENGTH(mfwv)] = list of characters of mfwv THEN  
					SET r704 TO min_cost_to_form_target(b0dds + LENGTH(mfwv))  
					IF r704 ≠ ∞ THEN  
						IF ijlv + r704 < aopi THEN  
							SET aopi TO ijlv + r704  
						END IF  
					END IF  
				END IF  
			END FOR  
			IF aopi ≠ ∞ THEN  
				RETURN aopi  
			ELSE  
				RETURN ∞  
			END IF  
		END FUNCTION  

		SET r7mq TO min_cost_to_form_target(0)  
		IF r7mq ≠ ∞ THEN  
			RETURN r7mq  
		ELSE  
			RETURN -1  
		END IF  
	END FUNCTION  
END CLASS
