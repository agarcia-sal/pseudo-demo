```pseudocode
CLASS Solution
    FUNCTION minimumArrayLength(nums)
        SET beta TO Infinity
        SET gamma TO 0
        FOR sigma FROM 0 TO LENGTH(nums) - 1
            IF nums[sigma] < beta
                SET beta TO nums[sigma]
                SET gamma TO 1
            ELSE IF nums[sigma] = beta
                SET gamma TO gamma + 1
            END IF
        END FOR
        SET delta TO 0
        IF NOT (gamma <> 1)
            SET delta TO 1
        ELSE
            SET delta TO ( (gamma + 1) / 2 )
        END IF
        RETURN delta
    END FUNCTION
END CLASS
```
