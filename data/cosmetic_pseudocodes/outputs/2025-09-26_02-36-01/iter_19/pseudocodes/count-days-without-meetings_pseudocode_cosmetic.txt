CLASS Solution
    FUNCTION countDays(omega, psi)
        PROCEDURE customSort(zeta)
            FOR alpha FROM 1 TO LENGTH(zeta) - 1
                FOR beta FROM 1 TO LENGTH(zeta) - alpha
                    IF zeta[beta][1] > zeta[beta + 1][1] THEN
                        SET temp TO zeta[beta]
                        SET zeta[beta] TO zeta[beta + 1]
                        SET zeta[beta + 1] TO temp
                    END IF
                END FOR
            END FOR
        END PROCEDURE

        CALL customSort(psi)
        SET lambda TO 1
        SET sigma TO 0
        PROCEDURE processMeetings(index, tau, upsilon)
            IF index > LENGTH(psi) THEN
                RETURN upsilon
            ELSE
                SET pi TO psi[index][1]
                SET rho TO psi[index][2]
                IF lambda < pi THEN
                    SET sigma TO upsilon + (pi - lambda)
                ELSE
                    SET sigma TO upsilon
                END IF
                SET lambda TO MAX(lambda, rho) + 1
                RETURN processMeetings(index + 1, lambda, sigma)
            END IF
        END PROCEDURE

        SET sigma TO processMeetings(1, lambda, sigma)
        IF lambda <= omega THEN
            SET sigma TO sigma + (omega - lambda + 1)
        END IF
        RETURN sigma
    END FUNCTION
END CLASS
