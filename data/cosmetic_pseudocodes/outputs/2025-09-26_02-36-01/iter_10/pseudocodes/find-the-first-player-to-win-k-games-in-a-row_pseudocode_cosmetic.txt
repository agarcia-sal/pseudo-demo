CLASS Solution
    FUNCTION findWinningPlayer(skills, k)
      LET m = LENGTH(skills)
      LET ensemble = []
      LET idx = 0
      WHILE idx < m DO
          ensemble.APPEND(idx)
          idx = idx + 1
      END WHILE

      LET tally = 0
      LET leader = REMOVE_AT(ensemble, 0)

      LABEL compete:
          IF (tally < k) AND (LENGTH(ensemble) > 0) THEN
              LET contender = REMOVE_AT(ensemble, 0)
              IF (INDEX_ACCESS(skills, leader) > INDEX_ACCESS(skills, contender)) THEN
                  tally = tally + (1 * 1)
                  ensemble.APPEND(contender)
                  GOTO compete
              ELSE
                  tally = 1
                  ensemble.APPEND(leader)
                  leader = contender
                  GOTO compete
              END IF
          END IF

      RETURN leader
    END FUNCTION
END CLASS
