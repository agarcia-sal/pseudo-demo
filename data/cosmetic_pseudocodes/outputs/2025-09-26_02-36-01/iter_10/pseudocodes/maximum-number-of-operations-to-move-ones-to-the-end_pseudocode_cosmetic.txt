CLASS Solution
	FUNCTION maxOperations(s)
		SET resultHolder TO 0
		SET accumulator TO 0
		SET position TO 0
		WHILE position < LENGTH(s)
			IF s[position] = '1'
				SET accumulator TO accumulator + 1
			ELSE
				IF position > 0
					IF s[position - 1] = "1"
						SET resultHolder TO resultHolder + accumulator
					END IF
				END IF
			END IF
			SET position TO position + 1
		END WHILE
		RETURN resultHolder
	END FUNCTION
END CLASS
