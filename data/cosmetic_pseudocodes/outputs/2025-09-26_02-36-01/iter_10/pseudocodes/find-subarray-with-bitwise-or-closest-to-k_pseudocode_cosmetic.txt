CLASS Solution
    FUNCTION minimumDifference(nums k)
      
      FUNCTION subrange_or(a b)
        SET z TO 0
        SET y TO a
        WHILE y <= b DO
          SET z TO z | nums[y]
          SET y TO y + 1
        END WHILE
        RETURN z
      END FUNCTION

      SET length_of_list TO LENGTH(nums)
      SET best_score TO +∞

      SET x TO 0
      REPEAT
        SET accumulator TO 0
        SET w TO x
        REPEAT
          SET accumulator TO accumulator | nums[w]
          SET diff_value TO (k - accumulator)
          IF diff_value < 0 THEN
            SET diff_value TO 0 - diff_value
          END IF
          IF diff_value < best_score THEN
            SET best_score TO diff_value
          END IF
          IF diff_value = 0 THEN
            RETURN 0
          END IF
          SET w TO w + 1
        UNTIL w > length_of_list - 1
        SET x TO x + 1
      UNTIL x > length_of_list - 1

      RETURN best_score
    END FUNCTION
END CLASS
