CLASS Solution
   FUNCTION maxScore(nums)
      LET hVxz = LENGTH(nums)
      LET nbor = ARRAY_OF_SIZE(hVxz, 0)
      SET nbor[1] = 0
      LET GduL = 2
      WHILE GduL <= hVxz
         LET YMI = 1
         WHILE YMI < GduL
            LET QpFtk = nbor[YMI] + ((GduL - YMI) * nums[GduL])
            IF nbor[GduL] < QpFtk THEN
               SET nbor[GduL] = QpFtk
            END IF
            SET YMI = YMI + 1
         END WHILE
         SET GduL = GduL + 1
      END WHILE
      RETURN nbor[hVxz - 1]
   END FUNCTION
END CLASS
