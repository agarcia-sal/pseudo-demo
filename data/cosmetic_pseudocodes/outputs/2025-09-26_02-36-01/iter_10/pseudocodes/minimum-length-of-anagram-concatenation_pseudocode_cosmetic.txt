CLASS Solution
	 FUNCTION minAnagramLength(s)
	 	DEFINE FUNCTION constructSet(inputString)
	 		DEFINE collection AS an empty associative container
	 		DEFINE indexCounter AS 0
	 		WHILE indexCounter < LENGTH(inputString)
	 			IF inputString[indexCounter] NOT IN collection
	 				SET collection[inputString[indexCounter]] TO TRUE
	 			END IF
	 			SET indexCounter TO indexCounter + 1
	 		END WHILE
	 		RETURN collection
	 	END FUNCTION
	 	
	 	DEFINE holder SET TO constructSet(s)
	 	
	 	DEFINE counter AS 0
	 	FOR EACH key IN holder DO
	 		SET counter TO counter + 1
	 	END FOR
	 	
	 	RETURN counter
	 END FUNCTION
END CLASS
