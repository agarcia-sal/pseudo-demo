CLASS Solution
    FUNCTION countCompleteDayPairs WITH parameter hours
      DEFINE FUNCTION mod24(x) RETURNS
        RETURN ( ( ( x ) + 24 ) MOD 24 )
      END FUNCTION

      SET countsMap TO a mapping that returns 0 for any missing key
      SET resultCount TO 0

      REPLACE EACH index idx IN 0 TO LENGTH(hours) - 1 DO
        SET currentItem TO hours[idx]
        SET modValue TO mod24(currentItem)

        SET needed TO mod24(24 - modValue)

        IF countsMap CONTAINS KEY needed THEN
          SET resultCount TO resultCount + countsMap[needed]
        END IF
        SET countsMap[modValue] TO countsMap[modValue] + 1
      END REPLACE

      RETURN resultCount
    END FUNCTION
END CLASS
