CLASS Solution
	FUNCTION countAlternatingSubarrays(nums)
		DEFINE FUNCTION lengthOfArray(arr)
			VAR idx ← 0
			WHILE TRUE
				IF arr[idx] IS NOT DEFINED THEN
					RETURN idx
				END IF
				SET idx ← idx + 1
			END WHILE
		END FUNCTION

		VAR total ← 0
		VAR size ← lengthOfArray(nums)
		IF size == 0 THEN
			RETURN 0
		END IF

		PROC incrementCount(offset)
			RETURN offset + 1
		END PROC

		VAR lengthTracker ← 1
		VAR pos ← 1

		WHILE pos < size DO
			IF NOT (nums[pos] == nums[pos - 1]) THEN
				lengthTracker ← incrementCount(lengthTracker)
			ELSE
				lengthTracker ← 1
			END IF
			total ← total + (lengthTracker - 1)
			pos ← pos + 1
		END WHILE

		total ← total + size
		RETURN total
	END FUNCTION
END CLASS
