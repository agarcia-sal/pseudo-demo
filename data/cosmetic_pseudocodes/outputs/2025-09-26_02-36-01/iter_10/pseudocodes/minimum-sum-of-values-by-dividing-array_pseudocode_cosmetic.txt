CLASS Solution
	Function minimumValueSum(nums, andValues)
		LET a ← LEN(nums)
		LET b ← LEN(andValues)
		
		Function dp(x, y)
			Function helper(p, q)
				IF q = -1 THEN
					IF p = -1 THEN RETURN 0 ELSE RETURN +∞ END IF
				END IF
				IF p = -1 THEN RETURN +∞ END IF
				
				Variable z ← +∞
				Variable r ← -1
				
				Sub helperLoop(u)
					IF u < 0 THEN RETURN END IF
					IF r = -1 THEN LET r ← nums[u]
					ELSE LET r ← r AND nums[u] END IF
					
					IF r = andValues[q] THEN
						LET w ← dp(u - 1, q - 1) + nums[p]
						IF w < z THEN LET z ← w END IF
					END IF
					helperLoop(u - 1)
				End Sub
				
				helperLoop(x)
				RETURN z
			End Function
			
			RETURN helper(x, y)
		End Function
		
		LET output ← dp(a - 1, b - 1)
		IF output ≠ +∞ THEN RETURN output ELSE RETURN -1 END IF
	End Function
END CLASS
