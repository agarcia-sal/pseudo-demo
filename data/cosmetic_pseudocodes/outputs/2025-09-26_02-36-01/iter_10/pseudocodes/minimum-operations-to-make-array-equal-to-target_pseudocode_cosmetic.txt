CLASS Solution
    FUNCTION minimumOperations(nums, target)
        SET q TO LENGTH(nums)
        FUNCTION absoluteValue(z)
            IF z < 0 THEN
                RETURN -z
            ELSE
                RETURN z
            END IF
        END FUNCTION

        SET g TO absoluteValue(target[0] - nums[0])

        FUNCTION loopHelper(r, s)
            IF r = q THEN
                RETURN s
            ELSE
                SET a TO target[r] - nums[r]
                SET b TO target[r - 1] - nums[r - 1]

                IF (a * b) > 0 THEN
                    SET c TO absoluteValue(a) - absoluteValue(b)
                    IF c > 0 THEN
                        RETURN loopHelper(r + 1, s + c)
                    ELSE
                        RETURN loopHelper(r + 1, s)
                    END IF
                ELSE
                    RETURN loopHelper(r + 1, s + absoluteValue(a))
                END IF
            END IF
        END FUNCTION

        RETURN loopHelper(1, g)
    END FUNCTION
END CLASS
