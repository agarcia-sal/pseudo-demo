CLASS Solution
	 FUNCTION numberOfStableArrays(zero zero, one one, limit limit) RETURNS integer
		 SET CONSTANT_MODULUS TO (10 ^ 9) + 1
		 
		 FUNCTION internalCompute(remainingZero remZero, remainingOne remOne, previous prev, streak streakCount) RETURNS integer
				IF (remZero = 0) AND (remOne = 0) THEN
					 RETURN 1
				END IF
				
				IF (remZero < 0) OR (remOne < 0) THEN
					 RETURN 0
				END IF
				
				SET accumulator TO 0
				
				IF prev = 0 THEN
					 IF streakCount < limit THEN
						  SET accumulator TO accumulator + internalCompute(remZero - 1, remOne, 0, streakCount + 1)
					 END IF
					 SET accumulator TO accumulator + internalCompute(remZero, remOne - 1, 1, 1)
				ELSE
					 IF remZero > 0 THEN
						  SET accumulator TO accumulator + internalCompute(remZero - 1, remOne, 0, 1)
					 END IF
					 IF streakCount < limit THEN
						  SET accumulator TO accumulator + internalCompute(remZero, remOne - 1, 1, streakCount + 1)
					 END IF
				END IF
				
				RETURN accumulator MOD CONSTANT_MODULUS
		 END FUNCTION
		 
		 RETURN internalCompute(zero, one - 1, 0, 1)
	 END FUNCTION
END CLASS
