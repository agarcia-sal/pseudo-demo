CLASS Solution
    FUNCTION waysToReachStair(k)
        FUNCTION recur(x, y, z)
            IF (x > k + 1) THEN
                RETURN 0
            END IF

            SET tempResult TO (x = k) * 1

            IF (x > 0 AND y = 0) THEN
                SET tempResult TO tempResult + recur(x - 1, 1, z)
            END IF

            SET tempResult TO tempResult + recur(((x + 1) * 2 * z), 0, z + 1)

            RETURN tempResult
        END FUNCTION

        RETURN recur(1, 0, 0)
    END FUNCTION
END CLASS
