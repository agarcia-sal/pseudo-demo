CLASS Solution
    FUNCTION countSubstrings(s c)
    BEGIN
        DEFINE uA as 0
        DEFINE Ls as LENGTH OF s
        DEFINE Xz as 0

        PROCEDURE countCharOccurrences()
        BEGIN
            SET yq TO 0
            WHILE yq < Ls DO
                IF s[yq] = c THEN
                    SET Xz TO Xz + 1
                END IF
                SET yq TO yq + 1
            END WHILE
        END

        PROCEDURE triangularSum(z)
        BEGIN
            RETURN (z * (z + 1)) / 2
        END

        CALL countCharOccurrences()

        SET uA TO triangularSum(Xz)

        SET RETURN_VALUE TO Xz * uA

        RETURN RETURN_VALUE
    END
END CLASS
