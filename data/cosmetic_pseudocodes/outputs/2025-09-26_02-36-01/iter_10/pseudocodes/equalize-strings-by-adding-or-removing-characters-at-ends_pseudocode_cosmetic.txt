CLASS Solution
    FUNCTION minOperations(initial parameter TO string, target parameter TO string) RETURNS integer
    BEGIN
        LET alpha BE LENGTH(initial)
        LET beta BE LENGTH(target)
        LET table BE LIST OF LISTS WITH DIMENSIONS (alpha + 1) Ã— (beta + 1) INITIALIZED TO ZERO
        LET peak_lcs BE 0

        DEFINE PROCEDURE processIndices(x INTEGER, y INTEGER)
        BEGIN
            IF x > alpha THEN RETURN END IF
            IF y > beta THEN
                CALL processIndices(x + 1, 1)
                RETURN
            END IF

            IF initial[x - 1] = target[y - 1] THEN
                table[x][y] := table[x - 1][y - 1] + 1
                IF peak_lcs < table[x][y] THEN
                    peak_lcs := table[x][y]
                END IF
            END IF

            CALL processIndices(x, y + 1)
        END

        CALL processIndices(1, 1)

        RETURN alpha + beta - 2 * peak_lcs
    END
END CLASS
