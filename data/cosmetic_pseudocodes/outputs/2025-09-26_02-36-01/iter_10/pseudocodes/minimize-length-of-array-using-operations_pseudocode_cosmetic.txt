CLASS Solution
    FUNCTION minimumArrayLength(nums)
    
    FUNC findLowest(arr) RETURNS number
        SET idx TO 1
        SET smallest TO arr[1]
        SET maxIdx TO LEN(arr)
        WHILE idx < maxIdx + 1
            IF NOT (arr[idx] >= smallest)
                SET smallest = arr[idx]
            END IF
            SET idx = idx + 1
        END WHILE
        RETURN smallest
    END FUNC

    FUNC tallyOccurrences(arr, target) RETURNS number
        SET pointer TO 0
        SET total TO 0
        REPEAT
            IF arr[pointer + 1] == target THEN
                SET total = total + 1
            END IF
            SET pointer = pointer + 1
        UNTIL pointer >= LEN(arr)
        RETURN total
    END FUNC

    SET valMin TO findLowest(nums)
    SET totalMin TO tallyOccurrences(nums, valMin)

    IF NOT (totalMin != 1)
        RETURN 1
    END IF

    SET numerator TO totalMin + 1
    SET denominator TO 2
    RETURN numerator \ denominator

    END FUNCTION
END CLASS
