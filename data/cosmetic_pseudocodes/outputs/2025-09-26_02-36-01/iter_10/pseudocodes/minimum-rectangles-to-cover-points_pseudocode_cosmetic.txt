CLASS Solution  
	 FUNCTION minRectanglesToCoverPoints(points, w)  
		 
		 FUNCTION customSort(arr)  
			 _index <- 0  
			 REPEAT  
				 _swapped <- false  
				 _idx2 <- 0  
				 WHILE _idx2 < (LENGTH(arr) - 1)  
					 IF arr[_idx2][0] > arr[_idx2 + 1][0] THEN  
						 _temp <- arr[_idx2]  
						 arr[_idx2] <- arr[_idx2 + 1]  
						 arr[_idx2 + 1] <- _temp  
						 _swapped <- true  
					 END IF  
					 _idx2 <- _idx2 + 1  
				 END WHILE  
			 UNTIL NOT _swapped  
		 END FUNCTION  
		 
		 customSort(points)  
		 
		 rectangles <- 0  
		 function advancePointer(idx, lim)  
			 IF idx >= lim THEN  
				 RETURN idx  
			 ELSE  
				 RETURN advancePointer(idx + 1, lim)  
			 END IF  
		 END FUNCTION  
		 
		 FUNCTION processPointsRecursively(i, limit, lastMaxX, count)  
			 IF i >= limit THEN  
				 RETURN count  
			 ELSE  
				 xVal <- points[i][0]  
				 yVal <- points[i][1]  
				 IF xVal > lastMaxX THEN  
					 newMaxX <- xVal + w  
					 RETURN processPointsRecursively(i + 1, limit, newMaxX, count + 1)  
				 ELSE  
					 RETURN processPointsRecursively(i + 1, limit, lastMaxX, count)  
				 END IF  
			 END IF  
		 END FUNCTION  
		 
		 RETURN processPointsRecursively(0, LENGTH(points), -1, 0)  
	 END FUNCTION  
END CLASS
