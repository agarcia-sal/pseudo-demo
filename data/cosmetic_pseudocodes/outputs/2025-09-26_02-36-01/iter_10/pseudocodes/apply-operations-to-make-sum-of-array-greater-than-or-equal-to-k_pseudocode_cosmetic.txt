CLASS Solution
  FUNCTION minOperations(k)
      SET ζ TO +∞
      SET υ TO 1
  _LOOP:
      IF NOT (υ ≤ FLOOR(SQRT(k)) + 1) THEN GOTO _ENDLOOP
      SET ρ TO INTEGER_DIVIDE(k + υ - 1, υ)
      SET σ TO (υ - 1) + (ρ - 1)
      IF σ < ζ THEN
         SET ζ TO σ
      END IF
      SET υ TO υ + 1
      GOTO _LOOP
  _ENDLOOP:
      RETURN ζ
  END FUNCTION
END CLASS
