CLASS Solution
	Function maximumLength(nums)
		Define hXjI = 0
		Define nLaE = 0
		Define rGzA = 1
		While (rGzA < LENGTH(nums))
			Define rUMkX = (nums[rGzA - 1] + nums[rGzA])
			If (rUMkX % 2 = 0)
				hXjI = MAX(hXjI + 1, nLaE)
			Else
				nLaE = MAX(nLaE + 1, hXjI)
			End If
			rGzA = rGzA + 1
		End While
		Define oIvB = MAX(hXjI, nLaE) + 1
		Return oIvB
	End Function
END CLASS
