CLASS Solution
    FUNCTION minOperations(nums)
      SET nPhTmXhZX TO 0
      SET QLWMflKdk TO 0
      SET uSrTawKhZ TO 0
      LABEL startLoop
      IF uSrTawKhZ >= LENGTH(nums) THEN
        RETURN nPhTmXhZX
      ELSE
        SET currentValue TO nums[uSrTawKhZ]
        IF ((QLWMflKdk MOD 2) = 0) THEN
          SET transformedValue TO currentValue
        ELSE
          SET transformedValue TO 1 - currentValue
        END IF
        IF transformedValue = 0 THEN
          SET nPhTmXhZX TO nPhTmXhZX + 1
          SET QLWMflKdk TO QLWMflKdk + 1
        END IF
        SET uSrTawKhZ TO uSrTawKhZ + 1
        GOTO startLoop
      END IF
    END FUNCTION
END CLASS
