CLASS Solution
    FUNCTION minimumTimeToInitialState(word PARAMETER k)
          DECLARE cA TO 1
          DECLARE nA TO LENGTH(word)
          
          FUNCTION checkSlices(subA, subB)
              RETURN subA = subB
          END FUNCTION
          
          FUNCTION getSubstringS(s, startI, endJ)
              DECLARE resR TO ""
              DECLARE idxK TO startI
              WHILE idxK < endJ
                  SET resR TO resR + s[idxK]
                  SET idxK TO idxK + 1
              END WHILE
              RETURN resR
          END FUNCTION
          
          LABEL loopStart:
              IF (cA * k) >= nA THEN
                  RETURN cA
              END IF
              
              DECLARE partOne TO getSubstringS(word, cA * k, nA)
              DECLARE partTwo TO getSubstringS(word, 0, nA - (cA * k))
              
              IF checkSlices(partOne, partTwo) THEN
                  RETURN cA
              END IF
              
              SET cA TO cA + 1
              GOTO loopStart
    END FUNCTION
END CLASS
