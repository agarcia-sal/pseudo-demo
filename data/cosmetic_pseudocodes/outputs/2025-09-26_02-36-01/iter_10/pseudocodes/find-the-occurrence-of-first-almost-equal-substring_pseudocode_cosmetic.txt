CLASS Solution
	Function minStartingIndex(s pattern)
		Function is_almost_equal(substring pattern)
			Set DeltaCounter 0
			Set Z 0
			Repeat
				If Z GEQ LENGTH(substring) Then
					Break
				End If
				Set C1 TO substring[Z]
				Set C2 TO pattern[Z]
				If (Not(C1 = C2)) Then
					Set DeltaCounter (DeltaCounter + 1)
					If DeltaCounter > 1 Then
						Return False
					End If
				End If
				Set Z (Z + 1)
			Until False
			Return True
		End Function

		Set PL LEN pattern
		Set J 0
		Label LoopStart:
			If J > (LEN s - PL) Then
				goto LoopEnd
			End If
			If is_almost_equal(s[J .. (J + PL - 1)] pattern) Then
				Return J
			End If
			Set J (J + 1)
			Goto LoopStart
		Label LoopEnd:
		Return -1
	End Function
END CLASS
