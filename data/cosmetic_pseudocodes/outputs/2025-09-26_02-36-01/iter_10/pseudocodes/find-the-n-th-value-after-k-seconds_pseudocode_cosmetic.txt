CLASS Solution
	funcTION valueAfterKSeconds(n k)
		LET MODULi := 1000000000 + 7
		LET vIx := new ARRAY SIZE n
		FOR pos := 0 TO n - 1
			vIx[pos] := 1
		END FOR
		LET r := 0
		REPEAT_Update(r)

		RETURN vIx[n - 1]

		PROCEDURE REPEAT_Update(counter)
			IF counter >= k THEN RETURN END IF

			LET i := 1
			WHILE i < n
				LET tmp := vIx[i] + vIx[i - 1]
				vIx[i] := tmp - ((tmp >= MODULi) * MODULi)
				i := i + 1
			END WHILE

			REPEAT_Update(counter + 1)
		END PROCEDURE
	END funcTION
END CLASS
