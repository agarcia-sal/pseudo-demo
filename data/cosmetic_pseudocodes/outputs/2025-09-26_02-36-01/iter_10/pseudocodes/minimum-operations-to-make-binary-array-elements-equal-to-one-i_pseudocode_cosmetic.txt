CLASS Solution
	func minOperations(nums)
		LET totalCount = 0
		LET upperLimit = LENGTH(nums) - 2
		LET index = 0

		WHILE index <= upperLimit
			IF nums[index] == 0 THEN
				call flipTriplet(nums, index)
				totalCount = totalCount + 1
			END IF
			index = index + 1
		END WHILE

		IF nums[LENGTH(nums) - 1] == 0 OR nums[LENGTH(nums) - 2] == 0 THEN
			RETURN -1
		END IF

		RETURN totalCount
	END FUNC

	sub func flipTriplet(arr, pos)
		FUNCTION invert(bit)
			RETURN 1 - bit
		END FUNCTION
		
		arr[pos] = invert(arr[pos])
		arr[pos + 1] = invert(arr[pos + 1])
		arr[pos + 2] = invert(arr[pos + 2])
	END SUB
END CLASS
