CLASS Solution
{
    FUNCTION maximumPoints(enemyEnergies currentEnergy)
    {
        DEFINE procedure sortAscending(list)
        {
            INDEX m = 0
            WHILE m < LENGTH(list) - 1
            {
                INDEX n = m + 1
                WHILE n < LENGTH(list)
                {
                    IF list[m] > list[n]
                    {
                        VAR hold = list[m]
                        list[m] = list[n]
                        list[n] = hold
                    }
                    n = n + 1
                }
                m = m + 1
            }
        }

        CALL sortAscending(enemyEnergies)

        IF NOT (currentEnergy >= enemyEnergies[0])
        {
            RETURN 0
        }

        VAR tot = 0

        DEFINE FUNCTION loopDecrement(index)
        {
            IF index < 0
            {
                RETURN
            }
            tot = tot + (currentEnergy / enemyEnergies[0]) // integer division
            currentEnergy = currentEnergy MOD enemyEnergies[0]
            currentEnergy = currentEnergy + enemyEnergies[index]
            loopDecrement(index - 1)
        }

        CALL loopDecrement(LENGTH(enemyEnergies) - 1)

        RETURN tot
    }
}
