CLASS Solution
    FUNCTION numberOfRightTriangles(grid)
        SET L0 = LENGTH(grid)
        SET L1 = LENGTH(grid[0])
        SET R = 0
        SET A = 0

        PROCEDURE SummateExceptColumn(currRowIndex, excludedCol, OUT sum)
            SET sum = 0
            SET idx = 0
            WHILE idx < L1
                IF idx ≠ excludedCol THEN
                    SET sum = sum + grid[currRowIndex][idx]
                END IF
                SET idx = idx + 1
            END WHILE
        END PROCEDURE

        PROCEDURE SummateExceptRow(excludedRow, currColIndex, OUT sum)
            SET sum = 0
            SET idx = 0
            WHILE idx < L0
                IF idx ≠ excludedRow THEN
                    SET sum = sum + grid[idx][currColIndex]
                END IF
                SET idx = idx + 1
            END WHILE
        END PROCEDURE

        SET OuterIndex = 0
        WHILE OuterIndex < L0
            SET InnerIndex = 0
            WHILE InnerIndex < L1
                IF grid[OuterIndex][InnerIndex] = 1 THEN
                    CALL SummateExceptColumn(OuterIndex, InnerIndex, A)
                    CALL SummateExceptRow(OuterIndex, InnerIndex, R)
                    SET total_triangles = total_triangles + A * R
                END IF
                SET InnerIndex = InnerIndex + 1
            END WHILE
            SET OuterIndex = OuterIndex + 1
        END WHILE

        RETURN total_triangles
    END FUNCTION
END CLASS
