CLASS Solution
	 FUNCTION countSubarrays(nums parameter LIST OF INTEGER k parameter INTEGER) RETURNS INTEGER
 	 	  	
 	  	    VAR totalLength, tally, indexStart, indexEnd, accumulator, limitValue
	
 		  	DEFINE bitwiseAND(x, y)
 				RETURN x & y
 			END DEFINE
	
 			SET totalLength = LENGTH OF nums
 			SET tally = 0
	
 			SET indexStart = 0
 			WHILE indexStart <= totalLength - 1
 				SET accumulator = nums[indexStart]
	
 				DEFINE innerLoop(currIndex, endIndex, acc)
 					IF endIndex > totalLength - 1 THEN
 						RETURN 0
 					END IF
	
 					SET acc = bitwiseAND(acc, nums[endIndex])
 					IF acc = k THEN
 						SET tally = tally + 1
 					END IF
	
 					IF acc < k THEN
 						RETURN 0
 					END IF
	
 					RETURN innerLoop(currIndex, endIndex + 1, acc)
 				END DEFINE
	
 				CALL innerLoop(indexStart, indexStart, accumulator)
 				SET indexStart = indexStart + 1
 			END WHILE
	
 			RETURN tally
	 END FUNCTION
END CLASS
