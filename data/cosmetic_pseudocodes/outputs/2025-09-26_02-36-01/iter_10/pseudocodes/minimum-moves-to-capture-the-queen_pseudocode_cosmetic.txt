CLASS Solution
    FUNCTION minMovesToCaptureTheQueen(a b c d e f)
        FUNCTION eq(x y)
            RETURN x = y
        END FUNCTION
        FUNCTION neq(x y)
            RETURN NOT eq(x y)
        END FUNCTION
        FUNCTION lt(x y)
            RETURN x < y
        END FUNCTION
        FUNCTION gt(x y)
            RETURN y < x
        END FUNCTION
        FUNCTION and_(x y)
            RETURN x AND y
        END FUNCTION
        FUNCTION or_(x y)
            RETURN x OR y
        END FUNCTION

        FUNCTION insideRange(u v w)
            RETURN or_(and_(lt(v u) lt(u w)) and_(lt(w u) lt(u v)))
        END FUNCTION

        FUNCTION checkMiddle(p q r s t u)
            IF eq(p s) THEN
                IF neq(r p) THEN
                    RETURN 1
                END IF
                IF or_(and_(lt(q t) lt(t u)) and_(lt(u t) lt(t q))) THEN
                    RETURN 2
                END IF
                RETURN 1
            END IF
            RETURN 0
        END FUNCTION

        FUNCTION checkSum(x1 y1 x2 y2 x3 y3)
            IF eq(x1 + y1 x3 + y3) THEN
                IF neq(x2 + y2 x1 + y1) THEN
                    RETURN 1
                END IF
                IF or_(and_(lt(x2 x1) lt(x1 x3) and_(lt(y2 y1) lt(y1 y3))) and_(and_(lt(x3 x1) lt(x1 x2) lt(y3 y1) lt(y1 y2)))) THEN
                    RETURN 2
                END IF
                RETURN 1
            END IF
            RETURN 0
        END FUNCTION

        FUNCTION checkDiff(a1 b1 a2 b2 a3 b3)
            IF eq(a1 - b1 a3 - b3) THEN
                IF neq(a2 - b2 a1 - b1) THEN
                    RETURN 1
                END IF
                IF or_(and_(lt(a2 a1) lt(a1 a3) and_(gt(b2 b1) gt(b1 b3))) and_(and_(lt(a3 a1) lt(a1 a2) gt(b3 b1) gt(b1 b2)))) THEN
                    RETURN 2
                END IF
                RETURN 1
            END IF
            RETURN 0
        END FUNCTION

        VARIABLE r1 <- checkMiddle(a b c d e f)
        IF r1 > 0 THEN
            RETURN r1
        END IF

        VARIABLE r2 <- 0
        IF eq(b f) THEN
            IF neq(d b) THEN
                RETURN 1
            END IF
            IF or_(and_(lt(a c) lt(c e)) and_(lt(e c) lt(c a))) THEN
                RETURN 2
            END IF
            RETURN 1
        END IF

        VARIABLE r3 <- checkSum(a b c d e f)
        IF r3 > 0 THEN
            RETURN r3
        END IF

        VARIABLE r4 <- checkDiff(c d e f a b)
        IF r4 > 0 THEN
            RETURN r4
        END IF

        RETURN 2
    END FUNCTION
END CLASS
