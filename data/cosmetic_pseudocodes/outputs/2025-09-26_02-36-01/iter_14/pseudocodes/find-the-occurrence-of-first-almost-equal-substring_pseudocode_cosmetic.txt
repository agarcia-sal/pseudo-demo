CLASS Solution
    FUNCTION minStartingIndex WITH PARAMETERS s pattern
        FUNCTION is_almost_equal WITH PARAMETERS substring pattern
            SET epsilon TO 0
            SET k TO 0
            WHILE k LESS THAN LENGTH(substring) AND epsilon LESS THAN 2 DO
                IF NOT (substring[k] EQUALS pattern[k]) THEN
                    SET epsilon TO epsilon + 1
                END IF
                SET k TO k + 1
            END WHILE
            IF epsilon GREATER THAN 1 THEN
                RETURN False
            ELSE
                RETURN True
            END IF
        END FUNCTION

        SET threshold TO LENGTH(pattern)
        SET counter TO 0
        WHILE counter LESS THAN (LENGTH(s) - threshold + 1) DO
            IF is_almost_equal WITH ARGUMENTS SLICE(s, counter, counter + threshold - 1) pattern THEN
                RETURN counter
            END IF
            SET counter TO counter + 1
        END WHILE
        RETURN -1
    END FUNCTION
END CLASS
