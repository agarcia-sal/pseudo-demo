CLASS Solution  
    FUNCTION minimumOperationsToMakeKPeriodic(word parameter k)  
        SET o TO 0  
        SET p TO LENGTH(word)  
        SET q TO EMPTY LIST  
        WHILE o < p  
            SET r TO SUBSTRING(word, o, o + k - 1)  
            ADD r TO q  
            SET o TO o + k  
        END WHILE  
        SET s TO EMPTY DICTIONARY  
        FOR EACH t IN q  
            IF NOT t IN s THEN  
                SET s[t] TO 0  
            END IF  
            SET s[t] TO s[t] + 1  
        END FOR  
        SET u TO 0  
        FOR EACH v IN s  
            IF s[v] > u THEN  
                SET u TO s[v]  
            END IF  
        END FOR  
        SET w TO LENGTH(q) - u  
        RETURN w  
    END FUNCTION  
END CLASS
