CLASS Solution
    FUNCTION maxScore(points, m)
        
        FUNCTION isPossible(valMin, m)
            SET totalMoves TO 0
            SET priorMoves TO 0
            SET pos TO 0
            WHILE pos < LENGTH(points)
                SET currPoint TO points[pos]
                SET calcReq TO CEIL((valMin + currPoint - 1) / currPoint)
                IF (calcReq - priorMoves) < 0 THEN
                    SET calcReq TO 0
                ELSE
                    SET calcReq TO calcReq - priorMoves
                END IF
                
                IF calcReq > 0 THEN
                    SET totalMoves TO totalMoves + (calcReq * 2 - 1)
                    SET priorMoves TO calcReq - 1
                ELSE IF (pos + 1) < LENGTH(points) THEN
                    SET totalMoves TO totalMoves + 1
                    SET priorMoves TO 0
                END IF

                IF totalMoves > m THEN
                    RETURN False
                END IF
                SET pos TO pos + 1
            END WHILE

            RETURN True
        END FUNCTION

        SET low TO 0
        SET high TO (((m + 1) // 2) * (points[0] + 1))
        
        REPEAT
            SET middle TO (low + high + 1) // 2
            IF isPossible(middle, m) THEN
                SET low TO middle
            ELSE
                SET high TO middle - 1
            END IF
        UNTIL low >= high
        
        RETURN low
    END FUNCTION
END CLASS
