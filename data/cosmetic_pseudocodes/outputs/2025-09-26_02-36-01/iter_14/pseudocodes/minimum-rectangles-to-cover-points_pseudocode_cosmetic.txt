CLASS Solution
    FUNCTION minRectanglesToCoverPoints(points, w)
  
        PROCEDURE arrangeAscendingByFirstElement(list)
            SET index TO 0
            REPEAT
                SET flagSorted TO true
                SET index TO 1
                WHILE index LESS THAN LENGTH(list) DO
                    IF (list[index - 1])[0] GREATER THAN (list[index])[0] THEN
                        SET tempElement TO list[index - 1]
                        SET list[index - 1] TO list[index]
                        SET list[index] TO tempElement
                        SET flagSorted TO false
                    END IF
                    SET index TO index PLUS 1
                END WHILE
            UNTIL flagSorted EQUALS true
        END PROCEDURE

        CALL arrangeAscendingByFirstElement(points)

        SET count_rectangles TO 0
        SET boundaryX TO (0 MULTIPLIED BY 1) MINUS 1

        FUNCTION processPoints(idx, total)
            IF idx GREATER OR EQUAL TO LENGTH(points) THEN
                RETURN total
            ELSE
                SET pointX TO (points[idx])[0]
                SET pointY TO (points[idx])[1]

                IF NOT (pointX LESS OR EQUAL boundaryX) THEN
                    SET boundaryX TO pointX PLUS w
                    SET updated_total TO total PLUS 1
                    RETURN processPoints(idx PLUS 1, updated_total)
                ELSE
                    RETURN processPoints(idx PLUS 1, total)
                END IF
            END IF
        END FUNCTION

        SET result TO processPoints(0, count_rectangles)

        RETURN result

    END FUNCTION
END CLASS
