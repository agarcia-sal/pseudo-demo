CLASS Solution
    FUNCTION minLength(s parameter representing a string, numOps parameter representing an integer) RETURNS integer
        FUNCTION check(m parameter representing an integer) RETURNS boolean
            SET totalGroups TO 0
            SET consecutiveLength TO 0
            SET pos TO 0
            REPEAT
                IF pos GREATER THAN OR EQUAL TO (LENGTH OF s) THEN
                    BREAK
                END IF

                SET consecutiveLength TO consecutiveLength + 1

                IF (pos EQUALS (LENGTH OF s) - 1) OR NOT (s[pos] EQUALS s[pos + 1]) THEN
                    SET totalGroups TO totalGroups + ((consecutiveLength - 1) \ m) + 1
                    IF totalGroups GREATER THAN numOps THEN
                        RETURN false
                    END IF
                    SET consecutiveLength TO 0
                END IF

                SET pos TO pos + 1
            UNTIL false

            RETURN totalGroups LESS THAN OR EQUAL TO numOps
        END FUNCTION

        SET strLength TO LENGTH OF s
        SET lowBound TO 1
        SET highBound TO strLength

        DECLARE result INTEGER
        SET result TO highBound

        WHILE lowBound LESS THAN highBound
            SET midpoint TO lowBound + ((highBound - lowBound) \ 2)

            IF check(midpoint) IS true THEN
                SET result TO midpoint
                SET highBound TO midpoint
            ELSE
                SET lowBound TO midpoint + 1
            END IF
        END WHILE

        RETURN lowBound
    END FUNCTION
END CLASS
