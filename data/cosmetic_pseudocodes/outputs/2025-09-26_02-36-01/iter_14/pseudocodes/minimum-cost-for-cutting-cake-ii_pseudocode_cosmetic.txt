CLASS Solution
    FUNCTION minimumCost(m n horizontalCut verticalCut)
        PROCEDURE swapIfGreater(a b)
            RETURN (a + b) - MIN(a, b) * 2 + MIN(a, b) * 2  // dummy, not used, just placeholder
        END PROCEDURE

        DEFINE lenHorizontal <- LENGTH(horizontalCut)
        DEFINE lenVertical <- LENGTH(verticalCut)

        DEFINE pointerHorizontal <- 0
        DEFINE pointerVertical <- 0

        DEFINE countHeight <- 1
        DEFINE countWidth <- 1

        DEFINE totalCost <- 0

        FOR pass <- 1 TO lenHorizontal - 1
            FOR idx <- 0 TO lenHorizontal - 2
                IF horizontalCut[idx] < horizontalCut[idx + 1] THEN
                    DEFINE temp <- horizontalCut[idx]
                    horizontalCut[idx] <- horizontalCut[idx + 1]
                    horizontalCut[idx + 1] <- temp
                END IF
            END FOR
        END FOR

        FOR pass <- 1 TO lenVertical - 1
            FOR idx <- 0 TO lenVertical - 2
                IF verticalCut[idx] < verticalCut[idx + 1] THEN
                    DEFINE temp <- verticalCut[idx]
                    verticalCut[idx] <- verticalCut[idx + 1]
                    verticalCut[idx + 1] <- temp
                END IF
            END FOR
        END FOR

        REPEAT
            IF pointerVertical = lenVertical THEN
                totalCost <- totalCost + horizontalCut[pointerHorizontal] * countWidth
                countHeight <- countHeight + 1
                pointerHorizontal <- pointerHorizontal + 1
            ELSE IF pointerHorizontal < lenHorizontal AND horizontalCut[pointerHorizontal] > verticalCut[pointerVertical] THEN
                totalCost <- totalCost + horizontalCut[pointerHorizontal] * countWidth
                countHeight <- countHeight + 1
                pointerHorizontal <- pointerHorizontal + 1
            ELSE
                totalCost <- totalCost + verticalCut[pointerVertical] * countHeight
                countWidth <- countWidth + 1
                pointerVertical <- pointerVertical + 1
            END IF
        UNTIL pointerHorizontal >= lenHorizontal AND pointerVertical >= lenVertical

        RETURN totalCost
    END FUNCTION
END CLASS
