CLASS Solution  
    FUNCTION maximumValueSum(nums k edges)  
      SET acc TO 0  
      SET cnt TO 0

      SET idx TO 0  
      WHILE idx < LENGTH(nums)  
          SET val TO nums[idx]  
          SET x TO val XOR k  
          SET diff TO x - val  
          IF NOT (diff <= 0) THEN  
              SET cnt TO cnt + 1  
          END IF  
          SET acc TO acc + (IF x > val THEN x ELSE val END IF)  
          IF mn > ABS(diff) THEN  
              SET mn TO ABS(diff)  
          END IF  
          SET idx TO idx + 1  
      END WHILE  
      
      IF (cnt MOD 2) != 0 THEN  
          SET acc TO acc - mn  
      END IF  
      
      RETURN acc  
    END FUNCTION  
END CLASS
