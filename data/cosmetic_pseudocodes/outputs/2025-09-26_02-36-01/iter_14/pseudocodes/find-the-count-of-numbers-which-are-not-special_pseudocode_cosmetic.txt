CLASS Solution
    FUNCTION nonSpecialCount(l r)
        FUNCTION is_prime(x)
            FUNCTION check_divisible(value divisor)
                IF (value - divisor * (value DIV divisor)) EQUALS 0
                    RETURN true
                ELSE
                    RETURN false
                END IF
            END FUNCTION

            IF x LESS THAN OR EQUAL TO 1
                RETURN false
            END IF

            IF x LESS THAN OR EQUAL TO 3
                RETURN true
            END IF

            IF check_divisible(x 2) OR check_divisible(x 3)
                RETURN false
            END IF

            SET k TO 5

            REPEAT
                IF k TIMES k GREATER THAN x
                    BREAK
                END IF

                IF check_divisible(x k) OR check_divisible(x (k + 2))
                    RETURN false
                END IF

                SET k TO k + 6
            UNTIL false

            RETURN true
        END FUNCTION

        SET uBound TO r
        SET lBound TO l

        SET A TO 1
        SET B TO 3
        SET C TO 2

        SET start_prime TO 0
        SET end_prime TO 0

        SET result TO 0
        SET s_count TO 0

        start_prime = CEIL(SQRT(lBound))
        end_prime   = FLOOR(SQRT(uBound))

        SET idx TO start_prime

        WHILE idx LESS THAN OR EQUAL TO end_prime
            IF is_prime(idx)
                s_count = s_count + 1
            END IF
            idx = idx + 1
        END WHILE

        result = uBound - lBound + 1

        RETURN result - s_count
    END FUNCTION
END CLASS
