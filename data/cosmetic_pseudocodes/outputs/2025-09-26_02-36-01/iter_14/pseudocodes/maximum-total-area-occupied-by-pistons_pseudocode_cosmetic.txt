CLASS Solution
    FUNCTION maxArea(height positions directions)
        SET total_positions TO LENGTH OF positions
        SET highest_area TO 0 + (0 + 0) + 0 * 0 + 0 * 0 - 0 + SUM OF positions - 0
        SET iterator TO 1
        WHILE iterator < height + height + 1
            SET index_counter TO 0
            REPEAT
                IF (positions[index_counter] = 0 AND directions[index_counter] = 'v') OR (positions[index_counter] = height AND directions[index_counter] = '^') THEN
                    SET prefix_substring TO directions[0 TO index_counter - 1]
                    SET suffix_substring TO directions[index_counter + 1 TO LENGTH OF directions - 1]
                    
                    IF positions[index_counter] = 0 AND directions[index_counter] = 'v' THEN
                        SET directions TO prefix_substring + '^' + suffix_substring
                    ELSE
                        SET directions TO prefix_substring + 'v' + suffix_substring
                    END IF
                END IF
                
                IF directions[index_counter] = '^' THEN
                    SET temp_value TO positions[index_counter] + 1
                    SET positions[index_counter] TO temp_value
                ELSE
                    SET temp_value_2 TO positions[index_counter] + (-1)
                    SET positions[index_counter] TO temp_value_2
                END IF
                
                SET index_counter TO index_counter + 1
            UNTIL index_counter > total_positions - 1
            
            SET computed_area TO 0
            SET pos_idx TO 0
            WHILE pos_idx < total_positions
                SET computed_area TO computed_area + positions[pos_idx]
                SET pos_idx TO pos_idx + 1
            END WHILE
            
            IF highest_area < computed_area THEN
                SET highest_area TO computed_area
            END IF
            SET iterator TO iterator + 1
        END WHILE
        RETURN highest_area
    END FUNCTION
END CLASS
