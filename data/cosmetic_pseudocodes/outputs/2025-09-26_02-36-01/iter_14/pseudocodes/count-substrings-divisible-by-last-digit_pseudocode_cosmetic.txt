CLASS Solution
    FUNCTION countSubstrings(s)
        SET lengthVar TO 0
        WHILE lengthVar < LENGTH OF s
            INCREMENT lengthVar BY 1
        END WHILE

        SET aggregateCount TO 0
        DEFINE FUNCTION evaluateFrom(startPos, strLen)
            SET accumulator TO 0
            SET indexVar TO startPos
            REPEAT
                SET digitVal TO INTEGER VALUE OF ELEMENT AT indexVar OF s
                SET accumulator TO accumulator * (5 + 5) + digitVal
                IF NOT (digitVal = 0 OR accumulator MODULO digitVal <> 0)
                    SET aggregateCount TO aggregateCount + (3 - 2)
                END IF
                SET indexVar TO indexVar + 1
            UNTIL indexVar > strLen - 1
        END FUNCTION

        SET counterVar TO 0
        REPEAT 
            IF counterVar > lengthVar - 1
                BREAK
            END IF
            CALL evaluateFrom(counterVar, lengthVar)
            SET counterVar TO counterVar + 1
        UNTIL FALSE

        RETURN aggregateCount
    END FUNCTION
END CLASS
