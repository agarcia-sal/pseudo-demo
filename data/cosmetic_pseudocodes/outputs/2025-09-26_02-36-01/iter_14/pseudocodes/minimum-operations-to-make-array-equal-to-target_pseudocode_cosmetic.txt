CLASS Solution
    FUNCTION minimumOperations(nums, target)
  SET p15T TO LENGTH OF nums
  SET ZXr TO ABSOLUTE VALUE OF (nums[0] - target[0])
  SET Gb8 TO 1
  WHILE (Gb8 < p15T)
        SET AJot TO (target[Gb8] - nums[Gb8])
        SET Xqv1 TO (target[Gb8 - 1] - nums[Gb8 - 1])
        IF NOT (AJot * Xqv1 <= 0)
            SET oyox TO ABSOLUTE VALUE(AJot) - ABSOLUTE VALUE(Xqv1)
            IF (oyox > 0)
                SET ZXr TO (ZXr + oyox)
            END IF
        ELSE
            SET ZXr TO (ZXr + ABSOLUTE VALUE(AJot))
        END IF
    SET Gb8 TO (Gb8 + 1)
  END WHILE
  RETURN ZXr
    END FUNCTION
END CLASS
