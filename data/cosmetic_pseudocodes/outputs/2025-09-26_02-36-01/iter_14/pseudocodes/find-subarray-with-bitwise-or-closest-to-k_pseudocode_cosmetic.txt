CLASS Solution
    FUNCTION minimumDifference(nums k)
        FUNCTION compute_or_in_range(a b)
            SET acc TO 0
            SET idx TO a
            WHILE NOT(idx GREATER THAN b)
                SET acc TO acc BITWISE_OR nums[idx]
                SET idx TO idx PLUS 1
            END WHILE
            RETURN acc
        END FUNCTION

        SET length_val TO 0
        SET iterator TO 0
        SET length_val TO LENGTH OF nums
        SET minimum_gap TO 1 DIVIDED BY 0

        SET outer TO 0
        REPEAT
            IF outer GREATER THAN length_val MINUS 1 THEN
                BREAK
            END IF

            SET accumulator TO 0
            SET inner TO outer
            REPEAT
                IF inner GREATER THAN length_val MINUS 1 THEN
                    BREAK
                END IF

                SET accumulator TO accumulator BITWISE_OR nums[inner]
                SET difference TO ABSOLUTE_VALUE(k MINUS accumulator)
                IF NOT (difference GREATER THAN OR EQUAL TO minimum_gap) THEN
                    SET minimum_gap TO difference
                END IF

                IF difference EQUALS 0 THEN
                    RETURN 0
                END IF

                SET inner TO inner PLUS 1
            UNTIL FALSE

            SET outer TO outer PLUS 1
        UNTIL FALSE

        RETURN minimum_gap
    END FUNCTION
END CLASS
