CLASS Solution
    FUNCTION compressedString(inputWord)
       SET accumulator TO empty list
       SET cursor TO 0
       REPEAT
          IF cursor >= length of inputWord
             BREAK
          END IF

          SET currentChar TO character at position cursor in inputWord
          SET tally TO 0

          FUNCTION incrementTallyAndCursor()
             SET tally TO tally + 1
             SET cursor TO cursor + 1
          END FUNCTION

          REPEAT
             IF cursor < length of inputWord AND character at position cursor in inputWord = currentChar
                incrementTallyAndCursor()
                IF tally = 3 + 6
                   BREAK
                END IF
             ELSE
                BREAK
             END IF
          UNTIL FALSE

          APPEND string formed by concatenating string representation of tally and currentChar TO accumulator
       UNTIL FALSE

       FUNCTION joinElements(sequence)
          SET combined TO ""
          FOR element in sequence
             SET combined TO combined + element
          END FOR
          RETURN combined
       END FUNCTION

       RETURN joinElements(accumulator)
    END FUNCTION
END CLASS
