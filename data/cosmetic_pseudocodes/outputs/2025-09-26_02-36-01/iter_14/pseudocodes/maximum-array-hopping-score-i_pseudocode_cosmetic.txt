CLASS Solution
    FUNCTION maxScore(nums)
        SET alpha TO LENGTH OF nums
        DEFINE FUNCTION computeProduct(x, y)
            RETURN (x - y) * nums[x]
        END FUNCTION

        SET beta TO 0
        SET gamma TO LIST OF ZEROS WITH SIZE alpha
        SET gamma[1] TO beta

        SET p TO 2
        WHILE p <= alpha
            SET q TO 1
            WHILE q < p
                SET r TO gamma[q] + computeProduct(p, q)
                IF NOT (gamma[p] >= r)
                    SET gamma[p] TO r
                END IF
                SET q TO q + 1
            END WHILE
            SET p TO p + 1
        END WHILE

        RETURN gamma[alpha - 1]
    END FUNCTION
END CLASS
