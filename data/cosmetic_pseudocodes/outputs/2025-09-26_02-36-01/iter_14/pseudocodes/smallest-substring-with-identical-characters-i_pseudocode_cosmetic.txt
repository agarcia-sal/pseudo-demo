CLASS Solution
    FUNCTION minLength(s parameter, numOps parameter)
        
        FUNCTION longest_uniform_substring(s parameter)
            SET alpha TO 0
            SET beta TO 1
            SET gamma TO 1
            WHILE gamma LESS THAN LENGTH OF s
                IF NOT (s[gamma] NOT EQUALS s[gamma - 1])
                    SET alpha TO alpha
                ELSE
                    IF alpha LESS THAN beta
                        SET beta TO beta
                    ELSE
                        SET beta TO alpha
                    END IF
                    SET alpha TO 1
                END IF
                SET gamma TO gamma + 1
            END WHILE
            
            IF beta GREATER THAN alpha
                RETURN beta
            ELSE
                RETURN alpha
            END IF
        END FUNCTION
        
        SET omega TO LENGTH OF s
        SET psi TO 1 LEFT SHIFTED BY omega
        SET theta TO omega
        
        SET delta TO 0
        WHILE delta LESS THAN psi
            # Count bits set in delta exceeds numOps?
            SET epsilon TO 0
            SET zeta TO delta
            WHILE zeta NOT EQUAL 0
                SET epsilon TO epsilon + (zeta BITWISE AND 1)
                SET zeta TO zeta RIGHT SHIFTED BY 1
            END WHILE
            IF epsilon GREATER THAN numOps
                SET delta TO delta + 1
                CONTINUE
            END IF
            
            SET eta TO LIST OF CHARACTERS OF s
            SET iota TO 0
            WHILE iota LESS THAN omega
                IF (delta BITWISE AND (1 LEFT SHIFTED BY iota)) NOT EQUAL 0
                    IF eta[iota] EQUALS '0'
                        SET eta[iota] TO '1'
                    ELSE
                        SET eta[iota] TO '0'
                    END IF
                END IF
                SET iota TO iota + 1
            END WHILE
            
            SET kappa TO longest_uniform_substring(eta)
            IF theta GREATER THAN kappa
                SET theta TO kappa
            END IF
            
            SET delta TO delta + 1
        END WHILE
        
        RETURN theta
    END FUNCTION
END CLASS
