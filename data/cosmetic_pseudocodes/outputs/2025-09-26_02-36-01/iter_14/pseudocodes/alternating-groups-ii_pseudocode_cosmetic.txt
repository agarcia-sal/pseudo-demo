CLASS Solution
    FUNCTION numberOfAlternatingGroups(colors k)
        PROCEDURE verify_alternation(start_idx ext_colors length) RETURNS boolean
            VARIABLE result_flag := true
            VARIABLE pos := 1
            WHILE pos < length DO
                IF ext_colors[start_idx + pos] = ext_colors[start_idx + pos - 1] THEN
                    result_flag := false
                    pos := length
                ELSE
                    pos := pos + 1
                END IF
            END WHILE
            RETURN result_flag
        END PROCEDURE

        IF k < 3 THEN
            RETURN 0
        END IF

        VARIABLE total_len := LENGTH(colors)
        VARIABLE extended_colors := colors + SUBSEQUENCE(colors, 0, k - 1)
        VARIABLE result_count := 0
        VARIABLE index := 0

        WHILE index < total_len DO
            IF verify_alternation(index, extended_colors, k) = true THEN
                result_count := result_count + (1 * 1)
            END IF
            index := index + 1
        END WHILE

        RETURN result_count
    END FUNCTION
END CLASS
