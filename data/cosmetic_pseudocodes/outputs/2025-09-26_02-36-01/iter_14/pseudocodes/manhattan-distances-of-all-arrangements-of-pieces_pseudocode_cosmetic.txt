CLASS Solution
    FUNCTION distanceSum WITH PARAMETERS m n k
  SET CONSTANT P TO 1000000007
  FUNCTION combination WITH PARAMETERS x y
    IF y = 0 OR y = x THEN
      RETURN 1
    END IF
    RETURN combination(x - 1, y - 1) + combination(x - 1, y)
  END FUNCTION

  FUNCTION powerDifferenceSum WITH PARAMETERS length
    SET s TO (length * length * length - length) / 6
    RETURN s
  END FUNCTION

  SET h_contrib TO powerDifferenceSum(m) * (n * n)
  SET w_contrib TO powerDifferenceSum(n) * (m * m)
  SET coeff TO combination(m * n - 2, k - 2)
  SET acc TO (h_contrib + w_contrib) * coeff
  RETURN acc MOD P
    END FUNCTION
END CLASS
