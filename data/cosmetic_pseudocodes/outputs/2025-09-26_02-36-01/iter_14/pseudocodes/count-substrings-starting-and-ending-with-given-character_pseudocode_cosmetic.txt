CLASS Solution
    FUNCTION countSubstrings(s c)
        SET alpha TO 0
        SET beta TO 0
        SET gamma TO LENGTH s
        WHILE NOT (beta EQUALS gamma)
            IF NOT (s[beta] NOT EQUAL c)
                SET alpha TO alpha PLUS 1
            END IF
            SET beta TO beta PLUS 1
        END WHILE
        SET delta TO (alpha MULTIPLIED BY ((alpha PLUS 1) DIVIDED BY 2))
        RETURN delta
    END FUNCTION
END CLASS
