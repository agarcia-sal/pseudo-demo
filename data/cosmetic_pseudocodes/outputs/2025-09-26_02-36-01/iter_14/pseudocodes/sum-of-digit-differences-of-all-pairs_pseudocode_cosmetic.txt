CLASS Solution
    FUNCTION sumDigitDifferences(nums)
        FUNCTION digit_difference(a, b)
            SET diffCount TO 0
            SET pos TO 1
            WHILE pos <= LENGTH(a)
                IF NOT (a[pos] = b[pos])
                    SET diffCount TO diffCount + 1
                END IF
                SET pos TO pos + 1
            END WHILE
            RETURN diffCount
        END FUNCTION
        
        SET result TO 0
        SET lengthNums TO LENGTH(nums)
        
        SET idxOuter TO 0
        WHILE idxOuter < lengthNums - 1
            SET idxInner TO idxOuter + 1
            WHILE idxInner < lengthNums
                SET result TO result + digit_difference(nums[idxOuter], nums[idxInner])
                SET idxInner TO idxInner + 1
            END WHILE
            SET idxOuter TO idxOuter + 1
        END WHILE
        
        RETURN result
    END FUNCTION
END CLASS
