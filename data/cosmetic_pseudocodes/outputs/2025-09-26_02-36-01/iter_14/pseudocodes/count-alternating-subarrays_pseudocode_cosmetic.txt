CLASS Solution
    FUNCTION countAlternatingSubarrays(nums)
        SET total_elements TO LENGTH(nums)
        IF NOT (total_elements > 0) THEN
            RETURN 0
        ELSE
            SET total_subarrays TO total_elements
            SET sequence_length TO 1
            SET index_tracker TO 2
            WHILE index_tracker <= total_elements
                IF NOT (nums[index_tracker] = nums[index_tracker - 1]) THEN
                    SET sequence_length TO sequence_length + 1
                ELSE
                    SET sequence_length TO 1
                END IF
                SET total_subarrays TO total_subarrays + (sequence_length - 1)
                SET index_tracker TO index_tracker + 1
            END WHILE
            RETURN total_subarrays
        END IF
    END FUNCTION
END CLASS
