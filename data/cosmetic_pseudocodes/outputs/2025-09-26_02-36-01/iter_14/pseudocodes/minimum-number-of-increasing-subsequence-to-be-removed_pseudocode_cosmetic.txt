CLASS Solution
    FUNCTION minOperations(nums)
      SET alpha TO 0
      SET beta TO LENGTH OF nums
      IF NOT (beta GREATER THAN 0)
          RETURN 0
      ELSE
          SET gamma TO new list of size beta WITH all elements 1
          SET delta TO 1
          WHILE delta LESS THAN beta
              SET epsilon TO 0
              REPEAT
                  IF nums[delta] GREATER THAN nums[epsilon] OR NOT (nums[delta] LESS THAN OR EQUAL TO nums[epsilon])
                      NOOP
                  ELSE
                      IF gamma[delta] LESS THAN OR EQUAL TO gamma[epsilon]
                          SET gamma[delta] TO gamma[epsilon] PLUS 1
                      END IF
                  END IF
                  SET epsilon TO epsilon PLUS 1
              UNTIL epsilon EQUALS delta
              SET delta TO delta PLUS 1
          END WHILE
          SET zeta TO gamma[0]
          SET eta TO 1
          WHILE eta LESS THAN beta
              IF zeta GREATER THAN gamma[eta]
                  NOOP
              ELSE
                  SET zeta TO gamma[eta]
              END IF
              SET eta TO eta PLUS 1
          END WHILE
          RETURN zeta
      END IF
    END FUNCTION
END CLASS
