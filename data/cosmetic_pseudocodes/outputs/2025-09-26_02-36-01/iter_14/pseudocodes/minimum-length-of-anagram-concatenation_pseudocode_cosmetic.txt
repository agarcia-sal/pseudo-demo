CLASS Solution
    FUNCTION minAnagramLength(s)
        FUNCTION buildCharSet(t)
            SET resultSet TO empty set
            SET index TO 0
            WHILE index < LENGTH(t)
                SET currChar TO t[index]
                IF NOT (currChar IN resultSet)
                    ADD currChar TO resultSet
                END IF
                SET index TO index + 1
            END WHILE
            RETURN resultSet
        END FUNCTION

        SET allFound TO buildCharSet(s)
        SET totalCount TO 0

        FOR EACH keyChar IN allFound
            SET totalCount TO totalCount + 1
        END FOR
        
        RETURN totalCount
    END FUNCTION
END CLASS
