CLASS Solution
    FUNCTION minimumValueSum(nums, andValues)
      SET omega TO LENGTH OF nums
      SET sigma TO LENGTH OF andValues

      FUNCTION delta(alpha, beta)
          IF beta = -1 THEN
              IF alpha = -1 THEN
                  RETURN 0
              ELSE
                  RETURN +∞
              END IF
          END IF

          IF alpha = -1 THEN
              RETURN +∞
          END IF

          SET psi TO +∞
          SET rho TO -1
          SET gamma TO alpha
          WHILE gamma >= -1 DO
              IF rho = -1 THEN
                  SET rho TO nums[gamma]
              ELSE
                  SET rho TO rho BITWISE_AND nums[gamma]
              END IF

              IF rho = andValues[beta] THEN
                  SET delta_candidate TO delta(gamma - 1, beta - 1) + nums[alpha]
                  IF delta_candidate < psi THEN
                      SET psi TO delta_candidate
                  END IF
              END IF

              SET gamma TO gamma - 1
          END WHILE

          RETURN psi
      END FUNCTION

      SET outcome TO delta(omega - 1, sigma - 1)
      IF outcome != +∞ THEN
          RETURN outcome
      ELSE
          RETURN -1
      END IF
    END FUNCTION
END CLASS
