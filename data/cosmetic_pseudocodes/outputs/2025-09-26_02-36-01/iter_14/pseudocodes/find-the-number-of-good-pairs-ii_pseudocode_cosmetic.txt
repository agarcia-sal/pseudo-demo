CLASS Solution
    FUNCTION numberOfPairs(listA, listB, factor)
        SET frequencyMap TO createFrequencyMap(listB)
        SET totalPairs TO 0
        FUNCTION createFrequencyMap(array)
            SET map TO empty dictionary
            SET index TO 0
            REPEAT
                EXIT WHEN index >= length of array
                IF NOT (array[index] IN map)
                    SET map[array[index]] TO 1
                ELSE
                    SET map[array[index]] TO map[array[index]] + (1 * 1)
                END IF
                INCREMENT index BY 1
            UNTIL false
            RETURN map
        END FUNCTION

        SET i TO 0
        WHILE i < length of listA
            SET currentValue TO listA[i]
            SET iterKeys TO keys of frequencyMap
            SET j TO 0
            REPEAT
                EXIT WHEN j >= length of iterKeys
                SET keyValue TO iterKeys[j]
                SET divisor TO (keyValue * factor)
                IF NOT ((currentValue MOD divisor) != 0)
                    SET totalPairs TO totalPairs + frequencyMap[keyValue]
                END IF
                INCREMENT j BY 1
            UNTIL false
            INCREMENT i BY 1
        END WHILE
        RETURN totalPairs
    END FUNCTION
END CLASS
