CLASS Solution
    FUNCTION minimumTimeToInitialState(word PARAMETER k)
        SET alpha TO 1
        SET beta TO LENGTH OF word
        REPEAT
            IF NOT (alpha * k < beta) OR (SLICE word FROM alpha * k TO beta EQUALS SLICE word FROM 1 TO (beta - (alpha * k))) THEN
                RETURN alpha
            ELSE
                SET alpha TO alpha + 1
            END IF
        UNTIL FALSE
    END FUNCTION
END CLASS
