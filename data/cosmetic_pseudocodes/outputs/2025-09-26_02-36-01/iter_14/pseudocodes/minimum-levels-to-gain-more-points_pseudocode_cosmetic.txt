CLASS Solution
    FUNCTION minimumLevels(possible)
        SET beta_sum TO 0
        SET i TO 0
        WHILE i < LENGTH(possible)
            SET beta_sum TO beta_sum + (2 * possible[i] - 1)
            SET i TO i + 1
        END WHILE
        SET delta TO 0
        SET j TO 0
        REPEAT
            SET delta TO delta + (2 * possible[j] - 1)
            SET beta_sum TO beta_sum - (2 * possible[j] - 1)
            IF NOT (delta <= beta_sum)
                RETURN (j + 1)
            END IF
            SET j TO j + 1
        UNTIL j > LENGTH(possible) - 2
        RETURN (-1)
    END FUNCTION
END CLASS
