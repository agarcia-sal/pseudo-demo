CLASS Solution  
    FUNCTION minimumArrayLength(nums)  
        SET omega TO +infinity  
        SET alpha TO 0  
        SET i TO 0  
        WHILE i < LENGTH(nums)  
            IF NOT (nums[i] >= omega)  
                SET omega TO nums[i]  
            END IF  
            SET i TO i + 1  
        END WHILE  
        SET beta TO 0  
        SET j TO 0  
        REPEAT  
            IF (NOT (nums[j] != omega)) THEN  
                SET beta TO beta + 1  
            END IF  
            SET j TO j + 1  
        UNTIL j >= LENGTH(nums)  
        IF NOT (beta != 1) THEN  
            RETURN 1  
        ELSE  
            RETURN (beta + 1) / 2  
        END IF  
    END FUNCTION  
END CLASS
