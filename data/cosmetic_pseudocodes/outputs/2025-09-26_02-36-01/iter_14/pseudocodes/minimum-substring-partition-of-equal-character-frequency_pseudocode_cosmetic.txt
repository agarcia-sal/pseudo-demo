CLASS Solution
    FUNCTION minimumSubstringsInPartition(s)
        FUNCTION dfs(k)
            IF k GREATER THAN OR EQUAL TO LENGTH(s) THEN
                RETURN 0
            END IF
            SET u TO new mapping defaulting to zero
            SET v TO new mapping defaulting to zero
            SET r TO LENGTH(s) - k
            SET m TO k
            WHILE m LESS THAN LENGTH(s)
                IF EXISTS u[s[m]] AND u[s[m]] NOT EQUAL TO 0 THEN
                    LET w = u[s[m]]
                    LET v[w] = v[w] - 1
                    IF v[w] EQUALS 0 THEN
                        DELETE v[w]
                    END IF
                END IF
                LET u[s[m]] = u[s[m]] + 1
                LET x = u[s[m]]
                LET v[x] = v.get(x,0) + 1
                IF SIZE OF KEYS(v) EQUALS 1 THEN
                    LET y = dfs(m + 1) + 1
                    IF y LESS THAN r THEN
                        SET r = y
                    END IF
                END IF
                m = m + 1
            END WHILE
            RETURN r
        END FUNCTION
        LET p = LENGTH(s)
        RETURN dfs(0)
    END FUNCTION
END CLASS
