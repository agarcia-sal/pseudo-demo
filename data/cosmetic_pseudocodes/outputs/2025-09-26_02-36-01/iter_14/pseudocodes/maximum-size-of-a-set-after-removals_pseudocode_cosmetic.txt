CLASS Solution
    FUNCTION maximumSetSize(nums1 PARAMETER List OF integers nums2 PARAMETER List OF integers) RETURNS integer
        SET l TO LENGTH OF nums1
        SET q TO l DIVIDED BY 2

        SET u TO EMPTY SET
        FOR EACH e IN nums1
            IF e NOT IN u THEN
                ADD e TO u
            END IF
        END FOR

        SET v TO EMPTY SET
        FOR EACH f IN nums2
            IF f NOT IN v THEN
                ADD f TO v
            END IF
        END FOR

        FUNCTION intersect(a SET, b SET) RETURNS SET
            SET r TO EMPTY SET
            FOR EACH x IN a
                IF x IN b THEN
                    ADD x TO r
                END IF
            END FOR
            RETURN r
        END FUNCTION

        SET w TO intersect(u, v)

        FUNCTION difference(a SET, b SET) RETURNS SET
            SET s TO EMPTY SET
            FOR EACH y IN a
                IF NOT (y IN b) THEN
                    ADD y TO s
                END IF
            END FOR
            RETURN s
        END FUNCTION

        SET p TO difference(u, w)
        SET m TO difference(v, w)

        FUNCTION minimum(x integer, y integer) RETURNS integer
            IF x < y THEN
                RETURN x
            ELSE
                RETURN y
            END IF
        END FUNCTION

        SET c TO minimum(q, LENGTH OF p)
        SET d TO minimum(q, LENGTH OF m)

        FUNCTION maximum(x integer, y integer) RETURNS integer
            IF x > y THEN
                RETURN x
            ELSE
                RETURN y
            END IF
        END FUNCTION

        SET r TO maximum(0, q - c) + maximum(0, q - d) 

        FUNCTION addIntegers(a integer, b integer) RETURNS integer
            RETURN a + b
        END FUNCTION

        SET total TO addIntegers(addIntegers(c, d), minimum(r, LENGTH OF w))

        RETURN total
    END FUNCTION
END CLASS
