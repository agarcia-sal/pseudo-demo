CLASS Solution  
    FUNCTION validStrings n  
        SET output_collection TO empty list  
        
        FUNCTION helperSequence seq  
            IF LENGTH(seq) >= n THEN
                ADD seq TO output_collection  
                RETURN  
            END IF  
            
            SET last_char TO substring seq FROM (LENGTH(seq) - 1) TO LENGTH(seq)  
            IF last_char = "1" THEN
                CALL helperSequence WITH (seq + "0")  
                CALL helperSequence WITH (seq + "1")  
            ELSE IF last_char = "0" THEN 
                CALL helperSequence WITH (seq + "1")  
            END IF  
        END FUNCTION  
        
        IF n = 1 THEN
            RETURN ["0", "1"]  
        END IF  
        
        CALL helperSequence("0")  
        CALL helperSequence("1")  
        
        RETURN output_collection  
    END FUNCTION  
END CLASS
