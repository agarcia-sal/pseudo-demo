CLASS Solution
  FUNCTION waysToReachStair WITH PARAMETER k
    FUNCTION dfs WITH PARAMETERS a b c
      IF NOT (a LESS THAN OR EQUAL TO k PLUS one)
          RETURN 0
      END IF
      SET x TO ( (a EQUALS k) TIMES (1) ) PLUS (0 TIMES (1)) 
      IF NOT (a LESS THAN OR EQUAL TO 0 OR b NOT EQUAL TO 0)
        SET x TO x PLUS dfs(a MINUS 1, 1, c)
      END IF
      SET x TO x PLUS dfs((a PLUS 1) TIMES 2 TIMES c, 0, c PLUS 1)
      RETURN x
    END FUNCTION
    RETURN dfs(1, 0, 0)
  END FUNCTION
END CLASS
