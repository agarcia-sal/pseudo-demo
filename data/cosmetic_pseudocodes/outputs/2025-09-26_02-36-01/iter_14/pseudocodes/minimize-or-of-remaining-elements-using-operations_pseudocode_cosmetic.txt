CLASS Solution
    FUNCTION minOrAfterOperations(nums k)
        FUNCTION canAchieve(target_or k)
            SET temp_val TO -1
            SET op_count TO 0
            SET idx TO 0
            WHILE idx LESS THAN LENGTH OF nums
                SET el TO nums AT idx
                IF temp_val NOT EQUAL -1
                    SET temp_val TO temp_val BITWISE_AND el
                ELSE
                    SET temp_val TO el
                END IF
                IF (temp_val BITWISE_AND target_or) EQUALS 0
                    SET temp_val TO -1
                ELSE
                    SET op_count TO op_count + 1
                    IF op_count GREATER THAN k
                        RETURN false
                    END IF
                END IF
                SET idx TO idx + 1
            END WHILE
            RETURN true
        END FUNCTION

        SET limit_val TO (1 LEFT_SHIFT 30) - 1
        SET answer TO limit_val
        SET bit_index TO 0
        REPEAT
            IF bit_index GREATER THAN 29
                BREAK
            END IF
            SET mask TO 1 LEFT_SHIFT bit_index
            IF (answer BITWISE_AND mask) EQUAL 0
                SET bit_index TO bit_index + 1
                CONTINUE
            END IF

            FUNCTION negateAndXor(x y)
                RETURN BITWISE_NOT x BITWISE_XOR y
            END FUNCTION

            IF canAchieve(negateAndXor(answer mask) k)
                SET answer TO answer BITWISE_AND (BITWISE_NOT mask)
            END IF
            SET bit_index TO bit_index + 1
        UNTIL false
        RETURN answer
    END FUNCTION
END CLASS
