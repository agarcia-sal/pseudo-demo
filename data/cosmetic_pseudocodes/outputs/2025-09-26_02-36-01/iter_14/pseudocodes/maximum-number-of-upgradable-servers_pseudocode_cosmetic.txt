CLASS Solution
    FUNCTION maxUpgrades(count LIST OF INTEGER upgrade LIST OF INTEGER sell LIST OF INTEGER money LIST OF INTEGER) RETURNS LIST OF INTEGER
        SET result TO empty list
        SET pos TO 0
        WHILE pos LESS THAN LENGTH OF count
            SET servers_qty TO count[pos]
            SET cost_upgrade TO upgrade[pos]
            SET price_sell TO sell[pos]
            SET funds TO money[pos]
            SET peak_upgrade TO 0
            SET try_sell TO servers_qty
            REPEAT
                SET left_servers TO servers_qty MINUS try_sell
                SET cash_gain TO try_sell MULTIPLIED BY price_sell
                SET total_funds TO funds PLUS cash_gain
                SET possible TO total_funds DIVIDED BY cost_upgrade
                IF possible GREATER THAN left_servers THEN
                    SET possible TO left_servers
                ELSE
                    NO OPERATION
                END IF
                IF possible GREATER THAN peak_upgrade THEN
                    SET peak_upgrade TO possible
                ELSE
                    NO OPERATION
                END IF
                SET try_sell TO try_sell MINUS 1
            UNTIL try_sell LESS THAN ZERO
            APPEND peak_upgrade TO result
            SET pos TO pos PLUS 1
        END WHILE
        RETURN result
    END FUNCTION
END CLASS
