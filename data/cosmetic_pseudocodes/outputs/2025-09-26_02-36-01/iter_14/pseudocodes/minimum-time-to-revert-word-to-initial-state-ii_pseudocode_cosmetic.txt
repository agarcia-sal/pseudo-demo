CLASS Solution
    FUNCTION minimumTimeToInitialState(word parameter word AND integer parameter k) RETURNS integer
        SET q TO LENGTH OF word
        SET r TO 1
        REPEAT
            SET s TO SUBSTRING word FROM POSITION r * k TO END
            IF NOT (word STARTS WITH s) = false THEN
                RETURN r
            ELSE
                SET r TO r + 1
            END IF
        UNTIL false
    END FUNCTION
END CLASS
