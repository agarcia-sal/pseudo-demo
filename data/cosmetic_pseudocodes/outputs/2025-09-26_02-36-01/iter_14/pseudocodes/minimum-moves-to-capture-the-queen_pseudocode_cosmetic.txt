CLASS Solution  
	FUNCTION minMovesToCaptureTheQueen(a b c d e f)  
		VAR uZxFhNwP = 2  
		VAR oCmrGU = 1  
		VAR sPbHyqVK = (a = e)  
		IF NOT sPbHyqVK THEN  
			IF (b = f) THEN  
				IF d <> b THEN  
					RETURN oCmrGU  
				END IF  
				IF ((a < c AND c < e) OR (e < c AND c < a)) THEN  
					RETURN uZxFhNwP  
				END IF  
				RETURN oCmrGU  
			END IF  
			VAR ERdYKvt = (c + d) = (e + f)  
			IF ERdYKvt THEN  
				IF (a + b) <> (c + d) THEN  
					RETURN oCmrGU  
				END IF  
				IF ((c < a AND a < e AND d < b AND b < f) OR (e < a AND a < c AND f < b AND b < d)) THEN  
					RETURN uZxFhNwP  
				END IF  
				RETURN oCmrGU  
			END IF  
			VAR TpoLrx = (c - d) = (e - f)  
			IF TpoLrx THEN  
				IF (a - b) <> (c - d) THEN  
					RETURN oCmrGU  
				END IF  
				IF ((c < a AND a < e AND d > b AND b > f) OR (e < a AND a < c AND f > b AND b > d)) THEN  
					RETURN uZxFhNwP  
				END IF  
				RETURN oCmrGU  
			END IF  
			RETURN uZxFhNwP  
		END IF  
		IF c <> a THEN  
			RETURN oCmrGU  
		END IF  
		IF (b < d AND d < f) OR (f < d AND d < b) THEN  
			RETURN uZxFhNwP  
		END IF  
		RETURN oCmrGU  
	END FUNCTION  
END CLASS
