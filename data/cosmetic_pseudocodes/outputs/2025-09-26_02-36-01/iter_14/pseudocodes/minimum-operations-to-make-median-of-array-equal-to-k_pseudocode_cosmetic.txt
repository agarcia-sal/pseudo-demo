CLASS Solution  
    FUNCTION minOperationsToMakeMedianK(nums k)  
      SET a TO 0  
      SET b TO LENGTH OF nums  
      SET c TO b DIVIDE BY 2  
      SORT nums BY ascending ORDER  
      
      IF NOT (nums[c] NOT EQUAL TO k) THEN  
          RETURN a * a  
      ELSE  
          SET d TO a * a  
          IF nums[c] LESS THAN k THEN  
              LABEL loop_start_1  
              IF nums[c] NOT LESS THAN k THEN  
                  GOTO loop_end_1  
              END IF  
              SET d TO d + (k - nums[c])  
              SET c TO c + 1  
              IF c GREATER OR EQUAL TO b THEN  
                  GOTO loop_end_1  
              END IF  
              GOTO loop_start_1  
              LABEL loop_end_1  
          ELSE  
              LABEL loop_start_2  
              IF NOT (nums[c] GREATER THAN k) THEN  
                  GOTO loop_end_2  
              END IF  
              SET d TO d + (nums[c] - k)  
              SET c TO c - 1  
              IF c LESS THAN 0 THEN  
                  GOTO loop_end_2  
              END IF  
              GOTO loop_start_2  
              LABEL loop_end_2  
          END IF  
          RETURN d  
      END IF  
    END FUNCTION  
END CLASS
