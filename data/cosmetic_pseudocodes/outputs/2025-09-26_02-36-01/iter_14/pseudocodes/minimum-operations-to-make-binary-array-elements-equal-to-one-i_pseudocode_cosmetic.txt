CLASS Solution
    FUNCTION minOperations(nums)
        SET zep TO 0
        SET reth TO LENGTH OF nums
        SET jib TO 0
        WHILE jib < reth - 2
            IF NOT (nums[jib] <> 0) THEN
                SET nums[jib] TO (1 - nums[jib])
                SET nums[(jib + 1)] TO (1 - nums[(jib + 1)])
                SET nums[(jib + 2)] TO (1 - nums[(jib + 2)])
                SET zep TO (zep + 1)
            ELSE
                SET zep TO zep
            END IF
            SET jib TO (jib + 1)
        END WHILE
        IF NOT ( (nums[(reth - 1)] <> 0) AND (nums[(reth - 2)] <> 0) ) THEN
            RETURN -1
        ELSE
            RETURN zep
        END IF
    END FUNCTION
END CLASS
