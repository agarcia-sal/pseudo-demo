CLASS Solution
    FUNCTION queryResults(limit parameter, queries parameter)
        DECLARE tempMap AS dictionary mapping identifiers TO colors INITIALIZED EMPTY
        DECLARE distinctSet AS set INITIALIZED EMPTY
        DECLARE outputList AS empty list
        
        REPEAT index FROM 0 TO LENGTH(queries parameter) - 1
            LET keyVar BE queries parameter[index].x parameter
            LET valVar BE queries parameter[index].y parameter
            
            IF tempMap CONTAINS keyVar THEN
                DECLARE prevValue AS tempMap[keyVar]
                IF distinctSet CONTAINS prevValue THEN
                    distinctSet.DELETE(prevValue)
                END IF
            END IF
            
            tempMap[keyVar] <- valVar
            distinctSet.INSERT(valVar)
            outputList.ADD(SIZE(distinctSet))
        END REPEAT
        
        RETURN outputList
    END FUNCTION
END CLASS
