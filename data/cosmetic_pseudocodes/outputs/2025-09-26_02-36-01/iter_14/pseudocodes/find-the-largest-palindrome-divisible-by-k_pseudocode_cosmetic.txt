CLASS Solution
    FUNCTION largestPalindrome(n k)
        IF n NOT EQUAL one THEN
            SET alpha TO ((n + 1) DIV 2)
            SET beta TO convert_to_int(repeat_characters('9', alpha))
            
            SET gamma TO beta
            WHILE gamma > 0 DO
                IF (n MOD 2) NOT EQUAL 1 THEN
                    SET delta TO convert_to_int(concatenate(convert_to_str(gamma), reverse_string(convert_to_str(gamma))))
                ELSE
                    SET epsilon TO substring(convert_to_str(gamma), 1, length_of(convert_to_str(gamma)) - 1)
                    SET delta TO convert_to_int(concatenate(convert_to_str(gamma), reverse_string(epsilon)))
                END IF
                
                IF delta MOD k = 0 THEN
                    RETURN convert_to_str(delta)
                END IF
                
                SET gamma TO gamma - 1
            END WHILE
            
            RETURN convert_to_str(0)
        ELSE
            SET i TO 9
            WHILE i >= 1 DO
                IF i MOD k = 0 THEN
                    RETURN convert_to_str(i)
                END IF
                SET i TO i - 1
            END WHILE
            
            RETURN convert_to_str(0)
        END IF
    END FUNCTION
END CLASS
