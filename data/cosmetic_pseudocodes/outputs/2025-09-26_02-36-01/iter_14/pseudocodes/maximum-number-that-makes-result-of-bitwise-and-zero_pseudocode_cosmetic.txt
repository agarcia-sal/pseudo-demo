CLASS Solution  
    FUNCTION maxNumber(n)  
        SET result TO 0  
        IF NOT (n NOT EQUALS 1) THEN  
            SET result TO 0  
        ELSE  
            SET alpha TO 1  
            REPEAT  
                SET alpha TO alpha + alpha  
            UNTIL alpha > n  
            SET alpha TO alpha / 2  
            SET result TO alpha - 1  
        END IF  
        RETURN result  
    END FUNCTION  
END CLASS
