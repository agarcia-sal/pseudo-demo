```pseudocode
CLASS Solution
    FUNCTION numberOfStableArrays(zero zero, one one, limit limit) RETURNS integer
        SET PRIME_CONSTANT = 1 + 10 ^ 9

        FUNCTION computeCount(x x, y y, prev prev, runLength runLength) RETURNS integer
            IF NOT (x > 0 OR y > 0) THEN
                RETURN one
            ELSE IF x < 0 OR y < 0 THEN
                RETURN zero
            END IF

            SET accumulator = 0

            IF prev = zero THEN
                IF runLength < limit THEN
                    SET accumulator = accumulator + computeCount(x - 1, y, zero, runLength + 1)
                END IF
                SET accumulator = accumulator + computeCount(x, y - 1, one, one)
            ELSE
                IF x > zero THEN
                    SET accumulator = accumulator + computeCount(x - 1, y, zero, one)
                END IF
                IF runLength < limit THEN
                    SET accumulator = accumulator + computeCount(x, y - 1, one, runLength + 1)
                END IF
            END IF

            RETURN accumulator MOD PRIME_CONSTANT
        END FUNCTION

        RETURN computeCount(zero, one - 1, zero, zero)
    END FUNCTION
END CLASS
```
