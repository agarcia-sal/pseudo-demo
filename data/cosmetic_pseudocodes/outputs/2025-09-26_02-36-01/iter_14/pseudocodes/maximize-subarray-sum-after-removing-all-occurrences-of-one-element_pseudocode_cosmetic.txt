CLASS Solution
    FUNCTION maxSubarraySum(nums)
        FUNCTION kadane(arr)
            SET alpha TO arr[1]
            SET beta TO arr[1]
            SET delta TO 2
            WHILE delta LESS THAN OR EQUAL TO LENGTH OF arr
                SET eta TO arr[delta]
                IF NOT (eta LESS THAN OR EQUAL TO alpha PLUS eta) THEN
                    SET alpha TO eta
                ELSE
                    SET alpha TO alpha PLUS eta
                END IF
                IF beta LESS THAN alpha
                    SET beta TO alpha
                END IF
                SET delta TO delta PLUS 1
            END WHILE
            RETURN beta
        END FUNCTION

        SET pi TO kadane(nums)
        SET sigma TO EMPTY SET
        SET iota TO 1
        WHILE iota LESS THAN OR EQUAL TO LENGTH OF nums
            ADD nums[iota] TO sigma
            SET iota TO iota PLUS 1
        END WHILE

        SET omega TO sigma
        SET gamma TO 1
        WHILE gamma LESS THAN OR EQUAL TO LENGTH OF omega
            SET theta TO omega[gamma]
            SET lambda TO EMPTY LIST
            SET mu TO 1
            WHILE mu LESS THAN OR EQUAL TO LENGTH OF nums
                SET nu TO nums[mu]
                IF nu EQUALS theta THEN
                    ; skip element
                ELSE
                    APPEND nu TO lambda
                END IF
                SET mu TO mu PLUS 1
            END WHILE
            IF LENGTH OF lambda GREATER THAN 0
                SET rho TO kadane(lambda)
                IF pi LESS THAN rho
                    SET pi TO rho
                END IF
            END IF
            SET gamma TO gamma PLUS 1
        END WHILE
        RETURN pi
    END FUNCTION
END CLASS
