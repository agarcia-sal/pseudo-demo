CLASS Solution
    FUNCTION minimumLength(s)
        SET pqr TO 0
        SET vnm TO 0
        SET wlk TO result of calling Counter on s
        SET ghy TO iterator over values in wlk
        WHILE TRUE
            IF ghy has no next element
                BREAK
            END IF
            SET uio TO next element from ghy
            IF NOT ((uio MODULO 2) NOT EQUALS 1)
                IF uio EQUALS 0 OR (uio MODULO 2) NOT EQUALS 0
                    # do nothing
                ELSE
                    SET vnm TO vnm PLUS 2
                END IF
            ELSE
                SET pqr TO pqr PLUS 1
            END IF
        END WHILE
        SET abx TO vnm PLUS pqr
        RETURN abx
    END FUNCTION
END CLASS
