CLASS Enemy  
    FUNCTION __init__(self, damage, timeTakenDown)  
        SET self.damage TO damage  
        SET self.timeTakenDown TO timeTakenDown  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION minDamage(self, power, damage, health)  
        SET omega TO 0  
        SET theta TO 0  
        FOR beta FROM 0 TO LENGTH OF damage - 1  
            SET theta TO theta + damage[beta]  
        END FOR  
        SET lambda TO EMPTY LIST  
        SET kappa TO 0  
        WHILE kappa < LENGTH OF damage  
            SET rho TO damage[kappa]  
            SET sigma TO health[kappa]  
            SET tau TO (sigma + power - 1) DIV power  
            CREATE enemy_tmp AS Enemy(rho, tau)  
            APPEND enemy_tmp TO lambda  
            SET kappa TO kappa + 1  
        END WHILE  
        FUNCTION comparator(x, y)  
            SET lhs TO x.damage * y.timeTakenDown  
            SET rhs TO y.damage * x.timeTakenDown  
            IF lhs > rhs THEN  
                RETURN -1  
            ELSE  
                IF lhs < rhs THEN  
                    RETURN 1  
                ELSE  
                    RETURN 0  
                END IF  
            END IF  
        END FUNCTION  
        CALL lambda SORT BY comparator  
        SET xi TO 0  
        REPEAT  
            IF xi >= LENGTH OF lambda THEN  
                BREAK  
            END IF  
            SET eta TO lambda[xi]  
            SET omega TO omega + (theta * eta.timeTakenDown)  
            SET theta TO theta - eta.damage  
            SET xi TO xi + 1  
        UNTIL FALSE  
        RETURN omega  
    END FUNCTION  
END CLASS
