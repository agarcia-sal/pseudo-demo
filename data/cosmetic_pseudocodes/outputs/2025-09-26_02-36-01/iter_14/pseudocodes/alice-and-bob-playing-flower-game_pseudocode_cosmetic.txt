CLASS Solution  
    FUNCTION flowerGame(x, y)  
        DEFINE FUNCTION computeHalfValues(a, b)  
            SET varP TO (a + 1) // 2  
            SET varQ TO a // 2  
            SET varR TO (b + 1) // 2  
            SET varS TO b // 2  
            RETURN (varP, varQ, varR, varS)  
        END FUNCTION  

        SET (alpha, beta, gamma, delta) TO computeHalfValues(x, y)  
        SET result TO (alpha * delta) + (beta * gamma)  
        RETURN result  
    END FUNCTION  
END CLASS
