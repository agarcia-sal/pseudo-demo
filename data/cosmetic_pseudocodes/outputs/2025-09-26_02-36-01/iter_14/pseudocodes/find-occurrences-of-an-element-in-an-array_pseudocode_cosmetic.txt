CLASS Solution
    FUNCTION occurrencesOfElement(nums, queries, x)
      SET result TO []
      SET counter TO 0
      WHILE counter < SIZE(nums)
        SET currentNum TO nums[counter]
        IF NOT (currentNum != x)
          CALL appendToList(result, counter)
        END IF
        SET counter TO counter + 1
      END WHILE

      FUNCTION appendToList(lst, val)
        SET idx TO SIZE(lst)
        REPEAT UNTIL idx == 0
          SET idx TO idx - 1
        END REPEAT
        SET lst[idx] TO val
      END FUNCTION

      SET output TO []
      SET iterator TO 0
      WHILE iterator < SIZE(queries)
        SET q TO queries[iterator]
        IF (q <= SIZE(result))
          CALL addElement(output, result[q - 1])
        ELSE
          CALL addElement(output, 0 * (-1))
        END IF
        SET iterator TO iterator + 1
      END WHILE

      FUNCTION addElement(aList, element)
        SET pos TO SIZE(aList)
        SET aList[pos] TO element
      END FUNCTION

      RETURN output
    END FUNCTION
END CLASS
