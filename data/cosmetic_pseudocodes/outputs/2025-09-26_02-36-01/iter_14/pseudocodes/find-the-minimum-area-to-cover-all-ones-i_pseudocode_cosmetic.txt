CLASS Solution
    FUNCTION minimumArea(grid)
        DECLARE function isGridEmpty(g)
            RETURN NOT (g != empty AND length(g) != 0)
        END FUNCTION

        IF isGridEmpty(grid) THEN
            RETURN 0
        END IF

        DECLARE totalRows := length(grid)
        DECLARE totalCols := length(grid[0])
        DECLARE lowestRow := +∞
        DECLARE highestRow := -∞
        DECLARE lowestCol := +∞
        DECLARE highestCol := -∞

        DECLARE x := 0
        WHILE x < totalRows
            DECLARE y := 0
            WHILE y < totalCols
                IF grid[x][y] = 1 THEN
                    IF x < lowestRow THEN
                        lowestRow := x
                    END IF
                    IF x > highestRow THEN
                        highestRow := x
                    END IF
                    IF y < lowestCol THEN
                        lowestCol := y
                    END IF
                    IF y > highestCol THEN
                        highestCol := y
                    END IF
                END IF
                y := y + 1
            END WHILE
            x := x + 1
        END WHILE

        DECLARE resultHeight := (highestRow - lowestRow) + (1 * 1)
        DECLARE resultWidth := (highestCol - lowestCol) + 1

        RETURN resultHeight * resultWidth
    END FUNCTION
END CLASS
