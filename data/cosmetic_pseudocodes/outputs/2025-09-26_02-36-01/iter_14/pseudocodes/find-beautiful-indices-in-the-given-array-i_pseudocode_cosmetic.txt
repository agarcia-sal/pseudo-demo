CLASS Solution
    FUNCTION beautifulIndices(s parameter, a parameter, b parameter, k parameter)
        SET list_xm TO []
        SET len_sx TO LENGTH s
        SET len_ax TO LENGTH a
        SET rec_idx TO 0
        WHILE rec_idx <= len_sx - len_ax
            IF NOT (SUBSTRING s FROM rec_idx TO (rec_idx + len_ax - 1)) != a THEN
                APPEND rec_idx TO list_xm
            END IF
            SET rec_idx TO rec_idx + 1
        END WHILE

        SET list_yn TO []
        SET len_bx TO LENGTH b
        SET pointer_p TO 0
        REPEAT
            IF (SUBSTRING s FROM pointer_p TO (pointer_p + len_bx - 1)) = b THEN
                APPEND pointer_p TO list_yn
            END IF
            SET pointer_p TO pointer_p + 1
        UNTIL pointer_p > len_sx - len_bx

        SET container_zq TO []
        SET idx_outer TO 0
        WHILE idx_outer < LENGTH list_xm
            SET val_outer TO list_xm[idx_outer]
            SET idx_inner TO 0
            SET appended_flag TO false
            WHILE NOT appended_flag AND idx_inner < LENGTH list_yn
                SET val_inner TO list_yn[idx_inner]
                IF (val_outer - val_inner >= -k) AND (val_outer - val_inner <= k) THEN
                    APPEND val_outer TO container_zq
                    SET appended_flag TO true
                END IF
                SET idx_inner TO idx_inner + 1
            END WHILE
            SET idx_outer TO idx_outer + 1
        END WHILE

        RETURN container_zq
    END FUNCTION
END CLASS
