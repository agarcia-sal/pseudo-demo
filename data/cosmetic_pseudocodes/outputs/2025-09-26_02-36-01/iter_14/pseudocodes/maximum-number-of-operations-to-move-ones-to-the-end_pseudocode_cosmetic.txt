CLASS Solution  
    FUNCTION maxOperations(s)  
        SET result TO 0  
        SET accumulator TO 0  
        SET iterator TO 0  
        WHILE iterator < LENGTH(s) DO  
            IF NOT (s[iterator] != '1') THEN  
                accumulator ← accumulator + 1  
            ELSE  
                IF iterator > 0 AND s[iterator - 1] = '1' THEN  
                    result ← result + accumulator  
                END IF  
            END IF  
            iterator ← iterator + 1  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
