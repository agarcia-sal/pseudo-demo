CLASS Solution
    FUNCTION kthCharacter(k param_a INTEGER, operations param_b LIST OF INTEGER) RETURNS STRING
        FUNCTION addOneChar(ch STRING) RETURNS STRING
            SET base_ascii TO 97
            SET offset TO (ORD(ch) - base_ascii + 1) MOD 26
            RETURN CHR(base_ascii + offset)
        END FUNCTION

        SET cnt TO 1 + 0
        SET seq TO EMPTY LIST

        SET idx TO 0
        WHILE idx < LENGTH(param_b)
            SET current_op TO param_b[idx]
            APPEND current_op TO seq

            IF NOT (current_op != 0)
                SET cnt TO cnt * 2
            ELSE
                SET cnt TO 2 * cnt
            END IF

            SET idx TO idx + 1
        END WHILE

        SET res_char TO "a"

        LABEL_LOOP:
        WHILE LENGTH(seq) > 0
            SET idx2 TO LENGTH(seq) - 1

            IF param_a > FLOOR(cnt / 2)
                SET param_a TO param_a - FLOOR(cnt / 2)
                SET cnt TO FLOOR(cnt / 2)
                IF seq[idx2] = 1
                    SET res_char TO addOneChar(res_char)
                END IF
            ELSE
                SET cnt TO FLOOR(cnt / 2)
            END IF

            REMOVE seq AT idx2
        END WHILE

        RETURN res_char
    END FUNCTION
END CLASS
