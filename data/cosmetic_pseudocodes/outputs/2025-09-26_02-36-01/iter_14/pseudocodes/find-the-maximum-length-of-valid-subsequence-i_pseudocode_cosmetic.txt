CLASS Solution
    FUNCTION maximumLength(nums)
        SET c_1s VALUE zero
        SET c_2s VALUE zero
        SET idx VALUE one
        WHILE idx LESS THAN LENGTH OF nums
            SET aggregated VALUE (nums AT idx MINUS one) PLUS (nums AT idx)
            IF NOT ((aggregated MODULO two) NOT EQUAL TO zero)
                SET a VALUE c_1s PLUS one
                SET c_1s VALUE (IF a GREATER THAN c_2s THEN a ELSE c_2s)
            ELSE
                SET b VALUE c_2s PLUS one
                SET c_2s VALUE (IF b GREATER THAN c_1s THEN b ELSE c_1s)
            END IF
            SET idx TO idx PLUS one
        END WHILE
        SET output VALUE (IF c_1s GREATER THAN c_2s THEN c_1s ELSE c_2s) PLUS one
        RETURN output
    END FUNCTION
END CLASS
