CLASS Solution  
    FUNCTION clearStars(s)  
        SET accumulator TO ""  
        SET counter TO 0  
        FUNCTION buildResult(lst)  
            IF counter = LENGTH(lst)  
                RETURN ""  
            ELSE  
                RETURN lst[counter] + buildResult(lst) WITH counter INCREMENTED BY 1  
            END IF  
        END FUNCTION  
        SET temp_storage TO an empty list  
        SET index TO 0  
        WHILE index < LENGTH(s) DO  
            SET current_char TO s[index]  
            IF NOT (current_char != "*") THEN  
                IF LENGTH(temp_storage) > 0 THEN  
                    REMOVE the element at POSITION LENGTH(temp_storage) - 1 FROM temp_storage  
                END IF  
            ELSE  
                APPEND current_char TO temp_storage  
            END IF  
            SET index TO index + 1  
        END WHILE  
        SET ptr TO 0  
        FUNCTION concatAll()  
            IF ptr = LENGTH(temp_storage) THEN  
                RETURN ""  
            ELSE  
                RETURN temp_storage[ptr] + concatAll() WITH ptr INCREMENTED BY 1  
            END IF  
        END FUNCTION  
        RETURN concatAll()  
    END FUNCTION  
END CLASS
