CLASS Solution
    FUNCTION maxScore(nums)
        SET k TO LENGTH(nums)
        SET cache TO array WITH k elements ALL EQUAL TO 0
        SET a TO k - 2
        WHILE a >= 0
            SET highest TO 0
            SET b TO a + 1
            REPEAT
                SET current_score TO (b - a) * nums[b]
                IF NOT (highest >= current_score + cache[b])
                    SET highest TO current_score + cache[b]
                END IF
                SET b TO b + 1
            UNTIL b > k - 1
            SET cache[a] TO highest
            SET a TO a - 1
        END WHILE
        SET result TO cache[0]
        RETURN result
    END FUNCTION
END CLASS
