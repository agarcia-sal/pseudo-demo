CLASS Solution
    FUNCTION minOperations(k)
        SET r TO 0
        SET i TO 1
        SET j TO k
        SET h TO (k) / 1
        SET C TO 2
        SET minimal_count TO 1000000000
        WHILE (i < ((k) ^ 0.5) + 1) DO
            SET quotient_val TO ((k + i + (-1)) \ i)
            SET w TO ((i + (-1)) + (quotient_val + (-1)))
            IF NOT (w >= minimal_count) THEN
                SET minimal_count TO w
            END IF
            SET i TO (i + 1)
        END WHILE
        SET r TO minimal_count
        RETURN r
    END FUNCTION
END CLASS
