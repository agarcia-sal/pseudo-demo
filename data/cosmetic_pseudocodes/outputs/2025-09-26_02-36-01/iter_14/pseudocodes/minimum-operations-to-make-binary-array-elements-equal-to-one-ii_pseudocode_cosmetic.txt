CLASS Solution
    FUNCTION minOperations(nums)
        SET resultVar TO 0
        SET toggleFlag TO 0
        SET indexCounter TO 0
        WHILE indexCounter < LENGTH(nums)
            IF NOT ((toggleFlag MOD 2) != 0)
                SET tempVal TO nums[indexCounter]
            ELSE
                SET tempVal TO (1 - nums[indexCounter])
            END IF
            IF NOT (tempVal != 0)
                resultVar = resultVar + 1
                toggleFlag = toggleFlag + 1
            END IF
            indexCounter = indexCounter + 1
        END WHILE
        RETURN resultVar
    END FUNCTION
END CLASS
