CLASS Solution
    FUNCTION minEnd(n a)
        FUNCTION canConstruct(z)
            SET r TO a
            SET q TO 1
            REPEAT
                IF (r BITWISE_AND a) NOT_EQUAL a THEN
                    SET q TO q
                ELSE
                    SET q TO q + 1
                    IF q EQUALS a THEN
                        RETURN True
                    ELSE
                        SET q TO q
                    END IF
                END IF
                IF r GREATER_OR_EQUAL z THEN
                    BREAK
                END IF
                SET r TO r + 1
            UNTIL False
            RETURN q EQUALS a
        END FUNCTION

        SET m TO a
        SET p TO 2 * 10 * 10 * 10 * 10 * 10 * 10 * 10 * 10

        WHILE m LESS_THAN p
            SET s TO (m + p) DIVIDED_BY 2
            IF canConstruct(s)
                SET p TO s
            ELSE
                SET m TO m + 1
            END IF
        END WHILE

        RETURN m
    END FUNCTION
END CLASS
