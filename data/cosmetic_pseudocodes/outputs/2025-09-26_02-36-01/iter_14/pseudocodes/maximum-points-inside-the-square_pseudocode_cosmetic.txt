CLASS Solution
    FUNCTION maxPointsInsideSquare(points s)
        SET total_points TO LENGTH OF s
        SET highest_points TO 0
        SET idx_outer TO 0
        WHILE idx_outer < total_points
            SET coordinate_a TO s[idx_outer][0]
            SET coordinate_b TO s[idx_outer][1]
            IF coordinate_a >= 0 THEN
                SET abs_a TO coordinate_a
            ELSE
                SET abs_a TO 0 - coordinate_a
            END IF
            IF coordinate_b >= 0 THEN
                SET abs_b TO coordinate_b
            ELSE
                SET abs_b TO 0 - coordinate_b
            END IF
            IF abs_a > abs_b THEN
                SET length_side TO abs_a
            ELSE
                SET length_side TO abs_b
            END IF
            SET markers_seen TO EMPTY DICTIONARY
            SET is_square_valid TO TRUE
            SET idx_inner TO 0
            WHILE idx_inner < total_points AND is_square_valid
                SET curr_x TO s[idx_inner][0]
                SET curr_y TO s[idx_inner][1]
                IF NOT (curr_x > length_side OR curr_x < (0 - length_side) OR curr_y > length_side OR curr_y < (0 - length_side)) THEN
                    SET current_tag TO s[idx_inner]
                    IF current_tag IN markers_seen THEN
                        SET is_square_valid TO FALSE
                    ELSE
                        markers_seen[current_tag] = TRUE
                    END IF
                END IF
                SET idx_inner TO idx_inner + 1
            END WHILE
            IF is_square_valid = TRUE THEN
                IF LENGTH OF markers_seen > highest_points THEN
                    SET highest_points TO LENGTH OF markers_seen
                END IF
            END IF
            SET idx_outer TO idx_outer + 1
        END WHILE
        RETURN highest_points
    END FUNCTION
END CLASS
