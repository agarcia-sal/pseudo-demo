CLASS Solution
    FUNCTION longestCommonPrefix(arr1 arr1 OF integers arr2 arr2 OF integers) RETURNS integer
        DEFINE FUNCTION convertToStr(value integer) RETURNS string
            RETURN CONCATENATE("" , value)
        END FUNCTION

        SET SetA TO EMPTY SET
        FOR INDEX idx1 = 0 TO (SIZE OF arr1) MINUS 1
            SET tempStr TO convertToStr(arr1[idx1])
            SET pos TO 1
            REPEAT
                ADD SUBSTRING(tempStr, 1, pos) TO SetA
                SET pos TO pos + 1
            UNTIL pos GREATER THAN LENGTH(tempStr)
        END FOR

        SET SetB TO EMPTY SET
        SET pointer TO 0
        WHILE pointer LESS THAN SIZE OF arr2
            SET currentVal TO arr2[pointer]
            SET strVal TO convertToStr(currentVal)
            SET counter TO 1
            REPEAT
                ADD SUBSTRING(strVal, 1, counter) TO SetB
                SET counter TO counter + 1
            UNTIL counter GREATER THAN LENGTH(strVal)
            SET pointer TO pointer + 1
        END WHILE

        SET maxLen TO 0
        SET iter TO SETA_ITERATOR(SetA)
        LOOP
            SET maybePrefix TO ITERATOR_NEXT(iter)
            IF maybePrefix EQUALS NULL THEN
                BREAK
            END IF
            IF NOT (maybePrefix IN SetB) THEN
                CONTINUE
            ELSE
                IF LENGTH(maybePrefix) GREATER THAN maxLen THEN
                    SET maxLen TO LENGTH(maybePrefix)
                END IF
            END IF
        END LOOP

        RETURN maxLen
    END FUNCTION
END CLASS
