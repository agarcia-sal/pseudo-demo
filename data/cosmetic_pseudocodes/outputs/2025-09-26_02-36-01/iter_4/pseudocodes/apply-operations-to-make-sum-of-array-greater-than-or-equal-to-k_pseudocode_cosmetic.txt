CLASS Solution  
    FUNCTION minOperations(k)  
        DECLARE minimalCount AS a very large number  
        DECLARE currentIndex AS 1  
        DECLARE limit AS FLOOR(SQRT(k)) + 1  
        WHILE currentIndex <= limit DO  
            DECLARE dividend AS k + currentIndex - 1  
            DECLARE quotient AS INTEGER_DIVIDE(dividend, currentIndex)  
            DECLARE currentOps AS (currentIndex - 1) + (quotient - 1)  
            IF NOT (currentOps >= minimalCount) THEN  
                SET minimalCount TO currentOps  
            END IF  
            INCREMENT currentIndex BY one  
        END WHILE  
        RETURN minimalCount  
    END FUNCTION  
END CLASS
