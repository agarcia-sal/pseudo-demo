CLASS Solution  
    FUNCTION minMoves(rooks)  
        LET total_rows TO 0  
        LET total_cols TO 0  
        LET length_rooks TO 0  
        LET idx TO 0  
        LET sorted_by_row TO empty list  
        LET sorted_by_col TO empty list  
        
        length_rooks = 0  
        WHILE length_rooks NOT EQUAL TO LENGTH(rooks)  
            INCREMENT length_rooks BY 1  
        END WHILE  
        
        sorted_by_row = rooks  
        sorted_by_col = rooks  
        
        LET i TO 1  
        WHILE i LESS THAN length_rooks  
            LET j TO 0  
            WHILE j LESS THAN length_rooks MINUS i  
                IF sorted_by_row[j][0] GREATER THAN sorted_by_row[j + 1][0] THEN  
                    LET temp TO sorted_by_row[j]  
                    SET sorted_by_row[j] TO sorted_by_row[j + 1]  
                    SET sorted_by_row[j + 1] TO temp  
                END IF  
                j = j + 1  
            END WHILE  
            i = i + 1  
        END WHILE  
        
        LET k TO 1  
        WHILE k LESS THAN length_rooks  
            LET l TO 0  
            WHILE l LESS THAN length_rooks MINUS k  
                IF sorted_by_col[l][1] GREATER THAN sorted_by_col[l + 1][1] THEN  
                    LET temp2 TO sorted_by_col[l]  
                    SET sorted_by_col[l] TO sorted_by_col[l + 1]  
                    SET sorted_by_col[l + 1] TO temp2  
                END IF  
                l = l + 1  
            END WHILE  
            k = k + 1  
        END WHILE  
        
        idx = 0  
        WHILE idx LESS THAN length_rooks  
            LET delta_row TO sorted_by_row[idx][0] MINUS idx  
            IF delta_row LESS THAN 0 THEN  
                delta_row = -delta_row  
            END IF  
            total_rows = total_rows + delta_row  
            idx = idx + 1  
        END WHILE  
        
        idx = 0  
        WHILE idx LESS THAN length_rooks  
            LET delta_col TO sorted_by_col[idx][1] MINUS idx  
            IF delta_col LESS THAN 0 THEN  
                delta_col = 0 - delta_col  
            END IF  
            total_cols = total_cols + delta_col  
            idx = idx + 1  
        END WHILE  
        
        LET result TO total_rows + total_cols  
        RETURN result  
    END FUNCTION  
END CLASS
