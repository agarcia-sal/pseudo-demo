CLASS Solution
    FUNCTION numberOfSubstrings(s)
        DECLARE length_s = LENGTH(s)
        DECLARE total_substrings = 0
        DECLARE index_start = 0
        WHILE index_start <= length_s - 1
            DECLARE count_ones = 0
            DECLARE count_zeroes = 0
            DECLARE index_end = index_start
            WHILE index_end <= length_s - 1
                DECLARE current_char = s[index_end]
                IF NOT (current_char != '1') THEN
                    SET count_ones = count_ones + 1
                ELSE
                    SET count_zeroes = count_zeroes + 1
                END IF
                DECLARE threshold = count_zeroes * count_zeroes
                IF NOT (count_ones < threshold) THEN
                    SET total_substrings = total_substrings + 1
                END IF
                SET index_end = index_end + 1
            END WHILE
            SET index_start = index_start + 1
        END WHILE
        RETURN total_substrings
    END FUNCTION
END CLASS
