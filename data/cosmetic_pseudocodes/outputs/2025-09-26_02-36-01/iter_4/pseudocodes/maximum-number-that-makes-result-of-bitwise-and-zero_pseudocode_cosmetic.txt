CLASS Solution
    FUNCTION maxNumber(n)
        DECLARE result
        IF NOT (n NOT EQUAL TO 1) THEN
            SET result AS 0
            RETURN result
        ENDIF

        LET bit_marker BE 1
        WHILE TRUE
            IF bit_marker > n THEN
                EXIT LOOP
            ENDIF
            LET temp_value BE bit_marker * 2
            SET bit_marker TO temp_value
        ENDWHILE

        LET adjusted_bit BE bit_marker / 2
        LET temp_result BE adjusted_bit - 1
        RETURN temp_result
    END FUNCTION
END CLASS
