CLASS Solution  
    FUNCTION valueAfterKSeconds(n k)  
        DECLARE CONSTANT MODULO_VALUE = (1000000000 + 7)  
        DECLARE LIST numbers WITH LENGTH n FILLED WITH 1  
        
        DECLARE counter_outer = 0  
        WHILE counter_outer < k DO  
            DECLARE counter_inner = 1  
            WHILE counter_inner < n DO  
                DECLARE previous_value = numbers[counter_inner - 1]  
                DECLARE current_value = numbers[counter_inner]  
                DECLARE updated_value = (current_value + previous_value) % MODULO_VALUE  
                numbers[counter_inner] = updated_value  
                SET counter_inner TO counter_inner + 1  
            END WHILE  
            SET counter_outer TO counter_outer + 1  
        END WHILE  
        
        RETURN numbers[n - 1]  
    END FUNCTION  
END CLASS
