CLASS Solution
    FUNCTION sumDigitDifferences(nums)
        FUNCTION digit_difference(num1, num2)
            DECLARE diff_counter = 0
            DECLARE pos = 0
            DECLARE limit = LENGTH OF num1
            WHILE pos < limit DO
                DECLARE d_one = CHARACTER AT pos IN num1
                DECLARE d_two = CHARACTER AT pos IN num2
                IF NOT (d_one = d_two) THEN
                    SET diff_counter TO diff_counter + 1
                END IF
                SET pos TO pos + 1
            END WHILE
            RETURN diff_counter
        END FUNCTION

        DECLARE aggregate = 0
        DECLARE length_of_nums = LENGTH OF nums
        DECLARE outer = 0
        WHILE outer < length_of_nums - 1 DO
            DECLARE inner = outer + 1
            WHILE inner < length_of_nums DO
                DECLARE current_diff = digit_difference(nums[outer], nums[inner])
                SET aggregate TO aggregate + current_diff
                SET inner TO inner + 1
            END WHILE
            SET outer TO outer + 1
        END WHILE
        RETURN aggregate
    END FUNCTION
END CLASS
