CLASS Solution  
    FUNCTION countSubarrays(nums parameter LIST OF INTEGER k parameter INTEGER) RETURNS INTEGER  
        DECLARE totalSubarrays AS zero  
        LET lengthNums BE THE LENGTH OF nums  
        DECLARE i AS zero  
        WHILE i IS LESS OR EQUAL TO lengthNums MINUS one DO  
            DECLARE andAccumulator AS nums AT INDEX i  
            DECLARE j AS i  
            WHILE j IS LESS OR EQUAL TO lengthNums MINUS one DO  
                LET andAccumulator TO andAccumulator BITWISE-AND nums AT INDEX j  
                IF andAccumulator EQUALS k THEN  
                    LET totalSubarrays TO totalSubarrays PLUS one  
                END IF  
                IF NOT (andAccumulator GREATER OR EQUAL TO k) THEN  
                    EXIT WHILE  
                END IF  
                LET j TO j PLUS one  
            END WHILE  
            LET i TO i PLUS one  
        END WHILE  
        RETURN totalSubarrays  
    END FUNCTION  
END CLASS
