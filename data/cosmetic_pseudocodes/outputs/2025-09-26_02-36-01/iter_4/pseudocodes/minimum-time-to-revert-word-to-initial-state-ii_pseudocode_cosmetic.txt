CLASS Solution  
    FUNCTION minimumTimeToInitialState(word parameter word AND integer parameter k) RETURNS integer  
        DECLARE length_of_word AS INTEGER  
        ASSIGN length_of_word WITH SIZE OF word  
        DECLARE elapsed AS INTEGER  
        ASSIGN elapsed TO 1  
        LOOP WHILE TRUE  
            DECLARE segment AS STRING  
            ASSIGN segment TO SLICE_OF word FROM (elapsed TIMES k) THROUGH END  
            DECLARE starts_match AS BOOLEAN  
            ASSIGN starts_match TO (word BEGINS WITH segment)  
            IF starts_match EQUALS TRUE THEN  
                RETURN elapsed  
            END IF  
            SET elapsed TO elapsed PLUS 1  
        END LOOP  
    END FUNCTION  
END CLASS
