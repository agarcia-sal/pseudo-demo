CLASS Solution  
    FUNCTION minOperationsToMakeMedianK(nums k)  
        CALL nums SORTED_ASCENDING  
        DECLARE length_of_list = SIZE OF nums  
        DECLARE mid_pos = length_of_list INTEGER_DIV 2  
        
        IF NOT (nums[mid_pos] NOT EQUAL TO k) THEN  
            RETURN 0  
        END IF  
        
        DECLARE total_operations TO 0  
        
        IF nums[mid_pos] < k THEN  
            DECLARE idx = mid_pos  
            WHILE (idx < length_of_list) AND (nums[idx] < k) DO  
                DECLARE difference = k - nums[idx]  
                total_operations = total_operations + difference  
                idx = idx + 1  
            END WHILE  
        ELSE  
            DECLARE j = mid_pos  
            WHILE (j >= 0) AND (nums[j] > k) DO  
                DECLARE diff_val = nums[j] - k  
                total_operations = total_operations + diff_val  
                j = j - 1  
            END WHILE  
        END IF  
        
        RETURN total_operations  
    END FUNCTION  
END CLASS
