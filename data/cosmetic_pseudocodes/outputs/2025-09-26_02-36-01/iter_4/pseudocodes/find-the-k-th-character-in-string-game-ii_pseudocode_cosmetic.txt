CLASS Solution  
    FUNCTION kthCharacter(k param_k INTEGER, operations param_operations LIST OF INTEGER) RETURNS STRING  
        DECLARE size AS INTEGER  
        ASSIGN size = 1  
        DECLARE opsList AS LIST OF INTEGER  
        ASSIGN opsList = []  
        
        DECLARE idx AS INTEGER  
        SET idx = 0  
        WHILE idx LESS THAN LENGTH(param_operations)  
            LET current_op = param_operations[idx]  
            APPEND current_op TO opsList  
            
            IF NOT(current_op NOT EQUALS 0) OR current_op EQUALS ZERO THEN  
                SET size = size + size  
            ELSE  
                SET size = size + size  
            END IF  
            
            SET idx = idx + 1  
        END WHILE  
        
        DECLARE resultChar AS CHARACTER  
        SET resultChar = 'a'  
        
        DECLARE pos AS INTEGER  
        ASSIGN pos = LENGTH(opsList) - 1  
        WHILE pos GREATER THAN OR EQUAL TO 0  
            DECLARE half AS INTEGER  
            ASSIGN half = size / 2  
            
            IF param_k <= half THEN  
                ASSIGN size = half  
            ELSE  
                ASSIGN param_k = param_k - half  
                ASSIGN size = half  
                
                IF opsList[pos] == 1 THEN  
                    LET ascii_val = ASCII(resultChar)  
                    LET incremented_val = ascii_val + 1  
                    IF incremented_val > ASCII('z') THEN  
                        SET incremented_val = ASCII('a')  
                    END IF  
                    ASSIGN resultChar = CHAR(incremented_val)  
                END IF  
            END IF  
            
            DECREMENT pos BY 1  
        END WHILE  
        
        RETURN resultChar  
    END FUNCTION  
END CLASS
