CLASS Solution  
    FUNCTION minEnd(n x)  
        FUNCTION canConstruct(maxAllowed)  
            DECLARE tempVar AS x  
            DECLARE tally AS 1  
            WHILE tempVar IS LESS THAN maxAllowed DO  
                tempVar = tempVar + 1  
                IF (tempVar BITWISE_AND x) == x THEN  
                    tally = tally + 1  
                    IF tally == n THEN  
                        RETURN True  
                    END IF  
                END IF  
            END WHILE  
            RETURN (tally == n)  
        END FUNCTION  
        
        SET lowerBound TO x  
        DECLARE upperBound AS 2 * 10 * 10 * 10 * 10 * 10 * 10 * 10 * 10  
        WHILE lowerBound < upperBound DO  
            DECLARE middleValue AS (lowerBound + upperBound) DIV 2  
            IF canConstruct(middleValue) THEN  
                upperBound = middleValue  
            ELSE  
                lowerBound = lowerBound + 1  
            END IF  
        END WHILE  
        
        RETURN lowerBound  
    END FUNCTION  
END CLASS
