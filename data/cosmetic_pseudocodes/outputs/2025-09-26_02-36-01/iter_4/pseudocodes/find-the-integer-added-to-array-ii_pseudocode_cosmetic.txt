CLASS Solution  
    FUNCTION minimumAddedInteger(nums1 RETURNS List OF integer, nums2 RETURNS List OF integer) RETURNS integer  
        CALL nums1.sort()  
        CALL nums2.sort()  
        
        DECLARE outerIndex = 0  
        WHILE outerIndex LESS THAN LENGTH(nums1)  
            DECLARE innerIndex = outerIndex PLUS 1  
            WHILE innerIndex LESS THAN LENGTH(nums1)  
                DECLARE filteredNums AS an empty List OF integer  
                
                DECLARE idx = 0  
                WHILE idx LESS THAN outerIndex  
                    APPEND nums1[idx] TO filteredNums  
                    INCREMENT idx BY 1  
                END WHILE  
                
                DECLARE idx = outerIndex PLUS 1  
                WHILE idx LESS THAN innerIndex  
                    APPEND nums1[idx] TO filteredNums  
                    idx = idx PLUS 1  
                END WHILE  
                
                DECLARE idx = innerIndex PLUS 1  
                WHILE idx LESS THAN LENGTH(nums1)  
                    APPEND nums1[idx] TO filteredNums  
                    idx = idx PLUS 1  
                END WHILE  
                
                LET differenceValue BE nums2[0] MINUS filteredNums[0]  
                
                LET isValidMatch BE TRUE  
                DECLARE position = 0  
                LOOP  
                    IF NOT (position LESS THAN LENGTH(nums2)) THEN EXIT LOOP  
                    IF filteredNums[position] PLUS differenceValue IS NOT EQUAL TO nums2[position] THEN  
                        SET isValidMatch TO FALSE  
                        EXIT LOOP  
                    END IF  
                    position = position PLUS 1  
                END LOOP  
                
                IF isValidMatch IS TRUE THEN  
                    RETURN differenceValue  
                END IF  
                
                innerIndex = innerIndex PLUS 1  
            END WHILE  
            outerIndex = outerIndex PLUS 1  
        END WHILE  
        
        RETURN None  
    END FUNCTION  
END CLASS
