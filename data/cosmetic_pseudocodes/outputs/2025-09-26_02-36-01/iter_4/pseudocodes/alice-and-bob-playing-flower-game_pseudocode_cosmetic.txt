CLASS Solution  
    FUNCTION flowerGame(n, m)  
        DECLARE half_ceil_n AS (n + 1) DIV 2  
        DECLARE half_floor_n AS n DIV 2  
        DECLARE half_ceil_m AS (m + 1) DIV 2  
        DECLARE half_floor_m AS m DIV 2  

        DECLARE product_one AS half_ceil_n TIMES half_floor_m  
        DECLARE product_two AS half_floor_n TIMES half_ceil_m  

        DECLARE sum_pairs AS product_one PLUS product_two  

        GIVE BACK sum_pairs  
    END FUNCTION  
END CLASS
