CLASS Solution  
    FUNCTION nonSpecialCount(l r)  
        FUNCTION is_prime(value)  
            IF NOT (value GREATER THAN one)  
                RETURN false  
            END IF  
            IF value LESS THAN OR EQUAL TO three  
                RETURN true  
            END IF  
            LET check_two AS (value MODULO two)  
            LET check_three AS (value MODULO three)  
            IF (check_two EQUALS zero) OR (check_three EQUALS zero)  
                RETURN false  
            END IF  
            DECLARE index = 5  
            WHILE (index * index) LESS THAN OR EQUAL TO value  
                LET mod_i = (value MODULO index)  
                LET mod_i_plus_two = (value MODULO (index + two))  
                IF (mod_i EQUALS zero) OR (mod_i_plus_two EQUALS zero)  
                    RETURN false  
                END IF  
                SET index TO index + six  
            END WHILE  
            RETURN true  
        END FUNCTION  
  
        LET lower_bound TO CEIL OF SQUARE ROOT OF l  
        LET upper_bound TO FLOOR OF SQUARE ROOT OF r  
  
        DECLARE prime_counter TO 0  
  
        SET current_num TO lower_bound  
        WHILE current_num LESS THAN OR EQUAL TO upper_bound  
            IF is_prime(current_num) IS true  
                SET prime_counter TO prime_counter + 1  
            END IF  
            SET current_num TO current_num + 1  
        END WHILE  
  
        LET range_span TO (r - l) + 1  
  
        LET result TO range_span - prime_counter  
  
        RETURN result  
    END FUNCTION  
END CLASS
