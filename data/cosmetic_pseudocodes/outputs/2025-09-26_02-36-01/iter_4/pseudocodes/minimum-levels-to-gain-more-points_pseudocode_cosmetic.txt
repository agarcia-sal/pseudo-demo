CLASS Solution
    FUNCTION minimumLevels(possible)
        DECLARE sumScores BE 0
        SET iterator TO 0
        WHILE iterator LESS THAN LENGTH OF possible
            DECLARE currentVal AS possible[iterator]
            LET incrementValue BE (currentVal TIMES 2) MINUS 1
            SET sumScores TO sumScores PLUS incrementValue
            SET iterator TO iterator PLUS 1
        END WHILE

        LET aliceSum BE 0
        LET position BE 0
        WHILE position LESS THAN LENGTH OF possible MINUS 2 PLUS 1
            LET val AT idx BE possible[position]
            LET incValue BE (val AT idx TIMES 2) MINUS 1
            aliceSum <- aliceSum PLUS incValue
            sumScores <- sumScores MINUS incValue

            IF aliceSum GREATER THAN sumScores THEN
                RETURN position PLUS 1
            END IF
            position <- position PLUS 1
        END WHILE

        RETURN -1
    END FUNCTION
END CLASS
