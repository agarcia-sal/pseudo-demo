CLASS Solution
    FUNCTION countCompleteDayPairs WITH parameter hours
        DECLARE remainderTracker MAPPING defaulting to zero
        DECLARE resultCount AS integer ASSIGNED zero
        DECLARE idx AS integer ASSIGNED zero
        WHILE idx LESS THAN LENGTH OF hours
            DECLARE currentHour AS integer ASSIGNED hours AT idx
            DECLARE modValue AS integer ASSIGNED currentHour MODULO 24
            DECLARE neededComplement AS integer ASSIGNED (24 - modValue) MODULO 24
            SET resultCount TO resultCount PLUS remainderTracker AT neededComplement
            SET remainderTracker AT modValue TO remainderTracker AT modValue PLUS one
            INCREMENT idx BY one
        END WHILE
        RETURN resultCount
    END FUNCTION
END CLASS
