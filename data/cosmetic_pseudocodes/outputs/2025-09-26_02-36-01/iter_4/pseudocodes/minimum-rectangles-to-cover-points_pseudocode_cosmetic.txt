CLASS Solution  
    FUNCTION minRectanglesToCoverPoints(points, w)  
        DECLARE sortedPoints AS points SORTED BY ascending order OF first coordinate  
        DECLARE countRectangles AS 0  
        DECLARE boundaryX AS negative one  
        DECLARE index AS 0  
        WHILE index < LENGTH(sortedPoints) DO  
            LET coordinatePair BE sortedPoints[index]  
            LET posX BE coordinatePair[0]  
            LET posY BE coordinatePair[1]  
            IF NOT (posX <= boundaryX) THEN  
                LET boundaryX TO posX + w  
                countRectangles = countRectangles + 1  
            END IF  
            index = index + 1  
        END WHILE  
        RETURN countRectangles  
    END FUNCTION  
END CLASS
