CLASS Solution
    FUNCTION numberOfRightTriangles(grid)
        LET total_triangles ASSIGN 0
        LET max_row ASSIGN LENGTH OF grid
        LET max_col ASSIGN LENGTH OF grid[0]

        LET row_idx ASSIGN 0
        WHILE row_idx LESS THAN max_row
            LET col_idx ASSIGN 0
            WHILE col_idx LESS THAN max_col
                LET current_cell ASSIGN grid[row_idx][col_idx]
                IF current_cell IS EQUAL TO 1
                    LET row_sum EXCLUDE CURRENT COL ASSIGN 0
                    LET col_sum EXCLUDE CURRENT ROW ASSIGN 0

                    LET col_being_checked ASSIGN 0
                    WHILE col_being_checked LESS THAN max_col
                        IF NOT (col_being_checked IS EQUAL TO col_idx)
                            LET row_sum ASSIGN row_sum PLUS grid[row_idx][col_being_checked]
                        END IF
                        INCREMENT col_being_checked BY 1
                    END WHILE

                    LET row_being_checked ASSIGN 0
                    WHILE row_being_checked LESS THAN max_row
                        IF NOT (row_being_checked IS EQUAL TO row_idx)
                            LET col_sum ASSIGN col_sum PLUS grid[row_being_checked][col_idx]
                        END IF
                        INCREMENT row_being_checked BY 1
                    END WHILE

                    LET triangles_from_cell ASSIGN row_sum MULTIPLIED BY col_sum
                    LET total_triangles ASSIGN total_triangles PLUS triangles_from_cell
                END IF
                INCREMENT col_idx BY 1
            END WHILE
            INCREMENT row_idx BY 1
        END WHILE

        RETURN total_triangles
    END FUNCTION
END CLASS
