CLASS Solution  
    FUNCTION minimumValueSum(nums, andValues)  
        DECLARE totalNums AS LENGTH OF nums  
        DECLARE totalAnds AS LENGTH OF andValues  
        
        FUNCTION dp(i, j)  
            IF NOT (j >= 0) THEN  
                IF NOT (i >= 0) THEN  
                    RETURN 0  
                ELSE  
                    RETURN ∞  
                END IF  
            END IF  
            
            IF NOT (i >= 0) THEN  
                RETURN ∞  
            END IF  
            
            DECLARE minSum AS ∞  
            DECLARE accAnd AS -1  
            DECLARE idx AS i  
            WHILE idx >= -1 DO  
                IF accAnd = -1 THEN  
                    SET accAnd TO nums[idx]  
                ELSE  
                    SET accAnd TO (accAnd & nums[idx])  
                END IF  
                
                IF accAnd = andValues[j] THEN  
                    DECLARE tempValue AS dp(idx - 1, j - 1) + nums[i]  
                    IF tempValue < minSum THEN  
                        SET minSum TO tempValue  
                    END IF  
                END IF  
                SET idx TO idx - 1  
            END WHILE  
            
            RETURN minSum  
        END FUNCTION  
        
        DECLARE answer AS dp(totalNums - 1, totalAnds - 1)  
        IF answer ≠ ∞ THEN  
            RETURN answer  
        ELSE  
            RETURN -1  
        END IF  
    END FUNCTION  
END CLASS
