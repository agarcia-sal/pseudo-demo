CLASS Solution  
    FUNCTION maximumLength(nums)  
        DECLARE evenCount AS 0  
        DECLARE oddCount AS 0  
        DECLARE index AS 1  
        WHILE index < (LENGTH OF nums)  
            LET sumAdjacent TO (nums[index - 1]) + (nums[index])  
            IF NOT (sumAdjacent MOD 2) THEN  
                LET tempEven TO evenCount + 1  
                IF tempEven > oddCount THEN  
                    SET evenCount TO tempEven  
                ELSE  
                    SET evenCount TO oddCount  
                END IF  
            ELSE  
                LET tempOdd TO oddCount + 1  
                IF tempOdd > evenCount THEN  
                    SET oddCount TO tempOdd  
                ELSE  
                    SET oddCount TO evenCount  
                END IF  
            END IF  
            INCREMENT index BY 1  
        END WHILE  
        LET maxLengthCandidate1 TO evenCount  
        LET maxLengthCandidate2 TO oddCount + 1  
        IF maxLengthCandidate1 > maxLengthCandidate2 THEN  
            RETURN maxLengthCandidate1  
        ELSE  
            RETURN maxLengthCandidate2  
        END IF  
    END FUNCTION  
END CLASS
