CLASS Solution  
    FUNCTION minimumLength(s)  
        DECLARE frequencyMap ASSIGN Counter(s)  
        DECLARE oddCount SET TO 0  
        DECLARE evenCount SET TO 0  
        LET iterator ASSIGN VALUES(frequencyMap)  
        DECLARE index SET TO 0  
        WHILE index LESS THAN LENGTH(iterator)  
            LET currentValue ASSIGN iterator[index]  
            IF NOT(currentValue MOD 2 EQUALS 0)  
                oddCount ASSIGN oddCount PLUS 1  
            ELSE IF (currentValue MOD 2 EQUALS 0) AND NOT(currentValue EQUALS 0)  
                evenCount ASSIGN evenCount PLUS 2  
            END IF  
            INCREMENT index BY 1  
        END WHILE  
        LET totalLength ASSIGN oddCount PLUS evenCount  
        RETURN totalLength  
    END FUNCTION  
END CLASS
