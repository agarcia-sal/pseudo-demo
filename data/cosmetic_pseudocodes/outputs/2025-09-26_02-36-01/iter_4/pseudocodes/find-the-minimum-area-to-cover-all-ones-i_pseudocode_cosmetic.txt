CLASS Solution
    FUNCTION minimumArea(grid)
        IF NOT (grid != empty AND (grid[0] != empty))
            RETURN 0
        END IF

        LET total_rows BE LENGTH(grid)
        LET total_columns BE LENGTH(grid[0])

        LET upper_bound BE +∞
        LET lower_bound BE -∞
        LET left_limit BE +∞
        LET right_limit BE -∞

        LET row_index BE 0
        WHILE row_index <= (total_rows - 1)
            LET col_index BE 0
            WHILE col_index <= (total_columns - 1)
                LET current_val BE grid[row_index][col_index]
                IF current_val == 1
                    IF upper_bound > row_index THEN
                        upper_bound = row_index
                    END IF

                    IF lower_bound < row_index THEN
                        lower_bound = row_index
                    END IF

                    IF left_limit > col_index THEN
                        left_limit = col_index
                    END IF

                    IF right_limit < col_index THEN
                        right_limit = col_index
                    END IF
                END IF
                col_index = col_index + 1
            END WHILE
            row_index = row_index + 1
        END WHILE

        LET height_calc BE (lower_bound - upper_bound) + 1
        LET width_calc BE (right_limit - left_limit) + 1

        RETURN height_calc * width_calc
    END FUNCTION
END CLASS
