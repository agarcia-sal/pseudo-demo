CLASS Solution  
    FUNCTION minAnagramLength(s)  
        DECLARE distinctLetters AS empty set  
        DECLARE index AS 0  
        WHILE index < LENGTH(s)  
            LET currentChar ASSIGN s[index]  
            IF NOT (currentChar IN distinctLetters) THEN  
                INSERT currentChar INTO distinctLetters  
            END IF  
            INCREMENT index BY 1  
        END WHILE  
        RETURN SIZE(distinctLetters)  
    END FUNCTION  
END CLASS
