CLASS Solution  
    FUNCTION minOperations(nums parameter k parameter)  
        EXECUTE nums.heapify()  
        INITIALIZE count TO 0  
        WHILE (NOT (nums[0] >= k)) AND (SIZE OF nums > 1)  
            ASSIGN first_element FROM nums.heappop()  
            ASSIGN second_element FROM nums.heappop()  
            ASSIGN smaller_value TO (IF first_element < second_element THEN first_element ELSE second_element)  
            ASSIGN larger_value TO (IF first_element > second_element THEN first_element ELSE second_element)  
            ASSIGN combined_value TO (smaller_value TIMES 2) PLUS larger_value  
            EXECUTE nums.heappush(combined_value)  
            ADD 1 TO count  
        END WHILE  
        RETURN count  
    END FUNCTION  
END CLASS
