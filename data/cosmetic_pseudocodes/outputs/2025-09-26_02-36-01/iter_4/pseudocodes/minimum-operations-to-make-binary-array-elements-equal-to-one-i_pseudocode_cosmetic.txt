CLASS Solution  
    FUNCTION minOperations(nums)  
        DECLARE length AS LENGTH OF nums  
        DECLARE countOps AS 0  
        DECLARE index AS 0  
        WHILE index LESS THAN length MINUS 2 DO  
            IF NOT (nums[index] NOT EQUAL TO 0) THEN  
                LET firstFlip BE 1 MINUS nums[index]  
                LET secondFlip BE 1 MINUS nums[index PLUS 1]  
                LET thirdFlip BE 1 MINUS nums[index PLUS 2]  
                SET nums[index] TO firstFlip  
                SET nums[index PLUS 1] TO secondFlip  
                SET nums[index PLUS 2] TO thirdFlip  
                SET countOps TO countOps PLUS 1  
            END IF  
            SET index TO index PLUS 1  
        END WHILE  
        IF (nums[length MINUS 1] EQUALS 0) OR (nums[length MINUS 2] EQUALS 0) THEN  
            RETURN -1  
        ELSE  
            RETURN countOps  
        END IF  
    END FUNCTION  
END CLASS
