CLASS Solution
    FUNCTION maximumStrength(nums k)
        LET length_nums BE THE COUNT OF nums
        DECLARE dp AS A TWO-DIMENSIONAL ARRAY WITH (length_nums PLUS 1) ROWS AND (k PLUS 1) COLUMNS
        FOR x FROM 0 TO length_nums DO
            FOR y FROM 0 TO k DO
                SET dp[x][y] TO NEGATIVE_INFINITY
            END FOR
        END FOR
        SET dp[0][0] TO 0

        LET index_i BE 1
        WHILE index_i <= length_nums DO
            LET index_j BE 1
            WHILE index_j <= k DO
                LET temp_sum BE 0
                LET end_idx BE index_i
                WHILE end_idx >= 1 DO
                    LET temp_sum TO temp_sum PLUS nums[end_idx MINUS 1]
                    IF (index_j MOD 2) NOT EQUAL TO 0 THEN
                        LET sign_multiplier BE k MINUS index_j MINUS 1 PLUS 1
                    ELSE
                        LET sign_multiplier BE NEGATIVE (k MINUS index_j MINUS 1 PLUS 1)
                    END IF
                    LET candidate_value BE dp[end_idx MINUS 1][index_j MINUS 1] PLUS sign_multiplier TIMES temp_sum
                    IF dp[index_i][index_j] < candidate_value THEN
                        SET dp[index_i][index_j] TO candidate_value
                    END IF
                    DECREMENT end_idx BY 1
                END WHILE
                LET dp[index_i][index_j] BE THE GREATER OF dp[index_i][index_j] AND dp[index_i MINUS 1][index_j]
                INCREMENT index_j BY 1
            END WHILE
            INCREMENT index_i BY 1
        END WHILE

        RETURN dp[length_nums][k]
    END FUNCTION
END CLASS
