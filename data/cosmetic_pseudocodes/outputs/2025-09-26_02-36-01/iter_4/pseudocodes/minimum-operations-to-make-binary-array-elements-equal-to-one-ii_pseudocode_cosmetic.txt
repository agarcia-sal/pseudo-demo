CLASS Solution  
    FUNCTION minOperations(nums)  
        DECLARE ops = 0  
        DECLARE toggle = 0  
        DECLARE idx = 0  
        WHILE idx < LENGTH(nums)  
            DECLARE val  
            IF (toggle MOD 2) != 1 THEN  
                SET val TO nums[idx]  
            ELSE  
                SET val TO 1 - nums[idx]  
            END IF  
            IF val == 0 THEN  
                LET ops TO ops + 1  
                LET toggle TO toggle + 1  
            END IF  
            LET idx TO idx + 1  
        END WHILE  
        RETURN ops  
    END FUNCTION  
END CLASS
