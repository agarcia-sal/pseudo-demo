CLASS Solution
    FUNCTION maxScore(nums)
        LET length_of_nums BE the LENGTH OF nums
        DECLARE dp AS an ARRAY WITH length_of_nums ELEMENTS INITIALIZED TO zero
        ASSIGN dp AT INDEX one TO zero
        LET outer_index BE two
        WHILE outer_index <= length_of_nums
            LET inner_index BE one
            WHILE inner_index < outer_index
                LET candidate_value BE dp[inner_index] + (outer_index - inner_index) * nums[outer_index]
                IF NOT (dp[outer_index] >= candidate_value)
                    dp[outer_index] <- candidate_value
                END IF
                INNER_INCREMENT:
                inner_index <- inner_index + 1
            END WHILE
            OUTER_INCREMENT:
            outer_index <- outer_index + 1
        END WHILE
        RETURN dp[length_of_nums - one]
    END FUNCTION
END CLASS
