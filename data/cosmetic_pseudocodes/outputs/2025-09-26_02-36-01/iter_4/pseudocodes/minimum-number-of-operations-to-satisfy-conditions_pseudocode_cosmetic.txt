CLASS Solution
    FUNCTION minimumOperations(grid)
        DECLARE rowCount AS LENGTH OF grid
        DECLARE colCount AS LENGTH OF grid AT index zero
        DECLARE totalOps AS 0
        
        SET colIndex TO 0
        WHILE colIndex < colCount
            SET rowIndex TO 0
            WHILE rowIndex < rowCount - 1
                DECLARE currentVal AS grid AT rowIndex THEN colIndex
                DECLARE nextVal AS grid AT (rowIndex + 1) THEN colIndex
                IF NOT (currentVal = nextVal)
                    totalOps = totalOps + 1
                    grid[rowIndex + 1][colIndex] = currentVal
                END IF
                rowIndex = rowIndex + 1
            END WHILE
            
            SET rowIndex TO 0
            WHILE rowIndex < rowCount
                IF (colIndex < colCount - 1) AND (grid[rowIndex][colIndex] = grid[rowIndex][colIndex + 1])
                    totalOps = totalOps + 1
                    DECLARE candidateValue AS 0
                    DECLARE replaced AS false
                    WHILE candidateValue <= 9 AND NOT replaced
                        IF NOT (candidateValue = grid[rowIndex][colIndex])
                            grid[rowIndex][colIndex + 1] = candidateValue
                            replaced = true
                        ELSE
                            candidateValue = candidateValue + 1
                        END IF
                    END WHILE
                END IF
                rowIndex = rowIndex + 1
            END WHILE
            
            colIndex = colIndex + 1
        END WHILE
        
        RETURN totalOps
    END FUNCTION
END CLASS
