CLASS Solution  
    FUNCTION countMatchingSubarrays(nums parameter, pattern parameter)  
        DECLARE lengthNums AS LENGTH OF nums parameter  
        DECLARE lengthPattern AS LENGTH OF pattern parameter  
        DECLARE totalMatches AS 0  
        DECLARE indexIter AS 0  
        WHILE indexIter <= (lengthNums - lengthPattern - 1)  
            DECLARE isValidMatch AS True  
            DECLARE innerCounter AS 0  
            WHILE innerCounter < lengthPattern  
                DECLARE currentPatternElement AS pattern parameter AT innerCounter  
                DECLARE currentNum AS nums parameter AT (indexIter + innerCounter)  
                DECLARE nextNum AS nums parameter AT (indexIter + innerCounter + 1)  
                IF currentPatternElement = 1 THEN  
                    IF nextNum <= currentNum THEN  
                        SET isValidMatch TO False  
                        EXIT WHILE  
                    END IF  
                ELSE IF currentPatternElement = 0 THEN  
                    IF nextNum != currentNum THEN  
                        SET isValidMatch TO False  
                        EXIT WHILE  
                    END IF  
                ELSE IF currentPatternElement = -1 THEN  
                    IF nextNum >= currentNum THEN  
                        SET isValidMatch TO False  
                        EXIT WHILE  
                    END IF  
                END IF  
                SET innerCounter TO innerCounter + 1  
            END WHILE  
            IF isValidMatch = True THEN  
                SET totalMatches TO totalMatches + 1  
            END IF  
            SET indexIter TO indexIter + 1  
        END WHILE  
        RETURN totalMatches  
    END FUNCTION  
END CLASS
