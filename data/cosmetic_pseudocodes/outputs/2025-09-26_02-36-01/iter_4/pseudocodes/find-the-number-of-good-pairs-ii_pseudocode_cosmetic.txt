CLASS Solution
    FUNCTION numberOfPairs(nums1, nums2, k)
        DECLARE frequency_map AS empty dictionary
        DECLARE index AS zero
        DECLARE length_nums2 AS size of nums2
        WHILE index < length_nums2
            LET element TO nums2[index]
            IF element EXISTS IN frequency_map THEN
                SET frequency_map[element] TO frequency_map[element] + one
            ELSE
                INITIALIZE frequency_map[element] AS one
            END IF
            INCREMENT index BY one
        END WHILE

        DECLARE total_pairs AS zero
        DECLARE pos AS zero
        DECLARE length_nums1 AS size of nums1
        WHILE pos < length_nums1
            LET current_value TO nums1[pos]
            FOR EACH key VALUE pair IN frequency_map
                LET divisor TO key * k
                LET division_remainder TO current_value MODULO divisor
                IF division_remainder = 0 THEN
                    SET total_pairs TO total_pairs + VALUE
                END IF
            END FOR
            INCREMENT pos BY one
        END WHILE
        RETURN total_pairs
    END FUNCTION
END CLASS
