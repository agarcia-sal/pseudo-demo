CLASS Solution
    FUNCTION minStartingIndex WITH PARAMETERS s pattern
        FUNCTION is_almost_equal WITH PARAMETERS substring pattern
            LET mismatch_counter BE 0
            LET pos BE 0
            WHILE pos LESS THAN THE LENGTH OF substring
                LET char_sub BE substring AT pos
                LET char_pat BE pattern AT pos
                IF NOT (char_sub EQUALS char_pat)
                    SET mismatch_counter AS mismatch_counter PLUS 1
                    IF mismatch_counter IS GREATER THAN 1
                        RETURN False
                    END IF
                END IF
                SET pos TO pos PLUS 1
            END WHILE
            RETURN True
        END FUNCTION

        LET plength BE LENGTH OF pattern
        LET start_idx BE 0
        WHILE start_idx IS LESS THAN OR EQUAL TO (LENGTH OF s) MINUS plength
            LET candidate_slice BE EMPTY SEQUENCE
            LET offset BE 0
            WHILE offset IS LESS THAN plength
                APPEND s AT (start_idx PLUS offset) TO candidate_slice
                SET offset TO offset PLUS 1
            END WHILE
            IF is_almost_equal WITH ARGUMENTS candidate_slice pattern THEN
                RETURN start_idx
            END IF
            SET start_idx TO start_idx PLUS 1
        END WHILE
        RETURN -1
    END FUNCTION
END CLASS
