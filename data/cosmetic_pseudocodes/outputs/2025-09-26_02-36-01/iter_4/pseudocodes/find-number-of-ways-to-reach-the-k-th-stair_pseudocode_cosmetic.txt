CLASS Solution
    FUNCTION waysToReachStair WITH PARAMETER k
        FUNCTION dfs WITH PARAMETERS i j jump
            IF NOT (i LESS THAN OR EQUAL TO k PLUS one) THEN
                RETURN 0
            END IF

            DECLARE result AS integer
            SET isAtTarget TO (i EQUALS k)
            SET result TO isAtTarget * 1

            IF (i GREATER THAN 0) THEN
                IF (j EQUALS 0) THEN
                    DECLARE tempValue AS integer
                    SET tempValue TO dfs(i MINUS one, 1, jump)
                    SET result TO result PLUS tempValue
                END IF
            END IF

            DECLARE incrementStep AS integer
            DECLARE nextI AS integer
            DECLARE nextJump AS integer

            SET incrementStep TO 2 TIMES jump
            SET nextI TO i PLUS incrementStep
            SET nextJump TO jump PLUS 1

            SET result TO result PLUS dfs(nextI, 0, nextJump)

            RETURN result
        END FUNCTION

        RETURN dfs(1, 0, 0)
    END FUNCTION
END CLASS
