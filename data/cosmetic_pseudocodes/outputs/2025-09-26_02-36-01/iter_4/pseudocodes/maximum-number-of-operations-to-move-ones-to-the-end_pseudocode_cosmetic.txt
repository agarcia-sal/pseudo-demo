CLASS Solution  
    FUNCTION maxOperations(collection)  
        DECLARE result AS 0  
        DECLARE tally AS 0  
        DECLARE cursor AS 0  
        WHILE cursor < LENGTH(collection) DO  
            DECLARE current_char AS collection[cursor]  
            IF current_char IS EQUAL TO '1' THEN  
                SET tally TO tally PLUS 1  
            ELSE  
                IF NOT (cursor IS EQUAL TO 0) THEN  
                    DECLARE previous_char AS collection[cursor MINUS 1]  
                    IF previous_char IS EQUAL TO '1' THEN  
                        SET result TO result PLUS tally  
                    END IF  
                END IF  
            END IF  
            INCREMENT cursor BY 1  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
