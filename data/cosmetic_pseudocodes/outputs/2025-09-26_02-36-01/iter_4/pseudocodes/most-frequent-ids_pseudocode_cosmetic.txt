CLASS Solution  
    FUNCTION mostFrequentIDs(nums PARAMETER, freq PARAMETER)  
        DECLARE freq_map AS a dictionary defaulting to zero  
        DECLARE heap_container AS an empty list  
        DECLARE output_list AS an empty list  
        DECLARE index TO 0  
        WHILE index < LENGTH(nums)  
            DECLARE current_num AS nums[index]  
            DECLARE current_freq AS freq[index]  
            LET freq_map[current_num] TO freq_map[current_num] + current_freq  
            DECLARE neg_freq AS 0 - freq_map[current_num]  
            APPEND (neg_freq, current_num) TO heap_container  
            WHILE LENGTH(heap_container) > 0 AND (- (heap_container[0][0]) != freq_map[heap_container[0][1]])  
                REMOVE heap_container[0] FROM heap_container  
            END WHILE  
            IF LENGTH(heap_container) > 0 THEN  
                APPEND (0 - heap_container[0][0]) TO output_list  
            ELSE  
                APPEND 0 TO output_list  
            END IF  
            INCREMENT index BY 1  
        END WHILE  
        GIVE BACK output_list  
    END FUNCTION  
END CLASS
