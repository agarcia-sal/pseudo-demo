CLASS Solution
    FUNCTION largestPalindrome(n k)
        IF NOT (n NOT EQUAL TO 1)
            LET val TO nine
            WHILE val IS GREATER OR EQUAL TO one
                LET remainder TO val MOD k
                IF remainder IS zero
                    RETURN STRING(val)
                END IF
                LET val TO val MINUS one
            END WHILE
            RETURN "0"
        END IF

        LET half_len TO (n PLUS one) DIVIDED BY 2
        LET nine_str TO EMPTY STRING
        LET count TO zero
        WHILE count LESS THAN half_len
            LET nine_str TO nine_str CONCATENATED WITH "9"
            LET count TO count PLUS one
        END WHILE
        LET half TO TO_INTEGER(nine_str)

        WHILE half GREATER THAN zero
            LET half_str TO STRING(half)
            IF (n MOD 2) IS zero
                LET rev_half_str TO EMPTY STRING
                LET index TO LENGTH(half_str) MINUS one
                WHILE index GREATER OR EQUAL TO zero
                    LET rev_half_str TO rev_half_str CONCATENATED WITH half_str[index]
                    LET index TO index MINUS one
                END WHILE
                LET full_str TO half_str CONCATENATED WITH rev_half_str
            ELSE
                LET rev_sub_str TO EMPTY STRING
                LET idx TO LENGTH(half_str) MINUS TWO
                WHILE idx GREATER OR EQUAL TO zero
                    LET rev_sub_str TO rev_sub_str CONCATENATED WITH half_str[idx]
                    LET idx TO idx MINUS one
                END WHILE
                LET full_str TO half_str CONCATENATED WITH rev_sub_str
            END IF

            LET full TO TO_INTEGER(full_str)
            IF full MOD k IS zero
                RETURN STRING(full)
            END IF
            LET half TO half MINUS one
        END WHILE

        RETURN "0"
    END FUNCTION
END CLASS
