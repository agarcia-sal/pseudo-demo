CLASS Solution  
    FUNCTION countAlternatingSubarrays(nums)  
        DECLARE totalCount AS 0  
        LET size BE LENGTH OF nums  
        IF size IS zero  
            RETURN 0  
        ENDIF  
        SET totalCount TO size  
        DECLARE lengthSeq AS 1  
        DECLARE index AS 1  
        WHILE index < size  
            IF NOT (nums[index] EQUALS nums[index - 1])  
                LET lengthSeq BE lengthSeq + 1  
            ELSE  
                SET lengthSeq TO 1  
            ENDIF  
            SET totalCount TO totalCount + (lengthSeq - 1)  
            INCREMENT index BY 1  
        ENDWHILE  
        RETURN totalCount  
    END FUNCTION  
END CLASS
