CLASS Solution  
    FUNCTION minimumArrayLength(nums)  
        LET smallest TO nums[0]  
        LET idx TO 1  
        WHILE idx LESS THAN LENGTH OF nums  
            IF nums[idx] LESS THAN smallest  
                SET smallest TO nums[idx]  
            END IF  
            INCREMENT idx BY one  
        END WHILE  

        LET how_many TO zero  
        LET i TO zero  
        WHILE i LESS THAN LENGTH OF nums  
            IF nums[i] EQUALS smallest  
                SET how_many TO how_many PLUS one  
            END IF  
            INCREMENT i BY one  
        END WHILE  

        IF NOT (how_many DIFFERENT FROM one)  
            RETURN one  
        END IF  

        LET result TO (how_many PLUS one) DIVIDED BY 2  
        RETURN result  
    END FUNCTION  
END CLASS
