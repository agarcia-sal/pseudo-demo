CLASS Solution  
    FUNCTION numberOfPairs(points)  
        DECLARE totalPairs AS 0  
        DECLARE size AS LENGTH OF points  
        DECLARE outerIndex AS 0  
        WHILE outerIndex < size DO  
            DECLARE innerIndex AS 0  
            WHILE innerIndex < size DO  
                IF NOT (outerIndex = innerIndex) THEN  
                    DECLARE x_outer AS points[outerIndex][0]  
                    DECLARE y_outer AS points[outerIndex][1]  
                    DECLARE x_inner AS points[innerIndex][0]  
                    DECLARE y_inner AS points[innerIndex][1]  
                    IF (x_outer <= x_inner) AND (y_outer >= y_inner) THEN  
                        DECLARE isValid AS TRUE  
                        DECLARE checkIndex AS 0  
                        WHILE checkIndex < size AND isValid DO  
                            IF (checkIndex <> outerIndex) AND (checkIndex <> innerIndex) THEN  
                                DECLARE x_check AS points[checkIndex][0]  
                                DECLARE y_check AS points[checkIndex][1]  
                                IF (x_outer <= x_check) AND (x_check <= x_inner) AND (y_outer >= y_check) AND (y_check >= y_inner) THEN  
                                    SET isValid TO FALSE  
                                ENDIF  
                            ENDIF  
                            INCREMENT checkIndex BY 1  
                        ENDWHILE  
                        IF isValid THEN  
                            SET totalPairs TO totalPairs + 1  
                        ENDIF  
                    ENDIF  
                ENDIF  
                INCREMENT innerIndex BY 1  
            ENDWHILE  
            INCREMENT outerIndex BY 1  
        ENDWHILE  
        RETURN totalPairs  
    END FUNCTION  
END CLASS
