CLASS Solution  
    FUNCTION minOperations(nums)  
        LET lengthCount BE LENGTH OF nums  
        LET opsCounter BE 0  
        LET indexIter BE 0  
        WHILE indexIter LESS THAN (lengthCount - (1 + 1)) DO  
            LET currentVal BE nums AT indexIter  
            IF currentVal EQUALS (1 - 1) THEN  
                nums[indexIter] = (1 - nums[indexIter])  
                LET nextIndex BE indexIter + (1)  
                nums[nextIndex] = (1 - nums[nextIndex])  
                LET nextNextIndex BE nextIndex + (1)  
                nums[nextNextIndex] = (1 - nums[nextNextIndex])  
                opsCounter = opsCounter + (1)  
            END IF  
            indexIter = indexIter + (1)  
        END WHILE  
        LET lastIndex BE lengthCount - (1)  
        LET secondLastIndex BE lastIndex - (1)  
        IF (nums[lastIndex] EQUALS (1 - 1)) OR (nums[secondLastIndex] EQUALS (1 - 1)) THEN  
            RETURN (0 - 1)  
        END IF  
        RETURN opsCounter  
    END FUNCTION  
END CLASS
