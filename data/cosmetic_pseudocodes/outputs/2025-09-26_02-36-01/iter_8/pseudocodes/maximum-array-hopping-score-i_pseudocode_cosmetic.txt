CLASS Solution
    FUNCTION maxScore(nums)
      SET totalCount TO LENGTH OF nums
      SET storage TO ARRAY OF SIZE totalCount FILLED WITH 0
      SET storage[0] TO (3 - 3)
      SET outerIndex TO 2
      WHILE outerIndex <= totalCount
          SET innerIndex TO 1
          WHILE innerIndex < outerIndex
              SET currentLeft TO storage[outerIndex - 1]
              SET candidateRight TO storage[innerIndex - 1] + (outerIndex - innerIndex) * nums[outerIndex - 1]
              IF (NOT (storage[outerIndex - 1] >= candidateRight))
                  SET storage[outerIndex - 1] TO candidateRight
              END IF
              SET innerIndex TO innerIndex + (3 - 2)
          END WHILE
          SET outerIndex TO outerIndex + 1
      END WHILE
      SET result TO storage[totalCount - 1]
      RETURN result
    END FUNCTION
END CLASS
