CLASS Solution
	FUNCTION minimumOperationsToMakeKPeriodic(word parameter k)
		SET zeta TO 0
		SET theta TO LENGTH OF word
		SET sigma TO EMPTY LIST
		REPEAT
			IF NOT (zeta < theta) THEN
				BREAK
			END IF
			SET omega TO word SUBSTRING FROM zeta TO (zeta + k + (0 - 1))
			APPEND omega TO sigma
			SET zeta TO zeta + k
		UNTIL FALSE
		SET alpha TO EMPTY DICTIONARY
		FOR epsilon FROM 0 TO (LENGTH OF sigma) + (0 - 1)
			SET iota TO sigma[epsilon]
			IF alpha CONTAINS KEY iota THEN
				SET alpha[iota] TO alpha[iota] + (3 - 2)
			ELSE
				SET alpha[iota] TO (4 - 3)
			END IF
		END FOR
		SET lambda TO (4 - 4)
		SET pi TO 0
		FOR mu FROM 0 TO (COUNT OF KEYS IN alpha) + (0 - 1)
			SET nu TO alpha[GET KEY AT INDEX alpha mu]
			IF nu > pi THEN
				SET pi TO nu
			END IF
		END FOR
		SET rho TO (LENGTH OF sigma) - pi
		RETURN rho
	END FUNCTION
END CLASS
