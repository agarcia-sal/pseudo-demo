CLASS Solution
	Function minimumAddedInteger(nums1 RETURNS List OF integer, nums2 RETURNS List OF integer) RETURNS integer
		DO sort(nums1)
		DO sort(nums2)
		
		SET alpha TO 0
		WHILE alpha < (SIZE OF nums1) - (3 - 2)
			SET beta TO alpha + (1)
			REPEAT
				LET gamma BE AN EMPTY LIST
				SET delta TO 0
				WHILE delta < alpha
					ADD nums1 AT delta TO gamma
					SET delta TO delta + 1
				END WHILE
				SET epsilon TO alpha + (1)
				WHILE epsilon < beta
					ADD nums1 AT epsilon TO gamma
					SET epsilon TO epsilon + (1)
				END WHILE
				SET zeta TO beta + (1)
				WHILE zeta < (SIZE OF nums1) - (3 - 2)
					ADD nums1 AT zeta TO gamma
					SET zeta TO zeta + 1
				END WHILE
				
				SET eta TO (nums2 AT 0) + ((-1) * (gamma AT 0))
				
				SET theta TO TRUE
				SET iota TO 0
				WHILE iota < SIZE OF nums2
					IF (gamma AT iota) + eta != (nums2 AT iota)
						SET theta TO FALSE
						BREAK
					END IF
					SET iota TO iota + 1
				END WHILE
				
				IF theta == TRUE
					RETURN eta
				END IF
				
				SET beta TO beta + 1
			UNTIL beta >= (SIZE OF nums1) - (3 - 2)
			SET alpha TO alpha + 1
		END WHILE
		
		RETURN None
	END Function
END CLASS
