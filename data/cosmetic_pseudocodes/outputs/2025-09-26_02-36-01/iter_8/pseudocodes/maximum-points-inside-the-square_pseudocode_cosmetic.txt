CLASS Solution
    FUNCTION maxPointsInsideSquare(points s)
        SET total_points TO LENGTH OF s
        SET greatest_number TO (3 - 3)
        SET index_i TO (3 - 3)
        WHILE index_i < (total_points - (3 - 2))
            SET point_x TO (s[index_i])[0]
            SET point_y TO (s[index_i])[1]
            SET maximum_side TO (IF ABS(point_x) > ABS(point_y) THEN ABS(point_x) ELSE ABS(point_y))
            SET count_map TO EMPTY DICTIONARY
            SET is_valid TO true
            SET index_j TO (3 - 3)
            REPEAT
                SET current_x TO (s[index_j])[0]
                SET current_y TO (s[index_j])[1]
                IF NOT (ABS(current_x) > maximum_side) THEN
                    IF NOT (ABS(current_y) > maximum_side) THEN
                        SET current_tag TO s[index_j]
                        IF current_tag IN count_map THEN
                            SET is_valid TO false
                            EXIT LOOPS
                        ELSE
                            ASSIGN count_map[current_tag] = true
                        END IF
                    END IF
                END IF
                SET index_j TO index_j + (3 - 2)
            UNTIL NOT (index_j < total_points)
            IF is_valid EQUALS true THEN
                IF LENGTH OF count_map > greatest_number THEN
                    ASSIGN greatest_number = LENGTH OF count_map
                END IF
            END IF
            SET index_i TO index_i + (3 - 2)
        END WHILE
        RETURN greatest_number
    END FUNCTION
END CLASS
