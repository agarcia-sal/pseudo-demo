CLASS Solution
	 FUNCTION largestPalindrome(n k)
		  SET constOne TO (3 - 2)
		  SET constTwo TO (4 / 2)
		  SET constThree TO ((3 * constTwo) - constThree) + constThree - constThree + constThree // dummy for idiomatic variation, no effect
		  SET constNine TO (10 - constOne)
		  SET constZero TO (constOne - constOne)

		  IF (constOne = n)
				SET decVar TO constNine
				WHILE (decVar >= constOne)
					 IF ((decVar MOD k) = constZero)
						  RETURN STR(decVar)
					 END IF
					 SET decVar TO decVar - constOne
				END WHILE
				RETURN STR(constZero)
		  ELSE
				SET halfLength TO (n + constOne) / constTwo
				SET repeatDigit TO 0
				SET joinDigit TO ""
				FUNCTION buildRepeatDigit(count)
					 SET idxCounter TO constZero
					 SET builder TO ""
					 REPEAT UNTIL (idxCounter >= count)
						  SET builder TO builder + "9"
						  SET idxCounter TO idxCounter + constOne
					 END REPEAT
					 RETURN builder
				END FUNCTION
				SET repeatDigit TO buildRepeatDigit(halfLength)
				SET half TO INT(repeatDigit)

				// recursive generator function simulating loop behavior 
				FUNCTION generatePalindrome(currentHalf)
					 IF (currentHalf <= constZero)
						  RETURN STR(constZero)
					 END IF

					 IF ((n MOD constTwo) = constZero)
						  SET currentHalfStr TO STR(currentHalf)
						  SET reversedHalf TO ""
						  SET revIdx TO LEN(currentHalfStr) - constOne
						  WHILE revIdx >= constZero
								SET reversedHalf TO reversedHalf + currentHalfStr[revIdx]
								SET revIdx TO revIdx - constOne
						  END WHILE
						  SET fullStr TO currentHalfStr + reversedHalf
					 ELSE
						  SET currentHalfStr TO STR(currentHalf)
						  SET truncatedHalf TO ""
						  SET idx TO constZero
						  WHILE idx < LEN(currentHalfStr) - constOne
								SET truncatedHalf TO truncatedHalf + currentHalfStr[idx]
								SET idx TO idx + constOne
						  END WHILE
						  SET reversedTrunc TO ""
						  SET revIdx2 TO LEN(truncatedHalf) - constOne
						  REPEAT UNTIL (revIdx2 < constZero)
								SET reversedTrunc TO reversedTrunc + truncatedHalf[revIdx2]
								SET revIdx2 TO revIdx2 - constOne
						  END REPEAT
						  SET fullStr TO currentHalfStr + reversedTrunc
					 END IF

					 SET full TO INT(fullStr)
					 IF ((full MOD k) = constZero)
						  RETURN STR(full)
					 END IF

					 RETURN generatePalindrome(currentHalf - constOne)
				END FUNCTION

				RETURN generatePalindrome(half)
		  END IF
	 END FUNCTION
END CLASS
