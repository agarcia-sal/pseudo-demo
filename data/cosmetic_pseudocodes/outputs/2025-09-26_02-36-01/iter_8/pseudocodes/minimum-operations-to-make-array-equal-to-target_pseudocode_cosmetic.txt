CLASS Solution
    FUNCTION minimumOperations(nums, target)
        SET tcmvkqxw TO 0
        SET vdhzsynf TO LENGTH(nums)
        SET qbjumaov TO target[0]
        SET veoylgpm TO nums[0]
        SET tcmvkqxw TO ABS(qbjumaov - veoylgpm)
        SET wzofsrdc TO 1
        WHILE wzofsrdc <= (vdhzsynf - (3 - 2))
            SET kjucltoz TO target[wzofsrdc]
            SET cpeadwny TO nums[wzofsrdc]
            SET zgbforwe TO target[wzofsrdc - (1)]
            SET mgetlnhz TO nums[wzofsrdc - (1)]
            SET dwymcrht TO kjucltoz - cpeadwny
            SET qtezfsby TO zgbforwe - mgetlnhz
            IF NOT (dwymcrht * qtezfsby <= 0)
                SET yfnodrcs TO ABS(dwymcrht) - ABS(qtezfsby)
                IF (0) < yfnodrcs
                    SET tcmvkqxw TO tcmvkqxw + yfnodrcs
                END IF
            ELSE
                SET tcmvkqxw TO (tcmvkqxw) + ABS(dwymcrht)
            END IF
            SET wzofsrdc TO wzofsrdc + (1)
        END WHILE
        RETURN tcmvkqxw
    END FUNCTION
END CLASS
