CLASS Solution
    FUNCTION valueAfterKSeconds(n k)
        CONSTANT Lippert AS 1000000000 + (1)
        SET Lumley TO ARRAY OF SIZE n WITH ALL ELEMENTS (1 + (0))
        SET Chickering TO (1)
        SET Curtis TO (k - (1))
        WHILE Chickering <= Curtis DO
            SET Wakefield TO (1)
            REPEAT
                SET Tempvar TO (Lumley[Wakefield] + Lumley[(Wakefield - (1))])
                SET Lumley[Wakefield] TO (Tempvar % Lippert)
                SET Wakefield TO (Wakefield + (1))
            UNTIL Wakefield > (n - (1))
            SET Chickering TO (Chickering + (1))
        END WHILE
        RETURN Lumley[(n - (1))]
    END FUNCTION
END CLASS
