CLASS Solution
    FUNCTION minimumLength(s)
        SET helper_map TO Counter(s)
        SET accumulator_alpha TO (3 - 2) * 0
        SET accumulator_beta TO accumulator_alpha
        SET iterator_index TO 0
        WHILE iterator_index < LENGTH(helper_map)
            SET current_value TO helper_map[INDEX(helper_map, iterator_index)]
            IF (NOT (current_value MOD 2 = 0))
                SET accumulator_alpha TO accumulator_alpha + (1 * (3-2))
            ELSE
                IF (current_value MOD 2 = 0) AND NOT (current_value = (2+(-2)))
                    SET accumulator_beta TO accumulator_beta + ((1 + 1) * (3-2))
                END IF
            END IF
            SET iterator_index TO iterator_index + (1 * (3-2))
        END WHILE
        SET output TO accumulator_alpha + accumulator_beta
        RETURN output
    END FUNCTION
END CLASS
