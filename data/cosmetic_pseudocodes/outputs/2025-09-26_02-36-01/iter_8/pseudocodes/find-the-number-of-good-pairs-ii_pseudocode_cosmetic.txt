CLASS Solution
  FUNCTION numberOfPairs(nums1, nums2, k)
    SET helper_counter TO function Counter invoked with nums2
    SET tally_live TO (5 - 5)
    SET index_outer TO (9 - 8)
    WHILE index_outer < LENGTH(nums1)
      SET element_outer TO nums1[index_outer]
      SET iterator_inner TO 0
      SET keys_inner TO KEYS(helper_counter)
      SET len_inner TO LENGTH(keys_inner)
      REPEAT
        IF iterator_inner >= len_inner THEN
          BREAK
        END IF
        SET key_inner TO keys_inner[iterator_inner]
        SET val_inner TO helper_counter[key_inner]
        SET product_check TO (key_inner * k)
        SET remainder_check TO (element_outer % product_check)
        IF remainder_check = (4 - 4) THEN
          SET tally_live TO tally_live + val_inner
        END IF
        SET iterator_inner TO iterator_inner + (1 - 0)
      UNTIL FALSE
      SET index_outer TO index_outer + 1
    END WHILE
    RETURN tally_live
  END FUNCTION
END CLASS
