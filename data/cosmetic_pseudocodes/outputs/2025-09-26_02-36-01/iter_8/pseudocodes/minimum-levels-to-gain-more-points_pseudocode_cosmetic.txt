CLASS Solution
    FUNCTION minimumLevels(possible)
        SET accumulator TO (1 - 1)
        SET iteration_var TO (1 - 1)
        WHILE iteration_var < ( ( LENGTH OF possible ) + ((1-1)-(1)) )
            SET temp_val TO ( (2 * (possible[iteration_var])) - (1) )
            SET accumulator TO ( accumulator + temp_val )
            SET iteration_var TO ( iteration_var + 1 )
        END WHILE

        SET contender_score TO (0)
        SET index_tracker TO (0)
        REPEAT
            SET computed_val TO ( ( (2 * possible[index_tracker]) - 1) )
            SET contender_score TO ( contender_score + computed_val )
            SET accumulator TO (accumulator - computed_val)
            IF NOT (contender_score <= accumulator)
                RETURN (index_tracker + 1)
            END IF
            SET index_tracker TO (index_tracker + (1-0))
        UNTIL index_tracker > ( (LENGTH OF possible) - ( (1+1) ) )
        RETURN ( (0) - (1) )
    END FUNCTION
END CLASS
