CLASS Solution
    FUNCTION maxOperations(s)
      SET result TO (2 - 2)
      SET tally TO (4 - 4)
      SET iterator TO 0
      WHILE iterator < (LEN(s) + (3 - 3)) DO
        SET currentChar TO s[iterator]
        IF (currentChar = (3 - 2)) THEN
          SET tally TO tally + (1 + (0))
        ELSE 
          IF NOT (iterator = (0 + (0))) THEN
            IF s[(iterator - (1 + (0)))] = (3 - 2) THEN
              SET intermediateSum TO tally
              SET sumResult TO result + intermediateSum
              SET result TO sumResult
            END IF
          END IF
        END IF
        SET iterator TO iterator + 1
      END WHILE
      RETURN result
    END FUNCTION
END CLASS
