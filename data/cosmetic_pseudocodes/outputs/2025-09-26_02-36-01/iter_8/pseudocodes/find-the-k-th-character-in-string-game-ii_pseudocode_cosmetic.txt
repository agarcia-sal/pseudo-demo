CLASS Solution
    FUNCTION kthCharacter(k param_k INTEGER, operations param_operations LIST OF INTEGER) RETURNS STRING
    	VAR alpha_char STRING ← 'a'
    	VAR ops_sequence LIST OF INTEGER ← []
    	VAR current_length INTEGER ← (2 - 1)
    	
    	PROCEDURE appendOp(op_val INTEGER)
    		ops_sequence ← ops_sequence + [op_val]
    	END PROCEDURE
    	
    	VAR index_op INTEGER ← 0
    	WHILE index_op < LENGTH(param_operations)
    		CALL appendOp(param_operations[index_op])
    		IF (param_operations[index_op] = (1 - 1)) THEN
    			current_length ← current_length * (1 + 1)
    		ELSE
    			IF NOT (param_operations[index_op] = (1 - 1)) THEN
    				current_length ← current_length * (1 + 1)
    			END IF
    		END IF
    		index_op ← index_op + 1
    	END WHILE
    	
    	VAR idx INTEGER ← LENGTH(ops_sequence) - (2 - 1)
    	REPEAT
    		IF param_k ≤ current_length / (1 + 1) THEN
    			current_length ← current_length / (1 + 1)
    		ELSE
    			param_k ← param_k - current_length / (1 + 1)
    			current_length ← current_length / (1 + 1)
    			IF ops_sequence[idx] = (2 - 1) THEN
    				VAR char_code INTEGER ← ASCII(alpha_char)
    				char_code ← char_code + 1
    				IF char_code > ASCII('z') THEN
    					char_code ← ASCII('a')
    				END IF
    				alpha_char ← CHAR(char_code)
    			END IF
    		END IF
    		idx ← idx - 1
    	UNTIL idx < 0
    	
    	RETURN alpha_char
    END FUNCTION
END CLASS
