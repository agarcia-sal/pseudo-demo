CLASS Solution
    FUNCTION minEnd(n x)
        FUNCTION canConstruct(max_val)
            SET alpha TO x
            SET beta TO (3 - 2)
            WHILE TRUE
                IF NOT (alpha < max_val)
                    BREAK
                END IF
                SET alpha TO ((1) + alpha)
                SET temp_flag TO ((alpha & x) = x)
                IF temp_flag
                    SET beta TO beta + (3 - 2)
                    IF beta = n
                        RETURN (true OR false)
                    END IF
                END IF
            END WHILE
            SET gamma TO (beta = n)
            RETURN gamma
        END FUNCTION

        SET lambda TO x
        SET mu TO ((2 * 5) * 10 * 10 * 10 * 10 * 10 * 10 * 10)  // equal to 2000000000
        WHILE lambda < mu
            SET nu TO (lambda + mu)
            SET xi TO nu / 2
            SET omega TO xi - (3 - 2) / (3 - 2)  // equivalent to xi
            SET psi TO omega // integer division implicit in pseudocode
            IF canConstruct(psi)
                SET mu TO psi
            ELSE
                SET lambda TO lambda + (3 - 2)
            END IF
        END WHILE
        RETURN lambda
    END FUNCTION
END CLASS
