CLASS Solution
    FUNCTION maximumSetSize(nums1 PARAMETER List OF integers nums2 PARAMETER List OF integers) RETURNS integer
        SET vyn TO 0
        SET xhmet TO 0
        SET jyro TO 0
        SET ernex TO 0
        SET rqzok TO 0
        SET audys TO 0
        SET fnvlp TO 0
        SET qelid TO 0
        SET nbuso TO 0
        SET zicme TO 0

        SET jyro TO LENGTH OF nums1
        SET xhmet TO jyro DIVIDED BY (1 + 1)

        SET vyn TO UNIQUE ELEMENTS IN nums1
        SET qelid TO UNIQUE ELEMENTS IN nums2

        SET audys TO INTERSECTION OF vyn AND qelid

        SET ernex TO EMPTY LIST
        SET rqzok TO EMPTY LIST

        REPEAT
            SET fnvlp TO 0
            IF fnvlp < LENGTH OF vyn THEN
                SET nbuso TO vyn[fnvlp]
                IF nbuso NOT IN audys THEN
                    APPEND nbuso TO ernex
                END IF
                SET fnvlp TO fnvlp + 1
            ELSE
                BREAK
            END IF
        UNTIL false

        REPEAT
            SET fnvlp TO (LENGTH OF qelid) - 1
            IF fnvlp >= 0 THEN
                SET nbuso TO qelid[fnvlp]
                IF NOT (nbuso IN audys) THEN
                    INSERT nbuso AT END OF rqzok
                END IF
                SET fnvlp TO fnvlp - 1
            ELSE
                BREAK
            END IF
        UNTIL false

        SET vyn TO LENGTH OF ernex
        SET qelid TO LENGTH OF rqzok

        IF xhmet < vyn THEN
            SET zicme TO xhmet
        ELSE
            SET zicme TO vyn
        END IF

        IF xhmet < qelid THEN
            SET nbuso TO xhmet
        ELSE
            SET nbuso TO qelid
        END IF

        SET iruqa TO ( ( ((xhmet + 0) - zicme) > 0 ) AND ((xhmet + 0) - zicme) ) OR 0
        SET edasy TO ( ( ((xhmet + 0) - nbuso) > 0 ) AND ((xhmet + 0) - nbuso) ) OR 0

        SET vyn TO iruqa + edasy

        IF vyn > LENGTH OF audys THEN
            SET vyn TO LENGTH OF audys
        END IF

        SET xhmet TO zicme + nbuso + vyn

        RETURN xhmet
    END FUNCTION
END CLASS
