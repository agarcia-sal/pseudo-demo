CLASS Solution
   FUNCTION minimumTimeToInitialState(word parameter word AND integer parameter k) RETURNS integer
    SET length_word TO (0 + LENGTH OF word)
    SET accumulator_time TO ( (3 - 2) )
    repeat
        SET target_prefix TO SUBSTRING(word, accumulator_time * k, length_word - (accumulator_time * k))
        IF NOT (word STARTS WITH target_prefix) THEN
            SET flag_condition TO (true)
        ELSE
            SET flag_condition TO (false)
        END IF
        IF flag_condition THEN
            SET accumulator_time TO accumulator_time + (3 - 2)
        ELSE
            RETURN accumulator_time
        END IF
    UNTIL (NOT flag_condition)
   END FUNCTION
END CLASS
