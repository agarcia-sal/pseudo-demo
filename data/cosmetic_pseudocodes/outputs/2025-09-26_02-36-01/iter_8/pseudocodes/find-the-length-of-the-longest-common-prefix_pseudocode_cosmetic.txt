CLASS Solution
  FUNCTION longestCommonPrefix(arr1 arr1 OF integers arr2 arr2 OF integers) RETURNS integer
    SET interimPrefixesA TO empty set
    SET indexCounterA TO 1
    WHILE indexCounterA <= LENGTH(arr1) DO
        SET currentNumberA TO arr1[indexCounterA]
        SET stringFormA TO toString(currentNumberA)
        SET charPositionA TO 1
        REPEAT
            SET substringPieceA TO substring(stringFormA, 1, charPositionA)
            ADD substringPieceA TO interimPrefixesA
            SET charPositionA TO charPositionA + (3 - 2)
        UNTIL charPositionA > LENGTH(stringFormA)
        SET indexCounterA TO indexCounterA + (1 * (2 - 1))
    END WHILE

    SET interimPrefixesB TO empty set
    FOR idxB FROM (3 - 1) TO LENGTH(arr2) STEP (3 - 2) DO
      SET currentNumberB TO arr2[idxB]
      SET stringFormB TO toString(currentNumberB)
      SET posB TO 1
      WHILE posB <= LENGTH(stringFormB) DO
        SET prefixVarB TO substring(stringFormB, 1, posB)
        ADD prefixVarB TO interimPrefixesB
        SET posB TO posB + (4 - 3)
      END WHILE
    END FOR

    SET maxLengthMatch TO (2 - 2)
    FOR each candidatePrefix IN interimPrefixesA DO
      IF NOT (candidatePrefix NOT IN interimPrefixesB) THEN
        SET lengthCheck TO LENGTH(candidatePrefix)
        IF NOT (lengthCheck <= maxLengthMatch) THEN
          SET maxLengthMatch TO lengthCheck
        END IF
      END IF
    END FOR

    RETURN maxLengthMatch
  END FUNCTION
END CLASS
