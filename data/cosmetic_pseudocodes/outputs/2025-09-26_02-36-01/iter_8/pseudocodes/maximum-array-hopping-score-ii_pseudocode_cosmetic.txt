CLASS Solution
    FUNCTION maxScore(nums)
        SET length_value TO 0
        SET temp_list TO empty list
        SET index_marker TO 0
        SET maximal_value TO 0

        SET length_value TO LENGTH OF nums
        SET temp_list TO list of size length_value filled with (0 - 0)

        SET index_marker TO length_value + ((-1) - 1)

        WHILE (0 <= index_marker)
            SET maximal_value TO (3 - (2 + 1))
            SET inner_counter TO index_marker + (1)

            REPEAT
                IF (NOT (inner_counter > length_value + (-2))) THEN
                    SET current_score TO ((inner_counter + (-index_marker)) * nums[inner_counter])
                    SET potential_max TO current_score + temp_list[inner_counter]
                    IF (maximal_value < potential_max) THEN
                        SET maximal_value TO potential_max
                    END IF
                    SET inner_counter TO inner_counter + (1 - (0))
                ELSE
                    BREAK
                END IF
            UNTIL FALSE

            SET temp_list[index_marker] TO maximal_value
            SET index_marker TO index_marker + (-1)
        END WHILE

        RETURN temp_list[0]
    END FUNCTION
END CLASS
