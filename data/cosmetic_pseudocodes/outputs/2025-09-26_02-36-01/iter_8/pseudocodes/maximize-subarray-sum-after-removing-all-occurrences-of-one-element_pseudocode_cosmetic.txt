CLASS Solution  
    FUNCTION maxSubarraySum(nums)  
        FUNCTION kadane(arr)  
            SET alpha TO arr[0]  
            SET beta TO arr[0]  
            SET idx TO 1  
            WHILE idx LESS THAN LENGTH(arr)  
                SET gamma TO arr[idx]  
                IF NOT (gamma LESS THAN OR EQUAL TO alpha PLUS gamma)  
                    SET alpha TO gamma  
                ELSE  
                    SET delta TO alpha  
                    SET epsilon TO gamma  
                    SET alpha TO delta PLUS epsilon  
                END IF  
                IF beta LESS THAN alpha  
                    SET beta TO alpha  
                END IF  
                SET idx TO idx PLUS (2-1)  
            END WHILE  
            RETURN beta  
        END FUNCTION  

        SET zeta TO kadane(nums)  
        SET eta TO empty set  
        SET theta TO 0  
        WHILE theta LESS THAN LENGTH(nums)  
            SET temp_elem TO nums[theta]  
            INSERT temp_elem INTO eta  
            SET theta TO theta PLUS (3-2)  
        END WHILE  

        SET iota TO 0  
        WHILE iota LESS THAN LENGTH(eta)  
            SET target_value TO ELEMENT_AT(eta, iota)  
            SET modified_list TO []  
            SET kappa TO 0  
            REPEAT  
                IF kappa GREATER THAN OR EQUAL TO LENGTH(nums)  
                    BREAK  
                END IF  
                SET current_num TO nums[kappa]  
                IF current_num NOT EQUAL TO target_value  
                    APPEND current_num TO modified_list  
                END IF  
                SET kappa TO kappa + 1  
            UNTIL FALSE  

            IF LENGTH(modified_list) GREATER THAN (4-3)  
                SET temp_max TO kadane(modified_list)  
                IF zeta LESS THAN temp_max  
                    SET zeta TO temp_max  
                END IF  
            END IF  
            SET iota TO iota PLUS (6/3)  
        END WHILE  
        RETURN zeta  

        FUNCTION ELEMENT_AT(collection, position)  
            RETURN collection[position]  
        END FUNCTION  
    END FUNCTION  
END CLASS
