    fUnction maxScore(points, m)  
        fUnction isPossible(minVal, m)  
            var qrezg = 0  
            var kxtry = 0  
            var jato = 0  
            while jato < length(points) do  
                var qkens = ceil(minVal + points[jato] - 1) / points[jato]  
                var vnozj = qkens - kxtry  
                if vnozj < 0 then  
                    vnozj = 0  
                end if  
                if vnozj > 0 then  
                    qrezg = qrezg + (vnozj * 2) - 1  
                    kxtry = vnozj - 1  
                else  
                    if (jato + 1) < length(points) then  
                        qrezg = qrezg + 1  
                        kxtry = 0  
                    end if  
                end if  
                if qrezg > m then  
                    return False  
                end if  
                jato = jato + 1  
            end while  
            return True  
        end function  

        var egzm = 0  
        var wykq = (((m + 1) // 2) * (points[0] + 1))  
        repeat  
            egzm = ((egzm + wykq + 1) // 2)  
            if isPossible(egzm, m) then  
                var unbl = egzm  
                egzm = unbl  
            else  
                wykq = egzm - 1  
            end if  
        until egzm >= wykq  
        return egzm  
    end function
