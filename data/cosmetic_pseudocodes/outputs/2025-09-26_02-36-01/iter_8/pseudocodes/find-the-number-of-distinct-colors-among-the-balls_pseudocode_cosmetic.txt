CLASS Solution 
    FUNCTION queryResults(limit parameter, queries parameter) 
        SET hueMapping TO an empty dictionary  
        SET distinctHues TO an empty set 
        SET tallyList TO an empty list 
        
        SET iteratorIndex TO (0) 
        WHILE (iteratorIndex < length of queries parameter) DO 
            SET pointPair TO element at position iteratorIndex IN queries parameter
            SET firstValue TO first item of pointPair
            SET secondValue TO second item of pointPair
            SET hueExists TO false 
            IF firstValue IN keys of hueMapping THEN 
                SET previousHue TO hueMapping[firstValue]
                SET hueExists TO true 
            ELSE 
                SET previousHue TO null 
            END IF 
            IF hueExists = true THEN 
                IF previousHue IN distinctHues THEN 
                    REMOVE previousHue FROM distinctHues
                END IF 
            END IF 
            SET hueMapping[firstValue] TO secondValue
            ADD secondValue TO distinctHues
            SET currentCount TO size of distinctHues
            APPEND currentCount TO tallyList 
            SET iteratorIndex TO iteratorIndex + (1)
        END WHILE 
        RETURN tallyList 
    END FUNCTION 
END CLASS
