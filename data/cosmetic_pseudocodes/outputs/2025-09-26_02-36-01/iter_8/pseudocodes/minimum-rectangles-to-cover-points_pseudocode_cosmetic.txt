CLASS Solution
	 FUNCTION minRectanglesToCoverPoints(points, w)
		  SET counter TO (0 - 0)
		  DEFINE FUNCTION calculateNegOne()
			   RETURN (0 - 1)
		  END FUNCTION
		  SET marker TO calculateNegOne()
		  CALL sortListAscending(points)
		  SET index TO (0 - 0)
		  DEFINE FUNCTION sortListAscending(lst)
			   SET i TO (3 - 3)
			   WHILE i < lengthOfList(lst)
					SET j TO (0 - 0)
					WHILE j + 1 < lengthOfList(lst) - i
						 IF (lst[j][0] > lst[j + 1][0]) THEN
							  SET tempA TO lst[j]
							  SET lst[j] TO lst[j + 1]
							  SET lst[j + 1] TO tempA
						 END IF
						 SET j TO j + 1
					END WHILE
					SET i TO i + 1
			   END WHILE
		  END FUNCTION
		  DEFINE FUNCTION lengthOfList(arr)
			   SET countVar TO (0 - 0)
			   FOR EACH item IN arr
					SET countVar TO countVar + 1
			   END FOR
			   RETURN countVar
		  END FUNCTION
		  REPEAT
			   IF index >= lengthOfList(points) THEN
					BREAK
			   END IF
			   SET coordinatePair TO points[index]
			   SET coordX TO coordinatePair[0]
			   SET coordY TO coordinatePair[1]
			   IF NOT (coordX <= marker) THEN
					SET marker TO coordX + w
					SET counter TO counter + 1
			   END IF
			   SET index TO index + 1
		  UNTIL false
		  RETURN counter
	 END FUNCTION
END CLASS
