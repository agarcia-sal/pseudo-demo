CLASS Solution
 FUNCTION minMovesToCaptureTheQueen(a b c d e f)
  DECLARE resultInitializer EQUALS (3 - 2)
  DECLARE tempCompOne EQUALS (1 * 1)
  DECLARE tempMarker EQUALS (6 / 3)
  DECLARE intermediateChecker EQUALS (2 + 2 - 2)
  DECLARE tempBooleanA EQUALS FALSE
  DECLARE tempBooleanB EQUALS FALSE
  DECLARE tempBooleanC EQUALS FALSE
  DECLARE tempBooleanD EQUALS FALSE
  DECLARE tempBooleanE EQUALS FALSE
  DECLARE tempBooleanF EQUALS FALSE
  DECLARE outVal EQUALS 0

  IF (a = e) THEN
   IF NOT(c = a) THEN
    outVal = tempCompOne
   ELSE
    IF NOT((b < d AND d < f) OR (f < d AND d < b)) THEN
     outVal = tempCompOne
    ELSE
     outVal = resultInitializer
    END IF
   END IF
  ELSE
   IF (b = f) THEN
    IF NOT(d = b) THEN
     outVal = tempCompOne
    ELSE
     IF NOT((a < c AND c < e) OR (e < c AND c < a)) THEN
      outVal = tempCompOne
     ELSE
      outVal = resultInitializer
     END IF
    END IF
   ELSE
    IF (c + d = e + f) THEN
     IF NOT(a + b = c + d) THEN
      outVal = tempCompOne
     ELSE
      IF NOT(((c < a AND a < e AND d < b AND b < f) OR (e < a AND a < c AND f < b AND b < d))) THEN
       outVal = tempCompOne
      ELSE
       outVal = resultInitializer
      END IF
     END IF
    ELSE
     IF (c - d = e - f) THEN
      IF NOT(a - b = c - d) THEN
       outVal = tempCompOne
      ELSE
       IF NOT(((c < a AND a < e AND d > b AND b > f) OR (e < a AND a < c AND f > b AND b > d))) THEN
        outVal = tempCompOne
       ELSE
        outVal = resultInitializer
       END IF
      END IF
     ELSE
      outVal = resultInitializer
     END IF
    END IF
   END IF
  END IF
  RETURN outVal
 END FUNCTION
END CLASS
