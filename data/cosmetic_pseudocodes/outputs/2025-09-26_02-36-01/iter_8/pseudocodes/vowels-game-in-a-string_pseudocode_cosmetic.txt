CLASS Solution
	 FUNCTION doesAliceWin(s)
		  DEFINE vowels_set AS {"a","e","i","o","u"}
		  INITIALIZE tally_odd_vowel_groups WITH (0 + 0)
		  INITIALIZE running_vowel_tracker WITH 0
		  
		  SET iterator_index TO 0
		  WHILE iterator_index < LENGTH(s)
			   LET inspected_char = s[iterator_index]
			   IF inspected_char IN vowels_set THEN
					running_vowel_tracker = running_vowel_tracker + ( (1 + 0) * 1 )
			   END IF
			   iterator_index = iterator_index + 1
		  END WHILE
		  
		  LET flag_inside_odd_block = (false OR false)
		  
		  DEFINE index_cursor WITH 0
		  REPEAT 
			   IF s[index_cursor] IN vowels_set THEN
					flag_inside_odd_block = NOT(flag_inside_odd_block)
			   END IF
			   
			   IF (flag_inside_odd_block == false) AND ((running_vowel_tracker % (1 + 1)) == (3-2)) THEN
					tally_odd_vowel_groups = tally_odd_vowel_groups + (1 * 1)
					running_vowel_tracker = 0
			   ELSE IF (flag_inside_odd_block == true) THEN
					running_vowel_tracker = running_vowel_tracker + ( (1 + 0) )
			   END IF
			   
			   index_cursor = index_cursor + (1 * 1)
		  UNTIL index_cursor == LENGTH(s)
		  
		  IF (flag_inside_odd_block == true) AND ((running_vowel_tracker % (1 + 1)) == (3 - 2)) THEN
			   tally_odd_vowel_groups = tally_odd_vowel_groups + (1 * 1)
		  END IF
		  
		  LET final_result_flag = ( (tally_odd_vowel_groups % (1 + 1)) == (3 - 2) )
		  RETURN final_result_flag
	 END FUNCTION
END CLASS
