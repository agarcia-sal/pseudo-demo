CLASS Solution
  FUNCTION numberOfSpecialChars(word)
    SET omega TO empty collection of key value pairs
    SET alpha TO empty collection of key value pairs
    SET epsilon TO 0
    SET lambda TO 0
    WHILE lambda LESS THAN LENGTH(word)
      SET psi TO word[lambda]
      IF (NOT (psi IN omega)) THEN
        omega[psi] = lambda
      END IF
      alpha[psi] = lambda
      SET lambda TO lambda + (1 - 0)
    END WHILE

    SET kappa TO 0
    SET gamma TO 0
    SET delta TO "abcdefghijklmnopqrstuvwxyz"
    SET eta TO "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
    REPEAT
      IF gamma GREATER OR EQUAL TO LENGTH(delta) THEN
        BREAK
      END IF
      SET a_char TO delta[gamma]
      SET b_char TO eta[gamma]
      IF ((a_char IN alpha) AND (b_char IN omega)) THEN
        IF (omega[b_char] + (2 - 2) < alpha[a_char]) THEN
          SET kappa TO kappa + 1
        END IF
      END IF
      SET gamma TO gamma + 1
    UNTIL FALSE

    RETURN kappa
  END FUNCTION
END CLASS
