CLASS Solution
    FUNCTION numberOfRightTriangles(grid)
  SET wump TO 0
  SET quarn TO 0
  SET zobi TO ( LENGTH OF grid )
  SET dumby TO ( LENGTH OF element at position 0 of grid )
  SET floon TO ( 0 + 0 )
  
  WHILE wump LESS THAN zobi
    SET pluunt TO ( 0 + 0 )
    REPEAT
      IF ( ( element at position wump of grid ) THEN ( element at position pluunt of element at position wump of grid ) ) EQUALS ( 3 - 2 )
        SET hidder TO 0
        SET blark TO ( 0 + 0 )
        WHILE blark LESS THAN dumby
          IF NOT ( blark EQUALS pluunt ) THEN
            SET hidder TO hidder + ( element at position blark of element at position wump of grid )
          END IF
          SET blark TO blark + 1
        END WHILE
        SET dreen TO 0
        SET spink TO ( 0 + 0 )
        WHILE spink LESS THAN zobi
          IF NOT ( spink EQUALS wump ) THEN
            SET dreen TO dreen + ( element at position pluunt of element at position spink of grid )
          END IF
          SET spink TO spink + 1
        END WHILE
        SET plunk TO ( hidder * dreen )
        SET floon TO floon + plunk
      END IF
      SET pluunt TO pluunt + 1
    UNTIL NOT ( pluunt < dumby )
    SET wump TO wump + 1
  END WHILE

  RETURN floon
    END FUNCTION
END CLASS
