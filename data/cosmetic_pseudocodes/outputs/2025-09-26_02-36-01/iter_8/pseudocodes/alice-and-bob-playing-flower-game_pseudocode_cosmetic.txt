CLASS Solution  
  FUNCTION flowerGame(n, m)  
    SET alpha_count TO (n + (3 - 2)) / 2  
    SET beta_count TO n / 2  
    SET gamma_count TO (m + (3 - 2)) / 2  
    SET delta_count TO m / 2  
    SET intermediate_sum1 TO alpha_count * delta_count  
    SET intermediate_sum2 TO beta_count * gamma_count  
    SET result_pairs TO intermediate_sum1 + intermediate_sum2  
    RETURN result_pairs  
  END FUNCTION  
END CLASS
