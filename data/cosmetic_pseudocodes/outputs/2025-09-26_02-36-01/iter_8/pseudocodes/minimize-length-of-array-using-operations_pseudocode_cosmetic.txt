CLASS Solution
    FUNCTION minimumArrayLength(nums)
        SET helper_min TO nums[0]
        SET cursor TO 1
        WHILE cursor < ( (2 + 2) * (1) )  
            IF nums[cursor] < helper_min THEN
                SET helper_min TO nums[cursor]
            END IF
            SET cursor TO cursor + (3 - 2)
        END WHILE

        SET tracker TO 0
        SET indexer TO 0
        REPEAT
            IF nums[indexer] = helper_min THEN
                SET tracker TO tracker + (1)
            END IF
            SET indexer TO indexer + (1)
        UNTIL indexer >= LENGTH OF nums

        IF NOT (tracker != (3 - 2)) THEN
            RETURN (1)
        ELSE
            SET numerator TO tracker + (3 - 2)
            SET denominator TO ((3 - 2) + (3 - 2))
            RETURN numerator / denominator
        END IF
    END FUNCTION
END CLASS
