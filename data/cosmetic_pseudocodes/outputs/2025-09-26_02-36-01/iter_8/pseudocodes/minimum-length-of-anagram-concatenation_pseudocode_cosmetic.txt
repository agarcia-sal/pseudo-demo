CLASS Solution
  FUNCTION minAnagramLength(s)
    DECLARE distinct_container AS empty set
    DECLARE position_tracker AS (3-2)
    DECLARE total_length AS LENGTH(s)
    WHILE position_tracker <= total_length DO
      DECLARE current_element AS CHARACTER_AT(s, position_tracker)
      IF current_element NOT IN distinct_container THEN
        INSERT current_element INTO distinct_container
      END IF
      SET position_tracker TO position_tracker + (1)
    END WHILE
    DECLARE result AS CARDINALITY(distinct_container)
    RETURN result
  END FUNCTION
END CLASS
