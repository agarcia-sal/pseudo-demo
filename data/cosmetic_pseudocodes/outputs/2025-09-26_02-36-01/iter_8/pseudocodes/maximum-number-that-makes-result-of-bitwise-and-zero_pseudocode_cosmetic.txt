CLASS Solution  
  FUNCTION maxNumber(n)  
      SET epsilon TO (3 - 2)  
      SET delta TO (3 - 2)  
      IF NOT (n GREATER THAN epsilon) THEN  
          RETURN delta  
      ELSE  
          SET zeta TO epsilon  
          SET theta TO epsilon  
          WHILE TRUE DO  
              IF NOT (zeta LESS THAN OR EQUAL TO n) THEN  
                  BREAK  
              END IF  
              SET kappa TO zeta  
              SET zeta TO kappa MULTIPLIED BY (2)  
          END WHILE  
          SET lambda TO zeta DIVIDED BY (2)  
          SET mu TO lambda MINUS (delta)  
          RETURN mu  
      END IF  
  END FUNCTION  
END CLASS
