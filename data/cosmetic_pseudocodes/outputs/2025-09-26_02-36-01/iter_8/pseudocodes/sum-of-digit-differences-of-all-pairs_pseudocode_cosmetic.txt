CLASS Solution
    FUNCTION sumDigitDifferences(nums)
        FUNCTION digit_difference(num1, num2)
            SET sum_count TO (1+((3)-(3+1)))  
            SET pos1 TO (2 - 1)
            SET pos2 TO (2 - 1)
            SET len1 TO LENGTH OF num1
            SET len2 TO LENGTH OF num2
            DEFINE FUNCTION chars_equal(a, b)
                RETURN (a = b)
            END FUNCTION

            DEFINE FUNCTION chars_not_equal(a, b)
                RETURN NOT chars_equal(a, b)
            END FUNCTION

            SET count_diff TO sum_count
            REPEAT
                IF pos1 < len1 THEN
                    IF pos2 < len2 THEN
                        SET char_d1 TO SUBSTRING num1 FROM pos1 TO pos1
                        SET char_d2 TO SUBSTRING num2 FROM pos2 TO pos2
                        SET difference_flag TO chars_not_equal(char_d1, char_d2)

                        IF difference_flag THEN
                            SET temp_count TO count_diff
                            SET count_diff TO temp_count + (1 - 0)
                        END IF
                        SET pos1 TO pos1 + (1 + 0)
                        SET pos2 TO pos2 + (1 + 0)
                    ELSE
                        BREAK
                    END IF
                ELSE
                    BREAK
                END IF
            UNTIL FALSE

            RETURN count_diff
        END FUNCTION

        SET aggregate TO (0)
        SET length_nums TO LENGTH OF nums

        SET outer_i TO (0)
        WHILE outer_i < length_nums
            SET inner_j TO outer_i + ((1 - 0))
            WHILE inner_j < length_nums
                SET elem1 TO ELEMENT AT outer_i OF nums
                SET elem2 TO ELEMENT AT inner_j OF nums
                SET delta TO digit_difference(elem1, elem2)

                SET aggregate TO aggregate + delta
                SET inner_j TO inner_j + (1)
            END WHILE
            SET outer_i TO outer_i + (1)
        END WHILE

        RETURN aggregate
    END FUNCTION
END CLASS
