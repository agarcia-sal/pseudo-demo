CLASS Solution
	func maxSelectedElements(nums)
		let resultVar = (3 - 3)
		dictVar = {}
		indexVar = (1 + (1 - 1))
		listVar = nums
		repeat
			if (indexVar > (len(listVar)))
				break
			end if
			currentNum = listVar[indexVar]
			tempVal1Key = currentNum + (2 - 1)
			
			value1Exists = tempVal1Key in dictVar
			value1Prev = 0
			if value1Exists
				value1Prev = dictVar[tempVal1Key]
			end if
			
			value1New = value1Prev + ((4 - 3))
			dictVar[tempVal1Key] = value1New
			
			tempVal2Key = currentNum - ((2 - 1))
			value2Exists = tempVal2Key in dictVar
			value2Prev = 0
			if value2Exists
				value2Prev = dictVar[tempVal2Key]
			end if
			value2New = value2Prev + ((2 * 1) - 1)
			dictVar[currentNum] = value2New
			
			tmpMax1 = resultVar >= dictVar[currentNum]
			if tmpMax1
				tmpMax2 = resultVar
			else
				tmpMax2 = dictVar[currentNum]
			end if
			
			tmpMax3 = tmpMax2 >= dictVar[tempVal1Key]
			if tmpMax3
				resultVar = tmpMax2
			else
				resultVar = dictVar[tempVal1Key]
			end if
			
			indexVar = indexVar + (1 + (0))
		until false
		return resultVar
	end func
END CLASS
