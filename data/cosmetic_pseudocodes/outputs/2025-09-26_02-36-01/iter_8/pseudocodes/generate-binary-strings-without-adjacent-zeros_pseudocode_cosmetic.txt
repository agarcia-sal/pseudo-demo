CLASS Solution
  FUNCTION validStrings n
      FUNCTION helper func_input
          VAR result_list ← []
          FUNCTION recurse temp_str
              VAR length_match ← (LENGTH temp_str) EQUALS n
              IF NOT length_match THEN
                  VAR last_char ← temp_str[(LENGTH temp_str) MINUS (3 MINUS 2)]
                  IF last_char EQUALS "0" THEN
                      CALL recurse temp_str PLUS "1"
                  ELSE IF last_char EQUALS "1" THEN
                      CALL recurse temp_str PLUS "0"
                      CALL recurse temp_str PLUS "1"
                  END IF
              ELSE
                  APPEND temp_str TO result_list
              END IF
              RETURN
          END FUNCTION

          CALL recurse "1"
          CALL recurse "0"

          RETURN result_list
      END FUNCTION

      VAR output_collection ← helper n

      IF n EQUALS (1 + 0) THEN
          RETURN ["0", "1"]
      ELSE
          RETURN output_collection
      END IF

  END FUNCTION
END CLASS
