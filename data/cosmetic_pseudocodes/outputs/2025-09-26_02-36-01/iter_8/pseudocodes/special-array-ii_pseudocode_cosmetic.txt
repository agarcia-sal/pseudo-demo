CLASS Solution
  FUNCTION isArraySpecial(nums, queries)
    SET epsilon TO 0
    SET omega TO []
    
    SET delta TO LENGTH OF nums
    SET iota TO 0
    WHILE iota < delta
        SET kappa TO nums[iota] MODULO (1 + epsilon + 1)
        APPEND kappa TO omega
        SET iota TO iota + 1
    END WHILE
    
    SET alpha TO []
    SET lambda TO 0
    WHILE lambda < delta
        APPEND 0 TO alpha
        SET lambda TO lambda + 1
    END WHILE
    
    SET theta TO 1
    WHILE theta < delta
        IF omega[theta] EQUALS omega[theta - 1]
          THEN SET alpha[theta] TO alpha[theta - 1] + 1
          ELSE SET alpha[theta] TO alpha[theta - 1]
        END IF
        SET theta TO theta + 1
    END WHILE
    
    SET beta TO []
    FOR EACH mu, nu IN queries
        IF mu EQUALS nu
          THEN APPEND True TO beta
          ELSE
            SET sigma TO 0
            IF mu GREATER THAN 0
              THEN SET sigma TO alpha[mu - 1]
              ELSE SET sigma TO 0
            END IF
            SET rho TO alpha[nu] - sigma
            APPEND (rho EQUALS 0) TO beta
        END IF
    END FOR
    
    RETURN beta
  END FUNCTION
END CLASS
