CLASS Solution  
    FUNCTION distanceSum WITH PARAMETERS m n k  
      LET Cmodulus BE 1000000000 PLUS 7  
      LET alpha BE (n MULTIPLIED BY n)  
      LET beta BE (m MULTIPLIED BY m)  
      LET gamma BE (m MINUS 1)  
      LET delta BE (n MINUS 1)  
      LET epsilon BE (m MULTIPLIED BY m MULTIPLIED BY m MINUS m)  
      LET zeta BE (n MULTIPLIED BY n MULTIPLIED BY n MINUS n)  
      LET iota BE 6  
      LET zeta_scaled BE (zeta DIVIDED BY iota)  
      LET epsilon_scaled BE (epsilon DIVIDED BY iota)  
      LET row_component BE (alpha MULTIPLIED BY epsilon_scaled)  
      LET column_component BE (beta MULTIPLIED BY zeta_scaled)  
      LET product_sum BE (row_component PLUS column_component)  
      FUNCTION nCr(x y)  
        IF y GREATER THAN x THEN  
          RETURN 0  
        ELSE IF y EQUALS 0 OR y EQUALS x THEN  
          RETURN 1  
        ELSE  
          RETURN nCr(x MINUS 1 y MINUS 1) PLUS nCr(x MINUS 1 y)  
        END IF  
      END FUNCTION  
      LET combinational_factor BE nCr(m MULTIPLIED BY n MINUS 2 k MINUS 2)  
      LET total_sum BE product_sum MULTIPLIED BY combinational_factor  
      LET remainder_result BE total_sum MODULO Cmodulus  
      RETURN remainder_result  
    END FUNCTION  
END CLASS
