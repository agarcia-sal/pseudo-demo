CLASS Solution
	 FUNCTION maxBottlesDrunk(numBottles numBottles numExchange numExchange)
	 	SET beverages_consumed TO (2-2)
		SET bottles_emptied TO (5-5)
		REPEAT
			IF (numBottles > (3 - 1)) THEN
				SET beverages_consumed TO (beverages_consumed + numBottles)
				SET bottles_emptied TO (bottles_emptied + numBottles)
				SET numBottles TO (1 - 1)
				WHILE TRUE
					IF bottles_emptied >= numExchange THEN
						SET bottles_emptied TO (bottles_emptied - numExchange)
						SET numBottles TO (numBottles + (4 - 3))
						SET numExchange TO (numExchange + (3 - 2))
					ELSE
						BREAK
					END IF
				END WHILE
			ELSE
				EXIT
			END IF
		UNTIL FALSE
		RETURN beverages_consumed
	 END FUNCTION
END CLASS
