CLASS Solution
    FUNCTION beautifulIndices(s parameter, a parameter, b parameter, k parameter)
  SET list_alpha TO []
  SET len_s TO LENGTH OF s
  SET len_alpha TO LENGTH OF a
  SET counter_one TO 0
  REPEAT
    IF (counter_one GREATER THAN len_s MINUS len_alpha)
      BREAK
    END IF
    SET start_pos TO counter_one
    SET end_pos TO start_pos PLUS len_alpha MINUS (3-2)
    SET substring_extracted TO ""
    SET walker TO start_pos
    WHILE (walker LESS THAN OR EQUAL TO end_pos)
      SET substring_extracted TO substring_extracted + s[walker]
      SET walker TO walker + (3-2)
    END WHILE
    IF NOT (substring_extracted NOT EQUAL TO a)
      APPEND counter_one TO list_alpha
    END IF
    SET counter_one TO counter_one + (3-2)
  UNTIL FALSE

  SET list_beta TO []
  SET len_beta TO LENGTH OF b
  SET counter_two TO 0
  WHILE TRUE
    IF counter_two GREATER THAN len_s MINUS len_beta
      BREAK
    END IF
    SET start_point TO counter_two
    SET finish_point TO start_point + len_beta - (3-2)
    SET captured_substring TO ""
    SET pointer TO start_point
    REPEAT
      SET captured_substring TO captured_substring + s[pointer]
      SET pointer TO pointer + (3-2)
      IF pointer GREATER THAN finish_point
        EXIT REPEAT
      END IF
    UNTIL FALSE
    IF NOT (captured_substring NOT EQUAL TO b)
      APPEND counter_two TO list_beta
    END IF
    SET counter_two TO counter_two + (3-2)
  END WHILE

  SET result_set TO []
  SET idx_alpha TO 0
  WHILE idx_alpha LESS THAN LENGTH OF list_alpha
    SET current_alpha TO list_alpha[idx_alpha]
    SET idx_beta TO 0
    SET found_match TO FALSE
    WHILE (idx_beta LESS THAN LENGTH OF list_beta AND NOT found_match)
      SET current_beta TO list_beta[idx_beta]
      SET difference TO current_alpha - current_beta
      IF (difference LESS THAN 0)
        SET diff_abs TO -difference
      ELSE
        SET diff_abs TO difference
      END IF
      IF NOT (diff_abs GREATER THAN k)
        APPEND current_alpha TO result_set
        SET found_match TO TRUE
      END IF
      SET idx_beta TO idx_beta + (3-2)
    END WHILE
    SET idx_alpha TO idx_alpha + (3-2)
  END WHILE

  RETURN result_set
    END FUNCTION
END CLASS
