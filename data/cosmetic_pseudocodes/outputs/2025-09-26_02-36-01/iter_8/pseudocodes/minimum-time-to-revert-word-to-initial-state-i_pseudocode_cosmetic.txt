CLASS Solution
   FUNCTION minimumTimeToInitialState(word PARAMETER k)
      SET totalLength TO (1 + (1 + (0)))
      SET incrementor TO (1 + (1 + (0)))
      SET lengthOfWord TO LENGTH OF word
      SET resultTime TO (1 + (0))
      REPEAT
         SET multiplierCheck TO (resultTime * k)
         SET suffixStartIndex TO multiplierCheck
         SET prefixStartIndex TO (1 + (0))
         SET prefixEndIndex TO (lengthOfWord - multiplierCheck)

         IF (multiplierCheck >= lengthOfWord) THEN
             RETURN resultTime
         ELSE
            SET suffixSubstring TO SUBSTRING of word FROM suffixStartIndex TO lengthOfWord
            SET prefixSubstring TO SUBSTRING of word FROM prefixStartIndex TO prefixEndIndex
            IF NOT (suffixSubstring <> prefixSubstring) THEN
               RETURN resultTime
            END IF
         END IF
         SET resultTime TO (resultTime + incrementor)
      UNTIL (FALSE)
   END FUNCTION
END CLASS
