CLASS Enemy 
    FUNCTION __init__(self, damage, timeTakenDown) 
        SET self.damage TO damage 
        SET self.timeTakenDown TO timeTakenDown 
    END FUNCTION 
END CLASS 

CLASS Solution 
    FUNCTION minDamage(self, power, damage, health) 
        SET kxjvn TO (0) 
        SET hqgma TO (0) 
        SET nqpcl TO (0) 
        SET wzmrf TO [] 
        FUNCTION _sumList(lst) 
            SET fycze TO 0 
            SET zikoq TO 0 
            WHILE zikoq < LENGTH(lst) 
                SET fycze TO fycze + lst[zikoq] 
                SET zikoq TO zikoq + 1 
            END WHILE 
            RETURN fycze 
        END FUNCTION 
        SET hqgma TO _sumList(damage) 
        SET nqpcl TO 0 
        WHILE nqpcl < LENGTH(damage) 
            SET sxupv TO damage[nqpcl] 
            SET ediwr TO health[nqpcl] 
            SET kvzfu TO ((ediwr + power) - 1) 
            SET ufvnm TO (kvzfu // power) 
            SET sdrwo TO NEW Enemy(sxupv, ufvnm) 
            APPEND sdrwo TO wzmrf 
            SET nqpcl TO nqpcl + 1 
        END WHILE 
        FUNCTION _compareEnemies(a, b) 
            SET ulnbt TO (a.damage * b.timeTakenDown) 
            SET prxgz TO (b.damage * a.timeTakenDown) 
            IF ulnbt < prxgz THEN RETURN +1 
            ELSE IF ulnbt > prxgz THEN RETURN -1 
            ELSE RETURN 0 
        END FUNCTION 
        SET sxupp TO True 
        WHILE sxupp 
            SET sxupp TO False 
            SET tovzm TO 0 
            WHILE tovzm < LENGTH(wzmrf) - 1 
                IF _compareEnemies(wzmrf[tovzm], wzmrf[tovzm + 1]) > 0 THEN 
                    SET tmpx TO wzmrf[tovzm] 
                    SET wzmrf[tovzm] TO wzmrf[tovzm + 1] 
                    SET wzmrf[tovzm + 1] TO tmpx 
                    SET sxupp TO True 
                END IF 
                SET tovzm TO tovzm + 1 
            END WHILE 
        END WHILE 
        SET qlyxv TO 0 
        REPEAT 
            IF qlyxv >= LENGTH(wzmrf) THEN 
                BREAK 
            END IF 
            SET xwgnz TO wzmrf[qlyxv] 
            SET kxjvn TO kxjvn + (hqgma * xwgnz.timeTakenDown) 
            SET hqgma TO hqgma - xwgnz.damage 
            SET qlyxv TO qlyxv + 1 
        UNTIL False 
        RETURN kxjvn 
    END FUNCTION 
END CLASS
