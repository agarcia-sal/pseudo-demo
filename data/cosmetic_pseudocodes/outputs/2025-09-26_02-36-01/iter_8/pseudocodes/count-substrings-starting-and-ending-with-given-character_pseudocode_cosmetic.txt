CLASS Solution
{
  FUNCTION countSubstrings(s c)
  {
    DECLARE tally INITIALIZED AS (0)
    DECLARE size INITIALIZED AS (0)
    DECLARE occurrence INITIALIZED AS (0)
    DECLARE output INITIALIZED AS (0)
    
    size = 0
    WHILE size < (LENGTH of s)
    {
      IF s[size] == c
      {
        occurrence = occurrence + (1)
      }
      size = size + (1)
    }
    
    tally = occurrence * ((occurrence + 1) / 2)
    
    output = tally
    
    RETURN output
  }
}
