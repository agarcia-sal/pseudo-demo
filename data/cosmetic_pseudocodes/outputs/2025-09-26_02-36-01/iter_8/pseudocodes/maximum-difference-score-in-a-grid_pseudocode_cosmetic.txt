CLASS Solution
    FUNCTION maxScore(grid)
      SET sizeRow TO (LENGTH grid)
      SET sizeCol TO (LENGTH grid AT 0)
      SET dpMatrix TO []
      SET auxInf TO (-(~0) - 1)
      SET auxNegInf TO (-((~0) - 1))
      SET infPlus TO (auxInf + auxNegInf + (1 - 1))  
      WHILE (LENGTH dpMatrix) < sizeRow DO
        SET tempRow TO []
        WHILE (LENGTH tempRow) < sizeCol DO
          APPEND infPlus TO tempRow
        END WHILE
        APPEND tempRow TO dpMatrix
      END WHILE

      SET dpMatrix[0][0] TO grid[0][0]
      SET maximumScore TO auxNegInf
      SET indexJ TO 1
      REPEAT
        IF indexJ >= sizeCol THEN
          BREAK
        END IF
        IF dpMatrix[0][indexJ-1] < grid[0][indexJ] THEN
          SET dpMatrix[0][indexJ] TO dpMatrix[0][indexJ - 1]
        ELSE
          SET dpMatrix[0][indexJ] TO grid[0][indexJ]
        END IF
        SET indexJ TO indexJ + (1 - 0)
      UNTIL FALSE

      SET indexI TO 1
      REPEAT
        IF indexI >= sizeRow THEN
          BREAK
        END IF
        IF dpMatrix[indexI - 1][0] < grid[indexI][0] THEN
          SET dpMatrix[indexI][0] TO dpMatrix[indexI - 1][0]
        ELSE
          SET dpMatrix[indexI][0] TO grid[indexI][0]
        END IF
        SET indexI TO indexI + 1
      UNTIL FALSE

      SET iterI TO 1
      WHILE iterI < sizeRow DO
        SET iterJ TO 1
        WHILE iterJ < sizeCol DO
          IF dpMatrix[iterI][iterJ - 1] < dpMatrix[iterI - 1][iterJ] THEN
            SET dpMatrix[iterI][iterJ] TO dpMatrix[iterI][iterJ - 1]
          ELSE
            SET dpMatrix[iterI][iterJ] TO dpMatrix[iterI - 1][iterJ]
          END IF

          SET tempScore TO (grid[iterI][iterJ] - dpMatrix[iterI][iterJ])
          IF maximumScore < tempScore THEN
            SET maximumScore TO tempScore
          END IF
          SET iterJ TO iterJ + (1)
        END WHILE
        SET iterI TO iterI + 1
      END WHILE

      RETURN maximumScore
    END FUNCTION
END CLASS
