CLASS Solution
    FUNCTION minOperations(k)
      SET omega TO 9999999999
      SET beta TO 1
      SET lambda TO TRUNCATE(POWER(k, 0.5))
      SET upsilon TO beta + lambda
      WHILE beta < upsilon
        SET alpha TO (k + beta - (1)) \ beta
        SET gamma TO (beta - (1)) + (alpha - (1))
        IF NOT (gamma >= omega)
          SET omega TO gamma
        END IF
        SET beta TO beta + (1)
      END WHILE
      RETURN omega
    END FUNCTION
END CLASS
