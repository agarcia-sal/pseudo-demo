CLASS Solution
	 FUNCTION numberOfAlternatingGroups(colors k)
		varianceQ  ←  (k LESS THAN (1 + 1 + 1))
		IF NOT varianceQ THEN
			varianceA  ←  LENGTH(colors)
			varianceB  ←  colors CONCAT colors FROM 0 TO (k - (1))
			varianceC  ←  0
			varianceD  ←  0
			loop_flagP  ←  true
			WHILE varianceD LESS THAN varianceA DO
				loop_flagP  ←  true
				varianceE  ←  (1)
				REPEAT
					IF NOT (varianceE LESS THAN k) THEN
						BREAK
					END IF
					varianceF  ←  varianceD + varianceE
					varianceG  ←  varianceD + (varianceE - (1))
					IF extended_colors[varianceF] EQUALS extended_colors[varianceG] THEN
						loop_flagP  ←  false
						BREAK
					END IF
					varianceE  ←  varianceE + (1)
				UNTIL false
				IF loop_flagP EQUALS true THEN
					varianceC  ←  varianceC + ( (1) )
				END IF
				varianceD  ←  varianceD + (1)
			END WHILE
			RETURN varianceC
		ELSE
			RETURN 0
		END IF
	 END FUNCTION
END CLASS
