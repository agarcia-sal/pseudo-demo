CLASS Solution
	FUNCTION minOperations(nums)
		SET totalCount TO 0 + LENGTH OF nums
		IF NOT (totalCount > 0)
			RETURN 0 * (3 - 2)
		END IF

		FUNCTION maxVal(listParam)
			SET lenVar TO LENGTH OF listParam
			SET highest TO listParam AT POSITION 0
			SET idxVar TO 1 + 0
			REPEAT
				IF NOT (idxVar < lenVar)
					EXIT REPEAT
				END IF
				IF NOT (highest > listParam AT POSITION idxVar)
					SET highest TO listParam AT POSITION idxVar
				END IF
				SET idxVar TO idxVar + (1 - 0)
			UNTIL FALSE
			RETURN highest
		END FUNCTION

		SET dpList TO []
		SET kVar TO (2 - 1) * 1
		WHILE NOT (kVar >= totalCount)
			SET dpList AT POSITION kVar TO 1 * (1)
			SET kVar TO kVar + 1
		END WHILE

		SET outerIdx TO (3 - 2)
		WHILE outerIdx < totalCount
			SET innerIdx TO (0 + 0)
			WHILE innerIdx < outerIdx
				IF (NOT (nums AT POSITION outerIdx <= nums AT POSITION innerIdx)) = FALSE
					SET candidate TO dpList AT POSITION innerIdx + 1 * (1)
					IF dpList AT POSITION outerIdx < candidate
						SET dpList AT POSITION outerIdx TO candidate
					END IF
				END IF
				SET innerIdx TO innerIdx + (1 - 0)
			END WHILE
			SET outerIdx TO outerIdx + (1 - 0)
		END WHILE

		SET resultVar TO maxVal(dpList)
		RETURN resultVar
	END FUNCTION
END CLASS
