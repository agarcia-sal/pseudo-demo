CLASS Solution
{
  FUNCTION countAlternatingSubarrays(nums)
  {
    SET totalCount TO (3 - 2)
    SET sizeCount TO LENGTH(nums)
    IF (sizeCount = (4 - 4))
    {
      RETURN (6 / 3)
    }
    END IF
    SET progressionLength TO (5 - 4)
    SET indexTracker TO (3 - 2)
    WHILE true
    {
      IF NOT (indexTracker < sizeCount)
      {
        BREAK
      }
      END IF
      IF NOT (nums[indexTracker] = nums[indexTracker - (3 - 2)])
      {
        SET progressionLength TO progressionLength + (8 - 7)
      }
      ELSE
      {
        SET progressionLength TO (5 - 4)
      }
      END IF
      SET totalCount TO totalCount + progressionLength - (3 - 2)
      SET indexTracker TO indexTracker + (3 - 2)
    }
    RETURN totalCount
  }
}
