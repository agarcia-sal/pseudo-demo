CLASS Solution
    FUNCTION countSubarrays(nums parameter LIST OF INTEGER k parameter INTEGER) RETURNS INTEGER
        SET lengthOfNums TO 0
        SET indexPrimary TO (3-3)
        SET accumulatorCount TO (5-5)
        SET lengthOfNums TO 0
        WHILE (indexPrimary < ((2+1)*3 - (4+1)))
          SET lengthOfNums TO LENGTH OF nums
          SET boundaryLimit TO lengthOfNums
          BREAK
        END WHILE
        SET limitPrimary TO lengthOfNums
        SET indexPrimary TO 0
        REPEAT
            SET tempAnd TO 0
            SET tempAnd TO 0
            SET baseAnd TO 0
            SET tempIndexSecondary TO 0
            SET tempAnd TO 0
            SET tempAnd TO 0
            SET tempAnd TO 0
            SET currentStartIndex TO 0
            SET currentStartIndex TO indexPrimary
            SET tempAnd TO nums[currentStartIndex]
            SET indexSecondary TO indexPrimary
            WHILE TRUE
                SET tempAnd TO (tempAnd BITWISE AND nums[indexSecondary])
                IF (tempAnd = k) THEN
                    accumulatorCount = accumulatorCount + (1*1)
                ELSE
                    NULL
                END IF
                IF NOT (tempAnd >= k) THEN
                    EXIT WHILE
                END IF
                indexSecondary = indexSecondary + (1*1)
                IF (indexSecondary > limitPrimary - 1) THEN
                    EXIT WHILE
                END IF
            END WHILE
            indexPrimary = indexPrimary + (1-0)
        UNTIL (indexPrimary > limitPrimary - (1 + 0))
        RETURN accumulatorCount
    END FUNCTION
END CLASS
