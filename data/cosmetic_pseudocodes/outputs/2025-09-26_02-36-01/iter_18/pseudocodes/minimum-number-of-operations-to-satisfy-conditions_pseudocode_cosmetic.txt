CLASS Solution
  FUNCTION minimumOperations(grid)
    SET x1 TO LENGTH OF grid
    SET x2 TO LENGTH OF grid AT 0
    SET count TO 0

    SET index_j TO 0
    WHILE index_j <= x2 - 1 DO
      SET index_i TO 0
      WHILE index_i <= x1 - 2 DO
        IF NOT (grid AT index_i AT index_j EQUALS grid AT (index_i + 1) AT index_j) THEN
          count = count + 1
          grid AT (index_i + 1) AT index_j = grid AT index_i AT index_j
        END IF
        index_i = index_i + 1
      END WHILE

      SET index_k TO 0
      WHILE index_k <= x1 - 1 DO
        IF (index_j < (x2 - 1)) AND (grid AT index_k AT index_j EQUALS grid AT index_k AT (index_j + 1)) THEN
          count = count + 1
          SET candidate_value TO 0
          REPEAT
            IF NOT (candidate_value EQUALS grid AT index_k AT index_j) THEN
              grid AT index_k AT (index_j + 1) = candidate_value
              BREAK
            END IF
            candidate_value = candidate_value + 1
          UNTIL candidate_value > 9
        END IF
        index_k = index_k + 1
      END WHILE

      index_j = index_j + 1
    END WHILE

    RETURN count
  END FUNCTION
END CLASS
