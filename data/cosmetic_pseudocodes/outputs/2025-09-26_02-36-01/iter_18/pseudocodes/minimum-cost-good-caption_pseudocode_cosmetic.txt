CLASS Solution
    FUNCTION minCostGoodCaption(caption)
        
        SET omega TO LENGTH OF caption
        IF omega LESS THAN 3 THEN
            RETURN ""
        END IF
        
        SET xi TO ARRAY OF characters IN caption
        
        SET alpha TO 0
        REPEAT
            IF alpha GREATER THAN OR EQUAL TO omega THEN
                BREAK
            END IF
            
            SET beta TO alpha
            DEFINE PROCEDURE advance_loop()
                IF beta LESS THAN omega AND xi[beta] EQUALS xi[alpha] THEN
                    INCREMENT beta BY 1
                    CALL advance_loop()
                END IF
            END PROCEDURE
            CALL advance_loop()
            
            SET delta TO beta - alpha
            
            IF delta < 3 THEN
                IF alpha > 0 AND xi[alpha - 1] = xi[alpha] THEN
                    SET xi[alpha - 1] TO xi[alpha]
                    DECREMENT alpha BY 1
                    INCREMENT delta BY 1
                END IF
                
                IF beta < omega AND xi[beta - 1] = xi[beta] THEN
                    SET xi[beta] TO xi[beta - 1]
                    INCREMENT beta BY 1
                    INCREMENT delta BY 1
                END IF
                
                IF delta < 3 THEN
                    IF alpha > 0 THEN
                        SET kappa TO xi[alpha - 1]
                    ELSE
                        SET kappa TO xi[beta]
                    END IF
                    
                    IF kappa = 'a' THEN
                        SET kappa TO 'b'
                    ELSE 
                        IF kappa = 'z' THEN
                            SET kappa TO 'y'
                        ELSE
                            SET kappa TO CHAR(FUNCTION_CODE( kappa ) + 1)
                        END IF
                    END IF
                    
                    SET gamma TO alpha
                    WHILE gamma <= alpha + delta - 1
                        SET xi[gamma] TO kappa
                        INCREMENT gamma BY 1
                    END WHILE
                    
                    SET alpha TO alpha + delta
                ELSE
                    SET alpha TO beta
                END IF
                
            ELSE
                SET alpha TO beta
            END IF
            
        UNTIL FALSE
        
        RETURN JOIN(xi)
    END FUNCTION
END CLASS
