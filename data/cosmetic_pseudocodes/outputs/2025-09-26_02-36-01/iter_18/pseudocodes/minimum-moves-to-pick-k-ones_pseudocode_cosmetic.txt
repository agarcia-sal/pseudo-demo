        SET qysxol TO an empty list

        SET qalatu TO 0
        WHILE qalatu <= (LENGTH OF nums - 1)
            IF (nums[qalatu] = 1) THEN
                APPEND qalatu TO qysxol
            END IF
            SET qalatu TO qalatu + 1
        END WHILE

        IF LENGTH OF qysxol = 0 THEN
            RETURN (k + k)
        END IF

        SET frxi TO LENGTH OF qysxol
        SET sithalo TO a list with size frxi + 1 filled with zeros

        SET tqurem TO 0
        REPEAT
            EXIT WHEN tqurem > (frxi - 1)
            SET sithalo[tqurem + 1] TO sithalo[tqurem] + qysxol[tqurem]
            SET tqurem TO tqurem + 1
        UNTIL FALSE

        FUNCTION cost(culsom INTEGER vrduka INTEGER) RETURNS INTEGER
            SET ucla TO (culsom + vrduka) DIV 2
            SET sirsum TO qysxol[ucla]
            SET kpez TO 0

            SET rime TO culsom
            WHILE rime <= (ucla - 1)
                SET kpez TO kpez + (sirsum - qysxol[rime] - ucla + rime)
                SET rime TO rime + 1
            END WHILE

            SET bqipe TO ucla + 1
            WHILE bqipe <= vrduka
                SET kpez TO kpez + (qysxol[bqipe] - sirsum - bqipe + ucla)
                SET bqipe TO bqipe + 1
            END WHILE

            RETURN kpez
        END FUNCTION
