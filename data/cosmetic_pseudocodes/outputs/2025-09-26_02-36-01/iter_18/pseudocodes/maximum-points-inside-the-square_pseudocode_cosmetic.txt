CLASS Solution
    FUNCTION maxPointsInsideSquare(points s)
        SET alpha TO LENGTH OF s
        SET omega TO 0
        SET upsilon TO 0
        WHILE upsilon < alpha
            SET psi TO s[upsilon][0]
            SET mu TO s[upsilon][1]
            SET kappa TO IF ABS(psi) > ABS(mu) THEN ABS(psi) ELSE ABS(mu)

            SET chi TO EMPTY DICTIONARY
            SET rho TO 1
            SET theta TO 0
            REPEAT
                SET delta TO s[theta][0]
                SET epsilon TO s[theta][1]
                IF NOT (ABS(delta) > kappa OR ABS(epsilon) > kappa) THEN
                    SET sigma TO s[theta]
                    IF chi CONTAINS KEY sigma THEN
                        SET rho TO 0
                        SET theta TO alpha
                    ELSE
                        chi[sigma] = TRUE
                    END IF
                END IF
                SET theta TO theta + 1
            UNTIL theta >= alpha

            IF rho == 1 THEN
                IF omega < LENGTH OF chi THEN
                    SET omega TO LENGTH OF chi
                END IF
            END IF
            SET upsilon TO upsilon + 1
        END WHILE
        RETURN omega
    END FUNCTION
END CLASS
