CLASS Solution
    FUNCTION nonSpecialCount(a b)
        FUNCTION is_prime(p)
            IF NOT (p > 1) THEN
                RETURN false
            END IF
            IF p <= 3 THEN
                RETURN true
            END IF
            IF (p MOD 2 = 0) OR (p MOD 3 = 0) THEN
                RETURN false
            END IF

            i = 5
            REPEAT
                IF (p MOD i = 0) OR (p MOD (i + 2) = 0) THEN
                    RETURN false
                END IF
                i = i + 6
            UNTIL NOT (i * i <= p)

            RETURN true
        END FUNCTION

        x = CEIL(SQRT(a))
        y = FLOOR(SQRT(b))

        countSpecial = 0
        z = x
        WHILE z <= y
            flag = is_prime(z)
            IF flag = true THEN
                countSpecial = countSpecial + 1
            END IF
            z = z + 1
        END WHILE

        fullRange = b - a + 1
        result = fullRange - countSpecial

        RETURN result
    END FUNCTION
END CLASS
