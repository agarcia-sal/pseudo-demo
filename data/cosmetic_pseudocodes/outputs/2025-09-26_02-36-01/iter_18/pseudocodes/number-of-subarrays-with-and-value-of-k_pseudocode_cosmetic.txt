```pseudocode
CLASS Solution
    FUNCTION countSubarrays(nums parameter LIST OF INTEGER k parameter INTEGER) RETURNS INTEGER
        SET delta TO 0
        SET dim TO LENGTH OF nums
        SET idx_outer TO 0

        WHILE idx_outer < dim
            SET val_mask TO nums[idx_outer]
            SET idx_inner TO idx_outer

            REPEAT
                SET val_mask TO val_mask BITWISE AND nums[idx_inner]
                IF (val_mask = k) THEN
                    SET delta TO delta + 1
                END IF

                IF NOT (val_mask >= k) THEN
                    EXIT REPEAT
                END IF

                SET idx_inner TO idx_inner + 1
            UNTIL idx_inner = dim

            SET idx_outer TO idx_outer + 1
        END WHILE

        RETURN delta
    END FUNCTION
END CLASS
```
