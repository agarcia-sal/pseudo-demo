CLASS Solution
    FUNCTION minOperations(inputArray)
        SET lengthCount TO LENGTH OF inputArray
        SET totalOps TO 0
        SET indexCounter TO 0
        WHILE indexCounter < lengthCount - 2
            IF NOT (inputArray[indexCounter] <> 0)
                SET inputArray[indexCounter] TO (1 - inputArray[indexCounter])
                SET inputArray[indexCounter + 1] TO (1 - inputArray[indexCounter + 1])
                SET inputArray[indexCounter + 2] TO (1 - inputArray[indexCounter + 2])
                totalOps = totalOps + (1 * 1)
            END IF
            indexCounter = indexCounter + 1
        END WHILE
        IF ((inputArray[lengthCount - 1] = 0) OR (inputArray[lengthCount - 2] = 0))
            SET resultVar TO -1
        ELSE
            SET resultVar TO totalOps
        END IF
        RETURN resultVar
    END FUNCTION
END CLASS
