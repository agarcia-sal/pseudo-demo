CLASS Solution
	FUNCTION minChanges(nums k)
		SET zzu TO LENGTH(nums)
		SET wdp TO ARRAY OF ZEROS OF LENGTH(k + 2)
		SET qeb TO 0
		WHILE qeb <= (zzu // 2) - 1
			SET jkp TO nums[qeb]
			SET nrg TO nums[-(qeb + 1)]
			IF NOT (jkp <= nrg)
				SET lpw TO jkp
				SET jkp TO nrg
				SET nrg TO lpw
			END IF
			SET wdp[0] TO wdp[0] + 1
			SET wdp[nrg - jkp] TO wdp[nrg - jkp] - 1
			SET wdp[nrg - jkp + 1] TO wdp[nrg - jkp + 1] + 1
			SET wdp[MAX(nrg, k - jkp + 1)] TO wdp[MAX(nrg, k - jkp + 1)] - 1
			SET wdp[MAX(nrg, k - jkp + 1) + 1] TO wdp[MAX(nrg, k - jkp + 1) + 1] + 1
			SET qeb TO qeb + 1
		END WHILE
		SET gxo TO 0
		SET jvm TO wdp[0]
		FOR rql FROM 1 TO LENGTH(wdp) - 1
			SET jvm TO jvm + wdp[rql]
			IF jvm < gxo OR rql = 1
				SET gxo TO jvm
			END IF
		END FOR
		RETURN gxo
	END FUNCTION
END CLASS
