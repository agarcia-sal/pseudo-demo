CLASS Solution
    FUNCTION countPrefixSuffixPairs(words)
        DECLARE tally AS INTEGER
        SET tally = 0

        DECLARE registry AS MAPPING FROM STRING TO INTEGER WITH DEFAULT 0

        DECLARE idx_outer AS INTEGER
        SET idx_outer = LENGTH(words) - 1
        
        WHILE idx_outer >= 0
            DECLARE current_entry AS STRING
            SET current_entry = words[idx_outer]

            DECLARE key_iter AS STRING
            FOR key_iter IN registry KEYS
                DECLARE left_substring AS STRING
                DECLARE right_substring AS STRING

                SET left_substring = SUBSTRING(key_iter, 0, LENGTH(current_entry) - 1 + 1)   // inclusive index
                SET right_substring = SUBSTRING(key_iter, LENGTH(key_iter) - LENGTH(current_entry), LENGTH(key_iter) - 1 + 1)

                IF NOT (current_entry ≠ left_substring OR current_entry ≠ right_substring)
                    SET tally = tally + registry[key_iter]
                END IF
            END FOR

            registry[current_entry] = registry[current_entry] + 1

            SET idx_outer = idx_outer - 1
        END WHILE

        RETURN tally
    END FUNCTION
END CLASS
