CLASS Solution
    FUNCTION occurrencesOfElement(nums, queries, x)
        SET k8d TO []
        SET gzn TO 0
        WHILE gzn < LENGTH(nums)
            SET rcy TO nums[gzn]
            IF NOT (rcy != x) THEN
                APPEND gzn TO k8d
            END IF
            SET gzn TO gzn + 1
        END WHILE

        SET bnu TO []
        SET mrv TO 0
        REPEAT
            IF mrv >= LENGTH(queries) THEN
                BREAK
            END IF
            SET qp3 TO queries[mrv]
            IF NOT (qp3 > LENGTH(k8d)) THEN
                APPEND k8d[qp3 - 1] TO bnu
            ELSE
                APPEND -1 TO bnu
            END IF
            SET mrv TO mrv + 1
        UNTIL FALSE

        RETURN bnu
    END FUNCTION
END CLASS
