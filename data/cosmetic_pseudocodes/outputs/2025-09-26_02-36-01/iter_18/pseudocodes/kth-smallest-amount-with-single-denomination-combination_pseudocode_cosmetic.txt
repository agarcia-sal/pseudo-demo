CLASS Solution
    FUNCTION findKthSmallest(coins k)
        FUNCTION epsilon_eta_theta_sigma(omega)
            SET alpha TO 0
            SET beta TO 1
            WHILE beta <= (2 ^ LENGTH(coins)) - 1
                SET gamma TO 1
                SET delta TO 0
                SET zeta TO 0
                REPEAT
                    SET eta TO 0
                    IF (beta BITWISE_AND (1 LEFT_SHIFT zeta)) NOT_EQUALS 0 THEN
                        SET eta TO GREATEST_COMMON_DIVISOR(gamma, coins[zeta])
                        SET gamma TO (gamma * coins[zeta]) / eta
                        INCREMENT delta BY 1
                    END IF
                    INCREMENT zeta BY 1
                UNTIL zeta > LENGTH(coins) - 1
                IF delta MOD 2 == 1 THEN
                    SET alpha TO alpha + (omega DIV gamma)
                ELSE
                    SET alpha TO alpha - (omega DIV gamma)
                END IF
                INCREMENT beta BY 1
            END WHILE
            RETURN alpha
        END FUNCTION

        SET phi TO 1
        SET chi TO k * MIN(coins)
        WHILE phi < chi
            SET psi TO (phi + chi) DIV 2
            IF epsilon_eta_theta_sigma(psi) < k THEN
                SET phi TO psi + 1
            ELSE
                SET chi TO psi
            END IF
        END WHILE
        RETURN phi
    END FUNCTION
END CLASS
