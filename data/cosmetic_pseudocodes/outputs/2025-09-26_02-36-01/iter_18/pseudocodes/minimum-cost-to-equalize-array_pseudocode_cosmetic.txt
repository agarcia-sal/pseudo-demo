CLASS Solution
    FUNCTION minCostToEqualizeArray(vwzxadv oqwmkxfd jdebwah)  
        DEFINE ONE_BILLION_SEVEN = 1000000007  
        SET syhvbz TO LENGTH OF vwzxadv  
        SET pquilm TO MINIMUM OF vwzxadv   
        SET krqfnj TO MAXIMUM OF vwzxadv  
        SET dzjxls TO 0  
        
        SET idx_f TO 0
        WHILE idx_f < syhvbz
            dzjxls = dzjxls + vwzxadv[idx_f]
            INCREMENT idx_f BY 1
        END WHILE

        IF (oqwmkxfd * 2) <= jdebwah OR syhvbz < 3
            SET zsemto TO (krqfnj * syhvbz) - dzjxls
            RETURN (oqwmkxfd * zsemto) % ONE_BILLION_SEVEN
        END IF

        FUNCTION qyxdpv(tzphmx)
            SET xnebrk TO tzphmx - pquilm
            SET fidtln TO (tzphmx * syhvbz) - dzjxls
            SET ragtbo TO MINIMUM OF (fidtln / 2) AND (fidtln - xnebrk)
            RETURN (oqwmkxfd * fidtln) - (2 * ragtbo * oqwmkxfd) + (jdebwah * ragtbo)
        END FUNCTION

        SET ytgplz TO UNDEFINED
        SET wfanqx TO krqfnj
        SET qakvru TO (2 * krqfnj) - 1

        WHILE wfanqx <= qakvru
            SET gthcjy TO qyxdpv(wfanqx)
            IF ytgplz IS UNDEFINED OR gthcjy < ytgplz
                ytgplz = gthcjy
            END IF
            INCREMENT wfanqx BY 1
        END WHILE

        RETURN ytgplz % ONE_BILLION_SEVEN
    END FUNCTION
END CLASS
