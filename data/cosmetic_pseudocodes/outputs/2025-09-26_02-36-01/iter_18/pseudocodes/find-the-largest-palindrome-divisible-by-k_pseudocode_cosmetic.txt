CLASS Solution
    FUNCTION largestPalindrome(n k)
        DECLARE alpha INTEGER
        DECLARE beta STRING
        DECLARE gamma INTEGER
        DECLARE delta INTEGER
        DECLARE epsilon STRING
        DECLARE zeta STRING
        DECLARE eta INTEGER
        DECLARE theta INTEGER
        DECLARE i INTEGER
        DECLARE temp STRING
        DECLARE palindromeVal INTEGER
        DECLARE result STRING
        
        IF NOT (n NOT EQUALS 1) THEN
            i = 9
            WHILE i GREATER THAN 0 DO
                IF i MODULO k EQUALS 0 THEN
                    RETURN STRING(i)
                ENDIF
                i = i - 1
            ENDWHILE
            RETURN "0"
        ENDIF

        alpha = (n + 1) DIV 2
        beta = ""
        FOR gamma = 1 TO alpha DO
            beta = beta + "9"
        END FOR
        delta = INTEGER(beta)

        REPEAT
            IF n MODULO 2 EQUALS 0 THEN
                epsilon = STRING(delta)
                zeta = ""
                i = LENGTH(epsilon)
                WHILE i GREATER THAN 0 DO
                    zeta = zeta + SUBSTRING(epsilon, i, 1)
                    i = i - 1
                ENDWHILE
                palindromeVal = INTEGER(epsilon + zeta)
            ELSE
                epsilon = STRING(delta)
                theta = LENGTH(epsilon)
                temp = SUBSTRING(epsilon, 1, theta - 1)
                zeta = ""
                i = LENGTH(temp)
                WHILE i GREATER THAN 0 DO
                    zeta = zeta + SUBSTRING(temp, i, 1)
                    i = i - 1
                ENDWHILE
                palindromeVal = INTEGER(epsilon + zeta)
            ENDIF

            IF palindromeVal MODULO k EQUALS 0 THEN
                RETURN STRING(palindromeVal)
            ENDIF
            delta = delta - 1
        UNTIL delta LESS THAN 1

        RETURN "0"
    END FUNCTION
END CLASS
