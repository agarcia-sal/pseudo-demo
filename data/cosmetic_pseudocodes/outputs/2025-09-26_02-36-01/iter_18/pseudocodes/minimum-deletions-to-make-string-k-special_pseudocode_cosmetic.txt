CLASS Solution
    FUNCTION minimumDeletions(word, k)
        SET alpha_counts TO empty map
        FOR idx FROM 0 TO LENGTH(word) - 1
            SET ch TO word[idx]
            IF ch IN alpha_counts THEN
                SET alpha_counts[ch] TO alpha_counts[ch] + 1
            ELSE
                SET alpha_counts[ch] TO 1
            END IF
        END FOR

        SET freq_list TO empty list
        FOR each val IN alpha_counts values
            APPEND val TO freq_list
        END FOR

        FOR i FROM 0 TO LENGTH(freq_list) - 2
            FOR j FROM 0 TO LENGTH(freq_list) - 2 - i
                IF freq_list[j] > freq_list[j + 1] THEN
                    SET tmp TO freq_list[j]
                    SET freq_list[j] TO freq_list[j + 1]
                    SET freq_list[j + 1] TO tmp
                END IF
            END FOR
        END FOR

        SET min_del TO inf
        SET p TO 0
        WHILE p < LENGTH(freq_list)
            SET max_freq TO freq_list[p] + k
            SET delt TO 0

            SET q TO p + 1
            WHILE q < LENGTH(freq_list)
                IF freq_list[q] > max_freq THEN
                    SET delt TO delt + (freq_list[q] - max_freq)
                END IF
                SET q TO q + 1
            END WHILE

            SET r TO 0
            WHILE r < p
                SET delt TO delt + freq_list[r]
                SET r TO r + 1
            END WHILE

            IF delt < min_del THEN
                SET min_del TO delt
            END IF

            SET p TO p + 1
        END WHILE

        RETURN min_del
    END FUNCTION
END CLASS
