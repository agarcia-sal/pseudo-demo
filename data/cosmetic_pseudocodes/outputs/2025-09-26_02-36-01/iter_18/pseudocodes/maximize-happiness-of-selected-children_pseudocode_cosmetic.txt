```pseudocode
CLASS Solution
    FUNCTION maximumHappinessSum(happiness, k)
        reorder happiness descending
        SET pc TO 0
        SET ua TO 0
        SET e TO 0
        while ua < k do
            SET r TO happiness[ua] - e
            if NOT (r >= 0) then
                SET r TO 0
            end if
            SET pc TO pc + r
            SET e TO e + 1
            SET ua TO ua + 1
        end while
        RETURN pc
    END FUNCTION
END CLASS
```
