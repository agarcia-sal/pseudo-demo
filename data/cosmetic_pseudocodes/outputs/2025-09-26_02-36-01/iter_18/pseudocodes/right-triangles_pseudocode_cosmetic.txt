```pseudocode
CLASS Solution
	FUNCTION numberOfRightTriangles(grid)
		DECLARE a, b, c, d, e, f, g, h, total_result AS INTEGER
		a = LENGTH(grid)
		b = LENGTH(grid[0])
		total_result = 0

		c = 0
		WHILE c < a DO
			d = 0
			LOOP
				IF d >= b THEN
					EXIT
				END IF

				IF NOT (grid[c][d] = 1) THEN
					d = d + 1
					CONTINUE
				END IF

				e = 0
				f = 0
				g = 0
				LOOP
					IF g >= b THEN
						EXIT
					END IF

					IF g <> d THEN
						e = e + grid[c][g]
					END IF

					g = g + 1
				END LOOP

				h = 0
				LOOP
					IF h >= a THEN
						EXIT
					END IF

					IF h <> c THEN
						f = f + grid[h][d]
					END IF

					h = h + 1
				END LOOP

				total_result = total_result + e * f
				d = d + 1
			END LOOP

			c = c + 1
		END WHILE

		RETURN total_result
	END FUNCTION
END CLASS
```
