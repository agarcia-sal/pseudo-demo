```pseudocode
CLASS Solution
    FUNCTION minimumOperations(nums, target)
        SET u TO LENGTH(nums)
        SET v TO ABS(target[0] - nums[0])
        SET w TO 1
        WHILE w < u
            SET p TO target[w] - nums[w]
            SET q TO target[w - 1] - nums[w - 1]
            IF (p * q) > 0 THEN
                SET r TO ABS(p) - ABS(q)
                IF r > 0 THEN
                    SET v TO v + r
                END IF
            ELSE
                SET v TO v + ABS(p)
            END IF
            SET w TO w + 1
        END WHILE
        RETURN v
    END FUNCTION
END CLASS
```
