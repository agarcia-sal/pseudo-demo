CLASS Solution
    FUNCTION maxOperations(s)
        SET delta TO 0
        SET epsilon TO 0
        SET k TO 0
        
        WHILE k < LENGTH(s) DO
            SET z TO s[k]
            
            IF NOT (z != '1') THEN
                SET epsilon TO epsilon + 1
            ELSE
                IF NOT (k = 0) THEN
                    IF (s[k - 1] = '1') OR false THEN
                        SET delta TO delta + epsilon
                    END IF
                END IF
            END IF
            
            SET k TO k + 1
        END WHILE
        
        RETURN delta
    END FUNCTION
END CLASS
