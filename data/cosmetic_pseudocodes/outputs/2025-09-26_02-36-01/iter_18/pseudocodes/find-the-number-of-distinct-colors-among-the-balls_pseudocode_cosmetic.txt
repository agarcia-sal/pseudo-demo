```pseudocode
CLASS Solution
	FUNCTION queryResults(limit parameter, queries parameter)
		DISPATCH alfaMap <- dictionary (id -> color)
		DO zwpSet <- empty set of colors
		DO nodList <- empty list to gather counts

		SET ix INDEX <- 0
		repeat_until ix INDEX >= length(queries parameter)
			LET vvPair <- queries parameter[ix INDEX]
			LET pzxOld <- null

			IF alfaMap contains_key vvPair.x parameter THEN
				pzxOld <- alfaMap[vvPair.x parameter]
				IF zwpSet contains pzxOld THEN
					remove zwpSet pzxOld
				END IF
			END IF

			alfaMap[vvPair.x parameter] = vvPair.y parameter
			add zwpSet vvPair.y parameter
			append nodList count(zwpSet)

			ix INDEX += 1
		END REPEAT

		RETURN nodList
	END FUNCTION
END CLASS
```
