CLASS Solution
    FUNCTION kthCharacter(param_y INTEGER, param_z LIST OF INTEGER) RETURNS STRING
        SET var_f TO 1
        SET var_m TO NEW LIST

        SET var_n TO 0
        WHILE var_n < LENGTH(param_z) DO
            APPEND param_z[var_n] TO var_m
            IF NOT (param_z[var_n] <> 0) THEN
                SET var_f TO var_f * 2
            ELSE
                SET var_f TO var_f * 2
            END IF
            SET var_n TO var_n + 1
        END WHILE

        SET var_w TO 'a'

        SET var_r TO LENGTH(var_m) - 1
        REPEAT
            IF param_y <= (var_f / 2) THEN
                SET var_f TO var_f / 2
            ELSE
                SET param_y TO param_y - (var_f / 2)
                SET var_f TO var_f / 2
                IF var_m[var_r] = 1 THEN
                    SET var_w TO CHR( ( ( ORD(var_w) - ORD('a') + 1 ) MOD 26 ) + ORD('a') )
                END IF
            END IF
            SET var_r TO var_r - 1
        UNTIL var_r < 0

        RETURN var_w
    END FUNCTION
END CLASS
