```pseudocode
CLASS Solution
    FUNCTION minStartingIndex WITH PARAMETERS s pattern
        FUNCTION is_almost_equal WITH PARAMETERS x y
            SET delta TO 0
            SET pos TO 0
            WHILE pos LESS THAN THE LENGTH OF x
                IF NOT (x[pos] EQUALS y[pos]) THEN
                    SET delta TO delta + 1
                    IF delta GREATER THAN 1 THEN
                        RETURN False
                    END IF
                END IF
                SET pos TO pos + 1
            END WHILE
            RETURN True
        END FUNCTION

        SET n TO LENGTH OF s
        SET m TO LENGTH OF pattern
        SET start TO 0
        REPEAT
            IF is_almost_equal WITH ARGUMENTS s FROM start TO (start + m - 1) pattern THEN
                SET output TO start
                SET start TO n  // to break the loop by condition
            ELSE
                SET start TO start + 1
            END IF
        UNTIL start GREATER THAN n - m
        IF start EQUALS n THEN
            RETURN output
        ELSE
            RETURN -1
        END IF
    END FUNCTION
END CLASS
```
