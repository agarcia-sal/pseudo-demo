CLASS Solution  
    FUNCTION valueAfterKSeconds(n k)  
        DEFINE BASE_MODULUS AS 1000000000 + 7  
        SET Q TO array of length n INITIALIZED with 1  
        
        DEFINE PROCEDURE iterateIndex(pos)  
            IF pos >= n THEN  
                RETURN  
            ELSE  
                SET Q[pos] TO (Q[pos] + Q[pos - 1]) % BASE_MODULUS  
                CALL iterateIndex(pos + 1)  
            END IF  
        END PROCEDURE

        DEFINE PROCEDURE iterateStep(count)  
            IF count == k THEN  
                RETURN  
            ELSE  
                CALL iterateIndex(1)  
                CALL iterateStep(count + 1)  
            END IF  
        END PROCEDURE
        
        CALL iterateStep(0)
        
        RETURN Q[n - 1]  
    END FUNCTION  
END CLASS
