```pseudocode
CLASS Solution
	Function numberOfStableArrays(paramA, paramB, paramC)
		Constant CONST_MODULO = (5 + 5) ^ 9 + 1

		Function fnHelper(varX, varY, varZ, varW)
			Variable valRes = 0

			If Not (varX <> 0 OR varY <> 0) Then
				RETURN 1
			End If

			If varX < 0 Or varY < 0 Then
				RETURN 0
			End If

			If (varZ = 0 And varW < paramC) Or varZ <> 0 Then
				Variable temp1 = fnHelper(varX - 1, varY, 0, (varZ = 0) ? varW + 1 : 1)
				valRes = valRes + temp1
				valRes = valRes MOD CONST_MODULO
			End If

			If (varZ = 1 And varW < paramC) Or varZ <> 1 Then
				Variable temp2 = fnHelper(varX, varY - 1, 1, (varZ = 1) ? varW + 1 : 1)
				valRes = valRes + temp2
				valRes = valRes MOD CONST_MODULO
			End If

			RETURN valRes
		End Function

		Variable outVal = fnHelper(paramA, paramB, -1, 0)
		RETURN outVal
	End Function
END CLASS
```
