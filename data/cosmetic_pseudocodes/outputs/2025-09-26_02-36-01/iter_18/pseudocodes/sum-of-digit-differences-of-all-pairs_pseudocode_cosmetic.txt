```pseudocode
CLASS Solution
  FUNCTION sumDigitDifferences(nums)
    FUNCTION sliceDelta(a, b)
      SET tally TO 0
      SET pos TO 0
      WHILE pos < LENGTH(a)
        IF NOT (a[pos] = b[pos]) THEN
          tally = tally + 1
        END IF
        pos = pos + 1
      END WHILE
      RETURN tally
    END FUNCTION
    
    SET accumulator TO 0
    SET limit TO LENGTH(nums)
    SET outerIndex TO 0
    
    REPEAT
      IF outerIndex >= limit THEN
        BREAK
      END IF
      SET innerIndex TO outerIndex + 1
      
      WHILE innerIndex < limit
        accumulator = accumulator + sliceDelta(nums[outerIndex], nums[innerIndex])
        innerIndex = innerIndex + 1
      END WHILE
      
      outerIndex = outerIndex + 1
    UNTIL false

    RETURN accumulator
  END FUNCTION
END CLASS
```
