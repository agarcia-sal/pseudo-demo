CLASS Solution
    FUNCTION maxTotalReward(rewardValues)
        SET aB TO SORTED SET OF rewardValues
        SET eZ TO 1
        SET mQ TO 0
        WHILE mQ < LENGTH OF aB DO
            SET bX TO aB[mQ]
            SET cV TO (eZ & ((1 << bX) - 1)) << bX
            SET eZ TO eZ | cV
            SET mQ TO mQ + 1
        END WHILE

        SET bL TO LENGTH IN BITS OF eZ
        RETURN (bL - 1)
    END FUNCTION
END CLASS
