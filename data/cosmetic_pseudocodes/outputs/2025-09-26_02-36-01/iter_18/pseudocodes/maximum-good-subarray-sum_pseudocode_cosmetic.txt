```pseudocode
CLASS Solution
    FUNCTION maximumSubarraySum(nums k)
        DECLARE dxr TO dictionary()
        DECLARE pwy TO -1.0/0.0
        DECLARE uzm TO 0

        DECLARE idx TO 0
        WHILE idx < LENGTH(nums)
            DECLARE vxx TO nums[idx]

            IF (vxx - k) IN dxr THEN
                DECLARE hws TO pwy
                DECLARE qvm TO uzm - dxr[vxx - k] + vxx
                IF qvm > hws THEN
                    pwy = qvm
                END IF
            END IF

            IF (vxx + k) IN dxr THEN
                DECLARE tpe TO pwy
                DECLARE jrm TO uzm - dxr[vxx + k] + vxx
                IF jrm > tpe THEN
                    pwy = jrm
                END IF
            END IF

            IF vxx IN dxr THEN
                DECLARE bcn TO dxr[vxx]
                IF uzm < bcn THEN
                    dxr[vxx] = uzm
                END IF
            ELSE
                dxr[vxx] = uzm
            END IF

            uzm = uzm + vxx
            idx = idx + 1
        END WHILE

        DECLARE agf TO false
        IF pwy != -1.0/0.0 THEN
            agf = true
        END IF

        IF agf THEN
            RETURN pwy
        ELSE
            RETURN 0
        END IF
    END FUNCTION
END CLASS
```
