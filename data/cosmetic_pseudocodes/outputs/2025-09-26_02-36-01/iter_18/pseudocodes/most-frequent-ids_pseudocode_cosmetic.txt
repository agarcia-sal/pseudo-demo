```pseudocode
CLASS Solution
    FUNCTION mostFrequentIDs(nums PARAMETER, freq PARAMETER)
        DECLARE freqMap AS dictionary with default integer zero
        DECLARE heapList AS empty list
        DECLARE outputList AS empty list

        DECLARE outerIdx AS 0
        WHILE outerIdx < LENGTH(nums)
            DECLARE valA AS nums[outerIdx]
            DECLARE valB AS freq[outerIdx]

            freqMap[valA] = freqMap[valA] + valB

            DECLARE newEntry AS PAIR(-freqMap[valA], valA)
            CALL heapList.ADD(newEntry)

            REPEAT
                IF heapList IS EMPTY THEN
                    BREAK
                END IF

                DECLARE topEntry AS heapList[0]
                DECLARE topNegFreq AS topEntry.FIRST
                DECLARE topNum AS topEntry.SECOND

                IF -topNegFreq != freqMap[topNum] THEN
                    REMOVE heapList[0]
                ELSE
                    BREAK
                END IF
            UNTIL FALSE

            DECLARE tempResult AS 0
            IF NOT (heapList IS EMPTY) THEN
                tempResult = -heapList[0].FIRST
            END IF

            CALL outputList.ADD(tempResult)
            SET outerIdx = outerIdx + 1
        END WHILE

        RETURN outputList
    END FUNCTION
END CLASS
```
