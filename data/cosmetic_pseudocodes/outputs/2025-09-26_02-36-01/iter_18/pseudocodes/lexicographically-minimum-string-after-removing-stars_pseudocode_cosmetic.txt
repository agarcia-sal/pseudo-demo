CLASS Solution
    FUNCTION clearStars(s)
        DECLARE tempList AS list = [] 
        DECLARE idx_1 AS integer = 0

        WHILE idx_1 < LENGTH(s)
            DECLARE nxtChar AS character = s[idx_1]
            IF NOT (nxtChar != '*') THEN
                IF LENGTH(tempList) > 0 THEN
                    REMOVE tempList[ LENGTH(tempList) - 1 ]
                END IF
            ELSE
                tempList = tempList + [nxtChar]
            END IF
            SET idx_1 = idx_1 + 1
        END WHILE

        DECLARE tempStr AS string = ""
        DECLARE idx_2 AS integer = 0

        REPEAT
            IF idx_2 >= LENGTH(tempList)
                BREAK
            END IF
            SET tempStr = tempStr + tempList[idx_2]
            SET idx_2 = idx_2 + 1
        UNTIL false

        RETURN tempStr
    END FUNCTION
END CLASS
