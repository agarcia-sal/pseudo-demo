```pseudocode
CLASS Solution
    FUNCTION flowerGame(gamma, delta)
        DEFINE alpha = (gamma + 1) / 2
        DEFINE beta = gamma / 2

        DEFINE sigma = (delta + 1) / 2
        DEFINE tau = delta / 2

        DEFINE epsilon = (beta * sigma) + (alpha * tau)

        RETURN epsilon
    END FUNCTION
END CLASS
```
