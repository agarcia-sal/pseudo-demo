```pseudocode
CLASS Solution
    FUNCTION getSmallestString(s parameter of type string k parameter of type integer) RETURNS string

        FUNCTION cyclic_distance(u parameter of type character v parameter of type character) RETURNS integer
            SET diff TO ABS(ORD(u) - ORD(v))
            SET alt TO 26 - diff
            IF alt < diff THEN
                RETURN alt
            ELSE
                RETURN diff
            END IF
        END FUNCTION

        SET w TO CHAR_ARRAY_FROM(s)
        SET x TO 0
        SET y TO LENGTH(s)

        REPEAT
            IF NOT (k > 0 AND x < y) THEN
                BREAK
            END IF

            IF w[x] = 'a' THEN
                SET x TO x + 1
                CONTINUE
            END IF

            SET z TO cyclic_distance(w[x], 'a')

            IF z <= k THEN
                SET w[x] TO 'a'
                SET k TO k - z
            ELSE
                SET w[x] TO CHAR(ORD(w[x]) - k)
                SET k TO 0
            END IF

            SET x TO x + 1
        UNTIL FALSE

        SET result TO ""
        FOR m FROM 0 TO LENGTH(w) - 1 DO
            SET result TO result + w[m]
        END FOR

        RETURN result
    END FUNCTION
END CLASS
```
