```pseudocode
CLASS Solution  
    FUNCTION numberOfWays(n)  
        CONSTANT threshold IS (10 ^ 9) + 7  
        ARRAY cache SET TO [0] * (n + 1)  
        cache[0] ← 1  
        FOR denomination IN [6, 1, 2]  
            index ← denomination  
            WHILE index ≤ n  
                cache[index] ← (cache[index] + cache[index - denomination]) MOD threshold  
                index ← index + 1  
            END WHILE  
        END FOR  
        accumulator ← 0  
        counter ← 0  
        REPEAT  
            IF (counter * 4) ≤ n THEN  
                accumulator ← (accumulator + cache[n - (counter * 4)]) MOD threshold  
            END IF  
            counter ← counter + 1  
        UNTIL counter > 2  
        RETURN accumulator  
    END FUNCTION  
END CLASS
```
