```pseudocode
CLASS Solution
    FUNCTION distanceSum WITH PARAMETERS alpha beta gamma
        SET moduloBase TO 1000000007
        SET termA TO (beta * beta * (alpha * alpha * alpha - alpha)) / 6
        SET termB TO (alpha * alpha * (beta * beta * beta - beta)) / 6
        SET comboCount TO combination((alpha * beta) - 2, gamma - 2)
        SET accumulatedSum TO (termA + termB) * comboCount

        DECLARE resultHolder
        SET resultHolder TO accumulatedSum MODULO moduloBase

        RETURN resultHolder
    END FUNCTION
END CLASS
```
