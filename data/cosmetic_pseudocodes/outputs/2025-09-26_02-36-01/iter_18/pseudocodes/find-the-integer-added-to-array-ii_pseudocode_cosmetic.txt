CLASS Solution
    FUNCTION minimumAddedInteger(nums1 RETURNS List OF integer, nums2 RETURNS List OF integer) RETURNS integer
        CALL sort ON nums1
        CALL sort ON nums2

        SET alpha TO 0
        WHILE alpha < LENGTH OF nums1 MINUS 1
            SET beta TO alpha PLUS 1
            WHILE beta < LENGTH OF nums1 MINUS 1
                SET gamma TO an empty list

                SET delta TO 0
                WHILE delta < alpha
                    APPEND nums1[delta] TO gamma
                    SET delta TO delta PLUS 1
                END WHILE

                SET epsilon TO alpha PLUS 1
                WHILE epsilon < beta
                    APPEND nums1[epsilon] TO gamma
                    SET epsilon TO epsilon PLUS 1
                END WHILE

                SET zeta TO beta PLUS 1
                WHILE zeta < LENGTH OF nums1
                    APPEND nums1[zeta] TO gamma
                    SET zeta TO zeta PLUS 1
                END WHILE

                SET eta TO nums2[0] MINUS gamma[0]

                SET theta TO TRUE
                SET iota TO 0
                WHILE iota < LENGTH OF nums2
                    IF (gamma[iota] PLUS eta) NOT EQUALS nums2[iota]
                        SET theta TO FALSE
                        SET iota TO LENGTH OF nums2
                    ELSE
                        SET iota TO iota PLUS 1
                    END IF
                END WHILE

                IF theta EQUALS TRUE
                    RETURN eta
                END IF

                SET beta TO beta PLUS 1
            END WHILE

            SET alpha TO alpha PLUS 1
        END WHILE

        RETURN None
    END FUNCTION
END CLASS
