```pseudocode
CLASS Solution
  FUNCTION countDays(days, meetings)
    PROCEDURE sortMeetingsAscending(list)
      VARIABLE p, q, temp
      p = 0
      WHILE p < LENGTH(list) - 1
        q = p + 1
        WHILE q < LENGTH(list)
          IF NOT (list[p][0] <= list[q][0])
            temp = list[p]
            list[p] = list[q]
            list[q] = temp
          END IF
          q = q + 1
        END WHILE
        p = p + 1
      END WHILE
    END PROCEDURE

    CALL sortMeetingsAscending(meetings)

    VARIABLE alpha = 1
    VARIABLE beta = 0
    VARIABLE idx = 0

    WHILE idx < LENGTH(meetings)
      VARIABLE mu = meetings[idx][0]
      VARIABLE nu = meetings[idx][1]

      IF alpha >= mu
        # do nothing
      ELSE
        beta = beta + (mu - alpha)
      END IF

      VARIABLE max_day = alpha
      IF nu > max_day
        max_day = nu
      END IF
      alpha = max_day + 1

      idx = idx + 1
    END WHILE

    IF alpha <= days
      beta = beta + ((days - alpha) + 1)
    END IF

    RETURN beta
  END FUNCTION
END CLASS
```
