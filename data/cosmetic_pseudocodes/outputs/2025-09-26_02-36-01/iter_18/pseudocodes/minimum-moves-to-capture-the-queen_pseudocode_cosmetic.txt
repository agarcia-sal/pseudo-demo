CLASS Solution
    FUNCTION minMovesToCaptureTheQueen(a b c d e f)
        VARIABLE u = 2
        VARIABLE xQwd = a
        VARIABLE qij = b
        VARIABLE fzst = c
        VARIABLE rweg = d
        VARIABLE mnlo = e
        VARIABLE pchv = f
        VARIABLE srik = 1
        VARIABLE vect = 0
        VARIABLE klgb = 1
        VARIABLE blzt = 1
        VARIABLE xcjv = 1
        VARIABLE lmku = 1

        LABEL start_check

        WHILE vect < 1 DO
            IF xQwd = mnlo THEN
                IF fzst <> xQwd THEN
                    vect = 1
                    u = klgb
                    BREAK
                END IF

                IF ((qij < rweg AND rweg < pchv) OR (pchv < rweg AND rweg < qij)) THEN
                    vect = 1
                    u = 2
                    BREAK
                END IF

                vect = 1
                u = blzt
                BREAK
            ELSIF qij = pchv THEN
                IF rweg <> qij THEN
                    vect = 1
                    u = srik
                    BREAK
                END IF

                IF ((xQwd < fzst AND fzst < mnlo) OR (mnlo < fzst AND fzst < xQwd)) THEN
                    vect = 1
                    u = 2
                    BREAK
                END IF

                vect = 1
                u = xcjv
                BREAK
            ELSIF fzst + rweg = mnlo + pchv THEN
                IF xQwd + qij <> fzst + rweg THEN
                    vect = 1
                    u = lmku
                    BREAK
                END IF

                IF ((fzst < xQwd AND xQwd < mnlo AND rweg < qij AND qij < pchv) OR (mnlo < xQwd AND xQwd < fzst AND pchv < qij AND qij < rweg)) THEN
                    vect = 1
                    u = 2
                    BREAK
                END IF

                vect = 1
                u = 1
                BREAK
            ELSIF fzst - rweg = mnlo - pchv THEN
                IF xQwd - qij <> fzst - rweg THEN
                    vect = 1
                    u = 1
                    BREAK
                END IF

                IF ((fzst < xQwd AND xQwd < mnlo AND rweg > qij AND qij > pchv) OR (mnlo < xQwd AND xQwd < fzst AND pchv > qij AND qij > rweg)) THEN
                    vect = 1
                    u = 2
                    BREAK
                END IF

                vect = 1
                u = 1
                BREAK
            ELSE
                vect = 1
                u = 2
                BREAK
            END IF
        END WHILE

        RETURN u
    END FUNCTION
END CLASS
