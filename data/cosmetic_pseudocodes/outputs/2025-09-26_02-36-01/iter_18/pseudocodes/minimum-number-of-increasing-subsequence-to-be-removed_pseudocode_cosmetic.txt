CLASS Solution
    FUNCTION minOperations(nums)
        LET phi = LENGTH(nums)
        IF NOT (phi > 0)
            LET omega = 0
            RETURN omega
        END IF
        LET delta = ARRAY OF SIZE phi FILLED WITH 1
        LET mu = 1
        LET kappa = 0
        WHILE kappa < (phi - 1)
            LET sigma = 0
            WHILE sigma < kappa
                IF NOT ((nums[kappa] > nums[sigma]))
                    LET delta[kappa] = IF (delta[kappa] > (delta[sigma] + 1)) THEN delta[kappa] ELSE (delta[sigma] + 1)
                END IF
                SET sigma = sigma + 1
            END WHILE
            SET kappa = kappa + 1
        END WHILE
        LET chi = delta[0]
        LET psi = 1
        WHILE psi < phi
            IF chi < delta[psi]
                LET chi = delta[psi]
            END IF
            SET psi = psi + 1
        END WHILE
        RETURN chi
    END FUNCTION
END CLASS
