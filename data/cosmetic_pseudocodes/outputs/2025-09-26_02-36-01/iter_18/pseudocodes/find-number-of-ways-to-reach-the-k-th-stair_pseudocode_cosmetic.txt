CLASS Solution
    FUNCTION waysToReachStair WITH PARAMETER k
        FUNCTION dfs WITH PARAMETERS x, y, z
            SET resultHolder TO 0
            IF NOT (x <= k + 1)
                RETURN 0
            END IF
            SET resultHolder TO ( (x = k) * 1 )
            
            IF (x > 0) AND NOT (y != 0)
                SET tempVal TO dfs(x - 1, 1, z)
                SET resultHolder TO resultHolder + tempVal
            END IF
            
            SET tempCalc TO dfs(x + (2 * z), 0, z + 1)
            SET resultHolder TO resultHolder + tempCalc
            
            RETURN resultHolder
        END FUNCTION
        
        RETURN dfs(1, 0, 0)
    END FUNCTION
END CLASS
