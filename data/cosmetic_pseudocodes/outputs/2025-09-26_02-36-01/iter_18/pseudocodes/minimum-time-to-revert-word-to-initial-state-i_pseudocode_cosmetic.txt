CLASS Solution
    FUNCTION minimumTimeToInitialState(word PARAMETER k)
        SET a TO LENGTH OF word
        SET b TO 1
        LOOP
            IF NOT (b * k < a) OR SUBSTRING(word, b * k, a) = SUBSTRING(word, 1, a - b * k) THEN
                RETURN b
            END IF
            SET b TO b + 1
        END LOOP
    END FUNCTION
END CLASS
