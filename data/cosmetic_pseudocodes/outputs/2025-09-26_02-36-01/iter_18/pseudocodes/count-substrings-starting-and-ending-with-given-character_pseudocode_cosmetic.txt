```pseudocode
CLASS Solution
    FUNCTION countSubstrings(s c)
        LET idx BE 0
        LET totalChars BE 0
        LET accumulator BE 0

        WHILE idx < LENGTH(s) DO
            IF NOT (s[idx] != c) THEN
                totalChars = totalChars + 1
            END IF
            idx = idx + 1
        END WHILE

        LET tempSum BE 0
        LET iter BE 1

        REPEAT
            tempSum = tempSum + iter
            iter = iter + 1
        UNTIL iter > totalChars

        accumulator = totalChars * tempSum

        LET output BE accumulator
        RETURN output
    END FUNCTION
END CLASS
```
