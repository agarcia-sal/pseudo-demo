CLASS Solution
    FUNCTION numberOfPairs(nums1, nums2, k)
        SET tally_map TO empty dictionary
        FOR element zeta FROM 0 UP TO LENGTH(nums2) - 1
            IF nums2[zeta] EXISTS IN tally_map THEN
                SET tally_map[nums2[zeta]] TO tally_map[nums2[zeta]] + 1
            ELSE
                SET tally_map[nums2[zeta]] TO 1
            END IF
        END FOR

        SET accumulator TO 0
        SET iterator_alpha TO 0
        WHILE iterator_alpha < LENGTH(nums1)
            SET current_num1 TO nums1[iterator_alpha]
            FOR EACH pair (key_beta, val_gamma) IN tally_map
                IF NOT ((current_num1 MOD (key_beta * k)) != 0) THEN
                    SET accumulator TO accumulator + val_gamma
                END IF
            END FOR
            SET iterator_alpha TO iterator_alpha + 1
        END WHILE

        RETURN accumulator
    END FUNCTION
END CLASS
