```pseudocode
CLASS Solution
    FUNCTION minimumArea(grid)
        DECLARE flagResult AS BOOLEAN
        DECLARE outerIndex AS INTEGER
        DECLARE innerIndex AS INTEGER
        DECLARE startRow AS INTEGER
        DECLARE endRow AS INTEGER
        DECLARE startCol AS INTEGER
        DECLARE endCol AS INTEGER
        DECLARE totalRows AS INTEGER
        DECLARE totalCols AS INTEGER
        DECLARE calculatedHeight AS INTEGER
        DECLARE calculatedWidth AS INTEGER
        DECLARE finalArea AS INTEGER

        IF NOT (grid != [] AND grid[0] != [])
            RETURN 0
        END IF

        totalRows ← 0
        WHILE totalRows < LENGTH(grid)
            totalRows ← totalRows + 1
        END WHILE

        totalCols ← LENGTH(grid[0])
        DECLARE stepCounter AS INTEGER
        stepCounter ← 0
        WHILE stepCounter < totalCols
            stepCounter ← stepCounter + 1
        END WHILE

        startRow ← +∞
        endRow ← -∞
        startCol ← +∞
        endCol ← -∞

        outerIndex ← 0
        WHILE outerIndex != totalRows
            innerIndex ← 0
            WHILE innerIndex != totalCols
                flagResult ← (grid[outerIndex][innerIndex] == 1)
                IF flagResult
                    IF startRow > outerIndex
                        startRow ← outerIndex
                    END IF

                    IF endRow < outerIndex
                        endRow ← outerIndex
                    END IF

                    IF startCol > innerIndex
                        startCol ← innerIndex
                    END IF

                    IF endCol < innerIndex
                        endCol ← innerIndex
                    END IF
                END IF
                innerIndex ← innerIndex + 1
            END WHILE
            outerIndex ← outerIndex + 1
        END WHILE

        calculatedHeight ← (endRow - startRow) + 1
        calculatedWidth ← endCol - startCol + 1

        finalArea ← calculatedHeight * calculatedWidth

        RETURN finalArea
    END FUNCTION
END CLASS
```
