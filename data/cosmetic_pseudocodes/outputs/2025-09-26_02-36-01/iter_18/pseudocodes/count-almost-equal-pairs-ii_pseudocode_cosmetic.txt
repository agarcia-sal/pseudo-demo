```pseudocode
CLASS Solution
	FUNCTION countPairs(nums)
		CALL sort nums IN PLACE
		
		SET eta TO 0
		SET phi TO a default dictionary RETURNING 0 ON missing keys
		
		SET epsilon TO 0
		WHILE epsilon < LENGTH OF nums
			SET alpha TO nums[epsilon]
			SET rho TO NEW EMPTY SET
			ADD alpha TO rho
			SET omega TO CONVERT alpha TO STRING AND SPLIT INTO CHARACTERS
			SET kappa TO LENGTH OF omega
			SET delta TO 0
			WHILE delta <= kappa - 1
				SET gamma TO 0
				WHILE gamma <= delta - 1
					// swap omega[gamma] WITH omega[delta]
					SET chi TO omega[gamma]
					SET omega[gamma] TO omega[delta]
					SET omega[delta] TO chi
					
					SET sigma TO TO_INTEGER(CONCATENATE ALL ELEMENTS OF omega)
					ADD sigma TO rho
					
					SET psi TO gamma + 1
					WHILE psi <= kappa - 1
						SET mu TO gamma + 1
						WHILE mu <= psi - 1
							// swap omega[mu] WITH omega[psi]
							SET tau TO omega[mu]
							SET omega[mu] TO omega[psi]
							SET omega[psi] TO tau
							
							SET upsilon TO TO_INTEGER(CONCATENATE ALL ELEMENTS OF omega)
							ADD upsilon TO rho
							
							// swap omega[mu] WITH omega[psi] back
							SET omega[psi] TO omega[mu]
							SET omega[mu] TO tau
							
							SET mu TO mu + 1
						END WHILE
						SET psi TO psi + 1
					END WHILE
					
					// swap omega[gamma] WITH omega[delta] back
					SET omega[delta] TO omega[gamma]
					SET omega[gamma] TO chi
					
					SET gamma TO gamma + 1
				END WHILE
				SET delta TO delta + 1
			END WHILE
			
			SET beta TO 0
			SET theta TO 0
			WHILE beta < LENGTH OF rho
				SET sigma TO ITEM AT POSITION beta IN rho
				SET theta TO theta + phi[sigma]
				SET beta TO beta + 1
			END WHILE
			SET eta TO eta + theta
			SET phi[alpha] TO phi[alpha] + 1
			SET epsilon TO epsilon + 1
		END WHILE
		
		RETURN eta
	END FUNCTION
END CLASS
```
