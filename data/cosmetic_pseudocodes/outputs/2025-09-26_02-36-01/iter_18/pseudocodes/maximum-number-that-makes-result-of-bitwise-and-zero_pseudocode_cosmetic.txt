CLASS Solution  
    FUNCTION maxNumber(n)  
        DECLARE resultVar AS INTEGER  
        DECLARE counterVar AS INTEGER  
        DECLARE limitFlag AS BOOLEAN  
        DECLARE tempVal AS INTEGER  
        
        IF NOT (n NOT EQUAL TO 1) THEN  
            resultVar = 0  
            RETURN resultVar  
        END IF  

        tempVal = 1  
        limitFlag = TRUE  

        WHILE limitFlag DO  
            (counterVar) = tempVal  
            IF NOT (counterVar <= n) THEN  
                limitFlag = FALSE  
            ELSE  
                tempVal = tempVal + tempVal  
            END IF  
        END WHILE  

        counterVar = tempVal / 2  
        resultVar = counterVar - 1  

        RETURN resultVar  
    END FUNCTION  
END CLASS
