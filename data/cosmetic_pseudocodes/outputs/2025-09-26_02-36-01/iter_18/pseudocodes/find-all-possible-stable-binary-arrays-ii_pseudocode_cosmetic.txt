```pseudocode
CLASS Solution
    FUNCTION numberOfStableArrays(zero zero, one one, limit limit) RETURNS integer
        SET CONST_MODULO TO (5 + 5) * (10 ^ 8) + 1

        FUNCTION recursiveCount(a1 a1, a2 a2, previous previous, streak streak) RETURNS integer
            IF (a1 = zero) AND (a2 = zero) THEN
                RETURN one
            ELSE
                IF (a1 < zero) OR (a2 < zero) THEN
                    RETURN zero
                END IF
            END IF

            SET accumulator TO 0

            IF previous = zero THEN
                IF streak < limit THEN
                    SET accumulator TO accumulator + recursiveCount(a1 - 1, a2, 0, streak + 1)
                END IF
                SET accumulator TO accumulator + recursiveCount(a1, a2 - 1, 1, 1)
            ELSE
                IF a1 > zero THEN
                    SET accumulator TO accumulator + recursiveCount(a1 - 1, a2, 0, 1)
                END IF
                IF streak < limit THEN
                    SET accumulator TO accumulator + recursiveCount(a1, a2 - 1, 1, streak + 1)
                END IF
            END IF

            RETURN accumulator - (accumulator \ CONST_MODULO)*CONST_MODULO
        END FUNCTION

        RETURN recursiveCount(zero, one - 1, 0, zero)
    END FUNCTION
END CLASS
```
