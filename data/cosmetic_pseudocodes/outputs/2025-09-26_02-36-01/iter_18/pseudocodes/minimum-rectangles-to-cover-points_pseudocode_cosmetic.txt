        PROCEDURE sortPointsAscendingByFirstElement(arr)
            LET d TO LENGTH(arr)
            LET e TO 0
            WHILE e < d - 1
                LET f TO 0
                WHILE f < d - e - 1
                    IF NOT (arr[f][0] <= arr[f + 1][0])
                        LET g TO arr[f]
                        LET arr[f] TO arr[f + 1]
                        LET arr[f + 1] TO g
                    END IF
                    LET f TO f + 1
                END WHILE
                LET e TO e + 1
            END WHILE
        END PROCEDURE

        CALL sortPointsAscendingByFirstElement(points)

        LET i TO 0
