CLASS Solution

    FUNCTION maxPalindromesAfterOperations(words)
        SET _m TO new counter CREATED FROM concatenation OF words
        SET _q TO 0
        SET _r TO 0

        FOR EACH _u IN VALUES OF _m
            SET _q TO _q + (_u DIV 2)
            SET _r TO _r + (_u MOD 2)
        END FOR

        CALL SortAscendingByLength(words)

        SET _o TO 0
        SET _x TO 0

        WHILE _x < LENGTH(words)
            SET _v TO LENGTH(words[_x]) DIV 2

            IF NOT (_q < _v)
                SET _q TO _q - _v
                SET _o TO _o + 1
            END IF

            SET _x TO _x + 1
        END WHILE

        RETURN _o
    END FUNCTION

    PROCEDURE SortAscendingByLength(arr)
        SET _a TO 0
        WHILE _a < LENGTH(arr) - 1
            SET _b TO 0
            WHILE _b < LENGTH(arr) - _a - 1
                IF LENGTH(arr[_b]) > LENGTH(arr[_b + 1])
                    SET _c TO arr[_b]
                    SET arr[_b] TO arr[_b + 1]
                    SET arr[_b + 1] TO _c
                END IF
                SET _b TO _b + 1
            END WHILE
            SET _a TO _a + 1
        END WHILE
    END PROCEDURE

END CLASS
