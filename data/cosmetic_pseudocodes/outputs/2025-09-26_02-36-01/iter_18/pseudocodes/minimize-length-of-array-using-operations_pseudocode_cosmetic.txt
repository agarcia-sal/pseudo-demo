CLASS Solution  
    FUNCTION minimumArrayLength(nums)  
        DECLARE c AS INTEGER  
        DECLARE j AS INTEGER  
        DECLARE p AS INTEGER  
        c = nums[0]  
        p = 1  
        j = 1  
        WHILE j < LENGTH(nums)  
            IF nums[j] < c THEN  
                c = nums[j]  
                p = 1  
            ELSE  
                IF NOT (nums[j] < c) THEN  
                    IF nums[j] = c THEN  
                        p = p + 1  
                    END IF  
                END IF  
            END IF  
            j = j + 1  
        END WHILE  
        DECLARE r AS INTEGER  
        IF NOT (p != 1) THEN  
            r = 1  
        ELSE  
            DECLARE s AS INTEGER  
            s = p + 1  
            r = s / 2  
        END IF  
        RETURN r  
    END FUNCTION  
END CLASS
