CLASS Solution
    FUNCTION compressedString(word)
        SET assembled TO empty list
        SET idx TO 0
        REPEAT 
            IF NOT (idx < LENGTH(word)) THEN
                BREAK
            END IF
            SET symbol TO word[idx]
            SET tally TO 0
            SET continueLoop TO TRUE
            WHILE continueLoop
                IF NOT (idx < LENGTH(word) AND word[idx] = symbol) THEN
                    continueLoop = FALSE
                ELSE
                    tally = tally + 1
                    idx = idx + 1
                    IF tally = 9 THEN
                        continueLoop = FALSE
                    END IF
                END IF
            END WHILE
            APPEND ("" + tally + symbol) TO assembled
        UNTIL FALSE
        SET finalString TO ""
        FOR each element IN assembled DO
            finalString = finalString + element
        END FOR
        RETURN finalString
    END FUNCTION
END CLASS
