CLASS Solution
    FUNCTION minOperations(m)
        SET counter TO 0
        SET extreme_value TO (m + 1) * (m + 1)
        SET minimal_count TO extreme_value
        SET iterator TO 1
        WHILE NOT (iterator > INTEGER_SQRT(m) + 1)
            SET divisor TO iterator
            SET quotient TO ((m + divisor - 1) DIV divisor)
            SET temp_result TO (divisor - 1) + (quotient - 1)
            IF NOT (temp_result >= minimal_count)
                SET minimal_count TO temp_result
            END IF
            SET iterator TO iterator + 1
        END WHILE
        SET final_answer TO minimal_count
        RETURN final_answer
    END FUNCTION
END CLASS
