CLASS Solution
    FUNCTION constructGridLayout(z3, w6)
        SET d1 TO a new list holding z3 empty lists

        SET i8 TO 0
        WHILE i8 LESS THAN length of w6
            SET a0, b7 TO w6 at i8
            INCREMENT i8 BY 1
            APPEND b7 TO d1 at position a0
            APPEND a0 TO d1 at position b7
        END WHILE

        SET c2 TO a list of five elements each initialized to ((0 - 1) - 1)

        SET k9, h5 TO 0, 0
        REPEAT
            IF k9 GREATER THAN OR EQUAL TO length of d1 THEN
                BREAK
            END IF
            SET r4 TO d1 at k9
            SET c2 at position length of r4 TO k9
            INCREMENT k9 BY 1
        UNTIL false

        IF NOT (c2 at 1 EQUALS ((0 - 1) - 1)) THEN
            SET v0 TO a list containing c2 at 1
        ELSE IF c2 at 4 EQUALS ((0 - 1) - 1) THEN
            SET z5 TO c2 at 2
            SET j2 TO 0
            REPEAT
                IF j2 GREATER THAN OR EQUAL TO length of d1 at z5 THEN
                    BREAK
                END IF
                SET p9 TO d1 at z5 at j2
                IF length of d1 at p9 EQUALS 2 THEN
                    SET v0 TO a list containing z5 and p9
                    BREAK
                END IF
                INCREMENT j2 BY 1
            UNTIL false
        ELSE
            SET z5 TO c2 at 2
            SET v0 TO a list containing z5
            SET o7 TO z5
            SET z5 TO d1 at z5 at 0
            WHILE length of d1 at z5 GREATER THAN 2
                APPEND z5 TO v0
                SET m8 TO 0
                REPEAT
                    IF m8 GREATER THAN OR EQUAL TO length of d1 at z5 THEN
                        BREAK
                    END IF
                    SET x4 TO d1 at z5 at m8
                    IF NOT x4 EQUALS o7 AND length of d1 at x4 LESS THAN 4 THEN
                        SET o7 TO z5
                        SET z5 TO x4
                        BREAK
                    END IF
                    INCREMENT m8 BY 1
                UNTIL false
            END WHILE
            APPEND z5 TO v0
        END IF

        SET j7 TO a list containing v0
        SET u6 TO a new list with z3 elements all false

        SET g0 TO 1
        WHILE g0 LESS THAN (z3 DIV length of v0) MINUS 1 + 1
            SET s3 TO 0
            WHILE s3 LESS THAN length of v0
                SET u6 at position v0 at s3 TO true
                INCREMENT s3 BY 1
            END WHILE

            SET d7 TO an empty list
            SET q5 TO 0
            WHILE q5 LESS THAN length of v0
                SET f9 TO v0 at q5
                SET q5 TO q5 + 1
                SET t1 TO 0
                REPEAT
                    IF t1 GREATER THAN OR EQUAL TO length of d1 at f9 THEN
                        BREAK
                    END IF
                    SET b6 TO d1 at f9 at t1
                    IF NOT u6 at b6 THEN
                        APPEND b6 TO d7
                        BREAK
                    END IF
                    INCREMENT t1 BY 1
                UNTIL false
            END WHILE

            APPEND d7 TO j7
            SET v0 TO d7
            INCREMENT g0 BY 1
        END WHILE

        RETURN j7
    END FUNCTION
END CLASS
