Class Solution
    Function maxArea(pos_vals dir_flags)
        Let m be Length(pos_vals)
        Let greatestSum be 0
        For counter from 1 to (height * 2)
            Let index be 0
            While index < m
                If pos_vals[index] = 0 And dir_flags[index] = "D" Then
                    dir_flags = Concat(Substr(dir_flags, 0, index - 1), "U", Substr(dir_flags, index + 1, Length(dir_flags) - 1))
                Else If pos_vals[index] = height And dir_flags[index] = "U" Then
                    dir_flags = Concat(Substr(dir_flags, 0, index - 1), "D", Substr(dir_flags, index + 1, Length(dir_flags) - 1))
                End If

                If dir_flags[index] = "U" Then
                    pos_vals[index] = pos_vals[index] + 1
                Else
                    pos_vals[index] = pos_vals[index] - 1
                End If

                index = index + 1
            End While

            Let tempSum be 0
            Let accIdx be 0
            Repeat
                tempSum = tempSum + pos_vals[accIdx]
                accIdx = accIdx + 1
            Until accIdx = m

            If greatestSum < tempSum Then
                greatestSum = tempSum
            End If
        End For
        Return greatestSum
    End Function
End Class
