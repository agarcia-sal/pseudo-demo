CLASS Solution
    FUNCTION minEnd(n x)
        FUNCTION canConstruct(threshold)
            SET alpha TO x
            SET beta TO 1
            repeat
                IF alpha >= threshold THEN
                    break
                END IF

                SET alpha TO alpha + 1

                IF (alpha AND x) = x THEN
                    SET beta TO beta + 1

                    IF beta = n THEN
                        RETURN True
                    END IF
                END IF
            UNTIL FALSE

            RETURN (beta = n)
        END FUNCTION

        SET gamma TO x
        SET delta TO ((((((((2 * 10) * 10) * 10) * 10) * 10) * 10) * 10) * 10)

        WHILE gamma < delta
            SET epsilon TO (gamma + delta) // 2

            IF canConstruct(epsilon) THEN
                SET delta TO epsilon
            ELSE
                SET gamma TO gamma + 1
            END IF
        END WHILE

        RETURN gamma
    END FUNCTION
END CLASS
