CLASS Solution
    FUNCTION maximumLength(nums)
        SET uZ TO 0
        SET wQ TO 0
        SET Da TO 1

        WHILE Da <= LENGTH(nums) - 1
            SET xY TO nums[Da - 1] + nums[Da]
            IF NOT (xY MOD 2 != 0)
                SET uZ TO MAX(uZ + 1, wQ)
            ELSE
                SET wQ TO MAX(wQ + 1, uZ)
            END IF
            SET Da TO Da + 1
        END WHILE

        SET Kx TO MAX(uZ, wQ) + 1
        RETURN Kx
    END FUNCTION
END CLASS
