CLASS Solution
	FUNCTION minAnagramLength(s)
		LET delta TO EMPTY MAP
		FOR i <- 0 WHILE i < LENGTH(s) DO
			IF NOT (s[i] IN delta) THEN
				SET delta[s[i]] <- 0
			END IF
			SET i <- i + 1
		END FOR

		LET omega <- 0
		LET keys <- LIST OF KEYS(delta)
		LET idx <- 0
		WHILE idx < LENGTH(keys) REPEAT
			SET omega <- omega + 1
			SET idx <- idx + 1
		END WHILE

		RETURN omega
	END FUNCTION
END CLASS
