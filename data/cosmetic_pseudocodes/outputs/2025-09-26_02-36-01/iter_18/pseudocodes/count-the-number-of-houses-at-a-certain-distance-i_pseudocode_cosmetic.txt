CLASS Solution
    FUNCTION countOfPairs(n parameter, x parameter, y parameter) RETURNS list of integer
        SET omega TO a list initialized with zeros, length n

        FUNCTION searchTraversal(root parameter)
            SET flagged TO a list filled with False, length n + 1
            SET steps TO a list filled with zero, length n + 1
            SET line TO a deque initialized with root parameter as sole element
            SET flagged at index root parameter TO True

            REPEAT
                IF line is empty THEN
                    BREAK
                END IF
                SET cursor TO deque popleft from line

                FOR candidate IN [cursor - 1, cursor + 1]
                    IF candidate >= 1 AND candidate <= n AND flagged[candidate] = False THEN
                        SET flagged[candidate] TO True
                        SET steps[candidate] TO steps[cursor] + 1
                        ENQUEUE candidate at end of line
                    END IF
                END FOR

                IF cursor = x AND flagged[y] = False THEN
                    SET flagged[y] TO True
                    SET steps[y] TO steps[cursor] + 1
                    APPEND y to line
                END IF

                IF cursor = y AND flagged[x] = False THEN
                    SET flagged[x] TO True
                    SET steps[x] TO steps[cursor] + 1
                    APPEND x to line
                END IF
            UNTIL False

            SET idx TO 1
            WHILE idx <= n
                IF steps[idx] > 0 THEN
                    SET result[steps[idx] - 1] TO result[steps[idx] - 1] + 1
                END IF
                SET idx TO idx + 1
            END WHILE
        END FUNCTION

        SET alpha TO 1
        WHILE alpha <= n
            CALL searchTraversal WITH alpha
            SET alpha TO alpha + 1
        END WHILE

        RETURN result
    END FUNCTION
END CLASS
