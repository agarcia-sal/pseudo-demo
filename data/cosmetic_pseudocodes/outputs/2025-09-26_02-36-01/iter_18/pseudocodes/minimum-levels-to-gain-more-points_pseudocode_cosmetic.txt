CLASS Solution
    FUNCTION minimumLevels(possible)
        SET omega TO 0
        SET chi TO 0
        SET phi TO 0
        SET psi TO LENGTH OF possible
        WHILE phi < psi
            SET sigma TO 0
            SET sigma TO 2 * possible[phi] - 1
            SET omega TO omega + sigma
            SET phi TO phi + 1
        END WHILE
        SET tau TO 0
        SET epsilon TO 0
        SET delta TO psi - 1
        SET theta TO 0
        REPEAT
            SET eta TO 2 * possible[theta] - 1
            SET tau TO tau + eta
            SET omega TO omega - eta
            IF (tau - omega) > 0 THEN
                SET delta TO theta + 1
                BREAK
            END IF
            SET theta TO theta + 1
        UNTIL theta > delta
        IF (tau - omega) <= 0 THEN
            RETURN -1
        ELSE
            RETURN delta
        END IF
    END FUNCTION
END CLASS
