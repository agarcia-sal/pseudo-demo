CLASS Solution
    FUNCTION maximumPoints(enemyEnergies currentEnergy)
        iJkvmJhQY <- 0
        xNqGZuGi <- 0
        hMRXLOAP <- LENGTH(enemyEnergies) - 1

        // Sorting enemyEnergies ascendingly with in-place repeated swaps
        REPEAT
            CphtvEeF <- FALSE
            JspqATjZ <- 0
            WHILE JspqATjZ < LENGTH(enemyEnergies) - 1 DO
                IF NOT (enemyEnergies[JspqATjZ] <= enemyEnergies[JspqATjZ + 1]) THEN
                    tmpVar <- enemyEnergies[JspqATjZ]
                    enemyEnergies[JspqATjZ] <- enemyEnergies[JspqATjZ + 1]
                    enemyEnergies[JspqATjZ + 1] <- tmpVar
                    CphtvEeF <- TRUE
                END IF
                JspqATjZ <- JspqATjZ + 1
            END WHILE
        UNTIL NOT CphtvEeF

        IF enemyEnergies[0] > currentEnergy THEN
            RETURN 0
        ELSE
            WHILE hMRXLOAP >= 0 DO
                xNqGZuGi <- xNqGZuGi + (currentEnergy - (currentEnergy MOD enemyEnergies[0])) / enemyEnergies[0]
                currentEnergy <- currentEnergy MOD enemyEnergies[0]
                currentEnergy <- currentEnergy + enemyEnergies[hMRXLOAP]
                hMRXLOAP <- hMRXLOAP - 1
            END WHILE
            RETURN xNqGZuGi
        END IF
    END FUNCTION
END CLASS
