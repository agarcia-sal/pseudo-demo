CLASS Solution
    FUNCTION countAlternatingSubarrays(nums)
        SET length_var TO LENGTH OF nums
        IF NOT (length_var > 0)
            RETURN 0
        END IF
        SET accumulator TO length_var
        SET tracker TO 1
        SET index_var TO 1
        WHILE index_var < length_var DO
            IF NOT (nums[index_var] == nums[index_var - 1])
                SET tracker TO tracker + 1
            ELSE
                SET tracker TO 1
            END IF
            SET accumulator TO accumulator + tracker - 1
            SET index_var TO index_var + 1
        END WHILE
        RETURN accumulator
    END FUNCTION
END CLASS
