```pseudocode
CLASS Solution
    FUNCTION answerString(word, numFriends)
        DECLARE r AS STRING
        DECLARE x AS STRING
        DECLARE m AS INTEGER
        IF NOT (numFriends NOT EQUALS 1) THEN
            r = word
        ELSE
            x = _lastSubstring(word)
            m = (LENGTH(word) - numFriends) + 1
            DECLARE t AS INTEGER
            t = LENGTH(x)
            IF m < t THEN
                t = m
            END IF
            r = SUBSTRING(x, 1, t)
        END IF
        RETURN r
    END FUNCTION

    FUNCTION _lastSubstring(s)
        DECLARE a AS INTEGER
        DECLARE b AS INTEGER
        DECLARE c AS INTEGER
        a = 0
        b = 1
        c = 0

        WHILE TRUE
            IF b + c >= LENGTH(s) THEN
                BREAK
            END IF

            IF s[a + c] = s[b + c] THEN
                c = c + 1
            ELSE
                IF s[a + c] > s[b + c] THEN
                    b = b + c + 1
                    c = 0
                ELSE
                    DECLARE temp AS INTEGER
                    temp = a + c + 1
                    a = MAX(temp, b)
                    b = a + 1
                    c = 0
                END IF
            END IF
        END WHILE

        RETURN SUBSTRING(s, a + 1, LENGTH(s) - a)
    END FUNCTION
END CLASS
```
