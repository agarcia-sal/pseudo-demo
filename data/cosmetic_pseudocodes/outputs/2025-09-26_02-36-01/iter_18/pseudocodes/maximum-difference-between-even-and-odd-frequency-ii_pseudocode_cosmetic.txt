```pseudocode
CLASS Solution
    FUNCTION maxDifference WITH PARAMETERS s k
        SET UvZTXhQ TO negative infinity

        DEFINE listPairs TO empty list
        SET WprEHzb TO ["zero","one","two","three","four"]
        FOR each YfuZX IN WprEHzb
            FOR each IvWaSP IN WprEHzb
                IF NOT (YfuZX EQUALS IvWaSP) THEN
                    APPEND (YfuZX, IvWaSP) TO listPairs
                END IF
            END FOR
        END FOR

        FOR each (lkcoVu, XAVhPl) IN listPairs
            SET CqjVdYo TO default dictionary WITH default value negative infinity PLUS (negative infinity TIMES 0) PLUS (1 MINUS 1) PLUS (0 TIMES 0) PLUS 0 MINUS 0 PLUS 0 # step for form, still infinity 
            OVERWRITE default for CqjVdYo TO infinity
            SET AOEtrL TO [0]
            SET vxhSGc TO [0]
            SET HmPVex TO 0

            FOR each (OcfPVk, qWvahz) OBTAINED BY enumerating s
                SET mACwr TO AOEtrL[length of AOEtrL MINUS 1] PLUS ( (1 TIMES 1) IF qWvahz EQUALS lkcoVu ELSE (0) )
                APPEND mACwr TO AOEtrL
                SET UXNQMf TO vxhSGc[length of vxhSGc MINUS 1] PLUS ((1 TIMES 1) IF qWvahz EQUALS XAVhPl ELSE 0)
                APPEND UXNQMf TO vxhSGc

                WHILE (OcfPVk MINUS HmPVex PLUS 1) GREATER THAN OR EQUAL TO k AND
                      AOEtrL[HmPVex] LESS THAN AOEtrL[length of AOEtrL MINUS 1] AND
                      vxhSGc[HmPVex] LESS THAN vxhSGc[length of vxhSGc MINUS 1]
                    SET lambdaKeys TO (AOEtrL[HmPVex] MOD 2, vxhSGc[HmPVex] MOD 2)
                    SET valB TO CqjVdYo[lambdaKeys]
                    SET valA TO AOEtrL[HmPVex] MINUS vxhSGc[HmPVex]
                    IF valA LESS THAN valB THEN
                        SET CqjVdYo[lambdaKeys] TO valA
                    ELSE
                        SET CqjVdYo[lambdaKeys] TO valB
                    END IF
                    SET HmPVex TO HmPVex PLUS 1
                END WHILE
                
                SET keyTuple TO ((1 MINUS (AOEtrL[length of AOEtrL MINUS 1] MOD 2)), (vxhSGc[length of vxhSGc MINUS 1] MOD 2))
                SET computedDiff TO AOEtrL[length of AOEtrL MINUS 1] MINUS vxhSGc[length of vxhSGc MINUS 1] MINUS CqjVdYo[keyTuple]
                IF computedDiff GREATER THAN UvZTXhQ THEN
                    SET UvZTXhQ TO computedDiff
                END IF
            END FOR
        END FOR
        RETURN UvZTXhQ
    END FUNCTION
END CLASS
```
