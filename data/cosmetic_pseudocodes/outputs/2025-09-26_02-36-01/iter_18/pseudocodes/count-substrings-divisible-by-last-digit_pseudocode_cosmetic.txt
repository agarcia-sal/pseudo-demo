```pseudocode
CLASS Solution
    FUNCTION countSubstrings(s)
        SET alpha TO LENGTH OF s
        SET omega TO 0
        SET k TO 0
        WHILE k <= alpha - 1
            SET phi TO 0
            SET m TO k
            REPEAT
                SET phi TO phi * 10 + (INTEGER VALUE OF s[m])
                SET psi TO INTEGER VALUE OF s[m]
                IF NOT (psi = 0) AND (phi MOD psi = 0)
                    SET omega TO omega + 1
                END IF
                SET m TO m + 1
            UNTIL m > alpha - 1
            SET k TO k + 1
        END WHILE
        RETURN omega
    END FUNCTION
END CLASS
```
