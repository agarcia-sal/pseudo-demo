```pseudocode
CLASS Solution
    FUNCTION maxScore(points, m)
        FUNCTION isPossible(minVal, m)
            SET alpha TO 0
            SET beta TO 0
            SET k TO 0
            WHILE k < LENGTH(points)
                SET phi TO ((minVal + points[k] - 1) DIV points[k])
                IF (phi - beta) < 0 THEN
                    SET phi TO 0
                ELSE
                    SET phi TO phi - beta
                END IF
                IF phi > 0 THEN
                    SET alpha TO alpha + ((2 * phi) - 1)
                    SET beta TO phi - 1
                ELSE IF (k + 1) < LENGTH(points) THEN
                    SET alpha TO alpha + 1
                    SET beta TO 0
                END IF
                IF alpha > m THEN
                    RETURN False
                END IF
                SET k TO k + 1
            END WHILE
            RETURN True
        END FUNCTION

        SET omega TO 0
        SET psi TO (((m + 1) DIV 2) * (points[0] + 1))
        REPEAT
            SET gamma TO ((omega + psi + 1) DIV 2)
            IF isPossible(gamma, m) THEN
                SET omega TO gamma
            ELSE
                SET psi TO gamma - 1
            END IF
        UNTIL omega >= psi

        RETURN omega
    END FUNCTION
END CLASS
```
