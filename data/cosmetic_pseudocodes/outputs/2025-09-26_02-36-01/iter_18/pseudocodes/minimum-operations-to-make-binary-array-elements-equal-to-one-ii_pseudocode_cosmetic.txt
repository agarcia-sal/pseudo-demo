CLASS Solution
    FUNCTION minOperations(nums)
        SET tally TO 0
        SET toggle_marker TO 0
        SET idx TO 0
        WHILE idx < LENGTH OF nums DO
            IF (toggle_marker MOD 2) = 0 THEN
                SET temp_val TO nums[idx]
            ELSE
                SET temp_val TO (1 + (0 - nums[idx]))
            END IF

            IF temp_val = 0 THEN
                tally ← tally + 1
                toggle_marker ← toggle_marker + 1
            END IF
            idx ← idx + 1
        END WHILE

        SET result TO tally
        RETURN result
    END FUNCTION
END CLASS
