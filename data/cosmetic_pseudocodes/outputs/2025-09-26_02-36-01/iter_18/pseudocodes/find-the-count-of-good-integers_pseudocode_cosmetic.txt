```pseudocode
CLASS Solution
    FUNCTION countGoodIntegers(n integer, k integer) RETURNS integer
        SET zym TO []
        SET bzv TO 0
        SET cdj TO 10
        SET xjl TO (n - 1) DIV 2
        FOR gaf FROM 0 TO (n + 1 - 1)
            APPEND factorial(gaf) TO zym
        END FOR
        SET bkp TO 0
        SET jmw TO SET()
        SET hfe TO cdj RAISED_TO_POWER xjl
        SET ueg TO hfe
        WHILE ueg <= (hfe * cdj - 1)
            SET lmi TO STRING(ueg)
            SET pna TO REVERSE(lmi)
            SET oxw TO n MOD 2
            SET mrf TO SUBSTRING(pna, oxw, LENGTH(pna))
            SET arb TO CONCAT(lmi, mrf)
            IF NOT ((INTEGER(arb) MOD k) = 0)
                ueg = ueg + 1
                CONTINUE
            END IF
            SET yiz TO SORTED_CHARACTERS(arb)  // returns string sorted ascending
            IF yiz IN jmw
                ueg = ueg + 1
                CONTINUE
            END IF
            ADD yiz TO jmw
            SET vdj TO DICTIONARY()
            FOR ch IN yiz
                IF ch IN vdj
                    vdj[ch] = vdj[ch] + 1
                ELSE
                    vdj[ch] = 1
                END IF
            END FOR
            IF '0' IN vdj AND vdj['0'] > 0
                SET zxk TO n - (vdj['0'] * zym[n - 1])
            ELSE
                SET zxk TO zym[n]
            END IF
            FOR cnt_val IN vdj.VALUES()
                SET zxk TO zxk DIV zym[cnt_val]
            END FOR
            SET bkp TO bkp + zxk
            ueg = ueg + 1
        END WHILE
        RETURN bkp
    END FUNCTION
END CLASS
```
