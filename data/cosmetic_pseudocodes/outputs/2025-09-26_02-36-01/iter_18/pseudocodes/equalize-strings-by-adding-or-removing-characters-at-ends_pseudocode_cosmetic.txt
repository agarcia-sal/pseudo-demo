```pseudocode
CLASS Solution
    FUNCTION minOperations(initial parameter TO string, target parameter TO string) RETURNS integer
        SET count_a TO 0
        SET count_b TO 0
        SET dimension_x TO LENGTH OF initial
        SET dimension_y TO LENGTH OF target
        CREATE matrix Q AS a (dimension_x + 1) BY (dimension_y + 1) 2D list FILLED WITH 0
        SET largest_lcs TO 0

        SET idx_x TO 1
        WHILE idx_x <= dimension_x
            SET idx_y TO 1
            WHILE idx_y <= dimension_y
                SET char_in_a TO initial AT POSITION (idx_x - 1)
                SET char_in_b TO target AT POSITION (idx_y - 1)
                IF NOT (char_in_a != char_in_b) THEN
                    SET Q[idx_x][idx_y] TO Q[idx_x - 1][idx_y - 1] + 1
                    IF largest_lcs < Q[idx_x][idx_y] THEN
                        SET largest_lcs TO Q[idx_x][idx_y]
                    END IF
                END IF
                SET idx_y TO idx_y + 1
            END WHILE
            SET idx_x TO idx_x + 1
        END WHILE

        SET total TO (dimension_x + dimension_y) - ((largest_lcs + largest_lcs))
        RETURN total
    END FUNCTION
END CLASS
```
