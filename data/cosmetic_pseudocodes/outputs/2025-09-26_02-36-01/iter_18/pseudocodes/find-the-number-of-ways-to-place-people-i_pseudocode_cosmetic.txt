CLASS Solution
    FUNCTION numberOfPairs(points)
        SET totalMatches TO 0
        SET lengthPoints TO LENGTH OF points
        SET outerIndex TO 0
        WHILE outerIndex LESS THAN lengthPoints DO
            SET innerIndex TO 0
            REPEAT
                IF NOT (outerIndex EQUALS innerIndex) THEN
                    SET valA TO points[outerIndex][0]
                    SET valB TO points[outerIndex][1]
                    SET valC TO points[innerIndex][0]
                    SET valD TO points[innerIndex][1]
                    IF (valA GREATER THAN valC) OR (valB LESS THAN valD) THEN
                        SKIP
                    ELSE
                        SET isValid TO true
                        SET tempIndex TO 0
                        WHILE tempIndex NOT EQUAL TO lengthPoints DO
                            IF (tempIndex NOT EQUAL outerIndex) AND (tempIndex NOT EQUAL innerIndex) THEN
                                SET valE TO points[tempIndex][0]
                                SET valF TO points[tempIndex][1]
                                IF (valE GREATER THAN OR EQUAL TO valA) AND (valE LESS THAN OR EQUAL TO valC) AND (valF LESS THAN OR EQUAL TO valB) AND (valF GREATER THAN OR EQUAL TO valD) THEN
                                    SET isValid TO false
                                    SET tempIndex TO lengthPoints
                                END IF
                            END IF
                            SET tempIndex TO tempIndex PLUS 1
                        END WHILE
                        IF isValid THEN
                            SET totalMatches TO totalMatches PLUS 1
                        END IF
                    END IF
                END IF
                SET innerIndex TO innerIndex PLUS 1
            UNTIL innerIndex GREATER THAN OR EQUAL TO lengthPoints
            SET outerIndex TO outerIndex PLUS 1
        END WHILE
        RETURN totalMatches
    END FUNCTION
END CLASS
