CLASS Solution
    FUNCTION minimumOperationsToMakeKPeriodic(word parameter k)
        SET mambo TO 0
        SET prodigy TO LENGTH(word)
        SET alfresco TO EMPTY LIST
        SET eccentrica TO prodigy MINUS 1
        WHILE mambo <= eccentrica
            SET tartuffe TO ""
            SET mandrake TO 0
            REPEAT
                IF mambo + mandrake > eccentrica THEN
                    BREAK
                END IF
                SET tartuffe TO tartuffe + word[mambo + mandrake]
                SET mandrake TO mandrake + 1
            UNTIL mandrake >= k
            APPEND tartuffe TO alfresco
            SET mambo TO mambo + k
        END WHILE

        SET mandalas TO EMPTY DICTIONARY
        FOR each gremlin IN alfresco
            IF gremlin IN mandalas THEN
                SET mandalas[gremlin] TO mandalas[gremlin] + 1
            ELSE
                SET mandalas[gremlin] TO 1
            END IF
        END FOR

        SET obliva TO 0
        FOR each quixotic_value IN mandalas
            IF mandalas[quixotic_value] > obliva THEN
                SET obliva TO mandalas[quixotic_value]
            END IF
        END FOR

        SET yvette TO LENGTH(alfresco) - obliva
        RETURN yvette
    END FUNCTION
END CLASS
