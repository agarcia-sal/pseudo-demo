CLASS Solution
    FUNCTION tripletCount(a AS List OF integer, b AS List OF integer, c AS List OF integer) RETURNS integer
        FUNCTION count_even_odd_bits(arr AS List OF integer) RETURNS tuple OF integer and integer
            SET alpha TO 0
            SET iota TO 0
            WHILE iota < LENGTH OF arr
                SET rho TO arr[iota]
                SET sigma TO 0
                SET tau TO rho
                WHILE tau > 0
                    SET sigma TO sigma + (tau AND 1)
                    SET tau TO tau SHIFT_RIGHT 1
                END WHILE
                IF (sigma BITWISE_AND 1) = 0 THEN
                    SET alpha TO alpha + 1
                END IF
                SET iota TO iota + 1
            END WHILE
            SET beta TO LENGTH OF arr - alpha
            RETURN (alpha, beta)
        END FUNCTION

        SET mu TO 0
        SET nu TO 0
        SET mu, nu TO count_even_odd_bits(a)

        SET xi TO 0
        SET omicron TO 0
        SET xi, omicron TO count_even_odd_bits(b)

        SET psi TO 0
        SET omega TO 0
        SET psi, omega TO count_even_odd_bits(c)

        SET delta TO mu * xi * psi
        SET epsilon TO (mu * omicron * omega) + (nu * xi * omega) + (nu * omicron * psi)

        SET zeta TO delta + epsilon
        RETURN zeta
    END FUNCTION
END CLASS
