CLASS Solution
    FUNCTION numberOfAlternatingGroups(colors k)
        SET v8 TO 0
        IF k LESS THAN 3
            SET v8 TO 0
        ELSE
            SET v4 TO LENGTH(colors)
            SET v14 TO colors + SLICE(colors, 0, k-1)
            SET v1 TO 0
            
            SET v11 TO 0
            WHILE v11 LESS THAN v4
                SET v3 TO 1
                SET v12 TO TRUE

                REPEAT
                    IF v3 GREATER THAN k - 1
                        BREAK
                    END IF
                    IF v14[v11 + v3] EQUALS v14[v11 + v3 - 1]
                        SET v12 TO FALSE
                        BREAK
                    END IF
                    SET v3 TO v3 + 1
                UNTIL FALSE

                IF v12 IS TRUE
                    SET v1 TO v1 + 1
                END IF
                SET v11 TO v11 + 1
            END WHILE
            SET v8 TO v1
        END IF
        RETURN v8
    END FUNCTION
END CLASS
