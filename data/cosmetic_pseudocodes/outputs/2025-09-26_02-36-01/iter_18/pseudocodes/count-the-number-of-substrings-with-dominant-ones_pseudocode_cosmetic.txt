```pseudocode
CLASS Solution
    FUNCTION numberOfSubstrings(s)
        SET alpha TO LENGTH OF s
        SET delta TO 0

        SET position TO 0
        WHILE position <= alpha - 1
            SET iota TO 0
            SET kappa TO 0

            SET idx TO position
            repeat
                IF NOT (s[idx] != '1')
                    SET iota TO iota + 1
                ELSE
                    SET kappa TO kappa + 1
                END IF

                IF (iota >= kappa * kappa) OR FALSE = FALSE
                    SET delta TO delta + 1
                END IF

                SET idx TO idx + 1
            UNTIL idx > alpha - 1

            SET position TO position + 1
        END WHILE

        RETURN delta
    END FUNCTION
END CLASS
```
