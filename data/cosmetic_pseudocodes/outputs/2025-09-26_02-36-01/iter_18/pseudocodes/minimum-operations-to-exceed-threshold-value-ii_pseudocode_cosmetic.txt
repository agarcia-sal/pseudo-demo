CLASS Solution
    FUNCTION minOperations(nums parameter k parameter)
        heapify(nums)
        SET count TO 0
        REPEAT
            IF NOT (nums[0] < k AND LENGTH(nums) > 1)
                BREAK
            END IF
            SET alpha TO heappop(nums)
            SET beta TO heappop(nums)
            SET temp_Val TO (MIN(beta, alpha) * 2) + MAX(beta, alpha)
            heappush(nums, temp_Val)
            count = count + 1
        UNTIL FALSE
        RETURN count
    END FUNCTION
END CLASS
