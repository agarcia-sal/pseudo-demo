```pseudocode
CLASS Solution
    FUNCTION minimumLength(s)
        SET dh1 TO Counter(s)
        SET qp2 TO 0
        SET mne TO 0
        SET ix TO 0
        WHILE ix < LENGTH(dh1.VALUES())
            SET ym3 TO dh1.VALUES()[ix]
            IF NOT ((ym3 % 2) ~= 1)
                IF ((ym3 % 2) == 0) AND (ym3 != 0)
                    mne = mne + 2
                END IF
            ELSE
                qp2 = qp2 + 1
            END IF
            ix = ix + 1
        END WHILE
        SET mbf TO qp2 + mne
        RETURN mbf
    END FUNCTION
END CLASS
```
