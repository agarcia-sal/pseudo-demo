```pseudocode
CLASS Solution

    FUNCTION maxTotalReward(rewardValues)
        
        FUNCTION dfs(zeta)
            SET omega TO 0
            SET tau TO 0
            WHILE tau LESS THAN LENGTH(rewardValues) AND rewardValues[tau] LESS THAN OR EQUAL TO zeta
                INCREMENT tau BY 1
            END WHILE

            SET psi TO 0
            WHILE psi LESS THAN LENGTH(rewardValues) MINUS tau
                SET upsilon TO rewardValues[tau + psi]
                IF (NOT (zeta + upsilon LESS THAN OR EQUAL TO zeta)) THEN
                    IF omega LESS THAN (upsilon + dfs(zeta + upsilon)) THEN
                        SET omega TO upsilon + dfs(zeta + upsilon)
                    END IF
                END IF
                INCREMENT psi BY 1
            END WHILE

            RETURN omega
        END FUNCTION

        SET alpha TO LENGTH(rewardValues) - 1
        LABEL StartSort:
            IF alpha GREATER THAN 0 THEN
                SET beta TO 0
                WHILE beta LESS THAN alpha
                    IF rewardValues[beta] GREATER THAN rewardValues[beta + 1] THEN
                        SET gamma TO rewardValues[beta]
                        SET rewardValues[beta] TO rewardValues[beta + 1]
                        SET rewardValues[beta + 1] TO gamma
                    END IF
                    INCREMENT beta BY 1
                END WHILE
                DECREMENT alpha BY 1
                GOTO StartSort
            END IF

        RETURN dfs(0)
    END FUNCTION

END CLASS
```
