CLASS Solution
    FUNCTION shortestSubstrings(arr)
        SET freqMapping TO new dictionary WITH default 0
        SET idxPrimary TO 0
        WHILE idxPrimary < LENGTH(arr)
            SET currentStr TO arr[idxPrimary]
            SET lenStr TO LENGTH(currentStr)
            SET idxStart TO 0
            REPEAT
                SET idxEnd TO idxStart + 1
                REPEAT
                    SET subsegment TO currentStr[idxStart..(idxEnd - 1)]
                    freqMapping[subsegment] = freqMapping[subsegment] + 1
                    SET idxEnd TO idxEnd + 1
                UNTIL idxEnd > lenStr
                SET idxStart TO idxStart + 1
            UNTIL idxStart > (lenStr - 1)
            SET idxPrimary TO idxPrimary + 1
        END WHILE

        SET resultList TO []
        SET posOuter TO 0
        LOOP
            EXIT WHEN posOuter >= LENGTH(arr)
            SET word TO arr[posOuter]
            SET lenWord TO LENGTH(word)
            SET candidateSubstr TO ""
            SET posI TO 0
            LOOP
                EXIT WHEN posI > (lenWord - 1)
                SET posJ TO posI + 1
                LOOP
                    EXIT WHEN posJ > lenWord
                    SET tempSubstr TO word[posI..(posJ - 1)]
                    IF NOT (freqMapping[tempSubstr] != 1) THEN
                        IF (candidateSubstr = "") OR (LENGTH(tempSubstr) < LENGTH(candidateSubstr)) OR ((LENGTH(tempSubstr) = LENGTH(candidateSubstr)) AND (tempSubstr < candidateSubstr)) THEN
                            SET candidateSubstr TO tempSubstr
                        END IF
                    END IF
                    SET posJ TO posJ + 1
                END LOOP
                SET posI TO posI + 1
            END LOOP
            resultList APPEND candidateSubstr
            SET posOuter TO posOuter + 1
        END LOOP

        RETURN resultList
    END FUNCTION
END CLASS
