CLASS Solution
    FUNCTION maximumScore(grid)
        SET alpha TO LENGTH OF grid
        SET omega TO a 2D list OF size (alpha + 1) BY (alpha + 1) FILLED WITH 0
        SET mu TO a list OF size (alpha + 1) FILLED WITH 0
        SET tau TO a list OF size (alpha + 1) FILLED WITH 0

        SET lambda TO 0
        WHILE lambda < alpha
            SET sigma TO 0
            rEpeat
                IF sigma >= alpha THEN
                    BREAK
                END IF
                SET omega[sigma + 1][lambda] TO omega[sigma][lambda] + grid[lambda][sigma]
                SET sigma TO sigma + 1
            UNTIL FALSE
            SET lambda TO lambda + 1
        END WHILE

        SET zeta TO 1
        REPEAT
            IF zeta >= alpha THEN
                BREAK
            END IF

            SET psi TO a list OF size (alpha + 1) FILLED WITH 0
            SET xi TO a list OF size (alpha + 1) FILLED WITH 0

            SET chi TO 0
            REPEAT
                IF chi > alpha THEN
                    BREAK
                END IF

                SET upsilon TO 0
                WHILE upsilon <= alpha
                    IF chi > upsilon THEN
                        SET phi TO omega[chi][zeta - 1] - omega[upsilon][zeta - 1]
                        SET psi[chi] TO MAX(psi[chi], tau[upsilon] + phi)
                        SET xi[chi] TO MAX(xi[chi], tau[upsilon] + phi)
                    ELSE
                        SET phi TO omega[upsilon][zeta] - omega[chi][zeta]
                        SET psi[chi] TO MAX(psi[chi], mu[upsilon] + phi)
                        SET xi[chi] TO MAX(xi[chi], xi[chi], mu[upsilon])
                    END IF
                    SET upsilon TO upsilon + 1
                END WHILE

                SET chi TO chi + 1
            UNTIL FALSE

            ASSIGN psi TO mu
            ASSIGN xi TO tau

            SET zeta TO zeta + 1
        UNTIL FALSE

        SET omegA TO -infinity
        SET beta TO 0
        WHILE beta <= alpha
            IF mu[beta] > omegA THEN
                SET omegA TO mu[beta]
            END IF
            SET beta TO beta + 1
        END WHILE

        RETURN omegA
    END FUNCTION
END CLASS
