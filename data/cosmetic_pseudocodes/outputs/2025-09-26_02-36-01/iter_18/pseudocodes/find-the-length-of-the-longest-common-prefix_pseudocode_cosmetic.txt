CLASS Solution
    FUNCTION longestCommonPrefix(arr1 arr1 OF integers arr2 arr2 OF integers) RETURNS integer
        SET vquxym TO empty set
        SET aznef TO 1
        SET wgupj TO 0
        WHILE aznef <= LENGTH(arr1)
            SET gjzto TO STRING(arr1[aznef])
            SET olexa TO 1
            REPEAT
                ADD SUBSTRING(gjzto, 1, olexa) TO vquxym
                SET olexa TO olexa + 1
            UNTIL olexa > LENGTH(gjzto)
            SET aznef TO aznef + 1
        END WHILE

        SET pmktr TO empty set
        SET ixdlk TO 1
        WHILE ixdlk <= LENGTH(arr2)
            SET cmrjs TO STRING(arr2[ixdlk])
            SET wybhm TO 1
            REPEAT
                ADD SUBSTRING(cmrjs, 1, wybhm) TO pmktr
                SET wybhm TO wybhm + 1
            UNTIL wybhm > LENGTH(cmrjs)
            SET ixdlk TO ixdlk + 1
        END WHILE

        SET kmerl TO 0
        SET idjnh TO vquxym
        SET wejds TO 0
        FOR EACH hyvrp IN idjnh
            IF NOT (hyvrp NOT IN pmktr) THEN
                IF LENGTH(hyvrp) > kmerl THEN
                    SET kmerl TO LENGTH(hyvrp)
                    SET wejds TO 1
                END IF
            END IF
        END FOR

        RETURN kmerl
    END FUNCTION
END CLASS
