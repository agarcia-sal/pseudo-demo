CLASS Solution
    FUNCTION makeAntiPalindrome(s)
        SET gprux TO list of characters in s
        SET mbqes TO length of gprux
        CALL sortAscending(gprux)
        SET ylndz TO mbqes DIV 2

        DEFINE FUNCTION sortAscending(arr)
            # Simple insertion sort implementation replicating ascending sort
            SET tpazc TO 1
            REPEAT
                EXIT WHEN tpazc >= length of arr
                SET qujob TO tpazc
                WHILE qujob > 0 AND arr[qujob] < arr[qujob - 1]
                    SET tempChar TO arr[qujob]
                    SET arr[qujob] TO arr[qujob - 1]
                    SET arr[qujob - 1] TO tempChar
                    SET qujob TO qujob - 1
                END WHILE
                SET tpazc TO tpazc + 1
            UNTIL false
        END FUNCTION

        IF NOT(gprux[ylndz] != gprux[ylndz - 1]) THEN
            SET kqvis TO ylndz
            WHILE kqvis < mbqes AND NOT(gprux[kqvis] != gprux[kqvis - 1])
                SET kqvis TO kqvis + 1
            END WHILE
            SET pwuge TO ylndz
            WHILE pwuge < mbqes AND NOT(gprux[pwuge] != gprux[mbqes - pwuge - 1])
                IF kqvis >= mbqes THEN
                    RETURN "-1"
                END IF
                # Swap elements gprux[kqvis] and gprux[pwuge]
                SET vcduh TO gprux[kqvis]
                SET gprux[kqvis] TO gprux[pwuge]
                SET gprux[pwuge] TO vcduh
    
                SET kqvis TO kqvis + 1
                SET pwuge TO pwuge + 1
            END WHILE
        END IF

        SET rwxmy TO 0
        REPEAT
            EXIT WHEN rwxmy >= ylndz
            IF NOT(gprux[rwxmy] != gprux[mbqes - rwxmy - 1]) THEN
                SET gswza TO false
                SET yfkna TO ylndz
                WHILE yfkna < mbqes
                    IF gprux[yfkna] != gprux[rwxmy] AND gprux[yfkna] != gprux[mbqes - rwxmy - 1] THEN
                        SET tempSwap TO gprux[yfkna]
                        SET gprux[yfkna] TO gprux[rwxmy]
                        SET gprux[rwxmy] TO tempSwap

                        SET gswza TO true
                        BREAK
                    END IF
                    SET yfkna TO yfkna + 1
                END WHILE
                IF NOT gswza THEN
                    RETURN "-1"
                END IF
            END IF
            SET rwxmy TO rwxmy + 1
        UNTIL false

        # Join array into string
        SET zrvel TO ""
        SET lbhtk TO 0
        WHILE lbhtk < mbqes
            SET zrvel TO zrvel + gprux[lbhtk]
            SET lbhtk TO lbhtk + 1
        END WHILE

        RETURN zrvel
    END FUNCTION
END CLASS
