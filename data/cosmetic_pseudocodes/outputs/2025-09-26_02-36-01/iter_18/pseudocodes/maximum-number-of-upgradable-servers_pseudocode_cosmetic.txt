CLASS Solution
    FUNCTION maxUpgrades(count LIST OF INTEGER upgrade LIST OF INTEGER sell LIST OF INTEGER money LIST OF INTEGER) RETURNS LIST OF INTEGER
        SET r04j TO []
        SET t2p1 TO 0
        WHILE t2p1 < LENGTH(count)
            SET j83x TO count[t2p1]
            SET v12m TO upgrade[t2p1]
            SET q9ln TO sell[t2p1]
            SET c57b TO money[t2p1]
            SET bx2u TO 0
            SET w8ks TO c57b
            SET dyt0 TO 0
            REPEAT
                SET fzp3 TO j83x - dyt0
                SET zmv6 TO dyt0 * q9ln
                SET o7tr TO w8ks + zmv6
                SET kx1n TO o7tr / v12m
                IF NOT (kx1n <= fzp3)
                    SET kx1n TO fzp3
                END IF
                IF NOT (kx1n <= bx2u)
                    SET bx2u TO kx1n
                END IF
                SET dyt0 TO dyt0 + 1
            UNTIL dyt0 > j83x
            APPEND bx2u TO r04j
            SET t2p1 TO t2p1 + 1
        END WHILE
        RETURN r04j
    END FUNCTION
END CLASS
