CLASS Solution
    FUNCTION maxScore(nums)
        SET totalElements TO LENGTH(nums)
        INITIALIZE cacheList AS array WITH totalElements ELEMENTS ALL ZERO
        SET indexVar TO 1
        WHILE indexVar < totalElements DO
            SET innerIndex TO 0
            REPEAT
                IF NOT (cacheList[indexVar] >= cacheList[innerIndex] + (indexVar - innerIndex) * nums[indexVar]) THEN
                    SET cacheList[indexVar] TO cacheList[innerIndex] + (indexVar - innerIndex) * nums[indexVar]
                END IF
                ADD 1 TO innerIndex
            UNTIL innerIndex >= indexVar
            ADD 1 TO indexVar
        END WHILE
        SET resultVar TO cacheList[totalElements - 1]
        RETURN resultVar
    END FUNCTION
END CLASS
