CLASS Solution
  FUNCTION beautifulIndices(s parameter a parameter b parameter k)
    SET xqlogn TO new list
    SET vgkzet TO 0
    SET umpfraw TO LENGTH OF s MINUS LENGTH OF a
    REPEAT
      IF vgkzet GREATER THAN umpfraw THEN
        BREAK
      END IF
      SET oqlhah TO SUBSTRING OF s FROM vgkzet TO (vgkzet + LENGTH OF a - 1)
      IF oqlhah EQUALS a THEN
        APPEND vgkzet TO xqlogn
      END IF
      SET vgkzet TO vgkzet + 1
    END REPEAT

    SET btzjil TO new list
    SET uvheaj TO 0
    SET jntxwy TO LENGTH OF s MINUS LENGTH OF b
    REPEAT
      IF uvheaj GREATER THAN jntxwy THEN
        BREAK
      END IF
      SET wumprx TO SUBSTRING OF s FROM uvheaj TO (uvheaj + LENGTH OF b - 1)
      IF wumprx EQUALS b THEN
        APPEND uvheaj TO btzjil
      END IF
      SET uvheaj TO uvheaj + 1
    END REPEAT

    SET vgepqi TO new list
    SET nknqzh TO 0
    SET uncirs TO 0

    WHILE TRUE
      IF NOT (nknqzh LESS THAN LENGTH OF xqlogn AND uncirs LESS THAN LENGTH OF btzjil) THEN
        BREAK
      END IF
      SET yfkwma TO xqlogn[nknqzh] - btzjil[uncirs]
      IF yfkwma LESS THAN 0 THEN
        SET yfkwma TO 0 - yfkwma
      END IF
      IF NOT (yfkwma GREATER THAN k) THEN
        APPEND xqlogn[nknqzh] TO vgepqi
        SET nknqzh TO nknqzh + 1
      ELSE
        IF xqlogn[nknqzh] LESS THAN btzjil[uncirs] THEN
          SET nknqzh TO nknqzh + 1
        ELSE
          SET uncirs TO uncirs + 1
        END IF
      END IF
    END WHILE

    RETURN vgepqi
  END FUNCTION
END CLASS
