CLASS Solution  
    FUNCTION minimumSubstringsInPartition(s)  
        FUNCTION rho(alpha)  
            IF alpha GREATER THAN OR EQUAL TO LENGTH(s) THEN  
                RETURN 0  
            ELSE  
                SET phi TO MAP DEFAULT 0  
                SET psi TO MAP DEFAULT 0  
                SET omega TO LENGTH(s) - alpha  
                SET beta TO alpha  
                WHILE beta LESS THAN LENGTH(s) DO  
                    SET gamma TO s[beta]  
                    IF gamma IN phi AND phi[gamma] NOT EQUAL 0 THEN  
                        SET psi[phi[gamma]] TO psi[phi[gamma]] - 1  
                        IF psi[phi[gamma]] EQUAL 0 THEN  
                            DELETE psi[phi[gamma]]  
                        END IF  
                    END IF  
                    SET phi[gamma] TO phi[gamma] + 1  
                    SET psi[phi[gamma]] TO psi.get(phi[gamma], 0) + 1  
                    IF LENGTH(psi) EQUAL 1 THEN  
                        SET delta TO 1 + rho(beta + 1)  
                        IF delta LESS THAN omega THEN  
                            SET omega TO delta  
                        END IF  
                    END IF  
                    SET beta TO beta + 1  
                END WHILE  
                RETURN omega  
            END IF  
        END FUNCTION  
        SET kappa TO LENGTH(s)  
        RETURN rho(0)  
    END FUNCTION  
END CLASS
