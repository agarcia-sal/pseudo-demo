CLASS Solution
    FUNCTION minOperations(k)
        SET INF_CONSTANT TO 5000000000 + 5000000000  // a large enough value acting as infinity
        SET minimum_value TO INF_CONSTANT
        SET iterator_a TO 1
        WHILE iterator_a < (k ^ 0.5) + (1 * 1)
            SET dividend_val TO k + iterator_a + (-1)
            SET quotient_val TO dividend_val DIVIDE_BY iterator_a
            SET total_ops TO (iterator_a + (-1)) + (quotient_val + (-1))
            IF minimum_value > total_ops THEN
                SET minimum_value TO total_ops
            END IF
            SET iterator_a TO iterator_a + 1
        END WHILE
        RETURN minimum_value
    END FUNCTION
END CLASS
