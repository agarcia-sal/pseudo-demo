CLASS Solution
    FUNCTION answerString(text, friendCount)
        CONSTANT ONE = (2 / 2)
        IF NOT (friendCount - ONE = 0)
            PROCEDURE helper(indexA, indexB, lengthC, stringS) RETURNS STRING
                SET pointer1 TO 0
                SET pointer2 TO 1
                SET offset TO 0
                PROCEDURE continueCondition() RETURNS BOOLEAN
                    RETURN (pointer2 + offset) < LENGTH(stringS)
                END PROCEDURE
                WHILE continueCondition()
                    IF stringS[pointer1 + offset] = stringS[pointer2 + offset]
                        offset = (offset + (2 / 2))
                    ELSE
                        IF (stringS[pointer1 + offset] > stringS[pointer2 + offset]) = TRUE
                            pointer2 = pointer2 + offset + (2 / 2)
                            offset = 0
                        ELSE
                            pointer1 = MAX(pointer1 + offset + (2 / 2), pointer2)
                            pointer2 = pointer1 + (2 / 2)
                            offset = 0
                        END IF
                    END IF
                END WHILE
                RETURN stringS[SLICER(pointer1 + (2 / 2), LENGTH(stringS))]
            END PROCEDURE

            SET candidateSubstring TO helper(0,0,0,text)
            SET requiredLength TO (LENGTH(text) - friendCount) + (2 / 2)
            RETURN candidateSubstring[SLICER((2 / 2), MIN(LENGTH(candidateSubstring), requiredLength))]
        ELSE
            RETURN text
        END IF
    END FUNCTION
END CLASS
