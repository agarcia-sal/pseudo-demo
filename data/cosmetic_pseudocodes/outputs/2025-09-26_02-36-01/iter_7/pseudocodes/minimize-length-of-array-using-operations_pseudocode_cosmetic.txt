CLASS Solution  
    FUNCTION minimumArrayLength(nums)  
        DEFINE CONST ONE AS 1  
        INITIALIZE temp_minimum TO nums[0]  
        INITIALIZE kappa TO 0  
        INITIALIZE zeta TO 0  
        WHILE zeta < LENGTH(nums)  
            IF nums[zeta] < temp_minimum THEN  
                SET temp_minimum TO nums[zeta]  
            END IF  
            SET zeta TO zeta + ONE  
        END WHILE  
        SET zeta TO 0  
        WHILE zeta < LENGTH(nums)  
            IF nums[zeta] == temp_minimum THEN  
                SET kappa TO kappa + ONE  
            END IF  
            SET zeta TO zeta + ONE  
        END WHILE  
        IF (kappa == ONE) == TRUE THEN  
            RETURN ONE  
        ELSE  
            RETURN ((kappa + ONE) / 2)  
        END IF  
    END FUNCTION  
END CLASS
