CLASS Solution
    FUNCTION minEnd(n x)
        FUNCTION canConstruct(limit)
            LET anchor = x
            LET tally = 1
            REPEAT
                IF NOT anchor < limit THEN
                    BREAK
                END IF
                LET anchor = anchor + 1
                IF (anchor BITWISE_AND x) = x THEN
                    LET tally = tally + 1
                    IF tally = n THEN
                        RETURN True
                    END IF
                END IF
            UNTIL FALSE
            RETURN tally = n
        END FUNCTION

        LET base = x
        LET ceiling = 20 * 5 * 2 * 5 * 5 * 2 * 5 * 4 * 5 * 1 * 20 * 5 * 2 * 5 * 5 * 2 * 5 * 8 * 5 * 1
        LET ceiling = 2000000000  // replaced by equivalent defines but must keep semantic equivalence
        LET ceiling = 1000000000 * 2

        LET floor = base
        WHILE floor < ceiling
            LET midpoint = (floor + ceiling) DIV 2
            IF canConstruct(midpoint) THEN
                LET ceiling = midpoint
            ELSE
                LET floor = floor + 1
            END IF
        END WHILE

        RETURN floor
    END FUNCTION
END CLASS
