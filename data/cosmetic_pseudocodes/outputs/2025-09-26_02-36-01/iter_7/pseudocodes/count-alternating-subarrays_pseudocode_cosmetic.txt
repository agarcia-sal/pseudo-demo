CLASS Solution  
    FUNCTION countAlternatingSubarrays(nums)  
        SET totalCount TO 0  
        SET size TO LENGTH OF nums  
        IF NOT (size > 0)  
            RETURN 0  
        END IF  
        SET totalCount TO size  
        SET seqLen TO 1  
        SET index TO 1  
        WHILE index <= (size - 1)  
            IF NOT (nums[index] = nums[(index - 1)])  
                SET seqLen TO (seqLen + 1)  
            ELSE  
                seqLen = 1  
            END IF  
            totalCount = totalCount + (seqLen - 1)  
            index = index + 1  
        END WHILE  
        RETURN totalCount  
    END FUNCTION  
END CLASS
