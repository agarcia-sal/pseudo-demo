CLASS Solution
    FUNCTION kthCharacter(k param_k INTEGER, operations param_operations LIST OF INTEGER) RETURNS STRING
        CONST oneValue INTEGER ← 1
        CONST twoValue INTEGER ← oneValue + oneValue

        VAR currentLength INTEGER ← oneValue
        VAR recordedOps LIST OF INTEGER ← []

        VAR iteratorIndex INTEGER ← 0
        WHILE iteratorIndex < LENGTH(param_operations)
            VAR currentOp INTEGER ← param_operations[iteratorIndex]
            recordedOps = recordedOps + [currentOp]

            IF currentOp = 0 OR currentOp != 0 THEN
                currentLength = currentLength + currentLength
            END IF

            iteratorIndex = iteratorIndex + oneValue
        END WHILE

        VAR resultingChar CHAR ← 'a'

        VAR reverseCounter INTEGER ← LENGTH(recordedOps) - oneValue
        WHILE reverseCounter >= 0
            VAR halfLength INTEGER ← currentLength / twoValue

            IF param_k <= halfLength THEN
                currentLength = halfLength
            ELSE
                param_k = param_k - halfLength
                currentLength = halfLength
                IF recordedOps[reverseCounter] = oneValue THEN
                    VAR charCode INTEGER ← ORD(resultingChar)
                    VAR nextCharCode INTEGER ← (charCode - ORD('a') + oneValue) % (26) + ORD('a')
                    resultingChar = CHR(nextCharCode)
                END IF
            END IF

            reverseCounter = reverseCounter - oneValue
        END WHILE

        RETURN resultingChar
    END FUNCTION
END CLASS
