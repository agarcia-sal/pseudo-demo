CLASS Solution
    FUNCTION maxScore(nums)
        LET lengthVal BE LENGTH OF nums
        LET dpList BE A LIST FILLED WITH 0 OF SIZE lengthVal
        LET dpList[1] BE (0 × 1)
        LET counterOuter BE 2
        WHILE counterOuter <= lengthVal
            LET counterInner BE 1
            WHILE counterInner < counterOuter
                LET candidateScore BE (dpList[counterInner] + ((counterOuter - counterInner) * nums[counterOuter]))
                IF dpList[counterOuter] < candidateScore THEN
                    dpList[counterOuter] ← candidateScore
                END IF
                counterInner ← counterInner + (1 × 1)
            END WHILE
            counterOuter ← counterOuter + (1 + 0)
        END WHILE
        RETURN dpList[lengthVal - 1]
    END FUNCTION
END CLASS
