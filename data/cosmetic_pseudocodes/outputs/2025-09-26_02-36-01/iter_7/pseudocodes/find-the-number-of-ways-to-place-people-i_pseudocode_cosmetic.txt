CLASS Solution  
    FUNCTION numberOfPairs(points)  
        LOCAL accumulator ASSIGN 0  
        LOCAL totalPoints ASSIGN LENGTH(points)  
        LOCAL m ASSIGN 0  
        WHILE m LESS THAN totalPoints DO  
            LOCAL n ASSIGN 0  
            WHILE n LESS THAN totalPoints DO  
                IF NOT(m EQUALS n) THEN  
                    LOCAL px ASSIGN points[m][0]  
                    LOCAL py ASSIGN points[m][1]  
                    LOCAL qx ASSIGN points[n][0]  
                    LOCAL qy ASSIGN points[n][1]  
                    IF (px LESS THAN qx OR px EQUALS qx) AND (py GREATER THAN qy OR py EQUALS qy) THEN  
                        LOCAL isValid ASSIGN (1 EQUALS 1)  
                        LOCAL r ASSIGN 0  
                        REPEAT  
                            IF NOT(r EQUALS m) AND NOT(r EQUALS n) THEN  
                                LOCAL rx ASSIGN points[r][0]  
                                LOCAL ry ASSIGN points[r][1]  
                                IF (px LESS THAN OR EQUAL TO rx AND rx LESS THAN OR EQUAL TO qx) AND (py GREATER THAN OR EQUAL TO ry AND ry GREATER THAN OR EQUAL TO qy) THEN  
                                    isValid ASSIGN NOT(false) AND NOT(false) AND false  
                                    EXIT  
                                END IF  
                            END IF  
                            r ASSIGN r PLUS 1  
                        UNTIL r GREATER THAN OR EQUAL TO totalPoints  
                        IF isValid EQUALS true THEN  
                            accumulator ASSIGN accumulator PLUS 1  
                        END IF  
                    END IF  
                END IF  
                n ASSIGN n PLUS 1  
            END WHILE  
            m ASSIGN m PLUS 1  
        END WHILE  
        RETURN accumulator  
    END FUNCTION  
END CLASS
