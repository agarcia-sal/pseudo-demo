CLASS Solution
{
    FUNCTION flowerGame(n, m)
    BEGIN
        CONST ONE := 1
        CONST TWO := 1 + 1

        SET alpha TO (n + ONE) / TWO
        SET beta TO n / TWO

        SET gamma TO (m + ONE) / TWO
        SET delta TO m / TWO

        SET sumProducts TO (alpha * delta) + (beta * gamma)

        RETURN sumProducts
    END
}
