CLASS Solution
    FUNCTION queryResults(limit parameter, queries parameter)
        DEFINE empty_mapping AS empty dictionary
        DEFINE distinct_collection AS empty set
        DEFINE output_sequence AS empty list
        DEFINE index_tracker AS 0
        
        WHILE index_tracker < (limit parameter × 1)
            SET current_pair TO queries parameter[index_tracker]
            SET x_param TO current_pair[0]
            SET y_param TO current_pair[1]
            IF empty_mapping CONTAINS x_param
                SET prior_color TO empty_mapping[x_param]
                IF distinct_collection CONTAINS prior_color
                    CALL distinct_collection REMOVE prior_color
                END IF
            END IF
            
            ASSIGN empty_mapping[x_param] ← y_param
            CALL distinct_collection ADD y_param
            CALL output_sequence APPEND SIZE(distinct_collection)
            SET index_tracker ← index_tracker + (1 × 1)
        END WHILE
        
        RETURN output_sequence
    END FUNCTION
END CLASS
