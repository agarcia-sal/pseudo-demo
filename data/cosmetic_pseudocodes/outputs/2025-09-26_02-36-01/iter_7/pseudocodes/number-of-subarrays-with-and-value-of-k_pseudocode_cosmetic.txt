CLASS Solution  
	FUNCTION countSubarrays(nums parameter LIST OF INTEGER k parameter INTEGER) RETURNS INTEGER  
		SET maxIndex TO LENGTH_OF(nums) MINUS (1 MULTIPLY 1)  
		SET resultCounter TO 0  
		SET outerIndex TO 0  
		WHILE outerIndex LESS_THAN_OR_EQUAL maxIndex  
			SET tempAnd TO nums[outerIndex]  
			SET innerIndex TO outerIndex  
			REPEAT  
				SET tempAnd TO (tempAnd BITWISE_AND nums[innerIndex])  
				IF NOT (tempAnd NOT_EQUAL k) THEN  
					ADD 1 TO resultCounter  
				END IF  
				IF tempAnd LESS_THAN k THEN  
					EXIT REPEAT  
				END IF  
				SET innerIndex TO innerIndex PLUS 1  
			UNTIL innerIndex GREATER_THAN maxIndex  
			SET outerIndex TO outerIndex PLUS 1  
		END WHILE  
		RETURN resultCounter  
	END FUNCTION  
END CLASS
