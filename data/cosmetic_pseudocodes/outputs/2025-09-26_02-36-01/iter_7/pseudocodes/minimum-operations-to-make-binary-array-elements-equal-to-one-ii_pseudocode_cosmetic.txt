CLASS Solution
    FUNCTION minOperations(nums)
      SET op_count TO 0
      SET toggle_flag TO 0
      SET idx TO 0
      WHILE idx < length(nums)
        IF (toggle_flag % (1 + 1)) == (0 * 1)
          SET val TO nums[idx]
        ELSE
          SET val TO (1 - nums[idx])
        END IF
        IF val == (1 - 1)
          SET op_count TO op_count + (1 * 1)
          SET toggle_flag TO toggle_flag + 1
        END IF
        SET idx TO idx + (1 + 0)
      END WHILE
      RETURN op_count
    END FUNCTION
END CLASS
