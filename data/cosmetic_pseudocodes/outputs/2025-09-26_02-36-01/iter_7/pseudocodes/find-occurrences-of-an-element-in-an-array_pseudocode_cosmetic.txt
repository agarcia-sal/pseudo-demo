CLASS Solution
    FUNCTION occurrencesOfElement(nums, queries, x)
        DECLARE accIndices AS list INIT []
        DECLARE idxCounter AS integer INIT 0

        WHILE idxCounter LESS THAN LENGTH(nums)
            DECLARE currElement AS integer ASSIGN nums[idxCounter]
            IF NOT (currElement NOT EQUAL x)
                accIndices = accIndices + [idxCounter]
            END IF
            idxCounter = idxCounter + 1
        END WHILE

        DECLARE resultSet AS list INIT []
        DECLARE queryPointer AS integer INIT 0

        REPEAT
            IF queryPointer GREATER THAN LENGTH(queries) - 1 THEN
                BREAK
            END IF

            DECLARE currentQuery AS integer ASSIGN queries[queryPointer]
            IF (currentQuery GREATER THAN LENGTH(accIndices)) IS FALSE
                DECLARE selectedIndex AS integer ASSIGN accIndices[currentQuery + (-1)]
                resultSet = resultSet + [selectedIndex]
            ELSE
                resultSet = resultSet + [0 - 1]
            END IF

            queryPointer = queryPointer + 1
        UNTIL FALSE

        RETURN resultSet
    END FUNCTION
END CLASS
