CLASS Solution
	 FUNCTION maxArea(height positions directions)
		 SET limit TO (height + height)
		 SET count TO LENGTH OF positions
		 SET peak_area TO 0
		 SET idx_outer TO 1
		 WHILE idx_outer <= limit
			 SET idx_inner TO 0
			 WHILE idx_inner < count
				 IF (positions[idx_inner] = 0 AND directions[idx_inner] = 'D')
					 SET left_substr TO directions[0 TO (idx_inner - 1)]
					 SET right_substr TO directions[(idx_inner + 1) TO END]
					 SET directions TO left_substr + 'U' + right_substr
				 ELSE IF (positions[idx_inner] = height AND directions[idx_inner] = 'U')
					 SET left_part TO directions[0 TO (idx_inner - 1)]
					 SET right_part TO directions[(idx_inner + 1) TO END]
					 SET directions TO left_part + 'D' + right_part
				 END IF
				 IF (directions[idx_inner] = 'U')
					 SET positions[idx_inner] TO positions[idx_inner] + (1 * 1)
				 ELSE
					 SET positions[idx_inner] TO (positions[idx_inner] - (1 + 0))
				 END IF
				 SET idx_inner TO idx_inner + 1
			 END WHILE
			 SET temp_area TO 0
			 SET idx_sum TO 0
			 WHILE idx_sum < count
				 SET temp_area TO temp_area + positions[idx_sum]
				 SET idx_sum TO idx_sum + 1
			 END WHILE
			 IF peak_area < temp_area
				 SET peak_area TO temp_area
			 END IF
			 SET idx_outer TO idx_outer + 1
		 END WHILE
		 RETURN peak_area
	 END FUNCTION
END CLASS
