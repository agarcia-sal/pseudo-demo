CLASS Solution
    FUNCTION minLength(s parameter representing a string, numOps parameter representing an integer) RETURNS integer
        FUNCTION check(m parameter representing an integer) RETURNS boolean
            SET totalOps TO 0
            SET runLength TO 0
            SET idx TO 0
            REPEAT UNTIL idx GREATER THAN LENGTH OF s MINUS 1
                INCREMENT runLength BY 1
                IF idx EQUALS LENGTH OF s MINUS 1 OR s[idx] NOT EQUALS s[idx + 1] THEN
                    SET groupOps TO (runLength DIV m) + 1
                    SET totalOps TO totalOps + groupOps
                    IF totalOps GREATER THAN numOps THEN
                        RETURN false
                    END IF
                    SET runLength TO 0
                END IF
                INCREMENT idx BY 1
            END REPEAT
            RETURN totalOps LESS THAN OR EQUAL TO numOps
        END FUNCTION

        SET lengthOfS TO LENGTH OF s
        SET lowBound TO 1
        SET highBound TO lengthOfS
        WHILE lowBound LESS THAN highBound DO
            SET midVal TO lowBound + (highBound - lowBound) DIV 2
            IF check(midVal) THEN
                SET highBound TO midVal
            ELSE
                SET lowBound TO midVal + 1
            END IF
        END WHILE
        RETURN lowBound
    END FUNCTION
END CLASS
