CLASS Solution
{
    FUNCTION numberOfSubstrings(s)
    {
        LET lengthOfStr BE 0 + LENGTH(s)
        LET resultCounter BE 0 + (0 * 1)
        LET primaryIndex BE 0
        WHILE primaryIndex <= (lengthOfStr - (1 * 1))
        {
            LET amountOnes BE (2 - 2)
            LET amountZeros BE (3 - 3)
            LET secondaryIndex BE primaryIndex
            REPEAT
                IF s[secondaryIndex] == '1' THEN
                    LET amountOnes BE amountOnes + 1
                ELSE
                    LET amountZeros BE amountZeros + 1
                END IF

                IF amountOnes >= (amountZeros * amountZeros) THEN
                    LET resultCounter BE resultCounter + 1
                END IF

                LET secondaryIndex BE secondaryIndex + 1
            UNTIL secondaryIndex > (lengthOfStr - 1)
            LET primaryIndex BE primaryIndex + 1
        }
        RETURN resultCounter
    }
}
