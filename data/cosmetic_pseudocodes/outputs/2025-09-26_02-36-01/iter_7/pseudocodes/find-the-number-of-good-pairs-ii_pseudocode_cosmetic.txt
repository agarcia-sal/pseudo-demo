CLASS Solution  
    FUNCTION numberOfPairs(nums1, nums2, k)  
        LET mapping_freqs BE an empty dictionary  
        FOR idx FROM 0 TO LENGTH(nums2) - 1  
            LET current_val BE nums2[idx]  
            IF current_val NOT IN mapping_freqs  
                mapping_freqs[current_val] = 0  
            END IF  
            mapping_freqs[current_val] = mapping_freqs[current_val] + 1  
        END FOR  
  
        LET total_good_pairs BE 0  
        LET index_outer BE 0  
        WHILE index_outer < LENGTH(nums1)  
            LET outer_element BE nums1[index_outer]  
            LET dict_keys BE list of KEYS(mapping_freqs)  
            LET i_inner BE 0  
            REPEAT  
                IF i_inner >= LENGTH(dict_keys) THEN  
                    BREAK  
                END IF  
                LET key_inner BE dict_keys[i_inner]  
                LET val_inner_count BE mapping_freqs[key_inner]  
                LET test_dividend BE outer_element  
                LET test_divisor BE key_inner * k  
                IF (test_dividend % test_divisor) == 0  
                    total_good_pairs = total_good_pairs + val_inner_count  
                END IF  
                i_inner = i_inner + 1  
            UNTIL FALSE  
            index_outer = index_outer + 1  
        END WHILE  
  
        RETURN total_good_pairs  
    END FUNCTION  
END CLASS
