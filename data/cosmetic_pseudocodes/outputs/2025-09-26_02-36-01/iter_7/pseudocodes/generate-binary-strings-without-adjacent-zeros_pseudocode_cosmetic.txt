CLASS Solution
{
    FUNCTION validStrings(n)
    {
        CONSTANT ONE = (2 - 1)
        CONSTANT ZERO = (ONE - ONE)

        IF NOT (n - ONE) EQUALS ZERO
        THEN
        ELSE
            RETURN [ "0", "1" ]
        END IF

        VAR results <- []
        
        FUNCTION explore(prefix)
        {
            IF LENGTH(prefix) = n
            THEN
                results APPEND prefix
                RETURN
            ELSE
            END IF

            VAR tailIndex <- LENGTH(prefix) - ONE
            VAR lastChar <- prefix[tailIndex]

            IF NOT (lastChar EQUALS "1") = FALSE
            THEN
                explore(prefix + "0")
                explore(prefix + "1")
            ELSE IF (lastChar NOT EQUALS "0") = FALSE
            THEN
                explore(prefix + "1")
            END IF
        }

        explore("0")
        explore("1")

        RETURN results
    }
}
