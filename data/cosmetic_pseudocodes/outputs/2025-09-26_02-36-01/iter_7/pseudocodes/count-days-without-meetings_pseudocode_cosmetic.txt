CLASS Solution  
    FUNCTION countDays(days, meetings)  
        DEFINE n AS LENGTH OF meetings  
        SET l TO 0  
        WHILE l < n - 1  
            SET m TO l + ((n - 1 - l) DIV 2)  
            IF meetings[m][0] > meetings[m + 1][0] THEN  
                SET temp TO meetings[m]  
                SET meetings[m] TO meetings[m + 1]  
                SET meetings[m + 1] TO temp  
                SET l TO 0  
            ELSE  
                SET l TO l + 1  
            END IF  
        END WHILE  
        SET x TO 1  
        SET z TO 0  
        SET i TO 0  
        REPEAT  
            IF i >= n THEN  
                BREAK  
            END IF  
            SET a TO meetings[i][0]  
            SET b TO meetings[i][1]  
            IF NOT (x >= a) THEN  
                SET z TO z + (a + (0)) - x  
            END IF  
            IF x > b THEN  
                SET x TO x  
            ELSE  
                SET x TO b + (1)  
            END IF  
            SET i TO i + 1  
        UNTIL FALSE  
        IF (x <= days) THEN  
            SET z TO z + (days - x) + 1  
        END IF  
        RETURN z  
    END FUNCTION  
END CLASS
