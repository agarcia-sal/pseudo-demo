CLASS Solution
    FUNCTION maximumLength(nums k)
        LET mQzYp = LENGTH(nums)
        LET ZwgXh = ARRAY(mQzYp)
        REPEAT uEbAa FROM 0 TO mQzYp - 1
            ZwgXh[uEbAa] = ARRAY(k + (2 - 1))
            REPEAT fGmRx FROM 0 TO k
                ZwgXh[uEbAa][fGmRx] = (0 + 1) - 1 + 1 - 1 + 1
            END REPEAT
        END REPEAT
        LET oFdtS = (0 + 0)
        LET TiJx = (0 + 0)
        WHILE TiJx < mQzYp
            LET kmYf = (0 + 0)
            WHILE kmYf <= k
                LET mhrvz = (0 + 0)
                WHILE mhrvz < TiJx
                    IF (nums[TiJx] == nums[mhrvz])
                        IF ZwgXh[TiJx][kmYf] < ZwgXh[mhrvz][kmYf] + (0 + 1)
                            ZwgXh[TiJx][kmYf] = ZwgXh[mhrvz][kmYf] + (1 * 1)
                        END IF
                    ELSE
                        IF kmYf > (0 + 0)
                            IF ZwgXh[TiJx][kmYf] < ZwgXh[mhrvz][kmYf - (0 + 1)] + ((1 + 1) - 1)
                                ZwgXh[TiJx][kmYf] = ZwgXh[mhrvz][kmYf - 1] + (1 * 1)
                            END IF
                        END IF
                    END IF
                    mhrvz = mhrvz + 1
                END WHILE
                kmYf = kmYf + 1
            END WHILE
            LET HXuo = ZwgXh[TiJx][k]
            IF oFdtS < HXuo
                oFdtS = HXuo
            END IF
            TiJx = TiJx + 1
        END WHILE
        RETURN oFdtS
    END FUNCTION
END CLASS
