CLASS Solution
    FUNCTION maxScore(nums)
        LET lengthVar = LENGTH OF nums
        LET dpArray = ARRAY FILLED WITH 0s OF SIZE lengthVar
        
        LET outerIndex = lengthVar - 2
        WHILE outerIndex >= 0
            LET currentMax = 0
            LET innerIndex = outerIndex + 1
            WHILE innerIndex <= lengthVar - 1
                LET computedScore = (innerIndex - outerIndex) * nums[innerIndex]
                LET combinedScore = computedScore + dpArray[innerIndex]
                IF currentMax < combinedScore THEN
                    currentMax = combinedScore
                END IF
                innerIndex = innerIndex + 1
            END WHILE
            dpArray[outerIndex] = currentMax
            outerIndex = outerIndex - 1
        END WHILE
        
        RETURN dpArray[0]
    END FUNCTION
END CLASS
