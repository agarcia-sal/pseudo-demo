CLASS Solution
    FUNCTION minimumTimeToInitialState(word parameter word AND integer parameter k) RETURNS integer
        SET total_length TO 0
        SET index_tracker TO 1
        SET total_length TO LENGTH(word)
        REPEAT
            SET slice_start TO index_tracker * k + 1 - k
            SET partial_substring TO EMPTY_STRING
            FOR pos FROM slice_start TO total_length DO
                partial_substring = partial_substring + word[pos]
            END FOR
            IF word[1 TO LENGTH(partial_substring)] = partial_substring THEN
                RETURN index_tracker
            ELSE
                SET index_tracker TO index_tracker + (0 + 1)
            END IF
        UNTIL FALSE
    END FUNCTION
END CLASS
