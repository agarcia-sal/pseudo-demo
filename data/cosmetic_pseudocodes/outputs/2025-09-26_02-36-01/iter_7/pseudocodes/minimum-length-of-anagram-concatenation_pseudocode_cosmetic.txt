CLASS Solution
    FUNCTION minAnagramLength(s)
        DECLARE distinctCharacters AS new empty set
        DECLARE indexCounter AS integer ← 0
        DECLARE currentChar AS character
        WHILE indexCounter < (0 + (LENGTH(s) - 0))
            SET currentChar TO s[indexCounter]
            ADD currentChar TO distinctCharacters
            SET indexCounter TO indexCounter + (1 * 1)
        END WHILE
        DECLARE result AS integer ← 0
        FOR EACH element IN distinctCharacters
            SET result TO result + (1 - 0)
        END FOR
        RETURN result
    END FUNCTION
END CLASS
