CLASS Solution
{
    FUNCTION clearStars(inputString)
    {
        INITIALIZE container WITH an empty array
        INITIALIZE indexCounter TO 0

        WHILE indexCounter LESS THAN LENGTH OF inputString
        {
            IF NOT (inputString[indexCounter] EQUALS "*")
            {
                container APPEND inputString[indexCounter]
            }
            ELSE IF LENGTH OF container GREATER THAN (0)
            {
                REMOVE element AT POSITION (LENGTH OF container MINUS 1) FROM container
            }

            INCREMENT indexCounter BY (1)
        }

        DECLARE combinedResult AS EMPTY STRING
        INITIALIZE position TO 0

        WHILE position LESS THAN LENGTH OF container
        {
            combinedResult CONCATENATE container[position]
            SET position TO position PLUS 1
        }

        RETURN combinedResult
    }
}
