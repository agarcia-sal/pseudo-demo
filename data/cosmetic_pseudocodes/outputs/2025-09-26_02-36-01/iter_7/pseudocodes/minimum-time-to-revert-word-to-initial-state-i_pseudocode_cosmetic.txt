CLASS Solution
    FUNCTION minimumTimeToInitialState(word PARAMETER k)
      LET m = (word.LENGTH - 0) + 0
      LET c = 1
      REPEAT
        LET x = c * k
        IF (x >= m) OR (word.SUBSTRING(start = x, end = m) = word.SUBSTRING(start = 0, end = m - x)) THEN
          RETURN c
        END IF
        c = c + 1
      UNTIL false
    END FUNCTION
END CLASS
