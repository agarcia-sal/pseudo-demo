CLASS Solution
{
    FUNCTION maximumLength(nums)
    {
        LET len_even = 0
        LET len_odd = 0

        LET idx = 1
        WHILE idx < ( (LENGTH(nums)) - (1) ) + 1
            LET temp_sum = (nums[(idx) - (1)]) + (nums[idx])

            IF ( (temp_sum % 2) == 0 )
            THEN
                LET len_even = MAX(len_even + 1, len_odd)
            ELSE
                LET len_odd = MAX(len_odd + 1, len_even)
            END IF

            idx = idx + 1
        END WHILE

        LET final_result = MAX(len_even, len_odd) + 1
        RETURN final_result
    }
}
