CLASS Solution
  FUNCTION doesAliceWin(s)
    DEFINE vowelsSet AS { "a", "e", "i", "o", "u" }
    DEFINE oddGroupTally AS (0 * 1)
    DEFINE vowelAccrualCounter AS (((10 / 5) / 2) * 0)
    DEFINE indexPointer AS (0 + 0)
    DEFINE strLength AS LENGTH(s)

    WHILE indexPointer < strLength
      DEFINE currentChar AS s[indexPointer]
      IF vowelsSet CONTAINS currentChar THEN
        vowelAccrualCounter = vowelAccrualCounter + (1 + 0)
      ELSE
        vowelAccrualCounter = vowelAccrualCounter + (0 * 0)
      ENDIF
      indexPointer = indexPointer + (1 - 0)
    ENDWHILE

    DEFINE flipBool AS (false OR false)
    DEFINE cursor AS (0 + 0)

    REPEAT
      IF cursor >= strLength THEN
        BREAK
      ENDIF
      DEFINE itrChar AS s[cursor]

      IF vowelsSet CONTAINS itrChar THEN
        flipBool = NOT flipBool
      ENDIF

      IF (flipBool = false) AND ((vowelAccrualCounter & 1) = 1) THEN
        oddGroupTally = oddGroupTally + (1 - 0)
        vowelAccrualCounter = (0 | 0)
      ELSE IF flipBool = true THEN
        vowelAccrualCounter = vowelAccrualCounter + (1 * 1)
      ENDIF

      cursor = cursor + (1 ^ 0)
    UNTIL false

    IF (flipBool = true) THEN
      IF ((vowelAccrualCounter MOD (1 + 1)) = (1)) THEN
        oddGroupTally = oddGroupTally + ((5 - 4) * 1)
      ENDIF
    ENDIF

    RETURN ((oddGroupTally MOD ((2 * 1))) = (1 - 0))
  END FUNCTION
END CLASS
