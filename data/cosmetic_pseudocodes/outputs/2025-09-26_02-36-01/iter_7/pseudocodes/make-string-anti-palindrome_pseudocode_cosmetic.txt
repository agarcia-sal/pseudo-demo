CLASS Solution
    FUNCTION makeAntiPalindrome(s)
        LET charArray BE characters from s sorted ascending
        LET lengthVal BE count of charArray
        LET halfLen BE lengthVal DIV 2

        LET pointerOne BE halfLen
        IF charArray[halfLen] == charArray[halfLen - 1] THEN
            WHILE pointerOne < lengthVal AND charArray[pointerOne] == charArray[pointerOne - 1] DO
                pointerOne = pointerOne + 1
            END WHILE

            LET pointerTwo BE halfLen
            WHILE pointerTwo < lengthVal AND charArray[pointerTwo] == charArray[lengthVal - pointerTwo - 1] DO
                IF pointerOne >= lengthVal THEN
                    RETURN "-1"
                END IF
                LET tempChar BE charArray[pointerOne]
                charArray[pointerOne] = charArray[pointerTwo]
                charArray[pointerTwo] = tempChar

                pointerOne = pointerOne + 1
                pointerTwo = pointerTwo + 1
            END WHILE
        END IF

        LET index BE 0
        WHILE index < halfLen DO
            IF charArray[index] == charArray[lengthVal - index - 1] THEN
                LET foundSwap BE FALSE
                LET kIndex BE halfLen
                REPEAT
                    IF kIndex >= lengthVal THEN
                        EXIT
                    END IF
                    IF charArray[kIndex] != charArray[index] AND charArray[kIndex] != charArray[lengthVal - index - 1] THEN
                        LET tmp BE charArray[kIndex]
                        charArray[kIndex] = charArray[index]
                        charArray[index] = tmp
                        foundSwap = TRUE
                        BREAK
                    END IF
                    kIndex = kIndex + 1
                UNTIL FALSE

                IF NOT foundSwap THEN
                    RETURN "-1"
                END IF
            END IF
            index = index + 1
        END WHILE

        LET response STRING BE ""
        LET buildIndex BE 0
        WHILE buildIndex < lengthVal DO
            response = response + charArray[buildIndex]
            buildIndex = buildIndex + 1
        END WHILE
        RETURN response
    END FUNCTION
END CLASS
