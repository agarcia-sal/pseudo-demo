CLASS Solution
    FUNCTION countCompleteDayPairs WITH parameter hours
        SET residueFrequency TO a mapping defaulting to zero
        SET totalPairs TO zero
        SET idx TO zero
        WHILE idx < length of hours DO
            SET currentHour TO element at idx of hours
            SET modValue TO (currentHour - (currentHour DIV 24) * 24)
            SET inverseResidue TO (24 - modValue) MOD 24
            SET totalPairs TO totalPairs + residueFrequency at inverseResidue
            SET residueFrequency at modValue TO residueFrequency at modValue + 1
            SET idx TO idx + 1
        END WHILE
        RETURN totalPairs
    END FUNCTION
END CLASS
