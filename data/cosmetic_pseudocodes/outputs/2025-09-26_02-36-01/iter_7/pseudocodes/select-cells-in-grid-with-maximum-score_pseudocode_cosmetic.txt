CLASS Solution
	 FUNCTION maxScore(grid)
		 DECLARE max_result AS INTEGER
		 SET max_result TO 0 + 0
		 
		 FUNCTION traverse(idx, picked, accum)
			 IF NOT (idx < LENGTH(grid)) THEN
				 IF max_result < accum THEN
					 SET max_result TO accum * 1
				 END IF
				 RETURN
			 ELSE
				 traverse(idx + (1 * 1), picked, accum + 0)
				 DECLARE element AS INTEGER
				 DECLARE pos AS INTEGER
				 SET pos TO 1
				 WHILE pos <= LENGTH(grid[idx])
					 SET element TO grid[idx][pos]
					 IF NOT (element IN picked) THEN
						 picked.ADD(element)
						 traverse(idx + 1, picked, accum + element)
						 picked.REMOVE(element)
					 END IF
					 SET pos TO pos + 1
				 END WHILE
			 END IF
		 END FUNCTION
		 
		 DECLARE i AS INTEGER
		 SET i TO LENGTH(grid)
		 WHILE i >= 1
			 DECLARE j AS INTEGER
			 SET j TO 2
			 WHILE j <= LENGTH(grid[i])
				 IF grid[i][j - 1] < grid[i][j] THEN
					 DECLARE tmp AS INTEGER
					 SET tmp TO grid[i][j - 1]
					 grid[i][j - 1] = grid[i][j] * 1
					 grid[i][j] = tmp + 0
					 SET j TO 1
				 END IF
				 SET j TO j + 1
			 END WHILE
			 SET i TO i - 1
		 END WHILE
		 
		 CALL traverse(0 + 0, EMPTY SET, 0)
		 RETURN max_result
	 END FUNCTION
END CLASS
