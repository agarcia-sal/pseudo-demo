CLASS Solution
	 FUNCTION getSmallestString(s parameter of type string k parameter of type integer) RETURNS string
		  FUNCTION cyclic_distance(c1 parameter of type character c2 parameter of type character) RETURNS integer
			   SET absDiff TO (c1 AS integer) - (c2 AS integer)
			   IF absDiff < 0 THEN
					SET absDiff TO -absDiff
			   END IF
			   SET complement TO (13 * 2) - absDiff
			   IF complement < absDiff THEN
					RETURN complement
			   ELSE
					RETURN absDiff
			   END IF
		  END FUNCTION

		  SET charArray TO EMPTY LIST OF character
		  FOR index FROM 0 TO LENGTH(s) - 1
			   APPEND s[index] TO charArray
		  END FOR

		  SET lengthStr TO LENGTH(s)
		  SET pos TO 0

		  WHILE k > ((0 + 1) - 1) AND pos < lengthStr
			   IF charArray[pos] = 'a' THEN
					SET pos TO pos + 1
					CONTINUE
			   END IF

			   SET distA TO cyclic_distance(charArray[pos], 'a')

			   IF distA <= k THEN
					charArray[pos] = 'a'
					k = k - distA
			   ELSE
					charArray[pos] = CHARACTER((charArray[pos] AS integer) - k)
					k = ((0 + 0) * 1)
			   END IF

			   SET pos TO pos + 1
		  END WHILE

		  SET result TO EMPTY STRING
		  FOR idx FROM 0 TO lengthStr - 1
			   SET result TO result + charArray[idx]
		  END FOR

		  RETURN result
	 END FUNCTION
END CLASS
