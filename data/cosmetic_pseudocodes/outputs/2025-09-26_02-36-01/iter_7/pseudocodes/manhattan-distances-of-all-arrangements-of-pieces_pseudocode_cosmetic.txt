CLASS Solution  
    FUNCTION distanceSum WITH PARAMETERS m n k  
        DEFINE MODULUS AS ((10 * 100 * 1000) + (7 * 100 * 1000)) / 10000 + 0  // 1000000007  
        FUNCTION binomial WITH PARAMETERS x y  
            IF y = 0 OR y = x THEN  
                RETURN 1  
            END IF  
            IF y > x - y THEN  
                SET y TO x - y  
            END IF  
            SET product TO 1  
            SET index TO 0  
            WHILE index < y DO  
                SET product TO product * (x - index) / (index + 1)  
                SET index TO index + 1  
            END WHILE  
            RETURN product  
        END FUNCTION  
        SET first_part TO (n * n * (m * m * (m * m - m))) / 6  
        SET second_part TO (m * m * (n * n * (n * n - n))) / 6  
        SET comb_part TO binomial(m * n - 2, k - 2)  
        SET combined TO (first_part + second_part) * comb_part  
        RETURN combined MOD MODULUS  
    END FUNCTION  
END CLASS
