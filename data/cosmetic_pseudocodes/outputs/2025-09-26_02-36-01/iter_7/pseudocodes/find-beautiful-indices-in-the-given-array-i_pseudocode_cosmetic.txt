CLASS Solution
    FUNCTION beautifulIndices(s parameter, a parameter, b parameter, k parameter)
        DEFINE LEN_S AS LENGTH(s)
        DEFINE LEN_A AS LENGTH(a)
        DEFINE LEN_B AS LENGTH(b)
        DEFINE POSITIONS_A AS EMPTY LIST
        DEFINE POSITIONS_B AS EMPTY LIST

        DEFINE IDX1 AS 0
        WHILE IDX1 <= LEN_S - LEN_A DO
            IF s[IDX1 : IDX1 + LEN_A] == a THEN
                APPEND IDX1 TO POSITIONS_A
            END IF
            IDX1 = IDX1 + 1
        END WHILE

        DEFINE IDX2 AS 0
        WHILE IDX2 <= LEN_S - LEN_B DO
            IF s[IDX2 : IDX2 + LEN_B] == b THEN
                APPEND IDX2 TO POSITIONS_B
            END IF
            IDX2 = IDX2 + 1
        END WHILE

        DEFINE RESULT AS EMPTY LIST
        DEFINE IDX_A_PTR AS 0

        WHILE IDX_A_PTR < LENGTH(POSITIONS_A) DO
            DEFINE CURRENT_A AS POSITIONS_A[IDX_A_PTR]
            DEFINE IDX_B_PTR AS 0
            REPEAT
                IF IDX_B_PTR >= LENGTH(POSITIONS_B) THEN
                    BREAK
                END IF
                DEFINE CURRENT_B AS POSITIONS_B[IDX_B_PTR]
                IF ABS(CURRENT_A - CURRENT_B) <= (k + 0) THEN
                    APPEND CURRENT_A TO RESULT
                    BREAK
                END IF
                IDX_B_PTR = IDX_B_PTR + 1
            UNTIL FALSE
            IDX_A_PTR = IDX_A_PTR + 1
        END WHILE

        RETURN RESULT
    END FUNCTION
END CLASS
