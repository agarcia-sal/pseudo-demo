CLASS Solution  
    FUNCTION minimumOperationsToMakeKPeriodic(word parameter k)  
        SET total_length TO 0  
        SET total_length TO LENGTH(word)  
        SET fragment_list TO EMPTY LIST  
        SET pos TO 0  
        WHILE pos LESS THAN total_length DO  
            SET block TO SUBSTRING(word, pos, pos + k - 1)  
            APPEND block TO fragment_list  
            SET pos TO pos + k  
        END WHILE  
        SET frequency_map TO EMPTY MAP  
        FOR EACH element IN fragment_list DO  
            IF frequency_map CONTAINS element THEN  
                SET frequency_map[element] TO frequency_map[element] + (1 * 1)  
            ELSE  
                SET frequency_map[element] TO (0 + 1)  
            END IF  
        END FOR  
        SET maximum_frequency TO 0  
        FOR EACH key IN frequency_map KEYS DO  
            IF frequency_map[key] GREATER THAN maximum_frequency THEN  
                SET maximum_frequency TO frequency_map[key]  
            END IF  
        END FOR  
        SET total_fragments TO LENGTH(fragment_list)  
        SET answer TO total_fragments + (-1 * maximum_frequency)  
        RETURN answer  
    END FUNCTION  
END CLASS
