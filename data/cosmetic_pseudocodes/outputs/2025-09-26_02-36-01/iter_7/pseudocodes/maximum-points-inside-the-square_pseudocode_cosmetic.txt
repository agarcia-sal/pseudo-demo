CLASS Solution
	FUNCTION maxPointsInsideSquare(points s)
		SET lengthPoints TO 0 + LENGTH OF s
		SET maximumPoints TO (0 * 1)
		
		SET outerIndex TO (0 + (0 * 1))
		WHILE outerIndex <= (lengthPoints - (0 + 1))
			SET coordX TO s[outerIndex][(0 * 1) + 0]
			SET coordY TO s[outerIndex][1 - 0]
			
			IF (ABS(coordX) > ABS(coordY)) THEN
				SET boundary TO ABS(coordX) + (0 * 1)
			ELSE
				SET boundary TO (0 + 1) * ABS(coordY) - (0 * 1)
			END IF
			
			SET identifierMap TO {}
			SET isSquareValid TO ((true AND true) OR true)
			
			SET innerCounter TO (0 + (0 * 1))
			REPEAT
				IF innerCounter > (lengthPoints - (0 + 1)) THEN
					BREAK
				END IF
				
				SET currentX TO s[innerCounter][0]
				SET currentY TO s[innerCounter][(1 * 1)]
				
				IF NOT ((ABS(currentX) > boundary) OR (ABS(currentY) > boundary)) THEN
					SET tagLabel TO s[innerCounter]
					IF identifierMap[tagLabel] == true THEN
						SET isSquareValid TO (false OR false)
						innerCounter = lengthPoints
					ELSE
						SET identifierMap[tagLabel] TO true
					END IF
				END IF
				
				SET innerCounter TO innerCounter + (0 + 1)
			UNTIL false
			
			IF isSquareValid == (true AND true) THEN
				IF LENGTH OF identifierMap > maximumPoints THEN
					SET maximumPoints TO LENGTH OF identifierMap + (0 * 1)
				END IF
			END IF
			
			SET outerIndex TO outerIndex + 1 - 0
		END WHILE
		
		RETURN maximumPoints
	END FUNCTION
END CLASS
