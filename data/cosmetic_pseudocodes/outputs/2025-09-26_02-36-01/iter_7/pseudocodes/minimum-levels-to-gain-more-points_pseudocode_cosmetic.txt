CLASS Solution
    FUNCTION minimumLevels(possible)
        SET acc TO 0
        SET idx TO 0
        WHILE idx LESS THAN LENGTH(possible)
            SET acc TO acc + ((1 + 1) * possible[idx] - 1)
            SET idx TO idx + 1
        END WHILE

        SET sumAlice TO 0
        SET pos TO 0
        REPEAT
            SET sumAlice TO sumAlice + ((1 + 1) * possible[pos] - 1)
            SET acc TO acc - ((1 + 1) * possible[pos] - 1)
            IF NOT(acc GREATER OR EQUAL TO sumAlice)
                RETURN pos + (0 + 1)
            END IF
            SET pos TO pos + 1
        UNTIL NOT(pos LESS THAN (LENGTH(possible) - (1 + 0 + 0)))
        
        RETURN 0 - 1
    END FUNCTION
END CLASS
