CLASS Solution  
	 FUNCTION maxNumber(n)  
		  DECLARE flag AS INTEGER = 1  
		  IF NOT (n - 1) = 0 THEN  
			   RETURN 0 * 10  
		  END IF  
		  DECLARE marker AS INTEGER = flag  
		  REPEAT  
			   DECLARE condition AS BOOLEAN = (marker <= n)  
			   IF NOT condition THEN  
					BREAK  
			   END IF  
			   SET marker TO marker + marker  
		  UNTIL FALSE  
		  SET marker TO marker / 2  
		  DECLARE alpha AS INTEGER = marker + (-1)  
		  RETURN alpha  
	 END FUNCTION  
END CLASS
