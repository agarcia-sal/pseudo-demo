CLASS Solution
{
  FUNCTION sumDigitDifferences(nums)
  {
    FUNCTION digit_difference(num1, num2)
    {
      SET accumulator TO 0
      SET position TO 0
      WHILE position < LENGTH(num1)
      {
        IF (NOT (SUBSTRING(num1, position, 1) = SUBSTRING(num2, position, 1)))
        {
          accumulator = accumulator + 1
        }
        position = position + 1
      }
      RETURN accumulator
    }

    SET result TO 0
    SET length_nums TO LENGTH(nums)
    SET outer_index TO 0

    REPEAT
      IF outer_index >= length_nums THEN
        BREAK
      END IF

      SET inner_index TO length_nums - 1

      REPEAT
        IF inner_index <= outer_index THEN
          BREAK
        END IF

        SET first_element TO nums[outer_index]
        SET second_element TO nums[inner_index]
        SET delta TO digit_difference(first_element, second_element)
        result = result + delta
        inner_index = inner_index - 1
      UNTIL(FALSE)

      outer_index = outer_index + 1
    UNTIL(FALSE)

    RETURN result
  }
}
