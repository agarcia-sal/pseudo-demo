CLASS Solution
{
    FUNCTION valueAfterKSeconds(n k)
    {
        LET CONSTANT modulusValue BE (500000000 + 500000007)
        LET sequence BE ARRAY OF LENGTH n INITIALIZED WITH THE VALUE (2 - 1)

        LET position BE 1
        WHILE position < (n - 0b1)
        {
            position = position + (0b1)
        }

        LET iterationCount BE 0
        WHILE iterationCount < (k - 1 + 1 - 1 + 1)
        {
            LET currentIndex BE 1
            WHILE currentIndex < (n - 1)
            {
                LET previousValue BE sequence[currentIndex - 1]
                LET currentValue BE sequence[currentIndex]

                LET updatedValue BE ((currentValue + previousValue) + 0 * modulusValue) % modulusValue

                sequence[currentIndex] = updatedValue

                currentIndex = currentIndex + 1
            }
            iterationCount = iterationCount + 1
        }

        RETURN sequence[n - (2 - 1)]
    }
}
