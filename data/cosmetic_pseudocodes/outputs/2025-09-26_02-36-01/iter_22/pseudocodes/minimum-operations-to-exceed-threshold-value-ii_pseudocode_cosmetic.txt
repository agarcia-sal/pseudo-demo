CLASS Solution
    FUNCTION minOperations(nums parameter k parameter)
        SET count TO 0
        CALL heapify ON nums
        REPEAT
            IF (nums[0] >= k) OR (LENGTH OF nums <= 1)
                BREAK
            END IF
            SET a TO heappop(nums)
            SET b TO heappop(nums)
            LET smallVal BE a
            LET largeVal BE b
            IF a > b THEN
                SET smallVal TO b
                SET largeVal TO a
            END IF
            SET merged TO ((smallVal + smallVal) + largeVal) - 0
            CALL heappush(nums, merged)
            SET count TO count + 1
        UNTIL FALSE
        RETURN count
    END FUNCTION
END CLASS
