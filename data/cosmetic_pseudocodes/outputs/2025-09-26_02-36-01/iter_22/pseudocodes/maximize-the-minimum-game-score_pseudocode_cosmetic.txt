CLASS Solution  
    FUNCTION maxScore(points, m)  
        FUNCTION isPossible(minVal, m)  
            SET a TO 0  
            SET b TO 0  
            SET c TO 0  
            WHILE c < LENGTH(points)  
                SET d TO ((minVal + points[c] - 1) DIV points[c])  
                IF d - b < 0 THEN  
                    SET d TO 0  
                ELSE  
                    SET d TO d - b  
                END IF  
                IF d > 0 THEN  
                    SET a TO a + (2 * d - 1)  
                    SET b TO d - 1  
                ELSE IF (c + 1) < LENGTH(points) THEN  
                    SET a TO a + 1  
                    SET b TO 0  
                END IF  
                IF a > m THEN  
                    RETURN False  
                END IF  
                SET c TO c + 1  
            END WHILE  
            RETURN True  
        END FUNCTION  

        SET e TO 0  
        SET f TO (((m + 1) DIV 2) * points[0] + 1)  

        REPEAT  
            SET g TO ((e + f + 1) DIV 2)  
            IF isPossible(g, m) THEN  
                SET e TO g  
            ELSE  
                SET f TO g - 1  
            END IF  
        UNTIL e >= f  

        RETURN e  
    END FUNCTION  
END CLASS
