CLASS Solution
    FUNCTION minimumLevels(possible)
        SET alpha TO 0
        SET beta TO 0
        SET gamma TO 0
        SET lenVal TO (LENGTH(possible) - 2)
        FOR ptr FROM 0 TO LENGTH(possible) - 1
            SET delta TO possible[ptr]
            SET gamma TO gamma + ((2 * delta) - 1)
        END FOR
        WHILE alpha <= lenVal
            SET epsilon TO possible[alpha]
            SET beta TO beta + ((2 * epsilon) - 1)
            SET gamma TO gamma - ((2 * epsilon) - 1)
            IF NOT (beta <= gamma)
                RETURN alpha + 1
            END IF
            SET alpha TO alpha + 1
        END WHILE
        RETURN (-1)
    END FUNCTION
END CLASS
