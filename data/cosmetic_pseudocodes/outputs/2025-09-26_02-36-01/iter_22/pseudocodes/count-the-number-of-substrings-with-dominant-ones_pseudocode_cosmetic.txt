CLASS Solution
    FUNCTION numberOfSubstrings(s)
        LET totalLength BE LENGTH s
        LET accumulator BE 0
        LET indexA BE 0
        WHILE indexA <= totalLength - 1
            LET sumOnes BE 0
            LET sumZeros BE 0
            LET indexB BE indexA
            REPEAT
                IF NOT (s[indexB] â‰  '1')
                    LET sumOnes BE sumOnes + 1
                ELSE
                    LET sumZeros BE sumZeros + 1
                END IF
                IF (sumOnes >= (sumZeros * sumZeros))
                    LET accumulator BE accumulator + 1
                END IF
                LET indexB BE indexB + 1
            UNTIL indexB > totalLength - 1
            LET indexA BE indexA + 1
        END WHILE
        RETURN accumulator
    END FUNCTION
END CLASS
