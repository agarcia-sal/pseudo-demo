CLASS Solution
    FUNCTION countCompleteDayPairs WITH parameter hours
        SET accumulator TO 0
        SET lookup TO an empty mapping defaulting to zero
        SET index TO 0
        WHILE index < LENGTH(hours)
            SET current_value TO (hours[index] - (hours[index] DIV 24) * 24)
            SET complement_value TO (24 - current_value + (current_value DIV 24) * 24) MOD 24
            SET accumulator TO accumulator + lookup[complement_value]
            SET lookup[current_value] TO lookup[current_value] + 1
            SET index TO index + 1
        END WHILE
        RETURN accumulator
    END FUNCTION
END CLASS
