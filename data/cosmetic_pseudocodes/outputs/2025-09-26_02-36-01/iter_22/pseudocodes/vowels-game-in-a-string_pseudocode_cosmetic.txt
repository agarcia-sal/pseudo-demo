CLASS Solution
    FUNCTION doesAliceWin(s)
        SET kf TO 0
        SET mcyh TO 0
        SET bfncdgq TO { "u", "o", "a", "i", "e" }
        LET pdiw = 1
        WHILE pdiw <= LENGTH(s)
            IF s[pdiw] IN bfncdgq THEN
                mcyh = mcyh + 1
            END IF
            pdiw = pdiw + 1
        END WHILE
        SET ybrehu TO false
        SET mxvte TO 1
        WHILE mxvte <= LENGTH(s)
            IF s[mxvte] IN bfncdgq THEN
                ybrehu = NOT ybrehu
            END IF
            IF ybrehu = false THEN
                IF (mcyh MOD 2) = 1 THEN
                    kf = kf + 1
                    mcyh = 0
                END IF
            ELSE
                mcyh = mcyh + 1
            END IF
            mxvte = mxvte + 1
        END WHILE
        IF ybrehu = true THEN
            IF (mcyh MOD 2) = 1 THEN
                kf = kf + 1
            END IF
        END IF
        RETURN (kf MOD 2) = 1
    END FUNCTION
END CLASS
