CLASS Solution
    FUNCTION minOperations(initial parameter TO string, target parameter TO string) RETURNS integer
        SET q TO 0
        SET f TO 0
        SET u TO LENGTH OF initial
        SET v TO LENGTH OF target
        SET w TO a list of lists of zeros with size (u + 1) by (v + 1)
        SET x TO 1
        WHILE x LESS THAN OR EQUAL TO u
            SET y TO 1
            WHILE y LESS THAN OR EQUAL TO v
                IF NOT (initial element at (x - 1) NOT EQUAL TO target element at (y - 1)) THEN
                    LET w[x][y] = w[x - 1][y - 1] + 1
                    IF w[x][y] GREATER THAN q THEN
                        SET q TO w[x][y]
                    END IF
                END IF
                SET y TO y + 1
            END WHILE
            SET x TO x + 1
        END WHILE
        RETURN u + v - 2 * q
    END FUNCTION
END CLASS
