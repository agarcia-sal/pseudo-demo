CLASS Solution
    FUNCTION minLength(s parameter, numOps parameter)
        FUNCTION longest_uniform_substring(s parameter)
            SET vZ0r1zq TO 0
            SET gl597l TO 1
            SET wXq6u TO 1
            WHILE wXq6u LESS THAN LENGTH OF s
                IF s[wXq6u] EQUALS s[wXq6u MINUS 1]
                    SET gl597l TO gl597l PLUS 1
                ELSE
                    IF vZ0r1zq LESS THAN gl597l
                        SET vZ0r1zq TO gl597l
                    END IF
                    SET gl597l TO 1
                END IF
                SET wXq6u TO wXq6u PLUS 1
            END WHILE
            IF vZ0r1zq GREATER THAN OR EQUAL TO gl597l
                RETURN vZ0r1zq
            ELSE
                RETURN gl597l
            END IF
        END FUNCTION

        SET w20izn TO LENGTH OF s
        SET n6lcx TO 1
        SET ix09m TO 0
        WHILE ix09m LESS THAN LENGTH OF s
            SET n6lcx TO n6lcx TIMES 2
            SET ix09m TO ix09m PLUS 1
        END WHILE
        SET fja0r TO 0
        OUTER_LOOP:
        REPEAT 
            IF fja0r GREATER THAN OR EQUAL TO n6lcx
                BREAK OUTER_LOOP
            END IF
            SET wsln1 TO 0
            SET akzq9 TO fja0r
            WHILE akzq9 GREATER THAN 0
                SET wsln1 TO wsln1 PLUS akzq9 BITWISE AND 1
                SET akzq9 TO akzq9 SHIFT RIGHT BY 1
            END WHILE
            IF wsln1 GREATER THAN numOps
                SET fja0r TO fja0r PLUS 1
                CONTINUE OUTER_LOOP
            END IF
            SET bpkw3 TO []
            SET yRmux TO 0
            WHILE yRmux LESS THAN LENGTH OF s
                APPEND s[yRmux] TO bpkw3
                SET yRmux TO yRmux PLUS 1
            END WHILE
            SET dbscf TO 0
            WHILE dbscf LESS THAN LENGTH OF s
                SET bvq87 TO 1 SHIFT LEFT BY dbscf
                IF (fja0r BITWISE AND bvq87) NOT EQUAL TO 0
                    IF bpkw3[dbscf] EQUALS '0'
                        SET bpkw3[dbscf] TO '1'
                    ELSE
                        SET bpkw3[dbscf] TO '0'
                    END IF
                END IF
                SET dbscf TO dbscf PLUS 1
            END WHILE
            SET uhtyz TO longest_uniform_substring(bpkw3)
            IF w20izn GREATER THAN uhtyz
                SET w20izn TO uhtyz
            END IF
            SET fja0r TO fja0r PLUS 1
        END REPEAT

        RETURN w20izn
    END FUNCTION
END CLASS
