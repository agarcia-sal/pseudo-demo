CLASS Solution  
    FUNCTION numberOfWays WITH PARAMETER n  
        SET X TO (10 * (10 ^ 8)) + 7  
        SET Z TO [0] REPEATED (n + 1) TIMES  
        Z[0] ← 1  
        SET C TO [6, 2, 1]  
        FOR M FROM 0 TO LENGTH(C) - 1  
            SET v TO C[M]  
            SET e TO v  
            WHILE e ≤ n  
                Z[e] ← (Z[e] + Z[e - v]) MOD X  
                SET e TO e + 1  
            END WHILE  
        END FOR  
        SET W TO 0  
        SET q TO 0  
        LOOP  
            EXIT WHEN q > 2  
            IF (q * 4) ≤ n THEN  
                W ← (W + Z[n - q * 4]) MOD X  
            END IF  
            SET q TO q + 1  
        END LOOP  
        RETURN W  
    END FUNCTION  
END CLASS
