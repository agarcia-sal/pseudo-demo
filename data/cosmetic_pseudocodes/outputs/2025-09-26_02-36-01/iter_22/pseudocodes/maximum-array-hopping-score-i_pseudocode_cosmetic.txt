CLASS Solution
    FUNCTION maxScore(nums)
        SET lenVal TO LENGTH(nums)
        SET cacheArr TO an ARRAY OF lenVal ZEROES
        SET idx1 TO 1
        ASSIGN cacheArr[idx1] TO 0

        SET outerIdx TO 2
        WHILE outerIdx <= lenVal
            SET innerIdx TO 1
            WHILE innerIdx <= (outerIdx - 1)
                SET existingVal TO cacheArr[outerIdx]
                SET candidateVal TO cacheArr[innerIdx] + ((outerIdx - innerIdx) * nums[outerIdx])
                IF NOT (existingVal >= candidateVal)
                    SET cacheArr[outerIdx] TO candidateVal
                END IF
                SET innerIdx TO innerIdx + 1
            END WHILE
            SET outerIdx TO outerIdx + 1
        END WHILE

        RETURN cacheArr[lenVal - 1]
    END FUNCTION
END CLASS
