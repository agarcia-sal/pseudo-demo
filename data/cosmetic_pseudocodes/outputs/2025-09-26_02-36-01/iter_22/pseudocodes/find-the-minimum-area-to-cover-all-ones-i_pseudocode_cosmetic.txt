CLASS Solution
    FUNCTION minimumArea(grid)
        IF NOT (grid IS NOT empty AND element at position zero of grid IS NOT empty)
            RETURN 0
        END IF
        SET a1 TO LENGTH OF grid
        SET a2 TO LENGTH OF element at position zero of grid
        SET uA TO +∞
        SET uB TO -∞
        SET uC TO +∞
        SET uD TO -∞
        SET p TO 0
        WHILE p < a1
            SET q TO 0
            WHILE q < a2
                IF element at position p of grid AT position q = 1 THEN
                    IF uA > p THEN
                        SET uA TO p
                    END IF
                    IF uB < p THEN
                        SET uB TO p
                    END IF
                    IF uC > q THEN
                        SET uC TO q
                    END IF
                    IF uD < q THEN
                        SET uD TO q
                    END IF
                END IF
                SET q TO q + 1
            END WHILE
            SET p TO p + 1
        END WHILE
        SET hgt TO (uB - uA) + 1
        SET wid TO (uD - uC) + 1
        RETURN hgt * wid
    END FUNCTION
END CLASS
