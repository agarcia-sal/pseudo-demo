CLASS Solution
		FUNCTION waysToReachStair WITH PARAMETER k
				FUNCTION dfs WITH PARAMETERS x y z
						IF NOT (x LESS THAN OR EQUAL TO k PLUS one) THEN
								RETURN 0
						END IF
						SET result TO 0
						IF x EQUALS k THEN
								SET result TO 1
						END IF
						IF (x GREATER THAN 0) AND (y EQUALS 0) THEN
								SET result TO result PLUS dfs(x MINUS 1, 1, z)
						END IF
						SET result TO result PLUS dfs(x PLUS (2 TIMES z), 0, z PLUS 1)
						RETURN result
				END FUNCTION
				RETURN dfs(1, 0, 0)
		END FUNCTION
END CLASS
