CLASS Solution  
    FUNCTION maxArea(height positions directions)  
        SET totalPositionsCount TO 0  
        SET maxAreaValue TO 0  
        SET indexCounter TO 0  
        WHILE indexCounter LESS THAN THE LENGTH OF positions  
            SET totalPositionsCount TO totalPositionsCount PLUS positions[indexCounter]  
            INCREMENT indexCounter BY 1  
        END WHILE  
        SET maxAreaValue TO totalPositionsCount  
        
        SET timeStep TO 1  
        WHILE timeStep LESS THAN OR EQUAL TO height TIMES 2  
            SET posIndex TO 0  
            WHILE posIndex LESS THAN THE LENGTH OF positions  
                IF positions[posIndex] EQUALS 0 AND directions[posIndex] EQUALS 'D'  
                    SET directions TO CONCAT(SUBSTRING(directions, 0, posIndex), 'U', SUBSTRING(directions, posIndex + 1))  
                ELSE  
                    IF positions[posIndex] EQUALS height AND directions[posIndex] EQUALS 'U'  
                        SET directions TO CONCAT(SUBSTRING(directions, 0, posIndex), 'D', SUBSTRING(directions, posIndex + 1))  
                    END IF  
                END IF  
                
                IF directions[posIndex] EQUALS 'U'  
                    SET positions[posIndex] TO positions[posIndex] PLUS 1  
                ELSE  
                    SET positions[posIndex] TO positions[posIndex] MINUS 1  
                END IF  
                
                SET posIndex TO posIndex PLUS 1  
            END WHILE  
            
            SET currentSum TO 0  
            SET sumCounter TO 0  
            WHILE sumCounter LESS THAN THE LENGTH OF positions  
                SET currentSum TO currentSum PLUS positions[sumCounter]  
                SET sumCounter TO sumCounter PLUS 1  
            END WHILE  
            
            IF maxAreaValue LESS THAN currentSum  
                SET maxAreaValue TO currentSum  
            END IF  
            
            SET timeStep TO timeStep PLUS 1  
        END WHILE  
        RETURN maxAreaValue  
    END FUNCTION  
END CLASS
