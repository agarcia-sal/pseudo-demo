CLASS Solution  
    FUNCTION sumOfPower(nums parameter, k parameter)  
        SET modulus TO 1000000007  
        SET lengthNums TO LENGTH OF nums  
        SET dpArray TO a COLLECTION  
        SET idxOuter TO 0  
        
        WHILE idxOuter LESS THAN OR EQUAL TO lengthNums  
            SET tempList TO an EMPTY COLLECTION  
            SET idxInner TO 0  
            WHILE idxInner LESS THAN OR EQUAL TO k  
                APPEND 0 TO tempList  
                SET idxInner TO idxInner PLUS 1  
            END WHILE  
            APPEND tempList TO dpArray  
            SET idxOuter TO idxOuter PLUS 1  
        END WHILE  
        
        SET dpArray[0][0] TO 1  
        
        SET posI TO 1  
        WHILE posI LESS THAN OR EQUAL TO lengthNums  
            SET posJ TO 0  
            WHILE posJ LESS THAN OR EQUAL TO k  
                SET dpArray[posI][posJ] TO dpArray[posI - 1][posJ]  
                
                IF posJ GREATER THAN OR EQUAL TO nums[posI - 1] THEN  
                    SET dpArray[posI][posJ] TO dpArray[posI][posJ] PLUS dpArray[posI - 1][posJ - nums[posI - 1]]  
                END IF  
                
                SET dpArray[posI][posJ] TO dpArray[posI][posJ] MOD modulus  
                SET posJ TO posJ PLUS 1  
            END WHILE  
            SET posI TO posI PLUS 1  
        END WHILE  
        
        SET indexCounter TO 1  
        SET totalPowerAccum TO 0  
        WHILE indexCounter LESS THAN (2 EXP lengthNums)  
            SET sumCurrent TO 0  
            SET countSetBits TO 0  
            SET bitPosition TO 0  
            WHILE bitPosition LESS THAN lengthNums  
                IF ((indexCounter DIV (2 EXP bitPosition)) MOD 2) EQUALS 1 THEN  
                    SET sumCurrent TO sumCurrent PLUS nums[bitPosition]  
                    SET countSetBits TO countSetBits PLUS 1  
                END IF  
                SET bitPosition TO bitPosition PLUS 1  
            END WHILE  
            
            IF sumCurrent EQUALS k THEN  
                SET totalPowerAccum TO totalPowerAccum PLUS (2 EXP (lengthNums MINUS countSetBits))  
                SET totalPowerAccum TO totalPowerAccum MOD modulus  
            END IF  
            SET indexCounter TO indexCounter PLUS 1  
        END WHILE  
        
        RETURN totalPowerAccum  
    END FUNCTION  
END CLASS
