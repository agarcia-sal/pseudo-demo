CLASS Solution  
    FUNCTION numberOfRightTriangles(grid)  
        SET U8L2WZ TO 0  
        SET PU7Q3S TO 0  
        SET TNZ6RJ TO 0  
        SET A7EYVM TO 0  
        SET BIR4SJ TO 0  
        SET XCM59P TO 0  
        SET GH0FXN TO 0  
        SET DE75U2 TO 0  
        SET OQZKW1 TO LENGTH(grid)  
        SET R5Y8MT TO LENGTH(grid[0])  
        WHILE U8L2WZ < OQZKW1  
            SET PU7Q3S TO 0  
            WHILE PU7Q3S < R5Y8MT  
                IF grid[U8L2WZ][PU7Q3S] = 1 THEN  
                    SET TNZ6RJ TO 0  
                    SET XCM59P TO 0  
                    SET GH0FXN TO 0  
                    WHILE GH0FXN < R5Y8MT  
                        IF GH0FXN ≠ PU7Q3S THEN  
                            SET YK3V1J TO grid[U8L2WZ][GH0FXN]  
                            SET TNZ6RJ TO TNZ6RJ + YK3V1J  
                        END IF  
                        SET GH0FXN TO GH0FXN + 1  
                    END WHILE  
                    SET DE75U2 TO 0  
                    WHILE DE75U2 < OQZKW1  
                        IF DE75U2 ≠ U8L2WZ THEN  
                            SET BLACKN TO grid[DE75U2][PU7Q3S]  
                            SET XCM59P TO XCM59P + BLACKN  
                        END IF  
                        SET DE75U2 TO DE75U2 + 1  
                    END WHILE  
                    SET BIR4SJ TO TNZ6RJ * XCM59P  
                    SET A7EYVM TO A7EYVM + BIR4SJ  
                END IF  
                SET PU7Q3S TO PU7Q3S + 1  
            END WHILE  
            SET U8L2WZ TO U8L2WZ + 1  
        END WHILE  
        RETURN A7EYVM  
    END FUNCTION  
END CLASS
