CLASS Solution
    FUNCTION minEnd(n x)
        FUNCTION canConstruct(max_val)
            SET delta TO x
            SET idx TO 1
            REPEAT
                IF NOT (delta < max_val) THEN BREAK
                SET delta TO delta + 1
                IF (delta & x) = x THEN
                    SET idx TO idx + 1
                    IF idx = n THEN
                        RETURN True
                    END IF
                END IF
            UNTIL FALSE
            RETURN idx = n
        END FUNCTION

        SET alpha TO x
        SET omega TO 2 * (10 * (10 * (10 * (10 * (10 * (10 * (10 * 10)))))))
        WHILE alpha <> omega
            SET beta TO (alpha + omega) \ 2
            IF canConstruct(beta)
                SET omega TO beta
            ELSE
                SET alpha TO alpha + 1
            END IF
            IF alpha >= omega THEN BREAK
        END WHILE

        RETURN alpha
    END FUNCTION
END CLASS
