CLASS Solution
    FUNCTION maxNumber(n)
        SET p TO 1
        IF NOT (n NOT EQUAL one)
            RETURN 0
        END IF
        WHILE NOT (p GREATER THAN n)
            SET p TO 1 + p + (p - 1)
        END WHILE
        SET q TO p DIV 2
        SET r TO q - 1
        RETURN r
    END FUNCTION
END CLASS
