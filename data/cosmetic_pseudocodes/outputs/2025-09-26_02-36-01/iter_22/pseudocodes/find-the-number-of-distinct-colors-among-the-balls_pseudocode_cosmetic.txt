CLASS Solution
    FUNCTION queryResults(limit parameter, queries parameter)
        DECLARE map_alpha AS DICTIONARY
        DECLARE set_beta AS SET
        DECLARE list_gamma AS LIST
        SET map_alpha TO EMPTY DICTIONARY
        SET set_beta TO EMPTY SET
        SET list_gamma TO EMPTY LIST

        SET idx_mu TO 0
        WHILE idx_mu < LENGTH(queries parameter)
            DECLARE val_rho, val_sigma
            SET val_rho, val_sigma TO queries parameter[idx_mu]

            IF CONTAINS KEY(map_alpha, val_rho) THEN
                DECLARE val_tau
                SET val_tau TO map_alpha[val_rho]

                IF INCLUDES(set_beta, val_tau) THEN
                    REMOVE val_tau FROM set_beta
                END IF
            END IF

            SET map_alpha[val_rho] TO val_sigma
            INSERT val_sigma INTO set_beta
            APPEND SIZE(set_beta) TO list_gamma

            SET idx_mu TO idx_mu + 1
        END WHILE

        RETURN list_gamma
    END FUNCTION
END CLASS
