CLASS Solution
    FUNCTION countSubstrings(s)
        SET lengthVar TO 0
        SET accCount TO 0
        WHILE lengthVar < LENGTH(s)
            SET innerCounter TO lengthVar
            SET convNum TO 0
            REPEAT
                SET digitVal TO INTEGER VALUE OF s AT innerCounter
                SET convNum TO (convNum * 10) + digitVal
                IF (digitVal â‰  0) AND ((convNum MOD digitVal) = 0)
                    SET accCount TO accCount + 1
                END IF
                SET innerCounter TO innerCounter + 1
            UNTIL innerCounter = LENGTH(s)
            SET lengthVar TO lengthVar + 1
        END WHILE
        RETURN accCount
    END FUNCTION
END CLASS
