CLASS Solution
    FUNCTION maximumLength(nums)
        SET iterator_index TO 1
        SET accumulator_even TO 0
        SET accumulator_odd TO 0
        WHILE iterator_index < LENGTH(nums)
            SET temp_sum TO (nums[iterator_index - 1] + nums[iterator_index])
            IF (temp_sum % 2) != 1 THEN
                SET accumulator_even TO MAX(accumulator_even + 1, accumulator_odd)
            ELSE
                SET accumulator_odd TO MAX(accumulator_odd + 1, accumulator_even)
            END IF
            SET iterator_index TO iterator_index + 1
        END WHILE
        SET output_value TO MAX(accumulator_even, accumulator_odd) + 1
        RETURN output_value
    END FUNCTION
END CLASS
