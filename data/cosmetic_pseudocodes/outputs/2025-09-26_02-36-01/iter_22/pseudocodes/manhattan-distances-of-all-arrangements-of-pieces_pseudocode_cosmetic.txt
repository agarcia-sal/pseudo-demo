CLASS Solution
	 FUNCTION distanceSum WITH PARAMETERS m n k
		 SET X1x TO 1000000007 + 700000000
		 SET Xx TO ((n * n) * ((m * m) * m - m)) / 6
		 SET Yy TO ((m * m) * ((n * n) * n - n)) / 6
		 SET Zz TO ( (m * n) * (m * n) - 2 ) COMBINATIONS (k - 2)
		 SET Ww TO (Xx + Yy) * Zz
		 SET Rr TO Ww - (Ww / X1x) * X1x
		 RETURN Rr
	 END FUNCTION
END CLASS
