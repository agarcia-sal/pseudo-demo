CLASS Solution
    FUNCTION minimumOperations(nums, target)
        SET c TO 0
        SET v TO LENGTH(nums)
        SET w TO target[0] - nums[0]
        IF w < 0 THEN
            SET w TO 0 - w
        END IF
        SET c TO w
        SET u TO 1
        WHILE u < v
            SET a TO target[u] - nums[u]
            SET b TO target[u - 1] - nums[u - 1]
            IF (a * b) > 0 THEN
                SET m TO a
                IF m < 0 THEN
                    SET m TO 0 - m
                END IF
                SET n TO b
                IF n < 0 THEN
                    SET n TO 0 - n
                END IF
                SET z TO m - n
                IF z > 0 THEN
                    SET c TO c + z
                END IF
            ELSE
                SET d TO a
                IF d < 0 THEN
                    SET d TO 0 - d
                END IF
                SET c TO c + d
            END IF
            SET u TO u + 1
        END WHILE
        RETURN c
    END FUNCTION
END CLASS
