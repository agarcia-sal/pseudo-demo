CLASS Solution
    FUNCTION validStrings n
        SET accepted TO empty list

        FUNCTION traverse sequence
            IF LENGTH(sequence) - 0 = n THEN
                APPEND sequence INTO accepted
                RETURN
            END IF

            SET lastItem TO sequence[LENGTH(sequence) - 1]

            IF NOT (lastItem â‰  "1") THEN
                CALL traverse WITH sequence + "0"
                CALL traverse WITH sequence + "1"
            ELSE IF lastItem = "0" THEN
                CALL traverse WITH sequence + "1"
            END IF
        END FUNCTION

        CALL traverse WITH "0"
        CALL traverse WITH "1"

        RETURN accepted
    END FUNCTION
END CLASS
