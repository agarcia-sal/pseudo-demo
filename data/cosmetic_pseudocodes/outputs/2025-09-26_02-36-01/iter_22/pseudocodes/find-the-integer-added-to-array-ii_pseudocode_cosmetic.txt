CLASS Solution
    FUNCTION minimumAddedInteger(nums1 RETURNS List OF integer, nums2 RETURNS List OF integer) RETURNS integer
        CALL sort ON nums1
        CALL sort ON nums2

        SET a TO 0
        WHILE a < LENGTH OF nums1 MINUS 1 DO
            SET b TO a PLUS 1
            WHILE b < LENGTH OF nums1 DO
                SET tempList TO an empty list
                SET c TO 0
                WHILE c < a DO
                    APPEND nums1 AT c TO tempList
                    SET c TO c PLUS 1
                END WHILE
                SET d TO a PLUS 1
                WHILE d < b DO
                    APPEND nums1 AT d TO tempList
                    SET d TO d PLUS 1
                END WHILE
                SET e TO b PLUS 1
                WHILE e < LENGTH OF nums1 DO
                    APPEND nums1 AT e TO tempList
                    SET e TO e PLUS 1
                END WHILE

                SET diff TO nums2 AT 0 MINUS tempList AT 0

                SET valid TO TRUE
                SET f TO 0
                LOOP
                    IF f >= LENGTH OF nums2 THEN
                        EXIT LOOP
                    END IF
                    IF tempList AT f PLUS diff NOT EQUALS nums2 AT f THEN
                        SET valid TO FALSE
                        EXIT LOOP
                    END IF
                    SET f TO f PLUS 1
                END LOOP

                IF valid THEN
                    RETURN diff
                END IF

                SET b TO b PLUS 1
            END WHILE
            SET a TO a PLUS 1
        END WHILE

        RETURN None
    END FUNCTION
END CLASS
