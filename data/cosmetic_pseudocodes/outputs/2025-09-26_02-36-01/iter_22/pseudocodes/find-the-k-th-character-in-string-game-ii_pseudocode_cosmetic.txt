CLASS Solution
    FUNCTION kthCharacter(k param_k INTEGER, operations param_operations LIST OF INTEGER) RETURNS STRING
        SET idx_var TO 0
        SET acc_len TO 1
        SET accumulator TO empty list

        WHILE idx_var LESS THAN LENGTH(param_operations)
            APPEND param_operations[idx_var] TO accumulator
            SET temp_len TO acc_len + acc_len
            SET acc_len TO temp_len
            SET idx_var TO idx_var + 1
        END WHILE

        SET current_char TO 'a'

        SET position TO LENGTH(accumulator) - 1
        WHILE position GREATER THAN OR EQUAL TO 0
            SET half_len TO acc_len / 2
            IF param_k GREATER THAN half_len THEN
                SET param_k TO param_k - half_len
                SET acc_len TO half_len
                IF accumulator[position] EQUALS 1 THEN
                    IF current_char EQUALS 'z' THEN
                        SET current_char TO 'a'
                    ELSE
                        SET current_char TO CHARACTER(ASCII(current_char) + 1)
                    END IF
                END IF
            ELSE
                SET acc_len TO half_len
            END IF
            SET position TO position - 1
        END WHILE

        RETURN current_char
    END FUNCTION
END CLASS
