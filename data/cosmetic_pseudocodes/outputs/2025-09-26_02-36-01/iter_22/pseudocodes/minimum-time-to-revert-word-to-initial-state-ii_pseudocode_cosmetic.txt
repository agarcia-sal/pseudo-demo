CLASS Solution
    FUNCTION minimumTimeToInitialState(word parameter word AND integer parameter k) RETURNS integer
        SET o0 TO LENGTH(word)
        SET o1 TO 1
        REPEAT
            SET o2 TO SUBSTRING(word, o1 * k, o0 - (o1 * k))
            IF NOT (word DOES NOT START WITH o2) THEN
                RETURN o1
            END IF
            SET o1 TO o1 + 1
        UNTIL FALSE
    END FUNCTION
END CLASS
