CLASS Solution
    FUNCTION largestPalindrome(n k)
        IF NOT (n NOT EQUALS 1) THEN
            SET c0 TO 9
            WHILE c0 GREATER_THAN_EQUALS 1
                IF (c0 MOD k) EQUALS 0 THEN
                    RETURN STRING(c0)
                END IF
                SET c0 TO c0 MINUS 1
            END WHILE
            RETURN STRING(0)
        END IF

        SET s_len TO (n PLUS 1) DIV 2
        SET base_str TO ""
        SET idx0 TO 1
        WHILE idx0 LESS_THAN_OR_EQUALS s_len
            SET base_str TO base_str CONCAT "9"
            SET idx0 TO idx0 PLUS 1
        END WHILE
        SET half_val TO INTEGER(base_str)

        REPEAT
            SET h_str TO STRING(half_val)
            IF (n MOD 2) EQUALS 0 THEN
                SET full_str TO h_str CONCAT REVERSE(h_str)
            ELSE
                SET sub_part TO ""
                SET idx1 TO 1
                WHILE idx1 LESS_THAN (LENGTH(h_str))
                    SET sub_part TO sub_part CONCAT h_str[idx1]
                    SET idx1 TO idx1 PLUS 1
                END WHILE
                SET full_str TO h_str CONCAT REVERSE(sub_part)
            END IF

            SET full_val TO INTEGER(full_str)
            IF (full_val MOD k) EQUALS 0 THEN
                RETURN STRING(full_val)
            END IF

            SET half_val TO half_val MINUS 1
        UNTIL half_val LESS_THAN_OR_EQUALS 0

        RETURN STRING(0)
    END FUNCTION
END CLASS
