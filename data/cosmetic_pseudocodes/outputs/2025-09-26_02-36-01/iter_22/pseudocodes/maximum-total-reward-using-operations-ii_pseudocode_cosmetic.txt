CLASS Solution
	FUNCTION maxTotalReward(rewardValues)
		SET auxiliary TO 1
		SET sortedUniqueValues TO an empty collection
		WHILE true
			IF rewardValues IS EMPTY
				BREAK
			END IF
			SET minElem TO rewardValues[0]
			FOR each element IN rewardValues
				IF element < minElem
					SET minElem TO element
				END IF
			END FOR
			IF minElem NOT IN sortedUniqueValues
				APPEND minElem TO sortedUniqueValues
			END IF
			SET filtered TO an empty collection
			FOR each elem IN rewardValues
				IF elem != minElem
					APPEND elem TO filtered
				END IF
			END FOR
			SET rewardValues TO filtered
		END WHILE

		SET index TO 0
		WHILE index < LENGTH(sortedUniqueValues)
			SET currentValue TO sortedUniqueValues[index]
			SET shiftLeftVal TO (1 << currentValue) - 1
			SET combinedShift TO shiftLeftVal << currentValue
			SET bitwiseAndRes TO auxiliary & combinedShift
			SET auxiliary TO auxiliary | bitwiseAndRes
			INCREMENT index BY 1
		END WHILE

		SET bitCount TO 0
		SET tempVal TO auxiliary
		WHILE tempVal > 0
			SET tempVal TO tempVal >> 1
			SET bitCount TO bitCount + 1
		END WHILE
		RETURN bitCount - 1
	END FUNCTION
END CLASS
