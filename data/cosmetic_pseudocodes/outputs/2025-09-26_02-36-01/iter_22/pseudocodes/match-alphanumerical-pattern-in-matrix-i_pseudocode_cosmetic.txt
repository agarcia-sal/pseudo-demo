CLASS Solution
    FUNCTION findPattern(board, pattern)
        SET v9 TO LENGTH(board)
        SET w0 TO LENGTH(board[0])
        SET x3 TO LENGTH(pattern)
        SET y7 TO LENGTH(pattern[0])
        
        FUNCTION matches(d1, e2)
            SET mapA TO new empty dictionary
            SET mapB TO new empty dictionary
            
            SET i4 TO 0
            WHILE i4 < x3
                SET j5 TO 0
                WHILE j5 < y7
                    SET chZ TO pattern[i4][j5]
                    SET digY TO board[d1 + i4][e2 + j5]

                    IF NOT (chZ IS NOT a digit)
                        IF (INTEGER(chZ) <> digY) RETURN False
                    ELSE
                        IF mapA CONTAINS chZ
                            IF mapA[chZ] <> digY RETURN False
                        ELSE
                            IF mapB CONTAINS digY RETURN False
                            mapA[chZ] = digY
                            mapB[digY] = chZ
                        END IF
                    END IF
                    SET j5 TO j5 + 1
                END WHILE
                SET i4 TO i4 + 1
            END WHILE
            RETURN True
        END FUNCTION
        
        SET m6 TO 0
        WHILE m6 <= v9 - x3
            SET n8 TO 0
            WHILE n8 <= w0 - y7
                IF matches(m6, n8) RETURN [m6, n8]
                SET n8 TO n8 + 1
            END WHILE
            SET m6 TO m6 + 1
        END WHILE
        
        RETURN [-1, -1]
    END FUNCTION
END CLASS
