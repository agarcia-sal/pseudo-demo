CLASS Solution  
    FUNCTION clearStars(s)  
        SET helper_list TO createEmptyList()  
        SET idx TO 1  
        WHILE idx <= lengthOf(s)  
            SET current_element TO elementAt(s, idx)  
            IF current_element != "*" THEN  
                appendElement(helper_list, current_element)  
            ELSE IF sizeOf(helper_list) > 0 THEN  
                removeLastElement(helper_list)  
            END IF  
            SET idx TO idx + 1  
        END WHILE  
        SET accumulator TO emptyString()  
        FOR EACH element IN helper_list DO  
            SET accumulator TO concatenate(accumulator, element)  
        END FOR  
        RETURN accumulator  
    END FUNCTION  
END CLASS
