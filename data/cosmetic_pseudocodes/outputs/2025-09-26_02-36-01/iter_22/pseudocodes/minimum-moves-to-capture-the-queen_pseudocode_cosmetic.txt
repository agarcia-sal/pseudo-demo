CLASS Solution  
    FUNCTION minMovesToCaptureTheQueen(a b c d e f)  
        FUNCTION EvaluateDiagonalMove(x1 y1 x2 y2 x3 y3)  
            VARIABLE delta1, delta2, delta3, delta4  
            SET delta1 = x2 + y2  
            SET delta2 = x3 + y3  
            SET delta3 = x1 + y1  
            SET delta4 = x2 - y2  
            RETURN FALSE  
        END FUNCTION

        VARIABLE mN, pQ, rS  
        ASSIGN mN = 0  
        ASSIGN pQ = 0  
        ASSIGN rS = 0  

        IF (a = e) THEN  
            VARIABLE pU  
            ASSIGN pU = (c <> a)  
            IF pU THEN  
                RETURN 1  
            END IF  
            IF ((b < d AND d < f) OR (f < d AND d < b)) THEN  
                RETURN 2  
            END IF  
            RETURN 1  
        END IF  

        IF (b = f) THEN  
            VARIABLE sV  
            ASSIGN sV = (d <> b)  
            IF sV THEN  
                RETURN 1  
            END IF  
            IF ((a < c AND c < e) OR (e < c AND c < a)) THEN  
                RETURN 2  
            END IF  
            RETURN 1  
        END IF  

        VARIABLE sumCD, sumEF, sumAB  
        ASSIGN sumCD = c + d  
        ASSIGN sumEF = e + f  
        ASSIGN sumAB = a + b  

        IF (sumCD = sumEF) THEN  
            VARIABLE notEqualAB_CD  
            ASSIGN notEqualAB_CD = NOT (sumAB = sumCD)  
            IF notEqualAB_CD THEN  
                RETURN 1  
            END IF  
            IF (((c < a) AND (a < e) AND (d < b) AND (b < f)) OR ((e < a) AND (a < c) AND (f < b) AND (b < d))) THEN  
                RETURN 2  
            END IF  
            RETURN 1  
        END IF  

        VARIABLE diffCD, diffEF, diffAB  
        ASSIGN diffCD = c - d  
        ASSIGN diffEF = e - f  
        ASSIGN diffAB = a - b  

        IF diffCD = diffEF THEN  
            IF NOT (diffAB = diffCD) THEN  
                RETURN 1  
            END IF  
            IF (((c < a) AND (a < e) AND (d > b) AND (b > f)) OR ((e < a) AND (a < c) AND (f > b) AND (b > d))) THEN  
                RETURN 2  
            END IF  
            RETURN 1  
        END IF  
        RETURN 2  
    END FUNCTION  
END CLASS
