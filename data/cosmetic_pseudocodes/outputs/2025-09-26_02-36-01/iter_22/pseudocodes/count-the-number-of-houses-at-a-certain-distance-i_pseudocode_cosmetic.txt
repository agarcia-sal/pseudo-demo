CLASS Solution  
    FUNCTION countOfPairs(n parameter, x parameter, y parameter) RETURNS list of integer  
        SET __Alpha TO a list of zeros repeated n times  

        FUNCTION bfs(start parameter)  
            SET __Beta TO a list of n PLUS one False booleans  
            SET __Gamma TO a list of n PLUS one zeros  
            SET __Delta TO a deque containing only start parameter  

            REPEAT  
                IF __Delta is empty THEN  
                    BREAK  
                END IF  
                SET __Epsilon TO the first element popped from __Delta  

                FOR each __Zeta IN [__Epsilon MINUS one, __Epsilon PLUS one]  
                    IF (__Zeta >= 1) AND (__Zeta <= n) AND (NOT __Beta[__Zeta]) THEN  
                        SET __Beta[__Zeta] TO True  
                        SET __Gamma[__Zeta] TO __Gamma[__Epsilon] PLUS 1  
                        APPEND __Zeta TO __Delta  
                    END IF  
                END FOR  

                IF (__Epsilon = x) AND (NOT __Beta[y]) THEN  
                    SET __Beta[y] TO True  
                    SET __Gamma[y] TO __Gamma[__Epsilon] PLUS 1  
                    APPEND y TO __Delta  
                END IF  

                IF (__Epsilon = y) AND (NOT __Beta[x]) THEN  
                    SET __Beta[x] TO True  
                    SET __Gamma[x] TO __Gamma[__Epsilon] PLUS 1  
                    APPEND x TO __Delta  
                END IF  
            UNTIL False  

            FOR __Eta FROM 1 TO n DO  
                IF __Gamma[__Eta] > 0 THEN  
                    SET __Alpha[__Gamma[__Eta] MINUS 1] TO __Alpha[__Gamma[__Eta] MINUS 1] PLUS 1  
                END IF  
            END FOR  
        END FUNCTION  

        FOR __Theta IN range(1, n PLUS 1) DO  
            CALL bfs WITH __Theta  
        END FOR  

        RETURN __Alpha  
    END FUNCTION  
END CLASS
