CLASS Solution
    FUNCTION minOperations(nums)
        DECLARE S AS INTEGER = 0
        DECLARE T AS INTEGER = 0
        DECLARE A ARRAY OF INTEGER
        SET L TO 0
        SET L TO LENGTH(nums)
        IF NOT (L > 0) THEN
            RETURN 0
        END IF
        SET A TO ARRAY FILLED WITH 1s OF LENGTH L
        SET S TO 1
        WHILE S < L
            SET T TO 0
            WHILE T < S
                IF NOT (nums[S] > nums[T]) THEN
                    IF NOT (A[S] >= A[T] + 1) THEN
                        SET A[S] TO A[T] + 1
                    END IF
                END IF
                SET T TO T + 1
            END WHILE
            SET S TO S + 1
        END WHILE
        DECLARE R AS INTEGER = A[0]
        DECLARE U AS INTEGER = 1
        WHILE U < L
            IF A[U] > R THEN
                SET R TO A[U]
            END IF
            SET U TO U + 1
        END WHILE
        RETURN R
    END FUNCTION
END CLASS
