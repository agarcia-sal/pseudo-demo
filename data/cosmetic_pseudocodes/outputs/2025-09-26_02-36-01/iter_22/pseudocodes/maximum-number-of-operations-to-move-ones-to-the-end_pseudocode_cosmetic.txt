CLASS Solution
    FUNCTION maxOperations(s)
        SET x0 TO 0
        SET y7 TO 0
        SET z3 TO 0
        WHILE z3 < LENGTH(s)
            SET a2 TO s[z3]
            IF NOT (a2 != '1')
                SET y7 TO y7 + 1
            ELSE
                IF (z3 > 0) AND ((s[z3 - 1]) = '1')
                    SET x0 TO x0 + y7
                END IF
            END IF
            SET z3 TO z3 + 1
        END WHILE
        RETURN x0
    END FUNCTION
END CLASS
