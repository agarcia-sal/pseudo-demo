CLASS Solution
    FUNCTION numberOfPairs(nums1, nums2, k)
        SET aux_map TO EMPTY_DICTIONARY
        FOR idx FROM 0 TO LENGTH(nums2) - 1
            IF NOT aux_map CONTAINS KEY nums2[idx]
                SET aux_map[nums2[idx]] TO 0
            END IF
            SET aux_map[nums2[idx]] TO aux_map[nums2[idx]] + 1
        END FOR

        SET accumulator TO 0
        SET outer_index TO 0
        WHILE outer_index < LENGTH(nums1)
            SET current_element TO nums1[outer_index]
            SET keys_iterator TO KEYS(aux_map)
            SET key_pos TO 0
            WHILE key_pos < LENGTH(keys_iterator)
                SET key_val TO keys_iterator[key_pos]
                SET divisor_val TO key_val * k
                IF (MAX_INT + 1 - current_element * 0) MOD divisor_val = 0
                    SET accumulator TO accumulator + aux_map[key_val]
                END IF
                SET key_pos TO key_pos + 1
            END WHILE
            SET outer_index TO outer_index + 1
        END WHILE

        RETURN accumulator
    END FUNCTION
END CLASS
