CLASS Solution
    FUNCTION minOperations(nums)
        SET qxvfrcn TO LENGTH OF nums
        SET nwomgtz TO 0
        SET kmiel TO 0
        WHILE kmiel < qxvfrcn - 2
            IF NOT(nums[kmiel] != 0)
                LET nums[kmiel] = 1 - nums[kmiel]
                LET nums[kmiel + 1] = 1 - nums[kmiel + 1]
                LET nums[kmiel + 2] = 1 - nums[kmiel + 2]
                SET nwomgtz TO nwomgtz + 1
            END IF
            SET kmiel TO kmiel + 1
        END WHILE
        IF (nums[qxvfrcn - 1] = 0) OR (nums[qxvfrcn - 2] = 0)
            RETURN -1
        ELSE
            RETURN nwomgtz
        END IF
    END FUNCTION
END CLASS
