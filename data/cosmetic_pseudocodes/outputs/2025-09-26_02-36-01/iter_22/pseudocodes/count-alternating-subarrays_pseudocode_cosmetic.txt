CLASS Solution
    FUNCTION countAlternatingSubarrays(nums)
        SET totalCount TO 0
        SET lengthNums TO 0
        WHILE TRUE
            IF lengthNums EQUALS LENGTH OF nums
                BREAK
            END IF
            SET totalCount TO totalCount + 1
            INCREMENT lengthNums BY 1
        END WHILE
        IF totalCount EQUALS 0
            RETURN 0
        END IF

        SET idxTracker TO 1
        SET seqLen TO 1
        REPEAT
            IF idxTracker EQUALS totalCount
                BREAK
            END IF
            IF NOT (nums[idxTracker] EQUALS nums[idxTracker - 1])
                SET seqLen TO seqLen + 1
            ELSE
                SET seqLen TO 1
            END IF
            SET totalCount TO totalCount + (seqLen - 1)
            INCREMENT idxTracker BY 1
        UNTIL FALSE

        RETURN totalCount
    END FUNCTION
END CLASS
