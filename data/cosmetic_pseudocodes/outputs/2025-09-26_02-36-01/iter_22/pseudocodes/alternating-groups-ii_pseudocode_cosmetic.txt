CLASS Solution  
    FUNCTION numberOfAlternatingGroups(colors k)  
        SET p TO 0  
        SET q TO LENGTH(colors)  
        IF NOT (k >= 3) THEN  
            RETURN 0  
        END IF  

        SET r TO CONCAT(colors, SLICE(colors, 0, k - 1))  

        SET s TO 0  
        SET t TO 0  
        WHILE t < q DO  
            SET u TO true  
            SET v TO 1  
            WHILE (v < k) AND u DO  
                IF r[t + v] = r[t + v - 1] THEN  
                    SET u TO false  
                END IF  
                SET v TO v + 1  
            END WHILE  
            IF u THEN  
                SET s TO s + 1  
            END IF  
            SET t TO t + 1  
        END WHILE  

        RETURN s  
    END FUNCTION  
END CLASS
