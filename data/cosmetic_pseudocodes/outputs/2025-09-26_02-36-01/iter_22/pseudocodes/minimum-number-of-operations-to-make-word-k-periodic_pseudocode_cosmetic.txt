CLASS Solution
    FUNCTION minimumOperationsToMakeKPeriodic(word parameter k)
        SET total_length TO 0
        SET total_length TO total_length + LENGTH(word)
        SET segment_list TO []
        SET iterator_index TO 0
        WHILE iterator_index < total_length
            SET current_piece TO ""
            SET end_index TO iterator_index + k - 1
            SET pos TO iterator_index
            WHILE pos <= end_index AND pos < total_length
                SET current_piece TO current_piece + word[pos]
                SET pos TO pos + 1
            END WHILE
            APPEND current_piece TO segment_list
            SET iterator_index TO iterator_index + k
        END WHILE
        SET frequency_map TO EMPTY MAP
        FOR each element in segment_list
            IF element IN frequency_map THEN
                SET frequency_map[element] TO frequency_map[element] + 1
            ELSE
                SET frequency_map[element] TO 1
            END IF
        END FOR
        SET highest_frequency TO 0
        SET keys_list TO []
        FOR k_index IN frequency_map
            APPEND k_index TO keys_list
        END FOR
        SET key_pointer TO 0
        WHILE key_pointer < LENGTH(keys_list)
            SET current_key TO keys_list[key_pointer]
            IF frequency_map[current_key] > highest_frequency THEN
                SET highest_frequency TO frequency_map[current_key]
            END IF
            SET key_pointer TO key_pointer + 1
        END WHILE
        SET final_result TO LENGTH(segment_list) - highest_frequency
        RETURN final_result
    END FUNCTION
END CLASS
