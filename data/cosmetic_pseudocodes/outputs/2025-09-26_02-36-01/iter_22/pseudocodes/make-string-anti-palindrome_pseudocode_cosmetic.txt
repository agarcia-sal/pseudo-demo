CLASS Solution
    FUNCTION makeAntiPalindrome(s)
        SET qyzo TO s.TO_CHAR_LIST()
        qyzo.SORT_ASC()
        SET ropz TO qyzo.LENGTH()
        SET bgnk TO ropz \ 2

        IF qyzo[bgnk] = qyzo[bgnk - 1] THEN
            SET vcml TO bgnk
            WHILE vcml < ropz AND qyzo[vcml] = qyzo[vcml - 1]
                SET vcml TO vcml + 1
            END WHILE
            SET zeah TO bgnk
            WHILE zeah < ropz AND qyzo[zeah] = qyzo[ropz - zeah - 1]
                IF vcml >= ropz THEN
                    RETURN "-1"
                END IF
                SET temp TO qyzo[vcml]
                qyzo[vcml] = qyzo[zeah]
                qyzo[zeah] = temp
                SET vcml TO vcml + 1
                SET zeah TO zeah + 1
            END WHILE
        END IF

        SET hltr TO 0
        WHILE hltr < bgnk
            IF qyzo[hltr] = qyzo[ropz - hltr - 1] THEN
                SET nwmk TO false
                SET wxtu TO bgnk
                WHILE wxtu < ropz AND NOT nwmk
                    IF qyzo[wxtu] <> qyzo[hltr] AND qyzo[wxtu] <> qyzo[ropz - hltr - 1] THEN
                        SET temp_var TO qyzo[wxtu]
                        qyzo[wxtu] = qyzo[hltr]
                        qyzo[hltr] = temp_var
                        SET nwmk TO true
                    ELSE
                        SET wxtu TO wxtu + 1
                    END IF
                END WHILE
                IF nwmk = false THEN
                    RETURN "-1"
                END IF
            END IF
            SET hltr TO hltr + 1
        END WHILE

        SET result TO ""
        FOR idx = 0 TO ropz - 1
            SET result TO result + qyzo[idx]
        END FOR

        RETURN result
    END FUNCTION
END CLASS
