CLASS Solution  
    FUNCTION minimumCost(m n horizontalCut verticalCut)  
        DEFINE FUNCTION descSort(arr)  
            SET p TO LENGTH OF arr  
            SET q TO 1  
            REPEAT  
                SET q TO q PLUS 1  
                SET r TO 1  
                WHILE r LESS THAN p  
                    IF element at position r - 1 OF arr LESS THAN element at position r OF arr  
                        SET tempSwap TO element at position r - 1 OF arr  
                        SET element at position r - 1 OF arr TO element at position r OF arr  
                        SET element at position r OF arr TO tempSwap  
                    END IF  
                    INCREMENT r BY 1  
                END WHILE  
                DECREMENT p BY 1  
            UNTIL q GREATER THAN LENGTH OF arr  
            RETURN arr  
        END FUNCTION  
        
        SET sortedHor TO descSort(horizontalCut)  
        SET sortedVer TO descSort(verticalCut)  
        
        SET totalSum TO 0  
        SET idxH TO 0  
        SET idxV TO 0  
        SET sectionH TO 1  
        SET sectionV TO 1  
        
        LOOP_LABEL:  
        IF NOT (idxH LESS THAN LENGTH OF sortedHor OR idxV LESS THAN LENGTH OF sortedVer) THEN GOTO END_LOOP  
        
        IF idxV EQUALS LENGTH OF sortedVer THEN  
            SET segCost TO element at idxH OF sortedHor MULTIPLIED BY sectionV  
            SET totalSum TO totalSum PLUS segCost  
            SET sectionH TO sectionH PLUS 1  
            SET idxH TO idxH PLUS 1  
            GOTO LOOP_LABEL  
        END IF  
        
        IF idxH LESS THAN LENGTH OF sortedHor THEN  
            IF element at idxH OF sortedHor GREATER THAN element at idxV OF sortedVer THEN  
                SET segCost TO element at idxH OF sortedHor MULTIPLIED BY sectionV  
                SET totalSum TO totalSum PLUS segCost  
                SET sectionH TO sectionH PLUS 1  
                SET idxH TO idxH PLUS 1  
                GOTO LOOP_LABEL  
            ELSE  
                SET segCost TO element at idxV OF sortedVer MULTIPLIED BY sectionH  
                SET totalSum TO totalSum PLUS segCost  
                SET sectionV TO sectionV PLUS 1  
                SET idxV TO idxV PLUS 1  
                GOTO LOOP_LABEL  
            END IF  
        ELSE  
            SET segCost TO element at idxV OF sortedVer MULTIPLIED BY sectionH  
            SET totalSum TO totalSum PLUS segCost  
            SET sectionV TO sectionV PLUS 1  
            SET idxV TO idxV PLUS 1  
            GOTO LOOP_LABEL  
        END IF  
        
        LABEL END_LOOP:  
        RETURN totalSum  
    END FUNCTION  
END CLASS
