CLASS Solution
    FUNCTION minLargest(nums1 nums1 AS list OF integers nums2 nums2 AS list OF integers) RETURNS integer
        FUNCTION nxt(x x AS integer y y AS integer) RETURNS integer
            IF (x AND ((1 XOR y) AND 1)) = 1 THEN
                RETURN x + 1
            ELSE
                RETURN (x + 2)
            END IF
        END FUNCTION

        SET p TO LENGTH(nums1)
        SET q TO LENGTH(nums2)
        SET arr TO a list WITH (p + 1) ELEMENTS EACH BEING a list WITH (q + 1) ZEROES

        SET u TO 1
        WHILE u <= p DO
            SET a TO nums1[u]
            SET arr[u][0] TO nxt(arr[u - 1][0], a)
            SET u TO u + 1
        END WHILE

        SET v TO 1
        WHILE v <= q DO
            SET b TO nums2[v]
            SET arr[0][v] TO nxt(arr[0][v - 1], b)
            SET v TO v + 1
        END WHILE

        SET w TO 1
        REPEAT
            IF w > p THEN
                BREAK
            END IF
            SET c TO nums1[w]
            SET z TO 1
            REPEAT
                IF z > q THEN
                    BREAK
                END IF
                SET d TO nums2[z]
                SET val1 TO nxt(arr[w - 1][z], c)
                SET val2 TO nxt(arr[w][z - 1], d)
                IF val1 < val2 THEN
                    SET arr[w][z] TO val1
                ELSE
                    SET arr[w][z] TO val2
                END IF
                SET z TO z + 1
            UNTIL FALSE
            SET w TO w + 1
        UNTIL FALSE

        RETURN arr[p][q]
    END FUNCTION
END CLASS
