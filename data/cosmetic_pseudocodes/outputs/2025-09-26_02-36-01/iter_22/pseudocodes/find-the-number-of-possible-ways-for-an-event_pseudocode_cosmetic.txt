CLASS Solution
    FUNCTION numberOfWays(n integer parameter, x integer parameter, y integer parameter) RETURNS integer
        SET CONST_MOD TO (10 * 10 * 10 * 10 * 10 * 10 * 10 * 10 * 10 * 10) + 7
        SET matrixF TO []
        SET idxA TO 0
        WHILE idxA <= n
            SET innerList TO []
            SET idxB TO 0
            WHILE idxB <= x
                APPEND 0 TO innerList
                SET idxB TO idxB + 1
            END WHILE
            APPEND innerList TO matrixF
            SET idxA TO idxA + 1
        END WHILE
        ASSIGN matrixF[0][0] = 1

        SET counterI TO 1
        WHILE counterI <= n
            SET counterJ TO 1
            WHILE counterJ <= x
                SET prevI TO counterI - 1
                SET val1 TO matrixF[prevI][counterJ] * counterJ
                SET val2 TO matrixF[prevI][counterJ - 1] * (x - (counterJ - 1))
                SET matrixF[counterI][counterJ] = (val1 + val2) MOD CONST_MOD
                SET counterJ TO counterJ + 1
            END WHILE
            SET counterI TO counterI + 1
        END WHILE

        SET totalAns TO 0
        SET powerVal TO 1
        SET counterK TO 1
        WHILE counterK <= x
            SET powerVal = (powerVal * y) MOD CONST_MOD
            SET addVal = (matrixF[n][counterK] * powerVal) MOD CONST_MOD
            SET totalAns = (totalAns + addVal) MOD CONST_MOD
            SET counterK TO counterK + 1
        END WHILE
        RETURN totalAns
    END FUNCTION
END CLASS
