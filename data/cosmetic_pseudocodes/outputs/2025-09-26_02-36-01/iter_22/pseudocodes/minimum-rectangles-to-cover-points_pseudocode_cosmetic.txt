CLASS Solution
    FUNCTION minRectanglesToCoverPoints(points, w)
        SET count_rectangles TO 0
        DEFINE index_cursor AS 0
        SORT points BY extracting the first coordinate IN ascending order
        WHILE index_cursor < LENGTH(points)
            SET base_position TO points[index_cursor][0]
            SET boundary_limit TO base_position
            SET extended_limit TO boundary_limit PLUS w
            INCREMENT count_rectangles BY 1
            REPEAT
                INCREMENT index_cursor BY 1
            UNTIL index_cursor >= LENGTH(points) OR points[index_cursor][0] >= extended_limit
        END WHILE
        RETURN count_rectangles
    END FUNCTION
END CLASS
