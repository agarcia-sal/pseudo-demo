CLASS Solution  
    FUNCTION maximumLength(nums k)  
        SET pQ TO LENGTH OF nums  
        SET qM TO a matrix filled with ones sized pQ rows and (k + 1) columns  
        SET cR TO 0  
        SET tU TO 0  
        WHILE tU < pQ  
            SET rL TO nums at index tU  
            SET mS TO 0  
            WHILE mS <= k  
                SET wD TO 0  
                WHILE wD < tU  
                    SET oV TO nums at index wD  
                    IF NOT (rL NOT EQUAL oV) THEN  
                        IF qM at position (tU, mS) < qM at position (wD, mS) + 1 THEN  
                            SET qM at position (tU, mS) TO qM at position (wD, mS) + 1  
                        END IF  
                    ELSE IF mS > 0 THEN  
                        IF qM at position (tU, mS) < qM at position (wD, mS - 1) + 1 THEN  
                            SET qM at position (tU, mS) TO qM at position (wD, mS - 1) + 1  
                        END IF  
                    END IF  
                    INCREMENT wD BY 1  
                END WHILE  
                INCREMENT mS BY 1  
            END WHILE  
            IF cR < qM at position (tU, k) THEN  
                SET cR TO qM at position (tU, k)  
            END IF  
            INCREMENT tU BY 1  
        END WHILE  
        RETURN cR  
    END FUNCTION  
END CLASS
