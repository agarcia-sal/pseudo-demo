CLASS Solution
    FUNCTION minAnagramLength(q)
        DECLARE b AS SET
        SET b TO EMPTY SET
        DECLARE e AS INTEGER
        SET e TO 0
        DECLARE p AS INTEGER
        SET p TO LENGTH(q)
        WHILE e < p DO
            IF NOT (q[e] IN b) THEN
                ADD q[e] TO b
            END IF
            SET e TO e + 1
        END WHILE
        RETURN LENGTH(b)
    END FUNCTION
END CLASS
