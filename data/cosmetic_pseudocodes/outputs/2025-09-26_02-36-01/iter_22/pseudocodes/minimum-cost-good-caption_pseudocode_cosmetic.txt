CLASS Solution
    FUNCTION minCostGoodCaption(caption)
        DECLARE q AS INTEGER
        SET q TO 0
        IF ( (LEN(caption) >= 3) = FALSE )
            RETURN ""
        END IF

        DECLARE arr AS LIST OF CHAR
        SET arr TO []
        FOR w IN caption
            APPEND w TO arr
        END FOR

        DECLARE x AS INTEGER
        SET x TO 0
        DECLARE y AS INTEGER
        DECLARE z AS INTEGER
        DECLARE c AS CHAR
        WHILE x < LEN(arr)
            SET y TO x
            WHILE (x < LEN(arr)) AND (arr[x] = arr[y])
                SET x TO x + 1
            END WHILE

            SET z TO x - y

            IF (z < 3)
                IF (y > 0) AND (arr[y - 1] = arr[y])
                    SET arr[y - 1] TO arr[y]
                    SET y TO y - 1
                    SET z TO z + 1
                END IF

                IF (x < LEN(arr)) AND (arr[x - 1] = arr[x])
                    SET arr[x] TO arr[x - 1]
                    SET x TO x + 1
                    SET z TO z + 1
                END IF

                IF (z < 3)
                    IF (y > 0)
                        SET c TO arr[y - 1]
                    ELSE
                        SET c TO arr[x]
                    END IF

                    IF c = 'a'
                        SET c TO 'b'
                    ELSE
                        IF c = 'z'
                            SET c TO 'y'
                        ELSE
                            SET c TO CHAR(ORD(c) + 1)
                        END IF
                    END IF

                    FOR index FROM y TO y + z - 1
                        SET arr[index] TO c
                    END FOR

                    SET x TO y + z
                END IF
            END IF
        END WHILE

        RETURN CONCATENATE(arr)
    END FUNCTION
END CLASS
