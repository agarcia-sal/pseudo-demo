CLASS Solution  
    FUNCTION maxPointsInsideSquare(points s)  
        SET totalCount TO LENGTH OF s  
        SET maximumResult TO 0  
        SET outerIndex TO 0  
        WHILE outerIndex < totalCount  
            SET coordX TO s[outerIndex][0]  
            SET coordY TO s[outerIndex][1]  
            SET limitSide TO (IF ABS(coordX) >= ABS(coordY) THEN ABS(coordX) ELSE ABS(coordY))  
            SET encounteredTags TO EMPTY DICTIONARY  
            SET isSquareValid TO TRUE  
            SET innerIndex TO 0  
            REPEAT  
                SET currentX TO s[innerIndex][0]  
                SET currentY TO s[innerIndex][1]  
                IF (ABS(currentX) <= limitSide) AND (ABS(currentY) <= limitSide) THEN  
                    SET currentTag TO s[innerIndex]  
                    IF currentTag IN encounteredTags THEN  
                        SET isSquareValid TO FALSE  
                        EXIT REPEAT  
                    ELSE  
                        SET encounteredTags[currentTag] TO TRUE  
                    END IF  
                END IF  
                INCREMENT innerIndex BY 1  
            UNTIL innerIndex = totalCount OR isSquareValid = FALSE  
            IF isSquareValid = TRUE THEN  
                IF LENGTH OF encounteredTags > maximumResult THEN  
                    SET maximumResult TO LENGTH OF encounteredTags  
                END IF  
            END IF  
            INCREMENT outerIndex BY 1  
        END WHILE  
        RETURN maximumResult  
    END FUNCTION  
END CLASS
