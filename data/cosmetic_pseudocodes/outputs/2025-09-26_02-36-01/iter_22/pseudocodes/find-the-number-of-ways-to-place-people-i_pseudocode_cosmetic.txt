CLASS Solution
    FUNCTION numberOfPairs(points)
        SET totalPairs TO 0
        SET limit TO LENGTH OF points
        SET outerIdx TO 0
        WHILE outerIdx < limit
            SET innerIdx TO 0
            REPEAT
                IF NOT (outerIdx = innerIdx) THEN
                    SET px TO points[outerIdx][0]
                    SET py TO points[outerIdx][1]
                    SET qx TO points[innerIdx][0]
                    SET qy TO points[innerIdx][1]
                    IF (px <= qx) AND (py >= qy) THEN
                        SET isValid TO 1
                        SET checker TO 0
                        WHILE checker < limit
                            IF (checker != outerIdx) AND (checker != innerIdx) THEN
                                SET rx TO points[checker][0]
                                SET ry TO points[checker][1]
                                IF ((px <= rx) AND (rx <= qx)) AND ((py >= ry) AND (ry >= qy)) THEN
                                    SET isValid TO 0
                                    SET checker TO limit
                                END IF
                            END IF
                            SET checker TO checker + 1
                        END WHILE
                        IF isValid = 1 THEN
                            SET totalPairs TO totalPairs + 1
                        END IF
                    END IF
                END IF
                SET innerIdx TO innerIdx + 1
            UNTIL innerIdx = limit
            SET outerIdx TO outerIdx + 1
        END WHILE
        RETURN totalPairs
    END FUNCTION
END CLASS
