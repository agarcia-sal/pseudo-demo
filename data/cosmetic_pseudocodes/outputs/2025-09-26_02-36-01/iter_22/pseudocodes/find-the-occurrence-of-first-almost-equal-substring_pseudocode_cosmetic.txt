CLASS Solution
    FUNCTION minStartingIndex WITH PARAMETERS s pattern
        FUNCTION is_almost_equal WITH PARAMETERS substring pattern
            SET epsilon TO 0
            SET alpha_beta TO 0
            WHILE alpha_beta LESS THAN LENGTH(substring) DO
                IF NOT substring[alpha_beta] = pattern[alpha_beta] THEN
                    SET epsilon TO epsilon + 1
                    IF NOT epsilon <= 1 THEN
                        RETURN False
                    END IF
                END IF
                SET alpha_beta TO alpha_beta + 1
            END WHILE
            RETURN True
        END FUNCTION

        SET delta TO LENGTH(pattern)
        SET omega TO 0
        WHILE omega LESS THAN (LENGTH(s) - delta + 1) DO
            SET gamma TO omega + delta - 1
            IF is_almost_equal WITH ARGUMENTS (SUBSTRING s FROM omega TO gamma) pattern THEN
                RETURN omega
            END IF
            SET omega TO omega + 1
        END WHILE
        RETURN -1
    END FUNCTION
END CLASS
