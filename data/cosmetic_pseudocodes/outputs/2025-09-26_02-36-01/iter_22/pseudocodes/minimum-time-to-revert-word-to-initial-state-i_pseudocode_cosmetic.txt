CLASS Solution
    FUNCTION minimumTimeToInitialState(word PARAMETER k)
        SET length TO LENGTH(word)
        SET counter TO 1
        REPEAT
            IF NOT (counter * k < length) OR SUBSTRING(word, counter * k, length) = SUBSTRING(word, 1, length - counter * k) THEN
                RETURN counter
            END IF
            SET counter TO counter + 1
        UNTIL FALSE
    END FUNCTION
END CLASS
