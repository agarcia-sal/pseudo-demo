CLASS Solution
    FUNCTION minimumDifference(nums k)
        FUNCTION bitwise_or_of_subarray(start end)
            SET accumulator TO 0
            SET index TO start
            WHILE index LESS THAN OR EQUAL TO end DO
                SET accumulator TO accumulator BITWISE_OR nums AT index
                SET index TO index PLUS 1
            END WHILE
            RETURN accumulator
        END FUNCTION

        SET length TO LENGTH OF nums
        SET bestDiff TO infinity
        SET outerIndex TO 0
        WHILE outerIndex LESS THAN length DO
            SET runningOr TO 0
            SET innerIndex TO outerIndex
            REPEAT
                SET runningOr TO runningOr BITWISE_OR nums AT innerIndex
                SET currentDiff TO k MINUS runningOr
                IF currentDiff LESS THAN 0 THEN
                    SET currentDiff TO 0 MINUS currentDiff
                END IF
                IF currentDiff LESS THAN bestDiff THEN
                    SET bestDiff TO currentDiff
                END IF
                IF bestDiff EQUALS 0 THEN
                    RETURN 0
                END IF
                SET innerIndex TO innerIndex PLUS 1
            UNTIL innerIndex GREATER THAN length MINUS 1
            SET outerIndex TO outerIndex PLUS 1
        END WHILE

        RETURN bestDiff
    END FUNCTION
END CLASS
