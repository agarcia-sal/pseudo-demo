CLASS Solution
	FUNCTION sumDigitDifferences(nums)
		FUNCTION digit_difference(num1, num2)
			SET result TO 0
			SET pos TO 0
Repeat_Diff_Check :
			IF pos < LENGTH(num1) THEN
				IF NOT (num1[pos] = num2[pos]) THEN
					SET result TO result + 1
				END IF
				SET pos TO pos + 1
				GOTO Repeat_Diff_Check
			END IF
			RETURN result
		END FUNCTION

		SET aggregate TO 0
		SET lengthVal TO LENGTH(nums)

		SET leftIndex TO 0
		WHILE leftIndex < lengthVal
			SET rightIndex TO leftIndex + 1
			REPEAT
				IF rightIndex >= lengthVal THEN
					EXIT REPEAT
				END IF
				ADD digit_difference(nums[leftIndex], nums[rightIndex]) TO aggregate
				SET rightIndex TO rightIndex + 1
			UNTIL FALSE
			SET leftIndex TO leftIndex + 1
		END WHILE

		RETURN aggregate
	END FUNCTION
END CLASS
