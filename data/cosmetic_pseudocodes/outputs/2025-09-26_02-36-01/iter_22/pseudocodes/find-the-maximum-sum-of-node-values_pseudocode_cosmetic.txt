CLASS Solution  
    FUNCTION maximumValueSum(nums k edges)  
        SET σ TO 0  
        SET ζ TO 0  
        SET μ TO +∞  
        SET ι TO 0  
        WHILE ι < LENGTH(nums)  
            SET ξ TO BITWISE_XOR(nums[ι] k)  
            SET η TO ξ - nums[ι]  
            IF η > 0 THEN  
                SET ζ TO ζ + 1  
            END IF  
            IF nums[ι] > ξ THEN  
                SET σ TO σ + nums[ι]  
            ELSE  
                SET σ TO σ + ξ  
            END IF  
            IF ABS(η) < μ THEN  
                SET μ TO ABS(η)  
            END IF  
            SET ι TO ι + 1  
        END WHILE  
        IF (ζ MOD 2) ≠ 0 THEN  
            SET σ TO σ - μ  
        END IF  
        RETURN σ  
    END FUNCTION  
END CLASS
