CLASS Solution
    FUNCTION beautifulIndices(s parameter a parameter b parameter k)
        SET tempList1 TO []
        SET pos1 TO 0
        REPEAT
            IF NOT (pos1 <= LENGTH(s) - LENGTH(a)) THEN
                BREAK
            END IF
            SET subStr1 TO ""
            SET idx1 TO 0
            WHILE idx1 < LENGTH(a)
                SET subStr1 TO subStr1 + s[pos1 + idx1]
                SET idx1 TO idx1 + 1
            END WHILE
            IF subStr1 = a THEN
                CALL tempList1.APPEND(pos1)
            END IF
            SET pos1 TO pos1 + 1
        UNTIL FALSE

        SET tempList2 TO LIST[]
        SET pos2 TO 0
        WHILE pos2 < LENGTH(s) - LENGTH(b) + 1
            SET accStr TO ""
            FOR step1 FROM 0 TO LENGTH(b) - 1
                SET accStr TO accStr + s[pos2 + step1]
            END FOR
            IF accStr = b THEN
                CALL tempList2.APPEND(pos2)
            END IF
            SET pos2 TO pos2 + 1
        END WHILE

        SET resultList TO []
        SET idxA TO 0
        SET idxB TO 0
        WHILE (idxA < LENGTH(tempList1)) AND (idxB < LENGTH(tempList2))
            SET diffVal TO tempList1[idxA] - tempList2[idxB]
            IF diffVal < 0 THEN
                SET diffVal TO -diffVal
            END IF
            IF diffVal <= k THEN
                CALL resultList.APPEND(tempList1[idxA])
                SET idxA TO idxA + 1
            ELSE
                IF tempList1[idxA] < tempList2[idxB] THEN
                    SET idxA TO idxA + 1
                ELSE
                    SET idxB TO idxB + 1
                END IF
            END IF
        END WHILE

        RETURN resultList
    END FUNCTION
END CLASS
