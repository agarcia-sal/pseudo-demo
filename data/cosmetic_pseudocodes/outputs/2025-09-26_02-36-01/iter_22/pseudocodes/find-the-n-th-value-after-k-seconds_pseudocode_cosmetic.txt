CLASS Solution
    FUNCTION valueAfterKSeconds(n k)
        CONST LIMIT = 10 * 10 * 10 * 10 * 10 * 10 * 10 * 10 * 10 * 10 + 7
        VAR buffer ← array filled with 1 repeated n times
        VAR outerCounter ← 0
        WHILE outerCounter < k
            VAR innerCounter ← 1
            WHILE innerCounter < n
                VAR tempVal ← buffer[innerCounter] + buffer[innerCounter - 1]
                VAR calcVal ← tempVal - (tempVal // LIMIT) * LIMIT
                buffer[innerCounter] ← calcVal
                innerCounter ← innerCounter + 1
            END WHILE
            outerCounter ← outerCounter + 1
        END WHILE
        RETURN buffer[n - 1]
    END FUNCTION
END CLASS
