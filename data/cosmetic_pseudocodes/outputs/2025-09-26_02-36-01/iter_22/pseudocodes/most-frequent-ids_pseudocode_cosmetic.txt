CLASS Solution
    FUNCTION mostFrequentIDs(nums PARAMETER, freq PARAMETER)
        SET dictCounter TO empty dictionary defaulting to integer zero
        SET heapStorage TO empty list
        SET outputList TO empty list
        SET indexX TO 0
        WHILE indexX < LENGTH(nums)
            SET currentNum TO nums[indexX]
            SET currentFreq TO freq[indexX]
            SET dictCounter[currentNum] TO dictCounter[currentNum] + currentFreq
            SET heapItem TO (NEGATE(dictCounter[currentNum]), currentNum)
            APPEND heapItem TO heapStorage
            WHILE LENGTH(heapStorage) > 0 AND NEGATE(heapStorage[0][0]) â‰  dictCounter[heapStorage[0][1]]
                REMOVE heapStorage[0]
            END WHILE
            IF LENGTH(heapStorage) > 0
                APPEND NEGATE(heapStorage[0][0]) TO outputList
            ELSE
                APPEND 0 TO outputList
            END IF
            SET indexX TO indexX + 1
        END WHILE
        RETURN outputList
    END FUNCTION
END CLASS
