CLASS Solution
    FUNCTION minOrAfterOperations(nums k)
        FUNCTION canAchieve(target_or k)
            SET state_tmp TO -1
            SET count_ops TO 0
            SET idx TO 0
            WHILE idx LESS THAN LENGTH(nums)
                SET el TO nums[idx]
                IF state_tmp EQUALS -1 THEN
                    SET state_tmp TO el
                ELSE
                    SET state_tmp TO state_tmp AND el
                END IF
                IF (state_tmp AND target_or) EQUALS 0 THEN
                    SET state_tmp TO -1
                ELSE
                    SET count_ops TO count_ops + 1
                    IF count_ops GREATER THAN k THEN
                        RETURN false
                    END IF
                END IF
                SET idx TO idx + 1
            END WHILE
            RETURN true
        END FUNCTION

        SET upper_bound TO (1 SHIFT_LEFT 30) - 1
        SET answer TO upper_bound
        SET bit_counter TO 0
        REPEAT
            IF bit_counter GREATER THAN 29 THEN
                BREAK
            END IF
            SET flag_mask TO 1 SHIFT_LEFT bit_counter
            IF (answer AND flag_mask) EQUALS 0 THEN
                SET bit_counter TO bit_counter + 1
                CONTINUE
            END IF
            SET complement TO NOT (answer XOR flag_mask)
            IF canAchieve(complement k) THEN
                SET answer TO answer AND complement
            END IF
            SET bit_counter TO bit_counter + 1
        UNTIL false
        RETURN answer
    END FUNCTION
END CLASS
