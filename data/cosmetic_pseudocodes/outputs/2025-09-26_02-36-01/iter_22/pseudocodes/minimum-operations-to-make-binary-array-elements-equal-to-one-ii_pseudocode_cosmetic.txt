CLASS Solution
    FUNCTION minOperations(nums)
        SET idx TO 0
        SET total_ops TO 0
        SET toggle_flag TO 0
        WHILE idx < LENGTH(nums)
            IF NOT (toggle_flag MOD 2) = 1 THEN
                SET temp_val TO nums[idx]
            ELSE
                SET temp_val TO 1 - nums[idx]
            END IF
            IF temp_val = 0 THEN
                SET total_ops TO total_ops + 1
                SET toggle_flag TO toggle_flag + 1
            END IF
            SET idx TO idx + 1
        END WHILE
        RETURN total_ops
    END FUNCTION
END CLASS
