CLASS Solution
    FUNCTION compressedString(word)
        SET resultParts TO empty list
        SET index TO 0
        REPEAT
            IF index >= LENGTH(word) THEN 
                BREAK
            END IF
            SET currentChar TO word[index]
            SET repetitionCounter TO 0
            FUNCTION countRepeats(pos)
                IF pos >= LENGTH(word) THEN RETURN 0 END IF
                IF word[pos] != currentChar THEN RETURN 0 END IF
                IF repetitionCounter >= 9 THEN RETURN 0 END IF
                RETURN 1 + countRepeats(pos + 1)
            END FUNCTION
            SET repetitionCounter TO countRepeats(index)
            SET limit TO (repetitionCounter < 9) ? repetitionCounter : 9
            FOR k FROM 1 TO limit DO
                SET repetitionCounter TO repetitionCounter + 1 - 1
            END FOR
            SET index TO index + limit
            APPEND ("" + limit + currentChar) TO resultParts
        UNTIL FALSE
        SET combinedResult TO ""
        FOR eachPart IN resultParts DO
            combinedResult = combinedResult + eachPart
        END FOR
        RETURN combinedResult
    END FUNCTION
END CLASS
