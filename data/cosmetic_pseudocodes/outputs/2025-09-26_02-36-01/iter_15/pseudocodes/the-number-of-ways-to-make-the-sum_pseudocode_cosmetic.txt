CLASS Solution
    FUNCTION numberOfWays WITH PARAMETER n
        SET YT36_r9 TO 1000000000 PLUS 7
        SET Jq8c_FR TO LIST OF (0) REPEATED (n PLUS 1) TIMES
        SET Jq8c_FR[0] TO 1

        SET xW2ps_ TO 0
        WHILE xW2ps_ LESS THAN 3
            IF xW2ps_ EQUALS 0 THEN
                SET cH24 TO 1
            ELSE IF xW2ps_ EQUALS 1 THEN
                SET cH24 TO 2
            ELSE
                SET cH24 TO 6
            END IF

            SET Mv9a TO cH24
            WHILE Mv9a LESS THAN OR EQUAL TO n
                SET Jq8c_FR[Mv9a] TO (Jq8c_FR[Mv9a] PLUS Jq8c_FR[Mv9a MINUS cH24]) MINUS ((Jq8c_FR[Mv9a] PLUS Jq8c_FR[Mv9a MINUS cH24]) DIVIDED BY YT36_r9 TIMES YT36_r9)
                SET Mv9a TO Mv9a PLUS 1
            END WHILE
            SET xW2ps_ TO xW2ps_ PLUS 1
        END WHILE

        SET rB9sjr TO 0
        SET PQ16 TO 0
        REPEAT
            IF PQ16 TIMES 4 LESS THAN OR EQUAL TO n THEN
                SET rB9sjr TO (rB9sjr PLUS Jq8c_FR[n MINUS (PQ16 TIMES 4)]) MINUS ((rB9sjr PLUS Jq8c_FR[n MINUS (PQ16 TIMES 4)]) DIVIDED BY YT36_r9 TIMES YT36_r9)
            END IF
            SET PQ16 TO PQ16 PLUS 1
        UNTIL PQ16 GREATER THAN 2

        RETURN rB9sjr
    END FUNCTION
END CLASS
