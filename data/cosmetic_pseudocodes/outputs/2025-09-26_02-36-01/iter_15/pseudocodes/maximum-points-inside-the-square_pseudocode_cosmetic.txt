CLASS Solution
    FUNCTION maxPointsInsideSquare(points s)
        SET total_entries TO LENGTH OF s
        SET maximum_marked TO 0
        SET outer_index TO 0
        WHILE outer_index < total_entries DO
            SET coord_x TO s[outer_index][0]
            SET coord_y TO s[outer_index][1]
            SET boundary_limit TO IF ABS(coord_x) >= ABS(coord_y) THEN ABS(coord_x) ELSE ABS(coord_y)
            SET encountered_tags TO EMPTY MAP
            SET is_square_valid TO 1 = 1
            SET inner_index TO 0
            REPEAT
                IF inner_index >= total_entries THEN
                    BREAK
                END IF
                SET curr_x TO s[inner_index][0]
                SET curr_y TO s[inner_index][1]
                IF ABS(curr_x) <= boundary_limit AND ABS(curr_y) <= boundary_limit THEN
                    SET current_tag TO s[inner_index]
                    IF current_tag IN encountered_tags THEN
                        SET is_square_valid TO NOT(is_square_valid)
                        BREAK
                    ELSE
                        SET encountered_tags[current_tag] TO TRUE
                    END IF
                END IF
                SET inner_index TO inner_index + 1
            UNTIL FALSE
            IF is_square_valid = TRUE THEN
                SET maximum_marked TO IF maximum_marked >= LENGTH OF encountered_tags THEN maximum_marked ELSE LENGTH OF encountered_tags
            END IF
            SET outer_index TO outer_index + 1
        END WHILE
        RETURN maximum_marked
    END FUNCTION
END CLASS
