CLASS Solution
    FUNCTION minimumOperationsToMakeKPeriodic(word parameter k)
        SET L TO 0
        SET M TO LENGTH(word)
        SET P TO EMPTY LIST
        WHILE L <= M - k
            SET Q TO ""
            SET R TO 0
            WHILE R < k AND L + R < M
                SET Q TO Q + word[L + R]
                SET R TO R + 1
            END WHILE
            APPEND Q TO P
            SET L TO L + k
        END WHILE

        FUNCTION T(S)
            SET U TO EMPTY MAP
            FOR V FROM 0 TO LENGTH(S) - 1
                IF S[V] IS NOT IN U THEN
                    SET U[S[V]] TO 1
                ELSE
                    SET U[S[V]] TO U[S[V]] + 1
                END IF
            END FOR
            RETURN U
        END FUNCTION

        SET W TO T(P)

        SET X TO 0
        FOR EACH Y IN KEYS(W)
            IF W[Y] > X THEN
                SET X TO W[Y]
            END IF
        END FOR

        SET Z TO LENGTH(P) - X
        RETURN Z
    END FUNCTION
END CLASS
