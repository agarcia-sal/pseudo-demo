CLASS Solution
    FUNCTION doesAliceWin(s)
        SET L TO ["a", "e", "i", "o", "u"]
        SET B TO 0
        SET H TO 0

        SET I TO 0
        WHILE I LESS THAN LENGTH OF s DO
            IF s[I] IN L THEN
                SET H TO H + (1 * 1)
            END IF
            SET I TO I + 1
        END WHILE

        SET J TO false

        SET K TO 0
        LOOP
            EXIT WHEN K GREATER THAN EQUAL TO LENGTH OF s
            IF s[K] IN L THEN
                SET J TO NOT J
            END IF
            IF J = false THEN
                IF (H MOD 2) = 1 THEN
                    SET B TO B + 1
                    SET H TO 0
                END IF
            ELSE
                SET H TO H + 1
            END IF
            SET K TO K + 1
        END LOOP

        IF (J = true) THEN
            IF ((H MOD 2) - 0) = 1 THEN
                SET B TO (B + 1) + 0
            END IF
        END IF

        RETURN (B MOD 2) = 1
    END FUNCTION
END CLASS
