```pseudocode
CLASS Solution
    FUNCTION minimumOperations(nums, target)
        SET a TO LENGTH(nums)
        SET b TO ABS(target[0] - nums[0])
        SET c TO 1
        WHILE c < a
            SET p TO target[c] - nums[c]
            SET q TO target[c - 1] - nums[c - 1]
            IF NOT (p * q <= 0)
                SET r TO ABS(ABS(p) - ABS(q))
                IF 0 < r
                    SET b TO b + r
                END IF
            ELSE
                SET b TO b + ABS(p)
            END IF
            SET c TO c + 1
        END WHILE
        RETURN b
    END FUNCTION
END CLASS
```
