```pseudocode
CLASS Solution
    FUNCTION findWinningPlayer(skills, k)
        SET p TO LENGTH(skills)
        SET c TO 0
        SET d TO ARRAY FROM 0 TO p - 1
        SET e TO d[0]
        REMOVE d[0] FROM d
        SET f TO 0
        WHILE NOT (f >= k OR LENGTH(d) = 0)
            SET g TO d[0]
            REMOVE d[0] FROM d
            IF (skills[e] - skills[g]) > 0
                SET f TO f + 1
                APPEND g TO d
            ELSE
                SET f TO 1
                APPEND e TO d
                SET e TO g
            END IF
        END WHILE
        RETURN e
    END FUNCTION
END CLASS
```
