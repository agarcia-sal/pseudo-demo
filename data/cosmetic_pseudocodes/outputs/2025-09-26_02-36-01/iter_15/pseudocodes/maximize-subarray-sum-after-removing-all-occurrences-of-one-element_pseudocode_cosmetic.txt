CLASS Solution
    FUNCTION maxSubarraySum(nums)
        FUNCTION kadane(arr)
            SET alpha TO arr[0]
            SET beta TO arr[0]
            SET indexGamma TO 1
            WHILE indexGamma < LENGTH(arr)
                SET delta TO arr[indexGamma]
                IF NOT (delta <= alpha + delta)
                    SET alpha TO delta
                ELSE
                    SET alpha TO alpha + delta
                END IF
                IF beta < alpha
                    SET beta TO alpha
                END IF
                SET indexGamma TO indexGamma + 1
            END WHILE
            RETURN beta
        END FUNCTION

        SET omega TO kadane(nums)
        SET sigmaSet TO EMPTY SET
        FOR each omegaElement IN nums
            SET sigmaSet[omegaElement] TO true
        END FOR

        SET uniqueIndex TO 0
        SET sigmaList TO EMPTY LIST
        FOR key IN sigmaSet KEYS
            APPEND key TO sigmaList
        END FOR

        SET iota TO 0
        WHILE iota < LENGTH(sigmaList)
            SET kappa TO sigmaList[iota]
            SET lambdaList TO EMPTY LIST
            SET zetaIndex TO 0
            WHILE zetaIndex < LENGTH(nums)
                SET mu TO nums[zetaIndex]
                IF NOT (mu = kappa)
                    APPEND mu TO lambdaList
                END IF
                SET zetaIndex TO zetaIndex + 1
            END WHILE
            IF LENGTH(lambdaList) > 0
                SET newSum TO kadane(lambdaList)
                IF omega < newSum
                    SET omega TO newSum
                END IF
            END IF
            SET iota TO iota + 1
        END WHILE

        RETURN omega
    END FUNCTION
END CLASS
