CLASS Solution
    FUNCTION largestPalindrome(n k)
        FUNCTION convertToInt(strval)
            SET result TO 0
            FOR idx FROM 0 TO LENGTH(strval) - 1
                SET result TO result * 10 + (ASCII_VALUE(strval[idx]) - ASCII_VALUE('0'))
            END FOR
            RETURN result
        END FUNCTION
        
        PROCEDURE decrementByOne(REF val)
            SET val TO val + (-1)
        END PROCEDURE
        
        FUNCTION strRepeat(ch count)
            SET acc TO ""
            SET counter TO 0
            WHILE counter LESS THAN count
                SET acc TO acc + ch
                SET counter TO counter + 1
            END WHILE
            RETURN acc
        END FUNCTION
        
        IF n = 1 THEN
            SET z TO 9
            WHILE z >= 1
                IF (z MOD k) = 0 THEN
                    RETURN "" + z
                END IF
                SET z TO z - 1
            END WHILE
            RETURN "0"
        END IF
        
        SET m TO (n + 1) / 2
        SET nineString TO strRepeat('9', m)
        SET halfValue TO convertToInt(nineString)
        
        FUNCTION strReverse(s)
            SET res TO ""
            SET pos TO LENGTH(s) - 1
            WHILE pos >= 0
                SET res TO res + s[pos]
                SET pos TO pos - 1
            END WHILE
            RETURN res
        END FUNCTION
        
        WHILE halfValue > 0
            SET halfStr TO "" + halfValue
            SET reversedStr TO strReverse(halfStr)
            IF (n MOD 2) = 0 THEN
                SET combined TO halfStr + reversedStr
            ELSE
                SET truncated TO SUBSTRING(halfStr, 0, LENGTH(halfStr) - 1)
                SET combined TO halfStr + strReverse(truncated)
            END IF
            
            SET combinedInt TO convertToInt(combined)
            IF (combinedInt MOD k) = 0 THEN
                RETURN combined
            END IF
            
            decrementByOne(halfValue)
        END WHILE
        
        RETURN "0"
    END FUNCTION
END CLASS
