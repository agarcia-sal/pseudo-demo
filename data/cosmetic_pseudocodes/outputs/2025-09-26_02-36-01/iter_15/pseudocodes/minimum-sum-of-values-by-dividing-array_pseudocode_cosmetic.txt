CLASS Solution
    FUNCTION minimumValueSum(nums, andValues)
        SET u7h TO LENGTH OF nums
        SET s9z TO LENGTH OF andValues

        FUNCTION dp(l2q, t8v)
            IF t8v = -1 THEN
                IF l2q = -1 THEN
                    RETURN 0
                ELSE
                    RETURN +∞
                END IF
            END IF

            IF l2q = -1 THEN
                RETURN +∞
            END IF

            SET l37 TO +∞
            SET wqh TO -1

            SET FUNCTION recurse(kh4)
                IF kh4 < -1 THEN
                    RETURN
                END IF

                IF wqh = -1 THEN
                    SET wqh TO nums[kh4]
                ELSE
                    SET wqh TO wqh BITWISE-AND nums[kh4]
                END IF

                IF wqh = andValues[t8v] THEN
                    SET zax TO dp(kh4 - 1, t8v - 1) + nums[l2q]
                    IF zax < l37 THEN
                        SET l37 TO zax
                    END IF
                END IF
                CALL recurse(kh4 - 1)
            END FUNCTION

            CALL recurse(l2q)
            RETURN l37
        END FUNCTION

        SET h1x TO dp(u7h - 1, s9z - 1)
        IF h1x ≠ +∞ THEN
            RETURN h1x
        ELSE
            RETURN -1
        END IF
    END FUNCTION
END CLASS
