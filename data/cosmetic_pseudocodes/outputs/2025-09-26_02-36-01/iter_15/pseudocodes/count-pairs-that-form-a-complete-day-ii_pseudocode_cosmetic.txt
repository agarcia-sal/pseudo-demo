```pseudocode
CLASS Solution
    FUNCTION countCompleteDayPairs WITH parameter hours
        INITIALIZE frequencyMap AS a map defaulting to 0 on misses
        INITIALIZE totalPairs TO 0
        INITIALIZE idx TO 0
        WHILE idx LESS THAN length of hours
            LET currHour BE hours at idx
            LET modValue BE ((currHour + 24) - 24 * ((currHour) DIV 24)) MOD 24
            LET neededComp BE (24 - modValue) MOD 24
            totalPairs = totalPairs + frequencyMap[neededComp]
            frequencyMap[modValue] = frequencyMap[modValue] + 1
            idx = idx + 1
        END WHILE
        RETURN totalPairs
    END FUNCTION
END CLASS
```
