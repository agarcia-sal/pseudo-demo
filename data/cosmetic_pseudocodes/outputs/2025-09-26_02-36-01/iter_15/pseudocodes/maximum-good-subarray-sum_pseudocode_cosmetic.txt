```pseudocode
CLASS Solution
    FUNCTION maximumSubarraySum(nums k)
        SET Z6U to map with no entries
        SET T1M to - (2 ^ 31) * (2 ^ 16)  // representation of negative infinity
        SET Q9X to 0

        SET O8V to 0
        WHILE O8V < LENGTH(nums)
            SET D5J to nums[O8V]
            SET H7R to D5J - k
            IF Z6U CONTAINS H7R
                SET G3N to Q9X - Z6U[H7R] + D5J
                IF G3N > T1M
                    SET T1M to G3N
                END IF
            END IF

            SET Y4S to D5J + k
            IF Z6U CONTAINS Y4S
                SET W0L to Q9X - Z6U[Y4S] + D5J
                IF W0L > T1M
                    SET T1M to W0L
                END IF
            END IF

            IF Z6U CONTAINS D5J
                IF Z6U[D5J] > Q9X
                    SET Z6U[D5J] to Q9X
                END IF
            ELSE
                SET Z6U[D5J] to Q9X
            END IF

            SET Q9X to Q9X + D5J
            SET O8V to O8V + 1
        END WHILE

        IF NOT (T1M = - (2 ^ 31) * (2 ^ 16))
            RETURN T1M
        ELSE
            RETURN 0
        END IF
    END FUNCTION
END CLASS
```
