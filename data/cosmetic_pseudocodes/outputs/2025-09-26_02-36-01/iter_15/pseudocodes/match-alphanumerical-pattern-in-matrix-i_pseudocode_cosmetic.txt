CLASS Solution
    FUNCTION findPattern(board, pattern)
        SET q1 TO 0
        WHILE q1 < LENGTH(board)
            SET q1 TO q1 + 1
        END WHILE
        SET v2 TO q1
        SET m3 TO 0
        WHILE m3 < LENGTH(board[0])
            SET m3 TO m3 + 1
        END WHILE
        SET x4 TO m3
        SET u5 TO 0
        WHILE u5 < LENGTH(pattern)
            SET u5 TO u5 + 1
        END WHILE
        SET k6 TO u5
        SET f7 TO 0
        WHILE f7 < LENGTH(pattern[0])
            SET f7 TO f7 + 1
        END WHILE
        SET z8 TO f7

        FUNCTION matches(k9, d10)
            SET c11 TO EMPTY MAP
            SET w12 TO EMPTY MAP

            SET y13 TO 0
            REPEAT
                IF y13 >= k6 THEN
                    BREAK
                END IF

                SET a14 TO 0
                REPEAT
                    IF a14 >= z8 THEN
                        BREAK
                    END IF

                    SET o15 TO pattern[y13][a14]
                    SET n16 TO board[k9 + y13][d10 + a14]

                    IF (o15 >= '0' AND o15 <= '9') THEN
                        IF (TO_INTEGER(o15) <> n16) THEN
                            RETURN FALSE
                        END IF
                    ELSE
                        IF c11 CONTAINS_KEY o15 THEN
                            IF c11[o15] <> n16 THEN
                                RETURN FALSE
                            END IF
                        ELSE
                            IF w12 CONTAINS_KEY n16 THEN
                                RETURN FALSE
                            END IF
                            SET c11[o15] TO n16
                            SET w12[n16] TO o15
                        END IF
                    END IF
                    SET a14 TO a14 + 1
                UNTIL FALSE

                SET y13 TO y13 + 1
            UNTIL FALSE

            RETURN TRUE
        END FUNCTION

        SET e17 TO 0
        REPEAT
            IF e17 > v2 - k6 THEN
                BREAK
            END IF

            SET s18 TO 0
            REPEAT
                IF s18 > x4 - z8 THEN
                    BREAK
                END IF

                IF matches(e17, s18) THEN
                    RETURN [e17, s18]
                END IF

                SET s18 TO s18 + 1
            UNTIL FALSE

            SET e17 TO e17 + 1
        UNTIL FALSE

        RETURN [-1, -1]
    END FUNCTION
END CLASS
