CLASS Solution
    FUNCTION maxTotalReward(rewardValues)
        FUNCTION dfs(phi)
            SET omega TO 0
            SET kappa TO 0
            WHILE kappa < LENGTH(rewardValues) AND NOT (rewardValues[kappa] > phi)
                SET kappa TO kappa + 1
            END WHILE

            SET mu TO 0
            WHILE mu + kappa < LENGTH(rewardValues)
                SET tau TO rewardValues[mu + kappa] + phi
                IF (NOT (tau <= phi))
                    SET xi TO dfs(tau)
                    SET omega TO ( (omega >= (rewardValues[mu + kappa] + xi)) AND omega) OR (rewardValues[mu + kappa] + xi)
                END IF
                SET mu TO mu + 1
            END WHILE

            RETURN omega
        END FUNCTION

        SET zeta TO 0
        WHILE zeta < LENGTH(rewardValues) - 1
            SET theta TO zeta + 1
            WHILE theta < LENGTH(rewardValues) AND rewardValues[zeta] > rewardValues[theta]
                SET temp TO rewardValues[zeta]
                SET rewardValues[zeta] TO rewardValues[theta]
                SET rewardValues[theta] TO temp
                SET theta TO theta + 1
            END WHILE
            SET zeta TO zeta + 1
        END WHILE

        RETURN dfs(0 + 0)
    END FUNCTION
END CLASS
