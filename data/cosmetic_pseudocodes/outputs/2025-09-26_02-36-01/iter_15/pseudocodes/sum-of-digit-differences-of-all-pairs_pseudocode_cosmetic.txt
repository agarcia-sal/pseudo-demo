CLASS Solution
    FUNCTION sumDigitDifferences(nums)
        FUNCTION digit_difference(num1, num2)
            SET c TO 0
            SET p TO 0
            WHILE p < LENGTH(num1)
                IF NOT (CHAR_AT(num1, p) = CHAR_AT(num2, p)) THEN
                    SET c TO c + 1
                END IF
                SET p TO p + 1
            END WHILE
            RETURN c
        END FUNCTION

        SET total TO 0
        SET length_of_nums TO LENGTH(nums)

        SET x TO 0
        WHILE x < length_of_nums
            SET y TO x + 1
            WHILE y < length_of_nums
                SET total TO total + digit_difference(nums[x], nums[y])
                SET y TO y + 1
            END WHILE
            SET x TO x + 1
        END WHILE

        RETURN total
    END FUNCTION
END CLASS
