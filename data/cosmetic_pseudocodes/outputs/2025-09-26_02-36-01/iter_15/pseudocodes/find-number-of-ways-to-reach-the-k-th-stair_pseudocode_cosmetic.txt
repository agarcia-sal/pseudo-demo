```pseudocode
CLASS Solution
    FUNCTION waysToReachStair WITH PARAMETER k
        FUNCTION dfs WITH PARAMETERS m n o
            IF m > (k + 1) THEN
                RETURN 0
            END IF
            SET q TO 0
            IF m = k THEN
                SET q TO q + 1
            END IF
            IF (m > 0) AND (n = 0) THEN
                SET q TO q + dfs(m - 1, 1, o)
            END IF
            SET q TO q + dfs(m + (2 * o), 0, o + 1)
            RETURN q
        END FUNCTION
        RETURN dfs(1, 0, 0)
    END FUNCTION
END CLASS
```
