CLASS Solution
    FUNCTION minEnd(n x)
        FUNCTION canConstruct(u A)
            SET b TO x
            SET c TO 1
            REPEAT
                IF b >= A THEN
                    EXIT REPEAT
                END IF
                SET b TO b + 1
                IF (b & x) == x THEN
                    SET c TO c + 1
                    IF c == n THEN
                        RETURN True
                    END IF
                END IF
            UNTIL False
            RETURN c == n
        END FUNCTION

        SET g TO x
        SET h TO 2 * 10 * 10 * 10 * 10 * 10 * 10 * 10 * 10
        WHILE g < h
            SET j TO (g + h) \ 2
            IF canConstruct(j) THEN
                SET h TO j
            ELSE
                SET g TO g + 1
            END IF
        END WHILE

        RETURN g
    END FUNCTION
END CLASS
