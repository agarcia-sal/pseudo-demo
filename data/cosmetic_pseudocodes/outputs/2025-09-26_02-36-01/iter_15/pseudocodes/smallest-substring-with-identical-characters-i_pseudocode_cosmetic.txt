CLASS Solution
    FUNCTION minLength(s parameter, numOps parameter)
        FUNCTION longest_uniform_substring(s parameter)
            SET a TO 0
            SET b TO 1
            SET c TO 1
            WHILE c < LENGTH(s)
                IF NOT (s[c] NOT_EQUALS s[c - 1])
                    SET b TO b + 1
                ELSE
                    IF a < b
                        SET a TO b
                    END IF
                    SET b TO 1
                END IF
                SET c TO c + 1
            END WHILE
            IF a > b
                RETURN a
            ELSE
                RETURN b
            END IF
        END FUNCTION

        SET d TO LENGTH(s)
        SET e TO 1 << LENGTH(s)
        SET f TO 0
        WHILE f < e
            SET g TO 0
            SET h TO f
            WHILE h > 0
                SET g TO g + (h & 1)
                SET h TO h >> 1
            END WHILE
            IF g > numOps
                SET f TO f + 1
                CONTINUE
            END IF

            SET i TO EMPTY LIST
            SET j TO 0
            WHILE j < LENGTH(s)
                APPEND s[j] TO i
                SET j TO j + 1
            END WHILE

            SET k TO 0
            WHILE k < LENGTH(s)
                IF (f & (1 << k)) != 0
                    IF i[k] == '0'
                        SET i[k] TO '1'
                    ELSE
                        SET i[k] TO '0'
                    END IF
                END IF
                SET k TO k + 1
            END WHILE

            SET l TO longest_uniform_substring(i)
            IF d > l
                SET d TO l
            END IF
            SET f TO f + 1
        END WHILE

        RETURN d
    END FUNCTION
END CLASS
