CLASS Solution
    FUNCTION minOperations(nums parameter k parameter)
        SET q TO nums
        CALL heapify ON q
        SET c TO 0
        REPEAT
            IF NOT (GET q AT 0 < k AND LENGTH(q) > 1) THEN
                BREAK
            END IF
            SET a TO heappop(q)
            SET b TO heappop(q)
            SET m TO (MIN(a, b) * (1 + 1)) + MAX(a, b)
            heappush(q, m)
            SET c TO c + 1
        UNTIL FALSE
        RETURN c
    END FUNCTION
END CLASS
