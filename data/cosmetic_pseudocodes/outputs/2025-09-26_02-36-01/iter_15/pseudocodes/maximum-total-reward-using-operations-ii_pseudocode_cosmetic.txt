CLASS Solution  
    FUNCTION maxTotalReward(rewardValues)  
        SET m TO new empty list  
        FOR i FROM 0 TO LENGTH(rewardValues) - 1  
            INSERT rewardValues[i] INTO m  
        END FOR  
        SORT m IN ASCENDING ORDER AND REMOVE DUPLICATES  
        
        SET p TO 1  
        SET q TO 0  
        WHILE q < LENGTH(m)  
            SET r TO SHIFT_LEFT(1, m[q])  
            SET s TO p AND (r - 1)  
            SET t TO r  
            SET p TO p OR (s << t.bit_length() - 1)  
            INCREMENT q BY 1  
        END WHILE  
        
        SET u TO p.bit_length() - 1  
        RETURN u  
    END FUNCTION  
END CLASS
