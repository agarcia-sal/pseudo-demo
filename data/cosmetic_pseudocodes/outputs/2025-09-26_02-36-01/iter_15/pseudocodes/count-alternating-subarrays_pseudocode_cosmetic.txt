CLASS Solution  
    FUNCTION countAlternatingSubarrays(nums)  
        SET a TO LENGTH OF nums  
        IF a EQUALS 0 THEN  
            RETURN 0  
        END IF  
        SET b TO a  
        SET c TO 1  
        SET d TO 1  
        WHILE d < a DO  
            IF (nums[d] =/= nums[d - 1]) THEN  
                SET c TO c + 1  
            ELSE  
                SET c TO 1  
            END IF  
            SET b TO b + (c - 1)  
            SET d TO d + 1  
        END WHILE  
        RETURN b  
    END FUNCTION  
END CLASS
