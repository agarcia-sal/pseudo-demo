CLASS Solution
    FUNCTION minOperations(nums)
        SET alpha TO LENGTH(nums)
        IF NOT (alpha > 0) THEN
            RETURN 0
        END IF

        SET omega TO ARRAY OF SIZE alpha FILLED WITH 1

        SET beta TO 1
        WHILE beta < alpha DO
            SET gamma TO 0
            WHILE gamma < beta DO
                IF nums[beta] <= nums[gamma] THEN
                    SET theta TO omega[gamma] + 1
                    IF omega[beta] < theta THEN
                        SET omega[beta] TO theta
                    END IF
                END IF
                SET gamma TO gamma + 1
            END WHILE
            SET beta TO beta + 1
        END WHILE

        SET delta TO omega[0]
        SET iota TO 1
        REPEAT
            IF omega[iota] > delta THEN
                SET delta TO omega[iota]
            END IF
            SET iota TO iota +1
        UNTIL iota >= alpha

        RETURN delta
    END FUNCTION
END CLASS
