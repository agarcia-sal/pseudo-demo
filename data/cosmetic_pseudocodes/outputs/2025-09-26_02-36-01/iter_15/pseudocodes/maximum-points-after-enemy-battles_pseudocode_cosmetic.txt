CLASS Solution
    FUNCTION maximumPoints(enemyEnergies currentEnergy)
        PROCEDURE customSort(array)
            SET swapped TO true
            WHILE swapped
                SET swapped TO false
                FOR idx FROM 0 TO LENGTH(array) - 2
                    IF array[idx+1] < array[idx]
                        SET temp TO array[idx]
                        SET array[idx] TO array[idx+1]
                        SET array[idx+1] TO temp
                        SET swapped TO true
                    END IF
                END FOR
            END WHILE
        END PROCEDURE

        CALL customSort(enemyEnergies)

        IF NOT (currentEnergy >= enemyEnergies[0])
            RETURN 0
        END IF

        SET accumulator TO 0
        SET pointer TO LENGTH(enemyEnergies) - 1

        WHILE pointer >= 0
            SET divisionResult TO FLOOR(currentEnergy / enemyEnergies[0])
            SET remainderValue TO currentEnergy - (divisionResult * enemyEnergies[0])
            SET accumulator TO accumulator + divisionResult
            SET currentEnergy TO remainderValue + enemyEnergies[pointer]
            DECREMENT pointer BY 1
        END WHILE

        RETURN accumulator

    END FUNCTION
END CLASS
