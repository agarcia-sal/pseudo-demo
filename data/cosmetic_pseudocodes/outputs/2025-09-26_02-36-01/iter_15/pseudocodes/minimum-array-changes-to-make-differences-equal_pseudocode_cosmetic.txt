CLASS Solution
    FUNCTION minChanges(nums k)
        SET l TO LENGTH OF nums
        SET r TO 0
        SET s TO LIST of zeros with length (k + 2)

        WHILE r < l / 2
            SET a TO nums[r]
            SET b TO nums[l - r - 1]

            IF a < b THEN
                SET c TO a
                SET a TO b
                SET b TO c
            END IF
            
            ADD 1 TO s[0]
            SUBTRACT 1 FROM s[b - a]
            ADD 1 TO s[b - a + 1]
            SUBTRACT 1 FROM s[MAX(b, k - a) + 1]
            ADD 2 TO s[MAX(b, k - a) + 2]

            SET r TO r + 1
        END WHILE

        SET t TO 0
        SET u TO s[0]
        SET v TO u

        WHILE t < (k + 1)
            SET u TO u + s[t + 1]
            IF u < v THEN
                SET v TO u
            END IF
            SET t TO t + 1
        END WHILE

        RETURN v
    END FUNCTION
END CLASS
