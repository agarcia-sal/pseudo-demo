```pseudocode
CLASS Solution
    FUNCTION mostFrequentIDs(nums PARAMETER, freq PARAMETER)
        SET xg64y TO a new dictionary defaulting to integer zero
        SET lvnz TO an empty list
        SET pexm TO an empty list
        SET idxA TO 0
        WHILE idxA < LENGTH(nums)
            SET kqro TO nums[idxA]
            SET wdpm TO freq[idxA]
            SET xg64y[kqro] TO xg64y[kqro] + wdpm
            SET pcopy TO - xg64y[kqro]
            APPEND (pcopy, kqro) TO lvnz
            LOOP
                IF lvnz IS EMPTY THEN
                    BREAK
                END IF
                SET uzpni TO lvnz[0]
                SET valCheck TO - uzpni[0]
                SET keyCheck TO uzpni[1]
                IF valCheck <> xg64y[keyCheck] THEN
                    REMOVE lvnz[0]
                ELSE
                    BREAK
                END IF
            END LOOP
            IF lvnz IS NOT EMPTY THEN
                SET wzjo TO - lvnz[0][0]
                APPEND wzjo TO pexm
            ELSE
                APPEND 0 TO pexm
            END IF
            SET idxA TO idxA + 1
        END WHILE
        RETURN pexm
    END FUNCTION
END CLASS
```
