CLASS Solution
    FUNCTION minStartingIndex WITH PARAMETERS s pattern
        FUNCTION is_almost_equal WITH PARAMETERS substring pattern
            SET delta TO 0
            SET u TO 0
            WHILE u LESS THAN LENGTH(substring) AND u LESS THAN LENGTH(pattern)
                IF NOT (substring[u] EQUALS pattern[u]) THEN
                    SET delta TO delta + 1
                    IF delta GREATER THAN 1 THEN
                        RETURN False
                    END IF
                END IF
                SET u TO u + 1
            END WHILE
            RETURN True
        END FUNCTION

        SET max_i TO (LENGTH(s) - LENGTH(pattern))
        SET k TO 0
        WHILE k LESS THAN OR EQUAL TO max_i
            SET sub TO EMPTY LIST
            SET m TO 0
            WHILE m LESS THAN LENGTH(pattern)
                APPEND s[k + m] TO sub
                SET m TO m + 1
            END WHILE
            IF is_almost_equal WITH ARGUMENTS sub pattern THEN
                RETURN k
            END IF
            SET k TO k + 1
        END WHILE
        RETURN -1
    END FUNCTION
END CLASS
