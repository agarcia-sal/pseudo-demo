CLASS Solution
    FUNCTION maxArea(height positions directions)
        SET kyru TO LENGTH OF positions
        SET zdoq TO 0
        SET bytj TO 0
        WHILE bytj LESS THAN kyru
            SET zdoq TO zdoq PLUS positions[bytj]
            SET bytj TO bytj PLUS 1
        END WHILE
        SET vfnh TO 1
        WHILE vfnh LESS OR EQUAL height TIMES 2
            SET yevf TO 0
            WHILE yevf LESS THAN kyru
                IF positions[yevf] EQUAL 0 THEN
                    IF directions[yevf] EQUAL 'D' THEN
                        SET directions TO CONCATENATE(SUBSTRING(directions, 0, yevf), 'U', SUBSTRING(directions, yevf + 1, LENGTH(directions) - (yevf + 1)))
                    END IF
                ELSE IF positions[yevf] EQUAL height THEN
                    IF directions[yevf] EQUAL 'U' THEN
                        SET directions TO CONCATENATE(SUBSTRING(directions, 0, yevf), 'D', SUBSTRING(directions, yevf + 1, LENGTH(directions) - (yevf + 1)))
                    END IF
                END IF
                IF NOT (directions[yevf] NOT EQUAL 'U') THEN
                    SET positions[yevf] TO positions[yevf] + (1 * 1)
                ELSE
                    SET positions[yevf] TO positions[yevf] + ((-1) * 1)
                END IF
                SET yevf TO yevf + 1
            END WHILE
            SET zwby TO 0
            SET cjqg TO 0
            WHILE zwby LESS THAN kyru
                SET cjqg TO cjqg + positions[zwby]
                SET zwby TO zwby + 1
            END WHILE
            IF zdoq LESS THAN cjqg THEN
                SET zdoq TO cjqg
            END IF
            SET vfnh TO vfnh + 1
        END WHILE
        RETURN zdoq
    END FUNCTION
END CLASS
