CLASS Solution
    FUNCTION countOfPairs(n parameter, x parameter, y parameter) RETURNS list of integer
        SET wv TO a list filled with zeros repeated n times
        
        FUNCTION bfs(start parameter)
            SET hh TO a Boolean array of size (n + 1) initialized with False
            SET dd TO an integer array of size (n + 1) initialized with 0
            SET qq TO a deque initialized with a single element: start

            SET hh[start] TO True

            WHILE NOT (qq is empty)
                SET vv TO remove element from front of qq
                
                FOR each uu IN list containing (vv - 1) and (vv + 1)
                    IF (uu >= 1) AND (uu <= n) AND (hh[uu] = False)
                        SET hh[uu] TO True
                        SET dd[uu] TO dd[vv] + 1
                        APPEND uu to end of qq
                    END IF
                END FOR

                IF (vv = x) AND (hh[y] = False)
                    SET hh[y] TO True
                    SET dd[y] TO dd[vv] + 1
                    APPEND y to end of qq
                END IF

                IF (vv = y) AND (hh[x] = False)
                    SET hh[x] TO True
                    SET dd[x] TO dd[vv] + 1
                    APPEND x to end of qq
                END IF
            END WHILE

            FOR jj FROM 1 TO n
                IF dd[jj] > 0
                    SET wv[dd[jj] - 1] TO wv[dd[jj] - 1] + 1
                END IF
            END FOR
        END FUNCTION

        SET zz TO 1
        WHILE zz <= n
            CALL bfs(zz)
            SET zz TO zz + 1
        END WHILE

        RETURN wv
    END FUNCTION
END CLASS
