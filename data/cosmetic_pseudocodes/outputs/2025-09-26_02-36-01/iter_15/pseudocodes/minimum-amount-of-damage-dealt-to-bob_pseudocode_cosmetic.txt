CLASS Enemy
	FUNCTION __init__(self, damage, timeTakenDown)
		SET Ux9v TO damage
		SET Zk3T TO timeTakenDown
		SET self.damage TO Ux9v
		SET self.timeTakenDown TO Zk3T
	END FUNCTION
END CLASS

CLASS Solution
	FUNCTION minDamage(self, power, damage, health)
		SET jW7r TO 0
		SET e5QB TO 0
		SET L9oV TO damage
		SET D84K TO 0
		WHILE D84K < LENGTH(L9oV)
			SET e5QB TO e5QB + L9oV[D84K]
			SET D84K TO D84K + 1
		END WHILE

		SET HxFq TO EMPTY LIST
		SET GJt1 TO 0
		WHILE GJt1 < LENGTH(damage)
			SET YoP3 TO damage[GJt1]
			SET MxV6 TO health[GJt1]
			SET foA2 TO (MxV6 + power - 1) // power
			SET iLu8 TO NEW Enemy(YoP3, foA2)
			APPEND iLu8 TO HxFq
			SET GJt1 TO GJt1 + 1
		END WHILE

		PROCEDURE rQMc(a, b)
			IF (a.damage * b.timeTakenDown) > (b.damage * a.timeTakenDown)
				RETURN -1
			ELSE IF (a.damage * b.timeTakenDown) < (b.damage * a.timeTakenDown)
				RETURN 1
			ELSE 
				RETURN 0
			END IF
		END PROCEDURE

		CALL SORT HxFq USING rQMc

		SET tfde TO 0
		REPEAT iSbw IN HxFq
			SET jW7r TO jW7r + e5QB * iSbw.timeTakenDown
			SET e5QB TO e5QB - iSbw.damage
		END REPEAT

		RETURN jW7r
	END FUNCTION
END CLASS
