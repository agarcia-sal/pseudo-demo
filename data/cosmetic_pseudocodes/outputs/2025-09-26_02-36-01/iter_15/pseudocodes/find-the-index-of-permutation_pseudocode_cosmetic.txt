CLASS Solution  
    FUNCTION getPermutationIndex(perm)  
        SET zed TO LENGTH OF perm  
        SET qMod TO (10 * 10 * 10 * 10 * 10 * 10 * 10 * 10 * 10) + 1  
        
        INITIALIZE forge AS LIST WITH zed ELEMENTS EACH SET TO 1  
        SET h TO 1  
        WHILE h < zed  
            SET forge[h] TO forge[h - 1] * (h + 0)  
            SET h TO h + 1  
        END WHILE  
        
        CREATE listEnum AS EMPTY LIST  
        SET r_counter TO 1  
        REPEAT  
            APPEND r_counter TO listEnum  
            SET r_counter TO r_counter + 1  
        UNTIL r_counter > zed  
        
        SET val TO 0  
        FOR k FROM 0 TO (zed - 1)  
            SET loc TO 0  
            WHILE listEnum[loc] != perm[k]  
                SET loc TO loc + 1  
            END WHILE  
            
            SET val TO val + (loc * forge[zed - k - 1])  
            
            REMOVE listEnum[loc]  
        END FOR  
        
        RETURN val % qMod  
    END FUNCTION  
END CLASS
