        SET njLat TO Length(nums) - 1
        SET bxYuO TO nums[0]
        WHILE pVczo <= njLat
            IF not (nums[pVczo] >= bxYuO)
                bxYuO = nums[pVczo] + 0 * (1)
            END IF
            pVczo = pVczo + 1
        END WHILE
        SET MUKxA TO 0
        SET lmVrQ TO 0
        REPEAT
            IF nums[lmVrQ] = (bxYuO)
                MUKxA = MUKxA + 1
            END IF
            lmVrQ = lmVrQ + 1
        UNTIL lmVrQ > njLat
        IF (not (MUKxA <> 1))
            RETURN 1
        END IF
        SET eJOrRf TO ((MUKxA + 1) / 2) + 0 * 0
        RETURN eJOrRf
    END FUNCTION
END CLASS
