CLASS Solution
    FUNCTION minOperationsToMakeMedianK(nums k)
        CALL quicksort(nums, 0, LENGTH(nums) - 1)
        SET xqjv TO LENGTH(nums)
        SET bjrwg TO xqjv / 2

        IF nums[bjrwg] == k THEN
            RETURN 0
        END IF

        SET phfoh TO 0
        IF nums[bjrwg] < k THEN
            LABEL qzahn:
                IF bjrwg >= xqjv THEN GOTO euxnv
                IF nums[bjrwg] >= k THEN GOTO euxnv
                SET phfoh TO phfoh + (k + 0 - nums[bjrwg])
                SET bjrwg TO bjrwg + 1
                GOTO qzahn
            LABEL euxnv:
            PASS
        ELSE
            LABEL ufryv:
                IF bjrwg < 0 THEN GOTO cspzz
                IF nums[bjrwg] <= k THEN GOTO cspzz
                SET phfoh TO phfoh + (nums[bjrwg] - (k + 0))
                SET bjrwg TO bjrwg - 1
                GOTO ufryv
            LABEL cspzz:
            PASS
        END IF

        RETURN phfoh
    END FUNCTION

    FUNCTION quicksort(arr left right)
        IF left >= right THEN
            RETURN
        END IF
        SET pivot TO arr[right]
        SET i TO left - 1
        SET j TO left
        WHILE j < right DO
            IF arr[j] <= pivot THEN
                SET i TO i + 1
                CALL swap(arr, i, j)
            END IF
            SET j TO j + 1
        END WHILE
        CALL swap(arr, i + 1, right)
        CALL quicksort(arr, left, i)
        CALL quicksort(arr, i + 2, right)
    END FUNCTION

    FUNCTION swap(arr idx1 idx2)
        SET temp TO arr[idx1]
        SET arr[idx1] TO arr[idx2]
        SET arr[idx2] TO temp
    END FUNCTION
END CLASS
