CLASS Solution  
    FUNCTION minOperations(k)  
        SET alpha TO 1000000000  // represents infinity  
        SET beta TO 1  
        WHILE beta <= FLOOR(SQRT(k)) + 1 DO  
            SET gamma TO (k + beta - 1) \ beta  
            SET delta TO (beta - 1) + (gamma - 1)  
            IF delta < alpha THEN  
                SET alpha TO delta  
            END IF  
            SET beta TO beta + 1  
        END WHILE  
        RETURN alpha  
    END FUNCTION  
END CLASS
