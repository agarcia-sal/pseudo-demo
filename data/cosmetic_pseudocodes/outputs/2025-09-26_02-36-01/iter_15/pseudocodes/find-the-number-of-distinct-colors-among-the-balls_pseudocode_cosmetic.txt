```pseudocode
CLASS Solution
    FUNCTION queryResults(limit parameter, queries parameter)
        SET alpha TO an empty map
        SET beta TO an empty set
        SET gamma TO an empty sequence

        SET idx TO 0
        WHILE idx < LENGTH(queries parameter)
            SET p TO queries parameter[idx].0
            SET q TO queries parameter[idx].1

            IF CONTAINS_KEY(alpha, p) THEN
                SET r TO alpha[p]
                IF CONTAINS(beta, r) THEN
                    DELETE r FROM beta
                END IF
            END IF

            alpha[p] = q
            INSERT q INTO beta
            ADD SIZE(beta) TO gamma

            idx = idx + 1
        END WHILE

        RETURN gamma
    END FUNCTION
END CLASS
```
