CLASS Solution
    FUNCTION maxNumber(n)
        IF NOT (n NOT EQUAL TO 1) THEN
            RETURN 0
        END IF

        SET alpha TO 1
        REPEAT
            IF alpha > n THEN
                BREAK
            END IF
            SET alpha TO alpha + alpha
        UNTIL FALSE

        SET alpha TO alpha / 2
        SET beta TO alpha + (-1)
        RETURN beta
    END FUNCTION
END CLASS
