```pseudocode
CLASS Solution
    FUNCTION findMaximumNumber(k, x)
        FUNCTION count_set_bits(n, pos)
            SET s TO 0
            SET b TO 1
            FOR c FROM 1 TO pos
                SET b TO b * 2
            END FOR
            SET f TO FLOOR(n / b)
            SET s TO s + (FLOOR(f / 2) * b)
            IF (f MOD 2) = 1 THEN
                SET s TO s + (n MOD b) + 1
            END IF
            RETURN s
        END FUNCTION

        FUNCTION accumulated_price(n)
            SET total TO 0
            SET idx TO 1
            LOOP
                SET exp_term TO idx * x - 1
                SET pow_val TO 1
                FOR j FROM 1 TO exp_term
                    SET pow_val TO pow_val * 2
                END FOR
                IF pow_val > n THEN
                    EXIT LOOP
                END IF
                SET total TO total + count_set_bits(n, exp_term)
                SET idx TO idx + 1
            END LOOP
            RETURN total
        END FUNCTION

        SET low_val TO 1
        SET hi_val TO 1
        FOR p FROM 1 TO 60
            SET hi_val TO hi_val * 2
        END FOR
        WHILE low_val <= hi_val DO
            SET median TO low_val + FLOOR((hi_val - low_val) / 2)
            IF accumulated_price(median) <= k THEN
                SET low_val TO median + 1
            ELSE
                SET hi_val TO median - 1
            END IF
        END WHILE
        RETURN hi_val
    END FUNCTION
END CLASS
```
