CLASS Solution
    FUNCTION maxOperations(s)
        SET alpha TO 0
        SET beta TO 0
        SET gamma TO 0
        WHILE gamma LESS THAN LENGTH(s)
            SET delta TO s[gamma]
            IF NOT (delta NOT EQUAL TO '1') THEN
                SET beta TO beta + 1
            ELSE
                IF gamma GREATER THAN 0 THEN
                    IF NOT (s[gamma - 1] NOT EQUAL TO '1') THEN
                        SET alpha TO alpha + beta
                    END IF
                END IF
            END IF
            SET gamma TO gamma + 1
        END WHILE
        RETURN alpha
    END FUNCTION
END CLASS
