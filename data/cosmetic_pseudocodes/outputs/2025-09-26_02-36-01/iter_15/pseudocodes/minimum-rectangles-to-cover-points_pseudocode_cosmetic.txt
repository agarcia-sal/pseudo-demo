CLASS Solution
    FUNCTION minRectanglesToCoverPoints(points, w)
        DEFINE FUNCTION sortByX(arr)
            PROCEDURE swapElements(a, b)
                TEMP = arr[a]
                arr[a] = arr[b]
                arr[b] = TEMP
            END PROCEDURE

            FOR i FROM 0 TO LENGTH(arr) - 2
                FOR j FROM 0 TO LENGTH(arr) - 2 - i
                    IF arr[j][0] > arr[j + 1][0] THEN
                        CALL swapElements(j, j + 1)
                    END IF
                END FOR
            END FOR
        END FUNCTION

        CALL sortByX(points)

        rectangles = 0
        current_x = -1
        idx = 0

        WHILE idx < LENGTH(points) DO
            px = points[idx][0]
            py = points[idx][1]

            IF px > current_x THEN

        RETURN rectangles
    END FUNCTION
