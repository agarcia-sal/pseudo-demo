```pseudocode
CLASS Solution
    FUNCTION minLargest(nums1 nums1 AS list OF integers nums2 nums2 AS list OF integers) RETURNS integer
        FUNCTION nxt(x x AS integer y y AS integer) RETURNS integer
            IF ((x BITWISE_AND 1) XOR y) = 1 THEN
                RETURN (x + 1)
            ELSE
                RETURN (x + 2)
            END IF
        END FUNCTION

        SET u TO LENGTH(nums1)
        SET v TO LENGTH(nums2)
        SET q TO a list of lists OF integer WITH u + 1 ROWS AND v + 1 COLUMNS INITIALIZED TO 0

        SET a TO 1
        WHILE a <= u DO
            SET p TO nums1[a]
            SET q[a][0] TO nxt(q[a - 1][0], p)
            SET a TO a + 1
        END WHILE

        SET b TO 1
        WHILE b <= v DO
            SET r TO nums2[b]
            SET q[0][b] TO nxt(q[0][b - 1], r)
            SET b TO b + 1
        END WHILE

        SET c TO 1
        WHILE c <= u DO
            SET s TO nums1[c]
            SET d TO 1
            WHILE d <= v DO
                SET t TO nums2[d]
                SET val1 TO nxt(q[c - 1][d], s)
                SET val2 TO nxt(q[c][d - 1], t)
                IF val1 < val2 THEN
                    SET q[c][d] TO val1
                ELSE
                    SET q[c][d] TO val2
                END IF
                SET d TO d + 1
            END WHILE
            SET c TO c + 1
        END WHILE

        RETURN q[u][v]
    END FUNCTION
END CLASS
```
