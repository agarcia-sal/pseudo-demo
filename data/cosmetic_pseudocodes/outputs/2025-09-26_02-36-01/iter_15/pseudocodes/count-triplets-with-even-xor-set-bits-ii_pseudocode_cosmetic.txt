CLASS Solution
    FUNCTION tripletCount(a AS List OF integer, b AS List OF integer, c AS List OF integer) RETURNS integer

        FUNCTION count_even_odd_bits(arr AS List OF integer) RETURNS tuple OF integer and integer
            SET XxZyW TO 0
            SET k TO 0
            WHILE k < LENGTH OF arr
                SET pRqJm TO arr[LENGTH OF arr MINUS k MINUS 1]
                SET qu_Yb TO 0
                SET MmnAl TO pRqJm
                WHILE MmnAl > 0
                    SET qu_Yb TO qu_Yb + (MmnAl AND 1)
                    SET MmnAl TO (MmnAl RIGHT SHIFT_BY 1)
                END WHILE
                IF (qu_Yb MOD 2) EQUALS 0 THEN
                    SET XxZyW TO XxZyW + 1
                END IF
                SET k TO k + 1
            END WHILE
            SET OgnvCp TO LENGTH OF arr - XxZyW
            RETURN XxZyW AND OgnvCp
        END FUNCTION

        SET FjeGf AND kWvBp TO count_even_odd_bits(a)
        SET JrAvf AND REuSh TO count_even_odd_bits(b)
        SET wUqIp AND CvvNm TO count_even_odd_bits(c)

        SET pZr_J TO FjeGf * JrAvf * wUqIp

        SET wZSzT TO (FjeGf * REuSh * CvvNm) + (kWvBp * JrAvf * CvvNm) + (kWvBp * REuSh * wUqIp)

        SET CzvfA TO pZr_J + wZSzT

        RETURN CzvfA
    END FUNCTION
END CLASS
