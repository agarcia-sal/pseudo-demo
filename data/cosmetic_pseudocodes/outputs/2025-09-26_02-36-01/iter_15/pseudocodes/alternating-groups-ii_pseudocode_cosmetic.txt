CLASS Solution
    FUNCTION numberOfAlternatingGroups(colors k)
        IF NOT (k GREATER OR EQUAL THAN 3) THEN
            RETURN 0
        END IF

        LET a TO LENGTH OF colors
        LET b TO EMPTY LIST
        FOR c FROM 0 TO a - 1
            APPEND colors[c] TO b
        END FOR

        FOR d FROM 0 TO k - 1
            APPEND colors[d] TO b
        END FOR

        LET e TO 0

        LET f TO 0
        WHILE f < a
            LET g TO true
            LET h TO 1
            REPEAT
                IF b[f + h] EQUALS b[f + h - 1] THEN
                    SET g TO false
                    BREAK
                END IF
                SET h TO h + 1
            UNTIL h >= k

            IF g THEN
                SET e TO e + 1
            END IF

            SET f TO f + 1
        END WHILE

        RETURN e
    END FUNCTION
END CLASS
