```pseudocode
CLASS Solution
    FUNCTION numberOfRightTriangles(grid)
        SET alpha TO 0
        SET beta TO LENGTH(grid)
        SET gamma TO LENGTH(grid[0])
        SET delta TO 0
        
        PROCEDURE incrementIfNotEqual(x, y, ref count, arr)
            IF x ≠ y THEN
                count ← count + arr[x]
            END IF
        END PROCEDURE
        
        FUNCTION sumExceptIndexLine(line, excludeIdx)
            SET accumulator TO 0
            SET iterator TO 0
            WHILE iterator < LENGTH(line)
                CALL incrementIfNotEqual(iterator, excludeIdx, accumulator, line)
                iterator ← iterator + 1
            END WHILE
            RETURN accumulator
        END FUNCTION
        
        FUNCTION sumExceptIndexColumn(matrix, columnIdx, excludeRow)
            SET summation TO 0
            SET indexer TO 0
            REPEAT
                IF indexer = excludeRow THEN
                    indexer ← indexer + 1
                    CONTINUE
                END IF
                summation ← summation + matrix[indexer][columnIdx]
                indexer ← indexer + 1
            UNTIL indexer ≥ LENGTH(matrix)
            RETURN summation
        END FUNCTION
        
        WHILE alpha < beta
            SET epsilon TO 0
            WHILE epsilon < gamma
                IF NOT (grid[alpha][epsilon] = 0) THEN
                    SET zeta TO sumExceptIndexLine(grid[alpha], epsilon)
                    SET eta TO sumExceptIndexColumn(grid, epsilon, alpha)
                    delta ← delta + zeta * eta
                END IF
                epsilon ← epsilon + 1
            END WHILE
            alpha ← alpha + 1
        END WHILE
        
        RETURN delta
    END FUNCTION
END CLASS
```
