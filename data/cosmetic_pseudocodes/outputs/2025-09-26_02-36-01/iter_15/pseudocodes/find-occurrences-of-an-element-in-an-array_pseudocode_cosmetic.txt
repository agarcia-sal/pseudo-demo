```pseudocode
CLASS Solution
    FUNCTION occurrencesOfElement(nums, queries, x)
        SET pqr TO 1
        SET stu TO LENGTH(nums)
        SET vwx TO 0
        SET yza TO []
        WHILE vwx < stu
            SET bcd TO nums[vwx]
            IF NOT (bcd <> x)
                yza = yza + [vwx]
            END IF
            SET vwx TO vwx + pqr
        END WHILE

        SET efgh TO []
        SET ijkl TO 0
        SET mnop TO LENGTH(queries)
        REPEAT
            IF ijkl >= mnop THEN
                BREAK
            END IF

            SET qrst TO queries[ijkl]
            IF NOT (qrst > LENGTH(yza))
                SET uvwx TO yza[qrst - 1]
                efgh = efgh + [uvwx]
            ELSE
                efgh = efgh + [-(1)]
            END IF
            SET ijkl TO ijkl + pqr
        UNTIL FALSE

        RETURN efgh
    END FUNCTION
END CLASS
```
