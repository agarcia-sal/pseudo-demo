```pseudocode
CLASS Solution
    FUNCTION minLength(s string, numOps integer) RETURNS integer
        FUNCTION check(m integer) RETURNS boolean
            SET a TO 0
            SET b TO 0
            SET j TO 0
            WHILE j LESS THAN LENGTH s
                SET b TO b + 1
                IF j EQUALS LENGTH s MINUS 1 OR s[j] NOT EQUALS s[j + 1] THEN
                    SET a TO a + ((b - 1) DIV m) + 1
                    IF a GREATER THAN numOps THEN
                        RETURN false
                    END IF
                    SET b TO 0
                END IF
                SET j TO j + 1
            END WHILE
            RETURN a <= numOps
        END FUNCTION

        SET p TO LENGTH s
        SET q TO 1
        SET r TO p
        REPEAT
            SET o TO (q + r) DIV 2
            IF check(o) THEN
                SET r TO o
            ELSE
                SET q TO o + 1
            END IF
        UNTIL NOT (q < r)
        RETURN q
    END FUNCTION
END CLASS
```
