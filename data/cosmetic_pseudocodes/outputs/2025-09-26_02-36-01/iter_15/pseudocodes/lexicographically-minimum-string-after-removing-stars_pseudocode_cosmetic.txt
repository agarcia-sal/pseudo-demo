CLASS Solution
    FUNCTION clearStars(s)
        SET gamma TO empty list
        SET delta TO 0
        WHILE delta < LENGTH(s)
            SET psi TO s[delta]
            IF NOT (psi != '*' )
                IF LENGTH(gamma) > 0
                    REMOVE element AT (LENGTH(gamma) - 1) FROM gamma
                END IF
            ELSE
                APPEND psi TO gamma
            END IF
            SET delta TO delta + 1
        END WHILE
        SET theta TO ""
        SET epsilon TO 0
        WHILE epsilon < LENGTH(gamma)
            SET theta TO theta + gamma[epsilon]
            SET epsilon TO epsilon + 1
        END WHILE
        RETURN theta
    END FUNCTION
END CLASS
