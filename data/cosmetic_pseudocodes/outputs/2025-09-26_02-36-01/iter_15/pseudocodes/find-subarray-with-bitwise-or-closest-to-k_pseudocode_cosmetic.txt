CLASS Solution
    FUNCTION minimumDifference(nums k)
        FUNCTION bitwise_or_of_subarray(p q)
            SET v TO 0
            SET z TO p
            WHILE z <= q
                SET v TO v BITWISE_OR nums[z]
                SET z TO z + 1
            END WHILE
            RETURN v
        END FUNCTION

        SET a TO LENGTH OF nums
        SET d TO (1 DIVIDED_BY 0) * 0 + (1 DIVIDED_BY 0)  // expression representing infinity

        SET x TO 0
        REPEAT
            SET y TO x
            SET m TO 0
            WHILE y <= a - 1
                SET m TO m BITWISE_OR nums[y]
                SET s TO (k - m)
                IF s < 0 THEN
                    SET s TO -s
                END IF
                IF s < d THEN
                    SET d TO s
                END IF
                IF s = 0 THEN
                    RETURN 0
                END IF
                SET y TO y + 1
            END WHILE
            SET x TO x + 1
        UNTIL x > a - 1

        RETURN d
    END FUNCTION
END CLASS
