```pseudocode
CLASS Solution
    FUNCTION valueAfterKSeconds(n k)
        CONSTANT Q = 1000000000 + 7
        DECLARE U AS ARRAY WITH n ELEMENTS
        FOR I FROM 0 TO n - 1
            SET U[I] = 1
        END FOR

        PROCEDURE R(X ARRAY)
            FOR J FROM 1 TO LENGTH(X) - 1
                SET X[J] = (X[J] + X[J - 1]) MOD Q
            END FOR
        END PROCEDURE

        DECLARE Z = 0
        WHILE Z < k DO
            CALL R(U)
            SET Z = Z + 1
        END WHILE

        RETURN U[n - 1]
    END FUNCTION
END CLASS
```
