CLASS Solution
    FUNCTION maxUpgrades(count LIST OF INTEGER upgrade LIST OF INTEGER sell LIST OF INTEGER money LIST OF INTEGER) RETURNS LIST OF INTEGER
        SET beta TO empty list
        SET theta TO 0
        WHILE theta LESS THAN LENGTH OF count
            SET xi TO count[theta]
            SET rho TO upgrade[theta]
            SET psi TO sell[theta]
            SET omega TO money[theta]
            SET delta TO 0
            SET tau TO omega
            SET phi TO 0
            WHILE phi LESS THAN OR EQUAL TO xi
                SET sigma TO xi MINUS phi
                SET upsold AS phi
                SET acumulated AS tau PLUS (upsold MULTIPLIED BY psi)
                SET presale AS acumulated
                SET purchasable AS presale DIVIDED BY rho
                IF NOT (purchasable LESS OR EQUAL TO sigma) THEN
                    SET purchasable TO sigma
                END IF
                IF delta LESS THAN purchasable THEN
                    SET delta TO purchasable
                END IF
                SET phi TO phi PLUS 1
            END WHILE
            APPEND delta TO beta
            SET theta TO theta PLUS 1
        END WHILE
        RETURN beta
    END FUNCTION
END CLASS
