CLASS Solution
    FUNCTION minOperations(nums)
        SET alpha TO 0
        SET beta TO 0
        SET idx TO 0
        WHILE idx < LENGTH(nums)
            IF (beta MOD 2) = 0 THEN
                SET omega TO nums[idx]
            ELSE
                SET omega TO 1 - nums[idx]
            END IF
            IF omega = 0 THEN
                SET alpha TO alpha + 1
                SET beta TO beta + 1
            END IF
            SET idx TO idx + 1
        END WHILE
        RETURN alpha
    END FUNCTION
END CLASS
