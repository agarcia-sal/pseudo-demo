CLASS Solution  
    FUNCTION isArraySpecial(nums, queries)  
        SET alpha TO empty collection  
        SET indexX TO 0  
        WHILE indexX LESS THAN LENGTH OF nums  
            SET beta TO nums[indexX] MOD 2  
            ADD beta TO alpha  
            INCREMENT indexX BY 1  
        END WHILE  
        
        SET gamma TO list FILLED WITH 0 OF SIZE LENGTH OF nums  
        SET indexY TO 1  
        REPEAT UNTIL indexY GREATER THAN LENGTH OF nums MINUS 1  
            IF NOT (alpha[indexY] XOR alpha[indexY - 1]) THEN  
                SET gamma[indexY] TO gamma[indexY - 1] + 1  
            ELSE  
                SET gamma[indexY] TO gamma[indexY - 1] + 0  
            END IF  
            INCREMENT indexY BY 1  
        END REPEAT  
        
        SET omega TO empty list  
        FOR EACH (phi, psi) IN queries DO  
            IF phi EQUALS psi THEN  
                ADD TRUE TO omega  
            ELSE  
                IF phi GREATER THAN 0 THEN  
                    SET delta TO gamma[psi] - gamma[phi]  
                ELSE  
                    SET delta TO gamma[psi] - 0  
                END IF  
                ADD (delta EQUALS 0) TO omega  
            END IF  
        END FOR  
        
        RETURN omega  
    END FUNCTION  
END CLASS
