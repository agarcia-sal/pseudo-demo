CLASS Solution
    FUNCTION minimumTimeToInitialState(word PARAMETER k)
        SET l TO LENGTH(word)
        SET c TO 1
        REPEAT
            SET s TO c * k
            IF (s >= l) OR (SUBSTRING(word, s, l - s) = SUBSTRING(word, 0, l - s)) THEN
                RETURN c
            END IF
            SET c TO c + 1
        UNTIL FALSE
    END FUNCTION
END CLASS
