```pseudocode
CLASS Solution
    FUNCTION minOperations(initial parameter TO string, target parameter TO string) RETURNS integer
        SET alpha TO LENGTH(initial)
        SET beta TO LENGTH(target)
        DEFINE matrix AS a (alpha + 1) BY (beta + 1) 2D list FILLED WITH 0
        SET peak_match TO 0

        SET outer_index TO 1
        WHILE outer_index <= alpha DO
            SET inner_index TO 1
            REPEAT
                IF (NOT (GET(initial, outer_index - 1) != GET(target, inner_index - 1))) THEN
                    SET matrix[outer_index][inner_index] TO matrix[outer_index - 1][inner_index - 1] + 1
                    IF peak_match < matrix[outer_index][inner_index] THEN
                        peak_match = matrix[outer_index][inner_index]
                    END IF
                END IF
                SET inner_index TO inner_index + 1
            UNTIL inner_index > beta
            SET outer_index TO outer_index + 1
        END WHILE

        RETURN alpha + beta - (2 * peak_match)
    END FUNCTION
END CLASS
```
