CLASS Solution
    FUNCTION findKthSmallest(coins k)
        FUNCTION count_smaller_or_equal(x)
            SET alpha TO 0
            SET beta TO 1
            SET gamma TO LENGTH_OF(coins)
            SET delta TO (2 ^ gamma) - 1
            WHILE beta <= delta
                SET eta TO 1
                SET theta TO 0
                SET i TO 0
                WHILE i < gamma
                    IF (beta BITWISE_AND (1 LEFT_SHIFT i)) <> 0
                        SET mu TO coins[i]
                        SET nu TO eta
                        SET xi TO GCD(nu, mu)
                        SET eta TO (eta * mu) / xi
                        SET theta TO theta + 1
                    END IF
                    SET i TO i + 1
                END WHILE
                IF (theta MOD 2) = 1
                    SET alpha TO alpha + (x / eta)
                ELSE
                    SET alpha TO alpha - (x / eta)
                END IF
                SET beta TO beta + 1
            END WHILE
            RETURN alpha
        END FUNCTION

        SET rho TO 1
        SET sigma TO k * MINIMUM(coins)
        SET tau TO rho
        WHILE tau < sigma
            SET upsilon TO (tau + sigma) / 2
            IF count_smaller_or_equal(upsilon) < k
                SET tau TO upsilon + 1
            ELSE
                SET sigma TO upsilon
            END IF
        END WHILE
        RETURN tau
    END FUNCTION
END CLASS
