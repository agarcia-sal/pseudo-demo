```pseudocode
CLASS Solution
    FUNCTION maxBottlesDrunk(numBottles numBottles numExchange numExchange)
        SET accumulatedConsumption TO 0
        SET recycledContainers TO 0
        REPEAT
            IF NOT (numBottles > 0) THEN
                BREAK
            END IF
            LET interimDrinks = numBottles
            SET accumulatedConsumption TO accumulatedConsumption + interimDrinks
            SET recycledContainers TO recycledContainers + interimDrinks
            SET numBottles TO 0
            WHILE recycledContainers >= numExchange
                SET recycledContainers TO recycledContainers - numExchange
                SET numBottles TO numBottles + 1
                SET numExchange TO numExchange + 1
            END WHILE
        UNTIL FALSE
        RETURN accumulatedConsumption
    END FUNCTION
END CLASS
```
