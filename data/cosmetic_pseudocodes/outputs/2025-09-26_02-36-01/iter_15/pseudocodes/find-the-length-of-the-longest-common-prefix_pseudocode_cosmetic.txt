```pseudocode
CLASS Solution
    FUNCTION longestCommonPrefix(arr1 arr1 OF integers arr2 arr2 OF integers) RETURNS integer
        SET alpha TO empty set
        SET beta TO empty set

        SET idx_a TO 0
        WHILE idx_a < LENGTH(arr1)
            SET val_a TO arr1[idx_a]
            SET str_a TO convert val_a TO string
            SET idx_b TO 1
            REPEAT
                ADD substring(str_a, 1, idx_b) TO alpha
                SET idx_b TO idx_b + 1
            UNTIL idx_b > LENGTH(str_a)
            SET idx_a TO idx_a + 1
        END WHILE

        SET idx_c TO 0
        WHILE idx_c < LENGTH(arr2)
            SET val_c TO arr2[idx_c]
            SET str_c TO convert val_c TO string
            FOR idx_d FROM 1 TO LENGTH(str_c)
                ADD substring(str_c, 1, idx_d) TO beta
            END FOR
            SET idx_c TO idx_c + 1
        END WHILE

        SET maxLen TO 0
        FOR each pfx IN alpha
            IF NOT (pfx NOT IN beta) THEN
                IF LENGTH(pfx) - maxLen > 0 THEN
                    SET maxLen TO LENGTH(pfx)
                END IF
            END IF
        END FOR

        RETURN maxLen
    END FUNCTION
END CLASS
```
