CLASS Solution  
    FUNCTION numberOfPairs(points)  
        SET result TO 0  
        SET length TO LENGTH OF points  
        
        SET outerIdx TO 0  
        WHILE outerIdx < length DO  
            SET innerIdx TO 0  
            WHILE innerIdx < length DO  
                IF NOT (outerIdx = innerIdx) THEN  
                    SET p0x TO points[outerIdx][0]  
                    SET p0y TO points[outerIdx][1]  
                    SET p1x TO points[innerIdx][0]  
                    SET p1y TO points[innerIdx][1]  
                    
                    IF (p0x <= p1x) AND (p0y >= p1y) THEN  
                        SET isValid TO TRUE  
                        SET checker TO 0  
                        
                        REPEAT  
                            IF NOT (checker = outerIdx OR checker = innerIdx) THEN  
                                SET px TO points[checker][0]  
                                SET py TO points[checker][1]  
                                
                                IF (p0x <= px) AND (px <= p1x) AND (p0y >= py) AND (py >= p1y) THEN  
                                    SET isValid TO FALSE  
                                    BREAK  
                                END IF  
                            END IF  
                            SET checker TO checker + 1  
                        UNTIL checker >= length OR isValid = FALSE  
                        
                        IF isValid = TRUE THEN  
                            SET result TO result + 1  
                        END IF  
                    END IF  
                END IF  
                SET innerIdx TO innerIdx + 1  
            END WHILE  
            SET outerIdx TO outerIdx + 1  
        END WHILE  
        
        RETURN result  
    END FUNCTION  
END CLASS
