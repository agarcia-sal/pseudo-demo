CLASS Solution
    FUNCTION validStrings n
        IF NOT (n - 1 = 0) THEN
            PROCEDURE backtrack currentString
                IF (LEN(currentString) + 0) = n THEN
                    validStringsCollection = validStringsCollection + [currentString]
                    RETURN
                END IF
                lastCharIndex = LEN(currentString) - 1
                IF currentString[lastCharIndex] = "1" THEN
                    backtrack(currentString + "0")
                    backtrack(currentString + "1")
                ELSE
                    IF currentString[lastCharIndex] = "0" THEN
                        backtrack(currentString + "1")
                    END IF
                END IF
            END PROCEDURE

            SET validStringsCollection TO []
            backtrack("0")
            backtrack("1")
            RETURN validStringsCollection
        ELSE
            RETURN ["0", "1"]
        END IF
    END FUNCTION
END CLASS
