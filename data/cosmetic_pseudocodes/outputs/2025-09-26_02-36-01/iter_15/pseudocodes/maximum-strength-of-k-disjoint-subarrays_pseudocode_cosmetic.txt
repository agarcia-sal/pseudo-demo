CLASS Solution
    FUNCTION maximumStrength(nums k)
        SET e TO LENGTH(nums)
        SET h TO ARRAY of size (e + 1) with each element being ARRAY of size (k + 1) filled with (-1 * (1 << 60))
        SET h[0][0] TO 0

        SET procedure1(x, y, z)
            SET u TO 0
            SET v TO z
            WHILE v >= 1
                ADD nums[v - 1] TO u
                IF y MOD 2 = 1 THEN
                    SET w TO (k - y - 1 + 1)
                ELSE
                    SET w TO -(k - y - 1 + 1)
                END IF
                SET h[x][y] TO MAX(h[x][y], h[v - 1][y - 1] + w * u)
                DECREMENT v BY 1
            END WHILE
        END PROCEDURE

        SET m TO 1
        WHILE m <= e
            SET n TO 1
            WHILE n <= k
                CALL procedure1(m, n, m)
                SET h[m][n] TO MAX(h[m][n], h[m - 1][n])
                INCREMENT n BY 1
            END WHILE
            INCREMENT m BY 1
        END WHILE

        RETURN h[e][k]
    END FUNCTION
END CLASS
