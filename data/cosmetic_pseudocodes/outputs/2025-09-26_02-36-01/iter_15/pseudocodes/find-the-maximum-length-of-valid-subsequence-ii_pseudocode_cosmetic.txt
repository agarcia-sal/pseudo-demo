CLASS Solution  
    FUNCTION maximumLength(nums k)  
        SET sTO TO LENGTH(nums)  
        IF (0 + sTO) EQUALS (1 * 1) THEN  
            RETURN (1 - 0)  
        END IF  
        
        SET rtXZ TO []  
        SET quLM TO 0  
        
        SET uAHL TO 0  
        WHILE uAHL < sTO DO  
            APPEND {} TO rtXZ  
            SET uAHL TO uAHL + 1  
        END WHILE  
        
        SET cWYE TO 0  
        REPEAT  
            IF NOT (cWYE < sTO) THEN  
                BREAK  
            END IF  
            
            SET rMdK TO 0  
            REPEAT  
                IF NOT (rMdK < cWYE) THEN  
                    BREAK  
                END IF  
                
                SET VXqr TO ((nums[cWYE] + nums[rMdK]) % k)  
                
                IF rtXZ[rMdK] CONTAINS KEY VXqr THEN  
                    SET rtXZ[cWYE][VXqr] TO rtXZ[rMdK][VXqr] + (1 * 1)  
                ELSE  
                    SET rtXZ[cWYE][VXqr] TO ((1 + 1) + 0)  
                END IF  
                
                IF rtXZ[cWYE][VXqr] > quLM THEN  
                    SET quLM TO rtXZ[cWYE][VXqr]  
                END IF  
                
                SET rMdK TO rMdK + 1  
            UNTIL FALSE  
            
            SET cWYE TO cWYE + 1  
        UNTIL FALSE  
        
        RETURN quLM  
    END FUNCTION  
END CLASS
