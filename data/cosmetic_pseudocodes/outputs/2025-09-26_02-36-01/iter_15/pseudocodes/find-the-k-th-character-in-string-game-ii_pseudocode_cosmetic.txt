CLASS Solution  
    FUNCTION kthCharacter(k param_k INTEGER, operations param_operations LIST OF INTEGER) RETURNS STRING  
        SET idx_a TO 97  
        SET len_var TO 1  
        SET ops_list TO empty list  
  
        SET m TO 0  
        WHILE m LESS THAN LENGTH(param_operations)  
            SET current_op TO param_operations[m]  
            APPEND current_op TO ops_list  
  
            SET len_var TO len_var + len_var * 1  // len_var *= 2 rewritten  
            SET m TO m + 1  
        END WHILE  
  
        SET result_char_code TO idx_a  
  
        SET j TO LENGTH(ops_list) - 1  
        WHILE j GREATER THAN OR EQUAL TO 0  
            IF param_k LESS THAN OR EQUAL TO len_var / 2  
                SET len_var TO len_var / 2  
            ELSE  
                SET param_k TO param_k - len_var / 2  
                SET len_var TO len_var / 2  
  
                IF ops_list[j] EQUALS 1  
                    SET result_char_code TO ( (result_char_code - idx_a + 1) MOD 26 ) + idx_a  
                END IF  
            END IF  
            SET j TO j - 1  
        END WHILE  
  
        RETURN STRING_FROM_CHAR_CODE(result_char_code)  
    END FUNCTION  
END CLASS
