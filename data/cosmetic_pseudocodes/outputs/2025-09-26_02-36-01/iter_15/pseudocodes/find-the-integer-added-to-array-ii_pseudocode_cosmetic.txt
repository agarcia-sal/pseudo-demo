CLASS Solution
    FUNCTION minimumAddedInteger(nums1 RETURNS List OF integer, nums2 RETURNS List OF integer) RETURNS integer
        CALL nums1.sort()
        CALL nums2.sort()

        SET alpha TO 0
        WHILE alpha <= (LENGTH(nums1) - 1)
            SET beta TO alpha + 1
            WHILE beta <= (LENGTH(nums1) - 1)
                SET gamma TO EMPTY LIST
                SET delta TO 0
                WHILE delta < alpha
                    CALL gamma.append(nums1[delta])
                    SET delta TO delta + 1
                END WHILE

                SET epsilon TO alpha + 1
                WHILE epsilon < beta
                    CALL gamma.append(nums1[epsilon])
                    SET epsilon TO epsilon + 1
                END WHILE

                SET zeta TO beta + 1
                WHILE zeta <= (LENGTH(nums1) - 1)
                    CALL gamma.append(nums1[zeta])
                    SET zeta TO zeta + 1
                END WHILE
                
                SET eta TO (nums2[0] - gamma[0])
                
                SET theta TO TRUE
                SET iota TO 0
                WHILE iota <= (LENGTH(nums2) - 1)
                    IF (gamma[iota] + eta) != nums2[iota]
                        SET theta TO FALSE
                        EXIT WHILE
                    END IF
                    SET iota TO iota + 1
                END WHILE
                
                IF theta == TRUE
                    RETURN eta
                END IF
                
                SET beta TO beta + 1
            END WHILE
            SET alpha TO alpha + 1
        END WHILE

        RETURN None
    END FUNCTION
END CLASS
