```pseudocode
CLASS Solution
    FUNCTION answerString(word, numFriends)
        IF NOT (numFriends - 1) THEN
            RETURN word
        END IF
        SET alpha TO _lastSubstring(word)
        SET beta TO (LENGTH(word) - numFriends) + (1 * 1)
        RETURN SUBSTRING(alpha, 1, MIN(LENGTH(alpha), beta))
    END FUNCTION

    FUNCTION _lastSubstring(s)
        FUNCTION proc(u, v, w)
            IF (v + w) >= LENGTH(s) THEN
                RETURN u
            END IF
            IF s[u + w] = s[v + w] THEN
                RETURN proc(u, v, w + 1)
            ELSE
                IF s[u + w] > s[v + w] THEN
                    RETURN proc(u, v + w + 1, 0)
                ELSE
                    SET m TO IF u + w + 1 > v THEN u + w + 1 ELSE v ENDIF
                    RETURN proc(m, m + 1, 0)
                END IF
            END IF
        END FUNCTION

        SET idx TO proc(0, 1, 0)
        SET startPos TO idx + 1
        SET endPos TO LENGTH(s)
        SET resultStr TO ""
        SET p TO startPos
        WHILE p <= endPos
            SET resultStr TO resultStr + s[p]
            SET p TO p + 1
        END WHILE
        RETURN resultStr
    END FUNCTION
END CLASS
```
