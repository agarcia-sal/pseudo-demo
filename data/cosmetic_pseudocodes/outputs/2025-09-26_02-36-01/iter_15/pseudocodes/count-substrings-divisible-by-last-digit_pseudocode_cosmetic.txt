CLASS Solution  
    FUNCTION countSubstrings(s)  
        SET x1 TO LENGTH OF s  
        SET x2 TO 0  
        SET x3 TO 0  
        WHILE x3 < x1 DO  
            SET x4 TO x3  
            SET x5 TO 0  
            REPEAT  
                SET x6 TO VALUE OF s[x4] CONVERTED TO INTEGER  
                SET x5 TO (x5 * 10) + x6  
                IF ((x6 ≠ 0) AND ((x5 - (x5 DIV 1)) MOD x6 = 0)) THEN  
                    SET x2 TO x2 + 1  
                END IF  
                SET x4 TO x4 + 1  
                IF x4 ≥ x1 THEN  
                    BREAK  
                END IF  
            UNTIL FALSE  
            SET x3 TO x3 + 1  
        END WHILE  
        RETURN x2  
    END FUNCTION  
END CLASS
