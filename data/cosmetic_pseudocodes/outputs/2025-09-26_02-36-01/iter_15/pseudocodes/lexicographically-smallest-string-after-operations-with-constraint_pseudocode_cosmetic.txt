```pseudocode
CLASS Solution
    FUNCTION getSmallestString(s parameter of type string k parameter of type integer) RETURNS string
        FUNCTION cyclic_distance(c1 parameter of type character c2 parameter of type character) RETURNS integer
            SET x TO ABSOLUTE_VALUE(ORDINAL_VALUE(c1) - ORDINAL_VALUE(c2))
            SET y TO 26 - x
            IF y < x THEN
                RETURN y
            ELSE
                RETURN x
            END IF
        END FUNCTION

        SET p TO CONVERT_TO_LIST_OF_CHARS(s)
        SET m TO LENGTH(s)
        SET q TO 0

        REPEAT
            IF NOT (k > 0 AND q < m) THEN
                BREAK
            END IF

            IF p[q] = 'a' THEN
                SET q TO q + 1
                CONTINUE
            END IF

            SET r TO cyclic_distance(p[q], 'a')

            IF r <= k THEN
                SET p[q] TO 'a'
                SET k TO k - r
            ELSE
                SET p[q] TO CHARACTER_FROM_ORDINAL(ORDINAL_VALUE(p[q]) - k)
                SET k TO 0
            END IF

            SET q TO q + 1
        UNTIL FALSE

        SET result TO EMPTY_STRING
        FOR h FROM 0 TO m - 1 DO
            SET result TO result + p[h]
        END FOR

        RETURN result
    END FUNCTION
END CLASS
```
