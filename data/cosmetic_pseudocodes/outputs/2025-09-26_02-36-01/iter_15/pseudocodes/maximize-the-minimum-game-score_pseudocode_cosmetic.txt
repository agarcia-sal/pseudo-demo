CLASS Solution  
    FUNCTION maxScore(points, m)  
        FUNCTION isPossible(minVal, m)  
            SET totalActions TO 0  
            SET previousActions TO 0  
            
            SET indexCounter TO 0  
            WHILE indexCounter < LENGTH(points)  
                SET elementValue TO points[indexCounter]  
                SET quotient TO (minVal + elementValue - 1) / elementValue  
                SET requiredMoves TO FLOOR(quotient)  
                
                IF requiredMoves - previousActions < 0 THEN  
                    SET requiredMoves TO 0  
                ELSE  
                    SET requiredMoves TO requiredMoves - previousActions  
                END IF  
                
                IF requiredMoves > 0 THEN  
                    SET totalActions TO totalActions + (requiredMoves * 2 - 1)  
                    SET previousActions TO requiredMoves - 1  
                ELSE IF (indexCounter + 1) < LENGTH(points) THEN  
                    SET totalActions TO totalActions + 1  
                    SET previousActions TO 0  
                END IF  
                
                IF totalActions > m THEN  
                    RETURN False  
                END IF  
                
                SET indexCounter TO indexCounter + 1  
            END WHILE  
            
            RETURN True  
        END FUNCTION  

        SET leftBound TO 0  
        SET rightBound TO ((m + 1) // 2) * (points[0] + 1)  

        WHILE (leftBound < rightBound)  
            SET middleVal TO (leftBound + rightBound + 1) // 2  
            IF isPossible(middleVal, m) THEN  
                SET leftBound TO middleVal  
            ELSE  
                SET rightBound TO middleVal - 1  
            END IF  
        END WHILE  

        RETURN leftBound  
    END FUNCTION  
END CLASS
