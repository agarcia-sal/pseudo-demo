CLASS Solution
    FUNCTION countDays(days, meetings)
        PROCEDURE reorder(list)
            SET i TO 1
            WHILE i < LENGTH(list)
                SET j TO i
                WHILE j > 0 AND list[j - 1][0] > list[j][0]
                    SET temp TO list[j - 1]
                    SET list[j - 1] TO list[j]
                    SET list[j] TO temp
                    DECREMENT j BY 1
                END WHILE
                INCREMENT i BY 1
            END WHILE
        END PROCEDURE

        CALL reorder(meetings)
        
        SET alpha TO 1
        SET beta TO 0

        INDEX_LOOP: FOR idx FROM 0 TO LENGTH(meetings) - 1
            SET item TO meetings[idx]
            SET omega TO item[0]
            SET psi TO item[1]

            IF NOT (alpha >= omega)
                SET increment TO omega - alpha
                SET beta TO beta + increment
            END IF

            IF psi + 1 > alpha
                SET alpha TO psi + 1
            END IF
        END FOR INDEX_LOOP

        IF alpha <= days
            SET gamma TO days - alpha + 1
            SET beta TO beta + gamma
        END IF

        RETURN beta
    END FUNCTION
END CLASS
