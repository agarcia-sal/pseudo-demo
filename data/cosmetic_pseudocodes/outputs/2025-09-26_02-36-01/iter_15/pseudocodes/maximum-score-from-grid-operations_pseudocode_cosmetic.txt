```pseudocode
CLASS Solution
    FUNCTION maximumScore(grid)
        SET alpha TO LENGTH grid
        CREATE delta AS 2D ARRAY OF SIZE (alpha + 1) x (alpha + 1) FILLED WITH 0
        CREATE omega AS ARRAY OF SIZE (alpha + 1) FILLED WITH 0
        CREATE psi AS ARRAY OF SIZE (alpha + 1) FILLED WITH 0

        SET k TO 0
        WHILE k < alpha
            SET m TO 0
            WHILE m < alpha
                delta[m + 1][k] = delta[m][k] + grid[k][m]
                m = m + 1
            END WHILE
            k = k + 1
        END WHILE

        SET alpha2 TO 1
        WHILE alpha2 < alpha
            CREATE lambda AS ARRAY OF SIZE (alpha + 1) FILLED WITH 0
            CREATE sigma AS ARRAY OF SIZE (alpha + 1) FILLED WITH 0

            SET zeta TO 0
            WHILE zeta <= alpha
                SET eta TO 0
                WHILE eta <= alpha
                    IF zeta > eta THEN
                        SET tau TO (delta[zeta][alpha2 - 1] - delta[eta][alpha2 - 1])
                        lambda[zeta] = MAX(lambda[zeta], psi[eta] + tau)
                        sigma[zeta] = MAX(sigma[zeta], psi[eta] + tau)
                    ELSE
                        SET tau TO (delta[eta][alpha2] - delta[zeta][alpha2])
                        lambda[zeta] = MAX(lambda[zeta], omega[eta] + tau)
                        sigma[zeta] = MAX(sigma[zeta], omega[eta])
                    END IF
                    eta = eta + 1
                END WHILE
                zeta = zeta + 1
            END WHILE

            omega = lambda
            psi = sigma
            alpha2 = alpha2 + 1
        END WHILE

        SET result TO omega[0]
        SET i2 TO 1
        WHILE i2 <= alpha
            IF omega[i2] > result THEN
                result = omega[i2]
            END IF
            i2 = i2 + 1
        END WHILE

        RETURN result
    END FUNCTION
END CLASS
```
