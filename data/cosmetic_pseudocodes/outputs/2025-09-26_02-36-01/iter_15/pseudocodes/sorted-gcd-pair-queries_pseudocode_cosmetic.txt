```pseudocode
CLASS Solution
    FUNCTION gcdValues(nums, queries)
        DECLARE alpha = nums
        DECLARE beta = queries
        DECLARE gamma = MIN_INT
        FOR z FROM 1 TO LENGTH(alpha)
            IF alpha[z] > gamma THEN
                SET gamma = alpha[z]
            END IF
        END FOR

        DECLARE delta = NEW map
        FOR xi FROM 1 TO LENGTH(alpha)
            IF delta CONTAINS alpha[xi] THEN
                SET delta[alpha[xi]] = delta[alpha[xi]] + 1
            ELSE
                SET delta[alpha[xi]] = 1
            END IF
        END FOR

        DECLARE epsilon = ARRAY OF SIZE (gamma + 1) FILLED WITH 0

        DECLARE function_omega(i, k)
            IF k > gamma THEN
                RETURN 0
            END IF
            DECLARE s = 0
            DECLARE t = epsilon[i] - epsilon[k]
            s = s + (IF delta CONTAINS k THEN delta[k] ELSE 0)
            RETURN s + function_omega(i, k + i) - t
        END FUNCTION

        DECLARE index_r FROM gamma DOWNTO 1 STEP -1
            DECLARE u = 0
            DECLARE v = index_r
            WHILE v <= gamma
                u = u + (IF delta CONTAINS v THEN delta[v] ELSE 0)
                epsilon[index_r] = epsilon[index_r] - epsilon[v]
                v = v + index_r
            END WHILE
            epsilon[index_r] = epsilon[index_r] + ((u * (u - 1)) / 2)
        END FOR

        DECLARE accumulation_s = ARRAY WITH FIRST ELEMENT 0
        FOR i FROM 1 TO LENGTH(epsilon) - 1
            accumulation_s[i] = accumulation_s[i-1] + epsilon[i]
        END FOR

        DECLARE omega_result = EMPTY LIST

        FOR phi IN beta
            DECLARE left = 1
            DECLARE right = LENGTH(accumulation_s) - 1
            DECLARE mid = 0
            WHILE left <= right
                mid = (left + right) DIV 2
                IF accumulation_s[mid] <= phi THEN
                    left = mid + 1
                ELSE
                    right = mid - 1
                END IF
            END WHILE
            APPEND left TO omega_result
        END FOR

        RETURN omega_result
    END FUNCTION
END CLASS
```
