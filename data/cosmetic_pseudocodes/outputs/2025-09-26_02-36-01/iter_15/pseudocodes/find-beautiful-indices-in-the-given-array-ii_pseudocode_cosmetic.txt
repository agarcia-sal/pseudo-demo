CLASS Solution
    FUNCTION beautifulIndices(s parameter a parameter b parameter k)
        DEFINE list_x AS new empty list
        SET var_m TO 0
        REPEAT
            IF var_m GREATER THAN (LENGTH OF s) MINUS (LENGTH OF a) THEN
                BREAK
            END IF
            DEFINE temp_1 AS SUBSTRING OF s FROM var_m TO var_m PLUS (LENGTH OF a) MINUS 1
            IF temp_1 EQUALS a THEN
                APPEND var_m TO list_x
            END IF
            SET var_m TO var_m PLUS 1
        UNTIL FALSE

        DEFINE list_y AS new empty list
        SET var_n TO 0
        REPEAT
            IF var_n GREATER THAN (LENGTH OF s) MINUS (LENGTH OF b) THEN
                BREAK
            END IF
            DEFINE temp_2 AS SUBSTRING OF s FROM var_n TO var_n PLUS (LENGTH OF b) MINUS 1
            IF temp_2 EQUALS b THEN
                APPEND var_n TO list_y
            END IF
            SET var_n TO var_n PLUS 1
        UNTIL FALSE

        DEFINE result_list AS new empty list
        SET idx_p TO 0
        SET idx_q TO 0

        WHILE idx_p LESS THAN LENGTH OF list_x DO
            IF idx_q GREATER THAN EQUAL TO LENGTH OF list_y THEN
                EXIT WHILE
            END IF

            DEFINE diff_val AS list_x AT idx_p MINUS list_y AT idx_q
            IF diff_val LESS THAN 0 THEN
                SET diff_val TO diff_val MULTIPLIED BY (-1)
            END IF

            IF diff_val LESS THAN OR EQUAL TO k THEN
                APPEND list_x AT idx_p TO result_list
                SET idx_p TO idx_p PLUS 1
            ELSE
                IF list_x AT idx_p LESS THAN list_y AT idx_q THEN
                    SET idx_p TO idx_p PLUS 1
                ELSE
                    SET idx_q TO idx_q PLUS 1
                END IF
            END IF
        END WHILE

        RETURN result_list
    END FUNCTION
END CLASS
