CLASS Solution  
    FUNCTION maxScore(nums)  
        SET u TO LENGTH(nums)  
        SET v TO ARRAY OF SIZE u FILLED WITH 0  
        SET v[1] TO 0  
        SET x TO 2  
        WHILE x <= u DO  
            SET y TO 1  
            WHILE y < x DO  
                SET a TO v[x]  
                SET b TO v[y] + ((x - y) * nums[x])  
                IF a < b THEN  
                    SET v[x] TO b  
                END IF  
                SET y TO y + 1  
            END WHILE  
            SET x TO x + 1  
        END WHILE  
        RETURN v[u - 1]  
    END FUNCTION  
END CLASS
