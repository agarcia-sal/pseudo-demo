        SET wHsGuQOY TO 0

                        END IF
                        SET GbwnoZKL TO GbwnoZKL + 1
                    END WHILE
                    SET YdcqdmKC TO YdcqdmKC + 1
                END WHILE
                SET wHsGuQOY TO (wHsGuQOY + rLpGeNDV) % WnqJGwrM

                END WHILE
            END IF
        END PROCEDURE

        CALL OuVZgxKb(nums, k, 0, NEW LIST OF INTEGERS WITH SIZE k INITIALIZED TO 0)

    END FUNCTION
