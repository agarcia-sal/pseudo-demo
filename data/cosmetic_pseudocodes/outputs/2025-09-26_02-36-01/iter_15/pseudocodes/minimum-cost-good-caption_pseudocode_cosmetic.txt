      SET w TO LENGTH(caption)
      IF NOT (w >= 3) THEN
          RETURN ""
      END IF

      SET x TO TO_LIST(caption)
      SET u TO 0

      LOOP
          IF NOT (u < w) THEN
              BREAK
          END IF

          SET p TO u
          SET q TO p

          END LOOP

          SET r TO q - p

          IF r < 3 THEN
              IF (p > 0) AND (x[p - 1] = x[p]) THEN
                  SET x[p - 1] TO x[p]
                  SET p TO p - 1
                  SET r TO r + 1
              END IF

              IF (q < w) AND (x[q - 1] = x[q]) THEN
                  SET x[q] TO x[q - 1]
                  SET q TO q + 1
                  SET r TO r + 1
              END IF

              IF r < 3 THEN
                  IF p > 0 THEN
                      SET v TO x[p - 1]
                  ELSE
                      SET v TO x[q]
                  END IF

                  IF v = 'a' THEN
                      SET v TO 'b'
                  ELSE IF v = 'z' THEN
                      SET v TO 'y'
                  ELSE
                      SET v TO CHAR(ORD(v) + 1)
                  END IF

                  SET s TO 0
                  REPEAT
                      IF NOT (s < r) THEN
                          BREAK
                      END IF
                      SET x[p + s] TO v
                      SET s TO s + 1
                  UNTIL FALSE

                  SET u TO p + r
                  CONTINUE
              END IF
          END IF

          SET u TO q
      END LOOP

      RETURN JOIN(x)
    END FUNCTION
END CLASS
