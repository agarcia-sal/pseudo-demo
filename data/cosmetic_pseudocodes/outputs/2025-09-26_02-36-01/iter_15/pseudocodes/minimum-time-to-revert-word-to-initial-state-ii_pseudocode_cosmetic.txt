CLASS Solution
    FUNCTION minimumTimeToInitialState(word parameter word AND integer parameter k) RETURNS integer
        SET alpha TO LENGTH OF word
        SET omega TO 1
        LOOP
            SET beta TO SUBSTRING word FROM POSITION omega * k TO alpha
            IF NOT (word DOES NOT START WITH beta) THEN
                RETURN omega
            END IF
            SET omega TO omega + 1
        END LOOP
    END FUNCTION
END CLASS
