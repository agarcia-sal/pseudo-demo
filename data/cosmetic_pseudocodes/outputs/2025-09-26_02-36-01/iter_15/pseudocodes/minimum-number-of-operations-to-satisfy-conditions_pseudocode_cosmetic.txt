```pseudocode
CLASS Solution
    FUNCTION minimumOperations(grid)
        LET x1 BE LENGTH(grid)
        LET x2 BE LENGTH(grid[0])
        LET r0 BE 0

        FUNCTION replaceFirstDiff(arr, idx)
            IF arr[idx] != arr[idx + 1] THEN
                LET t0 BE arr[idx]
                LET t1 BE arr[idx + 1]
                LET t2 BE t1
                SET r0 TO r0 + 1
                SET arr[idx + 1] TO t0
                RETURN
            END IF
        END FUNCTION

        FUNCTION findReplacement(val)
            FOR c0 FROM 0 TO 9
                IF c0 != val THEN
                    RETURN c0
                END IF
            END FOR
            RETURN val
        END FUNCTION

        FOR c1 FROM 0 TO x2 - 1
            FOR c2 FROM 0 TO x1 - 2
                replaceFirstDiff(grid[c2], c1)
            END FOR

            FOR c3 FROM 0 TO x1 - 1
                IF (c1 < x2 - 1) AND (grid[c3][c1] == grid[c3][c1 + 1]) THEN
                    LET a0 BE grid[c3][c1]
                    LET a1 BE findReplacement(a0)
                    SET grid[c3][c1 + 1] TO a1
                    SET r0 TO r0 + 1
                END IF
            END FOR
        END FOR

        RETURN r0
    END FUNCTION
END CLASS
```
