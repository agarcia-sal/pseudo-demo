CLASS Solution  
    FUNCTION maxScore(nums)  
        SET t1 TO LENGTH(nums)  
        SET arr1 TO ARRAY FILLED WITH 0 FOR t1 ELEMENTS  
        SET q1 TO t1 - 1  
        WHILE q1 >= 0  
            SET v1 TO 0  
            SET q2 TO q1 + 1  
            WHILE q2 < t1  
                SET temp1 TO (q2 - q1) * nums[q2]  
                SET temp2 TO temp1 + arr1[q2]  
                IF v1 < temp2 THEN  
                    SET v1 TO temp2  
                END IF  
                SET q2 TO q2 + 1  
            END WHILE  
            SET arr1[q1] TO v1  
            SET q1 TO q1 - 1  
        END WHILE  
        RETURN arr1[0]  
    END FUNCTION  
END CLASS
