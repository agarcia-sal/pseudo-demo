CLASS Solution
    FUNCTION maximumLength(nums)
        SET a1 TO 0
        SET b2 TO 0
        SET c3 TO 1
        WHILE c3 < LENGTH OF nums
            SET d4 TO (nums[c3 - 1] + nums[c3])
            IF NOT (d4 MOD 2 = 1)
                SET a1 TO MAX(a1 + 1, b2)
            ELSE
                SET b2 TO MAX(b2 + 1, a1)
            END IF
            SET c3 TO c3 + 1
        END WHILE
        IF a1 > b2
            RETURN a1 + 1
        ELSE
            RETURN b2 + 1
        END IF
    END FUNCTION
END CLASS
