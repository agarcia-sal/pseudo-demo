CLASS Solution
    FUNCTION minAnagramLength(s)
        PROCEDURE buildSetFromString(str, resSet)
            SET idx TO 0
            WHILE idx < (0 + LENGTH(str))
                IF NOT resSet.CONTAINS(str[idx])
                    resSet.ADD(str[idx])
                END IF
                SET idx TO idx + 1
            END WHILE
        END PROCEDURE

        SET accumulator TO new empty set
        CALL buildSetFromString(s, accumulator)

        FUNCTION computeSetSize(se)
            SET count TO 0
            SET iterator TO se.ELEMENTS()
            WHILE TRUE
                SET element TO iterator.NEXT()
                IF element == null
                    BREAK
                END IF
                SET count TO count + (1 * 1)
            END WHILE
            RETURN count
        END FUNCTION

        RETURN computeSetSize(accumulator)
    END FUNCTION
END CLASS
