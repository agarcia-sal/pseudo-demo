CLASS Solution
    FUNCTION nonSpecialCount(l r)
        FUNCTION is_prime(num)
            IF NOT (num GREATER THAN 1)
                RETURN false
            ELSE IF num LESS THAN OR EQUAL TO 3
                RETURN true
            ELSE
                IF (num MOD 2 = 0) OR (num MOD 3 = 0)
                    RETURN false
                ELSE
                    DECLARE x = 5
                    WHILE (x * x) <= num
                        IF (num MOD x = 0) OR (num MOD (x + 2) = 0)
                            RETURN false
                        END IF
                        SET x = x + 6
                    END WHILE
                    RETURN true
                END IF
            END IF
        END FUNCTION

        DECLARE a = CEIL(SQRT(l))
        DECLARE b = FLOOR(SQRT(r))

        DECLARE c = 0

        DECLARE d = a
        WHILE d <= b
            IF is_prime(d) = true
                SET c = c + 1
            END IF
            SET d = d + 1
        END WHILE

        DECLARE e = r - l + 1

        RETURN e - c
    END FUNCTION
END CLASS
