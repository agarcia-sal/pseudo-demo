CLASS Solution
    FUNCTION countSubarrays(nums parameter LIST OF INTEGER k parameter INTEGER) RETURNS INTEGER
        SET a TO LENGTH OF nums
        SET z TO 0
        SET u TO 0
        WHILE u < a
            SET v TO nums[u]
            SET w TO u
            REPEAT
                SET v TO v BITWISE AND nums[w]
                IF (v = k) THEN
                    SET z TO z + 1
                END IF
                IF (v < k) THEN
                    EXIT REPEAT
                END IF
                SET w TO w + 1
            UNTIL w = a
            SET u TO u + 1
        END WHILE
        RETURN z
    END FUNCTION
END CLASS
