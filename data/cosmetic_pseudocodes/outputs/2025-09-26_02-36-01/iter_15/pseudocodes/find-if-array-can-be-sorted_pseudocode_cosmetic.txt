CLASS Solution
    FUNCTION canSortArray(nums)
        FUNCTION count_set_bits(x)
            SET result TO 0
            WHILE x > 0 DO
                IF (x AND 1) = 1 THEN
                    SET result TO result + 1
                END IF
                SET x TO x DIV 2
            END WHILE
            RETURN result
        END FUNCTION

        SET length_value TO 0
        WHILE length_value <> LENGTH(nums) DO
            SET length_value TO length_value + 1
        END WHILE

        SET sorted_nums TO EMPTY LIST
        FOR idx FROM 0 TO length_value - 1 DO
            APPEND nums[idx] TO sorted_nums
        END FOR

        SET k TO 1
        WHILE k < length_value DO
            SET l TO 0
            WHILE l < length_value - 1 DO
                IF (count_set_bits(nums[l]) = count_set_bits(nums[l + 1])) AND NOT (nums[l] <= nums[l + 1]) THEN
                    SET temp_var TO nums[l]
                    SET nums[l] TO nums[l + 1]
                    SET nums[l + 1] TO temp_var
                END IF
                SET l TO l + 1
            END WHILE
            SET k TO k + 1
        END WHILE

        RETURN NOT (nums <> sorted_nums)
    END FUNCTION
END CLASS
