CLASS Solution
    FUNCTION minimumArea(grid)
        IF NOT (grid ≠ [] AND grid[0] ≠ []) THEN
            RETURN 0
        END IF

        LET a = LENGTH(grid)
        LET b = LENGTH(grid[0])
        LET c = +∞
        LET d = -∞
        LET e = +∞
        LET f = -∞

        LET g = 0
        WHILE g < a DO
            LET h = 0
            WHILE h < b DO
                IF grid[g][h] = 1 THEN
                    IF c > g THEN
                        c ← g
                    END IF
                    IF d < g THEN
                        d ← g
                    END IF
                    IF e > h THEN
                        e ← h
                    END IF
                    IF f < h THEN
                        f ← h
                    END IF
                END IF
                h ← h + 1
            END WHILE
            g ← g + 1
        END WHILE

        LET height = (d - c) + (1 * 1)
        LET width = (f - e) + 1
        RETURN height * width
    END FUNCTION
END CLASS
