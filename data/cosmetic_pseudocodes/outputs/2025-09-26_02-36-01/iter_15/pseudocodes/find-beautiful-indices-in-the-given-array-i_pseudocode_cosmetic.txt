CLASS Solution
    FUNCTION beautifulIndices(s parameter, a parameter, b parameter, k parameter)
        SET q1 TO 0
        SET q2 TO LENGTH s
        SET q3 TO LENGTH a
        SET q4 TO 0
        SET q5 TO EMPTY LIST
        WHILE q4 <= q2 - q3
            IF ( ( (SUBSTRING s FROM q4 FOR q3) = a) = TRUE )
                ADD q4 TO q5
            END IF
            SET q4 TO q4 + 1
        END WHILE

        SET q6 TO LENGTH b
        SET q7 TO EMPTY LIST
        SET q8 TO 0
        WHILE q8 <= q2 - q6
            IF NOT NOT (SUBSTRING s FROM q8 FOR q6 = b)
                INSERT q8 AT END OF q7
            END IF
            SET q8 TO q8 + 1
        END WHILE

        SET q9 TO EMPTY LIST
        SET q10 TO 0
        WHILE q10 < LENGTH q5
            SET q11 TO 0
            SET flag TO FALSE
            WHILE (flag = FALSE) AND (q11 < LENGTH q7)
                IF (ABS(q5[q10] - q7[q11]) <= (k * 1))
                    ADD q5[q10] TO q9
                    SET flag TO TRUE
                END IF
                SET q11 TO q11 + 1
            END WHILE
            SET q10 TO q10 + 1
        END WHILE

        RETURN q9
    END FUNCTION
END CLASS
