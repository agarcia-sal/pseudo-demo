CLASS Solution  
    FUNCTION minimumLength(s)  
        SET alpha TO Counter(s)  
        SET beta TO 0  
        SET gamma TO 0  
        SET idx TO 0  
        WHILE idx < LENGTH(alpha.values)  
            SET delta TO alpha.values[idx]  
            IF (delta - 2 * (delta / 2)) = 1 THEN  
                SET beta TO beta + 1  
            ELSE  
                IF delta <> 0 THEN  
                    IF ((delta / 2) * 2) = delta THEN  
                        SET gamma TO gamma + (1 + 1)  
                    END IF  
                END IF  
            END IF  
            SET idx TO idx + 1  
        END WHILE  
        SET epsilon TO beta + gamma  
        RETURN epsilon  
    END FUNCTION  
END CLASS
