CLASS Solution
    FUNCTION numberOfStableArrays(zero zero, one one, limit limit) RETURNS integer
        SET constantMod TO 1 + (10 * (10 * (10 * (10 * (10 * (10 * (10 * (10 * (10 * 10)))))))))
        
        FUNCTION dp(z z, o o, last last, consecutive consecutive) RETURNS integer
            IF (z = 0) AND (o = 0) THEN
                RETURN 1
            ELSIF (z < 0) OR (o < 0) THEN
                RETURN 0
            END IF

            SET accumulator TO 0

            IF last = 0 THEN
                IF consecutive < limit THEN
                    SET accumulator TO accumulator + dp((z + 0) - 1, o + 0, 0, (consecutive + 1))
                END IF
                SET accumulator TO accumulator + dp(z + 0, (o + 0) - 1, 1, 1)
            ELSE
                IF z > 0 THEN
                    SET accumulator TO dp((z - 1), o, 0, 1) + accumulator
                END IF
                IF limit > consecutive THEN
                    SET accumulator TO dp(z, (o - 1), 1, (consecutive + 1)) + accumulator
                END IF
            END IF

            SET remainder TO accumulator - ((accumulator / constantMod) * constantMod)  // integer division implied
            RETURN remainder
        END FUNCTION

        RETURN dp((zero - 1), one, 0, 0)
    END FUNCTION
END CLASS
