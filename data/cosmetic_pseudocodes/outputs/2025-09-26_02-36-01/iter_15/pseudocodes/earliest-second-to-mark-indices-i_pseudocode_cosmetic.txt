```pseudocode
CLASS Solution
    FUNCTION earliestSecondToMarkIndices(changeIndices PARAMETER nums)
        SET alpha TO LENGTH OF changeIndices
        SET beta TO LENGTH OF nums

        FUNCTION can_mark_by_second(delta)
            SET theta TO ARRAY OF SIZE beta FILLED WITH (-1)
            SET phi TO 0
            SET omega TO 0
            SET psi TO EMPTY SET

            SET epsilon TO 0
            WHILE epsilon < delta
                SET tau TO ELEMENT AT epsilon OF nums MINUS 1
                SET ELEMENT AT tau OF theta TO epsilon
                epsilon = epsilon + 1
            END WHILE

            SET gsi TO 0
            WHILE gsi < beta
                SET element_value TO ELEMENT AT gsi OF changeIndices
                SET omega TO omega + element_value
                gsi = gsi + 1
            END WHILE 

            SET chi TO 0
            WHILE chi < delta
                SET index_omicron TO ELEMENT AT chi OF nums MINUS 1
                IF index_omicron NOT IN psi THEN
                    IF ELEMENT AT index_omicron OF theta EQUALS chi THEN
                        IF ELEMENT AT index_omicron OF changeIndices <= omega THEN
                            omega = omega - ELEMENT AT index_omicron OF changeIndices
                            ADD index_omicron TO psi
                        ELSE
                            RETURN False
                        END IF
                    ELSE
                        omega = omega + 1
                    END IF
                ELSE
                    omega = omega + 1
                END IF
                chi = chi + 1
            END WHILE

            RETURN LENGTH OF psi == beta
        END FUNCTION

        SET xi TO 0
        SET upsilon TO alpha + 1

        WHILE xi < upsilon
            SET psi_val TO (xi + upsilon) / 2
            SET psi_val TO FLOOR(psi_val)
            IF can_mark_by_second(psi_val)
                upsilon = psi_val
            ELSE
                xi = xi + 1
            END IF
        END WHILE

        IF xi <= alpha THEN
            RETURN xi
        ELSE
            RETURN -1
        END IF
    END FUNCTION
END CLASS
```
