CLASS Solution
    FUNCTION doesAliceWin(s)
        DEFINE FUNCTION isVowel(ch)
            SET W TO ("a", "e", "i", "o", "u")
            SET Z TO false
            SET J TO 0
            WHILE J < LENGTH OF W
                IF W[J] = ch THEN
                    SET Z TO true
                    EXIT LOOP
                END IF
                SET J TO J + 1
            END WHILE
            RETURN Z
        END FUNCTION

        SET M TO 0
        SET Q TO 0
        SET X TO 0
        SET Y TO false

        SET R TO 0
        WHILE R < LENGTH OF s
            SET P TO s[R]
            IF isVowel(P) THEN
                SET Q TO Q + 1
            END IF
            SET R TO R + 1
        END WHILE

        SET S TO 0
        WHILE S < LENGTH OF s
            SET D TO s[S]
            IF isVowel(D) THEN
                SET Y TO (NOT Y)
            END IF
            IF (Y = false) AND ((Q MOD 2) = 1) THEN
                SET M TO M + 1
                SET Q TO 0
            ELSE IF Y = true THEN
                SET Q TO Q + 1
            END IF
            SET S TO S + 1
        END WHILE

        IF (Y = true) AND ((Q MOD 2) = 1) THEN
            SET M TO M + 1
        END IF

        RETURN ((M MOD 2) = 1)
    END FUNCTION
END CLASS
