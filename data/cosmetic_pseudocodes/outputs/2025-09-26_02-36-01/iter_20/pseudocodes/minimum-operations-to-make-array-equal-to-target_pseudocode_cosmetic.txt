CLASS Solution  
    FUNCTION minimumOperations(a, b)  
        FUNCTION customAbs(z)  
            IF z < 0 THEN RETURN -z ELSE RETURN z END IF  
        END FUNCTION  
  
        SET L TO LENGTH OF a  
        SET total TO customAbs(b[0] - a[0])  
        SET idx TO 1  
        WHILE idx < L DO  
            SET p TO b[idx] - a[idx]  
            SET q TO b[idx - 1] - a[idx - 1]  
  
            FUNCTION productPositive(m, n)  
                RETURN (m > 0 AND n > 0) OR (m < 0 AND n < 0)  
            END FUNCTION  
  
            IF productPositive(p, q) THEN  
                SET diff TO customAbs(customAbs(p) - customAbs(q))  
                IF diff > 0 THEN  
                    SET total TO total + diff  
                END IF  
            ELSE  
                SET total TO total + customAbs(p)  
            END IF  
  
            SET idx TO idx + 1  
        END WHILE  
        RETURN total  
    END FUNCTION  
END CLASS
