CLASS Solution
    FUNCTION valueAfterKSeconds(x y)
        CONSTANT W = 10^9 + 7

        FUNCTION add modulo(p, q)
            RETURN (p + q) - W * ((p + q) DIV W)
        END FUNCTION

        FUNCTION initialArray(z)
            RETURN array of length z FILLED with 1
        END FUNCTION

        SET arr TO initialArray(x)

        PROCEDURE performIterations(count, limit)
            IF limit <= 0 THEN
                RETURN
            ELSE
                SET idx TO 1
                PROCEDURE innerLoop()
                    IF idx >= count THEN
                        RETURN
                    ELSE
                        arr[idx] = add modulo(arr[idx], arr[idx - 1])
                        idx = idx + 1
                        innerLoop()
                    END IF
                END PROCEDURE
                innerLoop()
                performIterations(count, limit - 1)
            END IF
        END PROCEDURE

        performIterations(x, y)

        RETURN arr[x - 1]
    END FUNCTION
END CLASS
