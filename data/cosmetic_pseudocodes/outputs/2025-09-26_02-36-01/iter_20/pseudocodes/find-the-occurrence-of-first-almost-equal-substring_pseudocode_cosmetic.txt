CLASS Solution
    FUNCTION minStartingIndex WITH PARAMETERS s pattern
        FUNCTION is_almost_equal WITH PARAMETERS substring pattern
            SET alpha TO 0
            SET beta TO 0
            LABEL compare_loop_start
            IF beta GREATER OR EQUAL TO LENGTH(substring) THEN GOTO compare_loop_end
            SET gamma TO substring[beta]
            SET delta TO pattern[beta]
            IF gamma EQUALS delta THEN
                SET epsilon TO 0
            ELSE
                SET epsilon TO 1
            END IF
            SET alpha TO alpha + epsilon
            IF alpha GREATER THAN 1 THEN 
                RETURN False
            END IF
            SET beta TO beta + 1
            GOTO compare_loop_start
            LABEL compare_loop_end
            RETURN True
        END FUNCTION

        SET mu TO LENGTH(pattern)
        SET nu TO 0
        LABEL outer_loop
        IF nu GREATER THAN (LENGTH(s) - mu) THEN GOTO outer_loop_end
        SET xi TO 0
        SET eta TO ""
        LABEL substring_accumulation
        IF xi GREATER OR EQUAL TO mu THEN GOTO substring_end
        SET eta TO eta + s[nu + xi]
        SET xi TO xi + 1
        GOTO substring_accumulation
        LABEL substring_end
        IF is_almost_equal WITH ARGUMENTS eta pattern THEN
            RETURN nu
        END IF
        SET nu TO nu + 1
        GOTO outer_loop
        LABEL outer_loop_end
        RETURN -1
    END FUNCTION
END CLASS
