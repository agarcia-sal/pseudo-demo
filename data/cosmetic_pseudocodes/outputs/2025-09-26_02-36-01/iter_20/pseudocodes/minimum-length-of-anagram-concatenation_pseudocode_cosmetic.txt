CLASS Solution
    FUNCTION minAnagramLength(t)
        PROCEDURE buildSet(x, r)
            IF x IS EMPTY THEN
                RETURN r
            ELSE
                LET k BE FIRST ELEMENT OF x
                LET l BE REST OF x
                IF k IN r THEN
                    RETURN buildSet(l, r)
                ELSE
                    ADD k TO r
                    RETURN buildSet(l, r)
                END IF
            END IF
        END PROCEDURE

        LET z BE buildSet(CONVERT_TO_LIST(t), CREATE_EMPTY_SET())
        LET y BE 0
        PROCEDURE countElements(u)
            IF u IS EMPTY THEN
                RETURN 0
            ELSE
                RETURN 1 + countElements(REST_OF(u))
            END IF
        END PROCEDURE

        SET y TO countElements(LIST_FROM_SET(z))
        RETURN y
    END FUNCTION
END CLASS
