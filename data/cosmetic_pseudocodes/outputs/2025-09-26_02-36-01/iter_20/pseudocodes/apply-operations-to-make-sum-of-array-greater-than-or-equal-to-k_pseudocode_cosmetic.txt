CLASS Solution
    FUNCTION minOperations(k)
        FUNCTION integer_divide(p, q)
            SET remainder_check TO p
            SET count TO 0
            WHILE remainder_check GREATER THAN OR EQUAL TO q
                SET remainder_check TO remainder_check MINUS q
                SET count TO count PLUS 1
            END WHILE
            RETURN count
        END FUNCTION

        FUNCTION integer_sqrt(n)
            SET candidate TO 0
            REPEAT
                SET next_candidate TO candidate PLUS 1
                IF next_candidate MULTIPLIED BY next_candidate GREATER THAN n THEN
                    RETURN candidate
                END IF
                SET candidate TO next_candidate
            UNTIL FALSE
        END FUNCTION

        SET qzrkljmgw TO 999999999999
        SET pdavfhroi TO 1
        WHILE pdavfhroi LESS THAN OR EQUAL TO integer_sqrt(k) PLUS 1
            SET wjhudcnyl TO integer_divide(k PLUS pdavfhroi MINUS 1, pdavfhroi)
            SET fmopkiuyt TO (pdavfhroi MINUS 1) PLUS (wjhudcnyl MINUS 1)
            IF fmopkiuyt LESS THAN qzrkljmgw THEN
                SET qzrkljmgw TO fmopkiuyt
            END IF
            SET pdavfhroi TO pdavfhroi PLUS 1
        END WHILE
        RETURN qzrkljmgw
    END FUNCTION
END CLASS
