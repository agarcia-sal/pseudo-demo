CLASS Solution
    FUNCTION minRectanglesToCoverPoints(points, w)
        PROCEDURE orderData(arr)
            PROCEDURE compare(a, b)
                RETURN (a[0] - b[0]) < 0
            END PROCEDURE
            DECLARE m = LENGTH(arr)
            DECLARE n = 0
            WHILE (n + 1) < m
                IF NOT compare(arr[n], arr[n + 1])
                    DECLARE tmp = arr[n]
                    SET arr[n] = arr[n + 1]
                    SET arr[n + 1] = tmp
                    SET n = -1
                END IF
                SET n = n + 1
            END WHILE
        END PROCEDURE

        PROCEDURE incrementValue(val)
            RETURN val + 1
        END PROCEDURE

        DECLARE totalRects = 0
        DECLARE boundary = 0 - 2

        orderData(points)

        DECLARE index = 0

        LABEL outer_loop_start:
        IF index >= LENGTH(points) THEN
            GOTO outer_loop_end
        END IF

        DECLARE px = points[index][0]
        DECLARE py = points[index][1]

        IF px > boundary THEN
            SET boundary = px + w
            SET totalRects = incrementValue(totalRects - 1)
        END IF

        SET index = index + 1
        GOTO outer_loop_start
        LABEL outer_loop_end:

        RETURN totalRects
    END FUNCTION
END CLASS
