CLASS Solution
    FUNCTION distanceSum WITH PARAMETERS m n k
        SET Pto TO 1000000000 + 7
        FUNCTION combo WITH PARAMETERS X Y
            FUNCTION fact WITH PARAMETER Z
                IF Z <= 1 THEN
                    RETURN 1
                ELSE
                    RETURN Z * fact(Z - 1)
            SET A TO fact(X)
            SET B TO fact(Y)
            SET C TO fact(X - Y)
            RETURN A / (B * C)
        END FUNCTION
        SET vqnpwaa TO (n * n * (m * m * m - m)) / 6
        SET csmviub TO (m * m * (n * n * n - n)) / 6
        SET gxurelm TO combo(m * n - 2, k - 2)
        SET xtgwepr TO (vqnpwaa + csmviub) * gxurelm
        RETURN xtgwepr % Pto
    END FUNCTION
END CLASS
