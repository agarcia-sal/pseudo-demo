CLASS Solution
    FUNCTION countSubstrings(s c)
        FUNCTION computeOccurrences(array key, element val)
            SET tally TO 0
            SET idx TO 0
            REPEAT
                IF idx IS EQUAL TO LENGTH OF key THEN
                    EXIT REPEAT
                END IF
                IF key[idx] IS EQUAL TO val THEN
                    SET tally TO tally PLUS 1
                END IF
                SET idx TO idx PLUS 1
            UNTIL FALSE
            RETURN tally
        END FUNCTION

        SET alpha TO 0
        SET i TO 0
        SET len_s TO LENGTH OF s
        WHILE TRUE
            IF NOT (i < len_s) THEN
                BREAK
            END IF
            IF s[i] = c THEN
                SET alpha TO alpha + 1
            END IF
            SET i TO i + 1
        END WHILE

        FUNCTION multiply(a b)
            RETURN a * b
        END FUNCTION

        FUNCTION add(a b)
            RETURN a + b
        END FUNCTION

        FUNCTION divideByTwo(x)
            RETURN x / 2
        END FUNCTION

        SET intermediate_value TO add(alpha 1)
        SET product TO multiply(alpha intermediate_value)
        SET outcome TO divideByTwo(product)
        RETURN outcome
    END FUNCTION
END CLASS
