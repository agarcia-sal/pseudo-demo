CLASS Solution
	FUNCTION clearStars(s)
		DEFINE FUNCTION deleteLastElement(arr)
			IF LENGTH(arr) EQUALS 0
				RETURN arr
			ELSE
				RETURN SUBARRAY(arr, 0, LENGTH(arr) - 1)
			END IF
		END FUNCTION

		SET alpha TO []
		SET omega TO 0
		WHILE omega < LENGTH(s)
			SET chi TO SUBSTRING(s, omega, omega + 1)
			IF NOT (chi EQUALS "*")
				SET alpha TO alpha + [chi]
			ELSE
				SET alpha TO deleteLastElement(alpha)
			END IF
			SET omega TO omega + 1
		END WHILE

		DEFINE FUNCTION concatenateElements(elements)
			SET result TO ""
			SET i TO 0
			REPEAT
				IF i >= LENGTH(elements)
					BREAK
				END IF
				SET result TO result + elements[i]
				SET i TO i + 1
			UNTIL FALSE
			RETURN result
		END FUNCTION

		RETURN concatenateElements(alpha)
	END FUNCTION
END CLASS
