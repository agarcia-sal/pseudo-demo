CLASS Solution
    FUNCTION minOperations(nums)
        SET delta TO LENGTH(nums)
        SET omega TO 0
        SET psi TO 0
        WHILE psi < delta - 2
            IF nums[psi] = 0 THEN
                CALL flipBit(nums, psi)
                CALL flipBit(nums, psi + 1)
                CALL flipBit(nums, psi + 2)
                SET omega TO omega + 1
            END IF
            SET psi TO psi + 1
        END WHILE

        IF zeroCheck(nums[delta - 1]) OR zeroCheck(nums[delta - 2]) THEN
            RETURN -1
        END IF

        RETURN omega
    END FUNCTION

    FUNCTION flipBit(arr, pos)
        SET arr[pos] TO (1 - arr[pos])
    END FUNCTION

    FUNCTION zeroCheck(val)
        RETURN (val = 0)
    END FUNCTION
END CLASS
