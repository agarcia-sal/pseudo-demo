CLASS Solution  
    FUNCTION minOperations(nums)  
        DEFINE FUNCTION modTwo(value)  
            RETURN value - (value / 2) * 2  
        END FUNCTION  

        SET alpha TO 0  
        SET beta TO 0  

        SET gamma TO 0  
        WHILE gamma < LENGTH(nums)  
            SET delta TO nums[gamma]  
            SET epsilon TO 0  

            IF modTwo(beta) = 0 THEN  
                SET epsilon TO delta  
            ELSE  
                SET epsilon TO 1 - delta  
            END IF  

            IF epsilon = 0 THEN  
                SET alpha TO alpha + 1  
                SET beta TO beta + 1  
            END IF  

            SET gamma TO gamma + 1  
        END WHILE  

        RETURN alpha  
    END FUNCTION  
END CLASS
