CLASS Solution
    FUNCTION numberOfStableArrays(zero, one, limit)
        
        CONST CONST_A = (10 * 10 * 10 * 10 * 10 * 10 * 10 * 10 * 10) + 1
        
        FUNCTION delta(alpha, beta, gamma, deltaVal)
            
            FUNCTION modAdd(x, y)
                DECLARE res
                LET res = ((x MOD CONST_A) + (y MOD CONST_A)) MOD CONST_A
                RETURN res
            END FUNCTION
            
            IF (alpha = 0) AND (beta = 0) THEN
                RETURN 1
            END IF
            
            IF (alpha < 0) OR (beta < 0) THEN
                RETURN 0
            END IF
            
            DECLARE accumulator
            LET accumulator = 0
            
            IF (gamma <> 0) OR (deltaVal < limit) THEN
                DECLARE temp1
                LET temp1 = delta(alpha - 1,
                                  beta,
                                  0,
                                  IF (gamma = 0) THEN deltaVal + 1 ELSE 1 END IF)
                LET accumulator = modAdd(accumulator, temp1)
            END IF
            
            IF (gamma <> 1) OR (deltaVal < limit) THEN
                DECLARE temp2
                LET temp2 = delta(alpha,
                                  beta - 1,
                                  1,
                                  IF (gamma = 1) THEN deltaVal + 1 ELSE 1 END IF)
                LET accumulator = modAdd(accumulator, temp2)
            END IF
            
            RETURN accumulator
        END FUNCTION
        
        RETURN delta(zero, one, -1, 0)
    END FUNCTION
END CLASS
