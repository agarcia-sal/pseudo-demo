CLASS Solution  
    FUNCTION minMoves(rooks)  
        FUNCTION computeAbsolute(value)  
            IF value < 0 THEN  
                RETURN 0 - value  
            ELSE  
                RETURN value  
            END IF  
        END FUNCTION  
  
        FUNCTION sortByFirstElement(arr)  
            IF LENGTH(arr) <= 1 THEN  
                RETURN arr  
            ELSE  
                SET pivot TO arr[0]  
                SET left TO []  
                SET right TO []  
                SET i TO 1  
                WHILE i < LENGTH(arr) DO  
                    IF arr[i][0] < pivot[0] THEN  
                        APPEND arr[i] TO left  
                    ELSE  
                        APPEND arr[i] TO right  
                    END IF  
                    SET i TO i + 1  
                END WHILE  
                RETURN CONCATENATE(sortByFirstElement(left), [pivot], sortByFirstElement(right))  
            END IF  
        END FUNCTION  
  
        FUNCTION sortBySecondElement(arr)  
            IF LENGTH(arr) <= 1 THEN  
                RETURN arr  
            ELSE  
                SET pivot TO arr[0]  
                SET left TO []  
                SET right TO []  
                SET j TO 1  
                WHILE j < LENGTH(arr) DO  
                    IF arr[j][1] < pivot[1] THEN  
                        APPEND arr[j] TO left  
                    ELSE  
                        APPEND arr[j] TO right  
                    END IF  
                    SET j TO j + 1  
                END WHILE  
                RETURN CONCATENATE(sortBySecondElement(left), [pivot], sortBySecondElement(right))  
            END IF  
        END FUNCTION  
  
        SET totalRows TO LENGTH(rooks)  
        SET sortedRowWise TO sortByFirstElement(rooks)  
        SET sortedColWise TO sortBySecondElement(rooks)  
  
        SET sumRowDiff TO 0  
        SET idx1 TO 0  
        WHILE idx1 < totalRows DO  
            SET delta TO computeAbsolute(sortedRowWise[idx1][0] - idx1)  
            SET sumRowDiff TO sumRowDiff + delta  
            SET idx1 TO idx1 + 1  
        END WHILE  
  
        SET sumColDiff TO 0  
        SET idx2 TO 0  
        WHILE idx2 < totalRows DO  
            SET delta TO computeAbsolute(sortedColWise[idx2][1] - idx2)  
            SET sumColDiff TO sumColDiff + delta  
            SET idx2 TO idx2 + 1  
        END WHILE  
  
        RETURN sumRowDiff + sumColDiff  
    END FUNCTION  
END CLASS
