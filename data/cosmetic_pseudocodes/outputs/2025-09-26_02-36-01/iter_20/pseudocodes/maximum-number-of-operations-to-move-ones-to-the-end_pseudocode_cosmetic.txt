CLASS Solution
    FUNCTION maxOperations(s)
        FUNCTION helperX(y)
            RETURN y == 1
        END FUNCTION

        SET totalA TO 0
        SET trackerB TO 0
        SET posC TO 0

        WHILE posC < LENGTH(s)
            SET elemD TO s[posC]
            IF NOT helperX(elemD)
                IF posC != 0
                    SET prevE TO s[posC - 1]
                    IF helperX(prevE)
                        SET totalA TO totalA + trackerB
                    END IF
                END IF
            ELSE
                SET trackerB TO trackerB + 1
            END IF
            SET posC TO posC + 1
        END WHILE

        RETURN totalA
    END FUNCTION
END CLASS
