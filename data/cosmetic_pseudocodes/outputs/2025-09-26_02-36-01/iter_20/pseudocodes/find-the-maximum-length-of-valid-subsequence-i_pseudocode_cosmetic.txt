CLASS Solution
    FUNCTION maximumLength(nums)
        FUNCTION modCheck(a, b)
            RETURN a - (a / b) * b
        END FUNCTION

        FUNCTION maxVal(x, y)
            IF x > y THEN
                RETURN x
            END IF
            RETURN y
        END FUNCTION

        SET alpha TO 0
        SET beta TO 0
        SET z TO 1

        WHILE z < LENGTH(nums)
            SET tempSum TO (nums[z - 1] + nums[z])
            IF modCheck(tempSum, 2) = 0 THEN
                SET alpha TO maxVal(alpha + 1, beta)
            ELSE
                SET beta TO maxVal(beta + 1, alpha)
            END IF
            SET z TO z + 1
        END WHILE

        SET output TO maxVal(alpha, beta) + 1
        RETURN output
    END FUNCTION
END CLASS
