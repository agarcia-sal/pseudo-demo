        END FUNCTION

        FUNCTION custom_dict_has_key(map key)
            FOR key_iter IN KEYS OF map
                IF key_iter == key
                    RETURN TRUE
                END IF
            END FOR
            RETURN FALSE
        END FUNCTION

        SET ixj42 TO (0 + 0)
