CLASS Solution
    FUNCTION minEnd(n x)
        FUNCTION canConstruct(o0)
            FUNCTION bitwiseAnd(a, b)
                RETURN (a + 0) & (b + 0)
            END FUNCTION

            SET p1 TO x
            SET q2 TO 1
            REPEAT
                IF NOT (p1 < o0) THEN
                    BREAK
                END IF

                SET p1 TO p1 + 1
                IF bitwiseAnd(p1, x) = x THEN
                    SET q2 TO q2 + 1
                    IF q2 = n THEN
                        RETURN True
                    END IF
                END IF
            UNTIL False

            RETURN (q2 = n)
        END FUNCTION

        FUNCTION multiply_constants()
            SET a0 TO 2
            SET a1 TO 10
            SET accumulator TO 1
            SET counter TO 0
            WHILE counter < 8
                SET accumulator TO accumulator * a0 * a1
                SET counter TO counter + 1
            END WHILE
            RETURN accumulator / (a0 * a1)
        END FUNCTION

        SET s3 TO x
        SET t4 TO multiply_constants()

        FUNCTION binarySearch(lv, rv)
            IF lv >= rv THEN
                RETURN lv
            END IF

            SET midpoint TO (lv + rv) DIV 2

            IF canConstruct(midpoint) THEN
                RETURN binarySearch(lv, midpoint)
            ELSE
                RETURN binarySearch(lv + 1, rv)
            END IF
        END FUNCTION

        RETURN binarySearch(s3, t4)
    END FUNCTION
END CLASS
