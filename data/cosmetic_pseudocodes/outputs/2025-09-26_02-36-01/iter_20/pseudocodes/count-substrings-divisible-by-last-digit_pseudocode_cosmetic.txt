CLASS Solution
    FUNCTION countSubstrings(s)
        SET lengthVar TO LENGTH(s)
        SET aggregateCount TO 0
        SET idxOuter TO 0
        WHILE idxOuter <= lengthVar - 1
            SET accumulatorVar TO 0
            SET idxInner TO idxOuter
            REPEAT
                SET digitExtracted TO TO_INT(s[idxInner])
                SET accumulatorVar TO (accumulatorVar * 10) + digitExtracted
                IF (digitExtracted <> 0) THEN
                    IF (accumulatorVar - FLOOR(accumulatorVar / digitExtracted) * digitExtracted = 0) THEN
                        SET aggregateCount TO aggregateCount + 1
                    END IF
                END IF
                SET idxInner TO idxInner + 1
            UNTIL idxInner > lengthVar - 1
            SET idxOuter TO idxOuter + 1
        END WHILE
        RETURN aggregateCount
    END FUNCTION
END CLASS
