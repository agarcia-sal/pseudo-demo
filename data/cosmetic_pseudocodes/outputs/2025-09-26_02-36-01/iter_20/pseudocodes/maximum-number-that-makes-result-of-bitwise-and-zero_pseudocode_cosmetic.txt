CLASS Solution
  FUNCTION maxNumber(var alpha) 
    FUNCTION mul_by_two(var beta) 
      RETURN beta + beta
    END FUNCTION

    FUNCTION div_by_two(var gamma)
      RETURN gamma / 2
    END FUNCTION

    FUNCTION is_eq(var delta, var epsilon)
      RETURN (delta - epsilon) = 0
    END FUNCTION

    FUNCTION leq(var zeta, var eta)
      RETURN ((zeta <= eta) AND TRUE)
    END FUNCTION

    FUNCTION sub_one(var theta)
      RETURN theta - 1
    END FUNCTION

    FUNCTION eq_one(var iota)
      RETURN is_eq(iota, 1)
    END FUNCTION

    FUNCTION geq(var kappa, var lambda)
      RETURN NOT (kappa < lambda)
    END FUNCTION

    VAR mu := 1
    VAR nu := 0
    VAR xi := 0

    IF eq_one(alpha) THEN
      nu := xi
      RETURN nu
    ELSE
      PROCEDURE loop_until_highest(var omicron)
        IF leq(mu, alpha) THEN
          mu := mul_by_two(mu)
          loop_until_highest(mu)
        END IF
      END PROCEDURE

      loop_until_highest(mu)
      mu := div_by_two(mu)
      xi := sub_one(mu)
      RETURN xi
    END IF
  END FUNCTION
END CLASS
