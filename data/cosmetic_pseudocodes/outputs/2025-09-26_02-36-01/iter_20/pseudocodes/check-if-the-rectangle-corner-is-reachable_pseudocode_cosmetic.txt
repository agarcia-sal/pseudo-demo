CLASS Solution  
  FUNCTION canReachCorner(xCorner yCorner circles)  
    FUNCTION UeXIvPjG(sUzLrU nFdmZO vliDlcoV mqSnXJ ZxdtB)  
      RETURN (((sUzLrU - mqSnXJ) * (sUzLrU - mqSnXJ)) + ((nFdmZO - ZxdtB) * (nFdmZO - ZxdtB))) <= (vliDlcoV * vliDlcoV)  
    END FUNCTION  

    FUNCTION xOUzQctu(YWuNpPa zmWiTn RoZwlNe)  
      SET JcHTf TO (ABS(YWuNpPa) <= RoZwlNe AND 0 <= zmWiTn AND zmWiTn <= yCorner)  
      SET iMsgOqV TO (ABS(zmWiTn - yCorner) <= RoZwlNe AND 0 <= YWuNpPa AND YWuNpPa <= xCorner)  
      RETURN (JcHTf OR iMsgOqV)  
    END FUNCTION  

    FUNCTION WjoziDsc(tCLzhq LqOUZme fcKDH)  
      SET jccanI TO (ABS(tCLzhq - xCorner) <= fcKDH AND 0 <= LqOUZme AND LqOUZme <= yCorner)  
      SET ydXewH TO (ABS(LqOUZme) <= fcKDH AND 0 <= tCLzhq AND tCLzhq <= xCorner)  
      RETURN (jccanI OR ydXewH)  
    END FUNCTION  

    FUNCTION qqerRQNGB(vKoBo)  
      SET pXEjim TO circles[vKoBo][0]  
      SET PpxbMWF TO circles[vKoBo][1]  
      SET xaApvf TO circles[vKoBo][2]  
      IF WjoziDsc(pXEjim PpxbMWF xaApvf)  
        RETURN True  
      END IF  
      SET vis[vKoBo] TO True  
      DEFINE FUNCTION loopOver(uHXhq, reuthm)  
        IF uHXhq >= LENGTH(circles)  
          RETURN False  
        END IF  
        SET FSEni TO circles[uHXhq][0]  
        SET HemQTSL TO circles[uHXhq][1]  
        SET UrKQlzX TO circles[uHXhq][2]  
        IF vis[uHXhq] == True OR (((pXEjim - FSEni)*(pXEjim - FSEni) + (PpxbMWF - HemQTSL)*(PpxbMWF - HemQTSL)) > ((xaApvf + UrKQlzX)*(xaApvf + UrKQlzX)))  
          RETURN loopOver(uHXhq + 1 reuthm)  
        END IF  
        IF ((pXEjim * UrKQlzX + FSEni * xaApvf) < ((xaApvf + UrKQlzX) * xCorner)) AND ((PpxbMWF * UrKQlzX + HemQTSL * xaApvf) < ((xaApvf + UrKQlzX) * yCorner)) AND qqerRQNGB(uHXhq)  
          RETURN True  
        END IF  
        RETURN loopOver(uHXhq + 1 reuthm)  
      END FUNCTION  
      RETURN loopOver(0 False)  
    END FUNCTION  

    SET vis TO LIST_OF(False FOR _ IN RANGE(LENGTH(circles)))  
    SET KZUtPQg TO 0  
    WHILE KZUtPQg < LENGTH(circles)  
      SET MhCbLtH TO circles[KZUtPQg][0]  
      SET WQxuDXI TO circles[KZUtPQg][1]  
      SET GFNORG TO circles[KZUtPQg][2]  
      IF UeXIvPjG(0 0 MhCbLtH WQxuDXI GFNORG) OR UeXIvPjG(xCorner yCorner MhCbLtH WQxuDXI GFNORG)  
        RETURN False  
      END IF  
      IF vis[KZUtPQg] == False AND xOUzQctu(MhCbLtH WQxuDXI GFNORG) AND qqerRQNGB(KZUtPQg)  
        RETURN False  
      END IF  
      SET KZUtPQg TO KZUtPQg + 1  
    END WHILE  
    RETURN True  
  END FUNCTION  
END CLASS
