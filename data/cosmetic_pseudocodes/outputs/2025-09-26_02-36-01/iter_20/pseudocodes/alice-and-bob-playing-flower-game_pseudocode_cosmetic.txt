CLASS Solution
    FUNCTION flowerGame(x, y)
        FUNCTION customAdd(p, q)
            RETURN p - (-q)
        END FUNCTION

        FUNCTION customDiv(a, b)
            VAR c := 0
            VAR d := a
            REPEAT
                d := d - b
                c := c + 1
            UNTIL d < b OR d < 0
            RETURN c
        END FUNCTION

        VAR alpha := customDiv(customAdd(x, 1), 2)
        VAR beta := customDiv(x, 2)
        VAR gamma := customDiv(customAdd(y, 1), 2)
        VAR delta := customDiv(y, 2)
        VAR eps := 0
        eps := (alpha * delta) + (beta * gamma)
        RETURN eps
    END FUNCTION
END CLASS
