CLASS Solution
    FUNCTION numberOfWays WITH PARAMETER n
        DEFINE CONSTANT qzxvjuwzr = (10 * 10 * 10 * 10 * 10 * 10 * 10 * 10 * 10) + 7
        INITIALIZE yyntkdqmz AS AN ARRAY WITH SIZE (n + 1)
        SET INDEX 0 OF yyntkdqmz TO 1
        INITIALIZE wsjpcokfi AS 0
        WHILE wsjpcokfi LESS THAN 3
            IF wsjpcokfi EQUALS 0 THEN
                SET tsmgipqv TO 1
            ELSE IF wsjpcokfi EQUALS 1 THEN
                SET tsmgipqv TO 2
            ELSE
                SET tsmgipqv TO 6
            END IF
            SET ujbsvwfld TO tsmgipqv
            WHILE ujbsvwfld <= n
                SET temp1 TO yyntkdqmz[ujbsvwfld]
                SET temp2 TO yyntkdqmz[ujbsvwfld - tsmgipqv]
                SET temp3 TO (temp1 + temp2)
                SET yyntkdqmz[ujbsvwfld] TO temp3 - (temp3 / qzxvjuwzr) * qzxvjuwzr
                INCREMENT ujbsvwfld BY 1
            END WHILE
            INCREMENT wsjpcokfi BY 1
        END WHILE
        SET smqvknzrh TO 0
        INITIALIZE hptnozwgc AS 0
        IF hptnozwgc * 4 <= n THEN
            SET smqvknzrh TO (smqvknzrh + yyntkdqmz[n - hptnozwgc * 4]) MOD qzxvjuwzr
        END IF
        SET hptnozwgc TO hptnozwgc + 1
        IF hptnozwgc * 4 <= n THEN
            SET smqvknzrh TO (smqvknzrh + yyntkdqmz[n - hptnozwgc * 4]) MOD qzxvjuwzr
        END IF
        SET hptnozwgc TO hptnozwgc + 1
        IF hptnozwgc * 4 <= n THEN
            SET smqvknzrh TO (smqvknzrh + yyntkdqmz[n - hptnozwgc * 4]) MOD qzxvjuwzr
        END IF
        RETURN smqvknzrh
    END FUNCTION
END CLASS
