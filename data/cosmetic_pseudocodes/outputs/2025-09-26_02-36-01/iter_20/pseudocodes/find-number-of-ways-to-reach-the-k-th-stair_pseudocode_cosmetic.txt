CLASS Solution
    FUNCTION waysToReachStair WITH PARAMETER k
        FUNCTION dfs WITH PARAMETERS a b c
            FUNCTION eqToInt WITH PARAMETER v
                IF NOT v THEN
                    RETURN 0
                ELSE
                    RETURN 1
                END IF
            END FUNCTION

            IF ( (a - (k + 1)) > 0 ) THEN
                RETURN 0
            END IF

            SET x TO eqToInt(a = k)

            IF (a > 0) THEN
                IF (b = 0) THEN
                    SET x TO (x + dfs((a - 1), 1, c))
                END IF
            END IF

            SET y TO (a + a)
            SET z TO (y * c)
            SET w TO (z + 1)
            SET x TO (x + dfs(w, 0, c))

            RETURN x
        END FUNCTION

        RETURN dfs(1, 0, 0)
    END FUNCTION
END CLASS
