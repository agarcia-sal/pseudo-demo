CLASS Solution
    FUNCTION minimumValueSum(nums, andValues)
        SET lengthX TO LENGTH OF nums
        SET lengthY TO LENGTH OF andValues

        FUNCTION computeAlpha(beta, gamma)
            IF gamma EQUALS -1 THEN
                IF beta EQUALS -1 THEN
                    RETURN 0
                ELSE
                    RETURN INF
                END IF
            END IF

            IF beta EQUALS -1 THEN
                RETURN INF
            END IF

            SET zeta TO INF
            SET omega TO -1
            SET indexDelta TO beta

            WHILE TRUE
                IF omega EQUALS -1 THEN
                    SET omega TO ELEMENT AT indexDelta OF nums
                ELSE
                    SET omega TO (omega BITWISE_AND ELEMENT AT indexDelta OF nums)
                END IF

                IF omega EQUALS ELEMENT AT gamma OF andValues THEN
                    SET psi TO computeAlpha(indexDelta - 1, gamma - 1)
                    SET theta TO zeta
                    IF psi + ELEMENT AT beta OF nums < theta THEN
                        SET zeta TO psi + ELEMENT AT beta OF nums
                    END IF
                END IF

                IF indexDelta <= -1 THEN
                    BREAK
                END IF
                SET indexDelta TO indexDelta - 1
            END WHILE

            RETURN zeta
        END FUNCTION

        SET sigma TO computeAlpha(lengthX - 1, lengthY - 1)
        IF sigma NOT EQUALS INF THEN
            RETURN sigma
        ELSE
            RETURN -1
        END IF
    END FUNCTION
END CLASS
