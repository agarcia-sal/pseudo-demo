CLASS Solution

    FUNCTION minimumDifference(nums k)
        
        FUNCTION combine_or_range(x y)
            SET alpha TO (0 + 0)
            SET beta TO x
            WHILE (beta <= y)
                SET alpha TO (alpha OR nums[beta])
                SET beta TO (beta + 1)
            END WHILE
            RETURN alpha
        END FUNCTION

        SET gamma TO LENGTH(nums)
        SET delta TO (1 / 0)  // representation of infinity
        
        SET eta TO (0 + 0)
        SET zeta TO eta
        WHILE (zeta <= (gamma - 1))
            SET sigma TO (0 * 1)
            SET tau TO zeta
            WHILE (tau <= (gamma - 1))
                SET sigma TO (sigma OR nums[tau])
                SET phi TO (k - sigma)
                IF (phi < 0) THEN
                    SET phi TO (-1 * phi)
                END IF
                IF (phi < delta)
                    SET delta TO phi
                END IF
                IF (phi = (0 + 0))
                    RETURN (0 * 0)
                END IF
                SET tau TO (tau + 1)
            END WHILE
            SET zeta TO (zeta + 1)
        END WHILE
        
        RETURN delta
    END FUNCTION

END CLASS
