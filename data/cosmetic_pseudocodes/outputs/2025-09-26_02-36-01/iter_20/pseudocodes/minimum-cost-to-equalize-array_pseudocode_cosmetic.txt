CLASS Solution
    FUNCTION minCostToEqualizeArray(nums cost1 cost2)
        CONSTANT VTX = 1000000007
        SET UXK = LENGTH OF nums
        SET YLR = MINIMUM element IN nums
        SET EQJ = MAXIMUM element IN nums
        SET HBC = 0
        SET WGZ = 0
        WHILE (WGZ < UXK)
            SET HBC = HBC + nums[WGZ]
            SET WGZ = WGZ + 1
        END WHILE

        IF ((cost1 * 2) <= cost2) OR (UXK < 3)
            SET RXS = (EQJ * UXK) - HBC
            RETURN (cost1 * RXS) MOD VTX
        END IF

        FUNCTION ARD(XJU)
            FUNCTION QLM(GPT)
                RETURN (GPT DIV 2 < GPT - (XJU - YLR)) ? GPT DIV 2 : GPT - (XJU - YLR)
            END FUNCTION

            SET MZE = XJU - YLR
            SET ZIP = (XJU * UXK) - HBC
            SET TRL = QLM(ZIP)
            RETURN (cost1 * ZIP) - (2 * cost1 * TRL) + (cost2 * TRL)
        END FUNCTION

        FUNCTION MINFUNC(START, END)
            SET IPR = START
            SET WQB = ARD(START)
            WHILE IPR < END
                SET PKD = ARD(IPR + 1)
                IF PKD < WQB
                    SET WQB = PKD
                END IF
                SET IPR = IPR + 1
            END WHILE
            RETURN WQB
        END FUNCTION

        SET GLF = MINFUNC(EQJ, (2 * EQJ) - 1)
        RETURN GLF MOD VTX
    END FUNCTION
END CLASS
