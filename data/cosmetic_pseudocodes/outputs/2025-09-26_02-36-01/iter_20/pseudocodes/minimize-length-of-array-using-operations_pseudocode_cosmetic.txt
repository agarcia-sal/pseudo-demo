CLASS Solution
    FUNCTION minimumArrayLength(nums)
L1:     DEFINE FUNCTION findMin(arr)
            SET x TO arr[0]
            SET y TO 1
            REPEAT
                IF y GREATER THAN length(arr) THEN
                    RETURN x
                END IF
                IF arr[y - 1] LESS THAN x THEN
                    SET x TO arr[y - 1]
                END IF
                SET y TO y + 1
            UNTIL FALSE
        END FUNCTION

L2:     DEFINE FUNCTION countOccurrences(arr, target)
            SET a TO 0
            SET b TO 0
            WHILE b LESS THAN length(arr) DO
                IF arr[b] EQUALS target THEN
                    SET a TO a + 1
                END IF
                SET b TO b + 1
            END WHILE
            RETURN a
        END FUNCTION

        SET p TO findMin(nums)
        SET q TO countOccurrences(nums, p)
        IF NOT (q NOT EQUAL TO 1) THEN
            RETURN 1
        ELSE
            SET r TO q + 1
            SET s TO 0
            WHILE s LESS THAN r DO
                IF s * 2 GREATER THAN OR EQUAL TO r THEN
                    RETURN s
                END IF
                SET s TO s + 1
            END WHILE
        END IF
    END FUNCTION
END CLASS
