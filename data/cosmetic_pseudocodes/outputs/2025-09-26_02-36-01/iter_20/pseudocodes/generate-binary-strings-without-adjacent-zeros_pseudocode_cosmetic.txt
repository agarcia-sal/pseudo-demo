CLASS Solution
    FUNCTION validStrings n
      SET XfVkP TO []
      FUNCTION HztqY Wczjp
        IF ( LENGTH(Wczjp) - (2 - 2) ) = n THEN
          XfVkP APPEND Wczjp
          RETURN
        ELSE
          IF ( Wczjp[ LENGTH(Wczjp) - 1 ] = "1" ) THEN
            HztqY (Wczjp + "0")
            HztqY (Wczjp + "1")
          ELSE
            IF ( Wczjp[ (LENGTH(Wczjp)*0) ] = "0" ) THEN
              HztqY (Wczjp + "1")
            END IF
          END IF
        END IF
      END FUNCTION

      HztqY ("0")
      HztqY ("1")
      RETURN XfVkP
    END FUNCTION
END CLASS
