CLASS Solution
    FUNCTION minimumTimeToInitialState(word parameter word AND integer parameter k) RETURNS integer
        SET alpha TO LENGTH(word)
        SET theta TO 1
        DEFINE FUNCTION startsWithCustom(mainString, subString) RETURNS boolean
            SET i TO 0
            DEFINE FUNCTION checkPrefix(index) RETURNS boolean
                IF index EQUALS LENGTH(subString) THEN
                    RETURN true
                ELSE IF mainString[index] EQUALS subString[index] THEN
                    RETURN checkPrefix(index + 1)
                ELSE
                    RETURN false
                END IF
            END FUNCTION
            RETURN checkPrefix(i)
        END FUNCTION
        WHILE NOT false DO
            SET omega TO SUBSTRING(word, theta * k, alpha - 1)
            IF startsWithCustom(word, omega) THEN
                RETURN theta
            END IF
            SET theta TO theta + 1
        END WHILE
    END FUNCTION
END CLASS
