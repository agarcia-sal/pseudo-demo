CLASS Solution
   FUNCTION countCompleteDayPairs WITH parameter hours
      FUNCTION modulus24 WITH parameter value
         SET acc TO value
         WHILE acc IS GREATER THAN OR EQUAL TO 24
            SET acc TO acc MINUS 24
         END WHILE
         WHILE acc IS LESS THAN 0
            SET acc TO acc PLUS 24
         END WHILE
         RETURN acc
      END FUNCTION

      SET storage_map TO an empty mapping defaulting zero
      SET total_pairs TO 0
      SET i7jk TO 0

      WHILE i7jk LESS THAN LENGTH OF hours
         SET current_val TO hours AT i7jk
         SET r4xp TO modulus24(current_val)
         SET cmp6 TO modulus24(24 MINUS r4xp)
         SET total_pairs TO total_pairs PLUS (IF cmp6 EXISTS IN storage_map THEN storage_map[cmp6] ELSE 0)
         IF r4xp EXISTS IN storage_map THEN
            SET storage_map[r4xp] TO storage_map[r4xp] PLUS 1
         ELSE
            SET storage_map[r4xp] TO 1
         END IF
         SET i7jk TO i7jk PLUS 1
      END WHILE

      RETURN total_pairs
   END FUNCTION
END CLASS
