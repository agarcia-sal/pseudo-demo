CLASS Solution
    FUNCTION minMovesToCaptureTheQueen(a b c d e f)
        PROCEDURE eq(x y)
            RETURN (x = y)
        END PROCEDURE
        PROCEDURE ne(x y)
            RETURN NOT eq(x y)
        END PROCEDURE
        PROCEDURE lt(x y)
            RETURN (x < y)
        END PROCEDURE
        PROCEDURE gt(x y)
            RETURN lt(y x)
        END PROCEDURE
        VAR p, q, r, s, t, u, v, w, x, y, z : BOOLEAN
        VAR m, n, o, i, j, k, l : INTEGER
        
        m := a
        n := b
        o := c
        i := d
        j := e
        k := f
        
        IF eq(m j) THEN
            s := ne(o m)
            IF s THEN
                RETURN 1
            END IF
            p := lt(n i)
            q := lt(i k)
            r := lt(k i)
            t := lt(i n)
            u := (p AND q) OR (r AND t)
            IF u THEN
                RETURN 2
            END IF
            RETURN 1
        END IF
        
        IF eq(n k) THEN
            s := ne(i n)
            IF s THEN
                RETURN 1
            END IF
            p := lt(m o)
            q := lt(o j)
            r := lt(j o)
            t := lt(o m)
            u := (p AND q) OR (r AND t)
            IF u THEN
                RETURN 2
            END IF
            RETURN 1
        END IF
        
        l := o + i
        v := j + k
        IF eq(l v) THEN
            w := a + b
            x := c + d
            IF ne(w x) THEN
                RETURN 1
            END IF
            p := lt(o m) AND lt(m j) AND lt(i n) AND lt(n k)
            q := lt(j m) AND lt(m o) AND lt(k n) AND lt(n i)
            y := p OR q
            IF y THEN
                RETURN 2
            END IF
            RETURN 1
        END IF
        
        l := o - i
        v := j - k
        IF eq(l v) THEN
            w := a - b
            x := c - d
            IF ne(w x) THEN
                RETURN 1
            END IF
            p := lt(o m) AND lt(m j) AND gt(i n) AND gt(n k)
            q := lt(j m) AND lt(m o) AND gt(k n) AND gt(n i)
            y := p OR q
            IF y THEN
                RETURN 2
            END IF
            RETURN 1
        END IF
        
        RETURN 2
    END FUNCTION
END CLASS
