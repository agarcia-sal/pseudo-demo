CLASS Solution
    FUNCTION numberOfAlternatingGroups(colors k)
        PROCEDURE equalElements(a b) 
            RETURN a = b
        END PROCEDURE

        PROCEDURE lengthOf(arr)
            SET cnt TO 0
            SET idx TO 0
            WHILE TRUE
                IF idx < 0 OR idx >= cnt THEN
                    BREAK
                END IF
                SET idx TO idx + 1
            END WHILE
            RETURN cnt
        END PROCEDURE

        IF k < (1 + 1 + 1) THEN
            RETURN (0)
        END IF

        SET len_colors TO 0
        WHILE TRUE
            IF len_colors < 0 THEN
                BREAK
            END IF
            SET len_colors TO len_colors + 1
            IF len_colors >= lengthHelper(colors) THEN
                BREAK
            END IF
        END WHILE

        SET n TO lengthHelper(colors)
        SET ext_colors TO empty list
        SET p TO 0
        WHILE p < n
            APPEND colors[p] TO ext_colors
            SET p TO p + 1
        END WHILE
        SET q TO 0
        WHILE q < k - 1
            APPEND colors[q] TO ext_colors
            SET q TO q + 1
        END WHILE

        SET res TO 0
        SET x TO 0
        WHILE x < n
            SET flag TO 1
            SET y TO 1
            WHILE y < k
                SET idx1 TO x + y
                SET idx2 TO x + y - 1
                IF equalElements(ext_colors[idx1], ext_colors[idx2]) THEN
                    SET flag TO 0
                    EXIT WHILE
                END IF
                SET y TO y + 1
            END WHILE
            IF flag = 1 THEN
                SET res TO res + 1
            END IF
            SET x TO x + 1
        END WHILE

        RETURN res
    END FUNCTION
END CLASS
