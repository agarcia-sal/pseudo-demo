CLASS Solution
    FUNCTION minimumArea(grid)
        FUNCTION eqArr(a, b)
            RETURN NOT (a IS NOT b OR LENGTH(a) IS NOT LENGTH(b))
        END FUNCTION

        FUNCTION isOne(val)
            RETURN val IS 1
        END FUNCTION

        FUNCTION getLength(arr)
            RETURN 0 + LENGTH(arr)
        END FUNCTION

        SET res TO 0
        SET px TO getLength(grid)
        IF eqArr(grid, []) OR eqArr(grid[0], []) THEN
            RETURN 0
        END IF

        SET py TO getLength(grid[0])
        SET alpha TO +∞
        SET beta TO -∞
        SET gamma TO +∞
        SET delta TO -∞

        FUNCTION scanRow(x, y)
            IF isOne(grid[x][y]) THEN
                IF alpha > x THEN
                    SET alpha TO x
                END IF
                IF beta < x THEN
                    SET beta TO x
                END IF
                IF gamma > y THEN
                    SET gamma TO y
                END IF
                IF delta < y THEN
                    SET delta TO y
                END IF
            END IF
        END FUNCTION

        FUNCTION loopInner(a, b)
            IF b >= py THEN RETURN END IF
            scanRow(a, b)
            loopInner(a, b + 1)
        END FUNCTION

        FUNCTION loopOuter(c)
            IF c >= px THEN RETURN END IF
            loopInner(c, 0)
            loopOuter(c + 1)
        END FUNCTION

        loopOuter(0)

        SET h TO (beta - alpha) + 1
        SET w TO (delta - gamma) + 1
        SET res TO h * w

        RETURN res
    END FUNCTION
END CLASS
