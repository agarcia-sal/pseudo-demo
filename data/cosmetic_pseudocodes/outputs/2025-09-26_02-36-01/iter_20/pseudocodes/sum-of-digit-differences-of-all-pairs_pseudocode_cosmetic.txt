CLASS Solution
    FUNCTION sumDigitDifferences(nums)
        FUNCTION digit_difference(a, b)
            FUNCTION count_mismatches(x, y, pos, acc)
                IF pos >= LENGTH(x) THEN
                    RETURN acc
                ELSE
                    IF x[pos] != y[pos] THEN
                        RETURN count_mismatches(x, y, pos + 1, acc + 1)
                    ELSE
                        RETURN count_mismatches(x, y, pos + 1, acc)
                    END IF
                END IF
            END FUNCTION
            RETURN count_mismatches(a, b, 0, 0)
        END FUNCTION

        SET sum_accumulator TO 0
        SET len_nums TO LENGTH(nums)
        SET outer_idx TO 0
        WHILE outer_idx < len_nums DO
            SET inner_idx TO outer_idx + 1
            WHILE inner_idx < len_nums DO
                SET sum_accumulator TO sum_accumulator + digit_difference(nums[outer_idx], nums[inner_idx])
                SET inner_idx TO inner_idx + 1
            END WHILE
            SET outer_idx TO outer_idx + 1
        END WHILE
        RETURN sum_accumulator
    END FUNCTION
END CLASS
