CLASS Solution  
    FUNCTION maxScore(points, m)  
        FUNCTION isPossible(minVal, m)  
            SET alpha TO 0  
            SET beta TO 0  
            SET gamma TO 0  
            WHILE gamma LESS THAN THE LENGTH OF points DO  
                SET sigma TO CEIL_DIV(minVal + points[gamma] - 1, points[gamma])  
                IF (sigma - beta) < 0 THEN  
                    SET sigma TO 0  
                ELSE  
                    SET sigma TO sigma - beta  
                END IF  
                IF sigma > 0 THEN  
                    SET alpha TO alpha + (2 * sigma - 1)  
                    SET beta TO sigma - 1  
                ELSE  
                    IF (gamma + 1) < LENGTH(points) THEN  
                        SET alpha TO alpha + 1  
                        SET beta TO 0  
                    END IF  
                END IF  
                IF alpha > m THEN  
                    RETURN False  
                END IF  
                SET gamma TO gamma + 1  
            END WHILE  
            RETURN True  
        END FUNCTION  

        FUNCTION CEIL_DIV(x, y)  
            IF (x MOD y) == 0 THEN  
                RETURN x DIV y  
            ELSE  
                RETURN (x DIV y) + 1  
            END IF  
        END FUNCTION  

        SET A TO 0  
        SET B TO ((m + 1) DIV 2) * (points[0] + 1)  
        FOR LOOP_COND: WHILE A < B DO  
            SET C TO ((A + B + 1) DIV 2)  
            IF isPossible(C, m) THEN  
                SET A TO C  
            ELSE  
                SET B TO C - 1  
            END IF  
        END FOR LOOP_COND  

        RETURN A  
    END FUNCTION  
END CLASS
