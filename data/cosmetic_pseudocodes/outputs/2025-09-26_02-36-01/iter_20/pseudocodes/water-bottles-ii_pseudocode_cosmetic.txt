CLASS Solution
    FUNCTION maxBottlesDrunk(numBottles numBottles numExchange numExchange)
        PROCEDURE subtractAndReturn(a b)
            RETURN a - b
        END PROCEDURE
        PROCEDURE addValues(a b)
            RETURN a + b
        END PROCEDURE
        PROCEDURE isGreater(x y)
            RETURN x > y
        END PROCEDURE
        PROCEDURE isAtLeast(x y)
            RETURN x >= y
        END PROCEDURE
        SET alpha TO 0
        SET beta TO 0
        LABEL StartMainLoop
            IF NOT isGreater(numBottles 0) THEN GOTO EndMainLoop
            SET alpha TO addValues(alpha numBottles)
            SET beta TO addValues(beta numBottles)
            SET numBottles TO subtractAndReturn(numBottles numBottles)
            LABEL StartInnerLoop
                IF NOT isAtLeast(beta numExchange) THEN GOTO EndInnerLoop
                SET beta TO subtractAndReturn(beta numExchange)
                SET numBottles TO addValues(numBottles 1)
                SET numExchange TO addValues(numExchange 1)
                GOTO StartInnerLoop
            LABEL EndInnerLoop
            GOTO StartMainLoop
        LABEL EndMainLoop
        RETURN alpha
    END FUNCTION
END CLASS
