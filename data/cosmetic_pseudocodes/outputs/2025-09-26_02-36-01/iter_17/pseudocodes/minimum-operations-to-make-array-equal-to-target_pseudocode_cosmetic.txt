CLASS Solution
    FUNCTION minimumOperations(nums, target)
        DECLARE integer p = 0
        DECLARE integer m = 0
        DECLARE integer r = 0
        DECLARE integer s = 0
        DECLARE integer q = LENGTH OF nums
        SET p = target[0] - nums[0]
        IF p < 0 THEN
            SET p = -p
        END IF
        SET r = 1
        WHILE r < q
            SET m = target[r] - nums[r]
            SET s = target[r - 1] - nums[r - 1]
            IF (m * s) > 0 THEN
                DECLARE integer u = m
                IF u < 0 THEN
                    SET u = -u
                END IF
                DECLARE integer v = s
                IF v < 0 THEN
                    SET v = -v
                END IF
                DECLARE integer w = u - v
                IF w > 0 THEN
                    SET p = p + w
                END IF
            ELSE
                IF m < 0 THEN
                    SET m = -m
                END IF
                SET p = p + m
            END IF
            SET r = r + 1
        END WHILE
        RETURN p
    END FUNCTION
END CLASS
