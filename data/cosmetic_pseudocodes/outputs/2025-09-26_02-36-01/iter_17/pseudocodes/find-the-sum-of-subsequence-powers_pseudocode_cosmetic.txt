CLASS Solution
    FUNCTION sumOfPowers(nums parameter LIST OF INTEGERS k parameter INTEGER) RETURNS INTEGER
        CONST MAX_VAL <- 1000000000 + 7
        VAR acc_result <- 0

        FUNCTION abs_val(x) RETURNS INTEGER
            IF x < 0 THEN
                RETURN 0 - x
            ELSE
                RETURN x
            END IF
        END FUNCTION

        FUNCTION generateCombos(arr LIST OF INTEGERS, r INTEGER) RETURNS LIST OF LIST OF INTEGERS
            VAR results <- EMPTY LIST

            FUNCTION recurse(start INTEGER, path LIST OF INTEGERS)
                IF LEN(path) = r THEN
                    results.ADD(COPY(path))
                    RETURN
                END IF
                VAR idx <- start
                WHILE idx < LEN(arr) DO
                    path.ADD(arr[idx])
                    recurse(idx + 1, path)
                    path.REMOVE_LAST()
                    idx <- idx + 1
                END WHILE
            END FUNCTION

            recurse(0, EMPTY LIST)
            RETURN results
        END FUNCTION

        VAR combosList <- generateCombos(nums, k)
        VAR comboIdx <- 0

        WHILE comboIdx < LEN(combosList) DO
            VAR currentCombo <- combosList[comboIdx]
            VAR minDiff <- 9000000000000
            VAR p <- 0

            REPEAT
                IF p >= k THEN
                    BREAK
                END IF
                VAR q <- p + 1

                REPEAT
                    IF q >= k THEN
                        BREAK
                    END IF
                    VAR diffTemp <- abs_val(currentCombo[p] - currentCombo[q])
                    IF diffTemp < minDiff THEN
                        minDiff <- diffTemp
                    END IF
                    q <- q + 1
                UNTIL FALSE
                p <- p + 1
            UNTIL FALSE

            acc_result <- acc_result + minDiff
            acc_result <- acc_result - ((acc_result / MAX_VAL) * MAX_VAL)
            comboIdx <- comboIdx + 1
        END WHILE

        RETURN acc_result
    END FUNCTION
END CLASS
