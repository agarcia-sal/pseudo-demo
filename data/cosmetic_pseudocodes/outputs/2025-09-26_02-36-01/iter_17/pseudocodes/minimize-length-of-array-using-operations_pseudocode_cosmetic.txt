CLASS Solution
    FUNCTION minimumArrayLength(nums)
        BEGIN
            SET alpha TO +∞
            FOR EACH zeta IN nums DO
            BEGIN
                IF (zeta < alpha) THEN
                BEGIN
                    SET alpha TO zeta
                END
            END

            SET beta TO 0
            SET iota TO 0
            WHILE iota < LENGTH(nums) DO
            BEGIN
                IF nums[iota] = alpha THEN
                BEGIN
                    SET beta TO beta + 1
                END
                SET iota TO iota + 1
            END

            IF NOT (beta ≠ 1) THEN
            BEGIN
                RETURN 1
            END

            SET gamma TO beta + 1
            SET delta TO 2
            SET epsilon TO 0
            epsilon = (gamma - (gamma MOD delta)) / delta
            RETURN epsilon
        END
END CLASS
