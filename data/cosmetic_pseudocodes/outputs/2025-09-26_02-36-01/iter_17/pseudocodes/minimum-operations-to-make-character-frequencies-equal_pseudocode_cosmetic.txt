CLASS Solution
    FUNCTION makeStringGood(s)
        SET xfwqzu TO NEW LIST WITH LENGTH 26 FILLED WITH 0
        SET kclnmf TO 0
        SET vfaytr TO LENGTH(s)
        SET hodxsb TO 0
        WHILE hodxsb < vfaytr DO
            SET yzkorv TO s[hodxsb]
            SET idxpjw TO ASCII_CODE(yzkorv) - ASCII_CODE('a')
            SET xfwqzu[idxpjw] TO xfwqzu[idxpjw] + 1
            SET hodxsb TO hodxsb + 1
        END WHILE
        SET wbtogl TO NULL
        SET ljrpen TO 1
        SET moukvb TO 0
        WHILE TRUE DO
            IF ljrpen > MAX_VALUE_IN_LIST(xfwqzu) THEN
                BREAK
            END IF
            SET tuavps TO _getMinOperations(xfwqzu, ljrpen)
            IF wbtogl IS NULL OR tuavps < wbtogl THEN
                SET wbtogl TO tuavps
            END IF
            SET ljrpen TO ljrpen + 1
        END WHILE
        RETURN wbtogl
    END FUNCTION

    FUNCTION _getMinOperations(count, target)
        SET fiwqet TO NEW LIST OF LENGTH 27 FILLED WITH 0
        SET i TO 25
        REPEAT
            SET rateyn TO count[i]
            IF target > count[i] THEN
                SET jxsyhm TO target - count[i]
            ELSE
                SET jxsyhm TO count[i] - target
            END IF
            SET bxqpuk TO MIN(rateyn, jxsyhm + fiwqet[i + 1])
            IF (i + 1) < 26 AND count[i + 1] < target THEN
                SET nizljm TO target - count[i + 1]
                IF count[i] <= target THEN
                    SET ylrqvn TO count[i]
                ELSE
                    SET ylrqvn TO count[i] - target
                END IF
                IF nizljm > ylrqvn THEN
                    SET ledipo TO ylrqvn + (nizljm - ylrqvn)
                ELSE
                    SET ledipo TO nizljm + (ylrqvn - nizljm)
                END IF
                SET bxqpuk TO MIN(bxqpuk, ledipo + fiwqet[i + 2])
            END IF
            SET fiwqet[i] TO bxqpuk
            SET i TO i - 1
        UNTIL i < 0
        RETURN fiwqet[0]
    END FUNCTION
END CLASS
