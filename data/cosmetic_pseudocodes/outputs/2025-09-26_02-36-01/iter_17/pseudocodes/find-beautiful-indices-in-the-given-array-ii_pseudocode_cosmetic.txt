CLASS Solution
    FUNCTION beautifulIndices(s parameter a parameter b parameter k)
        DECLARE list_X AS empty list
        DECLARE variable_m AS integer ZERO
        DECLARE variable_n AS integer (LENGTH OF s) - (LENGTH OF a)
        REPEAT
            IF variable_m IS GREATER THAN variable_n THEN BREAK
            END IF

            DECLARE temp_sub AS substring OF s FROM variable_m TO (variable_m + (LENGTH OF a) - ONE)
            IF NOT (temp_sub IS NOT EQUAL TO a) THEN
                APPEND variable_m TO list_X
            END IF
            variable_m = variable_m + 1
        UNTIL FALSE

        DECLARE list_Y AS empty list
        DECLARE variable_p AS integer ZERO
        DECLARE variable_q AS integer (LENGTH OF s) - (LENGTH OF b)
        REPEAT
            IF variable_p IS GREATER THAN variable_q THEN BREAK
            END IF

            DECLARE temp_sub2 AS substring OF s FROM variable_p TO (variable_p + (LENGTH OF b) - ONE)
            IF temp_sub2 IS EQUAL TO b THEN
                APPEND variable_p TO list_Y
            END IF
            variable_p = variable_p + 1
        UNTIL FALSE

        DECLARE resultList AS empty list
        DECLARE alpha AS ZERO
        DECLARE beta AS ZERO
        LOOP
            IF (alpha >= LENGTH OF list_X) OR (beta >= LENGTH OF list_Y) THEN EXIT LOOP END IF
            DECLARE dist AS list_X[alpha] - list_Y[beta]
            IF dist < 0 THEN dist = -dist END IF

            IF dist <= k THEN
                APPEND list_X[alpha] TO resultList
                alpha = alpha + 1
            ELSE
                IF list_X[alpha] < list_Y[beta] THEN
                    alpha = alpha + 1
                ELSE
                    beta = beta + 1
                END IF
            END IF
        END LOOP

        RETURN resultList
    END FUNCTION
END CLASS
