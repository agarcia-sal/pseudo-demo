CLASS Solution
    FUNCTION minMoves(rooks)
        SET total_rooks TO LENGTH(rooks)
        SET ordered_by_row TO EMPTY_LIST
        SET ordered_by_col TO EMPTY_LIST
        SET count_row_moves TO 0
        SET count_col_moves TO 0

        BEGIN
            SET temp_list TO COPY(rooks)
            WHILE LENGTH(temp_list) > 0
                SET min_idx TO 0
                SET j TO 1
                WHILE j < LENGTH(temp_list)
                    IF temp_list[j][0] < temp_list[min_idx][0]
                        SET min_idx TO j
                    END IF
                    SET j TO j + 1
                END WHILE
                APPEND ordered_by_row WITH temp_list[min_idx]
                REMOVE temp_list[min_idx]
            END WHILE
        END

        BEGIN
            SET temp_list TO COPY(rooks)
            WHILE LENGTH(temp_list) > 0
                SET min_idx TO 0
                SET k TO 1
                WHILE k < LENGTH(temp_list)
                    IF temp_list[k][1] < temp_list[min_idx][1]
                        SET min_idx TO k
                    END IF
                    SET k TO k + 1
                END WHILE
                APPEND ordered_by_col WITH temp_list[min_idx]
                REMOVE temp_list[min_idx]
            END WHILE
        END

        SET r TO 0
        SET hm TO 0
        REPEAT
            IF r >= total_rooks
                BREAK
            END IF
            SET delta_row TO ordered_by_row[r][0]
            IF delta_row >= r
                SET diff_row TO delta_row - r
            ELSE
                SET diff_row TO r - delta_row
            END IF
            SET hm TO hm + diff_row
            SET r TO r + 1
        UNTIL FALSE

        SET s TO 0
        SET tm TO 0
        REPEAT
            IF s >= total_rooks
                BREAK
            END IF
            SET delta_col TO ordered_by_col[s][1]
            IF delta_col >= s
                SET diff_col TO delta_col - s
            ELSE
                SET diff_col TO s - delta_col
            END IF
            SET tm TO tm + diff_col
            SET s TO s + 1
        UNTIL FALSE

        RETURN hm + tm
    END FUNCTION
END CLASS
