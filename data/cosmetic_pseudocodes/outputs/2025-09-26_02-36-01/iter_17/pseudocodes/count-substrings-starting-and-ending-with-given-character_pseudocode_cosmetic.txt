CLASS Solution
    FUNCTION countSubstrings(s c)
        DECLARE alpha AS INTEGER
        DECLARE beta AS INTEGER
        DECLARE gamma AS INTEGER
        DECLARE delta AS INTEGER

        beta = 0
        WHILE beta < LENGTH(s)
            IF s[beta] = c THEN
                alpha = alpha + 1
            END IF
            beta = beta + 1
        END WHILE

        gamma = alpha + 1
        delta = (alpha * gamma) / 2

        RETURN delta
    END FUNCTION
END CLASS
