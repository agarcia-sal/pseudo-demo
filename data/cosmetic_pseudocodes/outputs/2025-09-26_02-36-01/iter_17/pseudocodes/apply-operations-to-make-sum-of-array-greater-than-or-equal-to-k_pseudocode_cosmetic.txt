CLASS Solution  
    FUNCTION minOperations(k)  
        SET omega TO a very large number  
        SET alpha TO 1  
        WHILE alpha < FLOOR( SQRT(k) ) + 1  
            BEGIN  
                SET beta TO (k + alpha - 1) DIV alpha  
                SET gamma TO (alpha - 1) + (beta - 1)  
                IF gamma < omega THEN  
                    SET omega TO gamma  
                END IF  
                SET alpha TO alpha + 1  
            END  
        RETURN omega  
    END FUNCTION  
END CLASS
