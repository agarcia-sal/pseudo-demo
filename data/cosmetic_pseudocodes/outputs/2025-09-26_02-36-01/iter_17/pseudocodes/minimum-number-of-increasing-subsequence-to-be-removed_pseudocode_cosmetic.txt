CLASS Solution  
    FUNCTION minOperations(nums)  
        SET length TO 0  
        SET tempIndexA TO 0  
        SET tempIndexB TO 0  
        SET maxVal TO 0  
        SET dpList TO EMPTY LIST  

        length = LENGTH(nums)

        IF length = 0 THEN
            RETURN 0
        END IF

        SET k TO 0
        WHILE k < length DO
            APPEND 1 TO dpList
            k = k + 1
        END WHILE

        tempIndexA = 1
        REPEAT
            IF tempIndexA >= length THEN
                BREAK
            END IF
            tempIndexB = 0

            WHILE tempIndexB < tempIndexA DO
                IF NOT(nums[tempIndexA] > nums[tempIndexB]) THEN
                    dpList[tempIndexA] = MAX(dpList[tempIndexA], dpList[tempIndexB] + 1)
                END IF
                tempIndexB = tempIndexB + 1
            END WHILE

            tempIndexA = tempIndexA + 1
        UNTIL FALSE

        maxVal = dpList[0]
        tempIndexA = 1
        WHILE tempIndexA < length DO
            IF dpList[tempIndexA] > maxVal THEN
                maxVal = dpList[tempIndexA]
            END IF
            tempIndexA = tempIndexA + 1
        END WHILE

        RETURN maxVal
    END FUNCTION  
END CLASS
