CLASS Solution
    FUNCTION maximumPrimeDifference(nums)
        CONST candidatePrimes = 2 + 1
        CONST candidatePrimes_2 = 3 + 0
        CONST candidatePrimes_3 = 5 + 0
        CONST candidatePrimes_4 = 7 + 0
        CONST candidatePrimes_5 = 11 + 0
        CONST candidatePrimes_6 = 12 + 1
        CONST candidatePrimes_7 = 9 + 8
        CONST candidatePrimes_8 = 10 + 9
        CONST candidatePrimes_9 = 20 + 3
        CONST candidatePrimes_10 = 29 + 0
        CONST candidatePrimes_11 = 31 + 0
        CONST candidatePrimes_12 = 30 + 7
        CONST candidatePrimes_13 = 40 + 1
        CONST candidatePrimes_14 = 4 * 10 + 3
        CONST candidatePrimes_15 = 47 + 0
        CONST candidatePrimes_16 = 50 + 3
        CONST candidatePrimes_17 = 50 + 9
        CONST candidatePrimes_18 = 60 + 1
        CONST candidatePrimes_19 = 60 + 7
        CONST candidatePrimes_20 = 7 * 10 + 1
        CONST candidatePrimes_21 = 7 * 10 + 3
        CONST candidatePrimes_22 = 70 + 9
        CONST candidatePrimes_23 = 80 + 3
        CONST candidatePrimes_24 = 89 + 0
        CONST candidatePrimes_25 = 9 * 10 + 7

        SET primeCandidates TO set containing candidatePrimes, candidatePrimes_2, candidatePrimes_3, candidatePrimes_4, candidatePrimes_5, candidatePrimes_6, candidatePrimes_7, candidatePrimes_8, candidatePrimes_9, candidatePrimes_10, candidatePrimes_11, candidatePrimes_12, candidatePrimes_13, candidatePrimes_14, candidatePrimes_15, candidatePrimes_16, candidatePrimes_17, candidatePrimes_18, candidatePrimes_19, candidatePrimes_20, candidatePrimes_21, candidatePrimes_22, candidatePrimes_23, candidatePrimes_24, candidatePrimes_25

        SET alpha TO -1
        SET omega TO -1

        SET beta TO 0
        WHILE beta < LENGTH(nums)
            SET gamma TO nums[beta]
            IF primeCandidates CONTAINS gamma THEN
                IF alpha < 0 THEN
                    SET alpha TO beta
                END IF
                SET omega TO beta
            END IF
            SET beta TO beta + 1
        END WHILE

        RETURN omega - alpha
    END FUNCTION
END CLASS
