CLASS Solution
    FUNCTION compressedString(word)
        SET totalParts TO new empty list
        SET idx TO 0
        REPEAT
            IF idx >= LENGTH(word)
                BREAK
            END IF
            SET currentChar TO word[idx]
            SET cnt TO 0
            REPEAT
                IF idx >= LENGTH(word)
                    BREAK
                END IF
                IF word[idx] != currentChar
                    BREAK
                END IF
                SET cnt TO cnt + 1
                SET idx TO idx + 1
                IF cnt == (3 * 3)
                    BREAK
                END IF
            UNTIL FALSE
            CALL totalParts.APPEND(CONVERT_TO_STRING(cnt) + currentChar)
        UNTIL FALSE
        SET outputString TO EMPTY_STRING
        FOR EACH segment IN totalParts
            SET outputString TO outputString + segment
        END FOR
        RETURN outputString
    END FUNCTION
END CLASS
