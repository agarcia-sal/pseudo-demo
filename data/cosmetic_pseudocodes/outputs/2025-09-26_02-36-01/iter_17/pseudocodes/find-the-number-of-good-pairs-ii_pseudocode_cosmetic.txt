CLASS Solution  
    FUNCTION numberOfPairs(nums1, nums2, k)  
        DECLARE frequencyMap AS map INITIALIZED TO empty  
        FOR each element val IN nums2  
            IF val NOT IN frequencyMap  
                frequencyMap[val] = 1  
            ELSE  
                frequencyMap[val] = frequencyMap[val] + 1  
            END IF  
        END FOR  
        
        DECLARE totalPairs AS 0  
        DECLARE idx1 AS 0  
        WHILE idx1 < LENGTH(nums1)  
            DECLARE currentNum AS nums1[idx1]  
            DECLARE keysIterator AS LIST of KEYS(frequencyMap)  
            
            DECLARE idx2 AS 0  
            WHILE idx2 < LENGTH(keysIterator)  
                DECLARE keyVal AS keysIterator[idx2]  
                DECLARE freq AS frequencyMap[keyVal]  
                
                DECLARE divisor AS (keyVal * k)  
                DECLARE modVal AS currentNum - CURRENTDIV(currentNum, divisor) * divisor  
                IF modVal = 0  
                    totalPairs = totalPairs + freq  
                END IF  
                
                idx2 = idx2 + 1  
            END WHILE  
            
            idx1 = idx1 + 1  
        END WHILE  
        
        RETURN totalPairs  
    END FUNCTION  
    
    
    FUNCTION CURRENTDIV(a, b)  
        DECLARE quotient AS 0  
        DECLARE remainder AS a  
        WHILE remainder >= b  
            remainder = remainder - b  
            quotient = quotient + 1  
        END WHILE  
        RETURN quotient  
    END FUNCTION  
END CLASS
