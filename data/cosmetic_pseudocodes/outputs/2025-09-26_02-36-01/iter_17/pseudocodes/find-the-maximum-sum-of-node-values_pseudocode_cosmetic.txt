CLASS Solution
    FUNCTION maximumValueSum(nums k edges)
        DECLARE acc = 0
        DECLARE cnt = 0
        DECLARE least = +âˆž

        DECLARE idx = 0
        WHILE idx < LENGTH(nums)
            DECLARE val = nums[idx]
            DECLARE xres = val XOR k
            DECLARE diff = xres - val
            IF diff > 0 THEN
                cnt = cnt + 1
            END IF

            IF val > xres THEN
                acc = acc + val
            ELSE
                acc = acc + xres
            END IF

            DECLARE abs_diff = diff
            IF diff < 0 THEN
                abs_diff = -diff
            END IF

            IF abs_diff < least THEN
                least = abs_diff
            END IF

            idx = idx + 1
        END WHILE

        IF (cnt MOD 2) <> 0 THEN
            acc = acc - least
        END IF

        RETURN acc
    END FUNCTION
END CLASS
