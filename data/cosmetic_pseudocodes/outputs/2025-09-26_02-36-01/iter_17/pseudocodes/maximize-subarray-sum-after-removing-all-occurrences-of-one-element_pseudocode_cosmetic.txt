CLASS Solution
    FUNCTION maxSubarraySum(nums)
        FUNCTION kadane(arr)
            SET a TO arr[0]
            SET b TO arr[0]
            SET i TO 1
            WHILE i LESS THAN LENGTH OF arr
                SET c TO arr[i]
                IF c GREATER THAN a PLUS c
                    SET a TO c
                ELSE
                    SET a TO a PLUS c
                END IF
                IF b LESS THAN a
                    SET b TO a
                END IF
                SET i TO i PLUS 1
            END WHILE
            RETURN b
        END FUNCTION

        SET d TO kadane(nums)
        SET e TO EMPTY SET
        SET f TO 0
        WHILE f LESS THAN LENGTH OF nums
            SET e TO e UNION {nums[f]}
            SET f TO f PLUS 1
        END WHILE

        FOR EACH g IN e
            SET h TO EMPTY LIST
            SET j TO 0
            WHILE j LESS THAN LENGTH OF nums
                IF nums[j] NOT EQUALS g
                    APPEND nums[j] TO h
                END IF
                SET j TO j PLUS 1
            END WHILE
            IF LENGTH OF h GREATER THAN 0
                SET k TO kadane(h)
                IF d LESS THAN k
                    SET d TO k
                END IF
            END IF
        END FOR

        RETURN d
    END FUNCTION
END CLASS
