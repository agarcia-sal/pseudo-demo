CLASS Solution
    FUNCTION maxArea(height positions directions)
        SET length TO LENGTH(positions)
        SET maximum TO 0
        SET counter TO 1

        WHILE counter <= height * 2
            SET index TO 0
            WHILE index < length
                SET pos_value TO positions[index]
                SET dir_char TO directions[index]

                IF (pos_value = 0) AND (dir_char = 'D')
                    SET directions TO SUBSTR(directions, 0, index) + 'U' + SUBSTR(directions, index + 1, LENGTH(directions))
                ELSE
                    IF (pos_value = height) AND (dir_char = 'U')
                        SET directions TO SUBSTR(directions, 0, index) + 'D' + SUBSTR(directions, index + 1, LENGTH(directions))
                    END IF
                END IF

                IF directions[index] = 'U' THEN
                    SET positions[index] TO positions[index] + 1
                ELSE
                    SET positions[index] TO positions[index] - 1
                END IF

                SET index TO index + 1
            END WHILE

            SET current_sum TO 0
            SET j TO 0
            WHILE j < length
                SET current_sum TO current_sum + positions[j]
                SET j TO j + 1
            END WHILE

            IF maximum < current_sum THEN
                SET maximum TO current_sum
            END IF

            SET counter TO counter + 1
        END WHILE

        RETURN maximum
    END FUNCTION
END CLASS
