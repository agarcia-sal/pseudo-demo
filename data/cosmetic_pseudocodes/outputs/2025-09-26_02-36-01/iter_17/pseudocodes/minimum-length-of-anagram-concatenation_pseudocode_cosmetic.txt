CLASS Solution  
    FUNCTION minAnagramLength(s)  
        DECLARE auxSet AS SET OF CHAR  
        SET auxSet TO EMPTY SET  
        DECLARE idx AS INTEGER  
        SET idx TO 0  
        WHILE idx < LENGTH(s)  
            BEGIN  
                DECLARE currentChar AS CHAR  
                SET currentChar TO s[idx]  
                SET auxSet = auxSet UNION {currentChar}  
                SET idx TO idx + 1  
            END  
        END WHILE  
        RETURN SIZE(auxSet)  
    END FUNCTION  
END CLASS
