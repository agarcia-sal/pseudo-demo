CLASS Solution
    FUNCTION countDays(days, meetings)
        DECLARE n IS LENGTH(meetings)
        FOR i FROM 1 TO n - 1
            FOR j FROM 0 TO n - i - 1
                IF meetings[j][0] > meetings[j + 1][0] THEN
                    DECLARE temp IS meetings[j]
                    SET meetings[j] TO meetings[j + 1]
                    SET meetings[j + 1] TO temp
                END IF
            END FOR
        END FOR

        DECLARE p IS 1
        DECLARE q IS 0

        DECLARE k IS 0
        WHILE k < n DO
            DECLARE s IS meetings[k][0]
            DECLARE e IS meetings[k][1]

            IF p < s THEN
                SET q TO q + (s - p)
            END IF

            IF p > e THEN
                SET p TO p
            ELSE
                SET p TO e + 1
            END IF

            SET k TO k + 1
        END WHILE

        IF p <= days THEN
            SET q TO q + (days - p + 1)
        END IF

        RETURN q
    END FUNCTION
END CLASS
