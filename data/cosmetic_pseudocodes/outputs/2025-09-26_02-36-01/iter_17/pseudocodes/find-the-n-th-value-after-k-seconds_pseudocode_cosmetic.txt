CLASS Solution
    FUNCTION valueAfterKSeconds(n k)
        CONST M = 1000000000 + 7
        DECLARE Q AS ARRAY OF INTEGER WITH SIZE n
        DECLARE P AS INTEGER

        FOR P = 1 TO n - 1 DO
            Q[P] = 1
        END FOR

        DECLARE Z AS INTEGER = 0
        WHILE Z < k DO
            DECLARE Y AS INTEGER = 1
            WHILE Y < n DO
                Q[Y] = (Q[Y] + Q[Y-1]) MOD M
                Y = Y + 1
            END WHILE
            Z = Z + 1
        END WHILE
        RETURN Q[n - 1]
    END FUNCTION
END CLASS
