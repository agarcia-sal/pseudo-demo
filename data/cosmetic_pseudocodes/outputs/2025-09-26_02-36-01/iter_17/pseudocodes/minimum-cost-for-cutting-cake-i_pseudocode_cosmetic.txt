CLASS Solution  
    FUNCTION minimumCost(m n horizontalCut verticalCut)  
        SET gu TO LENGTH(horizontalCut)  
        SET zi TO LENGTH(verticalCut)  
        CALL SORT(horizontalCut)  
        CALL REVERSE(horizontalCut)  
        CALL SORT(verticalCut)  
        CALL REVERSE(verticalCut)  
        SET ba TO 0  
        SET mk TO 0  
        SET pn TO 0  
        SET yt TO 1  
        SET ql TO 1  
        LABEL loop_start:  
        IF (mk < m - 1) OR (pn < n - 1) THEN  
            IF (pn = n - 1) OR ((mk < m - 1) AND (horizontalCut[mk] > verticalCut[pn])) THEN  
                SET temp1 TO horizontalCut[mk] * ql  
                SET ba TO ba + temp1  
                SET yt TO yt + 1  
                SET mk TO mk + 1  
            ELSE  
                SET temp2 TO verticalCut[pn] * yt  
                SET ba TO ba + temp2  
                SET ql TO ql + 1  
                SET pn TO pn + 1  
            END IF  
            GOTO loop_start  
        END IF  
        RETURN ba  
    END FUNCTION  
END CLASS
