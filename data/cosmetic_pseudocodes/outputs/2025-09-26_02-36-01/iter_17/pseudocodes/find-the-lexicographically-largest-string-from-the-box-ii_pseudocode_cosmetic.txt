CLASS Solution
	FUNCTION answerString(word, numFriends)
		DO
			IF numFriends <> 1 THEN BREAK END IF
			RETURN word
		WHILE FALSE
		DECLARE xU34 AS STRING
		DECLARE w7V2 AS INTEGER
		SET xU34 TO _lastSubstring(word)
		SET w7V2 TO LENGTH(word) - numFriends + 1
		DECLARE z90K AS INTEGER
		SET z90K TO LENGTH(xU34)
		IF w7V2 < z90K THEN
			SET z90K TO w7V2
		END IF
		RETURN substring(xU34, 1, z90K)
	END FUNCTION

	FUNCTION _lastSubstring(s)
		DECLARE a AS INTEGER
		DECLARE b AS INTEGER
		DECLARE c AS INTEGER
		SET a = 0
		SET b = 1
		SET c = 0

		WHILE TRUE
			IF (b + c) >= LENGTH(s) THEN BREAK END IF
			DECLARE p1 AS CHARACTER
			DECLARE p2 AS CHARACTER
			SET p1 = s[a + c + 1]
			SET p2 = s[b + c + 1]
			IF p1 = p2 THEN
				SET c = c + 1
			ELSE
				IF p1 > p2 THEN
					SET b = b + c + 1
					SET c = 0
				ELSE
					SET a = MAX(a + c + 1, b)
					SET b = a + 1
					SET c = 0
				END IF
			END IF
		END WHILE
		RETURN substring(s, a + 1, LENGTH(s) - a)
	END FUNCTION
END CLASS
