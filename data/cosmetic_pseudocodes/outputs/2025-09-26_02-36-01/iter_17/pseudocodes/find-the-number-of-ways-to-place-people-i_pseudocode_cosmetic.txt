CLASS Solution
    FUNCTION numberOfPairs(points)
        SET total TO 0
        SET length TO LENGTH(points)
        SET outerIndex TO 0
        WHILE outerIndex < length
            SET innerIndex TO 0
            REPEAT
                IF NOT (outerIndex = innerIndex) THEN
                    BEGIN
                        SET px TO points[outerIndex][0]
                        SET py TO points[outerIndex][1]
                        SET qx TO points[innerIndex][0]
                        SET qy TO points[innerIndex][1]
                        IF (NOT (qx < px)) AND (NOT (py < qy)) THEN
                            BEGIN
                                SET isValid TO true
                                SET middleIndex TO 0
                                WHILE middleIndex < length AND isValid = true
                                    IF (middleIndex <> outerIndex) AND (middleIndex <> innerIndex) THEN
                                        BEGIN
                                            SET rx TO points[middleIndex][0]
                                            SET ry TO points[middleIndex][1]
                                            IF ((px <= rx) AND (rx <= qx)) AND ((py >= ry) AND (ry >= qy)) THEN
                                                BEGIN
                                                    SET isValid TO false
                                                END
                                            END IF
                                        END
                                    middleIndex = middleIndex + 1
                                END WHILE
                                IF isValid = true THEN
                                    total = total + 1
                                END IF
                            END
                        END IF
                    END
                END IF
                innerIndex = innerIndex + 1
            UNTIL NOT (innerIndex < length)
            outerIndex = outerIndex + 1
        END WHILE
        RETURN total
    END FUNCTION
END CLASS
