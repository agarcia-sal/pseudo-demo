CLASS Solution
    FUNCTION minOperations(nums parameter k parameter)
        CALL heapify ON nums

        DECLARE c GIVING 0

        REPEAT
            IF (NOT (nums[0] < k)) OR (LENGTH(nums) <= 1) THEN
                BREAK
            END IF

            DECLARE a AS heappop(nums)
            DECLARE b AS heappop(nums)

            DECLARE v AS (MIN(b, a) * 2) + MAX(b, a)

            CALL heappush ON nums WITH v

            c = c + 1
        UNTIL FALSE

        RETURN c
    END FUNCTION
END CLASS
