CLASS Solution  
    FUNCTION findKthSmallest(coins k)  
        FUNCTION count_smaller_or_equal(x)  
            SET z TO 0  
            SET u TO 1  
            LOOP  
                IF u GREATER THAN 2 RAISED TO (LENGTH(coins) MINUS 1)  
                    BREAK  
                END IF  
                SET p TO 1  
                SET q TO 0  
                SET v TO 0  
                LOOP  
                    IF v GREATER OR EQUAL LENGTH(coins)  
                        BREAK  
                    END IF  
                    IF (u BITWISE_AND (1 LEFT_SHIFT v)) NOT_EQUALS 0  
                        SET a TO p  
                        SET b TO coins[v]  
                        SET r TO 0  
                        LOOP  
                            IF b EQUALS 0  
                                SET r TO a  
                                BREAK  
                            END IF  
                            SET temp_a TO a MOD b  
                            SET a TO b  
                            SET b TO temp_a  
                        END LOOP  
                        SET p TO (p MULTIPLY coins[v]) DIVIDED_BY r  
                        SET q TO q PLUS 1  
                    END IF  
                    SET v TO v PLUS 1  
                END LOOP  
                IF (q MOD 2) EQUALS 1  
                    SET z TO z PLUS (x DIV p)  
                ELSE  
                    SET z TO z MINUS (x DIV p)  
                END IF  
                SET u TO u PLUS 1  
            END LOOP  
            RETURN z  
        END FUNCTION  

        SET s TO 1  
        SET t TO k MULTIPLY coins[0]  
        WHILE s LESS_THAN t  
            SET m TO (s PLUS t) DIV 2  
            IF count_smaller_or_equal(m) LESS_THAN k  
                SET s TO m PLUS 1  
            ELSE  
                SET t TO m  
            END IF  
        END WHILE  
        RETURN s  
    END FUNCTION  
END CLASS
