CLASS Solution
    FUNCTION numberOfWays WITH PARAMETER n
        LET CONST_MODULO BE (10 * 10 * 10 * 10 * 10 * 10 * 10 * 10 * 10) + 7
        LET ARRAY dpArray BE ARRAY OF SIZE (n + 1) FILLED WITH 0
        SET dpArray[0] = 1
        
        LET coinIndex BE 0
        LET coinsList BE [1, 2, 6]
        WHILE coinIndex < LENGTH(coinsList) DO
            LET currentCoin BE coinsList[coinIndex]
            LET iteratorJ BE currentCoin
            WHILE iteratorJ <= n DO
                LET tempVal1 BE dpArray[iteratorJ]
                LET tempVal2 BE dpArray[iteratorJ - currentCoin]
                dpArray[iteratorJ] = (tempVal1 + tempVal2) MOD CONST_MODULO
                SET iteratorJ = iteratorJ + 1
            END WHILE
            SET coinIndex = coinIndex + 1
        END WHILE
        
        LET accumulator BE 0
        LET idxK BE 0
        REPEAT
            IF (idxK * 4) <= n THEN
                LET subIndex BE n - (idxK * 4)
                accumulator = (accumulator + dpArray[subIndex]) MOD CONST_MODULO
            END IF
            SET idxK = idxK + 1
        UNTIL idxK > 2
        
        RETURN accumulator
    END FUNCTION
END CLASS
