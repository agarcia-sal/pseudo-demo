CLASS Solution  
    FUNCTION sumDigitDifferences(nums)  
        FUNCTION digit_difference(num1, num2)  
            SET mismatchCount TO 0  
            SET pos TO 0  
            WHILE pos < LENGTH(num1) DO  
                IF NOT (CHAR_AT(num1, pos) = CHAR_AT(num2, pos)) THEN  
                    SET mismatchCount TO mismatchCount + 1  
                END IF  
                SET pos TO pos + 1  
            END WHILE  
            RETURN mismatchCount  
        END FUNCTION  
        SET aggregate TO 0  
        SET lengthNums TO LENGTH(nums)  
        SET outerIndex TO 0  
        REPEAT  
            IF outerIndex >= lengthNums THEN BREAK  
            SET innerIndex TO outerIndex + 1  
            REPEAT  
                IF innerIndex >= lengthNums THEN BREAK  
                SET firstVal TO nums[outerIndex]  
                SET secondVal TO nums[innerIndex]  
                SET diffValue TO digit_difference(firstVal, secondVal)  
                SET aggregate TO aggregate + diffValue  
                SET innerIndex TO innerIndex + 1  
            UNTIL FALSE  
            SET outerIndex TO outerIndex + 1  
        UNTIL FALSE  
        RETURN aggregate  
    END FUNCTION  
END CLASS
