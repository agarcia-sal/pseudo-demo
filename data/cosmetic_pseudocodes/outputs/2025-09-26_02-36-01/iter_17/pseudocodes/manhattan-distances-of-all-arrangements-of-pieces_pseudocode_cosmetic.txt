CLASS Solution
    FUNCTION distanceSum WITH PARAMETERS m n k
        BEGIN
            CONST limit ← 1000000007
            VAR varA, varB, varC, varD, varE, varF, varG, varH, varI, varJ
            SET varA ← m * m
            SET varB ← n * n
            SET varC ← m * m * m - m
            SET varD ← n * n * n - n
            SET varE ← varC / 6
            SET varF ← varD / 6
            SET varG ← m * m * n * n * n
            SET varH ← m * m * m * n * n
            SET varI ← (m * n) - 2
            SET varJ ← 1
            VAR idx ← 1
            VAR acc ← 1
            VAR res ← 1
            VAR x ← 0
            VAR y ← 0
            VAR z ← 1
            VAR w ← 1
            VAR temp ← 1
            VAR answer ← 0

            FUNCTION comb WITH PARAMETERS a b
                VAR numerator, denominator, i, j
                SET numerator ← 1
                SET denominator ← 1
                SET i ← a
                SET j ← 1
                WHILE j ≤ b DO
                    SET numerator ← numerator * i
                    SET denominator ← denominator * j
                    SET i ← i - 1
                    SET j ← j + 1
                END WHILE
                RETURN numerator / denominator
            END FUNCTION

            SET varJ ← comb(m * n - 2, k - 2)
            SET varK ← varE * n * n
            SET varL ← varF * m * m
            SET total_sum ← (varE * n * n + varF * m * m) * varJ
            RETURN total_sum MOD limit
        END
END CLASS
