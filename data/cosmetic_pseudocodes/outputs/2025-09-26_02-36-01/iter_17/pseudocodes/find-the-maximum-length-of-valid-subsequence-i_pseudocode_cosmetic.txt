CLASS Solution
    FUNCTION maximumLength(nums)
        DECLARE var_A AS INTEGER
        DECLARE var_B AS INTEGER
        DECLARE var_C AS INTEGER
        DECLARE var_D AS INTEGER
        SET var_A = 0
        SET var_B = 0
        SET var_C = 1
        WHILE var_C < (LENGTH OF nums)
            SET var_D = (nums[(var_C - 1)] + nums[var_C])
            IF ((var_D MOD 2) = 0)
                IF ((var_A + 1) > var_B)
                    SET var_A = var_A + 1
                ELSE
                    SET var_A = var_B
                END IF
            ELSE
                IF ((var_B + 1) > var_A)
                    SET var_B = var_B + 1
                ELSE
                    SET var_B = var_A
                END IF
            END IF
            SET var_C = var_C + 1
        END WHILE
        IF var_A > (var_B + 1)
            RETURN var_A
        ELSE
            RETURN var_B + 1
        END IF
    END FUNCTION
END CLASS
