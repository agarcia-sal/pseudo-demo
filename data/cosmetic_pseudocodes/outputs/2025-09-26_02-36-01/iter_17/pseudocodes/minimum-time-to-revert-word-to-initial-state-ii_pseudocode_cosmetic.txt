CLASS Solution
    FUNCTION minimumTimeToInitialState(word parameter word AND integer parameter k) RETURNS integer
        SET w_len TO 0
        SET t_counter TO 1

        SET w_len TO LENGTH(word)

        REPEAT
            BEGIN
                SET sub_pref TO SUBSTRING(word, t_counter * k, w_len - (t_counter * k))
                SET does_start TO false

                IF LEFT(word, LENGTH(sub_pref)) = sub_pref THEN
                    SET does_start TO true
                END IF

                IF does_start = true THEN
                    RETURN t_counter
                END IF

                SET t_counter TO t_counter + 1
            END
        UNTIL false
    END FUNCTION
END CLASS
