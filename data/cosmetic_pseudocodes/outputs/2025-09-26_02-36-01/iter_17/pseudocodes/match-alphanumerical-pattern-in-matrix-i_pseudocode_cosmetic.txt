CLASS Solution
    FUNCTION findPattern(board, pattern)
        SET a TO LENGTH board
        SET b TO LENGTH board[0]
        SET c TO LENGTH pattern
        SET d TO LENGTH pattern[0]

        FUNCTION matches(x, y)
            SET m TO EMPTY MAP
            SET n TO EMPTY MAP

            SET e TO 0
            WHILE e < c
                SET f TO 0
                REPEAT
                    SET u TO pattern[e][f]
                    SET v TO board[x + e][y + f]

                    IF u >= '0' AND u <= '9' THEN
                        IF INT(u) != v THEN
                            RETURN False
                        END IF
                    ELSE
                        IF u IN m THEN
                            IF m[u] != v THEN
                                RETURN False
                            END IF
                        ELSE
                            IF v IN n THEN
                                RETURN False
                            END IF
                            SET m[u] = v
                            SET n[v] = u
                        END IF
                    END IF

                    SET f = f + 1
                UNTIL f >= d
                SET e = e + 1
            END WHILE
            RETURN True
        END FUNCTION

        SET i TO 0
        WHILE i <= a - c
            SET j TO 0
            WHILE j <= b - d
                IF matches(i, j) THEN
                    RETURN [i, j]
                END IF
                SET j = j + 1
            END WHILE
            SET i = i + 1
        END WHILE

        RETURN [-1, -1]
    END FUNCTION
END CLASS
