CLASS Solution
    FUNCTION countAlternatingSubarrays(nums)
        SET size TO LENGTH OF nums
        IF size EQUALS 0 THEN
            RETURN 0
        END IF
        SET tally TO size
        SET streak TO 1
        SET index TO 1
        WHILE index LESS THAN size DO
            BEGIN
                IF NOT (nums[index] EQUALS nums[index MINUS 1]) THEN
                    streak = streak + 1
                ELSE
                    streak = 1
                END IF
                tally = tally + (streak MINUS 1)
                index = index + 1
            END
        END WHILE
        RETURN tally
    END FUNCTION
END CLASS
