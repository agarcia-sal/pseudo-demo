CLASS Enemy
    FUNCTION __init__(self, damage, timeTakenDown)
        SET self.damage TO damage
        SET self.timeTakenDown TO timeTakenDown
    END FUNCTION
END CLASS

CLASS Solution
    FUNCTION minDamage(self, power, damage, health)
        SET beta TO 0
        SET gamma TO 0
        FOR omega FROM 0 TO LENGTH OF damage MINUS 1
            SET gamma TO gamma + damage[omega]
        END FOR
        SET zeta TO []
        SET alpha TO 0
        WHILE alpha <= LENGTH OF damage - 1
            SET kappa TO damage[alpha]
            SET mu TO health[alpha]
            SET nu TO (mu + power - 1) DIV power
            CREATE delta AS Enemy(kappa, nu)
            APPEND delta TO zeta
            SET alpha TO alpha + 1
        END WHILE
        SET zeta TO SORTED(zeta, BY FUNCTION(e) RETURN e.damage / e.timeTakenDown END FUNCTION, REVERSE true)
        SET iota TO 0
        WHILE iota < LENGTH OF zeta
            SET beta TO beta + gamma * zeta[iota].timeTakenDown
            SET gamma TO gamma - zeta[iota].damage
            SET iota TO iota + 1
        END WHILE
        RETURN beta
    END FUNCTION
END CLASS
