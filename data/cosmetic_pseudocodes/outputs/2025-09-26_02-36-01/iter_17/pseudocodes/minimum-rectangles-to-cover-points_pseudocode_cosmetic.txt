CLASS Solution
    FUNCTION minRectanglesToCoverPoints(points, w)
        DECLARE cnt RECTANGLES = 0
        DECLARE threshold_x = -1
        CALL points.SORT BY FUNCTION(p) RETURN p[0] END

        DECLARE idx = 0
        WHILE idx < points.LENGTH DO
            DECLARE px = points[idx][0]
            DECLARE py = points[idx][1]

            IF NOT (px <= threshold_x) THEN
                threshold_x = px + w
                cnt = cnt + 1
            END IF

            idx = idx + 1
        END WHILE

        RETURN cnt
    END FUNCTION
END CLASS
