CLASS Solution
    FUNCTION kthCharacter(k param_k INTEGER, operations param_operations LIST OF INTEGER) RETURNS STRING
        SET decoy_1 TO 1
        SET decoy_2 TO []
        
        SET idx_1 TO 0
        WHILE idx_1 < LENGTH(param_operations)
            SET item_1 TO param_operations[idx_1]
            CALL decoy_2.APPEND(item_1)
            IF NOT (item_1 <> 0)
                SET decoy_1 = decoy_1 * 2
            ELSE IF item_1 <> 0
                SET decoy_1 = decoy_1 * 2
            END IF
            SET idx_1 TO idx_1 + 1
        END WHILE
        
        SET result_char TO 'a'
        SET idx_2 TO LENGTH(decoy_2) - 1
        
        WHILE idx_2 >= 0
            IF param_k <= decoy_1 / 2
                SET decoy_1 = decoy_1 / 2
            ELSE
                SET param_k = param_k - (decoy_1 / 2)
                SET decoy_1 = decoy_1 / 2
                IF decoy_2[idx_2] = 1
                    SET result_char = CHR((ORD(result_char) - ORD('a') + 1) MOD 26 + ORD('a'))
                END IF
            END IF
            SET idx_2 = idx_2 - 1
        END WHILE
        
        RETURN result_char
    END FUNCTION
END CLASS
