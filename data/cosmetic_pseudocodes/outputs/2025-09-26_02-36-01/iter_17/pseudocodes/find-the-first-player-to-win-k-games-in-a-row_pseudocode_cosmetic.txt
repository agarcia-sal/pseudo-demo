CLASS Solution  
    FUNCTION findWinningPlayer(skills, k)  
        LET lengthVar = (SKILLS LENGTH)  
        LET indexCollection = CREATE_EMPTY_LIST()  
        LET counter = 0  
        LET leader = NULL  
        LET i = 0  
        REPEAT  
            IF i < lengthVar THEN  
                APPEND i TO indexCollection  
                i = i + 1  
            ELSE  
                BREAK  
            END IF  
        UNTIL FALSE  
  
        LET battlesWon = 0  
        leader = REMOVE_FIRST_ELEMENT(indexCollection)  
  
        /* Main loop substitutions */  
        WHILE (battlesWon < k) AND (indexCollection LENGTH > 0) DO  
            LET challenger = REMOVE_FIRST_ELEMENT(indexCollection)  
            LET leaderSkill = skills[leader]  
            LET challengerSkill = skills[challenger]  
  
            LET leaderBeats = FALSE  
            IF leaderSkill > challengerSkill THEN  
                leaderBeats = TRUE  
            END IF  
  
            IF leaderBeats = TRUE THEN  
                battlesWon = battlesWon + 1  
                APPEND challenger TO indexCollection  
            ELSE  
                battlesWon = 1  
                APPEND leader TO indexCollection  
                leader = challenger  
            END IF  
        END WHILE  
  
        RETURN leader  
    END FUNCTION  
END CLASS
