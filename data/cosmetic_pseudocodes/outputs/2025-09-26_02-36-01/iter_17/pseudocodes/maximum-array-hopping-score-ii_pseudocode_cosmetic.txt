CLASS Solution  
    FUNCTION maxScore(nums)  
        DECLARE x AS INTEGER ← LENGTH(nums)  
        DECLARE y AS LIST OF INTEGER ← EMPTY LIST  
        WHILE LENGTH(y) < x DO  
            APPEND 0 TO y  
        END WHILE  
        DECLARE a AS INTEGER ← x - 2  
        REPEAT  
            IF a < 0 THEN BREAK END IF  
            DECLARE b AS INTEGER ← 0  
            DECLARE c AS INTEGER ← a + 1  
            WHILE c < x DO  
                DECLARE d AS INTEGER ← c - a  
                DECLARE e AS INTEGER ← nums[c] * d  
                DECLARE f AS INTEGER ← e + y[c]  
                IF b < f THEN  
                    b ← f  
                END IF  
                c ← c + 1  
            END WHILE  
            y[a] ← b  
            a ← a - 1  
        UNTIL FALSE  
        RETURN y[0]  
    END FUNCTION  
END CLASS
