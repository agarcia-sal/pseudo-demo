CLASS Solution
    FUNCTION flowerGame(n, m)
        BEGIN
            DECLARE xZqJ AS INTEGER
            DECLARE LyRt AS INTEGER
            DECLARE BlUw AS INTEGER
            DECLARE FyKG AS INTEGER
            DECLARE HsJN AS INTEGER

            xZqJ = (n + 1) / 2
            LyRt = n / 2
            BlUw = (m + 1) / 2
            FyKG = m / 2

            HsJN = (xZqJ * FyKG) + (LyRt * BlUw)

            RETURN HsJN
        END
END CLASS
