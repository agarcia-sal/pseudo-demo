CLASS Solution  
    FUNCTION maxOperations(s)  
        DECLARE result AS INTEGER := 0  
        DECLARE tally AS INTEGER := 0  
        DECLARE pos AS INTEGER := 0  
        WHILE pos < LENGTH(s) DO  
            BEGIN  
                DECLARE currentChar AS CHAR := s[pos]  
                IF currentChar = "1" THEN  
                    tally = tally + 1  
                ELSE   
                    IF pos > 0 THEN  
                        DECLARE prevChar AS CHAR := s[pos - 1]  
                        IF prevChar = "1" THEN  
                            result = result + tally  
                        END IF  
                    END IF  
                END IF  
                pos = pos + 1  
            END  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
