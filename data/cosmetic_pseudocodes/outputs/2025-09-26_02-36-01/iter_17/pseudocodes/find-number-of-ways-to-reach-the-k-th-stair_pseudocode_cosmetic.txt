CLASS Solution
    FUNCTION waysToReachStair WITH PARAMETER k
        FUNCTION dfs WITH PARAMETERS i j jump
            BEGIN
                IF NOT (i <= k + 1) THEN
                    RETURN 0
                END IF

                DECLARE xwra AS INTEGER
                SET xwra TO IF (i = k) THEN 1 ELSE 0

                IF i > 0 THEN
                    IF j = 0 THEN
                        SET xwra TO xwra + dfs(i - 1, 1, jump)
                    END IF
                END IF

                SET xwra TO xwra + dfs(i + (1 * 2 * jump), 0, jump + 1)
                RETURN xwra
            END
        END FUNCTION

        RETURN dfs(1, 0, 0)
    END FUNCTION
END CLASS
