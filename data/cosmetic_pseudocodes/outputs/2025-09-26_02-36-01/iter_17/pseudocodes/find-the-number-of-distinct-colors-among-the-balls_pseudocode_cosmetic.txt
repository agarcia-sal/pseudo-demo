CLASS Solution
    FUNCTION queryResults(limit parameter, queries parameter)
        DECLARE FQDK TO an empty map
        DECLARE UHJM TO an empty set
        DECLARE QTJN TO an empty sequence
        DECLARE IORX TO 0

        WHILE IORX < LENGTH(queries parameter)
            BEGIN
            DECLARE VNBW TO queries parameter[IORX][0]
            DECLARE ZCME TO queries parameter[IORX][1]

            IF MAP_HAS_KEY(FQDK, VNBW)
                BEGIN
                DECLARE LWYO TO MAP_GET(FQDK, VNBW)
                IF SET_CONTAINS(UHJM, LWYO)
                    SET = SET_REMOVE(UHJM, LWYO)
                END IF
                END
            END IF

            SET = MAP_SET(FQDK, VNBW, ZCME)
            SET = SET_ADD(UHJM, ZCME)
            SEQUENCE_APPEND(QTJN, SET_SIZE(UHJM))

            SET IORX TO IORX + 1
            END
        RETURN QTJN
    END FUNCTION
END CLASS
