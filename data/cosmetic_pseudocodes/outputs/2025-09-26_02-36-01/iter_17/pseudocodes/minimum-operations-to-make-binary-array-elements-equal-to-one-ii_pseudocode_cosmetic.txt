CLASS Solution
    FUNCTION minOperations(nums)
        SET totalOps TO 0
        SET toggleFlag TO 0
        SET index TO 0
        WHILE index < LENGTH(nums)
            {
            SET valAtIndex TO 0
            SET modResult TO toggleFlag
            SET modResult TO modResult - 2 * (INT(modResult / 2))
            IF modResult = 0
                SET valAtIndex TO nums[index]
            ELSE
                SET valAtIndex TO 1 - nums[index]
            END IF
            IF valAtIndex = 0
                SET totalOps TO totalOps + 1
                SET toggleFlag TO toggleFlag + 1
            END IF
            SET index TO index + 1
            }
        RETURN totalOps
    END FUNCTION
END CLASS
