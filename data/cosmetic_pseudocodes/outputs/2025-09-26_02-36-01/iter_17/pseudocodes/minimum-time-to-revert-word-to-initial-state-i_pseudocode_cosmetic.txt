CLASS Solution
    FUNCTION minimumTimeToInitialState(word PARAMETER k)
        SET lengthValue TO LENGTH OF word
        SET counter TO 1
        REPEAT
            BEGIN
                SET productValue TO counter * k
                SET substringA TO SUBSTRING(word, productValue, lengthValue - productValue)
                SET substringB TO SUBSTRING(word, 1, lengthValue - productValue)
                IF (productValue >= lengthValue) OR (substringA = substringB) THEN
                    RETURN counter
                END IF
                SET counter TO counter + 1
            END
        UNTIL FALSE
    END FUNCTION
END CLASS
