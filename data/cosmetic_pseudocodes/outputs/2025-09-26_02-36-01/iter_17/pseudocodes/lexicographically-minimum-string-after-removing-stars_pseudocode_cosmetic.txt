CLASS Solution
    FUNCTION clearStars(s)
        SET Xkqv TO NEW LIST
        SET i_otcdwb TO 0
        WHILE i_otcdwb < LENGTH(s)
            SET wqtxm TO s[i_otcdwb]
            IF wqtxm != '*'
                BEGIN
                    APPEND wqtxm TO Xkqv
                END
            ELSE
                BEGIN
                    IF LENGTH(Xkqv) > 0
                        BEGIN
                            REMOVE Xkqv[LENGTH(Xkqv) - 1]
                        END
                END
            SET i_otcdwb TO i_otcdwb + 1
        END WHILE
        SET tpmn_swivs TO ""
        FOR EACH el IN Xkqv
            BEGIN
                SET tpmn_swivs TO tpmn_swivs + el
            END
        END FOR
        RETURN tpmn_swivs
    END FUNCTION
END CLASS
