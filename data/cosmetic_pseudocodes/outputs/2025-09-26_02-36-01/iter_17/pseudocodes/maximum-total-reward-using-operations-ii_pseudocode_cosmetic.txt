CLASS Solution
    FUNCTION maxTotalReward(rewardValues)
        DECLARE uniqueElements AS ARRAY
        DECLARE accBitMask AS INTEGER
        DECLARE tempVal AS INTEGER
        DECLARE idx AS INTEGER

        uniqueElements = []
        FOR i FROM 0 TO LENGTH(rewardValues) - 1
            IF NOT (rewardValues[i] IN uniqueElements) THEN
                uniqueElements = uniqueElements + [rewardValues[i]]
            END IF
        END FOR
        uniqueElements = SORT(uniqueElements)

        accBitMask = 1
        idx = 0
        WHILE idx < LENGTH(uniqueElements)
            tempVal = 1 << uniqueElements[idx]
            accBitMask = accBitMask OR ((accBitMask AND (tempVal - 1)) << uniqueElements[idx])
            idx = idx + 1
        END WHILE

        RETURN BIT_LENGTH(accBitMask) - 1
    END FUNCTION
END CLASS
