CLASS Solution  
    FUNCTION maximumLength(nums k)  
        SET countElements TO 0  
        WHILE countElements < LENGTH OF nums  
            INCREMENT countElements BY 1  
        END WHILE  
        
        IF NOT (countElements NOT EQUAL TO 1) THEN  
            RETURN 1  
        END IF  
        
        SET dp TO EMPTY LIST  
        SET idx TO 0  
        WHILE idx < countElements  
            APPEND EMPTY MAPPING TO dp  
            SET idx TO idx + 1  
        END WHILE  
        
        SET longestSeq TO 1  
        
        SET outerIndex TO 0  
        WHILE outerIndex < countElements  
            SET innerIndex TO 0  
            WHILE innerIndex < outerIndex  
                SET val1 TO nums[outerIndex]  
                SET val2 TO nums[innerIndex]  
                SET sumMod TO (val2 + val1) - k * ((val2 + val1) / k)  
                
                IF dp[innerIndex] CONTAINS KEY sumMod THEN  
                    SET updatedLength TO dp[innerIndex][sumMod] + 1  
                    SET dp[outerIndex][sumMod] TO updatedLength  
                ELSE  
                    SET dp[outerIndex][sumMod] TO 2  
                END IF  
                
                IF dp[outerIndex][sumMod] > longestSeq THEN  
                    SET longestSeq TO dp[outerIndex][sumMod]  
                END IF  
                SET innerIndex TO innerIndex + 1  
            END WHILE  
            SET outerIndex TO outerIndex + 1  
        END WHILE  
        
        RETURN longestSeq  
    END FUNCTION  
END CLASS
