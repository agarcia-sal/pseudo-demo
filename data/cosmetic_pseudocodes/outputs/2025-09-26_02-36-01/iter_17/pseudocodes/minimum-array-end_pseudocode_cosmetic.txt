CLASS Solution
    FUNCTION minEnd(n x)
        FUNCTION canConstruct(max_val)
            SET pqpwax TO x
            SET yikeqg TO 1
            LOOP
                IF NOT (pqpwax < max_val) THEN
                    BREAK
                END IF
                SET pqpwax TO pqpwax + 1
                SET rlmzaj TO (pqpwax BIT_AND x)
                IF rlmzaj = x THEN
                    SET yikeqg TO yikeqg + 1
                    IF yikeqg = n THEN
                        RETURN True
                    END IF
                END IF
            END LOOP
            RETURN (yikeqg = n)
        END FUNCTION

        SET evrsti TO x
        SET pidgwf TO ((((2 * 10) * 10) * 10) * 10) * 10
        SET pidgwf TO (((pidgwf * 10) * 10) * 10)
        SET pidgwf TO pidgwf * 10
        SET pidgwf TO pidgwf * 10
        SET pidgwf TO pidgwf * 10
        SET pidgwf TO pidgwf * 10
        SET pidgwf TO pidgwf * 10

        REPEAT
            SET wtrmly TO (evrsti + pidgwf) / 2
            SET wtrmly TO FLOOR(wtrmly)
            IF canConstruct(wtrmly) THEN
                SET pidgwf TO wtrmly
            ELSE
                SET evrsti TO evrsti + 1
            END IF
        UNTIL NOT (evrsti < pidgwf)

        RETURN evrsti
    END FUNCTION
END CLASS
