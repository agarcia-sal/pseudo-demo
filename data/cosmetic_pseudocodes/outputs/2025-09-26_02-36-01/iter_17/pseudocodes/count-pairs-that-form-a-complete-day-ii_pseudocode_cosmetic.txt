CLASS Solution
    FUNCTION countCompleteDayPairs WITH parameter hours
        SET freq_map TO empty dictionary with default 0 values
        SET total_pairs TO 0
        SET idx TO 0
        WHILE idx < LENGTH OF hours DO
            BEGIN
                SET current_hour TO hours[idx]
                SET rem_val TO current_hour - ( (current_hour / 24) * 24 )
                SET comp_val TO (24 - rem_val) - ( ((24 - rem_val) / 24) * 24 )
                SET total_pairs TO total_pairs + freq_map[comp_val]
                SET freq_map[rem_val] TO freq_map[rem_val] + 1
                SET idx TO idx + 1
            END
        END WHILE
        RETURN total_pairs
    END FUNCTION
END CLASS
