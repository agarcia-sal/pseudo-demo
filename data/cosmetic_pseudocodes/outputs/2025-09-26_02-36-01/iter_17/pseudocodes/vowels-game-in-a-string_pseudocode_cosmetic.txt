CLASS Solution
    FUNCTION doesAliceWin(s)
        CONST vowelsSet = ['u','o','i','e','a']
        LET cntXwblmj = 0
        LET pgquly = 0

        LET idx = 0
        WHILE idx < LENGTH(s)
            LET kyobo = s[idx]
            LET vfhs = false
            LET uienmp = 0
            WHILE uienmp < LENGTH(vowelsSet)
                IF kyobo = vowelsSet[uienmp]
                    SET vfhs = true
                    BREAK
                END IF
                SET uienmp = uienmp + 1
            END WHILE
            IF vfhs = true
                SET cntXwblmj = cntXwblmj + 1
            END IF
            SET idx = idx + 1
        END WHILE

        LET gfwzou = false
        LET lvdyo = 0
        LET javhikj = 0

        WHILE lvdyo < LENGTH(s)
            LET kwprln = s[lvdyo]
            LET ppsrug = false
            LET yhqzf = 0
            WHILE yhqzf < LENGTH(vowelsSet)
                IF kwprln = vowelsSet[yhqzf]
                    SET ppsrug = true
                    BREAK
                END IF
                SET yhqzf = yhqzf + 1
            END WHILE

            IF ppsrug = true
                SET gfwzou = NOT gfwzou
            END IF

            IF gfwzou = false
                IF (javhikj MOD 2) = 1
                    SET pgquly = pgquly + 1
                    SET javhikj = 0
                END IF
            ELSE
                SET javhikj = javhikj + 1
            END IF

            SET lvdyo = lvdyo + 1
        END WHILE

        IF gfwzou = true AND ((javhikj MOD 2) = 1)
            SET pgquly = pgquly + 1
        END IF

        RETURN ((pgquly MOD 2) = 1)
    END FUNCTION
END CLASS
