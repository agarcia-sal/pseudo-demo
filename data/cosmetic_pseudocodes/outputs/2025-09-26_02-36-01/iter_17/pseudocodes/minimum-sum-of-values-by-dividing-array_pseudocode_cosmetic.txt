CLASS Solution
    FUNCTION minimumValueSum(nums, andValues)
        SET a TO LENGTH(nums)
        SET b TO LENGTH(andValues)

        FUNCTION dp(x, y)
            IF y = -1 THEN
                IF x = -1 THEN
                    RETURN 0
                ELSE
                    RETURN ∞
                END IF
            END IF

            IF x = -1 THEN
                RETURN ∞
            END IF

            SET z TO ∞
            SET w TO -1

            SET idx TO x
            WHILE idx >= -1 DO
                IF w = -1 THEN
                    SET w TO nums[idx]
                ELSE
                    SET w TO w AND nums[idx]
                END IF

                IF w = andValues[y] THEN
                    SET temp TO dp(idx - 1, y - 1)
                    SET temp2 TO temp + nums[x]
                    IF temp2 < z THEN
                        SET z TO temp2
                    END IF
                END IF

                SET idx TO idx - 1
            END WHILE

            RETURN z
        END FUNCTION

        SET output TO dp(a - 1, b - 1)
        IF NOT (output = ∞) THEN
            RETURN output
        ELSE
            RETURN -1
        END IF
    END FUNCTION
END CLASS
