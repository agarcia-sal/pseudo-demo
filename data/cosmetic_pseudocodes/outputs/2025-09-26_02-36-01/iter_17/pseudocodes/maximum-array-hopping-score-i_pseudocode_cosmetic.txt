CLASS Solution  
    FUNCTION maxScore(nums)  
        LET a TO LENGTH(nums)  
        LET b TO ARRAY_OF_SIZE(a) INITIALIZED_WITH(0)  
        b[0] = 0  
        LET x TO 2  
        WHILE x <= a DO  
        BEGIN  
            LET y TO 1  
            WHILE y < x DO  
            BEGIN  
                LET u TO b[y - 1] + (x - y) * nums[x - 1]  
                IF b[x - 1] < u THEN  
                    b[x - 1] = u  
                END IF  
                y = y + 1  
            END  
            x = x + 1  
        END  
        RETURN b[a - 1]  
    END FUNCTION  
END CLASS
