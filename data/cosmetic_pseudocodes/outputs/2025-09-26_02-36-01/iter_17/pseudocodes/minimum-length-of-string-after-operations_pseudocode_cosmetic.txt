CLASS Solution  
    FUNCTION minimumLength(s)  
        SET frequencyMap TO Counter(s)  
        SET oddCountSum TO 0  
        SET evenCountSum TO 0  
        
        SET iteratorValues TO frequencyMap.values()  
        SET index TO 0  
        WHILE index < LENGTH(iteratorValues)  
            SET currentValue TO iteratorValues[index]  
            SET remainder TO currentValue - 2 * (currentValue / 2)  
            IF remainder = 1 THEN  
                SET oddCountSum TO oddCountSum + 1  
            ELSE   
                IF NOT (remainder = 1) AND currentValue <> 0 THEN  
                    SET evenCountSum TO evenCountSum + 2  
                END IF  
            END IF  
            SET index TO index + 1  
        END WHILE  
        
        SET totalResult TO oddCountSum + evenCountSum  
        RETURN totalResult  
    END FUNCTION  
END CLASS
