CLASS Solution
    FUNCTION numberOfRightTriangles(grid)
        SET a TO 0
        SET b TO LENGTH(grid)
        SET c TO LENGTH(grid[0])
        SET d TO 0
        SET e TO 0
        SET f TO 0
        SET g TO 0
        SET h TO 0
        SET i_outer TO 0

        WHILE i_outer < b DO
            SET j_inner TO 0
            WHILE j_inner < c DO
                IF (grid[i_outer])[j_inner] = 1 THEN
                    SET k1 TO 0
                    SET m1 TO 0
                    WHILE k1 < c DO
                        IF k1 <> j_inner THEN
                            SET m1 TO m1 + (grid[i_outer])[k1]
                        END IF
                        SET k1 TO k1 + 1
                    END WHILE

                    SET k2 TO 0
                    SET m2 TO 0
                    WHILE k2 < b DO
                        IF k2 <> i_outer THEN
                            SET m2 TO m2 + (grid[k2])[j_inner]
                        END IF
                        SET k2 TO k2 + 1
                    END WHILE

                    SET d TO d + (m1 * m2)
                END IF
                SET j_inner TO j_inner + 1
            END WHILE
            SET i_outer TO i_outer + 1
        END WHILE
        RETURN d
    END FUNCTION
END CLASS
