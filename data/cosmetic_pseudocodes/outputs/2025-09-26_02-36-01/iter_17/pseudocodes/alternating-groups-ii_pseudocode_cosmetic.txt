CLASS Solution
    FUNCTION numberOfAlternatingGroups(colors k)
        IF k < 3 THEN
            RETURN 0
        END IF

        SET lengthColors TO LENGTH(colors)
        SET repeatedColors TO colors + SLICE(colors, 0, k - 1)

        SET total TO 0
        SET outerIndex TO 0

        WHILE outerIndex < lengthColors DO
            SET alternatingFlag TO 1
            SET innerOffset TO 1

            REPEAT
                IF repeatedColors[outerIndex + innerOffset] = repeatedColors[outerIndex + innerOffset - 1] THEN
                    SET alternatingFlag TO 0
                    BREAK
                END IF

                SET innerOffset TO innerOffset + 1
            UNTIL innerOffset = k

            IF alternatingFlag = 1 THEN
                SET total TO total + 1
            END IF

            SET outerIndex TO outerIndex + 1
        END WHILE

        RETURN total
    END FUNCTION
END CLASS
