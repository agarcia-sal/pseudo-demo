CLASS Solution
    FUNCTION numberOfSubstrings(s)
        SET length_s TO LENGTH(s)
        SET total_substr TO 0
        SET idx1 TO 0
        WHILE idx1 LESS_THAN length_s
            SET cnt_one TO 0
            SET cnt_zero TO 0
            SET idx2 TO idx1
            REPEAT
                IF s[idx2] EQUALS '1' THEN
                    SET cnt_one TO cnt_one + 1
                ELSE
                    SET cnt_zero TO cnt_zero + 1
                END IF
                IF cnt_one GREATER_THAN_OR_EQUAL_TO cnt_zero * cnt_zero THEN
                    SET total_substr TO total_substr + 1
                END IF
                SET idx2 TO idx2 + 1
            UNTIL idx2 EQUALS length_s
            SET idx1 TO idx1 + 1
        END WHILE
        RETURN total_substr
    END FUNCTION
END CLASS
