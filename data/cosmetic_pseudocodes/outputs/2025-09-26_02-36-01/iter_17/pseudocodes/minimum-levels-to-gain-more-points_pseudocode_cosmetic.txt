CLASS Solution
    FUNCTION minimumLevels(possible)
        SET wqzmnv TO 0
        SET vjkwcm TO 0
        SET mnkbzd TO 0
        WHILE mnkbzd < LENGTH(possible)
            SET wqzmnv TO wqzmnv + (2 * possible[mnkbzd] - 1)
            SET mnkbzd TO mnkbzd + 1
        END WHILE

        SET ieykpb TO 0
        SET torxjn TO 0
        REPEAT
            IF torxjn >= LENGTH(possible) - 1 THEN
                BREAK
            END IF

            SET ieykpb TO ieykpb + (2 * possible[torxjn] - 1)
            SET wqzmnv TO wqzmnv - (2 * possible[torxjn] - 1)

            IF ieykpb > wqzmnv THEN
                RETURN torxjn + 1
            END IF

            SET torxjn TO torxjn + 1
        UNTIL FALSE

        RETURN -1
    END FUNCTION
END CLASS
