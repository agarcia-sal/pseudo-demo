CLASS Solution
    FUNCTION minStartingIndex WITH PARAMETERS s pattern
        FUNCTION is_almost_equal WITH PARAMETERS substring pattern
            SET delta TO 0
            SET pos_sub TO 0
            WHILE pos_sub LESS THAN LENGTH(substring)
                SET ch_sub TO substring[pos_sub]
                SET ch_pat TO pattern[pos_sub]
                IF NOT (ch_sub = ch_pat) THEN
                    SET delta TO delta + 1
                    IF delta > 1 THEN
                        RETURN False
                    END IF
                END IF
                SET pos_sub TO pos_sub + 1
            END WHILE
            RETURN True
        END FUNCTION

        SET limit TO LENGTH(s) - LENGTH(pattern)
        SET idx TO 0
        REPEAT
            IF idx > limit THEN
                BREAK
            END IF
            SET segment TO []
            SET j TO idx
            WHILE j < idx + LENGTH(pattern)
                APPEND s[j] TO segment
                SET j TO j + 1
            END WHILE
            IF is_almost_equal(segment, pattern) THEN
                RETURN idx
            END IF
            SET idx TO idx + 1
        UNTIL False

        RETURN -1
    END FUNCTION
END CLASS
