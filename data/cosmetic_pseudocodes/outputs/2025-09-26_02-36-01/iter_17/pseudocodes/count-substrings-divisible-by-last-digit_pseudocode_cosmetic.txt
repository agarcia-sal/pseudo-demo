CLASS Solution
    FUNCTION countSubstrings(s)
        SET lengthOfStr TO 0
        WHILE lengthOfStr LESS THAN LENGTH OF s
            SET lengthOfStr TO lengthOfStr + 1
        END WHILE

        SET aggregateCount TO 0
        SET outerIndex TO 0

        REPEAT
            IF outerIndex GREATER THAN OR EQUAL TO lengthOfStr
                BREAK
            END IF

            SET numericAccumulator TO 0
            SET innerIndex TO outerIndex

            WHILE innerIndex LESS THAN lengthOfStr
                SET digitChar TO s AT POSITION innerIndex
                SET digitVal TO INTEGER VALUE OF digitChar
                SET numericAccumulator TO (numericAccumulator TIMES 10) + digitVal

                IF digitVal NOT EQUAL TO 0
                    IF (numericAccumulator MOD digitVal) EQUALS 0
                        SET aggregateCount TO aggregateCount + 1
                    END IF
                END IF

                SET innerIndex TO innerIndex + 1
            END WHILE

            SET outerIndex TO outerIndex + 1
        UNTIL FALSE

        RETURN aggregateCount
    END FUNCTION
END CLASS
