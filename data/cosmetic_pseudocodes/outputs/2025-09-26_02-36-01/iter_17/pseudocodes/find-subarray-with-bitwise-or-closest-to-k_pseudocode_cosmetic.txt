CLASS Solution
    FUNCTION minimumDifference(nums k)
        FUNCTION subarray_or(x y)
            SET acc TO 0
            SET idx TO x
            WHILE idx <= y DO
                SET acc = acc BITWISE_OR nums[idx]
                SET idx = idx + 1
            END WHILE
            RETURN acc
        END FUNCTION

        SET length TO LENGTH OF nums
        SET minimal_gap TO +âˆž

        SET a TO 0
        WHILE a <= length - 1 DO
            SET running_or TO 0
            SET b TO a
            WHILE b <= length - 1 DO
                SET running_or = running_or BITWISE_OR nums[b]
                SET gap TO (k - running_or)
                IF gap < 0 THEN
                    SET gap = -gap
                END IF
                IF gap < minimal_gap THEN
                    SET minimal_gap = gap
                END IF
                IF gap = 0 THEN
                    RETURN 0
                END IF
                SET b = b + 1
            END WHILE
            SET a = a + 1
        END WHILE

        RETURN minimal_gap
    END FUNCTION
END CLASS
