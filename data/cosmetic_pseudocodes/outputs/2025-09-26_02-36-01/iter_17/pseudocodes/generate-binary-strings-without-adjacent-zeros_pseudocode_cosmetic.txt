CLASS Solution  
    FUNCTION validStrings n  
        DECLARE FUNCTION backtrack tempStr  
            DECLARE lenReached BOOLEAN  
            SET lenReached TO (LENGTH(tempStr) EQUALS n)  
            
            IF lenReached THEN
                APPEND tempStr TO resultCollection  
                RETURN  
            END IF  
            
            DECLARE lastChar CHAR  
            SET lastChar TO CHARACTER_AT(tempStr, LENGTH(tempStr) MINUS 1)  
            
            IF lastChar EQUALS "1" THEN  
                backtrack(tempStr PLUS "0")  
                backtrack(tempStr PLUS "1")  
            ELSE  
                IF lastChar EQUALS "0" THEN  
                    backtrack(tempStr PLUS "1")  
                END IF  
            END IF  
        END FUNCTION  
        
        SET resultCollection TO NEW LIST()  
        backtrack("0")  
        backtrack("1")  
        RETURN resultCollection  
    END FUNCTION  
END CLASS
