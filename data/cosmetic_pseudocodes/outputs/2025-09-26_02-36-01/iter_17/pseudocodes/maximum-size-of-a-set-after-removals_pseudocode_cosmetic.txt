CLASS Solution  
    FUNCTION maximumSetSize(nums1 PARAMETER List OF integers nums2 PARAMETER List OF integers) RETURNS integer  
        BEGIN  
            DECLARE xAyz INTEGER  
            DECLARE pOqi INTEGER  
            DECLARE dLve SET OF integer  
            DECLARE yupu SET OF integer  
            DECLARE qvMw SET OF integer  
            DECLARE kXsn SET OF integer  
            DECLARE bLtw INTEGER  
            DECLARE tHzg INTEGER  
            DECLARE rZvQ INTEGER  
            DECLARE wEfi INTEGER  
            DECLARE vMnr INTEGER  
            
            SET xAyz = LENGTH(nums1)  
            SET pOqi = xAyz / 2  
            
            SET dLve = UNIQUE(nums1)  
            SET yupu = UNIQUE(nums2)  
            
            SET qvMw = dLve INTERSECT yupu  
            SET kXsn = dLve MINUS qvMw  
            SET bLtw = yupu MINUS qvMw  
            
            IF pOqi < LENGTH(kXsn) THEN  
                SET tHzg = pOqi  
            ELSE  
                SET tHzg = LENGTH(kXsn)  
            END IF  
            
            IF pOqi < LENGTH(bLtw) THEN  
                SET rZvQ = pOqi  
            ELSE  
                SET rZvQ = LENGTH(bLtw)  
            END IF  
            
            SET wEfi = (pOqi - tHzg)  
            IF wEfi < 0 THEN  
                SET wEfi = 0  
            END IF  
            
            SET vMnr = (pOqi - rZvQ)  
            IF vMnr < 0 THEN  
                SET vMnr = 0  
            END IF  
            
            SET wEfi = wEfi + vMnr  
            
            IF wEfi < LENGTH(qvMw) THEN  
                SET wEfi = wEfi  
            ELSE  
                SET wEfi = LENGTH(qvMw)  
            END IF  
            
            RETURN tHzg + rZvQ + wEfi  
        END  
END CLASS
