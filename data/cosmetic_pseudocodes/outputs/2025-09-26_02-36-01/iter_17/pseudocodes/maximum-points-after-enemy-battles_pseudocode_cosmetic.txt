CLASS Solution
    FUNCTION maximumPoints(enemyEnergies currentEnergy)
        DECLARE sortedList = []
        DECLARE resultCounter = 0
        DECLARE idx = 0
        BEGIN
            SET sortedList = enemyEnergies
            CALL sortAscending(sortedList)
            IF NOT (currentEnergy >= sortedList[0]) THEN
                RETURN 0
            END IF
            SET idx = LENGTH(sortedList) - 1
            WHILE idx >= 0 DO
                DECLARE divResult = INTEGER_DIVIDE(currentEnergy, sortedList[0])
                SET resultCounter = resultCounter + divResult
                SET currentEnergy = MOD(currentEnergy, sortedList[0])
                SET currentEnergy = currentEnergy + sortedList[idx]
                SET idx = idx - 1
            END WHILE
            RETURN resultCounter
        END
    END FUNCTION
END CLASS
