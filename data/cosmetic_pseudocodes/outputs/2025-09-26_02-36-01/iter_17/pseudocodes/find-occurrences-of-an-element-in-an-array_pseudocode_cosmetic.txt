CLASS Solution  
    FUNCTION occurrencesOfElement(nums, queries, x)  
        DECLARE accum_1 AS LIST  
        SET accum_1 TO []  
        DECLARE pos_2 AS INTEGER  
        DECLARE val_3 AS INTEGER  
        SET pos_2 TO 0  
        WHILE pos_2 < LENGTH(nums)  
            SET val_3 TO nums[pos_2]  
            IF NOT(val_3 != x) THEN  
                BEGIN  
                    SET accum_1 TO accum_1 + [pos_2]  
                END  
            END IF  
            SET pos_2 TO pos_2 + 1  
        END WHILE  
        DECLARE accum_4 AS LIST  
        SET accum_4 TO []  
        DECLARE idx_5 AS INTEGER  
        SET idx_5 TO 0  
        REPEAT  
            IF idx_5 >= LENGTH(queries) THEN  
                BREAK  
            END IF  
            DECLARE qry_6 AS INTEGER  
            SET qry_6 TO queries[idx_5]  
            IF NOT(qry_6 > LENGTH(accum_1)) THEN  
                BEGIN  
                    DECLARE elem_7 AS INTEGER  
                    SET elem_7 TO accum_1[qry_6 - 1]  
                    SET accum_4 TO accum_4 + [elem_7]  
                END  
            ELSE  
                SET accum_4 TO accum_4 + [-1]  
            END IF  
            SET idx_5 TO idx_5 + 1  
        UNTIL FALSE  
        RETURN accum_4  
    END FUNCTION  
END CLASS
