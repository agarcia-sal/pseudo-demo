CLASS Solution  
    FUNCTION maxNumber(n)  
        IF NOT (n NOT EQUALS 1) THEN  
            RETURN 0  
        END IF  

        DECLARE a  
        a = 1  

        WHILE TRUE DO  
            IF a > n THEN  
                BREAK  
            END IF  
            a = a * 2  
        END WHILE  

        DECLARE b = a / 2  
        DECLARE c = b - 1  

        RETURN c  
    END FUNCTION  
END CLASS
