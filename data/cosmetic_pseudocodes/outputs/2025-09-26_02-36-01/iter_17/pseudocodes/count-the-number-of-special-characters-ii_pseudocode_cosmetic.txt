CLASS Solution  
    FUNCTION numberOfSpecialChars(word)  
        BEGIN  
            DECLARE mapA AS EMPTY KEY VALUE COLLECTION  
            DECLARE mapB AS EMPTY KEY VALUE COLLECTION  
            DECLARE idx1 AS INTEGER  
            DECLARE ch1 AS CHARACTER  
            DECLARE idx2 AS INTEGER  
            DECLARE ch2 AS CHARACTER  
            DECLARE ctr AS INTEGER  
            DECLARE iter1 AS INTEGER  
            DECLARE iter2 AS INTEGER  
            ITERATE idx1, ch1 IN word WITH INDEX  
                IF NOT EXISTS KEY ch1 IN mapA THEN  
                    ASSIGN mapA[ch1] = idx1  
                END IF  
                ASSIGN mapB[ch1] = idx1  
            END ITERATE  
            ASSIGN ctr = 0  
            SET iter1 = 1  
            WHILE iter1 <= LENGTH("abcdefghijklmnopqrstuvwxyz") DO  
                SET iter2 = 1  
                WHILE iter2 <= LENGTH("ABCDEFGHIJKLMNOPQRSTUVWXYZ") DO  
                    ASSIGN ch1 = SUBSTRING("abcdefghijklmnopqrstuvwxyz", iter1, 1)  
                    ASSIGN ch2 = SUBSTRING("ABCDEFGHIJKLMNOPQRSTUVWXYZ", iter2, 1)  
                    DECLARE cond1 AS BOOLEAN  
                    DECLARE cond2 AS BOOLEAN  
                    DECLARE valA AS INTEGER  
                    DECLARE valB AS INTEGER  
                    ASSIGN cond1 = (ch1 IN mapB)  
                    ASSIGN cond2 = (ch2 IN mapA)  
                    IF cond1 AND cond2 THEN  
                        ASSIGN valA = mapB[ch1]  
                        ASSIGN valB = mapA[ch2]  
                        IF valA < valB THEN  
                            ASSIGN ctr = ctr + 1  
                        END IF  
                    END IF  
                    ASSIGN iter2 = iter2 + 1  
                END WHILE  
                ASSIGN iter1 = iter1 + 1  
            END WHILE  
            RETURN ctr  
        END  
    END FUNCTION  
END CLASS
