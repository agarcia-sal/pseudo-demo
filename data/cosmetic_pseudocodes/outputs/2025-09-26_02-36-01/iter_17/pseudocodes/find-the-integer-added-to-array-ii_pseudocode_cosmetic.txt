CLASS Solution
    FUNCTION minimumAddedInteger(nums1 RETURNS List OF integer, nums2 RETURNS List OF integer) RETURNS integer
        CALL sort ON nums1
        CALL sort ON nums2

        SET a TO 0
        WHILE a < LENGTH OF nums1 - 1
            SET b TO a + 1
            WHILE b < LENGTH OF nums1
                SET tempList TO EMPTY LIST

                SET c TO 0
                WHILE c < a
                    CALL append nums1[c] TO tempList
                    SET c TO c + 1
                END WHILE

                SET d TO a + 1
                WHILE d < b
                    CALL append nums1[d] TO tempList
                    SET d TO d + 1
                END WHILE

                SET e TO b + 1
                WHILE e < LENGTH OF nums1
                    CALL append nums1[e] TO tempList
                    SET e TO e + 1
                END WHILE

                SET offset TO nums2[0] - tempList[0]

                SET valid TO TRUE
                SET f TO 0
                WHILE f < LENGTH OF nums2
                    IF tempList[f] + offset != nums2[f] THEN
                        SET valid TO FALSE
                        SET f TO LENGTH OF nums2
                    ELSE
                        SET f TO f + 1
                    END IF
                END WHILE

                IF valid == TRUE
                    RETURN offset
                END IF

                SET b TO b + 1
            END WHILE
            SET a TO a + 1
        END WHILE

        RETURN None
    END FUNCTION
END CLASS
