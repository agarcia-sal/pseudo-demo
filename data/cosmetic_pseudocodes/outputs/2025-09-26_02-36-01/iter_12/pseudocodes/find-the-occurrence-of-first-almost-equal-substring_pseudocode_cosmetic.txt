CLASS Solution
    FUNCTION minStartingIndex WITH PARAMETERS s pattern
        FUNCTION is_almost_equal WITH PARAMETERS substring pattern
            SET n TO LENGTH(substring)
            SET m TO LENGTH(pattern)
            SET delta TO 0
            FUNCTION compare_chars WITH PARAMETERS a b
                RETURN NOT (a = b)
            END FUNCTION
            SET pos TO 0
            WHILE pos < n AND delta <= 1
                IF compare_chars(substring[pos], pattern[pos]) THEN
                    delta = delta + 1
                END IF
                pos = pos + 1
            END WHILE
            RETURN delta <= 1
        END FUNCTION

        SET plen TO LENGTH(pattern)
        SET result TO -1
        FUNCTION check_index WITH PARAMETERS idx
            IF is_almost_equal(Slice(s, idx, idx + plen - 1), pattern) THEN
                RETURN true
            ELSE
                RETURN false
            END IF
        END FUNCTION

        SET current TO 0
        SET found TO false
        REPEAT
            IF current + plen > LENGTH(s) THEN
                found = false
                BREAK
            END IF
            IF check_index(current) THEN
                found = true
                BREAK
            END IF
            current = current + 1
        UNTIL false

        IF found THEN
            result = current
        END IF
        RETURN result
    END FUNCTION
END CLASS
