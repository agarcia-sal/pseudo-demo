CLASS Solution
    FUNCTION flowerGame(n, m)
        FUNCTION halfRoundedUp(x)
            RETURN (x + 1) DIV 2
        END FUNCTION

        FUNCTION halfRoundedDown(x)
            RETURN x DIV 2
        END FUNCTION

        SET p TO halfRoundedUp(n)
        SET q TO halfRoundedDown(n)
        SET r TO halfRoundedUp(m)
        SET s TO halfRoundedDown(m)

        SET u TO (p * s) + (q * r)

        RETURN u
    END FUNCTION
END CLASS
