CLASS Solution
    FUNCTION countSubstrings(s c)
        VAR alpha_result <- 0
        VAR omega_length <- 0
        VAR delta_occurrences <- 0
        VAR auxiliary_sum <- 0

        PROCEDURE count_occurrences(input_string, target_char, OUT counter)
            VAR index_iter <- 0
            SET counter TO 0
            WHILE index_iter < LENGTH(input_string)
                IF input_string[index_iter] == target_char THEN
                    SET counter <- counter + 1
                END IF
                SET index_iter <- index_iter + 1
            END WHILE
        END PROCEDURE

        CALL count_occurrences(s, c, delta_occurrences)
        SET omega_length <- LENGTH(s)
        SET auxiliary_sum <- delta_occurrences + 1

        VAR intermediate_value <- 0
        PROCEDURE compute_half_sum(x, OUT result_val)
            SET result_val <- (x * auxiliary_sum) / 2
        END PROCEDURE

        CALL compute_half_sum(delta_occurrences, intermediate_value)
        SET alpha_result <- intermediate_value

        RETURN alpha_result
    END FUNCTION
END CLASS
