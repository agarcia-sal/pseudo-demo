CLASS Solution
    FUNCTION minMovesToCaptureTheQueen(a b c d e f)
        VAR resultKeeper
        PROCEDURE swapIfGreater(x y)
            IF x GREATER THAN y THEN
                VAR tempHolder
                tempHolder <- x
                x <- y
                y <- tempHolder
            END IF
        END PROCEDURE

        PROCEDURE isBetween(val start end)
            VAR low
            VAR high
            low <- start
            high <- end
            swapIfGreater(low high)
            RETURN (val GREATER THAN low AND val LESS THAN high)
        END PROCEDURE

        IF a EQUALS e THEN
            IF c NOT EQUALS a THEN
                resultKeeper <- 1
            ELSE
                IF (isBetween(d b f) OR isBetween(d f b)) THEN
                    resultKeeper <- 2
                ELSE
                    resultKeeper <- 1
                END IF
            END IF
            RETURN resultKeeper
        END IF

        IF b EQUALS f THEN
            IF d NOT EQUALS b THEN
                resultKeeper <- 1
            ELSE
                IF (isBetween(c a e) OR isBetween(c e a)) THEN
                    resultKeeper <- 2
                ELSE
                    resultKeeper <- 1
                END IF
            END IF
            RETURN resultKeeper
        END IF

        VAR sumCD
        VAR sumEF
        sumCD <- c PLUS d
        sumEF <- e PLUS f

        IF sumCD EQUALS sumEF THEN
            IF (a PLUS b) NOT EQUALS sumCD THEN
                resultKeeper <- 1
            ELSE
                IF ((c LESS THAN a AND a LESS THAN e AND d LESS THAN b AND b LESS THAN f) OR (e LESS THAN a AND a LESS THAN c AND f LESS THAN b AND b LESS THAN d)) THEN
                    resultKeeper <- 2
                ELSE
                    resultKeeper <- 1
                END IF
            END IF
            RETURN resultKeeper
        END IF

        VAR diffCD
        VAR diffEF
        diffCD <- c MINUS d
        diffEF <- e MINUS f

        IF diffCD EQUALS diffEF THEN
            IF (a MINUS b) NOT EQUALS diffCD THEN
                resultKeeper <- 1
            ELSE
                IF ((c LESS THAN a AND a LESS THAN e AND d GREATER THAN b AND b GREATER THAN f) OR (e LESS THAN a AND a LESS THAN c AND f GREATER THAN b AND b GREATER THAN d)) THEN
                    resultKeeper <- 2
                ELSE
                    resultKeeper <- 1
                END IF
            END IF
            RETURN resultKeeper
        END IF

        RETURN 2
    END FUNCTION
END CLASS
