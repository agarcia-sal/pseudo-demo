CLASS Solution
    FUNCTION minimumTimeToInitialState(word PARAMETER k)
        SET totalLength TO LENGTH(word)
        FUNCTION areSubstringsEqual(start1 PARAMETER s1, start2 PARAMETER s2, count PARAMETER c)
            SET index TO 0
            WHILE index < c
                IF word[s1 + index] NOT EQUAL TO word[s2 + index]
                    RETURN false
                END IF
                SET index TO index + 1
            END WHILE
            RETURN true
        END FUNCTION

        FUNCTION isConditionMet(currentTime PARAMETER t) RETURNS BOOLEAN
            SET position TO t * k
            IF position >= totalLength
                RETURN true
            END IF
            RETURN areSubstringsEqual(position, 0, totalLength - position)
        END FUNCTION

        SET accumulator TO 1
        SET conditionFlag TO false
        REPEAT
            SET conditionFlag TO isConditionMet(accumulator)
            IF conditionFlag
                RETURN accumulator
            END IF
            SET accumulator TO accumulator + 1
        UNTIL false
    END FUNCTION
END CLASS
