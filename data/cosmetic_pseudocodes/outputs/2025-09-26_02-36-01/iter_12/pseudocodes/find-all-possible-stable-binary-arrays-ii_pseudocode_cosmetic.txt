CLASS Solution
    FUNCTION numberOfStableArrays(zero zero, one one, limit limit) RETURNS integer
        SET CONSTRAINT TO (10 * 10 * 10 * 10 * 10 * 10 * 10 * 10 * 10) + 1

        FUNCTION recur(a a, b b, prev prev, streak streak) RETURNS integer
            IF (a = 0) AND (b = 0) THEN
                RETURN one
            END IF

            IF (a < 0) OR (b < 0) THEN
                RETURN zero
            END IF

            SET accumulator TO 0

            IF prev = zero THEN
                IF streak < limit THEN
                    SET accumulator TO accumulator + recur(a - 1, b, zero, streak + 1)
                END IF

                SET accumulator TO accumulator + recur(a, b - 1, one, 1)
            ELSE
                IF a > 0 THEN
                    SET accumulator TO accumulator + recur(a - 1, b, zero, 1)
                END IF

                IF streak < limit THEN
                    SET accumulator TO accumulator + recur(a, b - 1, one, streak + 1)
                END IF
            END IF

            SET remainder TO accumulator - ( (accumulator / CONSTRAINT) * CONSTRAINT )
            RETURN remainder
        END FUNCTION

        RETURN recur(zero, one, minus one, zero)
    END FUNCTION
END CLASS
