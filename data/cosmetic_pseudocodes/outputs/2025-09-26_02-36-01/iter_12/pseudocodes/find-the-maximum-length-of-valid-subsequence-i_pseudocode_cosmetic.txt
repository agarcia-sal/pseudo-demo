CLASS Solution

    FUNCTION maximumLength(nums)
        FUNCTION isEven(num)
            RETURN ((num MOD 2) = 0)
        END FUNCTION

        FUNCTION maxVal(a, b)
            IF a >= b THEN
                RETURN a
            ELSE
                RETURN b
            END IF
        END FUNCTION

        SET countE TO 0
        SET countO TO 0
        SET idx TO 1
        SET lenNums TO LENGTH(nums)
        
        WHILE idx < lenNums
            SET tempSum TO ADD(ACCESS(nums, idx - 1), ACCESS(nums, idx))
            IF isEven(tempSum) = TRUE THEN
                SET countE TO maxVal(ADD(countE, 1), countO)
            ELSE
                SET countO TO maxVal(ADD(countO, 1), countE)
            END IF
            SET idx TO ADD(idx, 1)
        END WHILE

        SET answerVar TO maxVal(countE, countO)
        SET finalResult TO ADD(answerVar, 1)

        RETURN finalResult
    END FUNCTION

    FUNCTION ADD(x, y)
        RETURN x + y
    END FUNCTION

    FUNCTION ACCESS(arr, pos)
        RETURN arr[pos]
    END FUNCTION

END CLASS
