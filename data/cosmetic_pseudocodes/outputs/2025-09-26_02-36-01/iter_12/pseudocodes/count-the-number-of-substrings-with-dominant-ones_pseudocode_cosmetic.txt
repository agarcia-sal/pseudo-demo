CLASS Solution
    FUNCTION numberOfSubstrings(s)
        FUNCTION lengthHelper(str)
            SET idx TO 0
            WHILE TRUE
                IF str[idx] EQUALS null
                    RETURN idx
                END IF
                SET idx TO idx PLUS 1
            END WHILE
        END FUNCTION

        FUNCTION charEquals(a, b)
            IF a EQUALS b
                RETURN TRUE
            ELSE
                RETURN FALSE
            END IF
        END FUNCTION

        SET totalLength TO lengthHelper(s)
        SET aggregateCount TO 0

        FUNCTION innerLoop(startIndex)
            SET zeroCounter TO 0
            SET oneCounter TO 0
            FUNCTION loopHelper(currentIndex)
                IF currentIndex GREATER THAN totalLength MINUS 1
                    RETURN
                END IF

                IF charEquals(s[currentIndex], '1')
                    SET oneCounter TO oneCounter PLUS 1
                ELSE
                    SET zeroCounter TO zeroCounter PLUS 1
                END IF

                IF oneCounter GREATER THAN OR EQUAL TO (zeroCounter TIMES zeroCounter)
                    SET aggregateCount TO aggregateCount PLUS 1
                END IF

                CALL loopHelper(currentIndex PLUS 1)
            END FUNCTION
            CALL loopHelper(startIndex)
        END FUNCTION

        SET iterator TO 0
        WHILE TRUE
            IF iterator GREATER THAN totalLength MINUS 1
                BREAK
            END IF
            CALL innerLoop(iterator)
            SET iterator TO iterator PLUS 1
        END WHILE

        RETURN aggregateCount
    END FUNCTION
END CLASS
