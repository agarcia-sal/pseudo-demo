CLASS Solution
    FUNCTION maxOperations(s)
        DEFINE FUNCTION isOne(x)
            RETURN (x = "1")
        END FUNCTION

        SET result = 0
        SET tally = 0
        SET position = 0

        WHILE position < LENGTH(s)
            SET currentChar = s[position]
            IF isOne(currentChar)
                SET tally = tally + 1
            ELSE 
                IF position > 0
                    SET prevChar = s[position - 1]
                    IF isOne(prevChar)
                        SET result = result + tally
                    END IF
                END IF
            END IF
            SET position = position + 1
        END WHILE

        RETURN result
    END FUNCTION
END CLASS
