CLASS Solution
    FUNCTION minOperations(k)
        FUNCTION intSqrt(n)
            SET r TO 0
            SET c TO 1
            WHILE c * c <= n
                SET r TO c
                SET c TO c + 1
            END WHILE
            RETURN r
        END FUNCTION

        FUNCTION intDivFloor(a, b)
            RETURN (a - (a MOD b)) / b
        END FUNCTION

        SET result TO 9223372036854775807  // very large number representing infinity
        SET idx TO 1
        SET limit TO intSqrt(k) + 1

        WHILE idx < limit
            SET numerator TO k + idx - 1
            SET denom TO idx
            SET valY TO intDivFloor(numerator, denom)
            SET sumOps TO (idx - 1) + (valY - 1)

            IF sumOps < result THEN
                SET result TO sumOps
            END IF

            SET idx TO idx + 1
        END WHILE

        RETURN result
    END FUNCTION
END CLASS
