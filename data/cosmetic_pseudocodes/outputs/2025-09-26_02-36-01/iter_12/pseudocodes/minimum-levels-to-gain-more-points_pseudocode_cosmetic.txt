CLASS Solution
    FUNCTION minimumLevels(possible)
        FUNCTION multiplyByTwo(value)
            RETURN value << 1
        END FUNCTION

        SET acc TO 0
        SET iterator TO 0

        WHILE iterator < LENGTH(possible)
            SET val TO possible[iterator]
            SET tmp TO multiplyByTwo(val) - 1
            SET acc TO acc + tmp
            SET iterator TO iterator + 1
        END WHILE

        SET sumA TO 0
        SET i TO 0
        WHILE i < LENGTH(possible) - 2 + 1
            SET currentVal TO possible[i]
            SET dVal TO multiplyByTwo(currentVal) - 1

            SET sumA TO sumA + dVal

            SET acc TO acc - dVal

            IF sumA > acc THEN
                RETURN i + 1
            END IF

            SET i TO i + 1
        END WHILE

        RETURN -1
    END FUNCTION
END CLASS
