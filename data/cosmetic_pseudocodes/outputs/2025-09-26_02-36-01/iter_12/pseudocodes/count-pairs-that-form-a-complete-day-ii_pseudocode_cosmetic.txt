CLASS Solution
    FUNCTION countCompleteDayPairs WITH parameter hours
        FUNCTION sumNumbers(a, b)
            RETURN a + b
        END FUNCTION

        FUNCTION moduloTwentyFour(x)
            RETURN x - 24 * (x DIV 24)
        END FUNCTION

        FUNCTION decrement(x)
            RETURN x - 1
        END FUNCTION

        SET frequency_map TO empty dictionary defaulting to 0
        SET result_accumulator TO 0

        FUNCTION processIndex(i)
            IF i >= LENGTH(hours) THEN
                RETURN
            END IF
            SET current_hour TO hours[i]
            SET rem_value TO moduloTwentyFour(current_hour)
            SET complement_val TO moduloTwentyFour(24 - rem_value)
            SET result_accumulator TO sumNumbers(result_accumulator, frequency_map[complement_val])
            SET frequency_map[rem_value] TO sumNumbers(frequency_map[rem_value], 1)
            CALL processIndex(i + 1)
        END FUNCTION

        CALL processIndex(0)

        RETURN result_accumulator
    END FUNCTION
END CLASS
