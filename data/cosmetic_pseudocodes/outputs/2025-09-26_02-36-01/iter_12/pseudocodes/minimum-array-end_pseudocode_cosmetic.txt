CLASS Solution
    FUNCTION minEnd(n x)
        FUNCTION canConstruct(max_limit)
            FUNCTION bitwiseAnd(a b)
                RETURN a AND b
            END FUNCTION

            FUNCTION incrementByOne(value)
                RETURN value PLUS 1
            END FUNCTION

            SET temp_var TO x
            SET tally TO 1

            REPEAT
                SET temp_var TO incrementByOne(temp_var)
                IF bitwiseAnd(temp_var x) EQUALS x THEN
                    tally = tally + 1
                    IF tally EQUALS n THEN
                        RETURN True
                    END IF
                END IF
            UNTIL temp_var >= max_limit

            RETURN tally EQUALS n
        END FUNCTION

        FUNCTION multiply(base exponent)
            SET result TO base
            SET count TO 1
            WHILE count LESS THAN exponent
                SET result TO result TIMES base
                SET count TO count + 1
            END WHILE
            RETURN result
        END FUNCTION

        SET low_bound TO x
        SET high_bound TO 2
        SET power_index TO 0

        WHILE power_index LESS THAN 8
            SET high_bound TO multiply(high_bound 10)
            SET power_index TO power_index + 1
        END WHILE

        WHILE low_bound LESS THAN high_bound
            SET aggregate TO low_bound PLUS high_bound
            SET pivot TO aggregate DIV 2
            IF canConstruct(pivot) THEN
                SET high_bound TO pivot
            ELSE
                SET low_bound TO incrementByOne(low_bound)
            END IF
        END WHILE

        RETURN low_bound
    END FUNCTION
END CLASS
