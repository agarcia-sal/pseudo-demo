CLASS Solution
    FUNCTION minimumSubstringsInPartition(s)
        FUNCTION zwr(alc)
            IF alc GREATER THAN OR EQUAL TO LENGTH(s) THEN
                RETURN 0
            ELSE
                SET qpx TO DICTIONARY WITH DEFAULT 0
                SET mev TO DICTIONARY WITH DEFAULT 0
                SET atp TO LENGTH(s) - alc
                PROCEDURE dba()
                    RETURN zwr(alc + 1)
                END PROCEDURE
                FUNCTION vqy(current)
                    IF s[current] IN qpx AND qpx[s[current]] NOT EQUAL 0 THEN
                        mev[qpx[s[current]]] = mev[qpx[s[current]]] - 1
                        IF mev[qpx[s[current]]] EQUAL 0 THEN
                            DELETE mev[qpx[s[current]]]
                        END IF
                    END IF
                    qpx[s[current]] = qpx[s[current]] + 1
                    mev[qpx[s[current]]] = mev.get(qpx[s[current]], 0) + 1
                END FUNCTION
                SET vpw TO alc
                SET pqb TO LENGTH(s)
                WHILE vpw LESS THAN pqb
                    CALL vqy(vpw)
                    IF LENGTH(mev.keys()) EQUAL 1 THEN
                        SET qmu TO 1 + zwr(vpw + 1)
                        IF qmu LESS THAN atp THEN
                            atp = qmu
                        END IF
                    END IF
                    vpw = vpw + 1
                END WHILE
                RETURN atp
            END IF
        END FUNCTION
        RETURN zwr(0)
    END FUNCTION
END CLASS
