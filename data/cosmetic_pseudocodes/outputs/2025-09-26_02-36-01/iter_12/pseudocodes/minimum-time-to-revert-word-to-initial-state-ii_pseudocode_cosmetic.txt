CLASS Solution
    FUNCTION minimumTimeToInitialState(word parameter word AND integer parameter k) RETURNS integer
        FUNCTION checkPrefix(mainStr, subStr) RETURNS boolean
            SET index TO 0
            SET isMatch TO true
            WHILE index < LENGTH(subStr) AND isMatch DO
                IF mainStr[index] NOT EQUAL subStr[index] THEN
                    SET isMatch TO false
                END IF
                INCREMENT index BY 1
            END WHILE
            RETURN isMatch
        END FUNCTION

        SET strLength TO LENGTH(word)
        SET counter TO 1

        REPEAT
            SET startPos TO counter * k
            SET tempPrefix TO ""
            SET pos TO startPos

            WHILE pos < strLength DO
                SET tempPrefix TO tempPrefix + word[pos]
                INCREMENT pos BY 1
            END WHILE

            IF checkPrefix(word, tempPrefix) THEN
                RETURN counter
            END IF

            SET counter TO counter + 1
        UNTIL false
    END FUNCTION
END CLASS
