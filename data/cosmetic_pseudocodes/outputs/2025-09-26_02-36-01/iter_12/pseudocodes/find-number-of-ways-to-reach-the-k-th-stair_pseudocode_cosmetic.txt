CLASS Solution
    FUNCTION waysToReachStair(k)
        FUNCTION explorePosition(current step, toggle flag, last jump)
            IF current step GREATER THAN k PLUS 1 THEN
                RETURN 0
            END IF

            SET result TO (IF current step EQUALS k THEN 1 ELSE 0)

            IF current step GREATER THAN 0 THEN
                IF toggle EQUALS 0 THEN
                    SET intermediate TO explorePosition(current step MINUS 1, 1, last jump)
                    SET result TO result PLUS intermediate
                END IF
            END IF

            SET next jump TO last jump PLUS 1
            SET forward step TO current step PLUS (2 TIMES last jump)
            SET increment TO explorePosition(forward step, 0, next jump)
            SET result TO result PLUS increment

            RETURN result
        END FUNCTION

        RETURN explorePosition(1, 0, 0)
    END FUNCTION
END CLASS
