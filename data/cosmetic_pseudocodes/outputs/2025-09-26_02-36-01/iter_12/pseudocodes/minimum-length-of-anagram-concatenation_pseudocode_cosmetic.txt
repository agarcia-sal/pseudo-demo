CLASS Solution
	 FUNCTION minAnagramLength(s)
		  DEFINE FUNCTION toBuildSet(str) RETURNS set
				DECLARE idx AS INTEGER INITIALIZED TO 0
				DECLARE resSet AS NEW set
				WHILE idx < lengthOf(str)
					LET charElement BE charAt(str, idx)
					IF contains(resSet, charElement) EQUALS false THEN
						addToSet(resSet, charElement)
					END IF
					idx = idx + 1
				END WHILE
				RETURN resSet
		  END FUNCTION

		  DECLARE counter AS INTEGER INITIALIZED TO 0
		  DECLARE temporarySet AS set ASSIGNED toBuildSet(s)
		  DECLARE iterVariableList AS list ASSIGNED toList(temporarySet)
		  WHILE counter < sizeOf(iterVariableList)
				; do nothing besides incrementing counter
				counter = counter + 1
		  END WHILE
		  RETURN counter
	 END FUNCTION
END CLASS
