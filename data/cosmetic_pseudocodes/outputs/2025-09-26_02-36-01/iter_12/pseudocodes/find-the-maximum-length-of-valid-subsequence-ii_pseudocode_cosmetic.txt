CLASS Solution
    FUNCTION maximumLength(nums k)
        
        FUNCTION computeMod(a b)
            RETURN ((a MOD b) + b) MOD b
        END FUNCTION
        
        SET length TO 0
        SET idx TO 0
        WHILE idx LESS THAN LENGTH OF nums
            SET length TO length + 1
            SET idx TO idx + 1
        END WHILE
        
        IF NOT(length NOT EQUAL TO 1)
            RETURN 1
        END IF
        
        SET mappings TO EMPTY LIST
        SET counter TO 0
        REPEAT
            EXIT WHEN counter EQUALS length
            APPEND EMPTY DICTIONARY TO mappings
            SET counter TO counter + 1
        UNTIL FALSE
        
        SET longest TO 1
        SET outer TO 0
        WHILE outer LESS THAN length
            SET inner TO 0
            WHILE inner LESS THAN outer
                SET sumVal TO nums[outer] + nums[inner]
                SET modVal TO computeMod(sumVal k)
                IF modVal IN mappings[inner]
                    SET currentLen TO mappings[inner][modVal] + 1
                    SET mappings[outer][modVal] TO currentLen
                ELSE
                    SET mappings[outer][modVal] TO 2
                    SET currentLen TO 2
                END IF
                IF currentLen GREATER THAN longest
                    SET longest TO currentLen
                END IF
                SET inner TO inner + 1
            END WHILE
            SET outer TO outer + 1
        END WHILE
        
        RETURN longest
    END FUNCTION
END CLASS
