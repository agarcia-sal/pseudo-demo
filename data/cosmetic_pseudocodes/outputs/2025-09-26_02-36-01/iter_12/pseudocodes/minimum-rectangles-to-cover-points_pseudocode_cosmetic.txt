CLASS Solution
	Function minRectanglesToCoverPoints(points, w)
		Define FUNCTION customSort(arr)
			Define FUNCTION compare(a, b)
                Return (a[0] - b[0]) < 0
            END FUNCTION
			Define VARIABLE i TO 0
			Define VARIABLE n TO LENGTH(arr)
			WHILE i < n - 1
				Define VARIABLE j TO i + 1
				WHILE j < n
					IF compare(arr[j], arr[i]) THEN
						Define VARIABLE tmp TO arr[i]
						SET arr[i] TO arr[j]
						SET arr[j] TO tmp
					END IF
					SET j TO j + 1
				END WHILE
				SET i TO i + 1
			END WHILE
			RETURN arr
		END FUNCTION
		
		SET sortedPoints TO customSort(points)
		Define rectanglesCount TO 0
		Define coverageLimit TO 0
		Define FUNCTION isGreater(a, b)
			RETURN a > b
		END FUNCTION
		
		Define FUNCTION incrementCounter(val)
			RETURN val + 1
		END FUNCTION
		
		Define VARIABLE index TO 0
		Define VARIABLE total TO LENGTH(sortedPoints)
		
		WHILE index < total
			Define VARIABLE pointX TO sortedPoints[index][0]
			Define VARIABLE pointY TO sortedPoints[index][1]
			
			IF isGreater(pointX, coverageLimit) THEN
				SET coverageLimit TO pointX + w
				SET rectanglesCount TO incrementCounter(rectanglesCount)
			END IF
			SET index TO index + 1
		END WHILE
		
		Define VARIABLE output TO rectanglesCount
		RETURN output
	END FUNCTION
END CLASS
