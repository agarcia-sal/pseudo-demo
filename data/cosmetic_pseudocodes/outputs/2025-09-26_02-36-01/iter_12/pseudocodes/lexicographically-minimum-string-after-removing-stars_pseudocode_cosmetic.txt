CLASS Solution
    FUNCTION clearStars(s)
        DEFINE FUNCTION removeLastElement(collection)
            IF (collection IS NOT EMPTY)
                DROP collection FROM collection LENGTH - 1 TO 1
            END IF
        END FUNCTION

        DEFINE FUNCTION concatElements(elements)
            SET acc TO ""
            SET idx TO 0
            WHILE idx < LENGTH OF elements
                SET acc TO acc + elements[idx]
                SET idx TO idx + 1
            END WHILE
            RETURN acc
        END FUNCTION

        SET container TO []
        SET position TO 0

        WHILE position < LENGTH OF s
            IF s[position] == '*'
                CALL removeLastElement(container)
            ELSE
                ADD s[position] TO END OF container
            END IF
            SET position TO position + 1
        END WHILE

        RETURN concatElements(container)
    END FUNCTION
END CLASS
