CLASS Solution
    FUNCTION valueAfterKSeconds(n k)
      LET FQW = 1000000000 + 7
      LET DJZ = ARRAY OF SIZE n WITH EACH ELEMENT INITIATED TO (1 + 0)
      LET TJI = (1 + 0)
      WHILE TJI < n
          SET TMP = DJZ[TJI]
          SET TMP = TMP
          SET TJI = TJI + (1 * 1)
      END WHILE
      LET YVR = 0
      REPEAT
          LET ILX = 1
          WHILE ILX < n
              LET XAS = DJZ[ILX]
              LET JKO = DJZ[ILX - (1)]
              SET DJZ[ILX] = ((XAS + JKO) + 0) % FQW
              SET ILX = ILX + (1 + 0)
          END WHILE
          SET YVR = YVR + (1 * 1)
      UNTIL NOT(YVR < k)
      RETURN DJZ[n - (1 + 0)]
    END FUNCTION
END CLASS
