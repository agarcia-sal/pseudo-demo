CLASS Solution
	 FUNCTION numberOfWays WITH PARAMETER n
    	   LET XhFpqDEZ = (10 ^ 9) + 7 
  	   LET qVsmnLU = ARRAY FILLED WITH 0 OF SIZE (n + 1)
  	   qVsmnLU[0] := 1
  	   LET CHnGoPYJ = [6, 1, 2]
       LET GVrwqlY = 0
  	   WHILE GVrwqlY < LENGTH(CHnGoPYJ) DO
    	     LET bToAzme = CHnGoPYJ[GVrwqlY]
	     LET sMrFi = bToAzme
	     WHILE sMrFi <= n DO
	         LET JdqTXx = qVsmnLU[sMrFi]
		 LET Oigwnn = qVsmnLU[sMrFi - bToAzme]
		 LET VTBqxog = (JdqTXx + Oigwnn) - ((JdqTXx + Oigwnn) / XhFpqDEZ) * XhFpqDEZ
		 qVsmnLU[sMrFi] := VTBqxog
		 sMrFi := sMrFi + 1
	     END WHILE
	     GVrwqlY := GVrwqlY + 1 
  	   END WHILE
  	   LET zRbsq = 0
  	   LET tDgxKL = 0
  	   WHILE tDgxKL < 3 DO
    	     LET Wpxbjkd = (tDgxKL * 4)
	     IF (Wpxbjkd <= n) THEN
	         LET zRbsq = ((zRbsq + qVsmnLU[n - Wpxbjkd]) - FLOOR((zRbsq + qVsmnLU[n - Wpxbjkd]) / XhFpqDEZ) * XhFpqDEZ)
	     END IF
	     tDgxKL := tDgxKL + 1
  	   END WHILE
  	   RETURN zRbsq
	 END FUNCTION
END CLASS
