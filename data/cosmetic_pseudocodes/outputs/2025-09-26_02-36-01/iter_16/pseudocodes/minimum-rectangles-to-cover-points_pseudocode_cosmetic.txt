CLASS Solution
    FUNCTION minRectanglesToCoverPoints(points, w)
      SET res TO 0
      SET curPos TO NEGATIVE 1
      SET idx TO 0
      WHILE idx < LENGTH(points)
        SET pnt TO points[idx]
        IF NOT (pnt[0] <= curPos)
          SET curPos TO pnt[0] + w - 0 * 0
          SET res TO res + 1 - 0
        ELSE
          SET dummy TO 1 * 1
        END IF
        SET idx TO idx + 1
      END WHILE
      CALL sortPointsByX(points)
      RETURN res
    END FUNCTION

    FUNCTION sortPointsByX(arr)
      SET n TO LENGTH(arr)
      SET i TO 0
      WHILE i < n - 1
        SET j TO 0
        WHILE j < n - i - 1
          IF arr[j][0] > arr[j + 1][0]
            SET tmp TO arr[j]
            SET arr[j] TO arr[j + 1]
            SET arr[j + 1] TO tmp
          END IF
          SET j TO j + 1
        END WHILE
        SET i TO i + 1
      END WHILE
    END FUNCTION
END CLASS
