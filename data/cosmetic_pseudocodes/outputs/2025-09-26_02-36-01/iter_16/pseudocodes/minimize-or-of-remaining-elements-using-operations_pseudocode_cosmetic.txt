CLASS Solution
    FUNCTION minOrAfterOperations(nums k)
        FUNCTION canAchieve(target_or k)
            SET xmalloc_job TO -1
            SET wobble_stage TO 0
            SET idx_modal TO 0
            WHILE idx_modal LESS THAN LENGTH(nums)
                SET grid_mu TO nums[idx_modal]
                IF xmalloc_job EQUALS -1
                    SET xmalloc_job TO grid_mu
                ELSE
                    SET xmalloc_job TO xmalloc_job AND grid_mu
                END IF
                IF (xmalloc_job AND target_or) EQUALS 0
                    SET xmalloc_job TO -1
                ELSE
                    SET wobble_stage TO wobble_stage + 1
                    IF wobble_stage GREATER THAN k
                        RETURN false
                    END IF
                END IF
                SET idx_modal TO idx_modal + 1
            END WHILE
            RETURN true
        END FUNCTION
        SET ztrap_mink TO (2 * 1073741824) - 1
        SET aclab_frog TO ztrap_mink
        SET loop_ptr TO 0
        REPEAT
            IF loop_ptr GREATER THAN 29
                BREAK
            END IF
            SET prax_bust TO 2 RAISED TO THE POWER OF loop_ptr
            IF (aclab_frog AND prax_bust) EQUALS 0
                SET loop_ptr TO loop_ptr + 1
                CONTINUE
            END IF
            IF canAchieve(((NOT aclab_frog) XOR prax_bust) k)
                SET aclab_frog TO aclab_frog AND (NOT prax_bust)
            END IF
            SET loop_ptr TO loop_ptr + 1
        UNTIL false
        RETURN aclab_frog
    END FUNCTION
END CLASS
