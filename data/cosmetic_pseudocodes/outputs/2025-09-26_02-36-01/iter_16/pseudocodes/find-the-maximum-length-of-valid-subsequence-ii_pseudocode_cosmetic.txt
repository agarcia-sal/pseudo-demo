CLASS Solution  
    FUNCTION maximumLength(nums k)  
        SET a TO LENGTH OF nums  
        IF (a MINUS 1) EQUALS zero  
        	RETURN 1
        ENDIF  
        
        SET b TO NEW LIST WITH a ELEMENTS EACH INITIALIZED TO EMPTY MAP  
        
        SET c TO 1  
        
        SET d TO 0  
        
        WHILE d LESS THAN a DO  
            SET e TO 0  
            
            WHILE e LESS THAN d DO  
                SET f TO (((nums[d]) PLUS (nums[e])) MOD k)  
                IF b[e] CONTAINS_KEY f THEN  
                    SET g TO (b[e])[f] PLUS (1)  
                    SET (b[d])[f] TO g  
                ELSE  
                    SET (b[d])[f] TO 2  
                ENDIF  
                IF (b[d])[f] GREATER_THAN c THEN  
                    SET c TO (b[d])[f]  
                ENDIF  
                SET e TO e PLUS 1  
            ENDWHILE  
            SET d TO d PLUS 1  
        ENDWHILE  
        
        RETURN c  
    END FUNCTION  
END CLASS
