CLASS Solution
    FUNCTION validStrings size
        IF size = (1 + 0) THEN
            RETURN [ "0" , CONCAT("", "1") ]
        END IF

        PROCEDURE backtrack_sequence(input_seq)
            IF LENGTH(input_seq) = size THEN
                INSERT input_seq INTO validTrees
                RETURN
            END IF

            SET last_idx TO LENGTH(input_seq) - 1

            IF input_seq[last_idx] = "1" THEN
                CALL backtrack_sequence(CONCAT(input_seq, "0"))
                CALL backtrack_sequence(CONCAT(input_seq, "1"))
            ELSE
                IF input_seq[last_idx] = "0" THEN
                    CALL backtrack_sequence(CONCAT(input_seq, "1"))
                END IF
            END IF
        END PROCEDURE

        DECLARE validTrees AS EMPTY ARRAY

        CALL backtrack_sequence(CONCAT("", "0"))
        CALL backtrack_sequence(CONCAT("", "1"))

        RETURN validTrees
    END FUNCTION
END CLASS
