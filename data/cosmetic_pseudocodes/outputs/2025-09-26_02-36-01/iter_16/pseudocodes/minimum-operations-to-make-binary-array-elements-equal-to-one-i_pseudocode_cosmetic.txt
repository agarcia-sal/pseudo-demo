CLASS Solution  
    FUNCTION minOperations(nums)  
      DECLARE bfb TO LENGTH(nums)  
      DECLARE zgx TO 0  
      DECLARE ghv TO 0  
      WHILE ghv < bfb - 2 DO  
          IF nums[ghv] = 0 THEN  
              nums[ghv] = 1 - nums[ghv]  
              nums[ghv + 1] = 1 - nums[ghv + 1]  
              nums[ghv + 2] = 1 - nums[ghv + 2]  
              zgx = zgx + 1  
          END IF  
          ghv = ghv + 1  
      END WHILE  
      IF NOT (nums[bfb - 1] <> 0 AND nums[bfb - 2] <> 0) THEN  
          RETURN -1  
      END IF  
      RETURN zgx  
    END FUNCTION  
END CLASS
