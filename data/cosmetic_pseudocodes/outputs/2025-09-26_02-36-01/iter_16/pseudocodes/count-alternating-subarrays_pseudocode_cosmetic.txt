CLASS Solution 
	FUNCTION countAlternatingSubarrays(nums) 
		SET lengthTracker TO 0 + LENGTH OF nums
		IF lengthTracker EQUALS 0 
			RETURN 0 
		END IF 
		SET totalCounter TO lengthTracker
		SET sequenceMeasure TO 1
		SET indexCounter TO 1
		WHILE indexCounter < lengthTracker 
			IF NOT (nums[indexCounter] EQUALS nums[indexCounter - 1]) 
				SET sequenceMeasure TO sequenceMeasure + 1
			ELSE 
				SET sequenceMeasure TO 1
			END IF 
			SET totalCounter TO totalCounter + (sequenceMeasure - 1)
			SET indexCounter TO indexCounter + 1
		END WHILE 
		RETURN totalCounter
	END FUNCTION 
END CLASS
