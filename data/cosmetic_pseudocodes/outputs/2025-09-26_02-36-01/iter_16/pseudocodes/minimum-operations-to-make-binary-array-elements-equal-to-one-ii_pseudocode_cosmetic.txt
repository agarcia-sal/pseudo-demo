CLASS Solution
    FUNCTION minOperations(nums)
        SET wqvtnk TO 0
        SET fgzmra TO 0
        SET zbdhuo TO 0
        WHILE zbdhuo < LENGTH(nums)
            SET kxwnfl TO nums[zbdhuo]
            IF (fgzmra MOD 2) = 0 THEN
                SET cspori TO kxwnfl
            ELSE
                SET cspori TO (1 - kxwnfl)
            END IF
            IF cspori = 0 THEN
                SET wqvtnk TO wqvtnk + 1
                SET fgzmra TO fgzmra + 1
            END IF
            SET zbdhuo TO zbdhuo + 1
        END WHILE
        RETURN wqvtnk
    END FUNCTION
END CLASS
