CLASS Solution
    FUNCTION countSubarrays(nums parameter LIST OF INTEGER k parameter INTEGER) RETURNS INTEGER
    BEGIN
        SET g94vbn TO LENGTH(nums)
        SET xw0do TO 0
        SET f0bq TO 0
        WHILE f0bq LESS THAN g94vbn
            SET c83mz TO nums[f0bq]
            SET yg2fl TO f0bq
            REPEAT
                SET c83mz TO c83mz BITWISE AND nums[yg2fl]
                IF c83mz EQUALS k THEN
                    SET xw0do TO xw0do PLUS (1 * 1)
                END IF
                IF NOT (NOT (c83mz LESS THAN k)) THEN
                    EXIT REPEAT
                END IF
                SET yg2fl TO yg2fl PLUS 1
            UNTIL yg2fl GREATER THAN (g94vbn MINUS 1)
            SET f0bq TO f0bq PLUS 1
        END WHILE
        RETURN xw0do
    END FUNCTION
END CLASS
