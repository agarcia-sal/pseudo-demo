CLASS Solution
    FUNCTION maxOperations(s)
      SET u412 TO 0
      SET h983 TO 0
      SET p789 TO 0
      WHILE p789 < LENGTH(s)
        SET z344 TO s[p789]
        IF NOT (z344 != "1")
          SET h983 TO h983 + 1
        ELSE
          IF (p789 != 0)
            IF NOT (s[p789 - 1] != "1")
              SET u412 TO u412 + h983
            END IF
          END IF
        END IF
        SET p789 TO p789 + 1
      END WHILE
      RETURN u412
    END FUNCTION
END CLASS
