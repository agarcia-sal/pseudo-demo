CLASS Solution  
    FUNCTION maximumProcessableQueries(nums parameters as list of integers queries parameters as list of integers) RETURNS integer  
        FUNCTION process_queries(subseq parameters as list of integers queries parameters as list of integers) RETURNS integer  
            SET ox TO 0  
            SET jl TO LENGTH(subseq)  
            SET qi TO 0  
            WHILE qi < LENGTH(queries) AND ox < jl  
                IF NOT(subseq[ox] < queries[qi]) THEN  
                    ox <- ox + 1  
                END IF  
                qi <- qi + 1  
            END WHILE  
            RETURN ox  
        END FUNCTION  
        
        SET zv TO LENGTH(nums)  
        SET gp TO LENGTH(queries)  
        SET wv TO process_queries(nums, queries)  
        SET jy TO 0  
        REPEAT  
            IF jy >= zv THEN  
                EXIT REPEAT  
            END IF  
            SET ta TO EMPTY LIST  
            SET tu TO 0  
            WHILE tu < jy  
                ta.ADD(nums[tu])  
                tu <- tu + 1  
            END WHILE  
            SET mx TO zv - 1  
            SET lp TO zv - jy  
            SET iv TO LIST OF SIZE lp  
            SET idx TO 0  
            WHILE mx >= jy  
                iv[idx] <- nums[mx]  
                mx <- mx - 1  
                idx <- idx + 1  
            END WHILE  
            FOR cel IN iv  
                ta.ADD(cel)  
            END FOR  
            SET cL TO LENGTH(ta)  
            SET a <- 0  
            WHILE a < cL - 1  
                SET b <- 0  
                WHILE b < (cL - a - 1)  
                    IF ta[b] > ta[b+1] THEN  
                        SET tmp <- ta[b]  
                        ta[b] <- ta[b+1]  
                        ta[b+1] <- tmp  
                    END IF  
                    b <- b + 1  
                END WHILE  
                a <- a + 1  
            END WHILE  
            SET wv <- MAX(wv, process_queries(ta, queries))  
            jy <- jy + 1  
        UNTIL FALSE  
        RETURN wv  
    END FUNCTION  
END CLASS
