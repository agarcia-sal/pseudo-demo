CLASS Solution  
    FUNCTION maxScore(nums)  
        SET a TO LENGTH OF nums  
        SET b TO LIST OF ZEROS WITH SIZE a  
        SET b[1] TO 0  
        SET c TO 2  
        WHILE c <= a  
            SET d TO 1  
            WHILE d < c  
                SET e TO b[d] + ((c - d) * nums[c])  
                IF b[c] < e THEN  
                    SET b[c] TO e  
                END IF  
                SET d TO d + 1  
            END WHILE  
            SET c TO c + 1  
        END WHILE  
        RETURN b[a - 1]  
    END FUNCTION  
END CLASS
