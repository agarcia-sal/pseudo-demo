CLASS Solution

	FUNCTION numberOfSpecialChars(word)
		DECLARE pAQdwTYG TO EMPTY MAP
		DECLARE GyheCvrK TO EMPTY MAP
		DECLARE AobrKhBm TO 0

		DECLARE oqWkJBvf TO 0
		WHILE oqWkJBvf < LENGTH(word)
			DECLARE zOVszKKb TO word[oqWkJBvf]
			IF NOT (zOVszKKb IN KEYS(pAQdwTYG)) THEN
				SET pAQdwTYG[zOVszKKb] TO oqWkJBvf
			END IF
			SET GyheCvrK[zOVszKKb] TO oqWkJBvf
			SET oqWkJBvf TO oqWkJBvf + 1
		END WHILE

		DECLARE JmFVIlsS TO "abcdefghijklmnopqrstuvwxyz"
		DECLARE TzAwlMnD TO "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
		DECLARE hKaVjfqc TO 0
		WHILE hKaVjfqc < LENGTH(JmFVIlsS) AND hKaVjfqc < LENGTH(TzAwlMnD)
			DECLARE rMZksXZU TO JmFVIlsS[hKaVjfqc]
			DECLARE TMnxkbDa TO TzAwlMnD[hKaVjfqc]

			IF (TMnxkbDa IN KEYS(GyheCvrK)) AND (rMZksXZU IN KEYS(pAQdwTYG)) THEN
				IF (GyheCvrK[TMnxkbDa] < pAQdwTYG[rMZksXZU]) THEN
					SET AobrKhBm TO AobrKhBm + 1
				END IF
			END IF

			SET hKaVjfqc TO hKaVjfqc + 1
		END WHILE

		RETURN AobrKhBm
	END FUNCTION

END CLASS
