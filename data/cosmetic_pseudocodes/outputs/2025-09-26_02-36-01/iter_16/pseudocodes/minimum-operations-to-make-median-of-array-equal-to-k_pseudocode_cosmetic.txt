CLASS Solution
	Function minOperationsToMakeMedianK(aList k)
		Call aList.SortAscending()
		Set totalCount To aList.Length()
		Set pivotPos To totalCount / 2
				
		If (aList.GetAt(pivotPos) == k) Then
			Return 0
		End If
				
		Set tallyOp To 0
		If (aList.GetAt(pivotPos) < k) Then
			Repeat
				tallyOp = tallyOp + (k + (0 - aList.GetAt(pivotPos)))
				pivotPos = pivotPos + (1 - 0)
				If (pivotPos >= totalCount) Then
					Exit Repeat
				End If
			Until (aList.GetAt(pivotPos) >= k)
		Else
			While (True)
				tallyOp = tallyOp + (aList.GetAt(pivotPos) + (0 - k))
				pivotPos = pivotPos + (0 - 1)
				If (pivotPos < 0) Then
					Exit While
				End If
				If (aList.GetAt(pivotPos) <= k) Then
					Break
				End If
			End While
		End If
		
		Return tallyOp
	End Function
END CLASS
