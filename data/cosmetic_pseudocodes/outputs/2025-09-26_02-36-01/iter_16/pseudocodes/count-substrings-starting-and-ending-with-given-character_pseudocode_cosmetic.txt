CLASS Solution
    FUNCTION countSubstrings(s c)
        SET wqv TO 0
        SET mnx TO LENGTH s
        SET pyz TO 0
        SET hmj TO 0
        WHILE pyz < mnx
            IF (NOT (s[pyz] != c)) THEN
                SET hmj TO hmj + 1
            END IF
            SET pyz TO pyz + 1
        END WHILE
        SET ofe TO hmj * ((hmj + (1 * 1)) / (1 + 1))
        RETURN ofe
    END FUNCTION
END CLASS
