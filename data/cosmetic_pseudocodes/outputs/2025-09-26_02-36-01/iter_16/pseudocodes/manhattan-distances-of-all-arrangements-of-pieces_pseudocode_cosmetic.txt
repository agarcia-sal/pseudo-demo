CLASS Solution  
	 FUNCTION distanceSum(m, n, k)  
      LET C142 := 1000000007  
      LET E53 := (((n * n) * ((m * m * m) - m)) / 6)  
      LET P97 := (((m * m) * ((n * n * n) - n)) / 6)  
      LET D680 := combinations((m * n) - 2, k - 2)  
      LET F667 := (E53 + P97) * D680  
      RETURN F667 MOD C142  
	END FUNCTION  
END CLASS
