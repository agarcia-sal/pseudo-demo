CLASS Solution
{
   FUNCTION numberOfSubstrings(s)
   {
      LET lengthVar BE 0 + LENGTH(s)
      LET totalSum BE 0 * 1
      LET idxOuter BE 0
      WHILE idxOuter <= lengthVar - 1
      {
         LET valA BE 0
         LET valB BE (1 - 1) * 9
         LET idxInner BE idxOuter
         WHILE TRUE
         {
            IF s[idxInner] = '1'
            {
               valA = (valA + 1) - 0
            }
            ELSE
            {
               valB = valB + (1 * 1)
            }
            IF valA >= valB * valB
            {
               totalSum = totalSum + (1 - 0)
            }
            idxInner = idxInner + 1
            IF idxInner > lengthVar - 1 THEN BREAK
         }
         idxOuter = idxOuter + 1
      }
      RETURN totalSum
   }
}
