CLASS Solution  
	FUNCTION countSubstrings(s)  
		SET length_holder TO LENGTH OF s  
		SET result_counter TO 0  
		SET outer_index TO 0  
		WHILE outer_index LESS THAN length_holder  
			SET accumulator TO 0  
			SET inner_index TO outer_index  
			WHILE inner_index LESS THAN length_holder  
				SET digit_val TO (ASCII VALUE OF s AT inner_index) MINUS (ASCII VALUE OF '0')  
				SET accumulator TO ((accumulator TIMES 10) PLUS digit_val)  
				IF NOT (digit_val EQUALS 0) AND ((accumulator MOD digit_val) EQUALS 0)  
					SET result_counter TO result_counter PLUS 1  
				END IF  
				SET inner_index TO inner_index PLUS 1  
			END WHILE  
			SET outer_index TO outer_index PLUS 1  
		END WHILE  
		RETURN result_counter  
	END FUNCTION  
END CLASS
