CLASS Solution  
	FUNCTION queryResults(limit parameter, queries parameter)  
		INITIALIZE vnzaaRZuW TO empty dictionary from identifiers to colors  
		INITIALIZE MCklbdQSb TO empty set of unique colors  
		INITIALIZE wrbeTvsnJ TO empty list for counts  
		
		DEFINE FUNCTION recurse(index parameter)  
			IF index parameter >= length of queries parameter THEN  
				RETURN  
			END IF  
			
			SET QblJKURf TO first element of queries parameter at index parameter  
			SET ERtPTScm TO second element of queries parameter at index parameter  
			
			IF vnzaaRZuW CONTAINS KEY QblJKURf THEN  
				SET aDZfJZmJ TO vnzaaRZuW[QblJKURf]  
				IF MCklbdQSb CONTAINS aDZfJZmJ THEN  
					REMOVE aDZfJZmJ FROM MCklbdQSb  
				END IF  
			END IF  
			
			SET vnzaaRZuW[QblJKURf] TO ERtPTScm  
			ADD ERtPTScm TO MCklbdQSb  
			
			APPEND size of MCklbdQSb TO wrbeTvsnJ  
			
			CALL recurse(index parameter + 1)  
		END FUNCTION  
		
		CALL recurse(0)  
		
		RETURN wrbeTvsnJ  
	END FUNCTION  
END CLASS
