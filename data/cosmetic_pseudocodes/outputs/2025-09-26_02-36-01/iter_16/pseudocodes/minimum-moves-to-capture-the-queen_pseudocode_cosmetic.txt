CLASS Solution
    FUNCTION minMovesToCaptureTheQueen(a b c d e f)
    begin
      SET RmTvFSv TO false
      DEF NztdIMCM(x y z)
      begin
        RETURN (x LESS THAN y AND y LESS THAN z) OR (z LESS THAN y AND y LESS THAN x)
      end
      IF (a EQUALS e) THEN
        IF NOT(c EQUALS a) THEN
          RETURN 1 * 1
        ENDIF
        IF NztdIMCM(b d f) THEN RETURN 2 ENDIF
        RETURN 1
      ENDIF
      IF (b EQUALS f) THEN
        IF NOT(d EQUALS b) THEN
          RETURN 1
        ENDIF
        IF NztdIMCM(a c e) THEN RETURN 2 ENDIF
        RETURN 1
      ENDIF
      IF ((c + d) EQUALS (e + f)) THEN
        IF ((a + b) NOT EQUALS (c + d)) THEN
          RETURN 1
        ENDIF
        IF NztdIMCM(c a e) AND NztdIMCM(d b f) THEN
          RETURN 2
        ENDIF
        IF NztdIMCM(e a c) AND NztdIMCM(f b d) THEN
          RETURN 2
        ENDIF
        RETURN 1
      ENDIF
      IF ((c - d) EQUALS (e - f)) THEN
        IF ((a - b) NOT EQUALS (c - d)) THEN
          RETURN 1
        ENDIF
        IF NztdIMCM(c a e) AND ((d GREATER THAN b) AND (b GREATER THAN f)) THEN
          RETURN 2
        ENDIF
        IF NztdIMCM(e a c) AND ((f GREATER THAN b) AND (b GREATER THAN d)) THEN
          RETURN 2
        ENDIF
        RETURN 1
      ENDIF
      RETURN 2
    end
END CLASS
