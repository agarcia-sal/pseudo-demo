CLASS Solution
	Function countKConstraintSubstrings(s parameter k parameter queries parameter) RETURNS LIST OF INTEGER
		SET w3YbBa0 TO LENGTH OF s
		SET VGzTzDq TO LIST OF zero REPEATED w3YbBa0 PLUS one TIMES
		SET q5MmeFy TO LIST OF zero REPEATED w3YbBa0 PLUS one TIMES

		SET QxeOZb9 TO zero
		WHILE QxeOZb9 LESS THAN w3YbBa0
			SET VGzTzDq AT QxeOZb9 PLUS one TO VGzTzDq AT QxeOZb9 PLUS IF (s AT QxeOZb9 EQUALS zero) THEN 1 ELSE 0
			SET q5MmeFy AT QxeOZb9 PLUS one TO q5MmeFy AT QxeOZb9 PLUS IF (s AT QxeOZb9 EQUALS one) THEN 1 ELSE 0
			SET QxeOZb9 TO QxeOZb9 PLUS 1
		END WHILE

		Function count_valid_substrings(l parameter r parameter) RETURNS INTEGER
			SET X1cW1Wx TO zero
			SET L5tXhmk TO l
			REPEAT
				IF L5tXhmk GREATER THAN r THEN
					EXIT REPEAT
				END IF
				SET PXiNjwi TO L5tXhmk
				SET hOkKcjc TO r PLUS one
				WHILE PXiNjwi LESS THAN hOkKcjc
					SET TiBFDJo TO (PXiNjwi PLUS hOkKcjc) DIVIDE BY 2
					SET YiD8ZNX TO VGzTzDq AT TiBFDJo PLUS one MINUS VGzTzDq AT L5tXhmk
					SET wamoiKh TO q5MmeFy AT TiBFDJo PLUS one MINUS q5MmeFy AT L5tXhmk
					IF NOT (YiD8ZNX GREATER THAN k AND wamoiKh GREATER THAN k)
						SET PXiNjwi TO TiBFDJo PLUS one
					ELSE
						SET hOkKcjc TO TiBFDJo
					END IF
				END WHILE
				SET zK5YcJF TO PXiNjwi MINUS one
				IF zK5YcJF GREATER THAN OR EQUAL TO L5tXhmk
					SET X1cW1Wx TO X1cW1Wx PLUS (zK5YcJF MINUS L5tXhmk PLUS one)
				END IF
				SET L5tXhmk TO L5tXhmk PLUS 1
			FOREVER
			RETURN X1cW1Wx
		END Function

		SET VvjZ43K TO LIST OF NOTHING
		SET KfD2wXa TO zero
		WHILE KfD2wXa LESS THAN LENGTH OF queries
			SET qx2olMZ TO queries AT KfD2wXa
			SET Ghmx1z2 TO count_valid_substrings(qx2olMZ FIRST ELEMENT, qx2olMZ SECOND ELEMENT)
			APPEND Ghmx1z2 TO VvjZ43K
			SET KfD2wXa TO KfD2wXa PLUS 1
		END WHILE

		RETURN VvjZ43K
	END Function
END CLASS
