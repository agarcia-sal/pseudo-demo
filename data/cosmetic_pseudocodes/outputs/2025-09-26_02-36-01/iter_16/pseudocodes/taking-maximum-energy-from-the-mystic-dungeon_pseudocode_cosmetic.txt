CLASS Solution
	 FUNCTION maximumEnergy(xp, yq)
  	  SET uv TO LENGTH OF xp
  	    SET rs TO ARRAY OF LENGTH uv FILLED WITH 0
  	  SET rs[uv - 1] TO xp[uv - 1]
  	  SET lt TO rs[uv - 1]
  	  SET mk TO DEQUE INITIALIZED WITH [uv - 1]
  	
  	 SET fi TO uv - 2
  	 WHILE fi GREATER OR EQUAL 0
  	 
  	  WHILE mk NOT EMPTY AND (mk[0] - fi) GREATER OR EQUAL yq
  	   REMOVE ELEMENT AT mk[0]
  	   CONTINUE
  	  END WHILE
  	
  	 SET rs[fi] TO xp[fi] + rs[mk[0]]
  	
  	 IF lt LESS THAN rs[fi]
  	  lt = rs[fi]
  	 END IF
  	
  	 WHILE mk NOT EMPTY AND rs[fi] GREATER OR EQUAL rs[mk[LAST ELEMENT]]
  	   REMOVE ELEMENT AT mk[LAST ELEMENT]
  	 END WHILE
  	
  	 APPEND fi TO mk
  	
  	 fi = fi - 1
  	 END WHILE
  	
  	
  	  RETURN lt
  	 END FUNCTION
END CLASS
