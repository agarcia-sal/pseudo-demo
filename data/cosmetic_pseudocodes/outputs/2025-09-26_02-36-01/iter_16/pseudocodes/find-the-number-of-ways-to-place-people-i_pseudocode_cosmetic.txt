CLASS Solution  
	FUNCTION numberOfPairs(points)  
		SET resultCounter TO 0  
		SET totalPoints TO LENGTH OF points  
		SET outerIndex TO 0  
		WHILE outerIndex < totalPoints DO  
			SET innerIndex TO 0  
			REPEAT  
				IF (outerIndex != innerIndex) THEN  
					SET xAlpha TO points AT outerIndex AT 0  
					SET yAlpha TO points AT outerIndex AT 1  
					SET xBeta TO points AT innerIndex AT 0  
					SET yBeta TO points AT innerIndex AT 1  
					IF ( (xAlpha <= xBeta) AND (yAlpha >= yBeta) ) THEN  
						SET isPairValid TO true  
						SET checkIndex TO 0  
						WHILE checkIndex < totalPoints AND isPairValid DO  
							IF ( (checkIndex != outerIndex) AND (checkIndex != innerIndex) ) THEN  
								SET xGamma TO points AT checkIndex AT 0  
								SET yGamma TO points AT checkIndex AT 1  
								IF ( (xAlpha <= xGamma) AND (xGamma <= xBeta) AND (yAlpha >= yGamma) AND (yGamma >= yBeta) ) THEN  
									SET isPairValid TO false  
								END IF  
							END IF  
							IF NOT isPairValid THEN  
								EXIT WHILE  
							END IF  
							INCREMENT checkIndex BY 1  
						END WHILE  
						IF (isPairValid == true) THEN  
							SET resultCounter TO resultCounter + 1  
						END IF  
					END IF  
				END IF  
				INCREMENT innerIndex BY 1  
			UNTIL innerIndex >= totalPoints  
			SET outerIndex TO outerIndex + 1  
		END WHILE  
		RETURN resultCounter  
	END FUNCTION  
END CLASS
