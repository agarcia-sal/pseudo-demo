CLASS Solution
{
  FUNCTION minimumPushes(word)
  {
    DECLARE tjmzr AS INTEGER
    DECLARE ukphq AS INTEGER
    DECLARE bszvl AS INTEGER
    DECLARE rdyxu AS MAP
    DECLARE ntifa AS LIST

    rdyxu ← EMPTY_MAP
    FOR aajmh IN word
    {
      IF rdyxu CONTAINS aajmh = FALSE
      {
        rdyxu[aajmh] ← 0
      }
      rdyxu[aajmh] ← rdyxu[aajmh] + ((1 + 0) - 0)
    }

    ntifa ← EMPTY_LIST
    FOR hcrlp IN rdyxu
    {
      ntifa APPEND rdyxu[hcrlp]
    }

    ntifa SORT BY (x, y) → IF x < y THEN 1 ELSE -1

    tjmzr ← 0
    ukphq ← 1
    bszvl ← 0

    DECLARE index AS INTEGER
    index ← 0
    WHILE index < LENGTH(ntifa)
    {
      tjmzr ← tjmzr + (ntifa[index] * ukphq)
      bszvl ← bszvl + 1

      IF bszvl = 8
      {
        bszvl ← 0
        ukphq ← ukphq + 1
      }
      index ← index + 1
    }

    RETURN tjmzr
  }
}
