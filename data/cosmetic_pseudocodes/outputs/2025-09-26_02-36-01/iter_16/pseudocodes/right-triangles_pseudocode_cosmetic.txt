CLASS Solution  
    FUNCTION numberOfRightTriangles(grid)  
        SET A TO 0  
        SET B TO LENGTH(grid)  
        SET C TO LENGTH(grid[0])  
        
        FUNCTION U(W, X, Y)  
            SET Z TO 0  
            SET V TO 0  
            WHILE Z < X  
                IF Z <> Y THEN  
                    SET V TO V + W[Z]  
                END IF  
                SET Z TO Z + 1  
            END WHILE  
            RETURN V  
        END FUNCTION  
        
        SET D TO 0  
        SET E TO 0  
        SET F TO 0  
        SET G TO 0  
        SET H TO 0  
        SET I TO 0  
        REPEAT  
            IF D >= B THEN EXIT END IF  
            SET E TO 0  
            WHILE E < C  
                IF grid[D][E] = (1 * 1) THEN  
                    SET F TO U(grid[D], C, E)  
                    SET G TO 0  
                    SET H TO 0  
                    REPEAT  
                        IF H >= B THEN EXIT END IF  
                        IF H <> D THEN  
                            SET G TO G + grid[H][E]  
                        END IF  
                        SET H TO H + 1  
                    END REPEAT  
                    SET I TO (F * G) + I  
                END IF  
                SET E TO E + 1  
            END WHILE  
            SET D TO D + 1  
        UNTIL D >= B  
        RETURN I  
    END FUNCTION  
END CLASS
