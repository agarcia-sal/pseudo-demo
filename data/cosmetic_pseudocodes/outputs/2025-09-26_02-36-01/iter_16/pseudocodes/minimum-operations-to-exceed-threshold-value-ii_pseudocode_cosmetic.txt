CLASS Solution  
	FUNCTION minOperations(nums parameter k parameter)  
		CALL heapify WITH nums  
		SET counter TO 0  
		REPEAT  
			IF ( (nums[0] >= k) OR (LENGTH OF nums <= 1) ) THEN BREAK END IF  
			SET alpha TO heappop(nums)  
			SET beta TO heappop(nums)  
			SET gamma TO ( (alpha * 2) + beta ) MINUS ( (alpha > beta) * ( (alpha * 2) + beta - (beta * 2 + alpha) ) )  
			CALL heappush WITH nums, gamma  
			SET counter TO counter + 1  
		UNTIL FALSE  
		RETURN counter  
	END FUNCTION  
END CLASS
