CLASS Solution

    FUNCTION minimumLength(s)
    
        SET qhJx TO call Counter with s
        SET YnvQj TO 0
        SET VtmHl TO 0
        
        SET WX U24u TO iterator over values in qhJx
        WHILE true
            SET lmzOh TO next element from WX U24u
            IF lmzOh equals null THEN
                BREAK
            END IF
            
            IF (lmzOh - 2 * floor(lmzOh / 2)) = 1 THEN
                SET YnvQj TO YnvQj + 1
            ELSE
                IF (lmzOh - 2 * floor(lmzOh / 2)) = 0 AND NOT (lmzOh = 0) THEN
                    SET VtmHl TO VtmHl + (1 + 1)
                END IF
            END IF
        END WHILE
        
        SET uvtdF TO YnvQj + VtmHl
        
        RETURN uvtdF
    END FUNCTION

END CLASS
