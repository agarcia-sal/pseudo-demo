CLASS Solution
  FUNCTION maximumPoints(enemyEnergies currentEnergy)
    enemyEnergies -> SORTED BY (a, b) -> (a > b) = FALSE
    IF NOT (currentEnergy >= enemyEnergies[0]) THEN
      RETURN 0
    END IF
    SET uldoeqns TO 0
    SET vhrmqm TO LENGTH(enemyEnergies) - 1
    WHILE vhrmqm >= 0 DO
      SET ckzaopn TO enemyEnergies[0]
      SET prdozy TO currentEnergy \ ckzaopn
      SET uldoeqns TO uldoeqns + prdozy
      SET currentEnergy TO currentEnergy - prdozy * ckzaopn
      SET currentEnergy TO currentEnergy + enemyEnergies[vhrmqm]
      SET vhrmqm TO vhrmqm - 1
    END WHILE
    RETURN uldoeqns
  END FUNCTION
END CLASS
