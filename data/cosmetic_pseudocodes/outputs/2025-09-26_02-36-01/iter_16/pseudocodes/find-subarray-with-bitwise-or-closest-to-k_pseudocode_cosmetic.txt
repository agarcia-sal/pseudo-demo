CLASS Solution

    FUNCTION minimumDifference(nums k)
        
        FUNCTION subarrayBitwiseOr(a b)
            SET w TO 0
            SET x TO a
            WHILE x <= b DO
                SET w TO w BITWISE_OR nums[x]
                SET x TO x + 1
            END WHILE
            RETURN w
        END FUNCTION
        
        SET m TO LENGTH OF nums
        SET y TO 1000000000 PLUS 1  // representing infinity

        SET z TO 0
        REPEAT
            IF z > m - 1 THEN
                BREAK
            END IF

            SET aa TO 0
            SET bb TO z
            WHILE bb <= m - 1 DO
                SET aa TO aa BITWISE_OR nums[bb]
                
                SET cc TO k MINUS aa
                IF cc < 0 THEN
                    SET dd TO 0 - cc
                ELSE
                    SET dd TO cc
                END IF

                IF dd < y THEN
                    SET y TO dd
                END IF
                
                IF dd = 0 THEN
                    RETURN 0
                END IF

                SET bb TO bb + 1
            END WHILE
            SET z TO z + 1
        UNTIL FALSE

        RETURN y
    END FUNCTION
END CLASS
