CLASS Solution
    FUNCTION clearStars(s)
        SET aux TO empty list
        SET idx TO 0
        WHILE idx < LENGTH(s)
            SET curr TO s[idx]
            IF curr == "*" THEN
                IF LENGTH(aux) > 0 THEN
                    REMOVE aux[LENGTH(aux) - 1]
                END IF
            ELSE
                APPEND curr TO aux
            END IF
            SET idx TO idx + 1
        END WHILE
        SET res_chars TO ""
        SET pos TO 0
        WHILE pos < LENGTH(aux)
            SET res_chars TO res_chars + aux[pos]
            SET pos TO pos + 1
        END WHILE
        RETURN res_chars
    END FUNCTION
END CLASS
