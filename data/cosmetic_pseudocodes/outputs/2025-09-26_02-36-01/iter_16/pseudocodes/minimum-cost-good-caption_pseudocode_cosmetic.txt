CLASS Solution  
  FUNCTION minCostGoodCaption(caption)  
      SET alpha TO 0  
      SET beta TO LENGTH OF caption  
      IF beta LESS THAN 3 THEN  
          RETURN ""  
      END IF  
  
      SET gamma TO SPLIT caption INTO characters  
  
      SET delta TO 0  
      WHILE delta < beta DO  
          SET epsilon TO delta  
          WHILE delta < beta AND gamma[delta] = gamma[epsilon] DO  
              SET delta TO delta + 1  
          END WHILE  
  
          SET zeta TO delta - epsilon  
  
          IF zeta < 3 THEN  
              IF epsilon > 0 AND gamma[epsilon - 1] = gamma[epsilon] THEN  
                  SET gamma[epsilon - 1] TO gamma[epsilon]  
                  SET epsilon TO epsilon - 1  
                  SET zeta TO zeta + 1  
              END IF  
  
              IF delta < beta AND gamma[delta - 1] = gamma[delta] THEN  
                  SET gamma[delta] TO gamma[delta - 1]  
                  SET delta TO delta + 1  
                  SET zeta TO zeta + 1  
              END IF  
  
              IF zeta < 3 THEN  
                  IF epsilon > 0 THEN  
                      SET eta TO gamma[epsilon - 1]  
                  ELSE  
                      SET eta TO gamma[delta]  
                  END IF  
  
                  IF eta = 'a' THEN  
                      SET eta TO 'b'  
                  ELSE IF eta = 'z' THEN  
                      SET eta TO 'y'  
                  ELSE  
                      SET theta TO ASCII_VALUE_OF eta + 1  
                      SET eta TO CHARACTER_CORRESPONDING_TO theta  
                  END IF  
  
                  SET iota TO epsilon  
                  REPEAT  
                      SET gamma[iota] TO eta  
                      SET iota TO iota + 1  
                  UNTIL iota = epsilon + zeta  
  
                  SET delta TO epsilon + zeta  
              END IF  
          END IF  
      END WHILE  
  
      RETURN CONCATENATE_ELEMENTS_OF gamma  
  END FUNCTION  
END CLASS
