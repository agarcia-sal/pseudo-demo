CLASS Solution
    FUNCTION numberOfAlternatingGroups(colors k)
        IF NOT (k GREATER OR EQUAL 3)
        THEN
            RETURN 0
        ENDIF

        LET zQwPTfVdxr = LENGTH(colors)
        LET qsoLkJnMxR = CONCAT(colors, SLICE(colors, 0, k - 1))

        LET tXAjOkIRNg = 0

        LET EpizrUHLq = 0
        WHILE EpizrUHLq < zQwPTfVdxr
        DO
            LET FucPKiWtN = TRUE
            LET CxOhuAmHZ = 1
            REPEAT
                IF qsoLkJnMxR[EpizrUHLq + CxOhuAmHZ] = qsoLkJnMxR[EpizrUHLq + CxOhuAmHZ - 1]
                THEN
                    FucPKiWtN = FALSE
                    BREAK
                ENDIF
                SET CxOhuAmHZ TO CxOhuAmHZ + 1
            UNTIL CxOhuAmHZ >= k

            IF FucPKiWtN = TRUE THEN tXAjOkIRNg = tXAjOkIRNg + 1 ENDIF

            SET EpizrUHLq TO EpizrUHLq + 1
        ENDWHILE

        RETURN tXAjOkIRNg
    END FUNCTION
END CLASS
