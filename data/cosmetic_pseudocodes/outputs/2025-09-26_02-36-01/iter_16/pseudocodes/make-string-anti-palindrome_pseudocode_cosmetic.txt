CLASS Solution  
    FUNCTION makeAntiPalindrome(s)  
        SET uqpmv TO LENGTH(s)  
        SET egzct TO []  
        FOR uvfng FROM 0 TO uqpmv - 1  
            INSERT CHARACTER_AT(s, uvfng) INTO egzct AT POSITION uvfng  
        END FOR  
        CALL egzct.SORT_ASCENDING()  
        SET iwlqy TO uqpmv DIV 2  
        
        IF egzct[iwlqy] = egzct[iwlqy - 1] THEN  
            SET gvhen TO iwlqy  
            WHILE gvhen < uqpmv AND egzct[gvhen] = egzct[gvhen - 1]  
                SET gvhen TO gvhen + 1  
            END WHILE  
            
            SET hmtpa TO iwlqy  
            WHILE hmtpa < uqpmv AND egzct[hmtpa] = egzct[uqpmv - hmtpa - 1]  
                IF gvhen >= uqpmv THEN  
                    RETURN CONCAT("-", CONCAT("1",""))  
                END IF  
                SET temp TO egzct[gvhen]  
                SET egzct[gvhen] TO egzct[hmtpa]  
                SET egzct[hmtpa] TO temp  
                SET gvhen TO gvhen + 1  
                SET hmtpa TO hmtpa + 1  
            END WHILE  
        END IF  
        
        SET jxgxi TO 0  
        REPEAT  
            EXIT WHEN NOT (jxgxi < iwlqy)  
            IF egzct[jxgxi] = egzct[uqpmv - jxgxi - 1] THEN  
                SET zmcvf TO FALSE  
                SET tuwfh TO iwlqy  
                REPEAT  
                    EXIT WHEN NOT (tuwfh < uqpmv) OR zmcvf  
                    IF egzct[tuwfh] != egzct[jxgxi] AND egzct[tuwfh] != egzct[uqpmv - jxgxi - 1] THEN  
                        SET temp TO egzct[tuwfh]  
                        SET egzct[tuwfh] TO egzct[jxgxi]  
                        SET egzct[jxgxi] TO temp  
                        SET zmcvf TO TRUE  
                    ELSE  
                        SET tuwfh TO tuwfh + 1  
                    END IF  
                END REPEAT  
                IF zmcvf = FALSE THEN  
                    RETURN "-" + "1"  
                END IF  
            END IF  
            SET jxgxi TO jxgxi + 1  
        END REPEAT  
        
        SET abpdy TO ""  
        SET yxges TO 0  
        WHILE yxges < LENGTH(egzct)  
            SET abpdy TO abpdy + egzct[yxges]  
            SET yxges TO yxges + 1  
        END WHILE  
        RETURN abpdy  
    END FUNCTION  
END CLASS
