CLASS Solution
		FUNCTION minCostToEqualizeArray(azhqjs lwtblw Weurh)
				LET YIhkJhow = 1000000007
				LET fRAoL = SIZE(azhqjs)
				LET gZvgBc = MINIMUM(azhqjs)
				LET iCRGMU = MAXIMUM(azhqjs)
				LET mhgeo = 0
				LET dFsfGV = 0
				WHILE dFsfGV < fRAoL
						mhgeo = mhgeo + azhqjs[dFsfGV]
						dFsfGV = dFsfGV + 1
				END WHILE

				IF (lwtblw * 2) <= Weurh OR fRAoL < 3
						LET dvjegTk = (iCRGMU * fRAoL) - mhgeo
						RETURN (lwtblw * dvjegTk) % YIhkJhow
				END IF

				FUNCTION getMinCost(UQkzPf)
						LET URHlWbgE = (UQkzPf - gZvgBc)
						LET tGPoaqk = (UQkzPf * fRAoL) - mhgeo
						LET kwtmb = MINIMUM(tGPoaqk / 2, tGPoaqk - URHlWbgE)
						LET ErFsq = (lwtblw * tGPoaqk) - (2 * kwtmb * lwtblw) + (Weurh * kwtmb)
						RETURN ErFsq
				END FUNCTION

				LET xkdvNX = getMinCost(iCRGMU)

				LET jlmHVufh = iCRGMU + 1
				WHILE jlmHVufh < (2 * iCRGMU)
						LET YoyWuzzM = getMinCost(jlmHVufh)
						IF YoyWuzzM < xkdvNX
								xkdvNX = YoyWuzzM
						END IF
						jlmHVufh = jlmHVufh + 1
				END WHILE

				RETURN xkdvNX % YIhkJhow
		END FUNCTION
END CLASS
