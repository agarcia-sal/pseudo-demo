CLASS Solution
    FUNCTION minEnd(n o)
        FUNCTION canConstruct(p)
            SET q TO o
            SET r TO 1
            WHILE TRUE
                IF q >= p THEN
                    BREAK
                END IF
                SET q TO q + 1
                IF ((q BITWISE_AND o) EQUALS o) THEN
                    SET r TO r + 1
                    IF r = n THEN
                        RETURN TRUE
                    END IF
                END IF
            END WHILE
            RETURN (r = n)
        END FUNCTION

        SET s TO o
        SET t TO 2
        SET t TO t * 10
        SET t TO t * 10
        SET t TO t * 10
        SET t TO t * 10
        SET t TO t * 10
        SET t TO t * 10
        SET t TO t * 10
        SET t TO t * 10
        SET u TO s
        SET v TO t

        WHILE u < v
            SET w TO (u + v) DIV 2
            IF canConstruct(w) THEN
                SET v TO w
            ELSE
                SET u TO u + 1
            END IF
        END WHILE

        RETURN u
    END FUNCTION
END CLASS
