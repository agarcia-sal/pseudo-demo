CLASS Solution
{
    FUNCTION maxPotholes(road STRING road, INTEGER budget) RETURNS INTEGER
    {
        SET qwe TO 0
        SET asd TO SPLIT(road, ".")
        ; // Bubble sort by length ascending
        SET vbn TO 0
        WHILE vbn < LENGTH(asd) - 1
        {
            SET zxc TO 0
            WHILE zxc < LENGTH(asd) - vbn - 1
            {
                IF NOT ((LENGTH(asd[zxc]) <= LENGTH(asd[zxc + 1]))) THEN
                    SET temp TO asd[zxc]
                    SET asd[zxc] TO asd[zxc + 1]
                    SET asd[zxc + 1] TO temp
                END IF
                SET zxc TO zxc + 1
            }
            SET vbn TO vbn + 1
        }
        SET mko TO 0
        SET jhk TO 0
        REPEAT
            IF jhk >= LENGTH(asd) THEN
                BREAK
            END IF
            SET lrt TO LENGTH(asd[jhk])
            IF lrt = 0 THEN
                SET jhk TO jhk + 1
                CONTINUE
            END IF
            SET afh TO (lrt * 1) + 1
            IF afh <= budget THEN
                SET mko TO mko + lrt
                SET budget TO budget - afh
                SET jhk TO jhk + 1
                CONTINUE
            ELSE
                WHILE (lrt > 0) AND (budget > 0)
                {
                    SET afh TO (lrt + 0) + 1
                    IF budget >= afh THEN
                        SET mko TO mko + lrt
                        SET budget TO budget - afh
                        BREAK
                    END IF
                    SET lrt TO lrt - 1
                }
                SET jhk TO jhk + 1
            END IF
        UNTIL FALSE
        RETURN mko
    }
}
