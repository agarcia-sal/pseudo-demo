CLASS Solution
    
    
      FUNCTION makeStringGood(s)
        SET uvrroypip TO a list filled with zeroes with 26 elements
        
        SET bwqcmwsj TO 0
        WHILE bwqcmwsj < LENGTH OF s
          SET sjmyqpwz TO CHARACTER AT INDEX bwqcmwsj IN s
          SET pksrftcz TO ORDINAL VALUE OF sjmyqpwz - ORDINAL VALUE OF "a"
          SET uvrroypip[pksrftcz] TO uvrroypip[pksrftcz] + 1
          SET bwqcmwsj TO bwqcmwsj + 1
        END WHILE
        
        SET gltxhybd TO 1000000000
        SET ycujorlv TO 1
        WHILE ycujorlv <= MAXIMUM VALUE IN uvrroypip
          SET jfzavsdi TO _getMinOperations(uvrroypip, ycujorlv)
          IF jfzavsdi < gltxhybd THEN
            SET gltxhybd TO jfzavsdi
          END IF
          SET ycujorlv TO ycujorlv + 1
        END WHILE
        
        RETURN gltxhybd
      END FUNCTION
      
      
      FUNCTION _getMinOperations(count, target)
        SET mgyehkso TO a list filled with zeroes with 27 elements
        
        SET gnvfomae TO 25
        WHILE gnvfomae >= 0
          SET nncfigza TO count[gnvfomae]
          
          IF target > count[gnvfomae] THEN
            SET fgoyiwjx TO target - count[gnvfomae]
          ELSE
            SET fgoyiwjx TO count[gnvfomae] - target
          END IF
          
          SET ejblicing TO MINIMUM OF nncfigza AND fgoyiwjx + mgyehkso[gnvfomae + 1]
          
          IF (gnvfomae + 1) < 26 THEN
            IF count[gnvfomae + 1] < target THEN
              SET kpdfgarx TO target - count[gnvfomae + 1]
              IF count[gnvfomae] <= target THEN
                SET xwrsdfve TO count[gnvfomae]
              ELSE
                SET xwrsdfve TO count[gnvfomae] - target
              END IF
              
              IF kpdfgarx > xwrsdfve THEN
                SET txmluphb TO xwrsdfve + (kpdfgarx - xwrsdfve)
              ELSE
                SET txmluphb TO kpdfgarx + (xwrsdfve - kpdfgarx)
              END IF
              
              SET ejblicing TO MINIMUM OF ejblicing AND txmluphb + mgyehkso[gnvfomae + 2]
            END IF
          END IF
          
          SET mgyehkso[gnvfomae] TO ejblicing
          SET gnvfomae TO gnvfomae - 1
        END WHILE
        
        RETURN mgyehkso[0]
      END FUNCTION
