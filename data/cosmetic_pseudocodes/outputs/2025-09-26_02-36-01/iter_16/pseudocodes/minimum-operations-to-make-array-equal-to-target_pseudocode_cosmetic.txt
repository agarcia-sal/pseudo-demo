CLASS Solution

    FUNCTION minimumOperations(arrA, arrB)
        
        SET len TO LENGTH(arrA)
        SET acc TO (arrB[0] - arrA[0]) * SIGN(arrB[0] - arrA[0])
        
        SET idx TO 1
        WHILE idx < len
            SET diffCurr TO (arrB[idx] - arrA[idx])
            SET diffPrev TO (arrB[idx - 1] - arrA[idx - 1])
            
            IF (diffCurr * diffPrev) > 0 THEN
                SET delta TO (ABS(diffCurr) - ABS(diffPrev)) * SIGN(ABS(diffCurr) - ABS(diffPrev))
                IF delta > 0 THEN
                    SET acc TO acc + delta
                ENDIF
            ELSE
                SET acc TO acc + diffCurr * SIGN(diffCurr)
            ENDIF

            SET idx TO idx + 1
        ENDWHILE

        RETURN acc
        
    END FUNCTION

END CLASS
