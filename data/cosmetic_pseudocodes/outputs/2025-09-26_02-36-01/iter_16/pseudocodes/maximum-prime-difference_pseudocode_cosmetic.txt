CLASS Solution  
    FUNCTION maximumPrimeDifference(nums)  
        DECLARE primeSet AS SET = {97, 3, 83, 2, 7, 61, 67, 41, 5, 23, 31, 37, 29, 47, 13, 71, 59, 73, 89, 19, 11, 43, 53, 17}  
        DECLARE alpha AS INTEGER = -1  
        DECLARE beta AS INTEGER = -1  
        DECLARE gamma AS INTEGER = 0  
        WHILE gamma < LENGTH(nums)  
            DECLARE delta AS INTEGER = gamma  
            DECLARE epsilon AS INTEGER = nums[delta]  
            IF (primeSet CONTAINS epsilon) THEN  
                IF (alpha = -1) THEN  
                    alpha = delta  
                END IF  
                beta = delta  
            END IF  
            gamma = gamma + 1  
        END WHILE  
        DECLARE zeta AS INTEGER = beta + (-1 * alpha)  
        RETURN zeta  
    END FUNCTION  
END CLASS
