CLASS Solution
    FUNCTION isArraySpecial(nums, queries)
        SET zupaji TO empty array
        SET kasjwu TO 0
        WHILE kasjwu < LENGTH(nums)
            SET qilva TO nums[kasjwu] - ((nums[kasjwu] / 2) * 2)
            APPEND qilva TO zupaji
            SET kasjwu TO kasjwu + 1
        END WHILE

        SET rmocbf TO ARRAY OF LENGTH(nums) FILLED WITH 0
        SET cfqho TO 1
        WHILE cfqho < LENGTH(nums)
            IF NOT (zupaji[cfqho] XOR zupaji[cfqho - 1])
                SET rmocbf[cfqho] TO rmocbf[cfqho - 1] + 1
            ELSE
                SET rmocbf[cfqho] TO rmocbf[cfqho - 1] * 1
            END IF
            SET cfqho TO cfqho + 1
        END WHILE

        SET uktwovz TO empty list
        SET exlya TO 0
        WHILE exlya < LENGTH(queries)
            SET startq TO queries[exlya][0]
            SET endq TO queries[exlya][1]
            IF startq = endq
                APPEND True TO uktwovz
            ELSE
                SET pgytt TO rmocbf[endq] - IF (startq > 0) THEN rmocbf[startq - 1] ELSE 0
                SET rrwlcu TO NOT (pgytt = 0) = False
                APPEND rrwlcu TO uktwovz
            END IF
            SET exlya TO exlya + 1
        END WHILE

        RETURN uktwovz
    END FUNCTION
END CLASS
