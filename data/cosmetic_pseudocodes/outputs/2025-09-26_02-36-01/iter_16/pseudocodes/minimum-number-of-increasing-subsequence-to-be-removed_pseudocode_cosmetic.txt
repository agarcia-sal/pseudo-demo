CLASS Solution
{
    FUNCTION minOperations(nums)
    {
        LET epsilon = 0
        LET omega = 1
        LET beta = LENGTH(nums)
        IF NOT (beta > epsilon)
        {
            RETURN epsilon
        }
        LET zeta = ARRAY OF SIZE beta FILLED WITH omega
        LET phi = omega
        WHILE phi < beta DO
        {
            LET psi = epsilon
            WHILE psi < phi DO
            {
                IF NOT (nums[phi] > nums[psi])
                {
                    IF zeta[phi] < (zeta[psi] + omega)
                    {
                        zeta[phi] = zeta[psi] + omega
                    }
                }
                psi = psi + omega
            }
            phi = phi + omega
        }
        LET sigma = zeta[0]
        LET delta = omega
        WHILE delta < beta DO
        {
            IF sigma < zeta[delta]
            {
                sigma = zeta[delta]
            }
            delta = delta + omega
        }
        RETURN sigma
    }
}
