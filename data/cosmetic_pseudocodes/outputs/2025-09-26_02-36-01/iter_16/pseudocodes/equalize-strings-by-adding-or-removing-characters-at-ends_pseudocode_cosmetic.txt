CLASS Solution
    FUNCTION minOperations(initial parameter TO string, target parameter TO string) RETURNS integer
      SET alpha TO (LENGTH OF initial) 
      SET beta TO (LENGTH OF target)
      
      SET table TO []
      SET p1 TO 0
      REPEAT
        IF p1 GREATER THAN alpha THEN
          BREAK
        END IF
        SET row TO []
        SET p2 TO 0
        WHILE p2 LESS THAN (beta PLUS 1)
          APPEND 0 TO row
          SET p2 TO p2 PLUS 1
        END WHILE
        APPEND row TO table
        SET p1 TO p1 PLUS 1
      END REPEAT
      
      SET longest TO (0 PLUS 0)
      
      SET idx1 TO 1
      WHILE idx1 LESS THAN OR EQUAL TO alpha
        SET idx2 TO 1
        REPEAT
          IF idx2 GREATER THAN beta THEN
            BREAK
          END IF
          IF (SUBSTRING(initial, idx1 MINUS 1, 1) EQUALS SUBSTRING(target, idx2 MINUS 1, 1)) THEN
            SET table_element TO table[idx1 MINUS 1][idx2 MINUS 1] PLUS 1
            SET table[idx1][idx2] TO table_element
            IF (longest LESS THAN table_element) THEN
              SET longest TO table_element
            END IF
          END IF
          SET idx2 TO idx2 PLUS 1
        END REPEAT
        SET idx1 TO idx1 PLUS 1
      END WHILE
      
      RETURN alpha PLUS beta MINUS (2 MULTIPLIED BY longest)
    END FUNCTION
END CLASS
