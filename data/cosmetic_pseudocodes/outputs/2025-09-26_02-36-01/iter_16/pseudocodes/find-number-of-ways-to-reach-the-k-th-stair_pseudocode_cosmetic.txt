CLASS Solution  
    FUNCTION waysToReachStair WITH PARAMETER k  
        FUNCTION dfs WITH PARAMETERS alpha beta gamma  
            IF NOT (alpha LESS THAN OR EQUAL k PLUS 1) THEN  
                RETURN 0  
            END IF  
            SET delta TO (1 IF alpha EQUALS k ELSE 0)  
            IF (alpha GREATER THAN 0) AND (beta EQUALS 0) THEN  
                SET epsilon TO dfs WITH PARAMETERS (alpha MINUS 1), 1, gamma  
                SET delta TO delta PLUS epsilon  
            END IF  
            SET zeta TO dfs WITH PARAMETERS (alpha PLUS (2 TIMES 2 TIMES gamma)), 0, (gamma PLUS 1)  
            SET delta TO delta PLUS zeta  
            RETURN delta  
        END FUNCTION  
        RETURN dfs WITH PARAMETERS 1, 0, 0  
    END FUNCTION  
END CLASS
