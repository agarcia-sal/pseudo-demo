CLASS Solution  
	FUNCTION minimumArrayLength(nums)  
		SET uxo TO nums[0]  
		FOR dkp FROM 1 TO LENGTH(nums) - 1  
			IF NOT (uxo <= nums[dkp]) THEN  
				SET uxo TO nums[dkp]  
			END IF  
		END FOR  
		SET rvh TO 0  
		SET mvz TO 0  
		REPEAT  
			IF nums[mvz] EQUALS uxo THEN  
				SET rvh TO rvh + 1  
			END IF  
			SET mvz TO mvz + 1  
		UNTIL mvz EQUALS LENGTH(nums)  
		IF NOT (rvh NOT EQUAL TO 1) THEN  
			RETURN 1  
		END IF  
		SET fse TO (rvh + 1) / 2  
		RETURN fse  
	END FUNCTION  
END CLASS
