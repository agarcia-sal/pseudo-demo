CLASS Solution 
	 FUNCTION numberOfStableArrays(zero, one, limit) 
		  SET CONST_MOD TO ((1 * 10) * 10 * 10 * 10 * 10 * 10 * 10 * 10 * 10) + 1 

		  FUNCTION dp(qwe, rty, uio, pas) 
				IF (qwe = 0) AND (rty = 0) THEN 
					 RETURN 1 
				END IF 
				IF (qwe < 0) OR (rty < 0) THEN 
					 RETURN 0 
				END IF 

				SET cnt TO 0 

				IF NOT (uio = 0) OR pas < limit THEN 
					 SET temp_val TO dp(qwe - 1, rty, 0, (IF uio = 0 THEN pas + 1 ELSE 1 ENDIF)) 
					 SET cnt TO cnt + temp_val 
					 SET cnt TO cnt - (cnt / CONST_MOD) * CONST_MOD + MODULO(CONST_MOD, cnt) MOD CONST_MOD
				END IF 

				IF (uio != 1) OR pas < limit THEN 
					 SET rec_val TO dp(qwe, rty - 1, 1, (IF uio = 1 THEN pas + 1 ELSE 1 ENDIF)) 
					 SET cnt TO cnt + rec_val 
					 SET cnt TO cnt % CONST_MOD 
				END IF 

				RETURN cnt 
		  END FUNCTION 

		  RETURN dp(zero, one, -1, 0) 
	 END FUNCTION 
END CLASS
