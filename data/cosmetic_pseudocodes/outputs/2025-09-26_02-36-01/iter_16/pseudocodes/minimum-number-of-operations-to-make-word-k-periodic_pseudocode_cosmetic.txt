CLASS Solution
    FUNCTION minimumOperationsToMakeKPeriodic(word parameter k)
        SET α TO LENGTH(word)
        SET Ω TO []
        SET ζ TO 0
        WHILE ζ < α
            SET Ψ TO ""
            SET ϑ TO 0
            WHILE ϑ < k AND (ζ + ϑ) < α
                SET Ψ TO Ψ + word[ζ + ϑ]
                SET ϑ TO ϑ + 1
            END WHILE
            APPEND Ψ TO Ω
            SET ζ TO ζ + k
        END WHILE
        SET Δ TO {}
        FOR υ IN Ω
            IF υ IN Δ
                SET Δ[υ] TO Δ[υ] + 1
            ELSE
                SET Δ[υ] TO 1
            END IF
        END FOR
        SET μ TO 0
        FOR ξ IN Δ
            IF Δ[ξ] > μ
                SET μ TO Δ[ξ]
            END IF
        END FOR
        SET ω TO LENGTH(Ω) - μ
        RETURN ω
    END FUNCTION
END CLASS
