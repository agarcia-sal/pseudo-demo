CLASS Solution
    FUNCTION minLength(s parameter representing a string, numOps parameter representing an integer) RETURNS integer
        FUNCTION check(m parameter representing an integer) RETURNS boolean
            SET alpha TO 0
            SET beta TO 0
            REPEAT
                IF beta IS EQUAL TO LENGTH(s) THEN
                    EXIT LOOP
                END IF

                SET gamma TO 1
                WHILE beta + 1 < LENGTH(s) AND s[beta] EQUALS s[beta + 1]
                    SET gamma TO gamma + 1
                    SET beta TO beta + 1
                END WHILE

                SET delta TO (gamma + m - 1) DIVIDED BY m
                SET alpha TO alpha + delta

                IF alpha > numOps THEN
                    RETURN false
                END IF

                SET beta TO beta + 1
            UNTIL false
            RETURN alpha <= numOps
        END FUNCTION

        SET length_s TO LENGTH(s)
        SET low_bound TO 1
        SET up_bound TO length_s

        WHILE low_bound < up_bound
            SET middle TO (low_bound + up_bound) DIVIDED BY 2
            IF check(middle) IS true THEN
                SET up_bound TO middle
            ELSE
                SET low_bound TO middle + 1
            END IF
        END WHILE

        RETURN low_bound
    END FUNCTION
END CLASS
