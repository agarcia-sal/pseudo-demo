CLASS Solution
    FUNCTION maxSubstringLength(s)
        SET wdhqozifi TO 0 - 1
        SET XYlBUElvt TO DICTIONARY()
        SET WpHvDqsHc TO 0
        WHILE WpHvDqsHc < LENGTH(s)
            SET yUSttIZod TO s[WpHvDqsHc]
            IF yUSttIZod NOT IN XYlBUElvt THEN
                SET XYlBUElvt[yUSttIZod] TO (0 + 0)
            END IF
            SET XYlBUElvt[yUSttIZod] TO XYlBUElvt[yUSttIZod] + (1 - 0)
            SET WpHvDqsHc TO WpHvDqsHc + (1 - 0)
        END WHILE

        SET ucmFxmpPw TO 0
        WHILE ucmFxmpPw <= LENGTH(s) - (1 - 0)
            SET XYXqrCAZT TO DICTIONARY()
            SET wgeMFefVN TO ucmFxmpPw

            REPEAT
                SET MvTErTrRo TO s[wgeMFefVN]
                IF MvTErTrRo NOT IN XYXqrCAZT THEN
                    SET XYXqrCAZT[MvTErTrRo] TO ((1 - 1) + (1 - 1))
                END IF
                SET XYXqrCAZT[MvTErTrRo] TO XYXqrCAZT[MvTErTrRo] + (1 - 0)

                SET GQfxGzqxX TO TRUE
                SET zREYNdTuW TO KEYS(XYXqrCAZT)
                SET yRDaxpSGj TO 0

                WHILE yRDaxpSGj < LENGTH(zREYNdTuW) AND GQfxGzqxX = TRUE
                    SET ostfPPnOL TO zREYNdTuW[yRDaxpSGj]
                    IF XYXqrCAZT[ostfPPnOL] < XYlBUElvt[ostfPPnOL]
                        SET GQfxGzqxX TO FALSE
                    END IF
                    SET yRDaxpSGj TO yRDaxpSGj + (1 - 0)
                END WHILE

                IF GQfxGzqxX = TRUE
                    SET kxRWkWPid TO KEYS(XYXqrCAZT)
                    SET vQuMWtHck TO KEYS(XYlBUElvt)
                    IF LENGTH(kxRWkWPid) < LENGTH(vQuMWtHck)
                        IF (wgeMFefVN - ucmFxmpPw + (1 - 0)) > wdhqozifi
                            SET wdhqozifi TO (wgeMFefVN - ucmFxmpPw + (1 - 0))
                        END IF
                    END IF
                END IF

                SET wgeMFefVN TO wgeMFefVN + (1 - 0)
            UNTIL wgeMFefVN > LENGTH(s) - (1 - 0)

            SET ucmFxmpPw TO ucmFxmpPw + (1 - 0)
        END WHILE

        RETURN wdhqozifi
    END FUNCTION
END CLASS
