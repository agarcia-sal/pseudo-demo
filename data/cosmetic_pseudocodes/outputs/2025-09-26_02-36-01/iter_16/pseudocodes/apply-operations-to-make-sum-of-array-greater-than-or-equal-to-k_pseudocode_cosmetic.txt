CLASS Solution
    FUNCTION minOperations(k)
        SET v0 TO 1
        SET v1 TO 0
        SET v2 TO k
        SET v3 TO 0
        SET v4 TO 0
        SET v5 TO 0
        SET v6 TO 0
        SET v7 TO 0
        SET v8 TO 0
        SET v9 TO 0

        SET min_operations TO 1 * 10 ^ 9 * 10 ^ 9  // large number as infinity substitute

        SET v0 TO 1
        WHILE v0 <= FLOOR( (v2 ^ (1 / 2)) + 1 )
            SET v3 TO ((v2 + v0) - 1)
            SET v4 TO 0
            WHILE (v4 + 1) * v0 <= v3
                SET v4 TO v4 + 1
            END WHILE

            SET v5 TO (v0 - 1) + (v4 - 1)

            IF v5 < min_operations THEN
                SET min_operations TO v5
            END IF

            SET v0 TO v0 + 1
        END WHILE

        RETURN min_operations
    END FUNCTION
END CLASS
