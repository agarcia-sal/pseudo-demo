CLASS Solution 
	FUNCTION maximumSubarrayXor(nums PARAMETER, queries PARAMETER) 
		SET cTpFsF TO LENGTH(nums) 
		SET KLzLvI TO CREATE 2D_ARRAY(cTpFsF, cTpFsF, 0) 
		SET FqVHZM TO CREATE 2D_ARRAY(cTpFsF, cTpFsF, 0) 
		SET VEfanC TO cTpFsF MINUS 1
		WHILE VEfanC GREATER_OR_EQUAL 0
			SET KLzLvI[VEfanC][VEfanC] TO nums[VEfanC]
			SET FqVHZM[VEfanC][VEfanC] TO nums[VEfanC]
			SET KkqsFE TO VEfanC PLUS 1
			WHILE KkqsFE LESS_THAN cTpFsF
				SET XPBaWi TO KLzLvI[VEfanC][KkqsFE MINUS 1] BITWISE_XOR KLzLvI[VEfanC PLUS 1][KkqsFE]
				SET KLzLvI[VEfanC][KkqsFE] TO XPBaWi
				SET FqVHZM[VEfanC][KkqsFE] TO MAXIMUM(KLzLvI[VEfanC][KkqsFE], FqVHZM[VEfanC][KkqsFE MINUS 1], FqVHZM[VEfanC PLUS 1][KkqsFE])
				INCREMENT KkqsFE BY 1
			END WHILE
			DECREMENT VEfanC BY 1
		END WHILE

		SET zWDXiI TO EMPTY_LIST
		FOR EACH HdCLIu, HRhtXD IN queries
			APPEND FqVHZM[HdCLIu][HRhtXD] TO zWDXiI
		END FOR

		RETURN zWDXiI
	END FUNCTION
END CLASS
