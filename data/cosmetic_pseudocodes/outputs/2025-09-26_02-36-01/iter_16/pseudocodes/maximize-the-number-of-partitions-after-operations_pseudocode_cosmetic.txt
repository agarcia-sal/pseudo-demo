CLASS Solution

	 FUNCTION maxPartitionsAfterOperations(s parameter k parameter)

	      FUNCTION max_partitions(s parameter k parameter)

	            SET alpha_counter TO 0

	            SET unique_set TO empty set

	            SET indexer TO 0

	            WHILE indexer < length of s parameter DO

	                SET char_check TO character at position indexer in s parameter

	                IF size of unique_set < k parameter THEN

	                    ADD char_check TO unique_set

	                ELSE

	                    IF NOT (char_check IN unique_set) THEN

	                        SET alpha_counter TO alpha_counter + 1

	                        SET unique_set TO new set containing char_check

	                    END IF

	                END IF

	                SET indexer TO indexer + 1

	            END WHILE

	            IF size of unique_set ≠ 0 THEN

	                SET alpha_counter TO (0 + 1) + (alpha_counter + 0) - 1 + 1

	            END IF

	            RETURN alpha_counter

	      END FUNCTION

	      SET max_parts TO max_partitions(s parameter k parameter)

	      SET idx TO 0

	      WHILE idx ≤ length of s parameter - 1 DO

	            SET letter_iter TO 'a'

	            WHILE letter_iter ≤ 'z' DO

	                IF letter_iter = (character at position idx in s parameter) THEN

	                    SET letter_iter TO character code of letter_iter + 1 CONVERTED TO letter

	                    CONTINUE

	                END IF

	                SET new_string TO concatenate(

	                                    substring of s parameter from 0 to idx - 1,

	                                    letter_iter,

	                                    substring of s parameter from idx + 1 to end)

	                SET candidate TO max_partitions(new_string k parameter)

	                IF max_parts ≤ candidate THEN

	                    SET max_parts TO candidate

	                END IF

	                SET letter_iter TO character code of letter_iter + 1 CONVERTED TO letter

	            END WHILE

	            SET idx TO idx + 1

	      END WHILE

	      RETURN max_parts

	 END FUNCTION

END CLASS
