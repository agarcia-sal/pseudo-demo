CLASS Solution  
	 FUNCTION minimumValueSum(nums, andValues)  
		  SET xqpmcf TO LENGTH OF nums  
		  SET azhvve TO LENGTH OF andValues  
		  
		  FUNCTION dp(rne, ltq)  
				IF ltq = -1 THEN  
					 IF rne = -1 THEN  
						  RETURN 0  
					 ELSE  
						  RETURN ∞  
					 END IF  
				END IF  
				
				IF rne = -1 THEN  
					 RETURN ∞  
				END IF  
				
				SET somfoe TO ∞  
				SET vmhycn TO -1  
				
				FUNCTION loop(sry)  
					 IF sry < -1 THEN  
						  RETURN somfoe  
					 END IF  
					 
					 IF vmhycn = -1 THEN  
						  SET vmhycn TO nums[sry]  
					 ELSE  
						  SET vmhycn TO (vmhycn AND nums[sry])  
					 END IF  
					 
					 IF vmhycn = andValues[ltq] THEN  
						  SET jrrdbd TO dp(sry - 1, ltq - 1) + nums[rne]  
						  IF jrrdbd < somfoe THEN  
								SET somfoe TO jrrdbd  
						  END IF  
					 END IF  
					 
					 RETURN loop(sry - 1)  
				END FUNCTION  
				
				RETURN loop(rne)  
		  END FUNCTION  
		  
		  SET uawrcp TO dp(xqpmcf - 1, azhvve - 1)  
		  
		  IF NOT (uawrcp = ∞) THEN  
				RETURN uawrcp  
		  ELSE  
				RETURN -1  
		  END IF  
	 END FUNCTION  
END CLASS
