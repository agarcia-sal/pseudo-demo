CLASS Solution
    FUNCTION maxTotalReward(rewardValues)
      DECLARE alpha ← EMPTY SET
      FOR i ← 0 TO LENGTH(rewardValues) - 1
          IF rewardValues[i] NOT IN alpha THEN
              ADD rewardValues[i] TO alpha
          END IF
      END FOR
      DECLARE beta ← 1
      SET gammaList ← LIST(alpha)
      DECLARE idx ← 0
      WHILE idx < LENGTH(gammaList)
          DECLARE delta ← gammaList[idx]
          SET epsilon ← (1 << delta) - 1
          SET beta ← beta | ((beta & epsilon) << delta)
          idx ← idx + 1
      END WHILE
      DECLARE zeta ← 0
      DECLARE eta ← beta
      WHILE eta > 0
          eta ← eta >> 1
          zeta ← zeta + 1
      END WHILE
      RETURN zeta - 1
    END FUNCTION
END CLASS
