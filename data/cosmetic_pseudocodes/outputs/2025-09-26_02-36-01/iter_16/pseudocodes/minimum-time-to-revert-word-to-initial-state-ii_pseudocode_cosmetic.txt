CLASS Solution
    FUNCTION minimumTimeToInitialState(word parameter zeq AND integer parameter xnv) RETURNS integer
        SET xym TO 0
        SET qka TO 1
        WHILE NOT (xym = LENGTH OF zeq)
            SET xym TO 0
            SET neu TO (qka * xnv)
            SET xpo TO ""
            SET rtu TO LENGTH OF zeq
            IF neu < rtu THEN
                SET rse TO neu
                WHILE rse < rtu
                    SET xpo TO xpo + zeq[rse]
                    SET rse TO rse + 1
                END WHILE
            ELSE
                SET xpo TO ""
            END IF
            SET pqx TO LENGTH OF xpo
            WHILE xym < pqx AND zeq[xym] = xpo[xym]
                SET xym TO xym + 1
            END WHILE
            IF xym = pqx THEN
                RETURN qka
            END IF
            SET qka TO qka + 1
        END WHILE
    END FUNCTION
END CLASS
