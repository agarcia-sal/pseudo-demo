CLASS Solution
    FUNCTION maximumStrength(nums k)
       SET a TO LENGTH OF nums
       SET B TO a + 1
       SET C TO k + 1
       
       DEFINE dp AS ARRAY WITH B ELEMENTS EACH CONTAINING ARRAY WITH C ELEMENTS EACH SET TO (-1) * (10 ^ 9)
       SET dp[0][0] TO 0
       
       SET X TO 1
       WHILE X <= a
           SET Y TO 1
           WHILE Y <= k
               SET Z TO 0
               SET W TO X
               WHILE W >= 1
                   SET Z TO Z + nums[W - 1]
                   IF (Y % 2) == 1 THEN
                      SET sign TO (k - Y) + 1
                   ELSE
                      SET sign TO -( (k - Y) + 1 )
                   END IF
                   SET dp[X][Y] TO MAX(dp[X][Y], dp[W - 1][Y - 1] + (sign * Z))
                   SET W TO W - 1
               END WHILE
               SET dp[X][Y] TO MAX(dp[X][Y], dp[X - 1][Y])
               SET Y TO Y + 1
           END WHILE
           SET X TO X + 1
       END WHILE
       
       RETURN dp[a][k]
    END FUNCTION
END CLASS
