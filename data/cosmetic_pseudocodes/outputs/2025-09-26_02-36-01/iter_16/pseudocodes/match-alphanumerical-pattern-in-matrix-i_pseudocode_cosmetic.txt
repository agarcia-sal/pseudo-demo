CLASS Solution
    FUNCTION findPattern(board, pattern)
      SET A TO LENGTH(board)
      SET B TO LENGTH(board[0])
      SET C TO LENGTH(pattern)
      SET D TO LENGTH(pattern[0])

      FUNCTION matches(X, Y)
        SET M TO EMPTY_MAP
        SET N TO EMPTY_MAP
        SET U TO 0
        WHILE U LESS_THAN C
          SET V TO 0
          REPEAT
            IF V GREATER_EQUAL D THEN BREAK
            SET W TO pattern[U][V]
            SET Z TO board[X+U][Y+V]
            IF W GREATER_EQUAL '0' AND W LESS_EQUAL '9' THEN
              IF INTEGER(W) NOT_EQUAL Z THEN
                RETURN False
              END IF
            ELSE
              IF M CONTAINS_KEY W THEN
                IF M[W] NOT_EQUAL Z THEN
                  RETURN False
                END IF
              ELSE
                IF N CONTAINS_KEY Z THEN
                  RETURN False
                END IF
                SET M[W] TO Z
                SET N[Z] TO W
              END IF
            END IF
            SET V TO V + 1
          UNTIL FALSE
          SET U TO U + 1
        END WHILE
        RETURN True
      END FUNCTION

      SET I TO 0
      REPEAT
        IF I GREATER_THAN A - C THEN BREAK
        SET J TO 0
        WHILE J LESS_EQUAL B - D
          IF matches(I, J) THEN
            RETURN [I, J]
          END IF
          SET J TO J + 1
        END WHILE
        SET I TO I + 1
      UNTIL FALSE

      RETURN [-1, -1]
    END FUNCTION
END CLASS
