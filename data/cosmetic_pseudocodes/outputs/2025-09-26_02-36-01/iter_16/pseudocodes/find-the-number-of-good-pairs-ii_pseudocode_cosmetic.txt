CLASS Solution
		FUNCTION numberOfPairs(nums1, nums2, k)
val_mxgc := dict()
bmitjr := 0
idx_paje := 1
WHILE idx_paje <= LENGTH(nums2)
	cwrqby := nums2[idx_paje]
	IF val_mxgc[cwrqby] IS UNDEFINED THEN
		val_mxgc[cwrqby] = 0
	END IF
	val_mxgc[cwrqby] = val_mxgc[cwrqby] + 1
	idx_paje = idx_paje + 1
END WHILE
sufreho := 1
WHILE sufreho <= LENGTH(nums1)
	aslpbr := nums1[sufreho]
	akruhw := GET_KEYS(val_mxgc)
	qypogr := 1
	WHILE qypogr <= LENGTH(akruhw)
		vemcot := akruhw[qypogr]
		acjrvy := aslpbr MOD (vemcot * k)
		IF (acjrvy = 0) THEN
			bmitjr = bmitjr + val_mxgc[vemcot]
		END IF
		qypogr = qypogr + 1
	END WHILE
	sufreho = sufreho + 1
END WHILE
RETURN bmitjr
		END FUNCTION
END CLASS
