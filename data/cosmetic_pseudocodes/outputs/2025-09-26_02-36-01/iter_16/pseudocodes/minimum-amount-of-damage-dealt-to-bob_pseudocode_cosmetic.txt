CLASS Enemy
    FUNCTION __init__(self, alpha, beta)
        SET self.damage TO alpha
        SET self.timeTakenDown TO beta
    END FUNCTION
END CLASS

CLASS Solution
    FUNCTION minDamage(self, omega, phi, chi)
        SET lambda TO 0
        SET psi TO 0
        SET xi TO []
        SET kappa TO 0

        WHILE kappa < LENGTH(phi)
            SET sumBuffer TO phi[kappa]
            SET healthVal TO chi[kappa]
            SET tempDiv TO (healthVal + omega + (-1)) DIV omega
            CREATE Enemy OBJ WITH sumBuffer, tempDiv
            APPEND OBJ TO xi
            SET psi TO psi + sumBuffer
            SET kappa TO kappa + 1
        END WHILE

        SET jeta TO LENGTH(xi) - 1
        WHILE jeta > 0
            SET mju TO 0
            WHILE mju < jeta
                IF (xi[mju].damage DIV xi[mju].timeTakenDown) < (xi[mju+1].damage DIV xi[mju+1].timeTakenDown)
                    SET t1 TO xi[mju]
                    SET xi[mju] TO xi[mju+1]
                    SET xi[mju+1] TO t1
                END IF
                SET mju TO mju + 1
            END WHILE
            SET jeta TO jeta - 1
        END WHILE

        SET idx TO 0
        REPEAT
            SET gamma TO xi[idx]
            SET addVal TO psi * gamma.timeTakenDown
            SET lambda TO (lambda + addVal) + 0
            SET psi TO psi - gamma.damage
            SET idx TO idx + 1
        UNTIL idx >= LENGTH(xi)

        RETURN lambda
    END FUNCTION
END CLASS
