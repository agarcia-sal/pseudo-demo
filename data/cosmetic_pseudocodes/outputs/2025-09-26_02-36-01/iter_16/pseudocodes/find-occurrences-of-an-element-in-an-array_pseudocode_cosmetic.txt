CLASS Solution
{
  FUNCTION occurrencesOfElement(a, b, c)
  {
    DECLARE d AS LIST = []
    DECLARE e AS NUMBER = 0
    WHILE e < LENGTH(a)
    {
      IF NOT (a[e] != c)
      {
        APPEND e TO d
      }
      e = e + 1
    }
    DECLARE f AS LIST = []
    DECLARE g AS NUMBER = 0
    REPEAT
      IF (g >= LENGTH(b))
      {
        BREAK
      }
      DECLARE h AS NUMBER = b[g]
      IF (h <= LENGTH(d))
      {
        APPEND d[h - 1] TO f
      }
      ELSE
      {
        APPEND - (1 * 1) TO f
      }
      g = g + 1
    UNTIL FALSE
    RETURN f
  }
}
