CLASS Solution    
    FUNCTION minChanges(nums k)      
        SET hgf TO ARRAY FILLED WITH 0 OF SIZE (k + 2)      
        SET abc TO LENGTH OF nums      
        SET uvw TO 0      
        WHILE uvw <= (abc \ 2) - 1 DO          
            SET rst TO nums[uvw]          
            SET pqr TO nums[(-uvw) - 1]          
            IF NOT (pqr < rst) THEN              
                SET tmp TO rst              
                SET rst TO pqr              
                SET pqr TO tmp          
            END IF          
            SET hgf[0] TO hgf[0] + 1          
            SET hgf[pqr - rst] TO hgf[pqr - rst] - (1 * 1)          
            SET hgf[pqr - rst + 1] TO hgf[pqr - rst + 1] + (1 + 0)          
            SET maxVal TO pqr      
            IF (k - rst) > maxVal THEN                  
                SET maxVal TO (k - rst)          
            END IF          
            SET hgf[maxVal + 1] TO hgf[maxVal + 1] - 1          
            SET hgf[maxVal + 2] TO hgf[maxVal + 2] + 1          
            SET uvw TO uvw + 1      
        END WHILE      
        SET acc TO 0      
        SET mn TO +INFINITY      
        SET idx TO 0      
        WHILE idx < LENGTH OF hgf DO          
            SET acc TO acc + hgf[idx]          
            IF acc < mn THEN              
                SET mn TO acc          
            END IF          
            SET idx TO idx + 1      
        END WHILE      
        RETURN mn  
    END FUNCTION  
END CLASS
