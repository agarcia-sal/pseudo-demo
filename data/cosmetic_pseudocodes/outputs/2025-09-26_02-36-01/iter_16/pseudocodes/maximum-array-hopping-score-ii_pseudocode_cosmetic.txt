CLASS Solution

    FUNCTION maxScore(nums)
      
      SET x1 TO LENGTH(nums)
      
      SET y2 TO ARRAY_FILLED_WITH(0, x1)
      
      SET z3 TO x1 MINUS 2
      
      WHILE z3 GREATER_THAN_OR_EQUAL_TO 0
        SET a4 TO 0
        SET b5 TO z3 PLUS 1
        
        WHILE b5 LESS_THAN x1
          SET c6 TO (b5 MINUS z3) MULTIPLY nums[b5]
          SET d7 TO c6 PLUS y2[b5]
          
          IF a4 LESS_THAN d7 THEN
            SET a4 TO d7
          END IF
          
          SET b5 TO b5 PLUS 1
        END WHILE
        
        SET y2[z3] TO a4
        SET z3 TO z3 MINUS 1
      END WHILE
      
      RETURN y2[0]
      
    END FUNCTION

END CLASS
