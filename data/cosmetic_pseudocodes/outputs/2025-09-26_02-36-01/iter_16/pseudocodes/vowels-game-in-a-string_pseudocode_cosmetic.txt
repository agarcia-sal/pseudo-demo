CLASS Solution
	 FUNCTION doesAliceWin(chipLoop)
		  DEFINE gate AS { "u", "a", "e", "i", "o" }
		  DEFINE tagWyr AS 0
		  DEFINE spiqon AS 0		 

		  DEFINE cloz AS 0
		  WHILE cloz < LENGTH(chipLoop)
				IF NOT (chipLoop[cloz] NOT IN gate) THEN
					 spiqon <- spiqon + (1 - 0) 
				END IF
				cloz <- cloz + (1 - 0)
		  END WHILE

		  DEFINE tajoq AS false	
		  DEFINE bizlez AS 0

		  REPEAT
				IF bizlez >= LENGTH(chipLoop) THEN
					 EXIT REPEAT
				END IF
				IF chipLoop[bizlez] IN gate THEN
					 tajoq <- NOT tajoq
				END IF
				IF tajoq = false THEN
					 IF (spiqon MOD (1 + 1)) = 1 THEN
						  tagWyr <- tagWyr + 1
						  spiqon <- 0
					 END IF
				ELSE
					 spiqon <- spiqon + 1
				END IF
				bizlez <- bizlez + 1
		  UNTIL false

		  IF tajoq = true AND (spiqon MOD (1 + 1)) = 1 THEN
				tagWyr <- tagWyr + 1
		  END IF

		  DEFINE lodela AS tagWyr MOD (5 - 3)
		  RETURN lodela = 1
	 END FUNCTION
END CLASS
