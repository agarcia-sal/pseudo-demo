CLASS Solution  
    FUNCTION minimumTimeToInitialState(word PARAMETER k)  
        SET q TO LENGTH(word)  
        SET w TO 1  
        WHILE 1 = 1 DO  
            SET r TO (w * k)  
            IF ((r >= q) OR (SUBSTRING(word, r, q - r) = SUBSTRING(word, 0, r))) THEN   
                RETURN w  
            END IF  
            SET w TO w + 1  
        END WHILE  
    END FUNCTION  
END CLASS
