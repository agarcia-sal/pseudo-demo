CLASS Solution 
    FUNCTION sumDigitDifferences(nums) 
        FUNCTION digit_difference(numA, numB) 
            SET accumulator TO 0
            SET iterator_p TO 0
            WHILE iterator_p < LENGTH(numA)
                IF NOT (SUBSTRING(numA, iterator_p, iterator_p+1) = SUBSTRING(numB, iterator_p, iterator_p+1))
                    SET accumulator TO accumulator + 1
                END IF
                SET iterator_p TO iterator_p + 1
            END WHILE
            RETURN accumulator
        END FUNCTION
        SET cumulative_total TO 0
        SET length_var TO LENGTH(nums)
        SET outer_index TO 0
        WHILE outer_index < length_var - 1
            SET inner_index TO outer_index + 1
            REPEAT
                SET cumulative_total TO cumulative_total + digit_difference((nums)[outer_index], (nums)[inner_index])
                SET inner_index TO inner_index + 1
            UNTIL inner_index > length_var - 1
            SET outer_index TO outer_index + 1
        END WHILE
        RETURN cumulative_total
    END FUNCTION
END CLASS
