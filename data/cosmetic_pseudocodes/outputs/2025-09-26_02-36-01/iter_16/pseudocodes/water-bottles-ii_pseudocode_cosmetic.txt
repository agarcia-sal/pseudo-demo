CLASS Solution
{
    FUNCTION maxBottlesDrunk(numBottles numBottles, numExchange numExchange)
    {
        DECLARE vAspqdm AS INTEGER = 0
        DECLARE uJxwrpz AS INTEGER = 0

        WHILE (numBottles > 0)
        {
            LET vAspqdm = vAspqdm + numBottles
            LET uJxwrpz = uJxwrpz + numBottles
            LET numBottles = 0

            WHILE (uJxwrpz >= numExchange)
            {
                LET uJxwrpz = uJxwrpz - numExchange
                LET numBottles = numBottles + 1
                LET numExchange = numExchange + 1
            }
        }

        RETURN vAspqdm
    }
}
