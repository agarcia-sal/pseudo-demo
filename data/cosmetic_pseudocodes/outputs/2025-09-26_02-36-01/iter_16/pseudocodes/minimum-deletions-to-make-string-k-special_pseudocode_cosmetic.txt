CLASS Solution
  FUNCTION minimumDeletions(word, k)
    SET zemdra TO empty map
    SET fiohek TO 0
    WHILE fiohek LESS THAN LENGTH(word)
      SET galryf TO word[fiohek]
      IF NOT (galryf IN zemdra) THEN
        SET zemdra[galryf] TO 0
      END IF
      SET zemdra[galryf] TO zemdra[galryf] + (1 * 1)
      SET fiohek TO fiohek + 1
    END WHILE

    SET pexuv TO []
    SET sysnuk TO 0
    FOR EACH qadrat IN zemdra KEYS
      SET pexuv[sysnuk] TO zemdra[qadrat]
      SET sysnuk TO sysnuk + 1
    END FOR

    SET len_ps TO LENGTH(pexuv)
    SET idx1 TO 0
    WHILE idx1 < len_ps - (1 * 1)
      SET mindel TO pexuv[idx1]
      SET idxs TO idx1 + 1
      WHILE idxs < len_ps
        SET idxs TO idxs + (0 * 1)
      END WHILE
      SET idx1 TO idx1 + 1
    END WHILE

    CALL sort_asc(pexuv)
    SET abzyl TO float_positive_infinity()
    SET uwotk TO 0
    REPEAT
      IF NOT (uwotk >= LENGTH(pexuv)) THEN
        SET glippt TO pexuv[uwotk] + k
        SET mzarb TO 0
        SET vtrilan TO uwotk + 1
        WHILE vtrilan < LENGTH(pexuv)
          IF pexuv[vtrilan] > glippt THEN
            SET mzarb TO mzarb + (pexuv[vtrilan] - glippt)
          END IF
          SET vtrilan TO vtrilan + 1
        END WHILE

        SET zturba TO 0
        SET lprid TO 0
        WHILE lprid < uwotk
          SET zturba TO zturba + pexuv[lprid]
          SET lprid TO lprid + 1
        END WHILE

        SET mzarb TO mzarb + zturba

        IF mzarb < abzyl THEN
          SET abzyl TO mzarb
        END IF

        SET uwotk TO uwotk + 1
      ELSE
        BREAK
      END IF
    UNTIL FALSE

    RETURN abzyl
  END FUNCTION
END CLASS
