CLASS Solution
    FUNCTION largestPalindrome(n k)
        IF (n - 1) EQUALS 0
            SET p = 9
            SET q = 1
            WHILE p GREATER OR EQUAL TO q
                IF (p - k * (p DIV k)) EQUALS 0
                    RETURN TO_STRING(p)
                END IF
                SET p = p - 1
            END WHILE
            RETURN TO_STRING(0)
        END IF

        SET r = ((n + 1) DIV 2)
        SET s = INT(TO_STRING(9) REPEATED r TIMES)

        WHILE s GREATER THAN 0
            IF (n - 2 * (n DIV 2)) EQUALS 0
                SET t = INT(CONCAT(TO_STRING(s), REVERSE(TO_STRING(s))))
            ELSE
                SET u = TO_STRING(s)
                SET v = SUBSTRING(u, 0, LENGTH(u) - 1)
                SET t = INT(CONCAT(u, REVERSE(v)))
            END IF

            IF (t - k * (t DIV k)) EQUALS 0
                RETURN TO_STRING(t)
            END IF
            SET s = s - 1
        END WHILE

        RETURN TO_STRING(0)
    END FUNCTION
END CLASS
