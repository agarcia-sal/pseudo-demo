CLASS Solution
    FUNCTION minOperations(m)
        DEFINE FUNCTION integerSqrt(n)
            SET p TO 0
            SET q TO n
            WHILE p <= q
                SET r TO (p + q) // 2
                IF r * r <= n THEN
                    SET p TO r + 1
                ELSE
                    SET q TO r - 1
                END IF
            END WHILE
            RETURN q
        END FUNCTION

        SET delta TO 2147483647
        SET alpha TO 1
        SET beta TO integerSqrt(m)
        SET gamma TO beta + 1

        PROCEDURE updateMin(z)
            IF z < delta THEN
                SET delta TO z
            END IF
        END PROCEDURE

        SET zeta TO alpha
        WHILE zeta < gamma
            SET eta TO ((m + zeta - 1) // zeta)
            SET theta TO (zeta - 1) + (eta - 1)
            CALL updateMin(theta)
            SET zeta TO zeta + 1
        END WHILE

        RETURN delta
    END FUNCTION
END CLASS
