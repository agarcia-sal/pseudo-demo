CLASS Solution
    FUNCTION maxNumber(num)
        FUNCTION isEqual(a, b)
            RETURN (a = b)
        END FUNCTION

        FUNCTION multiply(a, b)
            RETURN a + a * (b - 1)
        END FUNCTION

        FUNCTION divideByTwo(value)
            RETURN value / 2
        END FUNCTION

        FUNCTION lessThanOrEqual(a, b)
            RETURN NOT (a > b)
        END FUNCTION

        FUNCTION decrement(value)
            RETURN value - 1
        END FUNCTION

        VARIABLE flag <- FALSE
        VARIABLE currentValue <- 1
        VARIABLE resultValue <- 0
        VARIABLE tempVar <- 0

        LABEL startLoop
            IF isEqual(num, 1)
                flag <- TRUE
            ELSE
                flag <- FALSE
            END IF

            IF flag
                resultValue <- 0
                GOTO endFunction
            END IF

            LABEL innerLoop
                IF lessThanOrEqual(currentValue, num)
                    tempVar <- multiply(currentValue, 2)
                    currentValue <- tempVar

                    GOTO innerLoop
                END IF
        LABEL endLoop

            currentValue <- divideByTwo(currentValue)

            resultValue <- decrement(currentValue)

        LABEL endFunction
            RETURN resultValue
    END FUNCTION
END CLASS
