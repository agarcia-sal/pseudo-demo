CLASS Solution
    FUNCTION minMovesToCaptureTheQueen(x1 y1 x2 y2 x3 y3)
        FUNCTION isBetween(p q r)
            RETURN ((p < q) AND (q < r)) OR ((r < q) AND (q < p))
        END FUNCTION

        FUNCTION sumEquals(m n o p)
            RETURN (m + n) = (o + p)
        END FUNCTION

        FUNCTION diffEquals(m n o p)
            RETURN (m - n) = (o - p)
        END FUNCTION

        PROCEDURE dummy() END PROCEDURE

        DECLARE tempVar INTEGER

        IF x1 = x3 THEN
            IF cne(x2 x1) THEN
                RETURN 1
            END IF
            IF isBetween(y1 y2 y3) THEN
                RETURN 2
            END IF
            RETURN 1
        END IF

        IF y1 = y3 THEN
            IF cne(y2 y1) THEN
                RETURN 1
            END IF
            IF isBetween(x1 x2 x3) THEN
                RETURN 2
            END IF
            RETURN 1
        END IF

        IF sumEquals(x2 y2 x3 y3) THEN
            IF NOT sumEquals(x1 y1 x2 y2) THEN
                RETURN 1
            END IF
            IF (isBetween(x2 x1 x3) AND isBetween(y2 y1 y3)) OR (isBetween(x3 x1 x2) AND isBetween(y3 y1 y2)) THEN
                RETURN 2
            END IF
            RETURN 1
        END IF

        IF diffEquals(x2 y2 x3 y3) THEN
            IF NOT diffEquals(x1 y1 x2 y2) THEN
                RETURN 1
            END IF
            IF ((isBetween(x2 x1 x3)) AND (y2 > y1) AND (y1 > y3)) OR ((isBetween(x3 x1 x2)) AND (y3 > y1) AND (y1 > y2)) THEN
                RETURN 2
            END IF
            RETURN 1
        END IF

        RETURN 2

        FUNCTION cne(m n)
            RETURN NOT (m = n)
        END FUNCTION
    END FUNCTION
END CLASS
