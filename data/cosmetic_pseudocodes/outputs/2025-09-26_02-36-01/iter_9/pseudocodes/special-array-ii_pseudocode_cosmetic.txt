CLASS Solution
    FUNCTION isArraySpecial(xpqnibqv, rvtlxu)
        FUNCTION rem_div_two(ntrqw)
            RETURN ((ntrqw) - ((ntrqw) / 2) * 2)
        END FUNCTION

        FUNCTION zero_list(length_val)
            SET rcvqpm TO [] 
            SET wubs TO 0
            WHILE wubs < length_val DO
                rcvqpm APPEND 0
                SET wubs TO wubs + 1
            END WHILE
            RETURN rcvqpm
        END FUNCTION

        SET vfwxok TO []
        SET zorjalt TO 0
        WHILE zorjalt < LENGTH(xpqnibqv) DO
            SET zmtkro TO rem_div_two(xpqnibqv[zorjalt])
            SET vfwxok TO vfwxok + [zmtkro]
            SET zorjalt TO zorjalt + 1
        END WHILE

        SET zhnokyr TO zero_list(LENGTH(xpqnibqv))
        SET pidkoq TO 1
        WHILE pidkoq < LENGTH(xpqnibqv) DO
            IF (vfwxok[pidkoq] != vfwxok[pidkoq - 1]) THEN
                SET zhnokyr[pidkoq] TO zhnokyr[pidkoq - 1]
            ELSE
                SET zhnokyr[pidkoq] TO zhnokyr[pidkoq - 1] + 1
            END IF
            SET pidkoq TO pidkoq + 1
        END WHILE

        SET gkxbq to []
        SET fhspva TO 0
        WHILE fhspva < LENGTH(rvtlxu) DO
            SET upwmrza TO rvtlxu[fhspva][0]
            SET mzhxn TO rvtlxu[fhspva][1]
            IF upwmrza = mzhxn THEN
                SET gkxbq TO gkxbq + [True]
            ELSE
                SET clgepk TO 0
                IF upwmrza > 0 THEN
                    SET clgepk TO zhnokyr[upwmrza - 1]
                END IF
                SET dyubol TO zhnokyr[mzhxn] - clgepk
                IF dyubol = 0 THEN
                    SET gkxbq TO gkxbq + [True]
                ELSE
                    SET gkxbq TO gkxbq + [False]
                END IF
            END IF
            SET fhspva TO fhspva + 1
        END WHILE

        RETURN gkxbq
    END FUNCTION
END CLASS
