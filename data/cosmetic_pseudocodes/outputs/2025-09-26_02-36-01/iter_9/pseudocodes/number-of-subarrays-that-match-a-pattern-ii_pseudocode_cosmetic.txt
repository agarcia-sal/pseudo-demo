CLASS Solution
    FUNCTION countMatchingSubarrays(alpha, beta)
        FUNCTION derive_relation(a, b)
            IF (NOT (a >= b)) 
                RETURN 1
            ELSE
                IF (a = b) THEN
                    RETURN 0
                ELSE
                    RETURN -1
                END IF
            END IF
        END FUNCTION

        SET length_alpha TO 0
        SET cursor TO 0
        SET length_beta TO 0
        SET tally TO 0

        SET length_alpha TO SIZE alpha
        SET length_beta TO SIZE beta
        
        SET links TO []
        SET index_tracker TO 0
        
        WHILE index_tracker <= length_alpha - 2
            APPEND derive_relation(alpha[index_tracker], alpha[index_tracker + 1]) TO links
            SET index_tracker TO index_tracker + 1
        END WHILE

        SET scan_pos TO 0
        SET limit TO (length_alpha - length_beta) - 1

        WHILE scan_pos <= limit
            SET segment TO SLICE links FROM scan_pos TO scan_pos + length_beta - 1
            IF segment = beta THEN
                SET tally TO tally + 1
            END IF
            SET scan_pos TO scan_pos + 1
        END WHILE

        RETURN tally
    END FUNCTION
END CLASS
