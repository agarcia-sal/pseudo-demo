CLASS Solution
    FUNCTION minValidStrings(tangoFoxtrot, whiskeyZulu)
        PROCEDURE collectPrefixes(alphaBravo)
            SET kiloLima TO new EMPTY SET
            PROCEDURE accumulate(charlie)
                FUNCTION innerAccumulate(deltaEcho)
                    IF deltaEcho > LENGTH(charlie) THEN
                        RETURN
                    END IF
                    INSERT SUBSTRING(charlie, 1, deltaEcho) INTO kiloLima
                    CALL innerAccumulate(deltaEcho + 1)
                END FUNCTION
                CALL innerAccumulate(1)
            END PROCEDURE
            WHILE LENGTH(alphaBravo) > 0 DO
                SET hotelIndia TO alphaBravo[0]
                REMOVE FIRST ELEMENT FROM alphaBravo
                CALL accumulate(hotelIndia)
            END WHILE
            RETURN kiloLima
        END PROCEDURE

        PROCEDURE getMin(a, b)
            IF a < b THEN
                RETURN a
            ELSE
                RETURN b
            END IF
        END PROCEDURE

        SET kiloLima TO collectPrefixes(tangoFoxtrot)
        SET uniformVictor TO LENGTH(whiskeyZulu)
        SET yankeeXray TO []
        REPEAT
            INSERT 1000000000 INTO yankeeXray
        UNTIL LENGTH(yankeeXray) = uniformVictor + 1
        SET yankeeXray[0] TO 0

        FUNCTION substringExists(bravoChar, charStart, charEnd)
            RETURN SUBSTRING(bravoChar, charStart, charEnd) IN kiloLima
        END FUNCTION

        PROCEDURE fillDP(index)
            IF index > uniformVictor THEN
                RETURN
            END IF
            SET juliettTango TO 1
            WHILE juliettTango <= index DO
                SET subStr TO SUBSTRING(whiskeyZulu, juliettTango, index)
                IF subStr IN kiloLima THEN
                    SET yankeeXray[index] = getMin(yankeeXray[index], yankeeXray[juliettTango - 1] + 1)
                END IF
                SET juliettTango TO juliettTango + 1
            END WHILE
            CALL fillDP(index + 1)
        END PROCEDURE

        CALL fillDP(1)

        IF yankeeXray[uniformVictor] != 1000000000 THEN
            RETURN yankeeXray[uniformVictor]
        ELSE
            RETURN -1
        END IF
    END FUNCTION
END CLASS
