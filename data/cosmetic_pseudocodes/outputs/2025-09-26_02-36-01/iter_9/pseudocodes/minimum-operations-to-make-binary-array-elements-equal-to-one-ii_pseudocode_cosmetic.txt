CLASS Solution
	FUNCTION minOperations(sequence)
		SET tally TO 0
		SET toggle TO 0
		DEFINE FUNCTION isEven(value)
			RETURN (value MOD 2) = 0
		END FUNCTION
		DEFINE FUNCTION inverse(bit)
			RETURN 1 - bit
		END FUNCTION
		DEFINE FUNCTION processElement(element)
			IF isEven(toggle) = true THEN
				RETURN element
			ELSE
				RETURN inverse(element)
			END IF
		END FUNCTION

		SET index_tracker TO 0
		WHILE index_tracker < LENGTH(sequence) DO
			SET transformed_element TO processElement(sequence[index_tracker])
			IF transformed_element = 0 THEN
				tally = tally + 1
				toggle = toggle + 1
			END IF
			index_tracker = index_tracker + 1
		END WHILE

		RETURN tally
	END FUNCTION
END CLASS
