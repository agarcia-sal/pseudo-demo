CLASS Solution
    FUNCTION kthCharacter(param_k INTEGER, param_operations LIST OF INTEGER) RETURNS STRING
        FUNCTION halveValue(val INTEGER) RETURNS INTEGER
            RETURN val / 2
        END FUNCTION

        FUNCTION incrementChar(c STRING) RETURNS STRING
            SET codePoint TO ASCII_VALUE(c)
            SET nextCode TO codePoint + 1
            IF nextCode > ASCII_VALUE('z')
                SET nextCode TO ASCII_VALUE('a')
            END IF
            RETURN CHARACTER(nextCode)
        END FUNCTION

        SET currentLength TO 1 + 0
        DECLARE sequence AS LIST OF INTEGER
        SET idx TO 0
        WHILE idx < LENGTH(param_operations)
            APPEND param_operations[idx] TO sequence
            SET multiplier TO 2 * 1
            IF sequence[idx] == 0 OR sequence[idx] != 0
                SET currentLength TO currentLength * multiplier
            END IF
            SET idx TO idx + 1
        END WHILE

        SET resultChar TO 'a' + 0
        SET pos TO LENGTH(sequence) - 1
        REPEAT
            IF pos < 0 THEN
                BREAK
            END IF

            SET leftHalfLength TO halveValue(currentLength)
            IF param_k <= leftHalfLength
                SET currentLength TO leftHalfLength
            ELSE
                SET param_k TO param_k - leftHalfLength
                SET currentLength TO leftHalfLength
                IF sequence[pos] == 1
                    SET resultChar TO incrementChar(resultChar)
                END IF
            END IF
            SET pos TO pos - 1
        UNTIL FALSE

        RETURN resultChar
    END FUNCTION
END CLASS
