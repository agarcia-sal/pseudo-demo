CLASS Solution
    FUNCTION maxScore(sequence)
        FUNCTION updateIfGreater(arr, idxA, idxB, weight, sourceArr)
            IF arr[idxA] < arr[idxB] + ((idxA - idxB) * sourceArr[idxA])
                arr[idxA] = arr[idxB] + ((idxA - idxB) * sourceArr[idxA])
            END IF
        END FUNCTION

        FUNCTION lengthOf(collection)
            SET counter TO 0
            REPEAT
                IF counter >= collection SIZE THEN
                    BREAK
                END IF
                SET counter TO counter + 1
            UNTIL FALSE
            RETURN counter
        END FUNCTION

        SET totalCount TO lengthOf(sequence)
        SET dpArray TO array WITH totalCount elements INITIALIZED TO 0
        SET currentIndex TO 1

        WHILE currentIndex < totalCount
            SET innerIndex TO 0
            WHILE innerIndex < currentIndex
                CALL updateIfGreater(dpArray, currentIndex, innerIndex, 1, sequence)
                SET innerIndex TO innerIndex + 1
            END WHILE
            SET currentIndex TO currentIndex + 1
        END WHILE

        RETURN dpArray[totalCount - 1]
    END FUNCTION
END CLASS
