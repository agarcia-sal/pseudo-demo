CLASS Solution
	Function waysToReachStair WITH PARAMETER deltaK
		Function explore WITH PARAMETERS pos q flag
			IF NOT (pos <= deltaK + 1) THEN
				RETURN 0
			END IF

			SET count TO 0
			IF pos == deltaK THEN
				SET count TO 1
			END IF

			IF (pos > 0) THEN
				IF q == 0 THEN
					LET tmp1 BE explore(pos - 1, 1, flag)
					SET count TO count + tmp1
				END IF
			END IF

			LET tmp2 BE explore(pos + (2 * flag), 0, flag + 1)
			SET count TO count + tmp2

			RETURN count
		END FUNCTION

		RETURN explore(1, 0, 0)
	END FUNCTION
END CLASS
