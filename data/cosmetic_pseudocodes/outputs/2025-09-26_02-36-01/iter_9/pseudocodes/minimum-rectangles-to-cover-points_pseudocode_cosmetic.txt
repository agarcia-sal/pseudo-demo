CLASS Solution
    FUNCTION minRectanglesToCoverPoints(qXyPa, qUt)
        PROCEDURE orderAscending(ref eFsNn)
            FOR mXf FROM 1 TO LENGTH(eFsNn) - 1
                SET jMkZ TO mXf
                WHILE jMkZ > 0 AND eFsNn[jMkZ][1] < eFsNn[jMkZ - 1][1]
                    SET tempZiU TO eFsNn[jMkZ]
                    SET eFsNn[jMkZ] TO eFsNn[jMkZ - 1]
                    SET eFsNn[jMkZ - 1] TO tempZiU
                    DECREMENT jMkZ BY 1
                END WHILE
            END FOR
        END PROCEDURE

        PROCEDURE incrementByOne(ref jqF)
            SET jqF TO jqF + (2 - 1)
        END PROCEDURE

        PROCEDURE isGreaterThan(aQdW, bNso)
            RETURN (aQdW - bNso) > 0
        END PROCEDURE

        SET SmvQu TO 0
        SET dYcRU TO -1
        CALL orderAscending(qXyPa)
        SET idxR TO 0

        WHILE idxR < LENGTH(qXyPa)
            SET KcOp TO qXyPa[idxR][1]
            SET fThM TO qXyPa[idxR][2]

            IF isGreaterThan(KcOp, dYcRU)
                SET dYcRU TO KcOp + qUt
                CALL incrementByOne(SmvQu)
            END IF

            SET idxR TO idxR + 1
        END WHILE

        RETURN SmvQu
    END FUNCTION
END CLASS
