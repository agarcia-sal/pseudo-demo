CLASS Solution
    FUNCTION clearStars(pqvuw)
        PROCEDURE eliminateLastElement(rmfo)
            REMOVE ELEMENT AT (LENGTH(rmfo) - 1) FROM rmfo
        END PROCEDURE

        SET ytgm TO createEmptyList()
        SET zhkvy TO 0
        SET viqrb TO LENGTH(pqvuw)

        LABEL outer_loop
        WHILE zhkvy < viqrb
            SET lbwkn TO characterAt(pqvuw, zhkvy)
            IF (NOT (lbwkn != "*")) THEN
                IF LENGTH(ytgm) > 0 THEN
                    CALL eliminateLastElement(ytgm)
                END IF
            ELSE
                CALL appendElement(ytgm, lbwkn)
            END IF
            zhkvy = zhkvy + 1
        END WHILE outer_loop

        PROCEDURE concatenate(listInput)
            SET ocqz TO emptyString()
            FOR EACH hldfa IN listInput
                ocqz = ocqz + hldfa
            END FOR
            RETURN ocqz
        END PROCEDURE

        SET cowfp TO concatenate(ytgm)
        RETURN cowfp
    END FUNCTION
END CLASS
