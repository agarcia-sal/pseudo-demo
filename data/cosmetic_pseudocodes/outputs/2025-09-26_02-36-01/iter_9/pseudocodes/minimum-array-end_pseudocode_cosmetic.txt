CLASS Solution
    FUNCTION minEnd(n z)
        
        FUNCTION canConstruct(limit)
            FUNCTION bitwiseAnd(a, b)
                RETURN a & b
            END FUNCTION

            SET temp := z
            SET tally := 1

            REPEAT
                IF NOT (temp < limit) THEN
                    BREAK
                END IF

                SET temp := temp + 1

                IF bitwiseAnd(temp, z) = z THEN
                    SET tally := tally + 1

                    IF tally = n THEN
                        RETURN True
                    END IF
                END IF
            UNTIL False

            RETURN (tally = n)
        END FUNCTION

        SET low := z
        SET high := 2 * 10 * 10 * 10 * 10 * 10 * 10 * 10 * 10

        FUNCTION integerDivide(a, b)
            RETURN a // b
        END FUNCTION

        FUNCTION average(a, b)
            RETURN integerDivide(a + b, 2)
        END FUNCTION

        FUNCTION isLess(a, b)
            RETURN a < b
        END FUNCTION

        WHILE isLess(low, high)
            SET middle := average(low, high)

            IF canConstruct(middle) THEN
                SET high := middle
            ELSE
                SET low := low + 1
            END IF
        END WHILE
        
        RETURN low
    END FUNCTION
END CLASS
