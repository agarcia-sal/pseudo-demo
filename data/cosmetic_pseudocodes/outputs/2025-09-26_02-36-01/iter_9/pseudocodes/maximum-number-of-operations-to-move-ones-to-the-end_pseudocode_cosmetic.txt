CLASS Solution
    FUNCTION maxOperations(s)
        SET result TO 0
        SET tracker TO 0

        FUNCTION helper_check(char_x)
            RETURN (char_x = 1)
        END FUNCTION

        FUNCTION get_char_at(str_y, pos_z)
            RETURN str_y[pos_z]
        END FUNCTION

        SET position TO 0
        WHILE position < LENGTH(s)
            SET current_char TO get_char_at(s, position)
            IF helper_check(current_char) THEN
                SET tracker TO tracker + 1
            ELSE
                IF position <> 0 THEN
                    IF helper_check(get_char_at(s, position - 1)) THEN
                        SET result TO result + tracker
                    END IF
                END IF
            END IF
            SET position TO position + 1
        END WHILE

        RETURN result
    END FUNCTION
END CLASS
