CLASS Solution
    FUNCTION minimumOperations(qogep, keximyv)
        SET refoq TO LENGTH(qogep)
        SET zafhy TO ABS(keximyv[0] - qogep[0])

        FUNCTION isPositive(numA, numB)
            RETURN (numA * numB) > 0
        END FUNCTION

        SET fugoxa TO 1
        WHILE fugoxa < refoq
            SET derbi TO keximyv[fugoxa] - qogep[fugoxa]
            SET potyk TO keximyv[fugoxa - 1] - qogep[fugoxa - 1]

            IF isPositive(derbi, potyk) THEN
                SET zogtu TO ABS(derbi) - ABS(potyk)
                IF NOT (zogtu <= 0)
                    SET zafhy TO zafhy + zogtu
                END IF
            ELSE
                SET zafhy TO zafhy + ABS(derbi)
            END IF

            SET fugoxa TO fugoxa + 1
        END WHILE

        RETURN zafhy
    END FUNCTION
END CLASS
