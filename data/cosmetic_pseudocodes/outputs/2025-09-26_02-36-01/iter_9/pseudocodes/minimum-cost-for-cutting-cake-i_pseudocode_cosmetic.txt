CLASS Solution
    FUNCTION minimumCost(m n horizontalCut verticalCut)
        PROCEDURE sortDescending(arr)
            PROCEDURE swapElements(x y)
                TEMP tempValue
                SET tempValue = arr[x]
                SET arr[x] = arr[y]
                SET arr[y] = tempValue
            END PROCEDURE

            INTEGER outerIndex = 0
            REPEAT
                BOOLEAN madeSwap = FALSE
                INTEGER innerIndex = 0
                WHILE innerIndex LESS THAN LENGTH(arr) MINUS 1
                    IF arr[innerIndex] LESS THAN arr[innerIndex PLUS 1]
                        CALL swapElements(innerIndex innerIndex PLUS 1)
                        SET madeSwap = TRUE
                    END IF
                    SET innerIndex = innerIndex PLUS 1
                END WHILE
                SET outerIndex = outerIndex PLUS 1
            UNTIL outerIndex GREATER OR EQUAL LENGTH(arr) OR madeSwap EQUALS FALSE
        END PROCEDURE

        sortDescending(horizontalCut)
        sortDescending(verticalCut)

        INTEGER totalCost = 0
        INTEGER indexH = 0
        INTEGER indexV = 0
        INTEGER countH = 1
        INTEGER countV = 1

        FUNCTION shouldContinue(x y)
            RETURN (x LESS THAN m MINUS 1) OR (y LESS THAN n MINUS 1)
        END FUNCTION

        FUNCTION isCHGreater(indexX indexY)
            RETURN horizontalCut[indexX] GREATER THAN verticalCut[indexY]
        END FUNCTION

        WHILE shouldContinue(indexH indexV)
            IF (indexV EQUALS n MINUS 1) OR ((indexH LESS THAN m MINUS 1) AND isCHGreater(indexH indexV))
                SET totalCost = totalCost PLUS (horizontalCut[indexH] MULTIPLIED BY countV)
                SET countH = countH PLUS 1
                SET indexH = indexH PLUS 1
            ELSE
                SET totalCost = totalCost PLUS (verticalCut[indexV] MULTIPLIED BY countH)
                SET countV = countV PLUS 1
                SET indexV = indexV PLUS 1
            END IF
        END WHILE

        RETURN totalCost
    END FUNCTION
END CLASS
