CLASS Solution
    FUNCTION validStrings x
        SET results TO []
        
        FUNCTION recur z
            IF LENGTH(z) = x THEN
                ADD z TO results
                RETURN
            END IF
            
            SET lastChar TO z[ LENGTH(z) - 1 ]
            
            IF lastChar = "1" THEN
                CALL recur(z + "0")
                CALL recur(z + "1")
            ELSE
                IF lastChar = "0" THEN
                    CALL recur(z + "1")
                END IF
            END IF
        END FUNCTION
        
        CALL recur("0")
        CALL recur("1")
        
        IF x = 1 THEN
            SET tempList TO []
            ADD "0" TO tempList
            ADD "1" TO tempList
            RETURN tempList
        END IF
        
        RETURN results
    END FUNCTION
END CLASS
