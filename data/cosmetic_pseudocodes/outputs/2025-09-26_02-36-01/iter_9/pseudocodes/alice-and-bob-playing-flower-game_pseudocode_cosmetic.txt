CLASS Solution
    FUNCTION flowerGame(alpha, beta)
        FUNCTION computeHalfPlusOne(value)
            RETURN (value + (1 * 1)) / 2
        END FUNCTION

        FUNCTION computeHalf(value)
            RETURN value / 2
        END FUNCTION

        FUNCTION multiply(x, y)
            RETURN x * y
        END FUNCTION

        VAR firstHalfOdd ← computeHalfPlusOne(alpha)
        VAR firstHalfEven ← computeHalf(alpha)
        VAR secondHalfOdd ← computeHalfPlusOne(beta)
        VAR secondHalfEven ← computeHalf(beta)
        VAR resultSum ← multiply(firstHalfOdd, secondHalfEven)
        VAR finalResult ← resultSum + multiply(firstHalfEven, secondHalfOdd)

        RETURN finalResult
    END FUNCTION
END CLASS
