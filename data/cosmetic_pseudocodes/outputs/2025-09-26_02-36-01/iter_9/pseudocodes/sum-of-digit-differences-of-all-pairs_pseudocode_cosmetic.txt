CLASS Solution
    FUNCTION sumDigitDifferences(array_input)
        FUNCTION diffDigits(valA, valB)
            SET discrepancy TO 0
            SET pos_x TO 1
            WHILE pos_x <= LENGTH(valA)
                IF NOT (valA[pos_x] EQUALS valB[pos_x]) THEN
                    SET discrepancy TO discrepancy + 1
                END IF
                SET pos_x TO pos_x + 1
            END WHILE
            RETURN discrepancy
        END FUNCTION

        SET aggregate_result TO 0
        SET length_var TO LENGTH(array_input)
        SET outer_idx TO 0
        WHILE outer_idx < length_var
            SET inner_idx TO outer_idx + 1
            REPEAT
                IF inner_idx >= length_var THEN
                    EXIT REPEAT
                END IF
                SET aggregate_result TO aggregate_result + diffDigits(array_input[outer_idx], array_input[inner_idx])
                SET inner_idx TO inner_idx + 1
            UNTIL false
            SET outer_idx TO outer_idx + 1
        END WHILE
        RETURN aggregate_result
    END FUNCTION
END CLASS
