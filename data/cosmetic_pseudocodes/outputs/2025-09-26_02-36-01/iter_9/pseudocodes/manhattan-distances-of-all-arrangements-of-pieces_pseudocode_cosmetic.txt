CLASS Solution
    FUNCTION distanceSum WITH PARAMETERS alpha beta gamma
        DEFINE CONST constantVal := 1000000007
        DEFINE FUNCTION calcPart WITH PARAMETERS p, q
            DEFINE FUNCTION pow3term WITH PARAMETER x
                RETURN ((x * x) * (x - 1)) / 6
            END FUNCTION
            RETURN (p * p) * pow3term(q)
        END FUNCTION
        DEFINE FUNCTION nCr WITH PARAMETERS total, choose
            DEFINE FUNCTION factorial WITH PARAMETER num
                IF num <= 1 THEN RETURN 1 END IF
                RETURN num * factorial(num - 1)
            END FUNCTION
            RETURN factorial(total) / (factorial(choose) * factorial(total - choose))
        END FUNCTION
        DEFINE VARIABLE partialOne := calcPart(beta, alpha)
        DEFINE VARIABLE partialTwo := calcPart(alpha, beta)
        DEFINE VARIABLE combValue := nCr((alpha * beta) - 2, gamma - 2)
        DEFINE VARIABLE resultSum := (partialOne + partialTwo) * combValue
        RETURN resultSum MOD constantVal
    END FUNCTION
END CLASS
