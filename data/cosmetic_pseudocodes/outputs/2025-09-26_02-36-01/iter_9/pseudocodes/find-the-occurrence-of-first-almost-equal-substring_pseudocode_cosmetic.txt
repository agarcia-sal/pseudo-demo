CLASS Solution
    FUNCTION minStartingIndex WITH PARAMETERS s pattern
        FUNCTION is_almost_equal WITH PARAMETERS substr pat
            VAR mismatch_counter := 0
            VAR pos := 0
            WHILE pos < LENGTH(substr) DO
                VAR ch_sub := substr[pos]
                VAR ch_pat := pat[pos]
                IF ch_sub != ch_pat THEN
                    mismatch_counter := mismatch_counter + 1
                    IF mismatch_counter > 1 THEN
                        RETURN False
                    END IF
                END IF
                pos := pos + 1
            END WHILE
            RETURN True
        END FUNCTION

        VAR pat_len := LENGTH(pattern)
        VAR current_idx := 0
        REPEAT
            IF (current_idx + pat_len) > LENGTH(s) THEN
                BREAK
            END IF
            VAR segment := SLICE(s, current_idx, current_idx + pat_len)
            IF is_almost_equal(segment, pattern) THEN
                RETURN current_idx
            END IF
            current_idx := current_idx + 1
        UNTIL False

        RETURN -1
    END FUNCTION
END CLASS
