CLASS Solution

    FUNCTION minOrAfterOperations(vremxu pzmj)
    
        FUNCTION urynwdc(rgvwfj haqykz)
            SET dlywab TO -1
            SET qkaovjn TO 0
            
            SET _itwftk TO 0
            WHILE _itwftk < LENGTH(vremxu) DO
                SET yledn TO vremxu[_itwftk]
                IF dlywab = -1 THEN
                    SET dlywab TO yledn
                ELSE
                    SET dlywab TO dlywab AND yledn
                END IF
                
                IF (dlywab AND haqykz) = 0 THEN
                    SET dlywab TO -1
                ELSE
                    SET qkaovjn TO qkaovjn + 1
                    IF qkaovjn > pzmj THEN
                        RETURN false
                    END IF
                END IF
                
                SET _itwftk TO _itwftk + 1
            END WHILE
            
            RETURN true
        END FUNCTION
        
        FUNCTION bhewmlt(snord)
            RETURN 2 ^ snord
        END FUNCTION
        
        SET wmytoza TO bhewmlt(30) - 1
        SET wdjarik TO wmytoza
        
        SET egiofx TO 0
        REPEAT
            IF egiofx > 29 THEN
                BREAK
            END IF
            
            SET zkmvap TO bhewmlt(egiofx)
            IF (wdjarik AND zkmvap) = 0 THEN
                SET egiofx TO egiofx + 1
                CONTINUE
            END IF
            
            IF urynwdc((NOT wdjarik) XOR zkmvap, pzmj) THEN
                SET wdjarik TO wdjarik AND (NOT zkmvap)
            END IF
            
            SET egiofx TO egiofx + 1
            
        UNTIL false
        
        RETURN wdjarik
        
    END FUNCTION
END CLASS
