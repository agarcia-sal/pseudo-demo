CLASS Solution  
  FUNCTION minimumArrayLength(input_seq)  
    FUNCTION computeMinValue(sequence)  
      SET current_min TO sequence[0]  
      SET index_ptr TO 1  
      WHILE index_ptr < LENGTH(sequence)  
        IF sequence[index_ptr] < current_min THEN  
          SET current_min TO sequence[index_ptr]  
        END IF  
        SET index_ptr TO index_ptr + 1  
      END WHILE  
      RETURN current_min  
    END FUNCTION  
  
    FUNCTION countOccurrences(sequence, target)  
      SET total_count TO 0  
      SET cursor TO 0  
      REPEAT  
        IF sequence[cursor] = target THEN  
          SET total_count TO total_count + 1  
        END IF  
        SET cursor TO cursor + 1  
      UNTIL cursor >= LENGTH(sequence)  
      RETURN total_count  
    END FUNCTION  
  
    SET temp_min TO computeMinValue(input_seq)  
    SET freq_min TO countOccurrences(input_seq, temp_min)  
  
    IF NOT (freq_min <> 1) THEN    
      RETURN 1  
    ELSE  
      SET dividend TO freq_min + 1  
      SET result_val TO dividend / 2  
      RETURN result_val  
    END IF  
  
  END FUNCTION  
END CLASS
