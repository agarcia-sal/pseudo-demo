CLASS Solution
    FUNCTION minAnagramLength(prm_s)
        FUNCTION gatherUniqueElements(seq)
            DECLARE tracker AS empty dictionary
            DECLARE acc AS empty list
            DECLARE idx AS 0
            WHILE idx < LENGTH(seq)
                IF NOT (seq[idx] IN tracker)
                    SET tracker[seq[idx]] TO TRUE
                    CALL acc.APPEND(seq[idx])
                END IF
                SET idx TO idx + 1
            END WHILE
            RETURN acc
        END FUNCTION

        DECLARE distinct_set AS gatherUniqueElements(prm_s)
        DECLARE count AS 0
        DECLARE i AS 0
        REPEAT
            IF i >= LENGTH(distinct_set)
                BREAK
            END IF
            SET count TO count + 1
            SET i TO i + 1
        UNTIL FALSE

        RETURN count
    END FUNCTION
END CLASS
