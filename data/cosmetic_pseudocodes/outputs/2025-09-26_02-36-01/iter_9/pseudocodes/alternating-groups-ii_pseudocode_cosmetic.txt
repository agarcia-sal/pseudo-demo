CLASS Solution
    FUNCTION numberOfAlternatingGroups(colors k)
        IF NOT (k GREATER OR EQUAL TO 3) THEN
            RETURN 0
        END IF

        SET lengthColors TO 0
        WHILE lengthColors LESS THAN LENGTH(colors)
            INCREMENT lengthColors BY 1
        END WHILE

        SET duplicatedColors TO EMPTY ARRAY
        FOR idx FROM 0 TO lengthColors - 1
            APPEND colors[idx] TO duplicatedColors
        END FOR
        FOR idx2 FROM 0 TO k - 1
            APPEND colors[idx2] TO duplicatedColors
        END FOR

        SET validCount TO 0

        DEFINE FUNCTION checkSequence(startPos)
            IF k EQUALS 1 THEN
                RETURN true
            END IF

            SET posCheck TO 1
            DEFINE FUNCTION innerLoop()
                IF posCheck GREATER OR EQUAL TO k THEN
                    RETURN
                END IF
                IF duplicatedColors[startPos + posCheck] EQUALS duplicatedColors[startPos + posCheck - 1] THEN
                    RETURN false
                ELSE
                    SET tempPos TO posCheck + 1
                    SET posCheck TO tempPos
                    RETURN innerLoop()
                END IF
            END FUNCTION

            SET result TO innerLoop()
            IF result IS false THEN
                RETURN false
            ELSE
                RETURN true
            END IF
        END FUNCTION

        SET currentPos TO 0
        WHILE currentPos LESS THAN lengthColors
            IF checkSequence(currentPos) THEN
                SET validCount TO validCount + 1
            END IF
            SET currentPos TO currentPos + 1
        END WHILE

        RETURN validCount
    END FUNCTION
END CLASS
