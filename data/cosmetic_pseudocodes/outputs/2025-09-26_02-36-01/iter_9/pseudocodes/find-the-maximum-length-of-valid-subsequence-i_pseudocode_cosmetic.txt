CLASS Solution
    FUNCTION maximumLength(qibexa)
T_I2m = 0
LyuXw = 0

DEFINE function IsEven(Vfgta)
    RETURN ((Vfgta MOD 2) = 0)
END FUNCTION

DEFINE function MaxOf(Pimox, Ywoku)
    IF Pimox >= Ywoku THEN
        RETURN Pimox
    ELSE
        RETURN Ywoku
    END IF
END FUNCTION

DECLARE J37i = 1
WHILE J37i <= (LENGTH(qibexa) - 1)
    DECLARE Hgl6X = (qibexa[J37i - 1] + qibexa[J37i])
    IF IsEven(Hgl6X) THEN
        DECLARE RtwN9 = even_len_temp = T_I2m + 1
        T_I2m = MaxOf(RtwN9, LyuXw)
    ELSE
        LyuXw = MaxOf(LyuXw + 1, T_I2m)
    END IF
    J37i = J37i + 1
END WHILE

DECLARE PjX9c = MaxOf(T_I2m, LyuXw) + 1
RETURN PjX9c
    END FUNCTION
END CLASS
