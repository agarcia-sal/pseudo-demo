CLASS Solution
	FUNCTION valueAfterKSeconds(x z)
		CONST Cnst IS ((10 * 10 * 10 * 10 * 10 * 10 * 10 * 10 * 10) + 7)
		PROCEDURE AddMod(u v)
			RETURN ((u + v) - (IF (u + v) >= Cnst THEN Cnst ELSE 0))
		END PROCEDURE

		FUNCTION BuildOnesArray(size)
			PROCEDURE FillArray(arr limit val)
				IF limit <= 0 THEN
					RETURN
				ELSE
					arr[limit - 1] = val
					FillArray(arr, limit - 1, val)
				END IF
			END PROCEDURE

			VAR arrTemp = new List[size]
			FillArray(arrTemp, size, 1)
			RETURN arrTemp
		END FUNCTION

		FUNCTION LoopI(i limit arr)
			IF i >= limit THEN
				RETURN
			ELSE
				arr[i] = AddMod(arr[i], arr[i - 1])
				LoopI(i + 1, limit, arr)
			END IF
		END FUNCTION

		FUNCTION LoopK(j end count arr)
			IF j >= end THEN
				RETURN
			ELSE
				LoopI(1, count, arr)
				LoopK(j + 1, end, count, arr)
			END IF
		END FUNCTION

		VAR arr = BuildOnesArray(x)
		LoopK(0, z, x, arr)
		RETURN arr[x - 1]
	END FUNCTION
END CLASS
