CLASS Solution
    FUNCTION largestSquareArea(a, b)
        FUNCTION intersecting_square_area(c, d, e, f)
            FUNCTION _max(x, y)
                IF x GREATER THAN y THEN RETURN x ELSE RETURN y END IF
            END FUNCTION

            FUNCTION _min(x, y)
                IF x LESS THAN y THEN RETURN x ELSE RETURN y END IF
            END FUNCTION

            SET u TO _max(c[0], e[0])
            SET v TO _min(d[0], f[0])
            SET w TO _max(c[1], e[1])
            SET x TO _min(d[1], f[1])

            IF (NOT (u LESS THAN v)) OR (NOT (w LESS THAN x))
                RETURN 0
            END IF

            SET y TO _min(v - u, x - w)
            RETURN y * y
        END FUNCTION

        SET z TO 0
        SET A TO LENGTH(a)

        FUNCTION loop1(m)
            IF m GREATER THAN OR EQUAL TO A THEN RETURN END IF

            FUNCTION loop2(n)
                IF n GREATER THAN OR EQUAL TO A THEN RETURN END IF
                SET p TO intersecting_square_area(a[m], b[m], a[n], b[n])
                IF z LESS THAN p THEN
                    SET z TO p
                END IF
                loop2(n + 1)
            END FUNCTION

            loop2(m + 1)
            loop1(m + 1)
        END FUNCTION

        loop1(0)
        RETURN z
    END FUNCTION
END CLASS
