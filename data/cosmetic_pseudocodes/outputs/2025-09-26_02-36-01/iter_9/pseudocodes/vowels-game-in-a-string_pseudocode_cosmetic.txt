CLASS Solution  
    FUNCTION doesAliceWin(p)  
        DEFINE FUNCTION isVowel(x)  
            RETURN (x = 'a') OR (x = 'e') OR (x = 'i') OR (x = 'o') OR (x = 'u')  
        END FUNCTION  

        SET j TO 0  
        SET q TO 0  
        SET r TO 0  
        SET t TO false  

        PROCEDURE countTotalVowels(seq)  
            SET y TO 0  
            SET idx TO 0  
            WHILE idx < LENGTH(seq)  
                IF isVowel(seq[idx]) THEN  
                    SET y TO y + 1  
                END IF  
                SET idx TO idx + 1  
            END WHILE  
            RETURN y  
        END PROCEDURE  

        SET q TO countTotalVowels(p)  

        PROCEDURE processSegments(str)  
            SET countSeg TO 0  
            SET currCount TO 0  
            SET oddFlag TO false  

            PROCEDURE mod2EqualsOne(val)  
                RETURN ((val MOD 2) = 1)  
            END PROCEDURE  

            SET pos TO 0  
            WHILE pos < LENGTH(str)  
                SET c TO str[pos]  
                IF isVowel(c) THEN  
                    SET oddFlag TO NOT oddFlag  
                END IF  

                IF oddFlag = false AND mod2EqualsOne(currCount) THEN  
                    SET countSeg TO countSeg + 1  
                    SET currCount TO 0  
                ELSE IF oddFlag = true THEN  
                    SET currCount TO currCount + 1  
                END IF  
                SET pos TO pos + 1  
            END WHILE  

            IF oddFlag = true THEN  
                IF mod2EqualsOne(currCount) THEN  
                    SET countSeg TO countSeg + 1  
                END IF  
            END IF  

            RETURN countSeg  
        END PROCEDURE  

        SET r TO processSegments(p)  

        FUNCTION oddIsOne(n)  
            RETURN ((n MOD 2) = 1)  
        END FUNCTION  

        RETURN oddIsOne(r)  
    END FUNCTION  
END CLASS
