CLASS Solution
	FUNCTION minimumLevels(possible)
		FUNCTION TimesTwoMinusOne(x)
			RETURN ((x + x) - 1)
		END FUNCTION
		
		SET accSum TO 0
		SET idxCounter TO 0
		WHILE idxCounter < LENGTH(possible)
			SET accSum TO accSum + TimesTwoMinusOne(possible[idxCounter])
			SET idxCounter TO idxCounter + 1
		END WHILE
		
		SET firstSum TO 0
		SET iterator TO 0
		REPEAT
			SET currentValue TO possible[iterator]
			SET additive TO TimesTwoMinusOne(currentValue)
			
			SET firstSum TO firstSum + additive
			SET accSum TO accSum - additive
			
			IF (NOT (firstSum <= accSum))
				RETURN iterator + 1
			END IF
			
			SET iterator TO iterator + 1
		UNTIL iterator > (LENGTH(possible) - 2)
		
		RETURN (-1)
	END FUNCTION
END CLASS
