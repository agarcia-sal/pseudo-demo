CLASS Solution  
    FUNCTION maximumStrength(nums k)  
        LET length BE LENGTH OF nums  
        INITIALIZE dp AS ARRAY WITH (length + 1) ELEMENTS, EACH BEING AN ARRAY OF SIZE (k + 1) FILLED WITH NEGATIVE_INFINITY  
        ASSIGN dp[0][0] = 0  
        
        FOR index IN RANGE 1 TO length  
            FOR count IN RANGE 1 TO k  
                LET cumulativeSum = 0  
                LET maxVal = dp[index][count]  
                FOR startIndex IN REVERSE RANGE 1 TO index  
                    cumulativeSum = cumulativeSum + nums[startIndex - 1]  
                    IF (count MOD 2) == 1 THEN  
                        LET multiplier = (k - count) + 1  
                    ELSE  
                        LET multiplier = -((k - count) + 1)  
                    END IF  
                    maxVal = MAX(maxVal, dp[startIndex - 1][count - 1] + multiplier * cumulativeSum)  
                END FOR  
                dp[index][count] = MAX(maxVal, dp[index - 1][count])  
            END FOR  
        END FOR  
        
        RETURN dp[length][k]  
    END FUNCTION  
END CLASS
