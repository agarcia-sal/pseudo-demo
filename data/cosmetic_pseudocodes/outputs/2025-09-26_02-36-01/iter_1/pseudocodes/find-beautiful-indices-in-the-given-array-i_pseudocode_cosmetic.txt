CLASS Solution  
    FUNCTION beautifulIndices(s parameter, a parameter, b parameter, k parameter)  
        DEFINE found_a_positions AS empty collection  
        LET s_length BE LENGTH(s)  
        LET a_length BE LENGTH(a)  
        LET b_length BE LENGTH(b)  
        
        FOR position IN RANGE(0, s_length - a_length + 1)  
            IF s[position TO position + a_length - 1] == a THEN  
                INSERT position INTO found_a_positions  
            END IF  
        END FOR  
        
        DEFINE found_b_positions AS empty collection  
        FOR position IN RANGE(0, s_length - b_length + 1)  
            IF s[position TO position + b_length - 1] == b THEN  
                INSERT position INTO found_b_positions  
            END IF  
        END FOR  
        
        DEFINE result_indices AS empty collection  
        FOR pos_a IN found_a_positions  
            FOR pos_b IN found_b_positions  
                IF ABS(pos_a - pos_b) <= k THEN  
                    ADD pos_a TO result_indices  
                    EXIT INNER LOOP  
                END IF  
            END FOR  
        END FOR  
        
        RETURN result_indices  
    END FUNCTION  
END CLASS
