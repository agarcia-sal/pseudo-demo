CLASS Solution  
    FUNCTION numberOfWays WITH PARAMETER n  
        ASSIGN MODULO = 1000000007  
        INITIALIZE dp AS array WITH n + 1 ELEMENTS FILLED WITH 0  
        dp[0] = 1  
        DEFINE coins AS [1, 2, 6]  
        FOR coin IN coins DO  
            FOR index FROM coin TO n DO  
                dp[index] = (dp[index] + dp[index - coin]) % MODULO  
            END FOR  
        END FOR  
        SET totalWays = 0  
        FOR count FROM 0 TO 2 DO  
            IF (count * 4) <= n THEN  
                totalWays = (totalWays + dp[n - (count * 4)]) % MODULO  
            END IF  
        END FOR  
        RETURN totalWays  
    END FUNCTION  
END CLASS
