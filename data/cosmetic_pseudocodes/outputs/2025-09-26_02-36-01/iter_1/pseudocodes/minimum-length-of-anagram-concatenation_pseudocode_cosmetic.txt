CLASS Solution  
    FUNCTION minAnagramLength(s)  
        DECLARE char_map AS empty dictionary  
        FOR EACH character c IN s DO  
            IF c NOT IN char_map THEN  
                char_map[c] = TRUE  
            END IF  
        END FOR  
        DECLARE count AS 0  
        FOR EACH key IN char_map KEYS DO  
            INCREMENT count BY 1  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
