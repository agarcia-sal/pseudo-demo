CLASS Solution  
    FUNCTION beautifulIndices(s parameter a parameter b parameter k)  
        LET positionsOfA BE an empty collection  
        LET maxStartA BE LENGTH OF s MINUS LENGTH OF a  
        FOR pos FROM 0 TO maxStartA DO  
            LET segmentA BE substring OF s FROM pos LENGTH LENGTH OF a  
            IF segmentA IS EQUAL TO a THEN  
                ADD pos TO positionsOfA  
            END IF  
        END FOR  
        
        LET positionsOfB BE an empty collection  
        LET maxStartB BE LENGTH OF s MINUS LENGTH OF b  
        FOR pos FROM 0 TO maxStartB DO  
            LET segmentB BE substring OF s FROM pos LENGTH LENGTH OF b  
            IF segmentB IS EQUAL TO b THEN  
                ADD pos TO positionsOfB  
            END IF  
        END FOR  
        
        LET resultIndices BE a new empty list  
        LET pointerA BE 0  
        LET pointerB BE 0  
        WHILE pointerA < LENGTH OF positionsOfA AND pointerB < LENGTH OF positionsOfB DO  
            LET diff BE ABSOLUTE VALUE OF (positionsOfA[pointerA] - positionsOfB[pointerB])  
            IF diff <= k THEN  
                APPEND positionsOfA[pointerA] TO resultIndices  
                INCREMENT pointerA BY 1  
            ELSE IF positionsOfA[pointerA] < positionsOfB[pointerB] THEN  
                INCREMENT pointerA BY 1  
            ELSE  
                INCREMENT pointerB BY 1  
            END IF  
        END WHILE  
        
        RETURN resultIndices  
    END FUNCTION  
END CLASS
