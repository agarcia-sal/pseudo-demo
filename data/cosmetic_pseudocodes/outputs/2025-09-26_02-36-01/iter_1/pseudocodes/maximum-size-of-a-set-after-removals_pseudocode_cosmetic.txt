CLASS Solution  
    FUNCTION maximumSetSize(nums1 PARAMETER List OF integers nums2 PARAMETER List OF integers) RETURNS integer  
        LET total_count BE LENGTH(nums1)  
        LET half_limit BE total_count / 2  

        LET distinct1 BE EMPTY SET  
        FOR EACH num IN nums1 DO  
            ADD num TO distinct1  
        END FOR  

        LET distinct2 BE EMPTY SET  
        FOR EACH num IN nums2 DO  
            ADD num TO distinct2  
        END FOR  

        LET common_elements BE EMPTY SET  
        FOR EACH element IN distinct1 DO  
            IF element IN distinct2 THEN  
                ADD element TO common_elements  
            END IF  
        END FOR  

        LET exclusive1 BE EMPTY SET  
        FOR EACH element IN distinct1 DO  
            IF element NOT IN common_elements THEN  
                ADD element TO exclusive1  
            END IF  
        END FOR  

        LET exclusive2 BE EMPTY SET  
        FOR EACH element IN distinct2 DO  
            IF element NOT IN common_elements THEN  
                ADD element TO exclusive2  
            END IF  
        END FOR  

        LET pick_from_exclusive1 BE half_limit  
        IF LENGTH(exclusive1) < half_limit THEN  
            SET pick_from_exclusive1 TO LENGTH(exclusive1)  
        END IF  

        LET pick_from_exclusive2 BE half_limit  
        IF LENGTH(exclusive2) < half_limit THEN  
            SET pick_from_exclusive2 TO LENGTH(exclusive2)  
        END IF  

        LET leftover1 BE half_limit - pick_from_exclusive1  
        IF leftover1 < 0 THEN  
            SET leftover1 TO 0  
        END IF  

        LET leftover2 BE half_limit - pick_from_exclusive2  
        IF leftover2 < 0 THEN  
            SET leftover2 TO 0  
        END IF  

        LET pick_from_common BE leftover1 + leftover2  
        IF pick_from_common > LENGTH(common_elements) THEN  
            SET pick_from_common TO LENGTH(common_elements)  
        END IF  

        LET answer BE pick_from_exclusive1 + pick_from_exclusive2 + pick_from_common  
        RETURN answer  
    END FUNCTION  
END CLASS
