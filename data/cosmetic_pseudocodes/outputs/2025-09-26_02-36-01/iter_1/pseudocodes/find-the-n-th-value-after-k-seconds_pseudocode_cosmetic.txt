CLASS Solution  
    FUNCTION valueAfterKSeconds(n k)  
        DEFINE MODULO = 1000000007  
        INITIALIZE a AS array of length n, each element set to 1  
        
        FOR iteration IN RANGE 0 TO k - 1 DO  
            FOR idx IN RANGE 1 TO n - 1 DO  
                a[idx] = (a[idx] + a[idx - 1]) MODULO MODULO  
            END FOR  
        END FOR  
        
        RETURN a[n - 1]  
    END FUNCTION  
END CLASS
