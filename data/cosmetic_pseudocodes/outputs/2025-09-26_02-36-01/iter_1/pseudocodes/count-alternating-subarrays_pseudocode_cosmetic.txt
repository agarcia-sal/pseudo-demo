CLASS Solution  
    FUNCTION countAlternatingSubarrays(nums)  
        LET total_subarrays = 0  
        LET length_nums = SIZE(nums)  
        IF length_nums == 0 THEN  
            RETURN 0  
        END IF  
        SET total_subarrays = length_nums  
        SET streak_length = 1  
        FOR index FROM 2 TO length_nums  
            IF nums[index] != nums[index - 1] THEN  
                streak_length = streak_length + 1  
            ELSE  
                streak_length = 1  
            END IF  
            total_subarrays = total_subarrays + (streak_length - 1)  
        END FOR  
        RETURN total_subarrays  
    END FUNCTION  
END CLASS
