CLASS Solution  
    FUNCTION countSubarrays(nums parameter LIST OF INTEGER k parameter INTEGER) RETURNS INTEGER  
        LET total_subarrays = 0  
        LET length = LENGTH(nums)  
        FOR i IN 0 TO length - 1 DO  
            LET accum_and = nums[i]  
            FOR j IN i TO length - 1 DO  
                accum_and = accum_and BITWISE_AND nums[j]  
                IF accum_and == k THEN  
                    total_subarrays = total_subarrays + 1  
                ELSE IF accum_and < k THEN  
                    EXIT INNER LOOP  
                END IF  
            END FOR  
        END FOR  
        RETURN total_subarrays  
    END FUNCTION  
END CLASS
