CLASS Solution  
    FUNCTION minOperations(nums parameter k parameter)  
        INITIALIZE operations TO 0  
        EXECUTE heapify ON nums  
        WHILE (nums[0] < k) AND (length of nums > 1) DO  
            SET first_element TO heappop(nums)  
            SET second_element TO heappop(nums)  
            SET combined TO (2 * minimum of first_element, second_element) + maximum of first_element, second_element  
            heappush(nums, combined)  
            operations = operations + 1  
        END WHILE  
        RETURN operations  
    END FUNCTION  
END CLASS
