CLASS Solution  
    FUNCTION minimumOperations(grid)  
        LET rows BE LENGTH OF grid  
        LET cols BE LENGTH OF grid[0]  
        LET totalOps BE 0  
        
        FOR col FROM 0 TO cols - 1  
            FOR row FROM 0 TO rows - 2  
                IF grid[row][col] != grid[row + 1][col] THEN  
                    totalOps <- totalOps + 1  
                    grid[row + 1][col] <- grid[row][col]  
                END IF  
            END FOR  
            
            IF col < cols - 1 THEN  
                FOR row FROM 0 TO rows - 1  
                    IF grid[row][col] == grid[row][col + 1] THEN  
                        totalOps <- totalOps + 1  
                        FOR candidateVal FROM 0 TO 9  
                            IF candidateVal != grid[row][col] THEN  
                                grid[row][col + 1] <- candidateVal  
                                EXIT FOR  
                            END IF  
                        END FOR  
                    END IF  
                END FOR  
            END IF  
        END FOR  
        
        RETURN totalOps  
    END FUNCTION  
END CLASS
