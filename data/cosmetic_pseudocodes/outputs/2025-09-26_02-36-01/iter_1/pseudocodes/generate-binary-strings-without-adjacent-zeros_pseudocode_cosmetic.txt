CLASS Solution  
    FUNCTION validStrings n  
        DECLARE result AS empty list  

        FUNCTION exploreSequence seq  
            IF LENGTH(seq) IS n THEN  
                ADD seq TO result  
                RETURN  
            END IF  

            LET lastChar BE CHARACTER AT POSITION LENGTH(seq) - 1 IN seq  
            SWITCH lastChar  
                CASE "1":  
                    CALL exploreSequence WITH seq + "0"  
                    CALL exploreSequence WITH seq + "1"  
                CASE "0":  
                    CALL exploreSequence WITH seq + "1"  
            END SWITCH  
        END FUNCTION  

        IF n IS 1 THEN  
            RETURN ["0", "1"]  
        END IF  

        CALL exploreSequence("0")  
        CALL exploreSequence("1")  

        RETURN result  
    END FUNCTION  
END CLASS
