CLASS Solution  
    FUNCTION maxTotalReward(rewardValues)  
        LET uniqueValues BE A SORTED LIST OF DISTINCT ELEMENTS FROM rewardValues  
        LET flags BE 1  
        FOR value IN uniqueValues DO  
            LET mask BE ((1 << value) - 1) << value  
            flags = flags OR (flags AND mask)  
        END FOR  
        RETURN BIT_LENGTH(flags) - 1  
    END FUNCTION  
END CLASS
