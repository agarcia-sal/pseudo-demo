CLASS Solution  
    FUNCTION minStartingIndex(s, pattern)  
        FUNCTION is_almost_equal(substring, pattern)  
            SET differences TO 0  
            FOR position FROM 0 TO LENGTH(pattern) - 1  
                IF substring[position] ≠ pattern[position] THEN  
                    differences ← differences + 1  
                    IF differences > 1 THEN  
                        RETURN False  
                    END IF  
                END IF  
            END FOR  
            RETURN True  
        END FUNCTION  

        SET length_pattern TO LENGTH(pattern)  
        FOR start_pos IN 0 TO (LENGTH(s) - length_pattern)  
            SET candidate_substring TO s[start_pos .. start_pos + length_pattern - 1]  
            IF is_almost_equal(candidate_substring, pattern) THEN  
                RETURN start_pos  
            END IF  
        END FOR  
        RETURN -1  
    END FUNCTION  
END CLASS
