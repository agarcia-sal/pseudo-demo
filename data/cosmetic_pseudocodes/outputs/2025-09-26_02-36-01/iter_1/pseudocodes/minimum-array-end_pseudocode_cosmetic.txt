CLASS Solution  
    FUNCTION minEnd(n x)  
        FUNCTION canConstruct(max_val)  
            VARIABLE temp = x  
            VARIABLE found = 1  
            LOOP WHILE temp < max_val  
                temp = temp + 1  
                IF (temp BITWISE_AND x) IS x THEN  
                    found = found + 1  
                    IF found == n THEN  
                        RETURN True  
                    END IF  
                END IF  
            END LOOP  
            RETURN found == n  
        END FUNCTION  
        
        VARIABLE low = x  
        VARIABLE high = 2 * 10 * 10 * 10 * 10 * 10 * 10 * 10 * 10  
        LOOP WHILE low < high  
            VARIABLE middle = (low + high) // 2  
            IF canConstruct(middle) THEN  
                high = middle  
            ELSE  
                low = low + 1  
            END IF  
        END LOOP  
        
        RETURN low  
    END FUNCTION  
END CLASS
