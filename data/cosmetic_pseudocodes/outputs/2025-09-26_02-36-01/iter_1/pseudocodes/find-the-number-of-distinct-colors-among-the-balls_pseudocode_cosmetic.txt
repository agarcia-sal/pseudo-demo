CLASS Solution  
    FUNCTION queryResults(limit parameter, queries parameter)  
        DECLARE ball_colors AS dictionary (keyed by identifiers, values are colors) INITIALIZED empty  
        DECLARE unique_colors AS set INITIALIZED empty  
        DECLARE result AS empty list  

        FOR EACH query IN queries parameter DO  
            LET x parameter = query[0]  
            LET y parameter = query[1]  

            IF ball_colors CONTAINS KEY x parameter THEN  
                LET old_color = ball_colors[x parameter]  
                IF unique_colors CONTAINS old_color THEN  
                    unique_colors REMOVE old_color  
                END IF  
            END IF  

            ball_colors[x parameter] = y parameter  
            unique_colors ADD y parameter  
            result APPEND SIZE(unique_colors)  
        END FOR  

        RETURN result  
    END FUNCTION  
END CLASS
