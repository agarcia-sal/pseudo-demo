CLASS Solution  
    FUNCTION maxNumber(n)  
        IF n IS 1 THEN  
            RETURN 0  
        END IF  
        
        DECLARE highest_bit = 1  
        
        REPEAT  
            highest_bit = highest_bit * 2  
        UNTIL highest_bit > n  
        
        highest_bit = highest_bit / 2  
        
        DECLARE x = highest_bit - 1  
        
        RETURN x  
    END FUNCTION  
END CLASS
