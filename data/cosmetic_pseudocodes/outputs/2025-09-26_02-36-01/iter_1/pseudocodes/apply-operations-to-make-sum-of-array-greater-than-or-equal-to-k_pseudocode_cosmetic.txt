CLASS Solution  
    FUNCTION minOperations(k)  
        INITIALIZE min_operations WITH a very large number  
        FOR candidate_x IN RANGE 1 TO FLOOR(SQRT(k)) + 1 DO  
            CALCULATE candidate_y AS (k + candidate_x - 1) DIVIDED BY candidate_x USING INTEGER DIVISION  
            CALCULATE current_operations AS (candidate_x - 1) + (candidate_y - 1)  
            IF current_operations < min_operations THEN  
                ASSIGN min_operations TO current_operations  
            END IF  
        END FOR  
        RETURN min_operations  
    END FUNCTION  
END CLASS
