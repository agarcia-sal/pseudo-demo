CLASS Solution
    FUNCTION waysToReachStair(k)
        FUNCTION dfs(i, j, jump)
            IF i > k + 1 THEN
                RETURN 0
            END IF

            ans = TO_INT(i == k)

            IF i > 0 AND j == 0 THEN
                ans = ans + dfs(i - 1, 1, jump)
            END IF

            ans = ans + dfs(i * 2 * jump, 0, jump + 1)

            RETURN ans
        END FUNCTION

        RETURN dfs(1, 0, 0)
    END FUNCTION
END CLASS
