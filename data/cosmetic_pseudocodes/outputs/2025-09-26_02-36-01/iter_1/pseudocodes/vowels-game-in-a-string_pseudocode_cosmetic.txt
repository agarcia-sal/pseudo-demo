CLASS Solution
    FUNCTION doesAliceWin(s)
        DEFINE vowelSet AS {"a", "e", "i", "o", "u"}
        DECLARE oddSegmentCount = 0
        DECLARE vowelCount = 0
        
        FOR char IN s
            IF char IN vowelSet THEN
                vowelCount = vowelCount + 1
            END IF
        END FOR
        
        DECLARE toggleSegment = false
        
        FOR char IN s
            IF char IN vowelSet THEN
                toggleSegment = NOT toggleSegment
            END IF
            
            IF toggleSegment THEN
                vowelCount = vowelCount + 1
            ELSE IF NOT toggleSegment AND (vowelCount MOD 2) = 1 THEN
                oddSegmentCount = oddSegmentCount + 1
                vowelCount = 0
            END IF
        END FOR
        
        IF toggleSegment AND (vowelCount MOD 2) = 1 THEN
            oddSegmentCount = oddSegmentCount + 1
        END IF
        
        RETURN (oddSegmentCount MOD 2) = 1
    END FUNCTION
END CLASS
