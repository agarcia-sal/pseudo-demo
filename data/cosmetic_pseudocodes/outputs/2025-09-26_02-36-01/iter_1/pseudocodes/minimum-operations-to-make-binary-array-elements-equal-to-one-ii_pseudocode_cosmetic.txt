CLASS Solution  
    FUNCTION minOperations(nums)  
        INITIALIZE operations AS 0  
        INITIALIZE flip AS 0  
        FOR INDEX FROM 0 TO LENGTH(nums) - 1  
            IF (flip MOD 2) == 0 THEN  
                LET current_num BE nums[INDEX]  
            ELSE  
                LET current_num BE (1 - nums[INDEX])  
            END IF  
            IF current_num == 0 THEN  
                operations = operations + 1  
                flip = flip + 1  
            END IF  
        END FOR  
        RETURN operations  
    END FUNCTION  
END CLASS
