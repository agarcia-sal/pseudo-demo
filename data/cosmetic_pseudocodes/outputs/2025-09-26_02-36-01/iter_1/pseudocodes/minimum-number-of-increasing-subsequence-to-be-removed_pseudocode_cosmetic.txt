CLASS Solution  
    FUNCTION minOperations(nums)  
        LET length BE LENGTH(nums)  
        IF length IS 0 THEN  
            RETURN 0  
        END IF  
        CREATE array dp WITH length elements INITIALIZED TO 1  
        FOR current FROM 1 TO length - 1 DO  
            FOR previous FROM 0 TO current - 1 DO  
                IF nums[current] <= nums[previous] THEN  
                    dp[current] = MAX(dp[current], dp[previous] + 1)  
                END IF  
            END FOR  
        END FOR  
        LET result BE dp[0]  
        FOR val IN dp DO  
            IF val > result THEN  
                result = val  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
