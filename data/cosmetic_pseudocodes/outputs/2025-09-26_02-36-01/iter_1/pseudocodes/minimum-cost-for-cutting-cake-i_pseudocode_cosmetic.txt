CLASS Solution  
    FUNCTION minimumCost(m n horizontalCut verticalCut)  
        REVERSE SORT horizontalCut  
        REVERSE SORT verticalCut  
        INITIALIZE totalCost TO 0  
        INITIALIZE idxH TO 0  
        INITIALIZE idxV TO 0  
        INITIALIZE countH TO 1  
        INITIALIZE countV TO 1  
        LOOP UNTIL idxH == m - 1 AND idxV == n - 1  
            IF idxV == n - 1 OR (idxH < m - 1 AND horizontalCut[idxH] > verticalCut[idxV]) THEN  
                totalCost = totalCost + horizontalCut[idxH] * countV  
                countH = countH + 1  
                idxH = idxH + 1  
            ELSE  
                totalCost = totalCost + verticalCut[idxV] * countH  
                countV = countV + 1  
                idxV = idxV + 1  
            END IF  
        END LOOP  
        RETURN totalCost  
    END FUNCTION  
END CLASS
