CLASS Solution  
    FUNCTION flowerGame(n, m)  
        LET count_odd_in_n = (n + 1) // 2  
        LET count_even_in_n = n // 2  
        LET count_odd_in_m = (m + 1) // 2  
        LET count_even_in_m = m // 2  
        LET pairs = (count_odd_in_n * count_even_in_m) + (count_even_in_n * count_odd_in_m)  
        RETURN pairs  
    END FUNCTION  
END CLASS
