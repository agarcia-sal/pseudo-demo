CLASS Solution  
    FUNCTION minOperationsToMakeMedianK(nums k)  
        LET sortedNums BE nums SORTED IN ascending order  
        LET length BE SIZE OF sortedNums  
        LET mid BE length DIV 2  

        IF sortedNums[mid] IS k THEN  
            RETURN 0  
        END IF  

        LET totalOps BE 0  
        
        IF sortedNums[mid] < k THEN  
            WHILE mid < length AND sortedNums[mid] < k DO  
                totalOps = totalOps + (k - sortedNums[mid])  
                mid = mid + 1  
            END WHILE  
        ELSE  
            WHILE mid >= 0 AND sortedNums[mid] > k DO  
                totalOps = totalOps + (sortedNums[mid] - k)  
                mid = mid - 1  
            END WHILE  
        END IF  

        RETURN totalOps  
    END FUNCTION  
END CLASS
