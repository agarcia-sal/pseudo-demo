CLASS Solution
    FUNCTION minimumOperations(nums, target)
        LET length BE LENGTH(nums)
        LET totalOperations BE ABS(target[0] - nums[0])

        FOR index FROM 1 TO length - 1
            LET diffCurrent BE target[index] - nums[index]
            LET diffPrevious BE target[index - 1] - nums[index - 1]

            IF diffCurrent * diffPrevious > 0 THEN
                LET differenceAbs BE ABS(ABS(diffCurrent) - ABS(diffPrevious))
                IF differenceAbs > 0 THEN
                    totalOperations = totalOperations + differenceAbs
                END IF
            ELSE
                totalOperations = totalOperations + ABS(diffCurrent)
            END IF
        END FOR

        RETURN totalOperations
    END FUNCTION
END CLASS
