CLASS Solution  
    FUNCTION minimumTimeToInitialState(word PARAMETER k)  
        LET length BE LENGTH(word)  
        LET counter BE 1  
        LOOP  
            LET index BE counter * k  
            IF index >= length OR word[index TO length] == word[0 TO length - index] THEN  
                RETURN counter  
            END IF  
            counter = counter + 1  
        END LOOP  
    END FUNCTION  
END CLASS
