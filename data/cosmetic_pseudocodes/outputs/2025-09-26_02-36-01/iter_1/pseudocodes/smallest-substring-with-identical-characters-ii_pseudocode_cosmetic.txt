CLASS Solution  
    FUNCTION minLength(s parameter representing a string, numOps parameter representing an integer) RETURNS integer  
        FUNCTION check(m parameter representing an integer) RETURNS boolean  
            SET operationsUsed TO 0  
            SET currentSeqLength TO 0  
            FOR i FROM 0 TO LENGTH OF s MINUS 1  
                INCREMENT currentSeqLength BY 1  
                IF i IS LAST INDEX OR s[i] IS NOT EQUAL TO s[i + 1]  
                    SET operationsUsed TO operationsUsed + INTEGER_DIVISION(currentSeqLength, m + 1)  
                    IF operationsUsed > numOps  
                        RETURN false  
                    END IF  
                    SET currentSeqLength TO 0  
                END IF  
            END FOR  
            RETURN operationsUsed <= numOps  
        END FUNCTION  

        SET n TO LENGTH OF s  
        SET low TO 1  
        SET high TO n  
        WHILE low < high  
            SET middle TO FLOOR((low + high) / 2)  
            IF check(middle) == true  
                SET high TO middle  
            ELSE  
                SET low TO middle + 1  
            END IF  
        END WHILE  
        RETURN low  
    END FUNCTION  
END CLASS
