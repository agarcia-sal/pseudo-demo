CLASS Solution  
    FUNCTION maxOperations(s)  
        DECLARE ans = 0  
        DECLARE cnt = 0  
        FOR i FROM 0 TO LENGTH(s) - 1  
            LET c = s[i]  
            IF c == '1' THEN  
                cnt = cnt + 1  
            ELSE IF i > 0 AND s[i - 1] == '1' THEN  
                ans = ans + cnt  
            END IF  
        END FOR  
        RETURN ans  
    END FUNCTION  
END CLASS
