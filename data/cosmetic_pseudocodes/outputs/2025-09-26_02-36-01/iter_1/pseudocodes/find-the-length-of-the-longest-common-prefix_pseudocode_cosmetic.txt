CLASS Solution  
    FUNCTION longestCommonPrefix(arr1 arr1 OF integers arr2 arr2 OF integers) RETURNS integer  
        DECLARE prefixesFirst AS a new empty set  
        FOR index FROM 0 TO LENGTH(arr1) - 1  
            LET currentNumStr BE STRINGIFY(arr1[index])  
            LET lengthNumStr BE LENGTH(currentNumStr)  
            FOR subLen FROM 1 TO lengthNumStr  
                INSERT SUBSTRING(currentNumStr, 1, subLen) INTO prefixesFirst  
            END FOR  
        END FOR  
  
        DECLARE prefixesSecond AS a new empty set  
        FOR index FROM 0 TO LENGTH(arr2) - 1  
            LET currentNumStr BE STRINGIFY(arr2[index])  
            LET lengthNumStr BE LENGTH(currentNumStr)  
            FOR subLen FROM 1 TO lengthNumStr  
                INSERT SUBSTRING(currentNumStr, 1, subLen) INTO prefixesSecond  
            END FOR  
        END FOR  
  
        LET longestPrefixSize BE 0  
        FOR EACH prefixElement IN prefixesFirst  
            IF prefixesSecond CONTAINS prefixElement THEN  
                LET prefixLen BE LENGTH(prefixElement)  
                IF prefixLen > longestPrefixSize THEN  
                    longestPrefixSize = prefixLen  
                END IF  
            END IF  
        END FOR  
  
        RETURN longestPrefixSize  
    END FUNCTION  
END CLASS
