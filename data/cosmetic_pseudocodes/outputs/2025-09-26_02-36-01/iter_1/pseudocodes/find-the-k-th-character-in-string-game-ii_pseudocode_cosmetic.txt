CLASS Solution  
    FUNCTION kthCharacter(k param_k INTEGER, operations param_operations LIST OF INTEGER) RETURNS STRING  
        INITIALIZE current_length AS 1  
        INITIALIZE operation_list AS new empty list  

        FOR operation_element IN param_operations DO  
            ADD operation_element TO operation_list  
            current_length = current_length * 2  
        END FOR  

        SET result_char TO 'a'  

        FOR index FROM (SIZE OF operation_list) - 1 DOWNTO 0 DO  
            HALF_LENGTH = current_length / 2  
            IF param_k > HALF_LENGTH THEN  
                param_k = param_k - HALF_LENGTH  
                IF operation_list[index] == 1 THEN  
                    IF result_char == 'z' THEN  
                        result_char = 'a'  
                    ELSE  
                        result_char = CHARACTER_ASCII_VALUE(result_char) + 1 CONVERTED TO CHAR  
                    END IF  
                END IF  
            END IF  
            current_length = HALF_LENGTH  
        END FOR  

        RETURN result_char  
    END FUNCTION  
END CLASS
