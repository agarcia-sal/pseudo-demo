CLASS Solution  
    FUNCTION occurrencesOfElement(nums, queries, x)  
        DECLARE positions AS new list  
        DECLARE results AS new list  
        FOR i FROM 0 TO LENGTH(nums) - 1  
            IF nums[i] == x THEN  
                CALL positions.ADD(i)  
            END IF  
        END FOR  
        FOR EACH q IN queries DO  
            IF q <= LENGTH(positions) THEN  
                CALL results.ADD(positions[q - 1])  
            ELSE  
                CALL results.ADD(-1)  
            END IF  
        END FOR  
        RETURN results  
    END FUNCTION  
END CLASS
