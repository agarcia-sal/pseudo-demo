CLASS Solution  
    FUNCTION numberOfSubstrings(s)  
        LET length = LENGTH(s)  
        LET total = 0  
        FOR i IN RANGE(length)  
            LET cntOne = 0  
            LET cntZero = 0  
            FOR j IN RANGE(i, length)  
                IF s[j] == '1' THEN  
                    cntOne = cntOne + 1  
                ELSE  
                    cntZero = cntZero + 1  
                END IF  
                IF cntOne >= (cntZero * cntZero) THEN  
                    total = total + 1  
                END IF  
            END FOR  
        END FOR  
        RETURN total  
    END FUNCTION  
END CLASS
