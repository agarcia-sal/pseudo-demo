CLASS Solution  
    FUNCTION largestSquareArea(bottomLeft, topRight)  
        FUNCTION intersecting_square_area(bl1, tr1, bl2, tr2)  
            SET x_start TO MAX(bl1[0], bl2[0])  
            SET x_end TO MIN(tr1[0], tr2[0])  
            SET y_start TO MAX(bl1[1], bl2[1])  
            SET y_end TO MIN(tr1[1], tr2[1])  

            IF x_start >= x_end OR y_start >= y_end THEN  
                RETURN 0  
            END IF  

            SET side_length TO MIN(x_end - x_start, y_end - y_start)  
            RETURN side_length * side_length  
        END FUNCTION  

        SET count TO LENGTH(bottomLeft)  
        SET maximum_area TO 0  

        REPEAT WITH first FROM 0 TO count - 2  
            REPEAT WITH second FROM first + 1 TO count - 1  
                SET area TO intersecting_square_area(bottomLeft[first], topRight[first], bottomLeft[second], topRight[second])  
                IF area > maximum_area THEN  
                    SET maximum_area TO area  
                END IF  
            END REPEAT  
        END REPEAT  

        RETURN maximum_area  
    END FUNCTION  
END CLASS
