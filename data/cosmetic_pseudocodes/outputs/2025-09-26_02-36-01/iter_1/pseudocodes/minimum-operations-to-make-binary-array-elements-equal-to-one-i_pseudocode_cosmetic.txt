CLASS Solution  
    FUNCTION minOperations(nums)  
        LET length BE LENGTH OF nums  
        LET countOperations BE 0  
        LET index BE 0  
        WHILE index <= length - 3  
            IF nums[index] == 0 THEN  
                nums[index] = 1 - nums[index]  
                nums[index + 1] = 1 - nums[index + 1]  
                nums[index + 2] = 1 - nums[index + 2]  
                countOperations = countOperations + 1  
            END IF  
            index = index + 1  
        END WHILE  
        IF nums[length - 1] == 0 OR nums[length - 2] == 0 THEN  
            RETURN -1  
        END IF  
        RETURN countOperations  
    END FUNCTION  
END CLASS
