CLASS Solution  
    FUNCTION maximumLength(nums k)  
        LET length BE LENGTH OF nums  
        IF length IS 1 THEN  
            RETURN 1  
        END IF  
  
        INITIALIZE dp AS ARRAY OF length ELEMENTS, EACH BEING AN EMPTY DICTIONARY  
  
        max_len := 1  
  
        FOR current FROM 0 TO length - 1 DO  
            FOR previous FROM 0 TO current - 1 DO  
                sum_mod := (nums[current] + nums[previous]) MOD k  
  
                IF sum_mod EXISTS IN dp[previous] THEN  
                    dp[current][sum_mod] := dp[previous][sum_mod] + 1  
                ELSE  
                    dp[current][sum_mod] := 2  
                END IF  
  
                IF dp[current][sum_mod] > max_len THEN  
                    max_len := dp[current][sum_mod]  
                END IF  
            END FOR  
        END FOR  
  
        RETURN max_len  
    END FUNCTION  
END CLASS
