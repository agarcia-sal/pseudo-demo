CLASS Solution  
    FUNCTION minMovesToCaptureTheQueen(a b c d e f)  
        IF a == e THEN  
            IF c != a THEN  
                RETURN 1  
            ELSE IF ( (b < d AND d < f) OR (f < d AND d < b) ) THEN  
                RETURN 2  
            ELSE  
                RETURN 1  
            END IF  
        ELSE IF b == f THEN  
            IF d != b THEN  
                RETURN 1  
            ELSE IF ( (a < c AND c < e) OR (e < c AND c < a) ) THEN  
                RETURN 2  
            ELSE  
                RETURN 1  
            END IF  
        ELSE IF (c + d) == (e + f) THEN  
            IF (a + b) != (c + d) THEN  
                RETURN 1  
            ELSE IF ( (c < a AND a < e AND d < b AND b < f) OR (e < a AND a < c AND f < b AND b < d) ) THEN  
                RETURN 2  
            ELSE  
                RETURN 1  
            END IF  
        ELSE IF (c - d) == (e - f) THEN  
            IF (a - b) != (c - d) THEN  
                RETURN 1  
            ELSE IF ( (c < a AND a < e AND d > b AND b > f) OR (e < a AND a < c AND f > b AND b > d) ) THEN  
                RETURN 2  
            ELSE  
                RETURN 1  
            END IF  
        ELSE  
            RETURN 2  
        END IF  
    END FUNCTION  
END CLASS
