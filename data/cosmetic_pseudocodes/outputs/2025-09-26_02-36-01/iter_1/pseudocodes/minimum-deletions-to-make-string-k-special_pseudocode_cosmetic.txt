CLASS Solution  
    FUNCTION minimumDeletions(word, k)  
        DECLARE freqDict AS empty dictionary  
        FOR character IN word DO  
            IF character IN freqDict THEN  
                freqDict[character] = freqDict[character] + 1  
            ELSE  
                freqDict[character] = 1  
            END IF  
        END FOR  
        
        DECLARE freqList AS list of values in freqDict  
        SORT freqList IN ascending order  
        
        DECLARE bestDeletionCount AS infinity  
        
        FOR i FROM 0 TO LENGTH(freqList) - 1 DO  
            DECLARE capFrequency AS freqList[i] + k  
            DECLARE currentDeletionCount AS 0  
            
            FOR j FROM i + 1 TO LENGTH(freqList) - 1 DO  
                IF freqList[j] > capFrequency THEN  
                    currentDeletionCount = currentDeletionCount + (freqList[j] - capFrequency)  
                END IF  
            END FOR  
            
            FOR j FROM 0 TO i - 1 DO  
                currentDeletionCount = currentDeletionCount + freqList[j]  
            END FOR  
            
            IF currentDeletionCount < bestDeletionCount THEN  
                bestDeletionCount = currentDeletionCount  
            END IF  
        END FOR  
        
        RETURN bestDeletionCount  
    END FUNCTION  
END CLASS
