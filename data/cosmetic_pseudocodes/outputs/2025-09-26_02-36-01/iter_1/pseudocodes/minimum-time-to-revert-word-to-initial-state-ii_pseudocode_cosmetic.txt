CLASS Solution  
    FUNCTION minimumTimeToInitialState(word STRING, k INTEGER) RETURNS INTEGER  
        LET length = LENGTH(word)  
        LET current_time = 1  
        REPEAT  
            LET segment = SUBSTRING(word, current_time * k, length - current_time * k)  
            IF word BEGINS_WITH segment THEN  
                RETURN current_time  
            END IF  
            current_time = current_time + 1  
        UNTIL FALSE  
    END FUNCTION  
END CLASS
