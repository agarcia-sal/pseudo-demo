CLASS Solution  
    FUNCTION minimumCost(m n horizontalCut verticalCut)  
        SORT horizontalCut IN DESCENDING ORDER  
        SORT verticalCut IN DESCENDING ORDER  
        SET totalCost TO 0  
        SET idxH TO 0  
        SET idxV TO 0  
        SET segmentsH TO 1  
        SET segmentsV TO 1  
        LOOP WHILE idxH < LENGTH(horizontalCut) OR idxV < LENGTH(verticalCut)  
            IF idxV == LENGTH(verticalCut) OR (idxH < LENGTH(horizontalCut) AND horizontalCut[idxH] > verticalCut[idxV]) THEN  
                totalCost = totalCost + horizontalCut[idxH] * segmentsV  
                segmentsH = segmentsH + 1  
                idxH = idxH + 1  
            ELSE  
                totalCost = totalCost + verticalCut[idxV] * segmentsH  
                segmentsV = segmentsV + 1  
                idxV = idxV + 1  
            END IF  
        END LOOP  
        RETURN totalCost  
    END FUNCTION  
END CLASS
