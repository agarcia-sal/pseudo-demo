CLASS Enemy  
    FUNCTION __init__(self, damage, timeTakenDown)  
        self.damage = damage  
        self.timeTakenDown = timeTakenDown  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION minDamage(self, power, damage, health)  
        totalDamage = 0  
        damageSum = 0  
        enemyList = []  

        FOR i FROM 0 TO LENGTH(damage) - 1  
            damageSum = damageSum + damage[i]  
        END FOR  

        FOR i FROM 0 TO LENGTH(damage) - 1  
            hitsNeeded = (health[i] + power - 1) // power  
            enemyInstance = CREATE Enemy(damage[i], hitsNeeded)  
            APPEND enemyInstance TO enemyList  
        END FOR  

        SORT enemyList BY (enemy.damage / enemy.timeTakenDown) IN DESCENDING ORDER  

        FOR enemy IN enemyList  
            totalDamage = totalDamage + (damageSum * enemy.timeTakenDown)  
            damageSum = damageSum - enemy.damage  
        END FOR  

        RETURN totalDamage  
    END FUNCTION  
END CLASS
