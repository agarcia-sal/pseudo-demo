CLASS Solution  
    FUNCTION minRectanglesToCoverPoints(points, w)  
        LET rectangles COUNT BE 0  
        LET latest_covered_x BE -1  
        SORT points BY their x-coordinate IN ascending order  
        FOR EACH coordinate_pair IN points  
            LET x = coordinate_pair[0]  
            IF x > latest_covered_x THEN  
                latest_covered_x = x + w  
                rectangles COUNT = rectangles COUNT + 1  
            END IF  
        END FOR  
        RETURN rectangles COUNT  
    END FUNCTION  
END CLASS
