CLASS Solution  
    FUNCTION numberOfRightTriangles(grid)  
        LET total_triangles BE 0  
        LET rows BE LENGTH(grid)  
        LET cols BE LENGTH(grid[0])  

        FOR row_index IN RANGE(rows)  
            FOR col_index IN RANGE(cols)  
                IF grid[row_index][col_index] == 1 THEN  
                    LET row_sum BE 0  
                    LET col_sum BE 0  

                    FOR c IN RANGE(cols)  
                        IF c != col_index THEN  
                            row_sum = row_sum + grid[row_index][c]  
                        END IF  
                    END FOR  

                    FOR r IN RANGE(rows)  
                        IF r != row_index THEN  
                            col_sum = col_sum + grid[r][col_index]  
                        END IF  
                    END FOR  

                    total_triangles = total_triangles + (row_sum * col_sum)  
                END IF  
            END FOR  
        END FOR  

        RETURN total_triangles  
    END FUNCTION  
END CLASS
