CLASS Solution  
    FUNCTION numberOfWays(n integer parameter, x integer parameter, y integer parameter) RETURNS integer  
        DEFINE MODULO AS 1000000007  
        INITIALIZE DP AS a two-dimensional array with dimensions (x + 1) by (n + 1), filled with zeros  
        ASSIGN DP[0][0] TO 1  
        FOR index FROM 1 TO x DO  
            FOR total FROM 1 TO n DO  
                COMPUTE val1 AS (DP[index][total - 1] * index) MOD MODULO  
                COMPUTE val2 AS (DP[index - 1][total - 1] * (x - index + 1)) MOD MODULO  
                ASSIGN DP[index][total] TO (val1 + val2) MOD MODULO  
            END FOR  
        END FOR  
        SET result TO 0  
        SET powerY TO 1  
        FOR count FROM 1 TO x DO  
            UPDATE powerY TO (powerY * y) MOD MODULO  
            INCREMENT result BY (DP[count][n] * powerY) MOD MODULO  
            APPLY MODULO TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
