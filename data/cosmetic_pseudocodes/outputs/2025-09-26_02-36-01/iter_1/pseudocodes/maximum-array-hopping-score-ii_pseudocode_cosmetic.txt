CLASS Solution  
    FUNCTION maxScore(nums)  
        LET length BE LENGTH OF nums  
        INITIALIZE dp AS ARRAY OF ZEROES WITH SIZE length  
        FOR index FROM length - 2 DOWNTO 0  
            LET highest_score BE 0  
            FOR next_index FROM index + 1 TO length - 1  
                LET current_score BE (next_index - index) * nums[next_index]  
                LET total_score BE current_score + dp[next_index]  
                IF total_score > highest_score THEN  
                    highest_score = total_score  
                END IF  
            END FOR  
            dp[index] = highest_score  
        END FOR  
        RETURN dp[0]  
    END FUNCTION  
END CLASS
