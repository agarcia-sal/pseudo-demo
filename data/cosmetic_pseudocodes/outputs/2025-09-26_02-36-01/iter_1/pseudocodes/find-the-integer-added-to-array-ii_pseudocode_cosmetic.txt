CLASS Solution  
    FUNCTION minimumAddedInteger(nums1 RETURNS List OF integer, nums2 RETURNS List OF integer) RETURNS integer  
        CALL sort ON nums1  
        CALL sort ON nums2  
        
        FOR start_index FROM 0 TO LENGTH(nums1) - 2  
            FOR end_index FROM start_index + 1 TO LENGTH(nums1) - 1  
                
                SET filtered_nums1 TO []  
                
                FOR idx IN RANGE 0 TO LENGTH(nums1) - 1  
                    IF idx != start_index AND idx != end_index  
                        APPEND nums1[idx] TO filtered_nums1  
                    END IF  
                END FOR  
                
                SET offset TO nums2[0] - filtered_nums1[0]  
                
                SET is_match TO TRUE  
                FOR pos IN RANGE 0 TO LENGTH(nums2) - 1  
                    IF filtered_nums1[pos] + offset != nums2[pos]  
                        SET is_match TO FALSE  
                        BREAK  
                    END IF  
                END FOR  
                
                IF is_match  
                    RETURN offset  
                END IF  
            END FOR  
        END FOR  
        
        RETURN None  
    END FUNCTION  
END CLASS
