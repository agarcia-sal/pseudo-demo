CLASS Solution  
    FUNCTION maximumPoints(enemyEnergies currentEnergy)  
        enemyEnergies SORTED AS ascendingList BY ascending order  
        IF currentEnergy < ascendingList[0] THEN  
            RETURN 0  
        END IF  
        
        ans ← 0  
        idx ← LENGTH(ascendingList) - 1  
        
        WHILE idx >= 0 DO  
            quotient ← currentEnergy DIV ascendingList[0]  
            ans ← ans + quotient  
            currentEnergy ← currentEnergy MOD ascendingList[0]  
            currentEnergy ← currentEnergy + ascendingList[idx]  
            idx ← idx - 1  
        END WHILE  
        
        RETURN ans  
    END FUNCTION  
END CLASS
