CLASS Solution  
    FUNCTION maxScore(nums)  
        LET length = LENGTH OF nums  
        INITIALIZE dp AS ARRAY OF ZEROS WITH SIZE length  
        ASSIGN dp[0] = 0  
        FOR index FROM 1 TO length - 1  
            FOR prev FROM 0 TO index - 1  
                LET potential = dp[prev] + (index - prev) * nums[index]  
                IF dp[index] < potential THEN  
                    dp[index] = potential  
                END IF  
            END FOR  
        END FOR  
        RETURN dp[length - 1]  
    END FUNCTION  
END CLASS
