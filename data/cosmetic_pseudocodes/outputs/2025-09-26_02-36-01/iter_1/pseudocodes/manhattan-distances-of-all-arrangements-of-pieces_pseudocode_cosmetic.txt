CLASS Solution  
    FUNCTION distanceSum WITH PARAMETERS m n k  
        DEFINE MODULO_VALUE AS 1000000007  
        
        CALCULATE row_part AS ((n * n) * (m * m) * (m * m - m)) / 6  
        CALCULATE col_part AS ((m * m) * (n * n) * (n * n - n)) / 6  
        
        CALCULATE total_points AS m * n  
        CALCULATE combination_count AS COMBINATION(total_points - 2, k - 2)  
        
        COMPUTE combined_sum AS (row_part + col_part) * combination_count  
        RETURN combined_sum MODULO MODULO_VALUE  
    END FUNCTION  
END CLASS
