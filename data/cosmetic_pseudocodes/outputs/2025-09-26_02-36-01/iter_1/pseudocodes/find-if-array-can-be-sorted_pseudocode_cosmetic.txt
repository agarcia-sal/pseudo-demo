CLASS Solution  
    FUNCTION canSortArray(nums)  
        FUNCTION bitCount(x)  
            SET total TO 0  
            WHILE x GREATER THAN zero  
                INCREMENT total BY x BITWISE-AND 1  
                SET x TO x RIGHT-SHIFTED BY 1  
            END WHILE  
            RETURN total  
        END FUNCTION  
        
        SET length TO LENGTH OF nums  
        SET target TO COPY OF nums  
        SORT target IN ASCENDING ORDER  
        
        FOR index1 FROM zero TO length MINUS one  
            FOR index2 FROM zero TO length MINUS two  
                IF bitCount(nums[index2]) EQUALS bitCount(nums[index2 + 1]) AND nums[index2] GREATER THAN nums[index2 + 1]  
                    SET swapVar TO nums[index2]  
                    SET nums[index2] TO nums[index2 + 1]  
                    SET nums[index2 + 1] TO swapVar  
                END IF  
            END FOR  
        END FOR  
        
        RETURN nums IS EQUAL TO target  
    END FUNCTION  
END CLASS
