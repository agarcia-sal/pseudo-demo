CLASS Solution  
    FUNCTION minimumLevels(possible)  
        INITIALIZE sum_points TO 0  
        FOR value IN possible DO  
            sum_points = sum_points + (2 * value - 1)  
        ENDFOR  
  
        SET alice_score TO 0  
        FOR i FROM 0 TO LENGTH(possible) - 2 DO  
            alice_score = alice_score + (2 * possible[i] - 1)  
            sum_points = sum_points - (2 * possible[i] - 1)  
  
            IF alice_score > sum_points THEN  
                RETURN i + 1  
            END IF  
        ENDFOR  
  
        RETURN -1  
    END FUNCTION  
END CLASS
