CLASS Solution  
    FUNCTION countDays(days, meetings)  
        INITIALIZE available_days TO 0  
        INITIALIZE current_day TO 1  
        SORT meetings IN ASCENDING ORDER BASED ON meeting[0]  
        
        FOR meeting IN meetings DO  
            LET start = meeting[0]  
            LET end = meeting[1]  
            
            IF current_day < start THEN  
                LET available_days = available_days + (start - current_day)  
            END IF  
            
            IF end + 1 > current_day THEN  
                LET current_day = end + 1  
            END IF  
        END FOR  
        
        IF current_day <= days THEN  
            LET available_days = available_days + (days - current_day + 1)  
        END IF  
        
        RETURN available_days  
    END FUNCTION  
END CLASS
