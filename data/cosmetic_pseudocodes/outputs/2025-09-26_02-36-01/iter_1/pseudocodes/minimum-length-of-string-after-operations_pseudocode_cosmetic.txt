CLASS Solution  
    FUNCTION minimumLength(s)  
        DECLARE frequencyMap AS Counter of characters in s  
        INITIALIZE oddCount TO 0  
        INITIALIZE evenSum TO 0  
        FOR EACH frequency IN frequencyMap values DO  
            IF frequency MOD 2 IS NOT 0 THEN  
                oddCount = oddCount + 1  
            ELSE IF frequency != 0 THEN  
                evenSum = evenSum + 2  
            END IF  
        END FOR  
        RETURN oddCount + evenSum  
    END FUNCTION  
END CLASS
