CLASS Solution  
    FUNCTION countSubstrings(s)  
        INITIALIZE count TO 0  
        LET length BE THE LENGTH OF s  
        FOR startIndex IN RANGE FROM 0 TO length - 1  
            INITIALIZE num TO 0  
            FOR endIndex IN RANGE FROM startIndex TO length - 1  
                num = num * 10 + CONVERT s[endIndex] TO INTEGER  
                LET digit = CONVERT s[endIndex] TO INTEGER  
                IF digit != 0 AND num % digit == 0 THEN  
                    count = count + 1  
                END IF  
            END FOR  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
