CLASS Solution
    FUNCTION numberOfAlternatingGroups(colors k)
        IF k LESS THAN 3 THEN
            RETURN 0
        ENDIF

        LET length = LENGTH(colors)
        LET extended_colors = EMPTY LIST

        FOR index FROM 0 TO length + k - 2
            APPEND colors[index MOD length] TO extended_colors
        ENDFOR

        LET group_count = 0

        INDEX_LOOP: FOR start FROM 0 TO length - 1
            FOR offset FROM 1 TO k - 1
                IF extended_colors[start + offset] == extended_colors[start + offset - 1] THEN
                    CONTINUE INDEX_LOOP
                ENDIF
            ENDFOR
            group_count = group_count + 1
        ENDFOR

        RETURN group_count
    END FUNCTION
END CLASS
