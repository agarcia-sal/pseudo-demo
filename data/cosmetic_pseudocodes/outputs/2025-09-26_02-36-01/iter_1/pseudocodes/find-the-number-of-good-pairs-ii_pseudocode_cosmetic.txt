CLASS Solution  
    FUNCTION numberOfPairs(nums1, nums2, k)  
        SET freq_map TO DICTIONARY  
        FOR value IN nums2  
            IF value EXISTS IN freq_map THEN  
                INCREMENT freq_map[value] BY 1  
            ELSE  
                SET freq_map[value] TO 1  
            END IF  
        END FOR  
  
        SET total_pairs TO 0  
        FOR element1 IN nums1  
            FOR key, occurrences IN freq_map  
                IF (element1 MOD (key * k)) == 0 THEN  
                    total_pairs = total_pairs + occurrences  
                END IF  
            END FOR  
        END FOR  
  
        RETURN total_pairs  
    END FUNCTION  
END CLASS
