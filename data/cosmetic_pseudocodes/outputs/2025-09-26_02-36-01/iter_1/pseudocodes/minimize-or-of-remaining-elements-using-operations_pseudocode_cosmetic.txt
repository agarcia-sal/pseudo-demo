CLASS Solution  
    FUNCTION minOrAfterOperations(nums, k)  
        FUNCTION canAchieve(target_or, k)  
            SET current_and TO -1  
            SET operations TO 0  
            FOR num IN nums DO  
                IF current_and = -1 THEN  
                    current_and ← num  
                ELSE  
                    current_and ← current_and AND num  
                END IF  
                IF (current_and AND target_or) = 0 THEN  
                    current_and ← -1  
                ELSE  
                    operations ← operations + 1  
                    IF operations > k THEN  
                        RETURN false  
                    END IF  
                END IF  
            END FOR  
            RETURN true  
        END FUNCTION  

        SET max_val TO (1 << 30) - 1  
        SET answer TO max_val  

        FOR bit FROM 0 TO 29 DO  
            SET mask TO 1 << bit  
            IF (answer AND mask) = 0 THEN  
                CONTINUE  
            END IF  
            IF canAchieve((NOT answer) XOR mask, k) THEN  
                answer ← answer AND (NOT mask)  
            END IF  
        END FOR  

        RETURN answer  
    END FUNCTION  
END CLASS
