CLASS Solution  
    FUNCTION minimumDifference(nums k)  
        SET array_length TO LENGTH OF nums  
        SET closest_difference TO +âˆž  

        FOR left_index FROM 0 TO array_length - 1  
            SET accumulated_or TO 0  
            FOR right_index FROM left_index TO array_length - 1  
                accumulated_or = accumulated_or BITWISE_OR nums[right_index]  
                SET current_diff TO ABS(k - accumulated_or)  
                IF current_diff < closest_difference THEN  
                    closest_difference = current_diff  
                END IF  
                IF current_diff == 0 THEN  
                    RETURN 0  
                END IF  
            END FOR  
        END FOR   

        RETURN closest_difference  
    END FUNCTION  
END CLASS
