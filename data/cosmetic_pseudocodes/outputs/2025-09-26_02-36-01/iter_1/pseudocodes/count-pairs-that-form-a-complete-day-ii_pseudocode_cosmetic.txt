CLASS Solution  
    FUNCTION countCompleteDayPairs(hours)  
        CREATE dictionary remainderCount WITH default 0  
        INITIALIZE totalPairs TO 0  
        FOR hour IN hours DO  
            COMPUTE currentRemainder AS hour MOD 24  
            COMPUTE neededRemainder AS (24 - currentRemainder) MOD 24  
            ADD remainderCount[neededRemainder] TO totalPairs  
            INCREMENT remainderCount[currentRemainder] BY 1  
        END FOR  
        RETURN totalPairs  
    END FUNCTION  
END CLASS
