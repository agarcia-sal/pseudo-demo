CLASS Solution  
    FUNCTION minimumArrayLength(nums)  
        LET smallest = INFINITY  
        LET occurrences = 0  
        FOR EACH num IN nums  
            IF num < smallest THEN  
                smallest = num  
                occurrences = 1  
            ELSE IF num == smallest THEN  
                occurrences = occurrences + 1  
            END IF  
        END FOR  
        IF occurrences == 1 THEN  
            RETURN 1  
        ELSE  
            RETURN (occurrences + 1) // 2  
        END IF  
    END FUNCTION  
END CLASS
