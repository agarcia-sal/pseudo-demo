CLASS Solution  
    FUNCTION maxPointsInsideSquare(points s)  
        LET total_points = LENGTH OF s  
        LET maximum_covered = 0  
        FOR index IN 0 TO total_points - 1 DO  
            LET current_x = s[index][0]  
            LET current_y = s[index][1]  
            LET boundary = MAX(ABS(current_x), ABS(current_y))  
            LET encountered_tags = EMPTY MAP  
            LET is_square_valid = TRUE  
            FOR idx IN 0 TO total_points - 1 DO  
                LET candidate_x = s[idx][0]  
                LET candidate_y = s[idx][1]  
                IF ABS(candidate_x) <= boundary AND ABS(candidate_y) <= boundary THEN  
                    LET candidate_tag = s[idx]  
                    IF candidate_tag IN encountered_tags THEN  
                        is_square_valid = FALSE  
                        BREAK  
                    ELSE  
                        encountered_tags[candidate_tag] = TRUE  
                    END IF  
                END IF  
            END FOR  
            IF is_square_valid THEN  
                LET current_count = COUNT OF KEYS IN encountered_tags  
                maximum_covered = MAX(maximum_covered, current_count)  
            END IF  
        END FOR  
        RETURN maximum_covered  
    END FUNCTION  
END CLASS
