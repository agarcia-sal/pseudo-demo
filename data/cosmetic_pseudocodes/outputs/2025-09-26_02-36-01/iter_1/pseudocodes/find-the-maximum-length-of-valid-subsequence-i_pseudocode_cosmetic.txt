CLASS Solution  
    FUNCTION maximumLength(nums)  
        INITIALIZE even_length AS 0  
        INITIALIZE odd_length AS 0  
        
        FOR index FROM 1 TO LENGTH(nums) - 1  
            LET sum_pair = nums[index - 1] + nums[index]  
            
            IF (sum_pair % 2) == 0 THEN  
                even_length = MAX(even_length, odd_length) + 1  
            ELSE  
                odd_length = MAX(odd_length, even_length) + 1  
            END IF  
        END FOR  
        
        RETURN MAX(even_length, odd_length) + 1  
    END FUNCTION  
END CLASS
