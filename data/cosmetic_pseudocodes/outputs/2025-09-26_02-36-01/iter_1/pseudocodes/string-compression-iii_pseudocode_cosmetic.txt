CLASS Solution  
    FUNCTION compressedString(word)  
        DECLARE comp AS empty list  
        DECLARE i AS 0  
        WHILE i < LENGTH(word) DO  
            DECLARE currentChar AS word[i]  
            DECLARE frequency AS 0  
            LOOP  
                IF i >= LENGTH(word) OR word[i] != currentChar THEN  
                    EXIT LOOP  
                END IF  
                frequency = frequency + 1  
                i = i + 1  
                IF frequency == 9 THEN  
                    EXIT LOOP  
                END IF  
            END LOOP  
            comp.APPEND(TO_STRING(frequency) + currentChar)  
        END WHILE  
        DECLARE result AS EMPTY_STRING  
        FOR eachPart IN comp DO  
            result = result + eachPart  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
