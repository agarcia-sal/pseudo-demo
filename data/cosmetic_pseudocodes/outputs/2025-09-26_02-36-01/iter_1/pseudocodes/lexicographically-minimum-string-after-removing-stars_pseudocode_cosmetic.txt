CLASS Solution  
    FUNCTION clearStars(s)  
        INITIALIZE buffer AS an empty list  
        FOR index FROM 0 TO LENGTH(s) - 1  
            IF s[index] IS "*" THEN  
                IF buffer IS NOT EMPTY THEN  
                    DISCARD last element FROM buffer  
                END IF  
            ELSE  
                INSERT s[index] AT END OF buffer  
            END IF  
        END FOR  
        INITIALIZE output_string AS EMPTY STRING  
        FOR element IN buffer  
            CONCATENATE element TO output_string  
        END FOR  
        RETURN output_string  
    END FUNCTION  
END CLASS
