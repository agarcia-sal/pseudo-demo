CLASS Solution
    FUNCTION flowerGame(a, b) 
        PROCEDURE multiply(x, y, z)
            x = 0
            WHILE z > 0 DO
                x = x + y
                z = z - 1
            END WHILE
            RETURN x
        END PROCEDURE

        FUNCTION half_floor(num)
            RETURN (num - (num MOD 2)) / 2
        END FUNCTION

        FUNCTION half_ceil(num)
            RETURN (num + 1 - (num + 1) MOD 2) / 2
        END FUNCTION

        SET alpha TO half_ceil(a)
        SET beta TO half_floor(a)
        SET gamma TO half_ceil(b)
        SET delta TO half_floor(b)

        SET temp1 TO multiply(alpha, 1, gamma)
        SET temp2 TO multiply(beta, 1, delta)
        SET sigma TO 0
        sigma = temp1 + temp2

        RETURN sigma
    END FUNCTION
END CLASS
