CLASS Solution
    FUNCTION minimumDifference(nums k)
        FUNCTION sub_or(a b)
            SET zed TO 0
            SET q TO a
            WHILE q <= b
                SET zed TO (zed BITWISE_OR nums[q])
                SET q TO q + 1
            END WHILE
            RETURN zed
        END FUNCTION

        SET length_val TO LENGTH OF nums
        SET ultra_min TO +âˆž

        SET alpha TO 0
        REPEAT
            SET beta TO alpha
            SET curr_combined TO 0
            REPEAT
                SET curr_combined TO (curr_combined OR nums[beta])
                SET temp_diff TO (k - curr_combined)
                IF temp_diff < 0 THEN
                    SET temp_diff TO (-temp_diff)
                END IF

                IF temp_diff < ultra_min THEN
                    SET ultra_min TO temp_diff
                END IF

                IF temp_diff = 0 THEN
                    RETURN 0
                END IF

                SET beta TO beta + 1
            UNTIL beta > (length_val - 1)

            SET alpha TO alpha + 1
        UNTIL alpha > (length_val - 1)

        RETURN ultra_min
    END FUNCTION
END CLASS
