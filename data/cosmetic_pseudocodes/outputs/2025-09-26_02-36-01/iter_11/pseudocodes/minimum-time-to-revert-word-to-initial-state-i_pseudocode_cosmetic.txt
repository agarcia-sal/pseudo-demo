CLASS Solution
    FUNCTION minimumTimeToInitialState(word PARAMETER k)
        SET fz TO LENGTH OF word
        SET sq TO 1
        REPEAT
            IF (sq * k >= fz) OR (SUBSTRING(word, sq * k, fz) = SUBSTRING(word, 1, fz - sq * k)) THEN
                RETURN sq
            END IF
            SET sq TO sq + 1
        UNTIL FALSE
    END FUNCTION
END CLASS
