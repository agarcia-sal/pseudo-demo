CLASS Solution
	FUNCTION countSubstrings(s c)
		DEFINE idxVar ← 0
		DEFINE lengthVal ← 0
		DEFINE freqCounter ← 0
		DEFINE outputVal ← 0

		lengthVal ← 0
		WHILE lengthVal < LENGTH(s) DO
			lengthVal ← lengthVal + 1
		END WHILE

		freqCounter ← 0
		idxVar ← 0
		REPEAT
			IF s[(LENGTH(s) - 1) - idxVar] = c THEN
				freqCounter ← freqCounter + 1
			END IF
			idxVar ← idxVar + 1
		UNTIL idxVar ≥ LENGTH(s)

		outputVal ← ((freqCounter * (freqCounter + 1)) / 2)

		RETURN outputVal
	END FUNCTION
END CLASS
