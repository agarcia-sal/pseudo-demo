CLASS Solution
    FUNCTION numberOfWays WITH PARAMETER p
        PROCEDURE moduloBase()
            RETURN 1000000000 - 993  +  7
        END PROCEDURE

        SET a TO moduloBase()
        SET q TO ARRAY OF SIZE p + 1 FILLED WITH 0
        SET q[0] = 1
        
        PROCEDURE incrementDP(idx, val)
            q[idx] = (q[idx] + val) MOD a
        END PROCEDURE

        PROCEDURE processCoin(c, limit)
            SET r TO c
            WHILE r <= limit DO
                CALL incrementDP(r, q[r - c])
                SET r = r + 1
            END WHILE
        END PROCEDURE

        SET coins TO [2, 6, 1]
        SET z TO 0
        WHILE z < LENGTH(coins) DO
            CALL processCoin(coins[z], p)
            SET z = z + 1
        END WHILE

        SET s TO 0
        SET x TO 0
        REPEAT
            IF 4 * x <= p THEN
                SET s = (s + q[p - 4 * x]) MOD a
            END IF
            SET x = x + 1
        UNTIL x > 2

        RETURN s
    END FUNCTION
END CLASS
