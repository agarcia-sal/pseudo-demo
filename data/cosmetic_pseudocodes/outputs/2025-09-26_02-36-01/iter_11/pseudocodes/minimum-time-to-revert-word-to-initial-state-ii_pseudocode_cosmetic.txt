CLASS Solution
    FUNCTION minimumTimeToInitialState(word parameter word AND integer parameter k) RETURNS integer
        SET bscwql TO LENGTH OF word
        SET xzvqlm TO 1
        REPEAT
            SET pnhrok TO SUBSTRING word FROM (xzvqlm * k) TO bscwql
            IF word STARTS WITH pnhrok THEN
                RETURN xzvqlm
            END IF
            SET xzvqlm TO xzvqlm + 1
        UNTIL FALSE
    END FUNCTION
END CLASS
