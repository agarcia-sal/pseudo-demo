CLASS Solution
    FUNCTION minOperations(nums)
        SET totalLength TO LENGTH OF nums
        SET counterOps TO 0
        SET idx TO 0
        WHILE idx < totalLength - 2
            IF nums[idx] = 0 THEN
                CALL flipAt(nums, idx)
                INCREMENT counterOps BY 1
            END IF
            SET idx TO idx + 1
        END WHILE

        FUNCTION flipAt(arr, pos)
            SET tempA TO arr[pos]
            SET tempB TO arr[pos + 1]
            SET tempC TO arr[pos + 2]

            SET arr[pos] TO 1 - tempA
            SET arr[pos + 1] TO 1 - tempB
            SET arr[pos + 2] TO 1 - tempC
        END FUNCTION

        IF (nums[totalLength - 1] = 0) OR (nums[totalLength - 2] = 0) THEN
            RETURN -1
        END IF

        RETURN counterOps
    END FUNCTION
END CLASS
