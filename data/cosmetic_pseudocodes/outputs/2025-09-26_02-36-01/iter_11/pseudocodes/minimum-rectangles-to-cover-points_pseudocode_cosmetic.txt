CLASS Solution
    FUNCTION minRectanglesToCoverPoints(points, w)
        FUNCTION local_sorter(a, b)
            RETURN (a[0] < b[0])
        END FUNCTION

        CALL local_sort(points, local_sorter)

        SET kvw TO 0
        SET togr TO -1

        FUNCTION proc1(idx)
            IF idx >= LENGTH(points) THEN
                RETURN
            END IF

            SET prg TO points[idx]
            SET vky TO prg[0]
            SET fqk TO prg[1]

            IF vky > togr THEN
                SET togr TO vky + w
                SET kvw TO kvw + 1
            END IF

            CALL proc1(idx + 1)
        END FUNCTION

        CALL proc1(0)
        
        RETURN kvw
    END FUNCTION
END CLASS
