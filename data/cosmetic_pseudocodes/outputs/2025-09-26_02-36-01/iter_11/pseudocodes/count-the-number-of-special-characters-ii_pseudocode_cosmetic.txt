CLASS Solution
    FUNCTION numberOfSpecialChars(ζ)
        VAR αξ ← empty collection of key value pairs
        VAR μω ← empty collection of key value pairs

        VAR θ₁ ← 0
        WHILE θ₁ < LENGTH(ζ)
            VAR κφ ← ζ[θ₁]
            IF NOT (κφ IN KEYS(αξ))
                αξ[κφ] ← θ₁
            END IF
            μω[κφ] ← θ₁
            θ₁ ← θ₁ + 1
        END WHILE

        VAR ψω ← 0
        VAR λβ ← 0
        VAR πσ ← LENGTH("abcdefghijklmnopqrstuvwxyz")
        WHILE λβ < πσ
            VAR δξ ← "abcdefghijklmnopqrstuvwxyz"[λβ]
            VAR η₀ ← "ABCDEFGHIJKLMNOPQRSTUVWXYZ"[λβ]
            IF (δξ IN KEYS(μω)) AND (η₀ IN KEYS(αξ)) AND (μω[δξ] < αξ[η₀])
                ψω ← ψω + 1
            END IF
            λβ ← λβ + 1
        END WHILE

        RETURN ψω
    END FUNCTION
END CLASS
