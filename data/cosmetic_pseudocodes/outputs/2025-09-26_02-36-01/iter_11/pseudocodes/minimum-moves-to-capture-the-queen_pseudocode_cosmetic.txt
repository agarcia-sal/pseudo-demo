CLASS Solution
    FUNCTION minMovesToCaptureTheQueen(x1 y1 x2 y2 x3 y3)
        FUNCTION EQ(m n)
            RETURN m EQUALS n
        END FUNCTION

        FUNCTION NEQ(p q)
            RETURN NOT EQ(p q)
        END FUNCTION

        FUNCTION LT(u v)
            RETURN u LESS THAN v
        END FUNCTION

        FUNCTION ANDOP(a b)
            RETURN a AND b
        END FUNCTION

        FUNCTION OROP(a b)
            RETURN a OR b
        END FUNCTION

        FUNCTION SUM(r s)
            RETURN r PLUS s
        END FUNCTION

        FUNCTION DIFF(t w)
            RETURN t MINUS w
        END FUNCTION

        FUNCTION PATHCOND1()
            IF NEQ(x2 x1) THEN RETURN one END IF            
            IF OROP(
                ANDOP(LT(y1 y2) AND LT(y2 y3)),
                ANDOP(LT(y3 y2) AND LT(y2 y1))
            ) THEN RETURN two END IF
            RETURN one
        END FUNCTION

        FUNCTION PATHCOND2()
            IF NEQ(y2 y1) THEN RETURN one END IF           
            IF OROP(
                ANDOP(LT(x1 x2) AND LT(x2 x3)),
                ANDOP(LT(x3 x2) AND LT(x2 x1))
            ) THEN RETURN two END IF
            RETURN one
        END FUNCTION

        FUNCTION PATHCOND3()
            IF NEQ(SUM(x1 y1) SUM(x2 y2)) THEN RETURN one END IF          
            IF OROP(
                ANDOP(LT(x2 x1) AND LT(x1 x3) AND LT(y2 y1) AND LT(y1 y3)),
                ANDOP(LT(x3 x1) AND LT(x1 x2) AND LT(y3 y1) AND LT(y1 y2))
            ) THEN RETURN two END IF
            RETURN one
        END FUNCTION

        FUNCTION PATHCOND4()
            IF NEQ(DIFF(x1 y1) DIFF(x2 y2)) THEN RETURN one END IF          
            IF OROP(
                ANDOP(LT(x2 x1) AND LT(x1 x3) AND (y2 GREATER THAN y1) AND (y1 GREATER THAN y3)),
                ANDOP(LT(x3 x1) AND LT(x1 x2) AND (y3 GREATER THAN y1) AND (y1 GREATER THAN y2))
            ) THEN RETURN two END IF
            RETURN one
        END FUNCTION

        FUNCTION MAINRECUR(i R e q u i r e d)
            IF EQ(x1 x3) THEN
                RETURN PATHCOND1()
            ELSE
                IF EQ(y1 y3) THEN
                    RETURN PATHCOND2()
                ELSE
                    IF EQ(SUM(x2 y2) SUM(x3 y3)) THEN
                        RETURN PATHCOND3()
                    ELSE
                        IF EQ(DIFF(x2 y2) DIFF(x3 y3)) THEN
                            RETURN PATHCOND4()
                        ELSE
                            RETURN two
                        END IF
                    END IF
                END IF
            END IF
        END FUNCTION

        RETURN MAINRECUR(0 0 0 0 0)
    END FUNCTION
END CLASS
