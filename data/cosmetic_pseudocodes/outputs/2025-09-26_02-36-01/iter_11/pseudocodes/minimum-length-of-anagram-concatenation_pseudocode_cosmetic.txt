CLASS Solution
    FUNCTION minAnagramLength(t)
        DEFINE FUNCTION buildCharSet(inputStr)
            DECLARE tempSet AS empty set
            DECLARE idx AS 0
            REPEAT
                IF idx >= LENGTH(inputStr) THEN
                    BREAK
                END IF
                DECLARE ch AS inputStr[idx]
                tempSet ADD ch
                SET idx TO idx + 1
            UNTIL FALSE
            RETURN tempSet
        END FUNCTION
        DECLARE charCollection AS buildCharSet(t)
        DECLARE countChars AS (0 + LENGTH(charCollection))
        RETURN countChars
    END FUNCTION
END CLASS
