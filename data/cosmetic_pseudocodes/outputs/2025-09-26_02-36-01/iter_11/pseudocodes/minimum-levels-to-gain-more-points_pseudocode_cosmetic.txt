CLASS Solution
    FUNCTION minimumLevels(possible)
        FUNCTION multiplyByTwoMinusOne(x)
            RETURN (x + x) - 1
        END FUNCTION

        SET accumulator_x TO 0
        SET iterator_j TO 0
        SET length_k TO LENGTH OF possible
        WHILE iterator_j < length_k THEN
            SET accumulator_x TO accumulator_x + multiplyByTwoMinusOne(possible[iterator_j])
            SET iterator_j TO iterator_j + 1
        END WHILE

        SET count_p TO 0
        SET counter_i TO 0
        REPEAT
            SET count_p TO count_p + multiplyByTwoMinusOne(possible[counter_i])
            SET accumulator_x TO accumulator_x - multiplyByTwoMinusOne(possible[counter_i])
            IF count_p > accumulator_x THEN
                RETURN counter_i + 1
            END IF
            SET counter_i TO counter_i + 1
        UNTIL counter_i > length_k - 2

        RETURN -1
    END FUNCTION
END CLASS
