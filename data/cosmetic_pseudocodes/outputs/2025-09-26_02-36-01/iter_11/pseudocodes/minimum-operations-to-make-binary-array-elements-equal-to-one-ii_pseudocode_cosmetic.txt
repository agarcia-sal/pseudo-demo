CLASS Solution
    FUNCTION minOperations(inputArray)
        DEFINE counter AS 0
        DEFINE toggleFlag AS 0
        DEFINE index AS 0

        WHILE index < LENGTH(inputArray)
            DEFINE tempValue AS 0

            IF (toggleFlag % 2) = 0 THEN
                SET tempValue TO inputArray[index]
            ELSE
                SET tempValue TO (1 - inputArray[index])
            END IF

            IF tempValue = 0 THEN
                SET counter TO counter + 1
                SET toggleFlag TO toggleFlag + 1
            END IF

            SET index TO index + 1
        END WHILE

        RETURN counter
    END FUNCTION
END CLASS
