CLASS Solution
    FUNCTION clearStars(inputString)
        PROCEDURE deleteLastElement(LIST)
            SET count TO LENGTH OF LIST
            IF count GREATER THAN zero THEN
                REMOVE element AT index (count MINUS 1) FROM LIST
            END IF
        END PROCEDURE

        SET buffer TO an empty list
        SET indexVar TO 0

        REPEAT
            IF indexVar GREATER OR EQUAL TO LENGTH OF inputString THEN 
                EXIT REPEAT
            END IF

            SET currentChar TO inputString AT indexVar

            IF (NOT (currentChar NOT EQUAL TO '*')) THEN
                CALL deleteLastElement(buffer)
            ELSE
                APPEND currentChar TO buffer
            END IF

            SET indexVar TO indexVar PLUS 1
        UNTIL FALSE

        SET assembledString TO ""
        SET pointer TO 0

        WHILE pointer LESS THAN LENGTH OF buffer DO
            SET assembledString TO assembledString CONCATENATED WITH buffer AT pointer
            SET pointer TO pointer PLUS 1
        END WHILE

        RETURN assembledString
    END FUNCTION
END CLASS
