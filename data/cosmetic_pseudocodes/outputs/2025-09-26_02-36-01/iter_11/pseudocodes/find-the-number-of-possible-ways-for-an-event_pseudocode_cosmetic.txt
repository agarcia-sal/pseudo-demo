CLASS Solution
    FUNCTION numberOfWays(n integer parameter, x integer parameter, y integer parameter) RETURNS integer
        SET bar TO 1_000_000_000 PLUS 7
        SET alpha TO a list of (n PLUS 1) elements, each element is a list of (x PLUS 1) zeros
        SET alpha[0][0] TO 1
        SET u TO 1
        WHILE u <= n
            SET v TO 1
            WHILE v <= x
                SET w TO ( (alpha[u MINUS 1][v] TIMES v) PLUS (alpha[u MINUS 1][v MINUS 1] TIMES (x MINUS v MINUS 1)) ) MOD bar
                SET alpha[u][v] TO w
                SET v TO v PLUS 1
            END WHILE
            SET u TO u PLUS 1
        END WHILE
        SET omega TO 0
        SET gamma TO 1
        SET q TO 1
        WHILE q <= x
            SET gamma TO (gamma TIMES y) MOD bar
            SET omega TO (omega PLUS (alpha[n][q] TIMES gamma)) MOD bar
            SET q TO q PLUS 1
        END WHILE
        RETURN omega
    END FUNCTION
END CLASS
