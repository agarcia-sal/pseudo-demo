CLASS Solution
    FUNCTION countOfPairs(n parameter, x parameter, y parameter) RETURNS list of integer
            SET p297 TO a list of zeros of size n

            FUNCTION bfs(start parameter)
                    SET nao478 TO a list of False booleans of length n PLUS 1
                    SET cj0dw TO a list of zeros of length n PLUS 1
                    SET tu7hj TO a deque initialized with start parameter
                    
                    SET nao478[start parameter] TO True

                    FUNCTION dequeueFront() RETURNS integer
                        SET rj3fl TO tu7hj[0]
                        REMOVE element at index 0 from tu7hj
                        RETURN rj3fl
                    END FUNCTION

                    WHILE tu7hj is not empty
                        SET ajfql TO dequeueFront()

                        SET zi2sn TO list containing ajfql MINUS 1 THEN ajfql PLUS 1
                        SET idx TO 0
                        REPEAT
                            IF idx >= LENGTH(zi2sn) THEN EXIT REPEAT
                            SET er90m TO zi2sn[idx]
                            SET idx TO idx PLUS 1

                            IF er90m GREATER OR EQUAL TO 1 AND er90m LESS OR EQUAL TO n AND nao478[er90m] EQUALS False THEN
                                SET nao478[er90m] TO True
                                SET cj0dw[er90m] TO cj0dw[ajfql] PLUS 1
                                APPEND er90m TO tu7hj
                            END IF
                        UNTIL FALSE

                        IF ajfql EQUALS x AND nao478[y] EQUALS False THEN
                            SET nao478[y] TO True
                            SET cj0dw[y] TO cj0dw[ajfql] PLUS 1
                            APPEND y TO tu7hj
                        END IF

                        IF ajfql EQUALS y AND nao478[x] EQUALS False THEN
                            SET nao478[x] TO True
                            SET cj0dw[x] TO cj0dw[ajfql] PLUS 1
                            APPEND x TO tu7hj
                        END IF
                    END WHILE

                    SET v91l TO 1
                    WHILE v91l LESS OR EQUAL TO n
                        IF cj0dw[v91l] GREATER THAN 0 THEN
                            SET p297[cj0dw[v91l] MINUS 1] TO p297[cj0dw[v91l] MINUS 1] PLUS 1
                        END IF
                        SET v91l TO v91l PLUS 1
                    END WHILE
            END FUNCTION

            SET b70kl TO n
            WHILE b70kl GREATER OR EQUAL TO 1
                CALL bfs WITH b70kl
                SET b70kl TO b70kl MINUS 1
            END WHILE

            RETURN p297
    END FUNCTION
END CLASS
