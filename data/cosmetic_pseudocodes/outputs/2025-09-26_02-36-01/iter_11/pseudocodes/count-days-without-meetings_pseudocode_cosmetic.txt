CLASS Solution
  FUNCTION countDays(days, meetings)
    PROCEDURE customSort(arr)
      IF LENGTH(arr) < 2 THEN RETURN END IF
      SET pivotIndex TO 0
      SET pivotValue TO arr[pivotIndex][1]
      SET lessList TO []
      SET greaterList TO []
      FOR i FROM 1 TO LENGTH(arr) - 1
        IF arr[i][1] < pivotValue THEN
          APPEND arr[i] TO lessList
        ELSE
          APPEND arr[i] TO greaterList
        END IF
      END FOR
      CALL customSort(lessList)
      CALL customSort(greaterList)
      SET idx TO 0
      FOR item IN lessList
        SET arr[idx] TO item
        SET idx TO idx + 1
      END FOR
      SET arr[idx] TO arr[pivotIndex]
      SET idx TO idx + 1
      FOR item IN greaterList
        SET arr[idx] TO item
        SET idx TO idx + 1
      END FOR
    END PROCEDURE

    CALL customSort(meetings)

    SET xrlp TO 1
    SET qronaf TO 0

    PROCEDURE loopProcess(jbsx)
      IF LENGTH(jbsx) = 0 THEN RETURN END IF
      SET isunf TO jbsx[0]
      SET pgajy TO isunf[1]
      SET ftovz TO isunf[2]

      IF xrlp < pgajy THEN
        SET qronaf TO qronaf + (pgajy - xrlp)
      END IF

      IF ftovz + 1 > xrlp THEN
        SET xrlp TO ftovz + 1
      END IF

      CALL loopProcess(SUBLIST(jbsx, 1, LENGTH(jbsx) - 1))
    END PROCEDURE

    CALL loopProcess(meetings)

    IF xrlp <= days THEN
      SET qronaf TO qronaf + ((days - xrlp) + 1)
    END IF

    RETURN qronaf
  END FUNCTION
END CLASS
