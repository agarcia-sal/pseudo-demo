CLASS Solution
    FUNCTION kthCharacter(k param_k INTEGER, operations param_operations LIST OF INTEGER) RETURNS STRING
        SET alpha TO 'a'
        SET count TO 1
        DECLARE sequence AS LIST OF INTEGER INITIALLY EMPTY

        SET idx TO 0
        REPEAT
            EXIT WHEN idx EQUALS LENGTH OF param_operations
            SET current TO param_operations[idx]
            CALL sequence.APPEND(current)
            CALL doubleLength(current, count)
            SET idx TO idx + 1
        UNTIL FALSE

        FUNCTION doubleLength(val INTEGER, REF len INTEGER)
            IF val EQUALS 0 OR val NOT EQUALS 0 THEN
                SET len TO len + len
            END IF
        END FUNCTION

        DECLARE position INTEGER
        SET position TO LENGTH OF sequence - 1

        WHILE position GREATER OR EQUALS 0 DO
            IF param_k LESS OR EQUALS count / 2 THEN
                SET count TO count / 2
            ELSE
                SET param_k TO param_k - (count / 2)
                SET count TO count / 2
                IF sequence[position] EQUALS 1 THEN
                    SET alpha TO nextChar(alpha)
                END IF
            END IF
            SET position TO position - 1
        END WHILE

        FUNCTION nextChar(ch STRING) RETURNS STRING
            SET code INTEGER TO ASCII(ch)
            SET base INTEGER TO ASCII('a')
            RETURN CHAR(base + (code - base + 1) MOD 26)
        END FUNCTION

        RETURN alpha
    END FUNCTION
END CLASS
