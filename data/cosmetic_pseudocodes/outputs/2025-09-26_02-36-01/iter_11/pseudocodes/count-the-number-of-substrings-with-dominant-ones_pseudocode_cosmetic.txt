CLASS Solution
    FUNCTION numberOfSubstrings(s)
  LET totalLen := LENGTH s
  LET aggregate := 0
  LET outerIdx := 0
  WHILE outerIdx <= totalLen - 1 DO
    LET accumulatorA := 0
    LET accumulatorB := 0
    LET innerIdx := outerIdx
    REPEAT
      IF s[innerIdx] <> '0' THEN
        LET accumulatorA := accumulatorA + 1
      ELSE
        LET accumulatorB := accumulatorB + 1
      END IF
      IF accumulatorA >= (accumulatorB * accumulatorB) THEN
        LET aggregate := aggregate + 1
      END IF
      LET innerIdx := innerIdx + 1
    UNTIL innerIdx > totalLen - 1
    LET outerIdx := outerIdx + 1
  END WHILE
  RETURN aggregate
    END FUNCTION
END CLASS
