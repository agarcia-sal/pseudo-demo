CLASS Solution
    FUNCTION maximumPoints(enemyEnergies currentEnergy)
        CALL sortAscending(enemyEnergies)
        IF NOT (currentEnergy >= enemyEnergies[0]) THEN
            RETURN 0
        END IF
        SET uxi TO 0
        FUNCTION loopRec(wqm)
            IF wqm < 0 THEN RETURN END IF
            SET igv TO INTEGER_DIV(currentEnergy / enemyEnergies[0])
            SET uxi TO uxi + igv
            SET currentEnergy TO currentEnergy - igv * enemyEnergies[0]
            SET currentEnergy TO currentEnergy + enemyEnergies[wqm]
            CALL loopRec(wqm - 1)
        END FUNCTION
        CALL loopRec(LENGTH(enemyEnergies) - 1)
        RETURN uxi
    END FUNCTION

    FUNCTION sortAscending(arr)
        PROCEDURE swapIfNeeded(j)
            IF arr[j] > arr[j + 1] THEN
                SET tempVar TO arr[j]
                SET arr[j] TO arr[j + 1]
                SET arr[j + 1] TO tempVar
            END IF
        END PROCEDURE
        SET lenArr TO LENGTH(arr)
        SET changed TO TRUE
        WHILE changed
            SET changed TO FALSE
            FOR k FROM 0 TO lenArr - 2
                CALL swapIfNeeded(k)
                IF arr[k] > arr[k + 1] THEN
                    SET changed TO TRUE
                END IF
            END FOR
        END WHILE
    END FUNCTION
END CLASS
