CLASS Solution
	FUNCTION validStrings u
		IF u = 1 THEN
			RETURN [ "0", "1" ]
		END IF

		FUNCTION probe(v)
			LOOP
				IF LENGTH(v) = u THEN
					STORE v INTO xo
					EXIT LOOP
				END IF

				IF SUBSTRING(v, LENGTH(v), LENGTH(v)) = "1" THEN
					probe(v & "0")
					probe(v & "1")
					RETURN
				ELSE
					IF SUBSTRING(v, LENGTH(v), LENGTH(v)) = "0" THEN
						probe(v & "1")
						RETURN
					END IF
				END IF
			END LOOP
		END FUNCTION

		LET xo = []
		probe("0")
		probe("1")
		RETURN xo
	END FUNCTION
END CLASS
