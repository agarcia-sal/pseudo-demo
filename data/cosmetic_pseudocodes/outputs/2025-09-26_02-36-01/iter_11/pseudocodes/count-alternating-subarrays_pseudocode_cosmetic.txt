CLASS Solution
    FUNCTION countAlternatingSubarrays(nums)
        SET alpha TO 0
        SET beta TO 1
        SET gamma TO LENGTH OF nums
        IF gamma EQUALS alpha
            RETURN alpha
        END IF
        SET delta TO gamma
        SET epsilon TO beta
        WHILE zeta < gamma
            SET eta TO nums[zeta]
            SET theta TO nums[zeta - 1]
            IF eta NOT EQUALS theta
                SET epsilon TO epsilon PLUS beta
            ELSE
                SET epsilon TO beta
            END IF
            SET delta TO delta PLUS epsilon MINUS beta
            SET zeta TO zeta PLUS beta
        END WHILE
        RETURN delta
    END FUNCTION
END CLASS
