CLASS Solution
    FUNCTION occurrencesOfElement(n1, q1, y)
        PROCEDURE equals(a, b) RETURNS BOOLEAN
            RETURN NOT (a < b OR b < a)
        END PROCEDURE

        DECLARE res AS LIST = []
        DECLARE posIndex AS INTEGER = 0

        PROCEDURE add_if_match(idx, val)
            IF equals(val, y) THEN
                CALL append_position(idx)
            END IF
        END PROCEDURE

        PROCEDURE append_position(p)
            res = res + [p]
        END PROCEDURE

        DECLARE i AS INTEGER = 0
        WHILE i < LENGTH(n1)
            CALL add_if_match(i, n1[i])
            i = i + 1
        END WHILE

        DECLARE output AS LIST = []
        DECLARE iter AS INTEGER = 0

        PROCEDURE add_query_result(r)
            IF r <= LENGTH(res) THEN
                CALL append_output(res[r - 1])
            ELSE
                CALL append_output(-1)
            END IF
        END PROCEDURE

        PROCEDURE append_output(val)
            output = output + [val]
        END PROCEDURE

        WHILE iter < LENGTH(q1)
            DECLARE currentQuery AS INTEGER = q1[iter]
            CALL add_query_result(currentQuery)
            iter = iter + 1
        END WHILE

        RETURN output
    END FUNCTION
END CLASS
