CLASS Solution
    FUNCTION maxScore(nums)
        SET s TO LENGTH OF nums
        FUNCTION helper(p)
            IF p EQUALS 1 THEN
                RETURN 0
            ELSE
                SET x TO 1
                SET y TO helper(p MINUS 1)
                SET z TO 0
                REPEAT  
                    IF x LESS THAN p THEN
                        SET cond TO (GET_ELEMENT(dp, p) LESS THAN (GET_ELEMENT(dp, x) PLUS ((p MINUS x) MULTIPLIED BY GET_ELEMENT(nums, p))))
                        IF cond THEN
                            SET z TO GET_ELEMENT(dp, x) PLUS ((p MINUS x) MULTIPLIED BY GET_ELEMENT(nums, p))
                            SET dp[p] TO z
                        END IF
                        SET x TO x PLUS 1
                    END IF
                UNTIL x GREATER_OR_EQUAL p
                RETURN dp[p]
            END IF
        END FUNCTION

        SET dp TO ARRAY_OF_ZEROS(s)
        SET dp[1] TO 0
        SET k TO 1
        WHILE k LESS_OR_EQUAL s DO
            CALL helper(k)
            SET k TO k PLUS 1
        END WHILE
        RETURN dp[s]
    END FUNCTION
END CLASS
