CLASS Solution
    FUNCTION findPattern(board, pattern)
        SET α TO LENGTH OF board
        SET β TO LENGTH OF element at position zero of board
        SET γ TO LENGTH OF pattern
        SET δ TO LENGTH OF element at position zero of pattern

        FUNCTION matches(x, y)
            SET Ω TO empty mapping
            SET Σ TO empty mapping
            
            FUNCTION isDigit(ch)
                RETURN (ch >= "0") AND (ch <= "9")
            END FUNCTION
            
            SET m TO zero
            WHILE m <= γ MINUS one DO
                SET n TO zero
                REPEAT
                    SET u TO element at position m of pattern
                    SET v TO element at position n of u
                    SET w TO element at position x PLUS m of board
                    SET z TO element at position y PLUS n of w

                    IF isDigit(v) THEN
                        IF (INTEGER VALUE OF v) ≠ z THEN
                            RETURN False
                        END IF
                    ELSE
                        IF v IN Ω THEN
                            IF Ω at key v ≠ z THEN
                                RETURN False
                            END IF
                        ELSE
                            IF z IN Σ THEN
                                RETURN False
                            END IF
                            ASSIGN z TO Ω at key v
                            ASSIGN v TO Σ at key z
                        END IF
                    END IF
                    
                    SET n TO n PLUS one
                UNTIL n > δ MINUS one
                SET m TO m PLUS one
            END WHILE
            
            RETURN True
        END FUNCTION

        SET i TO zero
        WHILE i <= α MINUS γ DO
            SET j TO zero
            REPEAT
                IF matches(i, j) THEN
                    RETURN list containing i and j
                END IF
                SET j TO j PLUS one
            UNTIL j > β MINUS δ
            SET i TO i PLUS one
        END WHILE

        RETURN list containing minus one and minus one
    END FUNCTION
END CLASS
