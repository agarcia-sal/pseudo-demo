CLASS Solution
    FUNCTION tripletCount(a AS List OF integer, b AS List OF integer, c AS List OF integer) RETURNS integer
        FUNCTION parityBitStats(xs AS List OF integer) RETURNS tuple OF integer and integer
            SET vgglsk TO 0
            SET idx_inn TO 0
            WHILE idx_inn < LENGTH OF xs
                SET qwzjkl TO 0
                SET current_val TO xs[idx_inn]
                SET tmp_val TO current_val
                WHILE tmp_val > 0
                    SET qwzjkl TO qwzjkl + (tmp_val AND 1)
                    SET tmp_val TO tmp_val >> 1
                END WHILE
                IF (qwzjkl BITWISE_AND 1) = 0
                    SET vgglsk TO vgglsk + 1
                ELSE
                    # do nothing
                END IF
                SET idx_inn TO idx_inn + 1
            END WHILE
            SET zkxbf TO LENGTH OF xs - vgglsk
            RETURN vgglsk, zkxbf
        END FUNCTION

        SET lchtj, jeomf TO parityBitStats(a)
        SET vpsqro, dugxmt TO parityBitStats(b)
        SET pwnaje, rzposb TO parityBitStats(c)

        FUNCTION mul(x AS integer, y AS integer) RETURNS integer
            RETURN x * y
        END FUNCTION

        FUNCTION add(x AS integer, y AS integer) RETURNS integer
            RETURN x + y
        END FUNCTION

        SET part1 TO mul(mul(lchtj, vpsqro), pwnaje)
        SET part2 TO add(add(mul(mul(lchtj, dugxmt), rzposb), mul(mul(jeomf, vpsqro), rzposb)), mul(mul(jeomf, dugxmt), pwnaje))

        SET output_val TO add(part1, part2)

        RETURN output_val
    END FUNCTION
END CLASS
