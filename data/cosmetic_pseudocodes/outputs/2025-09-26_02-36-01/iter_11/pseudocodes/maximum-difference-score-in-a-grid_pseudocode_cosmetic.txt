CLASS Solution
    FUNCTION maxScore(grid)
        SET T TO LENGTH(grid)
        SET U TO LENGTH(grid[0])
        SET W TO []
        WHILE LENGTH(W) < T DO
            APPEND W, []
        END WHILE
        REPEAT V FROM 0 TO T - 1
            WHILE LENGTH(W[V]) < U DO
                APPEND W[V], +∞
            END WHILE
        END REPEAT
        SET W[0][0] TO grid[0][0]
        SET K TO -∞
        
        SET A TO 1
        WHILE A < U DO
            SET W[0][A] TO MIN(W[0][A - 1], grid[0][A])
            SET A TO A + 1
        END WHILE
        
        SET B TO 1
        REPEAT
            IF B = T THEN
                BREAK
            END IF
            SET W[B][0] TO MIN(W[B - 1][0], grid[B][0])
            SET B TO B + 1
        UNTIL FALSE
        
        DEFINE FUNCTION HuntScore(X, Y)
            RETURN MIN(W[X][Y - 1], W[X - 1][Y])
        END FUNCTION
        
        SET I TO 1
        WHILE I < T DO
            SET J TO 1
            WHILE J < U DO
                SET W[I][J] TO HuntScore(I, J)
                SET L TO grid[I][J] - W[I][J]
                IF K < L THEN
                    SET K TO L
                END IF
                SET J TO J + 1
            END WHILE
            SET I TO I + 1
        END WHILE
        
        RETURN K
    END FUNCTION
END CLASS
