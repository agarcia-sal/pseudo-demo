CLASS Solution
    FUNCTION numberOfPairs(points)
        SET accumulator TO 0
        SET limit TO LENGTH OF points
        SET outerIndex TO 0
        WHILE outerIndex < limit DO
            SET innerIndex TO 0
            REPEAT
                IF outerIndex ≠ innerIndex THEN
                    SET alpha TO points[outerIndex][0]
                    SET beta TO points[outerIndex][1]
                    SET gamma TO points[innerIndex][0]
                    SET delta TO points[innerIndex][1]
                    IF (NOT (alpha > gamma)) AND (NOT (beta < delta)) THEN
                        SET flag TO TRUE
                        SET checkerIndex TO 0
                        WHILE checkerIndex < limit AND flag DO
                            IF checkerIndex ≠ outerIndex AND checkerIndex ≠ innerIndex THEN
                                SET epsilon TO points[checkerIndex][0]
                                SET zeta TO points[checkerIndex][1]
                                IF (alpha ≤ epsilon AND epsilon ≤ gamma) AND (beta ≥ zeta AND zeta ≥ delta) THEN
                                    SET flag TO FALSE
                                END IF
                            END IF
                            SET checkerIndex TO checkerIndex + 1
                        END WHILE
                        IF flag = TRUE THEN
                            SET accumulator TO accumulator + 1
                        END IF
                    END IF
                END IF
                SET innerIndex TO innerIndex + 1
            UNTIL innerIndex ≥ limit
            SET outerIndex TO outerIndex + 1
        END WHILE
        RETURN accumulator
    END FUNCTION
END CLASS
