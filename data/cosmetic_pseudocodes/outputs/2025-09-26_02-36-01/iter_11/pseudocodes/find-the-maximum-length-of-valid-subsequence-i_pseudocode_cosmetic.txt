CLASS Solution
    FUNCTION maximumLength(nums)
        SET alpha TO 0
        SET beta TO 0

        SET delta TO 1
        WHILE delta < LENGTH(nums) DO
            SET epsilon TO (nums[delta - 1] + nums[delta]) 
            SET zeta TO (epsilon % 2)
            IF zeta = 0 THEN
                SET alpha TO MAX(alpha + 1, beta)
            ELSE
                SET beta TO MAX(beta + 1, alpha)
            END IF

            SET delta TO delta + 1
        END WHILE

        SET eta TO MAX(alpha, beta) + 1
        RETURN eta
    END FUNCTION
END CLASS
