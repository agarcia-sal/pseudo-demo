CLASS Solution
    FUNCTION minimumOperationsToMakeKPeriodic(word parameter k)
        SET h TO 0
        SET mls TO LENGTH OF word
        SET bpd TO AN EMPTY LIST

        WHILE h < mls
            SET vof TO h PLUS k
            SET bfz TO ""
            REPEAT
                IF h >= vof OR h >= mls THEN
                    BREAK
                END IF
                SET bfz TO bfz CONCATENATED WITH CHARACTER AT word[h]
                SET h TO h PLUS 1
            UNTIL FALSE
            APPEND bfz TO bpd
        END WHILE

        FUNCTION HGLX(lrn)
            SET ume TO EMPTY MAP
            FOR EACH rtq IN lrn
                IF ume CONTAINS KEY rtq THEN
                    SET ume[rtq] TO ume[rtq] PLUS 1
                ELSE
                    SET ume[rtq] TO 1
                END IF
            END FOR
            RETURN ume
        END FUNCTION

        SET rkb TO HGLX(bpd)

        SET pxq TO -1
        FOR EACH vss IN KEYS OF rkb
            IF rkb[vss] > pxq THEN
                SET pxq TO rkb[vss]
            END IF
        END FOR

        SET tgk TO LENGTH OF bpd MINUS pxq

        RETURN tgk
    END FUNCTION
END CLASS
