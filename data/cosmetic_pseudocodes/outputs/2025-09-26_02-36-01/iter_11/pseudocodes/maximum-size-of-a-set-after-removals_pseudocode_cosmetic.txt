CLASS Solution
    FUNCTION maximumSetSize(nums1 PARAMETER List OF integers nums2 PARAMETER List OF integers) RETURNS integer
        SET beta TO 0
        SET omega TO 0
        SET delta TO LENGTH(nums1)
        SET gamma TO delta DIV 2
        SET alpha TO EMPTY SET
        SET zeta TO EMPTY SET
        SET kappa TO EMPTY SET
        SET lambda TO EMPTY SET
        SET mu TO 0
        SET nu TO 0
        SET xi TO 0
        SET rho TO 0
        SET sigma TO 0

        FUNCTION uniqueElements(inputList PARAMETER List OF integers) RETURNS Set
            SET tau TO EMPTY SET
            SET i TO 0
            WHILE i < LENGTH(inputList)
                SET e TO inputList[i]
                IF e NOT IN tau THEN
                    ADD e TO tau
                END IF
                SET i TO i + 1
            END WHILE
            RETURN tau
        END FUNCTION

        FUNCTION intersectionSets(a PARAMETER Set, b PARAMETER Set) RETURNS Set
            SET u TO EMPTY SET
            SET iterList TO a AS List
            SET j TO 0
            WHILE j < LENGTH(iterList)
                SET val TO iterList[j]
                IF val IN b THEN
                    ADD val TO u
                END IF
                SET j TO j + 1
            END WHILE
            RETURN u
        END FUNCTION

        FUNCTION differenceSets(orig PARAMETER Set, sub PARAMETER Set) RETURNS Set
            SET w TO EMPTY SET
            FOR EACH item IN orig
                IF item NOT IN sub THEN
                    ADD item TO w
                END IF
            END FOR EACH
            RETURN w
        END FUNCTION

        FUNCTION minVal(a PARAMETER integer, b PARAMETER integer) RETURNS integer
            IF a < b THEN
                RETURN a
            ELSE
                RETURN b
            END IF
        END FUNCTION

        FUNCTION maxVal(a PARAMETER integer, b PARAMETER integer) RETURNS integer
            IF a > b THEN
                RETURN a
            ELSE
                RETURN b
            END IF
        END FUNCTION

        SET alpha TO uniqueElements(nums1)
        SET zeta TO uniqueElements(nums2)

        SET kappa TO intersectionSets(alpha, zeta)

        SET lambda TO differenceSets(alpha, kappa)
        SET beta TO differenceSets(zeta, kappa)

        SET mu TO minVal(gamma, LENGTH(lambda))
        SET nu TO minVal(gamma, LENGTH(beta))
        SET xi TO maxVal(0, gamma - mu)
        SET rho TO maxVal(0, gamma - nu)

        SET sigma TO mu + nu + minVal(xi + rho, LENGTH(kappa))

        RETURN sigma
    END FUNCTION
END CLASS
