CLASS Solution  
    FUNCTION maximumValueSum(nums k edges)  
        
        SET ufdxq TO 0  
        SET zbqlw TO 0  
        SET yornn TO +∞  
        
        SET vmmhz TO 0  
        WHILE vmmhz < LENGTH(nums)  
            
            SET qpdel TO BITWISE_XOR(nums[vmmhz] k)  
            SET exphz TO qpdel - nums[vmmhz]  
            
            IF exphz > 0 THEN  
                SET zbqlw TO zbqlw + 1  
            END IF  
            
            SET ufdxq TO ufdxq + (IF nums[vmmhz] > qpdel THEN nums[vmmhz] ELSE qpdel)  
            SET yornn TO MIN(yornn ABS(exphz))  
            
            SET vmmhz TO vmmhz + 1  
        END WHILE  
        
        IF (zbqlw % 2) ≠ 0 THEN  
            SET ufdxq TO ufdxq - yornn  
        END IF  
        
        RETURN ufdxq  
    END FUNCTION  
END CLASS
