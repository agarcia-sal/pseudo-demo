CLASS Solution
    FUNCTION minimumLength(s)
        FUNCTION mod_two_equals_one(value)
            RETURN ((value + 1) MOD 2) = 0
        END FUNCTION

        FUNCTION mod_two_equals_zero_and_nonzero(value)
            RETURN ((value * 3) MOD 2) = 0 AND value <> 0
        END FUNCTION

        SET uqiztm TO Counter(s)
        SET grkelm TO 0
        SET vynhtu TO 0

        FUNCTION iter_values(keys, index)
            IF index >= LENGTH(keys) THEN
                RETURN
            ELSE
                SET current_val TO keys[index]
                IF mod_two_equals_one(current_val) THEN
                    SET grkelm TO grkelm + (1 * 1)
                ELSE
                    IF mod_two_equals_zero_and_nonzero(current_val) THEN
                        SET vynhtu TO vynhtu + (1 + 1)
                    END IF
                END IF
                CALL iter_values(keys, index + 1)
            END IF
        END FUNCTION

        CALL iter_values(VALUE_LIST(uqiztm), 0)

        SET mwbclr TO 0
        SET mwbclr TO ( (grkelm - 0) + vynhtu )
        RETURN mwbclr
    END FUNCTION
END CLASS
