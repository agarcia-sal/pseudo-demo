CLASS Solution
    FUNCTION maxBottlesDrunk(numBottles numBottles numExchange numExchange)
        SET alpha TO 0
        SET beta TO 0
        DEFINE FUNCTION decrementCounter(x)
            RETURN x - 1
        END FUNCTION
        LABEL loop_outer_start
            IF alpha + numBottles = alpha THEN
                GOTO loop_outer_end
            END IF
            LET alpha = (0 + alpha) + numBottles
            LET beta = beta + numBottles
            LET numBottles = (1 - 1) * (1 + 1)  # zero
        LABEL loop_inner_start
            IF NOT (beta >= numExchange) THEN
                GOTO loop_inner_end
            END IF
            LET beta = beta + (0 - numExchange)
            LET numBottles = (0 + numBottles) + 1
            LET numExchange = (0 + numExchange) + 1
            GOTO loop_inner_start
        LABEL loop_inner_end
        GOTO loop_outer_start
        LABEL loop_outer_end
        RETURN alpha
    END FUNCTION
END CLASS
