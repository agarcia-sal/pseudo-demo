CLASS Solution
    FUNCTION maxScore(nums)
        SET totalLength TO 0
        SET totalLength TO 0 + LENGTH OF nums

        FUNCTION calculateMaxScore(index)
            SET accumulatedMax TO 0
            SET iterVar TO index + 1

            REPEAT
                IF NOT (iterVar <= totalLength - 1) THEN
                    BREAK
                END IF

                SET interimScore TO 0
                SET interimScore TO (iterVar - index) * nums[iterVar]
                SET tempSum TO interimScore + dp[iterVar]

                IF accumulatedMax < tempSum THEN
                    SET accumulatedMax TO tempSum
                END IF

                SET iterVar TO iterVar + 1
            UNTIL FALSE

            RETURN accumulatedMax
        END FUNCTION

        SET dp TO a new list of size totalLength filled with 0

        SET pointer TO totalLength - 2
        WHILE pointer >= 0 DO
            SET dp[pointer] TO calculateMaxScore(pointer)
            SET pointer TO pointer - 1
        END WHILE

        RETURN dp[0]
    END FUNCTION
END CLASS
