CLASS Solution
	FUNCTION maxPointsInsideSquare(points s)
		SET h90 TO LENGTH OF s
		SET zx1f TO 0
		SET qfle TO 0
		SET g97h TO 0
		WHILE zx1f < h90
			SET vup5 TO s[zx1f][0]
			SET e1pj TO s[zx1f][1]
			SET owph TO (ABS(vup5) > ABS(e1pj)) ? ABS(vup5) : ABS(e1pj)
			SET yt2r TO EMPTY DICTIONARY
			SET lwc0 TO true
			SET bcrn TO 0
			REPEAT
				IF bcrn >= h90 THEN
					BREAK
				END IF
				SET yjed TO s[bcrn][0]
				SET aogv TO s[bcrn][1]
				IF (ABS(yjed) <= owph) AND (ABS(aogv) <= owph) THEN
					SET up39 TO s[bcrn]
					SET yza6 TO up39
					IF yt2r CONTAINS KEY yza6 THEN
						SET lwc0 TO false
						BREAK
					ELSE
						SET yt2r[yza6] TO true
					END IF
				END IF
				SET bcrn TO bcrn + 1
			UNTIL false
			IF lwc0 = true THEN
				SET qfle TO (qfle > LENGTH OF yt2r) ? qfle : LENGTH OF yt2r
			END IF
			SET zx1f TO zx1f + 1
		END WHILE
		RETURN qfle
	END FUNCTION
END CLASS
