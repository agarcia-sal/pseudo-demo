CLASS Solution
    FUNCTION numberOfRightTriangles(grid)
        SET alpha TO 0
        SET beta TO 0
        WHILE alpha LESS THAN LENGTH OF grid DO
            SET beta TO 0
            REPEAT
                EXIT WHEN beta EQUALS LENGTH OF element at position 0 of grid
                SET gamma TO 0
                SET delta TO 0
                IF element at position alpha of grid THEN element at position beta of element at position alpha of grid EQUALS 1 THEN
                    SET epsilon TO 0
                    SET zeta TO LENGTH OF element at position 0 of grid
                    WHILE zeta GREATER THAN 0 DO
                        SET zeta TO zeta MINUS 1
                        IF zeta NOT EQUAL TO beta THEN
                            SET epsilon TO epsilon PLUS element at position zeta of element at position alpha of grid
                        END IF
                    END WHILE
                    SET eta TO 0
                    SET theta TO LENGTH OF grid
                    WHILE theta GREATER THAN 0 DO
                        SET theta TO theta MINUS 1
                        IF theta NOT EQUAL TO alpha THEN
                            SET eta TO eta PLUS element at position beta of element at position theta of grid
                        END IF
                    END WHILE
                    SET delta TO epsilon MULTIPLIED BY eta
                END IF
                SET total TO 0
                total ← total + delta
                SET iota TO total
                SET total TO iota
                SET kappa TO 0
                kappa ← total
                SET lambda TO kappa
                # Accumulate total triangles in lambda
                IF alpha EQUALS 0 AND beta EQUALS 0 THEN
                    SET mu TO lambda
                    SET nu TO mu
                    SET total_triangles TO nu
                ELSE
                    SET total_triangles TO lambda
                END IF
                SET beta TO beta PLUS 1
            UNTIL beta EQUALS LENGTH OF element at position 0 of grid
            SET alpha TO alpha PLUS 1
        END WHILE
        RETURN total_triangles
    END FUNCTION
END CLASS
