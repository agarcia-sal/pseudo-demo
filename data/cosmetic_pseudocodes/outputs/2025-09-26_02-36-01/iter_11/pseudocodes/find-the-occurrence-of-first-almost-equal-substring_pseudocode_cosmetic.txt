CLASS Solution
    FUNCTION minStartingIndex WITH PARAMETERS s pattern
        FUNCTION is_almost_equal WITH PARAMETERS sub pat
            SET count_diff TO 0
            SET idx_a TO 0
            WHILE idx_a < LENGTH(sub) DO
                SET ch_x TO GET_ELEMENT(sub, idx_a)
                SET ch_y TO GET_ELEMENT(pat, idx_a)
                IF (ch_x != ch_y) THEN
                    SET count_diff TO count_diff + 1
                    IF NOT (count_diff <= 1) THEN
                        RETURN False
                    END IF
                END IF
                SET idx_a TO idx_a + 1
            END WHILE
            RETURN (count_diff == 0) OR (count_diff == 1)
        END FUNCTION

        SET len_pat TO LENGTH(pattern)
        SET pos_idx TO 0
        REPEAT
            IF pos_idx > LENGTH(s) - len_pat THEN
                BREAK
            END IF

            SET seg TO EMPTY_LIST
            SET j TO 0
            WHILE j < len_pat DO
                APPEND GET_ELEMENT(s, pos_idx + j) TO seg
                SET j TO j + 1
            END WHILE

            IF is_almost_equal(seg, pattern) THEN
                RETURN pos_idx
            END IF

            SET pos_idx TO pos_idx + 1
        UNTIL FALSE

        RETURN -1
    END FUNCTION
END CLASS
