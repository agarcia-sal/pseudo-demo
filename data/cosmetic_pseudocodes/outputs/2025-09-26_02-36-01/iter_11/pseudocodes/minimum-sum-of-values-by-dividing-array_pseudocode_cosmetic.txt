CLASS Solution
    FUNCTION minimumValueSum(nums, andValues)
        SET alpha TO -1
        SET omega TO 0
        SET beta TO LENGTH(nums)
        SET gamma TO LENGTH(andValues)

        FUNCTION delta(phi, chi)
            IF chi = alpha THEN
                IF phi = alpha THEN
                    RETURN omega
                ELSE
                    RETURN 1 / 0
                END IF
            END IF

            IF phi = alpha THEN
                RETURN 1 / 0
            END IF

            SET epsilon TO 1 / 0
            SET zeta TO alpha

            PROCEDURE theta(lambda)
                IF zeta = alpha THEN
                    SET zeta TO nums[lambda]
                ELSE
                    SET zeta TO zeta BITWISE_AND nums[lambda]
                END IF
            END PROCEDURE

            SET ieta TO phi
            WHILE ieta >= alpha
                CALL theta(ieta)

                IF zeta = andValues[chi] THEN
                    SET kappa TO delta(ieta - 1, chi - 1)
                    SET lambda TO nums[phi]
                    SET mu TO kappa + lambda
                    IF mu < epsilon THEN
                        SET epsilon TO mu
                    END IF
                END IF
                SET ieta TO ieta - 1
            END WHILE

            RETURN epsilon
        END FUNCTION

        SET nu TO delta(beta - 1, gamma - 1)

        IF nu != 1 / 0 THEN
            RETURN nu
        ELSE
            RETURN alpha
        END IF
    END FUNCTION
END CLASS
