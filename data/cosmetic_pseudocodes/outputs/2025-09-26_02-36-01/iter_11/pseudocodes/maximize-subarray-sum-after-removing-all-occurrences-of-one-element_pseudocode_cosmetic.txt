CLASS Solution
    FUNCTION maxSubarraySum(nums)
        FUNCTION kadane(arr)
            SET beta_1 TO element at position one of arr
            SET gamma_2 TO element at position one of arr
            SET delta_3 TO 2
            WHILE delta_3 LESS THAN OR EQUAL TO length of arr
                LET omega_7 BE element at position delta_3 of arr
                IF omega_7 GREATER THAN (beta_1 PLUS omega_7) EQUAL TO false
                    SET beta_1 TO (beta_1 PLUS omega_7)
                ELSE
                    SET beta_1 TO omega_7
                END IF
                IF gamma_2 LESS THAN beta_1
                    SET gamma_2 TO beta_1
                END IF
                SET delta_3 TO delta_3 PLUS 1
            END WHILE
            RETURN gamma_2
        END FUNCTION

        SET alpha_9 TO kadane(nums)
        SET sigma_4 TO the set of unique elements of nums
        SET theta_8 TO 1
        SET zeta_5 TO size of sigma_4
        REPEAT
            IF theta_8 GREATER THAN zeta_5
                BREAK
            END IF
            LET chi_6 BE element at position theta_8 of sigma_4
            SET psi_10 TO empty list
            SET lambda_11 TO 1
            WHILE lambda_11 LESS THAN OR EQUAL TO length of nums
                LET phi_12 BE element at position lambda_11 of nums
                IF phi_12 EQUALS chi_6 EQUAL TO false
                    APPEND phi_12 TO psi_10
                END IF
                SET lambda_11 TO lambda_11 PLUS 1
            END WHILE
            IF length of psi_10 GREATER THAN 0
                LET xi_13 BE kadane(psi_10)
                IF alpha_9 LESS THAN xi_13
                    SET alpha_9 TO xi_13
                END IF
            END IF
            SET theta_8 TO theta_8 PLUS 1
        UNTIL false
        RETURN alpha_9
    END FUNCTION
END CLASS
