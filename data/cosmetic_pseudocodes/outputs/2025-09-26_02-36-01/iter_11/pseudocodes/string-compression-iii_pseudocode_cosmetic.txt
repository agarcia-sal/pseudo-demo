CLASS Solution
    FUNCTION compressedString(word)
        SET vne TO 0
        FUNCTION rptCount(x, y)
            SET zrd TO 0
            LABEL loopTop
            IF (y < LENGTH(word)) AND (word[y] = x) AND (zrd < 9)
                SET zrd TO zrd + 1
                SET y TO y + 1
                GOTO loopTop
            END IF
            RETURN (zrd, y)
        END FUNCTION

        SET hsm TO []
        LABEL outerLoop
        IF vne < LENGTH(word)
            SET mep TO word[vne]
            SET (jxf, vne) TO rptCount(mep, vne)
            SET mid TO STRING(jxf) + mep
            PUSH mid TO hsm
            GOTO outerLoop
        END IF
        SET zlg TO ""
        FOR qm FROM 0 TO LENGTH(hsm)-1
            SET zlg TO zlg + hsm[qm]
        END FOR
        RETURN zlg
    END FUNCTION
END CLASS
