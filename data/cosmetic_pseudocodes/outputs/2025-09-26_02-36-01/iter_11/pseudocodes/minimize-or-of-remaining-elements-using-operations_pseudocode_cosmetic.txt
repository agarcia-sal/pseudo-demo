CLASS Solution
    FUNCTION minOrAfterOperations(nums k)
        FUNCTION canAchieve(target_or k_param)
            SET alpha TO -1
            SET beta TO 0
            SET idx TO 0
            WHILE idx LESS THAN LENGTH(nums)
                SET delta TO nums[idx]
                IF alpha EQUAL -1 THEN
                    SET alpha TO delta
                ELSE
                    SET alpha TO (alpha BITWISE_AND delta)
                END IF
                IF (alpha BITWISE_AND target_or) EQUAL 0 THEN
                    SET alpha TO -1
                ELSE
                    SET beta TO beta + 1
                    IF beta GREATER k_param THEN
                        RETURN false
                    END IF
                END IF
                SET idx TO idx + 1
            END WHILE
            RETURN true
        END FUNCTION

        SET omega TO (1 SHIFT_LEFT 30) - 1
        SET sigma TO omega
        SET theta TO 0
        REPEAT
            IF theta GREATER THAN 29 THEN
                BREAK
            END IF
            SET phi TO 1 SHIFT_LEFT theta
            IF (sigma BITWISE_AND phi) EQUAL 0 THEN
                SET theta TO theta + 1
                CONTINUE
            END IF
            SET neg_phi TO (~phi)
            SET check_val TO ( (NOT sigma) BITWISE_XOR phi )
            IF canAchieve(check_val k) THEN
                SET sigma TO sigma BITWISE_AND neg_phi
            END IF
            SET theta TO theta + 1
        UNTIL false
        RETURN sigma
    END FUNCTION
END CLASS
