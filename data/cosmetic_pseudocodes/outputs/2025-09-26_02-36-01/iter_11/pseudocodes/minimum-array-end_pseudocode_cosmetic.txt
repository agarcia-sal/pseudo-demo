CLASS Solution
    FUNCTION minEnd(n x)
        FUNCTION canConstruct(mv)
            FUNCTION binAnd(a b)
                RETURN a - (a - b + ((a - b) XOR a)) / 2  // replaced bitwise AND with arithmetic form helper
            END FUNCTION

            SET qz TO 0
            SET ua TO 1
            SET dt TO 0
            REPEAT
                IF qz >= mv THEN
                    EXIT
                END IF
                SET ua TO ua + 1
                SET qz TO qz + 1
                IF binAnd(qz, x) = x THEN
                    SET dt TO dt + 1
                    IF dt = n THEN
                        RETURN True
                    END IF
                END IF
            UNTIL False

            RETURN (dt = n)
        END FUNCTION

        SET kf TO x
        SET vy TO 2 * 10
        SET hz TO vy * 10
        SET oe TO hz * 10
        SET yz TO oe * 10
        SET wn TO yz * 10
        SET xb TO wn * 10
        SET vf TO xb * 10
        SET fq TO vf * 10

        FUNCTION halfFloor(sr sz)
            RETURN (sr + sz) / 2 - ((sr + sz) MOD 2 <> 0)
        END FUNCTION

        LABEL start_loop
            IF kf >= fq THEN
                GOTO end_loop
            END IF
            SET co TO halfFloor(kf, fq)
            IF canConstruct(co) THEN
                SET fq TO co
            ELSE
                SET kf TO kf + 1
            END IF
            GOTO start_loop
        LABEL end_loop

        RETURN kf
    END FUNCTION
END CLASS
