CLASS Solution
    FUNCTION largestPalindrome(n k)
        IF n EQUALS 1 THEN
            SET m TO 9
            WHILE m GREATER OR EQUAL TO 1 DO
                IF m MOD k EQUALS 0 THEN
                    RETURN STRING(m)
                END IF
                SET m TO m - 1
            END WHILE
            RETURN "0"
        END IF

        FUNCTION makeFull(value)
            SET s TO STRING(value)
            SET t TO ""
            IF n MOD 2 EQUALS 0 THEN
                SET t TO s + REVERSE(s)
            ELSE
                SET t TO s + REVERSE(SUBSTRING(s, 0, LENGTH(s) - 1))
            END IF
            RETURN INTEGER(t)
        END FUNCTION

        SET limitValue TO INTEGER(CONCAT(REPEAT("9", (n + 1) / 2)))

        FUNCTION decreaseAndCheck(num)
            IF num LESS OR EQUAL 0 THEN
                RETURN "0"
            ELSE
                SET candidate TO makeFull(num)
                IF candidate MOD k EQUALS 0 THEN
                    RETURN STRING(candidate)
                ELSE
                    RETURN decreaseAndCheck(num - 1)
                END IF
            END IF
        END FUNCTION

        RETURN decreaseAndCheck(limitValue)
    END FUNCTION
END CLASS
