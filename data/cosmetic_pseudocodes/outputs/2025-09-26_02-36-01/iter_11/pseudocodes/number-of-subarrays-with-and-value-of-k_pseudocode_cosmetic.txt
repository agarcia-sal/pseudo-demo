CLASS Solution
    FUNCTION countSubarrays(nums parameter LIST OF INTEGER k parameter INTEGER) RETURNS INTEGER
        SET u42 TO 0
        SET yq7 TO LENGTH(nums)
        LABEL outer_loop:
        SET k83 TO 0
        WHILE k83 < yq7
            SET t19 TO GET_ELEMENT_AT(nums, k83)
            SET p04 TO k83
            LABEL inner_loop:
            WHILE p04 < yq7
                SET t19 TO (t19 BITWISE_AND GET_ELEMENT_AT(nums, p04))
                IF (t19 = k) THEN
                    SET u42 TO (u42 + 1)
                END IF
                IF (t19 < k) THEN
                    BREAK inner_loop
                END IF
                SET p04 TO (p04 + 1)
            END WHILE
            SET k83 TO (k83 + 1)
        END WHILE
        RETURN u42
    END FUNCTION
END CLASS
