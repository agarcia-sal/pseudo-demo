CLASS Solution
    FUNCTION numberOfAlternatingGroups(colors k)
        IF NOT (k GREATER THAN 2) THEN
            RETURN 0
        END IF

        SET w TO LENGTH(colors)
        SET v TO EMPTY LIST
        SET u TO 0
        WHILE u LESS THAN w DO
            ADD colors[u] TO v
            SET u TO u PLUS 1
        END WHILE

        SET z TO 0
        SET s TO 0
        REPEAT
            ADD colors[s] TO v
            SET s TO s PLUS 1
        UNTIL s EQUALS k MINUS 1

        SET m TO 0
        WHILE m LESS THAN w DO
            SET p TO 1
            SET q TO TRUE
            WHILE p LESS THAN k DO
                IF v[m PLUS p] EQUALS v[m PLUS p MINUS 1] THEN
                    SET q TO FALSE
                    EXIT WHILE
                END IF
                SET p TO p PLUS 1
            END WHILE
            IF q THEN
                SET z TO z PLUS 1
            END IF
            SET m TO m PLUS 1
        END WHILE

        RETURN z
    END FUNCTION
END CLASS
