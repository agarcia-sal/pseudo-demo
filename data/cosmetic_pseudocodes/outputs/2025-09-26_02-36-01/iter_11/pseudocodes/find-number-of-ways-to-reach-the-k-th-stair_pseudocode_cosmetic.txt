CLASS Solution
    FUNCTION waysToReachStair WITH PARAMETER k
        FUNCTION dfs WITH PARAMETERS x y z
            IF (k + 1) < x THEN
                RETURN 0
            END IF
            SET omega TO (x = k) * 1
            IF (0 < x) AND (0 = y) THEN
                SET phi TO dfs(x - 1, 1, z)
                SET omega TO omega + phi
            END IF
            SET psi TO dfs(x + (2 * z), 0, z + 1)
            SET omega TO omega + psi
            RETURN omega
        END FUNCTION
        SET result TO dfs(1, 0, 0)
        RETURN result
    END FUNCTION
END CLASS
