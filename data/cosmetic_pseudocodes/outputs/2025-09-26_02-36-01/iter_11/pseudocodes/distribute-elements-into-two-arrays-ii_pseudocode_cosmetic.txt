CLASS Solution
    FUNCTION resultArray(nums)
        SET vawlyj TO new list containing nums[0]
        SET xlotf TO new list containing nums[1]
        SET ziqep TO new list containing nums[0]
        SET uhten TO new list containing nums[1]

        FUNCTION lornvp(agtej, grmof)
            SET mgphs TO 0
            SET tcehw TO LENGTH OF agtej
            WHILE mgphs LESS THAN tcehw
                SET xhdup TO FLOOR((mgphs + tcehw) / 2)
                IF grmof LESS THAN agtej[xhdup]
                    SET tcehw TO xhdup
                ELSE
                    SET mgphs TO xhdup + 1
                END IF
            END WHILE
            RETURN LENGTH OF agtej - mgphs
        END FUNCTION
        
        SET kbvql TO 2
        WHILE kbvql LESS THAN LENGTH OF nums
            SET uyroj TO nums[kbvql]
            SET npfox TO lornvp(ziqep, uyroj)
            SET yxplj TO lornvp(uhten, uyroj)

            IF npfox GREATER THAN yxplj
                SET vawlyj_new TO vawlyj
                SET vawlyj_new APPEND uyroj
                SET vawlyj TO vawlyj_new

                SET ftyus TO LENGTH OF ziqep
                SET hqpza TO ftyus
                WHILE 0 LESS THAN hqpza
                    SET midr TO FLOOR((ftyus + hqpza) / 2)
                    IF uyroj LESS THAN ziqep[midr]
                        SET hqpza TO midr
                    ELSE
                        SET ftyus TO midr + 1
                    END IF
                END WHILE
                INSERT uyroj AT POSITION ftyus IN ziqep
            ELSE IF npfox LESS THAN yxplj
                xlotf APPEND uyroj
                
                SET pvwus TO LENGTH OF uhten
                SET btilj TO pvwus
                WHILE 0 LESS THAN btilj
                    SET qrhek TO FLOOR((pvwus + btilj) / 2)
                    IF uyroj LESS THAN uhten[qrhek]
                        SET btilj TO qrhek
                    ELSE
                        SET pvwus TO qrhek + 1
                    END IF
                END WHILE
                INSERT uyroj AT POSITION pvwus IN uhten
            ELSE
                IF LENGTH OF vawlyj LESS THAN OR EQUAL TO LENGTH OF xlotf
                    SET vawlyj_tmp TO vawlyj
                    vawlyj_tmp APPEND uyroj
                    SET vawlyj TO vawlyj_tmp

                    SET sopzm TO LENGTH OF ziqep
                    SET helfd TO sopzm
                    WHILE 0 LESS THAN helfd
                        SET qcxfj TO FLOOR((sopzm + helfd) / 2)
                        IF uyroj LESS THAN ziqep[qcxfj]
                            SET helfd TO qcxfj
                        ELSE
                            SET sopzm TO qcxfj + 1
                        END IF
                    END WHILE
                    INSERT uyroj AT POSITION sopzm IN ziqep
                ELSE
                    xlotf APPEND uyroj

                    SET tlboz TO LENGTH OF uhten
                    SET smzin TO tlboz
                    WHILE 0 LESS THAN smzin
                        SET arbkf TO FLOOR((tlboz + smzin) / 2)
                        IF uyroj LESS THAN uhten[arbkf]
                            SET smzin TO arbkf
                        ELSE
                            SET tlboz TO arbkf + 1
                        END IF
                    END WHILE
                    INSERT uyroj AT POSITION tlboz IN uhten
                END IF
            END IF
            SET kbvql TO kbvql + 1
        END WHILE

        RETURN vawlyj + xlotf
    END FUNCTION
END CLASS
