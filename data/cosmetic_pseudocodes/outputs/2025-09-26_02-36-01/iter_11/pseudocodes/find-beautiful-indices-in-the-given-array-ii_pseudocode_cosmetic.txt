CLASS Solution
    FUNCTION beautifulIndices(s parameter a parameter b parameter k)
        PROCEDURE cmp_substring(t parameter, x parameter) RETURNS list
            SET result TO new empty list
            SET m TO LENGTH OF t
            SET n TO LENGTH OF x
            SET counter TO zero
            WHILE counter NOT EQUAL TO m - n + 1
                IF SUBSTRING of t from counter TO counter + n - 1 EQUALS x THEN
                    APPEND counter TO result
                END IF
                SET counter TO counter + 1
            END WHILE
            RETURN result
        END PROCEDURE

        SET zyx TO cmp_substring(s, a)
        SET wvu TO cmp_substring(s, b)
        SET pqr TO new empty list

        PROCEDURE abs_diff(u parameter, v parameter) RETURNS integer
            IF u GREATER THAN OR EQUAL TO v THEN
                RETURN u - v
            ELSE
                RETURN v - u
            END IF
        END PROCEDURE

        SET alpha TO 0
        SET beta TO 0

        REPEAT
            IF alpha GREATER OR EQUAL TO LENGTH OF zyx OR beta GREATER OR EQUAL TO LENGTH OF wvu THEN
                BREAK
            END IF
            SET delta TO abs_diff(zyx[alpha], wvu[beta])
            IF delta LESS THAN OR EQUAL TO k THEN
                APPEND zyx[alpha] TO pqr
                SET alpha TO alpha + 1
            ELSE
                IF zyx[alpha] LESS THAN wvu[beta] THEN
                    SET alpha TO alpha + 1
                ELSE
                    SET beta TO beta + 1
                END IF
            END IF
        UNTIL FALSE
        
        RETURN pqr
    END FUNCTION
END CLASS
