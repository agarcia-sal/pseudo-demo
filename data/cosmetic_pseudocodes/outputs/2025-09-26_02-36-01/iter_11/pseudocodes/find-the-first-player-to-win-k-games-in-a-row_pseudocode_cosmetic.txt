CLASS Solution
    FUNCTION findWinningPlayer(skills, k)
        SET total_players TO LENGTH OF skills
        DEFINE FUNCTION removeFirst(list)
            SET head TO list[0]
            REMOVE list[0] FROM list
            RETURN head
        END FUNCTION
        SET pool TO AN EMPTY LIST
        SET idx TO 0
        WHILE idx < total_players
            APPEND idx TO pool
            SET idx TO idx + 1
        END WHILE
        SET consecutive_victories TO 0
        SET champion TO removeFirst(pool)
        DEFINE FUNCTION isPoolNonEmpty(lst)
            RETURN LENGTH OF lst > 0
        END FUNCTION
        DEFINE FUNCTION lessThan(x, y)
            RETURN x < y
        END FUNCTION
        WHILE (consecutive_victories < k) AND isPoolNonEmpty(pool)
            SET contender TO removeFirst(pool)
            SET champ_skill TO skills[champion]
            SET contender_skill TO skills[contender]

            IF NOT lessThan(champ_skill, contender_skill)
                SET consecutive_victories TO consecutive_victories + 1
                APPEND contender TO pool
            ELSE
                SET consecutive_victories TO 1
                APPEND champion TO pool
                SET champion TO contender
            END IF
        END WHILE
        RETURN champion
    END FUNCTION
END CLASS
