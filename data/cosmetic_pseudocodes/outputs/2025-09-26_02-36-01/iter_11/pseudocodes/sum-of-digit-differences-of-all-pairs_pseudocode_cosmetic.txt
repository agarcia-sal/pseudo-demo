CLASS Solution
    FUNCTION sumDigitDifferences(nums)
        FUNCTION delta_digits(a, b)
            SET acc TO 0
            SET pos TO 0
            REPEAT
                IF pos >= LENGTH(a) OR pos >= LENGTH(b) THEN
                    BREAK
                END IF
                SET ca TO CHAR_AT(a, pos)
                SET cb TO CHAR_AT(b, pos)
                IF NOT (ca = cb) THEN
                    SET acc TO acc + 1
                END IF
                SET pos TO pos + 1
            UNTIL FALSE
            RETURN acc
        END FUNCTION

        SET aggregate TO 0
        SET limit TO LENGTH(nums)
        SET outer TO 0
        WHILE outer < limit
            SET inner TO outer + 1
            WHILE inner < limit
                SET pair_diff TO delta_digits(nums[outer], nums[inner])
                SET aggregate TO aggregate + pair_diff
                SET inner TO inner + 1
            END WHILE
            SET outer TO outer + 1
        END WHILE
        RETURN aggregate
    END FUNCTION
END CLASS
