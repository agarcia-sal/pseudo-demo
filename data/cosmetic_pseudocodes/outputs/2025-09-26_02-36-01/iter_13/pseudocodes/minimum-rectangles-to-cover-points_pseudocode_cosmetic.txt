CLASS Solution
    FUNCTION minRectanglesToCoverPoints(points, w)
        FUNCTION compareFirstElement(a, b)
            IF a[0] LESS THAN b[0] THEN
                RETURN true
            ELSE
                RETURN false
            END IF
        END FUNCTION

        FUNCTION sortPoints(arr)
            FUNCTION quicksort(low, high)
                IF low GREATER OR EQUAL high THEN
                    RETURN
                END IF
                SET pivotIndex TO low
                SET pivotValue TO arr[high]
                SET i TO low
                FUNCTION swap(i, j)
                    SET temp TO arr[i]
                    SET arr[i] TO arr[j]
                    SET arr[j] TO temp
                END FUNCTION
                FOR j FROM low TO high - 1
                    IF compareFirstElement(arr[j], pivotValue) THEN
                        swap(i, j)
                        SET i TO i + 1
                    END IF
                END FOR
                swap(i, high)
                quicksort(low, i - 1)
                quicksort(i + 1, high)
            END FUNCTION

            quicksort(0, LENGTH(points) - 1)
        END FUNCTION

        CALL sortPoints(points)

        SET counter TO 0
        SET boundary TO 0
        FUNCTION process(index, limit)
            IF index GREATER OR EQUAL limit THEN
                RETURN
            END IF
            SET p TO points[index]
            IF p[0] GREATER THAN boundary
                SET boundary TO p[0] + w
                SET counter TO counter + 1
            END IF
            process(index + 1, limit)
        END FUNCTION

        SET boundary TO -1
        CALL process(0, LENGTH(points))

        RETURN counter
    END FUNCTION
END CLASS
