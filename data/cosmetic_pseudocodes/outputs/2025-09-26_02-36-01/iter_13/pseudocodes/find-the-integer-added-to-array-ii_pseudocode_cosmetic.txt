CLASS Solution
    FUNCTION minimumAddedInteger(nums1 RETURNS List OF integer, nums2 RETURNS List OF integer) RETURNS integer
        PROCEDURE qdrmwt(l RETURNS List OF integer)
            INTEGER a, b, c, d, e
            SET a TO LENGTH OF l
            e ← 1
            WHILE e < a
                b ← 0
                WHILE b < a - e
                    IF l[b] > l[b + 1] THEN
                        c ← l[b]
                        l[b] ← l[b + 1]
                        l[b + 1] ← c
                    END IF
                    b ← b + 1
                END WHILE
                e ← e + 1
            END WHILE
        END PROCEDURE

        CALL qdrmwt(nums1)
        CALL qdrmwt(nums2)

        FUNCTION jhsrmb(p RETURNS List OF integer, q RETURNS List OF integer, r RETURNS integer, s RETURNS integer) RETURNS List OF integer
            VARIABLE w RETURNS List OF integer ← EMPTY LIST
            PROCEDURE vloah(i RETURNS integer, j RETURNS integer)
                IF i >= j THEN RETURN END IF
                APPEND p[i] TO w
                CALL vloah(i + 1, j)
            END PROCEDURE
            CALL vloah(0, r)
            CALL vloah(r + 1, s)
            CALL vloah(s + 1, LENGTH OF p)
            RETURN w
        END FUNCTION

        FUNCTION hfjrso(v RETURNS List OF integer, u RETURNS List OF integer, t RETURNS integer) RETURNS boolean
            VARIABLE y RETURNS boolean ← TRUE
            VARIABLE z RETURNS integer ← 0
            WHILE z < LENGTH OF u
                IF v[z] + t ≠ u[z] THEN
                    y ← FALSE
                    EXIT WHILE
                END IF
                z ← z + 1
            END WHILE
            RETURN y
        END FUNCTION

        INTEGER o, m, n
        SET o TO 0
        WHILE o < LENGTH OF nums1
            SET m TO o + 1
            REPEAT
                IF m ≥ LENGTH OF nums1 THEN
                    BREAK
                END IF
                VARIABLE lnum RETURNS List OF integer ← jhsrmb(nums1, nums1, o, m)
                VARIABLE diff RETURNS integer ← nums2[0] - lnum[0]
                IF hfjrso(lnum, nums2, diff) THEN
                    RETURN diff
                END IF
                m ← m + 1
            UNTIL FALSE
            o ← o + 1
        END WHILE

        RETURN None
    END FUNCTION
END CLASS
