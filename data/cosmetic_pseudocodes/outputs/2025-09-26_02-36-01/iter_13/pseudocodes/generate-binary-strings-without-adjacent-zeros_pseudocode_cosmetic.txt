CLASS Solution  
  FUNCTION validStrings n  
    SET results TO empty list  
    FUNCTION recurse sequence  
      IF LENGTH(sequence) EQUALS n THEN  
        APPEND sequence TO results  
        RETURN  
      END IF  
      SET last_char TO sequence AT (LENGTH(sequence) MINUS 1)  
      IF last_char EQUALS "1" THEN  
        SET temp1 TO sequence CONCAT "0"  
        SET temp2 TO sequence CONCAT "1"  
        CALL recurse WITH temp1  
        CALL recurse WITH temp2  
      ELSE IF last_char EQUALS "0" THEN  
        CALL recurse WITH sequence CONCAT "1"  
      END IF  
    END FUNCTION  
    IF n EQUALS 1 THEN  
      RETURN LIST("0","1")  
    END IF  
    CALL recurse WITH "0"  
    CALL recurse WITH "1"  
    RETURN results  
  END FUNCTION  
END CLASS
