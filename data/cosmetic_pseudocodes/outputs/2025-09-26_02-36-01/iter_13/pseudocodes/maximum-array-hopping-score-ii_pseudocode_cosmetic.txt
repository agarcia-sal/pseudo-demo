CLASS Solution
	FUNCTION maxScore(nums)
		SET m TO LENGTH(nums)
		INITIALIZE arr TO ARRAY OF SIZE m WITH ALL ELEMENTS ZERO
		SET idx TO m - 2
		WHILE idx GREATER OR EQUAL TO 0
			SET curr_max TO 0
			DEFINE INNER_FUN(k)
				IF k EQUALS m THEN RETURN END IF
				SET val TO (k - idx) * nums[k]
				SET temp TO val + arr[k]
				IF curr_max LESS THAN temp THEN curr_max = temp END IF
				CALL INNER_FUN(k + 1)
			END
			CALL INNER_FUN(idx + 1)
			SET arr[idx] TO curr_max
			SET idx TO idx - 1
		END WHILE
		RETURN arr[0]
	END FUNCTION
END CLASS
