class Solution
function maxBottlesDrunk(numBottles numBottles numExchange numExchange)
    var accumulated int
    var scraps int
    accumulated ← 0
    scraps ← 0
    repeat
        if numBottles ≤ 0 then
            break
        end if
        accumulated ← accumulated + numBottles
        scraps ← scraps + numBottles
        numBottles ← 0
        while scraps ≥ numExchange do
            scraps ← scraps - numExchange
            numBottles ← numBottles + 1
            numExchange ← numExchange + 1
        end while
    until false
    return accumulated
end function
end class
