CLASS Solution
	 FUNCTION minimumOperations(nums, target)
		  SET m TO 0
		  SET p TO SIZE(nums)
		  SET m TO ABS((target[0]) - (nums[0]))
		  SET q TO 1
		  WHILE q < p
				SET r TO (target[q]) - (nums[q])
				SET s TO (target[q - 1]) - (nums[q - 1])
				IF (r * s) > 0 THEN
					 SET t TO ABS(r) - ABS(s)
					 IF t > 0 THEN
						  SET m TO m + t
					 END IF
				ELSE
					 SET m TO m + ABS(r)
				END IF
				SET q TO q + 1
		  END WHILE
		  RETURN m
	 END FUNCTION
END CLASS
