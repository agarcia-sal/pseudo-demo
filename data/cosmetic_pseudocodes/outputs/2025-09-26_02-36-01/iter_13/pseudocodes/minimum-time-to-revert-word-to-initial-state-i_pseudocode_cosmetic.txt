CLASS Solution
    FUNCTION minimumTimeToInitialState(word PARAMETER k)
        SET p TO 1
        SET q TO LENGTH(word)
        DEFINE FUNCTION equalSegments(s1, start1, s2, start2, length)
            SET r TO 0
            WHILE r LESS THAN length
                IF s1[start1 + r] NOT EQUAL TO s2[start2 + r] THEN
                    RETURN false
                END IF
                SET r TO r + 1
            END WHILE
            RETURN true
        END FUNCTION

        DEFINE FUNCTION conditionMet(current)
            SET threshold TO current * k
            IF threshold >= q THEN
                RETURN true
            ELSE
                RETURN equalSegments(word, threshold, word, 0, q - threshold)
            END IF
        END FUNCTION

        WHILE true
            IF conditionMet(p) THEN
                RETURN p
            END IF
            SET p TO p + 1
        END WHILE
    END FUNCTION
END CLASS
