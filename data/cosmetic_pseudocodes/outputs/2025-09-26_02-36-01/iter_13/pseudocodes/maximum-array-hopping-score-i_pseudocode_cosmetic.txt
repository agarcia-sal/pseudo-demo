    FUNCTION maxScore(nums)
        LET Z TO LENGTH(nums)
        LET dp TO AN ARRAY OF Z ZEROES
        LET k TO 1
        WHILE k < Z
            LET m TO 0
            LET v TO k - 1
            WHILE m < v
                LET a TO dp[m]
                LET b TO dp[k] 
                LET c TO (k - m) * nums[k]
                IF NOT (b >= a + c)
                    LET dp_k_new TO a + c
                    dp[k] = dp_k_new
                END IF
                m = m + 1
            END WHILE
            k = k + 1
        END WHILE
        RETURN dp[Z - 1]
    END FUNCTION
END CLASS
