CLASS Solution
    FUNCTION clearStars(s)
        FUNCTION eraseLastElement(arr)
            REPEAT
                IF LENGTH OF arr IS EQUAL TO 0 THEN
                    BREAK
                ELSE
                    SET tempList TO EMPTY LIST
                    SET idx TO 0
                    WHILE idx < LENGTH OF arr - 1 DO
                        APPEND arr[idx] TO tempList
                        SET idx TO idx + 1
                    END WHILE
                    SET arr TO tempList
                END IF
            UNTIL TRUE
            RETURN arr
        END FUNCTION

        SET accumulator TO EMPTY LIST
        SET position TO 0

        WHILE position < LENGTH OF s DO
            SET current_symbol TO GET_CHAR_AT(s, position)
            IF NOT (current_symbol = '*') THEN
                APPEND current_symbol TO accumulator
            ELSE
                SET accumulator TO eraseLastElement(accumulator)
            END IF
            SET position TO position + 1
        END WHILE

        FUNCTION concatList(chars)
            SET output TO ""
            SET counter TO 0
            REPEAT
                IF counter = LENGTH OF chars THEN
                    EXIT REPEAT
                ELSE
                    SET output TO output + chars[counter]
                    SET counter TO counter + 1
                END IF
            UNTIL FALSE
            RETURN output
        END FUNCTION

        RETURN concatList(accumulator)
    END FUNCTION
END CLASS
