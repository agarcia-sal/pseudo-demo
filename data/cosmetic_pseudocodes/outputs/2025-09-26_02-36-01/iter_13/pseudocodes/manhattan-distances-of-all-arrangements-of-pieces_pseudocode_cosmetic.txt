class Solution
   function distanceSum(m, n, k)
      define constant P = 1_000_000_007
      define function calc_contrib(x)
         define var acc = 0
         define var i = 1
         repeat
            acc = acc + i * i * i - i
            i = i + 1
         until i > x
         return acc / 6
      end function
      define function comb(a, b)
         if b > a then return 0 end if
         define var numerator = 1
         define var denominator = 1
         define var idx = 0
         while idx < b do
            numerator = numerator * (a - idx)
            denominator = denominator * (idx + 1)
            idx = idx + 1
         end while
         return numerator / denominator
      end function
      define var r_contrib = calc_contrib(m) * n * n
      define var c_contrib = calc_contrib(n) * m * m
      define var arr_contrib = comb(m * n - 2, k - 2)
      define var total = (r_contrib + c_contrib) * arr_contrib
      return total % P
   end function
end class
