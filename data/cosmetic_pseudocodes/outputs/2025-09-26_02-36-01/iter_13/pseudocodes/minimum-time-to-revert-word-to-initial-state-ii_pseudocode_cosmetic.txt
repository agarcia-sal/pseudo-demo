CLASS Solution
    FUNCTION minimumTimeToInitialState(word parameter p AND integer parameter q) RETURNS integer
        SET s TO LENGTH(p)
        SET u TO 1
        REPEAT
            SET v TO substringFromIndex(p, u * q)
            IF beginsWith(p, v) = TRUE THEN
                RETURN u
            END IF
            SET u TO u + 1
        UNTIL FALSE
    END FUNCTION

    FUNCTION substringFromIndex(word parameter x, integer parameter y) RETURNS word
        SET z TO ""
        SET a TO y
        WHILE a < LENGTH(x)
            SET z TO z + charAt(x, a)
            SET a TO a + 1
        END WHILE
        RETURN z
    END FUNCTION
