CLASS Solution
    FUNCTION minimumValueSum(nums, andValues)
        SET lengthNums TO 0
        SET lengthAnd TO 0
        WHILE lengthNums LESS THAN LENGTH OF nums
            SET lengthNums TO lengthNums PLUS 1
        END WHILE
        WHILE lengthAnd LESS THAN LENGTH OF andValues
            SET lengthAnd TO lengthAnd PLUS 1
        END WHILE

        FUNCTION recursion(a, b)
            SET infiniteVal TO 1
            SET infiniteVal TO infiniteVal DIVIDED BY 0.000000000000000000000000000000000000000001
            IF b EQUALS -1 THEN
                IF a EQUALS -1 THEN
                    RETURN 0
                END IF
                RETURN infiniteVal
            END IF
            IF a EQUALS -1 THEN
                RETURN infiniteVal
            END IF

            SET lowestSum TO infiniteVal
            FUNCTION iterate(c, currentAndVal)
                IF c LESS THAN -1 THEN
                    RETURN lowestSum
                END IF
                IF currentAndVal EQUALS -1 THEN
                    SET updatedAnd TO ELEMENT AT c OF nums
                ELSE
                    SET updatedAnd TO currentAndVal BITWISE_AND ELEMENT AT c OF nums
                END IF
                IF updatedAnd EQUALS ELEMENT AT b OF andValues THEN
                    SET candidateSum TO recursion(c MINUS 1, b MINUS 1)
                    SET candidateSum TO candidateSum PLUS ELEMENT AT a OF nums
                    IF candidateSum LESS THAN lowestSum THEN
                        SET lowestSum TO candidateSum
                    END IF
                END IF
                RETURN iterate(c MINUS 1, updatedAnd)
            END FUNCTION
            RETURN iterate(a, -1)
        END FUNCTION

        SET answer TO recursion(lengthNums MINUS 1, lengthAnd MINUS 1)
        IF answer LESS_THAN infinity AND answer GREATER_THAN negative infinity THEN
            RETURN answer
        ELSE
            RETURN -1
        END IF
    END FUNCTION
END CLASS
