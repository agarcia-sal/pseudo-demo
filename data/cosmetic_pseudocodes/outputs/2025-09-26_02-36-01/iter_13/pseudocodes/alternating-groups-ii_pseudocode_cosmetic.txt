CLASS Solution
    FUNCTION numberOfAlternatingGroups(colors k)
        SET m TO 0
        SET length_colors TO LENGTH colors
        IF NOT (k GREATER OR EQUAL 3) THEN
            RETURN m
        END IF

        SET extended_seq TO CONCAT colors SUBSEQ colors 0 (k - 1)

        SET idx TO 0
        WHILE idx LESS THAN length_colors
            SET alternate_flag TO true
            SET offset TO 1
            REPEAT
                IF offset GREATER OR EQUAL k THEN
                    BREAK
                END IF
                IF extended_seq[idx + offset] EQUALS extended_seq[idx + offset - 1] THEN
                    SET alternate_flag TO false
                    BREAK
                END IF
                SET offset TO offset + 1
            UNTIL FALSE
            IF alternate_flag THEN
                SET m TO m + 1
            END IF
            SET idx TO idx + 1
        END WHILE

        RETURN m
    END FUNCTION
END CLASS
