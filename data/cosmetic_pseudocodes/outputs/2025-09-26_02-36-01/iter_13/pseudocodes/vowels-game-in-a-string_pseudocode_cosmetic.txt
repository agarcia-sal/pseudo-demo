cLass Solution
{
	Function doesAliceWin(s)
		Define vowelsSet = CreateSetWithElements('a','e','i','o','u')
		Define oddSegmentCounter = 0
		Define countCurrentVowels = 0

		Proc countVowels(index)
			If index > StringLength(s)
				Return 0
			EndIf
			Var ch = GetCharAt(s,index)
			Var increment = Contains(vowelsSet,ch) ? 1 : 0
			Return increment + countVowels(index + 1)
		EndProc
		
		countCurrentVowels = countVowels(1)

		Define flagInOddSegment = false
		Define pos = 1

		Proc iterateChars(i)
			If i > StringLength(s)
				Return
			EndIf
			Var currentChar = CharAt(s,i)
			If Contains(vowelsSet,currentChar)
				flagInOddSegment = Not flagInOddSegment
			EndIf
			If flagInOddSegment == false And Mod(countCurrentVowels,2) == 1
				oddSegmentCounter = oddSegmentCounter + 1
				countCurrentVowels = 0
			ElseIf flagInOddSegment == true
				countCurrentVowels = countCurrentVowels + 1
			EndIf
			iterateChars(i + 1)
		EndProc

		iterateChars(1)

		If flagInOddSegment == true And Mod(countCurrentVowels,2) == 1 Then
			oddSegmentCounter = oddSegmentCounter + 1
		EndIf

		Var resultFlag = false
		If Mod(oddSegmentCounter,2) == 1
			resultFlag = true
		EndIf
		Return resultFlag
	EndFunction
}
