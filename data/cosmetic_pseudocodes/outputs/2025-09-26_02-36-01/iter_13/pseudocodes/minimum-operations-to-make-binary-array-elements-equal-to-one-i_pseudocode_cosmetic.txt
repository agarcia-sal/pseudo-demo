CLASS Solution
	FUNCTION minOperations(nums)
		SET count TO 0
		SET length TO LENGTH OF nums
		SET index TO 0
		REPEAT
			IF index GREATER THAN length MINUS 3 THEN
				EXIT REPEAT
			END IF
			IF nums[index] EQUALS 0 THEN
				DEFINE invert AS FUNCTION(x)
					RETURN 1 MINUS x
				END FUNCTION
				SET nums[index] TO invert(nums[index])
				SET nums[index PLUS 1] TO invert(nums[index PLUS 1])
				SET nums[index PLUS 2] TO invert(nums[index PLUS 2])
				SET count TO count PLUS 1
			END IF
			SET index TO index PLUS 1
		UNTIL FALSE
		IF nums[length MINUS 1] EQUALS 0 OR nums[length MINUS 2] EQUALS 0 THEN
			RETURN -1
		END IF
		RETURN count
	END FUNCTION
END CLASS
