CLASS Solution
    FUNCTION valueAfterKSeconds(n k)
        SET MODULUS TO 10 * 10 * 10 * 10 * 10 * 10 * 10 * 10 * 10 * 10 + 7
        DEFINE FUNCTION modAdd(x y)
            SET sum TO x + y
            IF sum >= MODULUS THEN
                SET sum TO sum - MODULUS
            END IF
            RETURN sum
        END FUNCTION
        SET arr TO new list
        REPEAT
            APPEND 1 TO arr
        UNTIL LENGTH OF arr = n
        DEFINE FUNCTION updateArray(pos currentArr)
            IF pos = LENGTH OF currentArr THEN
                RETURN currentArr
            ELSE
                SET newArr TO copy of currentArr
                SET newArr[pos] TO modAdd(currentArr[pos], currentArr[pos - 1])
                RETURN updateArray(pos + 1, newArr)
            END IF
        END FUNCTION
        SET i TO 0
        WHILE i < k
            SET arr TO updateArray(1, arr)
            SET i TO i + 1
        END WHILE
        RETURN arr[n - 1]
    END FUNCTION
END CLASS
