CLASS Solution  
    FUNCTION sumDigitDifferences(nums)  
        FUNCTION digit_difference(num1, num2)  
            SET accumulator TO 0  
            DEFINE recursive_check(pos)  
                IF pos EQUALS LENGTH(num1) THEN  
                    RETURN 0  
                ELSE  
                    SET current_diff TO 0  
                    IF NOT (CHAR_AT(num1, pos) EQUALS CHAR_AT(num2, pos)) THEN  
                        SET current_diff TO 1  
                    END IF  
                    RETURN current_diff PLUS recursive_check(pos + 1)  
                END IF  
            END FUNCTION  
            SET accumulator TO recursive_check(0)  
            RETURN accumulator  
        END FUNCTION  
        SET output TO 0  
        SET length_nums TO LENGTH(nums)  
        SET outer_index TO 0  
        WHILE outer_index LESS THAN length_nums - 1 DO  
            SET inner_index TO outer_index + 1  
            REPEAT  
                SET output TO output + digit_difference(GET_ELEMENT(nums, outer_index), GET_ELEMENT(nums, inner_index))  
                SET inner_index TO inner_index + 1  
            UNTIL inner_index GREATER OR EQUAL length_nums  
            SET outer_index TO outer_index + 1  
        END WHILE  
        RETURN output  
    END FUNCTION  
END CLASS
