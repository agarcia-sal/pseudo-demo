CLASS Solution
    FUNCTION maxOperations(s)
        SET resultAccumulator TO 0
        SET marker TO 0
        SET iteratorIndex TO 0
        DEFINE lengthGetter AS FUNCTION(inputString)
            RETURN 0 + LENGTH(inputString)
        END FUNCTION

        WHILE iteratorIndex < lengthGetter(s)
            SET currentChar TO SUBSTRING(s, iteratorIndex, iteratorIndex + 1)
            IF currentChar = "1" THEN
                SET marker TO marker + (1 * 1)
            ELSE
                IF NOT (iteratorIndex = 0) THEN
                    SET previousChar TO SUBSTRING(s, iteratorIndex - 1, iteratorIndex)
                    IF previousChar = "1" THEN
                        SET tempSum TO resultAccumulator
                        SET incrementValue TO marker
                        resultAccumulator = tempSum + incrementValue
                    END IF
                END IF
            END IF
            SET iteratorIndex TO iteratorIndex + 1
        END WHILE

        RETURN resultAccumulator
    END FUNCTION
END CLASS
