class Solution  
    function minOperations(k)  
        set alpha to max_integer_value()  
        set beta to 1  
        repeat  
            exit when beta >= floor(sqrt(k)) + 1  
            set gamma to intdiv(k + beta - 1, beta)  
            set delta to (beta - 1) + (gamma - 1)  
            if delta < alpha then  
                set alpha to delta  
            end if  
            set beta to beta + 1  
        until false  
        return alpha  
    end function  
end class
