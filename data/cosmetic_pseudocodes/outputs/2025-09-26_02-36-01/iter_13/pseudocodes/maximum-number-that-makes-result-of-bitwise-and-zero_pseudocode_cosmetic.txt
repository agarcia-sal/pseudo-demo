CLASS Solution
    FUNCTION maxNumber(n)
        SET p TO 1
        SET result TO 0
        IF NOT (n = 1) THEN
            SET continue_loop TO TRUE
            WHILE continue_loop DO
                IF p <= n THEN
                    SET p TO p + p
                ELSE
                    SET continue_loop TO FALSE
                END IF
            END WHILE
            SET p TO p / 2
            SET tempVar TO p - 1
            SET result TO tempVar
        END IF
        RETURN result
    END FUNCTION
END CLASS
