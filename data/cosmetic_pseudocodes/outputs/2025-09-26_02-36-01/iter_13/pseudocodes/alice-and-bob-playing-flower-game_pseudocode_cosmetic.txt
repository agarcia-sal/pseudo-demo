CLASS Solution
    FUNCTION flowerGame(u, v)
        FUNCTION multiply(x, y)
            VAR result := 0
            VAR counter := 0
            WHILE counter LESS THAN y DO
                result := result PLUS x
                counter := counter PLUS 1
            END WHILE
            RETURN result
        END FUNCTION

        FUNCTION half_ceil(z)
            VAR temp1 := z MINUS 1
            VAR temp2 := temp1 DIVIDED BY 2
            VAR temp3 := temp2 PLUS 1
            RETURN temp3
        END FUNCTION

        FUNCTION half_floor(z)
            VAR temp := z DIVIDED BY 2
            RETURN temp
        END FUNCTION

        VAR w := half_ceil(u)
        VAR x := half_floor(u)
        VAR y := half_ceil(v)
        VAR z := half_floor(v)

        VAR a := multiply(w, z)
        VAR b := multiply(x, y)
        VAR c := a PLUS b

        RETURN c
    END FUNCTION
END CLASS
