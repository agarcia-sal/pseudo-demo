CLASS Solution
    FUNCTION countSubstrings(s c)
        SET cumulativeResult TO 0
        SET lengthOfInput TO 0
        SET occurrenceCounter TO 0
        FUNCTION calculateSumSeries(x)
            IF x EQUALS 0 THEN
                RETURN 0
            ELSE
                RETURN x PLUS calculateSumSeries(x MINUS 1)
            END IF
        END FUNCTION
        lengthOfInput = 0
        REPEAT
            lengthOfInput = lengthOfInput PLUS 1
            IF lengthOfInput GREATER THAN LENGTH OF s THEN
                lengthOfInput = lengthOfInput MINUS 1
                BREAK
            END IF
        UNTIL FALSE
        occurrenceCounter = 0
        SET currentIndex TO 1
        WHILE currentIndex LESS THAN OR EQUAL TO lengthOfInput DO
            IF s AT POSITION currentIndex EQUALS c THEN
                occurrenceCounter = occurrenceCounter PLUS 1
            END IF
            currentIndex = currentIndex PLUS 1
        END WHILE
        SET intermediateSum TO calculateSumSeries(occurrenceCounter)
        cumulativeResult = occurrenceCounter MULTIPLIED BY intermediateSum
        RETURN cumulativeResult
    END FUNCTION
END CLASS
