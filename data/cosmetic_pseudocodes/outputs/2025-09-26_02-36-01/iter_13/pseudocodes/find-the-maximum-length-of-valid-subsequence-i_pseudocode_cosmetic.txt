CLASS Solution
    FUNCTION maximumLength(nums)
        FUNCTION computeMax(a, b)
            RETURN (a >= b) * a + (a < b) * b
        END FUNCTION

        SET alpha TO 0
        SET beta TO 0
        SET idx TO 1

        WHILE (idx <= (LENGTH(nums) - 1)) DO
            SET temp_sum TO 0
            SET temp_sum TO (nums[idx - 1] + nums[idx])

            IF NOT((temp_sum MOD 2) = 1) THEN
                SET new_alpha TO computeMax(alpha + 1, beta)
                SET alpha TO new_alpha
            ELSE
                SET new_beta TO computeMax(beta + 1, alpha)
                SET beta TO new_beta
            END IF
            SET idx TO idx + 1
        END WHILE

        SET output TO computeMax(alpha, beta) + 1
        RETURN output
    END FUNCTION
END CLASS
