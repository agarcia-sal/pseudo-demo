CLASS Solution  
    FUNCTION countAlternatingSubarrays(nums)  
        SET length_var TO 0  
        PROCEDURE calc_length(arr)  
            SET idx TO 0  
            SET res TO 0  
            SET cur_len TO 0  
            FUNCTION recurse(k)  
                IF k GREATER THAN OR EQUAL TO length_var THEN  
                    RETURN  
                END IF  
                IF k EQUALS 0 THEN  
                    SET cur_len TO 1  
                    SET res TO res PLUS 1  
                ELSE  
                    IF NOT(arr[k] EQUALS arr[k MINUS 1]) THEN  
                        SET cur_len TO cur_len PLUS 1  
                    ELSE  
                        SET cur_len TO 1  
                    END IF  
                    SET res TO res PLUS (cur_len MINUS 1)  
                END IF  
                recurse(k PLUS 1)  
            END FUNCTION  
            recurse(0)  
            RETURN res  
        END PROCEDURE  
        SET length_var TO LENGTH OF nums  
        IF length_var EQUALS 0 THEN  
            RETURN 0  
        END IF  
        RETURN calc_length(nums)  
    END FUNCTION  
END CLASS
