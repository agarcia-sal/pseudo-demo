class Solution
   function minOperations(nums)
      var result ← 0
      var toggle ← 0
      var idx ← 0
      repeat
         if idx >= length(nums) then
            break
         end if
         var val ← 0
         if toggle % 2 ≠ 1 then
            val ← nums[idx]
         else
            val ← 1 - nums[idx]
         end if
         if val = 0 then
            result ← result + 1
            toggle ← toggle + 1
         end if
         idx ← idx + 1
      until false
      return result
   end function
end class
