CLASS Solution
    FUNCTION waysToReachStair WITH PARAMETER k
        FUNCTION dfs WITH PARAMETERS x y z
            SET result TO 0
            IF (x GREATER THAN k PLUS 1) THEN
                RETURN 0
            END IF
            IF (x EQUALS k) THEN
                SET result TO 1
            END IF
            IF (x GREATER THAN 0) THEN
                IF NOT (y NOT EQUAL TO 0) THEN
                    SET temp TO dfs(x MINUS 1, 1, z)
                    SET result TO result PLUS temp
                END IF
            END IF
            SET nextStep TO dfs(x PLUS (z MULTIPLIED BY 2), 0, z PLUS 1)
            SET result TO result PLUS nextStep
            RETURN result
        END FUNCTION

        RETURN dfs(1, 0, 0)
    END FUNCTION
END CLASS
