CLASS Solution
    FUNCTION minimumTimeToInitialState(word parameter word AND integer parameter k) RETURNS integer
        SET length_indicator TO word.LENGTH()
        SET counter TO 1
        REPEAT
            SET segment TO word.SUBSTRING(counter TIMES k, length_indicator MINUS counter TIMES k)
            IF word.INDEX_OF(segment) = 0 THEN
                RETURN counter
            END IF
            counter += 1
        UNTIL FALSE
    END FUNCTION
END CLASS
