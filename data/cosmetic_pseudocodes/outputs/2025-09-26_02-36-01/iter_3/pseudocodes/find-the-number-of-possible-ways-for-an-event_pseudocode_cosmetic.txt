CLASS Solution
    FUNCTION numberOfWays(n integer parameter, x integer parameter, y integer parameter) RETURNS integer
        SET modulus TO 1000000000 PLUS 7
        SET dp TO an array of length n PLUS 1 where each element is an array of length x PLUS 1 filled with zero
        SET dp[0][0] TO 1
        SET indexI TO 1
        WHILE indexI IS LESS THAN OR EQUAL TO n
            SET indexJ TO 1
            WHILE indexJ IS LESS THAN OR EQUAL TO x
                SET temp1 TO dp[indexI MINUS 1][indexJ] TIMES indexJ
                SET temp2 TO dp[indexI MINUS 1][indexJ MINUS 1] TIMES (x MINUS indexJ MINUS 1)
                SET sumVal TO temp1 PLUS temp2
                SET dp[indexI][indexJ] TO sumVal MOD modulus
                INCREMENT indexJ BY 1
            END WHILE
            INCREMENT indexI BY 1
        END WHILE
        SET result TO 0
        SET powerVal TO 1
        SET counter TO 1
        WHILE counter IS LESS THAN OR EQUAL TO x
            SET powerVal TO (powerVal TIMES y) MOD modulus
            SET result TO (result PLUS dp[n][counter] TIMES powerVal) MOD modulus
            INCREMENT counter BY 1
        END WHILE
        RETURN result
    END FUNCTION
END CLASS
