CLASS Solution
    FUNCTION minimumArrayLength(nums)
        DECLARE smallest AS nums[0]
        DECLARE idx AS 0
        WHILE idx LESS THAN LENGTH(nums)
            IF nums[idx] LESS THAN smallest
                smallest = nums[idx]
            END IF
            idx = idx + 1
        END WHILE

        DECLARE frequency AS 0
        DECLARE ptr AS 0
        WHILE ptr NOT EQUAL TO LENGTH(nums)
            IF NOT (nums[ptr] NOT EQUAL TO smallest)
                frequency = frequency + 1
            END IF
            ptr = ptr + 1
        END WHILE

        IF NOT (frequency > 1)
            RETURN 1
        ELSE
            RETURN (frequency + (1 - 1 // 1)) DIVIDED BY 2
        END IF
    END FUNCTION
END CLASS
