CLASS Solution
    FUNCTION numberOfAlternatingGroups(colors k)
        IF k GREATER THAN OR EQUAL TO 3 THEN
            SET total TO 0
            LET length BE THE LENGTH OF colors
            LET doubled_colors BE AN EMPTY LIST
            SET idx TO 0
            WHILE idx LESS THAN length
                APPEND colors[idx] TO doubled_colors
                INCREMENT idx BY 1
            END WHILE
            LET extra_limit BE k - 1
            SET idx TO 0
            WHILE idx LESS THAN extra_limit
                APPEND colors[idx] TO doubled_colors
                INCREMENT idx BY 1
            END WHILE

            SET start_pos TO 0
            WHILE start_pos NOT EQUAL TO length
                SET valid_group TO 1
                SET offset TO 1
                WHILE offset NOT EQUAL TO k
                    IF doubled_colors[start_pos + offset] EQUALS doubled_colors[start_pos + offset - 1] THEN
                        SET valid_group TO 0
                        EXIT INNER LOOP
                    END IF
                    INCREMENT offset BY 1
                END WHILE
                total = total + valid_group
                INCREMENT start_pos BY 1
            END WHILE

            RETURN total
        ELSE
            RETURN 0
        END IF
    END FUNCTION
END CLASS
