CLASS Solution
    FUNCTION minimumAddedInteger(nums1 RETURNS List OF integer, nums2 RETURNS List OF integer) RETURNS integer
        CALL sort ON nums1
        CALL sort ON nums2

        SET len_nums1 TO LENGTH OF nums1
        SET len_nums2 TO LENGTH OF nums2

        SET outer_idx TO 0
        WHILE outer_idx < len_nums1 - 1
            SET inner_idx TO outer_idx + 1
            WHILE inner_idx < len_nums1
                SET filtered_nums1 TO EMPTY LIST

                SET cur_pos TO 0
                WHILE cur_pos < outer_idx
                    APPEND nums1[cur_pos] TO filtered_nums1
                    SET cur_pos TO cur_pos + 1
                END WHILE

                SET cur_pos TO outer_idx + 1
                WHILE cur_pos < inner_idx
                    APPEND nums1[cur_pos] TO filtered_nums1
                    SET cur_pos TO cur_pos + 1
                END WHILE

                SET cur_pos TO inner_idx + 1
                WHILE cur_pos < len_nums1
                    APPEND nums1[cur_pos] TO filtered_nums1
                    SET cur_pos TO cur_pos + 1
                END WHILE

                SET first_diff TO nums2[0] MINUS filtered_nums1[0]

                SET valid_shift TO TRUE

                SET idx_check TO 0
                WHILE idx_check < len_nums2
                    SET check_val TO filtered_nums1[idx_check] PLUS first_diff
                    IF NOT (check_val EQUALS nums2[idx_check])
                        SET valid_shift TO FALSE
                        BREAK
                    END IF
                    SET idx_check TO idx_check + 1
                END WHILE

                IF valid_shift EQUALS TRUE
                    RETURN first_diff
                END IF

                SET inner_idx TO inner_idx + 1
            END WHILE
            SET outer_idx TO outer_idx + 1
        END WHILE

        RETURN None
    END FUNCTION
END CLASS
