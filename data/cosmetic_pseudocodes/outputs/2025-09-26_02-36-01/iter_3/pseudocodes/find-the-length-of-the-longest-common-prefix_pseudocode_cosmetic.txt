CLASS Solution  
    FUNCTION longestCommonPrefix(arr1 arr1 OF integers arr2 arr2 OF integers) RETURNS integer  
        DECLARE collectedPrefixes1 : set  
        INITIALIZE collectedPrefixes1 AS empty set  
        DECLARE currentNum1, currentStr1, idx1 : integer/string  
        SET idx1 TO 0  
        WHILE idx1 < LENGTH(arr1) DO  
            SET currentNum1 TO VALUE AT arr1[idx1]  
            SET currentStr1 TO CONVERT_TO_STRING(currentNum1)  
            DECLARE length1 : integer  
            SET length1 TO LENGTH(currentStr1)  
            DECLARE pos1 : integer  
            SET pos1 TO 1  
            WHILE pos1 <= length1 DO  
                DECLARE subStr1 : string  
                SET subStr1 TO SLICE(currentStr1, 0, pos1)  
                INSERT subStr1 INTO collectedPrefixes1  
                SET pos1 TO pos1 + 1  
            END WHILE  
            SET idx1 TO idx1 + 1  
        END WHILE  

        DECLARE collectedPrefixes2 : set  
        INITIALIZE collectedPrefixes2 AS empty set  
        DECLARE currentNum2, currentStr2, idx2 : integer/string  
        SET idx2 TO 0  
        WHILE idx2 < LENGTH(arr2) DO  
            SET currentNum2 TO GET_ELEMENT(arr2, idx2)  
            SET currentStr2 TO TO_STRING(currentNum2)  
            DECLARE length2 : integer  
            SET length2 TO LENGTH(currentStr2)  
            DECLARE pos2 : integer  
            SET pos2 TO 1  
            WHILE pos2 <= length2 DO  
                DECLARE subStr2 : string  
                SET subStr2 TO SUBSTRING(currentStr2, 0, pos2)  
                ADD subStr2 TO collectedPrefixes2  
                SET pos2 TO pos2 + 1  
            END WHILE  
            INCREMENT idx2 BY 1  
        END WHILE  

        DECLARE maxCommonLen : integer  
        SET maxCommonLen TO 0  
        FOR prefix IN collectedPrefixes1 DO  
            DECLARE existsInSecond : boolean  
            SET existsInSecond TO prefix IN collectedPrefixes2  
            IF existsInSecond THEN  
                DECLARE lenPrefix : integer  
                SET lenPrefix TO LENGTH(prefix)  
                IF lenPrefix > maxCommonLen THEN  
                    SET maxCommonLen TO lenPrefix  
                END IF  
            END IF  
        END FOR  

        RETURN maxCommonLen  
    END FUNCTION  
END CLASS
