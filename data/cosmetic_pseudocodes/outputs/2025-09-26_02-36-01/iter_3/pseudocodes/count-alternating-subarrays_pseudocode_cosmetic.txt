CLASS Solution
    FUNCTION countAlternatingSubarrays(nums)
        LET totalElements BE LENGTH(nums)
        IF totalElements = 0 THEN
            RETURN 0
        END IF
        LET totalCount BE totalElements
        LET seqLength BE 1
        LET index BE 1
        WHILE index < totalElements
            IF NOT(nums[index] = nums[index - 1]) THEN
                seqLength := seqLength + 1
            ELSE
                seqLength := 1
            END IF
            totalCount := totalCount + (seqLength - 1)
            index := index + 1
        END WHILE
        RETURN totalCount
    END FUNCTION
END CLASS
