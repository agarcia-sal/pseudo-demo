CLASS Solution
    FUNCTION maximumLength(nums)
        SET idx TO 1
        SET len_even TO 0
        SET len_odd TO 0
        WHILE idx < (SIZE_OF nums)
            SET temp_sum TO (nums {idx - 1}) + (nums {idx})
            IF (temp_sum & 1) = 0 THEN
                SET len_even TO MAX(len_even + 1, len_odd)
            ELSE
                SET len_odd TO MAX(len_odd + 1, len_even)
            END IF
            INCREMENT idx BY 1
        END WHILE
        SET final_res TO MAX(len_even, len_odd) + 1
        RETURN final_res
    END FUNCTION
END CLASS
