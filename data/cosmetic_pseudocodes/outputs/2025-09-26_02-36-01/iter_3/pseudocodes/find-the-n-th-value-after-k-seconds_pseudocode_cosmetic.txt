CLASS Solution
    FUNCTION valueAfterKSeconds(n k)
        LET MODULUS BE 10^9 + 7
        LET sequence BE new array of length n
        LET pos BE 0
        WHILE pos < n
            sequence[pos] = 1
            pos = pos + 1
        END WHILE

        LET counter BE 0
        WHILE counter < k
            LET idx BE 1
            WHILE idx < n
                LET sum_val BE sequence[idx] + sequence[idx - 1]
                LET remainder BE sum_val - (sum_val / MODULUS) * MODULUS
                sequence[idx] = remainder
                idx = idx + 1
            END WHILE
            counter = counter + 1
        END WHILE

        RETURN sequence[n - 1]
    END FUNCTION
END CLASS
