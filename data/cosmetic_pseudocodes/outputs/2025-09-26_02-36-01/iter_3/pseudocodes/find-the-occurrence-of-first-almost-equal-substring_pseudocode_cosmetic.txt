CLASS Solution
    FUNCTION minStartingIndex WITH PARAMETERS s pattern
        FUNCTION is_almost_equal WITH PARAMETERS substring pattern
            SET mismatch_count TO 0
            SET pos TO 0
            WHILE pos < LENGTH(substring)
                SET char_sub TO substring[pos]
                SET char_pat TO pattern[pos]
                IF NOT (char_sub = char_pat)
                    mismatch_count = mismatch_count + 1
                    IF mismatch_count > 1
                        RETURN False
                    END IF
                END IF
                pos = pos + 1
            END WHILE
            RETURN True
        END FUNCTION

        SET length_pat TO LENGTH(pattern)
        SET start_pos TO 0
        WHILE start_pos <= LENGTH(s) - length_pat
            SET segment TO EMPTY LIST
            SET offset TO 0
            WHILE offset < length_pat
                APPEND s[start_pos + offset] TO segment
                offset = offset + 1
            END WHILE
            IF is_almost_equal WITH ARGUMENTS segment pattern
                RETURN start_pos
            END IF
            start_pos = start_pos + 1
        END WHILE
        RETURN -1
    END FUNCTION
END CLASS
