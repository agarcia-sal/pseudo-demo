CLASS Solution
    FUNCTION minOperations(nums parameter k parameter)
        CALL heapify ON nums
        SET op_count TO 0
        WHILE 1 < (length of nums) AND (nums[0] < k)
            SET first_min TO heappop(nums)
            SET second_min TO heappop(nums)
            SET doubled_min TO first_min + first_min
            SET max_val TO second_min
            IF max_val < first_min THEN
                SET max_val TO first_min
            END IF
            SET combined TO doubled_min + max_val - first_min
            IF max_val = first_min THEN
                SET combined TO doubled_min + max_val
            END IF
            IF first_min < second_min THEN
                SET combined TO (2 * first_min) + second_min
            ELSE
                SET combined TO (2 * second_min) + first_min
            END IF
            CALL heappush ON nums WITH combined
            SET op_count TO op_count + 1
        END WHILE
        RETURN op_count
    END FUNCTION
END CLASS
