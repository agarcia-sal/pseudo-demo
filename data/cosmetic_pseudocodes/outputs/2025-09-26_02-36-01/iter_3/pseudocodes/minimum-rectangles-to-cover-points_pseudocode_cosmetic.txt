CLASS Solution  
    FUNCTION minRectanglesToCoverPoints(points, w)  
        ASSIGN 0 TO count  
        ASSIGN -1 TO boundary  
        INITIALIZE index TO 0  
        SORT points BY their first coordinate IN non-decreasing order  
        WHILE index LESS THAN LENGTH of points  
            LET coord_x BE points[index][0]  
            IF boundary LESS THAN coord_x THEN  
                boundary = coord_x + w  
                count = count + 1  
            END IF  
            index = index + 1  
        END WHILE  
        RETURN count  
    END FUNCTION  
END CLASS
