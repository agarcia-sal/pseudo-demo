CLASS Solution
    FUNCTION minOperationsToMakeMedianK(nums k)
        CALL nums.sortAscending()
        LET size = nums.length()
        LET center = size DIV 2

        LET ops = 0
        IF NOT (nums[center] != k) THEN
            RETURN 0
        END IF

        IF nums[center] < k THEN
            LET pos = center
            LOOP WHILE pos < size AND nums[pos] < k
                LET diff = k - nums[pos]
                ops += diff
                pos = pos + 1
            END LOOP
        ELSE
            LET idx = center
            LOOP WHILE idx >= 0 AND nums[idx] > k
                LET diff = nums[idx] - k
                ops = ops + diff
                idx = idx - 1
            END LOOP
        END IF

        RETURN ops
    END FUNCTION
END CLASS
