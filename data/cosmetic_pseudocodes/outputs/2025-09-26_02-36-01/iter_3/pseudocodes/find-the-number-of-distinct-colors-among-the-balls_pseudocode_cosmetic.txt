CLASS Solution
    FUNCTION queryResults(limit_param, queries_param)
        INITIALIZE color_map AS empty map from keys to values
        INITIALIZE distinct_color_collection AS empty set
        INITIALIZE output_list AS empty array

        SET index_counter TO 0
        WHILE index_counter < LENGTH(queries_param)
            EXTRACT current_pair FROM queries_param AT index_counter
            EXTRACT first_value FROM current_pair AT POSITION 0
            EXTRACT second_value FROM current_pair AT POSITION 1

            IF first_value IS PRESENT IN KEYS OF color_map THEN
                SET prev_color TO color_map[first_value]
                IF prev_color IS IN distinct_color_collection THEN
                    DELETE prev_color FROM distinct_color_collection
                END IF
            END IF

            ASSIGN second_value TO color_map[first_value]
            INSERT second_value INTO distinct_color_collection
            APPEND SIZE OF distinct_color_collection TO output_list

            INCREMENT index_counter BY 1
        END WHILE

        RETURN output_list
    END FUNCTION
END CLASS
