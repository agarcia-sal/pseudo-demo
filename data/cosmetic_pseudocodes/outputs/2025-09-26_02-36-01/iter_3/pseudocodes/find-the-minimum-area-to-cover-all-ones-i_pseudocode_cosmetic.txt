CLASS Solution
    FUNCTION minimumArea(grid)
        DEFINE total_rows AS LENGTH grid
        IF total_rows IS zero OR LENGTH grid[0] IS zero
            RETURN 0
        END IF
        DEFINE total_cols AS LENGTH grid[0]

        SET top_edge TO +∞
        SET bottom_edge TO -∞
        SET left_edge TO +∞
        SET right_edge TO -∞

        SET row_index TO 0
        WHILE row_index < total_rows
            SET col_index TO 0
            WHILE col_index < total_cols
                SET cell_value TO grid[row_index][col_index]
                IF cell_value == 1
                    IF top_edge > row_index THEN top_edge = row_index END IF
                    IF bottom_edge < row_index THEN bottom_edge = row_index END IF
                    IF left_edge > col_index THEN left_edge = col_index END IF
                    IF right_edge < col_index THEN right_edge = col_index END IF
                END IF
                INCREMENT col_index BY 1
            END WHILE
            INCREMENT row_index BY 1
        END WHILE

        DEFINE rect_height AS (bottom_edge - top_edge + 1)
        DEFINE rect_width AS (right_edge - left_edge + 1)
        RETURN rect_height * rect_width
    END FUNCTION
END CLASS
