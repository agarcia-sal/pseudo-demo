CLASS Solution
    FUNCTION maximumHappinessSum(happiness, k)
        ARRANGE happiness so that each item is not smaller than the one after it
        INITIALIZE total_happiness WITH 0
        INITIALIZE subtractor WITH 0
        SET counter TO 0
        WHILE counter < k
            ASSIGN temp_score TO (happiness[counter] + (-subtractor))
            IF NOT (temp_score >= 0)
                temp_score = 0
            END IF
            total_happiness = total_happiness PLUS temp_score
            subtractor = subtractor + 1
            counter = counter + 1
        END WHILE
        RETURN total_happiness
    END FUNCTION
END CLASS
