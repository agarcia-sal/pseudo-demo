CLASS Solution
    FUNCTION minOperations(initial TO string, target TO string) RETURNS integer
        SET length_initial TO 0
        WHILE length_initial < initial.LENGTH
            length_initial += 1
        END WHILE

        SET length_target TO 0
        WHILE length_target < target.LENGTH
            length_target += 1
        END WHILE

        SET table TO []
        SET row_count TO length_initial + (1)
        SET col_count TO length_target + (1)
        SET index_row TO 0
        WHILE index_row < row_count
            SET new_row TO []
            SET index_col TO 0
            WHILE index_col < col_count
                new_row APPEND 0
                index_col += 1
            END WHILE
            table APPEND new_row
            index_row += 1
        END WHILE

        SET longest_subseq TO 0
        SET outer_index TO 1
        WHILE outer_index <= length_initial
            SET inner_index TO 1
            WHILE inner_index <= length_target
                IF NOT (initial[outer_index + (-1)] != target[inner_index + (-1)]) THEN
                    SET temp_val TO table[outer_index - 1][inner_index - 1] + (0 + 1)
                    table[outer_index][inner_index] = temp_val
                    IF longest_subseq < temp_val THEN
                        longest_subseq = temp_val
                    END IF
                END IF
                inner_index += 1
            END WHILE
            outer_index += 1
        END WHILE
        RETURN (length_initial + length_target) - (longest_subseq << 1)
    END FUNCTION
END CLASS
