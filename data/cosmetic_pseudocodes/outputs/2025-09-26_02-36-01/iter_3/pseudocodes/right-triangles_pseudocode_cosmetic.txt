CLASS Solution
    FUNCTION numberOfRightTriangles(grid)
        LET number_of_rows := SIZE grid
        LET number_of_columns := SIZE FIRST grid
        LET count_of_triangles := 0

        LET row_index := 0
        WHILE row_index < number_of_rows
            LET column_index := 0
            WHILE column_index < number_of_columns
                LET current_cell := grid[row_index][column_index]
                IF current_cell == 1 THEN
                    LET row_sum_exclusive := 0
                    LET col_sum_exclusive := 0

                    LET col_iter := 0
                    WHILE col_iter < number_of_columns
                        IF NOT (col_iter == column_index) THEN
                            LET row_sum_exclusive := row_sum_exclusive + grid[row_index][col_iter]
                        ENDIF
                        LET col_iter := col_iter + 1
                    ENDWHILE

                    LET row_iter := 0
                    WHILE row_iter < number_of_rows
                        IF NOT (row_iter == row_index) THEN
                            LET col_sum_exclusive := col_sum_exclusive + grid[row_iter][column_index]
                        ENDIF
                        LET row_iter := row_iter + 1
                    ENDWHILE

                    LET increment_value := row_sum_exclusive * col_sum_exclusive
                    LET count_of_triangles := count_of_triangles + increment_value
                ENDIF
                LET column_index := column_index + 1
            ENDWHILE
            LET row_index := row_index + 1
        ENDWHILE

        RETURN count_of_triangles
    END FUNCTION
END CLASS
