CLASS Solution
    FUNCTION distanceSum WITH PARAMETERS m n k
        CONST MODULO ← 1000000000 + 7
        LET a ← m * m * (m * m * m - m) / 6
        LET b ← n * n * (n * n * n - n) / 6
        LET total_points ← m * n
        LET comb ← 1
        LET count ← k - 2
        LET limit ← total_points - 2
        LET numerator ← 1
        LET denominator ← 1
        LET idx ← 0
        WHILE idx < count
            numerator ← numerator * (limit - idx)
            denominator ← denominator * (idx + 1)
            idx ← idx + 1
        END WHILE
        LET comb ← numerator / denominator
        LET dist_sum ← (a * b + b * a) * comb
        RETURN dist_sum MOD MODULO
    END FUNCTION
END CLASS
