CLASS Solution
    FUNCTION validStrings length
        IF length == 1 THEN
            RETURN ["0", "1"]
        END IF

        SET resultCollection TO []
        
        FUNCTION recurse(sequence)
            SET seqLen TO LENGTH(sequence)
            IF seqLen >= length THEN
                resultCollection += [sequence]
                RETURN
            END IF

            SET lastChar TO sequence[seqLen - 1]

            IF lastChar != "0" THEN
                recurse(sequence + "0")
                recurse(sequence + "1")
            ELSE
                recurse(sequence + "1")
            END IF
        END FUNCTION

        SET stack TO ["0", "1"]
        WHILE LENGTH(stack) > 0 DO
            SET currentSeq TO stack[-1]
            stack = stack[0:-1]
            IF LENGTH(currentSeq) == length THEN
                resultCollection += [currentSeq]
            ELSE
                SET tailChar TO currentSeq[-1]
                IF tailChar == "1" THEN
                    stack += [currentSeq + "0"]
                    stack += [currentSeq + "1"]
                ELSE
                    stack += [currentSeq + "1"]
                END IF
            END IF
        END WHILE

        RETURN resultCollection
    END FUNCTION
END CLASS
