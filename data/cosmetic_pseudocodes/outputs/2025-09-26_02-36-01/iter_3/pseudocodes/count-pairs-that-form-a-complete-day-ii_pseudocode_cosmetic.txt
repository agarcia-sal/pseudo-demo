CLASS Solution
    FUNCTION countCompleteDayPairs WITH parameter hours
        remainder_map ← empty map defaulting to 0
        total_pairs ← 0
        index ← 0
        max_index ← length of hours
        WHILE index < max_index
            current_hour ← hours[index]
            mod_val ← ((current_hour MOD 24) + 24) MOD 24
            needed ← ((24 - mod_val) MOD 24)
            total_pairs ← total_pairs + remainder_map[needed]
            remainder_map[mod_val] ← remainder_map[mod_val] + 1
            index ← index + 1
        END WHILE
        RETURN total_pairs
    END FUNCTION
END CLASS
