CLASS Solution
    FUNCTION maxOperations(s)
        LET result <- 0
        LET tally <- 0
        LET position <- 0
        WHILE position < LENGTH(s)
            LET currentChar <- s[position]
            IF currentChar = '1' THEN
                tally <- tally + 1
            ELSE
                IF position > 0 THEN
                    LET previousChar <- s[position - 1]
                    IF previousChar = '1' THEN
                        result <- result + tally
                    END IF
                END IF
            END IF
            position <- position + 1
        END WHILE
        RETURN result
    END FUNCTION
END CLASS
