CLASS Solution
 FUNCTION waysToReachStair(k)
  FUNCTION dfs(i, j, jump)
   IF (k + 1) < i THEN
    RETURN 0
   END IF
   local count ← 0
   IF i = k THEN
    count ← 1
   ELSE
    count ← 0
   END IF
   IF (0 < i) AND (j = 0) THEN
    count ← count + dfs(i - 1, 1, jump)
   END IF
   count ← count + dfs((i + 1) * 2 * jump, 0, jump + 1)
   RETURN count
  END FUNCTION
  RETURN dfs(1, 0, 0)
 END FUNCTION
END CLASS
