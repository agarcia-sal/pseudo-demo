CLASS Solution
    FUNCTION countSubstrings(s)
        LET lengthVar BE 0
        LET resultCount BE 0
        LET startIndex BE 0
        WHILE startIndex <= LENGTH s - 1
            LET accumulator BE 0
            LET endIndex BE startIndex
            WHILE endIndex <= LENGTH s - 1
                LET digitValue BE TO_INTEGER s[endIndex]
                accumulator <- accumulator * 10
                accumulator <- accumulator + digitValue
                IF (digitValue != 0) AND ((accumulator - (accumulator / digitValue) * digitValue) = 0)
                    resultCount <- resultCount + 1
                END IF
                endIndex <- endIndex + 1
            END WHILE
            startIndex <- startIndex + 1
        END WHILE
        RETURN resultCount
    END FUNCTION
END CLASS
