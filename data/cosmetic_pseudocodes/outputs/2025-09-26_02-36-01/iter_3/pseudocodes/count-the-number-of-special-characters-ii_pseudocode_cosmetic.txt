CLASS Solution
    FUNCTION numberOfSpecialChars(word)
        DECLARE beginningMap AS empty map
        DECLARE endingMap AS empty map
        DECLARE positionCounter AS 0
        WHILE positionCounter < LENGTH(word)
            DECLARE currentChar AS word[positionCounter]
            IF NOT (currentChar IN_KEYS beginningMap)
                beginningMap[currentChar] := positionCounter
            END IF
            endingMap[currentChar] := positionCounter
            positionCounter := positionCounter + 1
        END WHILE

        DECLARE resultCount AS 0
        DECLARE lowerIndex AS 0
        DECLARE upperIndex AS 0
        DECLARE lowercaseSeq AS "abcdefghijklmnopqrstuvwxyz"
        DECLARE uppercaseSeq AS "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
        WHILE lowerIndex < LENGTH(lowercaseSeq) AND upperIndex < LENGTH(uppercaseSeq)
            DECLARE lowerChar AS lowercaseSeq[lowerIndex]
            DECLARE upperChar AS uppercaseSeq[upperIndex]

            IF (lowerChar IN_KEYS endingMap)
                AND (upperChar IN_KEYS beginningMap)
                AND ((endingMap[lowerChar] < beginningMap[upperChar]) = TRUE)
                    resultCount := resultCount + 1
            END IF
            lowerIndex := lowerIndex + 1
            upperIndex := upperIndex + 1
        END WHILE

        RETURN resultCount
    END FUNCTION
END CLASS
