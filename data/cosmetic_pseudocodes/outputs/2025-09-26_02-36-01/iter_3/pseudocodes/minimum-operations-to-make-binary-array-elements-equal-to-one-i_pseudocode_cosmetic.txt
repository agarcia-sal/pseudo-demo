CLASS Solution  
    FUNCTION minOperations(nums)  
        LET lengthVar BE LENGTH(nums)  
        LET countOps BE 0  
        LET indexVar BE 0  
        WHILE indexVar <= (lengthVar - 3)  
            IF nums[indexVar] = 0 THEN  
                LET nums[indexVar] BE 1 - nums[indexVar]  
                LET nums[indexVar + 1] BE 1 - nums[indexVar + 1]  
                LET nums[indexVar + 2] BE 1 - nums[indexVar + 2]  
                countOps ← countOps + 1  
            END IF  
            indexVar ← indexVar + 1  
        END WHILE  
        IF (nums[lengthVar - 1] = 0) OR (nums[lengthVar - 2] = 0) THEN  
            RETURN (-1)  
        END IF  
        RETURN countOps  
    END FUNCTION  
END CLASS
