CLASS Solution
    FUNCTION sumDigitDifferences(nums)
        FUNCTION digit_difference(numA, numB)
            SET mismatchCount TO 0
            SET pos TO 0
            WHILE pos < LENGTH(numA)
                IF NOT (SUBSTRING(numA, pos, 1) = SUBSTRING(numB, pos, 1)) THEN
                    mismatchCount = mismatchCount + 1
                END IF
                pos = pos + 1
            END WHILE
            RETURN mismatchCount
        END FUNCTION

        SET aggregate TO 0
        SET totalNums TO LENGTH(nums)
        SET outerIndex TO 0
        WHILE outerIndex <> totalNums - 1
            SET innerIndex TO outerIndex + 1
            WHILE innerIndex < totalNums
                aggregate += digit_difference(nums[outerIndex], nums[innerIndex])
                innerIndex = innerIndex + 1
            END WHILE
            outerIndex = outerIndex + 1
        END WHILE

        RETURN aggregate
    END FUNCTION
END CLASS
