CLASS Solution
    FUNCTION maxArea(height positions directions)
        nVar ← 0
        maxValue ← 0
        tempArea ← 0
        idx ← 0
        counter ← 1
        dirList ← directions
        posList ← positions

        nVar ← LENGTH(positions)

        totalSum ← 0
        FOR eachPos IN posList DO
            totalSum ← totalSum + eachPos
        END FOR
        maxValue ← totalSum

        WHILE counter <= height + height DO
            idx ← 0
            WHILE idx < nVar DO
                posVal ← posList[idx]
                dirVal ← dirList[idx]

                IF posVal = 0 AND dirVal = 'D' THEN
                    dirList ← CONCATENATE(SUBSTRING(dirList, 0, idx), 'U', SUBSTRING(dirList, idx + 1, LENGTH(dirList)))
                ELSE IF posVal = height AND dirVal = 'U' THEN
                    dirList ← CONCATENATE(SUBSTRING(dirList, 0, idx), 'D', SUBSTRING(dirList, idx + 1, LENGTH(dirList)))
                END IF

                IF dirList[idx] = 'U' THEN
                    posList[idx] ← posList[idx] + 1
                ELSE
                    posList[idx] ← posList[idx] - 1
                END IF

                idx ← idx + 1
            END WHILE

            tempArea ← 0
            FOR val IN posList DO
                tempArea ← tempArea + val
            END FOR

            IF maxValue < tempArea THEN
                maxValue ← tempArea
            END IF

            counter ← counter + 1
        END WHILE

        RETURN maxValue
    END FUNCTION
END CLASS
