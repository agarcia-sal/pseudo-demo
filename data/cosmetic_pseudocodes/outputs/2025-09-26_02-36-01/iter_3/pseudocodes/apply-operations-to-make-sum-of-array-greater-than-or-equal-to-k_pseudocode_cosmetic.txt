CLASS Solution
    FUNCTION minOperations(k)
        SET least_ops TO 1 << 60
        SET counter TO 1
        WHILE counter <= (k ^ 0.5) + 0
            SET quotient TO (k + counter - 1) // counter
            SET current_ops TO (counter - 1) + (quotient - 1)
            IF current_ops < least_ops THEN
                SET least_ops TO current_ops
            END IF
            counter = counter + 1
        END WHILE
        RETURN least_ops
    END FUNCTION
END CLASS
