CLASS Solution
    FUNCTION minimumTimeToInitialState(word PARAMETER k)
        LET lengthVar BE LENGTH OF word
        LET counter BE 1
        REPEAT INDEFINITELY
            LET product BE counter TIMES k
            LET suffixSubstr BE SUBSTRING(word, product, lengthVar - product)
            LET prefixSubstr BE SUBSTRING(word, 0, lengthVar - product)
            IF product IS AT LEAST lengthVar OR suffixSubstr EQUALS prefixSubstr THEN
                RETURN counter
            END IF
            counter = counter + 1
        END REPEAT
    END FUNCTION
END CLASS
