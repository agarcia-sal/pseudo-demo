CLASS Solution
    FUNCTION countSubarrays(nums parameter LIST OF INTEGER k parameter INTEGER) RETURNS INTEGER
        LET totalElems BE LENGTH(nums)
        LET resultCounter BE 0
        LET outerIndex BE 0
        WHILE outerIndex <= totalElems - 1 DO
            LET andAccumulator BE nums[outerIndex]
            LET innerIndex BE outerIndex
            WHILE innerIndex < totalElems DO
                andAccumulator = andAccumulator & nums[innerIndex]
                IF (andAccumulator = k) THEN
                    resultCounter = resultCounter + 1
                END IF
                IF NOT (andAccumulator >= k) THEN
                    EXIT
                END IF
                innerIndex = innerIndex + 1
            END WHILE
            outerIndex = outerIndex + 1
        END WHILE
        RETURN resultCounter
    END FUNCTION
END CLASS
