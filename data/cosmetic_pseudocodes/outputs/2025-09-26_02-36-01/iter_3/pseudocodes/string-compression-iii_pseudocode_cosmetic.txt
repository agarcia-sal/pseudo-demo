CLASS Solution
    FUNCTION compressedString(word)
        SET resultPieces TO []
        SET position TO 0
        LOOP 
            IF NOT (position < LENGTH(word))
                BREAK
            END IF
            SET currentChar TO word[position]
            SET tally TO 0
            LOOP
                IF NOT (position < LENGTH(word) AND word[position] = currentChar)
                    BREAK
                END IF
                tally = tally + 1
                position = position + 1
                IF tally >= 9
                    BREAK
                END IF
            END LOOP
            SET segment TO STRING(tally) + currentChar
            resultPieces = resultPieces + [segment]
        END LOOP
        SET output TO EMPTY_STRING
        SET idx TO 0
        WHILE idx < LENGTH(resultPieces)
            output = output + resultPieces[idx]
            idx = idx + 1
        END WHILE
        RETURN output
    END FUNCTION
END CLASS
