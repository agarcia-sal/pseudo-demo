CLASS Solution
    FUNCTION maximumValueSum(nums k edges)
        LET aggregateSum BE 0
        LET positiveGainCount BE 0
        LET smallestGain BE infinity_value

        LET index BE 0
        WHILE index < LENGTH(nums)
            LET currentNum BE nums[index]
            LET xorResult BE currentNum XOR k
            LET netGain BE xorResult - currentNum

            IF netGain > 0 THEN
                positiveGainCount <- positiveGainCount + 1
            END IF

            LET maxVal BE currentNum
            IF xorResult > maxVal THEN
                maxVal <- xorResult
            END IF
            aggregateSum <- aggregateSum + maxVal

            LET absGain BE netGain
            IF absGain < 0 THEN
                absGain <- 0 - absGain
            END IF
            IF absGain < smallestGain THEN
                smallestGain <- absGain
            END IF

            index <- index + 1
        END WHILE

        IF positiveGainCount % 2 != 0 THEN
            aggregateSum <- aggregateSum - smallestGain
        END IF

        RETURN aggregateSum
    END FUNCTION
END CLASS
