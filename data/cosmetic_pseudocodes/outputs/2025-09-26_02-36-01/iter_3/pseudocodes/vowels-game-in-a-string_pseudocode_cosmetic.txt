CLASS Solution
    FUNCTION doesAliceWin(s)
        SET vowelSet TO map v from ["a","e","i","o","u"] into v
        SET countOddSegments TO 0
        SET vowelCounter TO 0

        SET index TO 0
        WHILE index < LENGTH(s)
            SET char TO s[index]
            IF char IN vowelSet THEN
                vowelCounter ← vowelCounter + 1
            ENDIF
            SET index TO index + 1
        ENDWHILE

        SET isOddSegmentActive TO false

        SET pos TO 0
        WHILE pos < LENGTH(s)
            SET letter TO s[pos]

            IF letter IN vowelSet THEN
                isOddSegmentActive ← NOT isOddSegmentActive
            ENDIF

            IF isOddSegmentActive THEN
                vowelCounter += 1
            ELSE
                IF (vowelCounter % 2) ≠ 0 THEN
                    countOddSegments ← countOddSegments + 1
                    vowelCounter ← 0
                ENDIF
            ENDIF
            pos ← pos + 1
        ENDWHILE

        IF isOddSegmentActive AND (vowelCounter % 2) ≠ 0 THEN
            countOddSegments ← countOddSegments + 1
        ENDIF

        RETURN ((countOddSegments MOD 2) = 1)
    END FUNCTION
END CLASS
