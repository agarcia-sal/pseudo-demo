CLASS Solution
    FUNCTION flowerGame(n, m)
        LET half_n_integer_part BE n DIV 2
        LET half_n_rounded_up BE (n + 1) DIV 2
        LET half_m_integer_part BE m DIV 2
        LET half_m_rounded_up BE (m + 1) DIV 2
        LET product_one BE half_n_rounded_up TIMES half_m_integer_part
        LET product_two BE half_n_integer_part TIMES half_m_rounded_up
        LET result_sum BE product_one + product_two
        RETURN result_sum
    END FUNCTION
END CLASS
