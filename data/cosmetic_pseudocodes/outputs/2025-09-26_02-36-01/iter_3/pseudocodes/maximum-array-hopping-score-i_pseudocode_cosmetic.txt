CLASS Solution  
    FUNCTION maxScore(nums)  
        SET length TO LENGTH OF nums  
        CREATE dp AS ARRAY OF length ELEMENTS INITIALIZED TO 0  
        SET dp[0] TO 0  
        SET index TO 1  
        WHILE index < length  
            SET inner TO 0  
            WHILE inner < index  
                SET candidate TO dp[inner] + ((index - inner) * nums[index])  
                IF dp[index] < candidate THEN  
                    dp[index] = candidate  
                END IF  
                SET inner TO inner + 1  
            END WHILE  
            SET index TO index + 1  
        END WHILE  
        RETURN dp[length - 1]  
    END FUNCTION  
END CLASS
