CLASS Solution
    FUNCTION minimumLength(s)
        SET frequency_map TO Counter(s)
        SET odd_total TO 0
        SET even_total TO 0
        SET iterator TO getIterator(frequency_map.values())
        WHILE hasNext(iterator)
            SET current_val TO next(iterator)
            IF (current_val - 1) MOD 2 == 0
                odd_total = odd_total + 1
            ELSE
                IF NOT (current_val == 0)
                    IF (current_val MOD 2) == 0
                        even_total += 2
                    END IF
                END IF
            END IF
        END WHILE
        SET aggregate TO (odd_total) + (even_total)
        RETURN aggregate
    END FUNCTION
END CLASS
