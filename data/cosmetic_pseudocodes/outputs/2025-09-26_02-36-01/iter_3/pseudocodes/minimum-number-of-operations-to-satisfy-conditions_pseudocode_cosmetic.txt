CLASS Solution
    FUNCTION minimumOperations(grid)
        LET rowCount EQUAL TO LENGTH grid
        LET colCount EQUAL TO LENGTH grid[0]
        LET opsCounter BE 0

        SET colIndex TO 0
        WHILE colIndex < colCount
            LET rowCursor TO 0
            WHILE rowCursor < rowCount - 1
                LET currentVal EQUAL TO grid[rowCursor][colIndex]
                LET nextVal EQUAL TO grid[rowCursor + 1][colIndex]
                IF NOT (currentVal = nextVal)
                    opsCounter += 1
                    grid[rowCursor + 1][colIndex] = currentVal
                ENDIF
                rowCursor += 1
            ENDWHILE

            LET checkRow TO 0
            REPEAT UNTIL checkRow = rowCount
                IF (colIndex < colCount - 1) AND (grid[checkRow][colIndex] = grid[checkRow][colIndex + 1])
                    opsCounter = opsCounter + 1
                    LET newValue TO 0
                    WHILE newValue <= 9
                        IF NOT (newValue = grid[checkRow][colIndex])
                            grid[checkRow][colIndex + 1] = newValue
                            EXIT WHILE
                        ENDIF
                        newValue += 1
                    ENDWHILE
                ENDIF
                checkRow = checkRow + 1
            ENDREPEAT

            colIndex += 1
        ENDWHILE

        RETURN opsCounter
    END FUNCTION
END CLASS
