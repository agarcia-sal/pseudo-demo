CLASS Solution
    FUNCTION countSubstrings(s c)
        INITIALIZE total TO 0
        ASSIGN length TO LENGTH(s)
        INITIALIZE occurrence TO 0
        INITIALIZE index TO 0
        WHILE index < length
            IF s[index] = c THEN
                occurrence ← occurrence + 1
            END IF
            index ← index + 1
        END WHILE
        SET temp TO occurrence + 1
        SET result TO (occurrence * temp) / 2
        RETURN result
    END FUNCTION
END CLASS
