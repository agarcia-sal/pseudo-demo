CLASS Solution
    FUNCTION occurrencesOfElement(nums, queries, x)
        SET positions TO []
        SET idx TO 0
        WHILE idx < LENGTH(nums)
            SET val TO nums[idx]
            IF NOT(val NOT EQUAL TO x)
                INSERT idx AT END OF positions
            END IF
            SET idx TO idx PLUS 1
        END WHILE
        SET results TO []
        SET qIndex TO 0
        WHILE qIndex < LENGTH(queries)
            SET req TO queries[qIndex]
            IF req <= LENGTH(positions)
                SET elementIndex TO req MINUS 1
                APPEND positions[elementIndex] TO results
            ELSE
                APPEND (-1) TO results
            END IF
            SET qIndex TO qIndex PLUS 1
        END WHILE
        RETURN results
    END FUNCTION
END CLASS
