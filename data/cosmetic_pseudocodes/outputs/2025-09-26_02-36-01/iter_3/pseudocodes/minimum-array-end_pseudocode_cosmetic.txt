CLASS Solution
    FUNCTION minEnd(n x)
        FUNCTION canConstruct(limit)
            SET val_tracker TO x
            SET total_found TO 1
            WHILE val_tracker NOT GREATER_OR_EQUAL limit
                SET val_tracker TO val_tracker + 1
                IF (val_tracker BITWISE_AND x) == x THEN
                    total_found = total_found + 1
                    IF total_found == n THEN
                        RETURN True
                    END IF
                END IF
            END WHILE
            RETURN total_found == n
        END FUNCTION

        SET boundary_left TO x
        SET exponent_base TO 10
        SET exponent_result TO 1
        SET exponent_times TO 9
        FOR i FROM 1 TO exponent_times
            exponent_result = exponent_result * exponent_base
        END FOR
        SET boundary_right TO 2 * exponent_result

        WHILE boundary_left < boundary_right
            SET middle_point TO (boundary_left + boundary_right) DIV 2
            IF canConstruct(middle_point)
                SET boundary_right TO middle_point
            ELSE
                SET boundary_left TO boundary_left + 1
            END IF
        END WHILE

        RETURN boundary_left
    END FUNCTION
END CLASS
