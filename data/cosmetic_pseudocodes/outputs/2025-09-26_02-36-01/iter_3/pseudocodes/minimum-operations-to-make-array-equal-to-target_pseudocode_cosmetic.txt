CLASS Solution
    FUNCTION minimumOperations(nums, target)
        DEFINE size = LENGTH(nums)
        DEFINE totalDiff = 0
        SET totalDiff = ABS(target[0] - nums[0])
        SET index = 1
        WHILE index < size
            DEFINE currDiff = target[index] + (-1 * nums[index])
            DEFINE prevDiff = target[index - 1] - nums[index - 1]
            IF (currDiff * prevDiff) > 0 THEN
                DEFINE diffVal = ABS(ABS(currDiff) - ABS(prevDiff))
                IF diffVal > 0 THEN
                    totalDiff = totalDiff + diffVal
                END IF
            ELSE
                totalDiff = totalDiff + ABS(currDiff)
            END IF
            index = index + 1
        END WHILE
        RETURN totalDiff
    END FUNCTION
END CLASS
