CLASS Solution
    FUNCTION maximumLength(nums k)
      LET totalElements BE LENGTH(nums)
      IF totalElements IS EQUAL TO 1 THEN RETURN 1 END IF

      LET dpArray BE ARRAY OF SIZE totalElements FILLED WITH EMPTY DICTIONARIES

      LET longestChain BE 1

      LET idxOuter BE 0
      WHILE idxOuter < totalElements
        LET idxInner BE 0
        WHILE idxInner < idxOuter
          LET currentSum BE nums[idxOuter] + nums[idxInner]
          LET modValue BE currentSum - (k * (currentSum DIV k))
          IF modValue IN dpArray[idxInner] THEN
            dpArray[idxOuter][modValue] = dpArray[idxInner][modValue] + 1
          ELSE
            dpArray[idxOuter][modValue] = 2
          END IF

          IF dpArray[idxOuter][modValue] > longestChain THEN
            longestChain = dpArray[idxOuter][modValue]
          END IF
          idxInner = idxInner + 1
        END WHILE
        idxOuter = idxOuter + 1
      END WHILE

      RETURN longestChain
    END FUNCTION
END CLASS
