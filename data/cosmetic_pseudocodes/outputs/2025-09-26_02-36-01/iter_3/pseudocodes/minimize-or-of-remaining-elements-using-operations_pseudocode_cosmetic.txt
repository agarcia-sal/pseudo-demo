CLASS Solution
    FUNCTION minOrAfterOperations(nums k)
        FUNCTION canAchieve(expectation k)
            SET accumulator TO -1
            SET count TO 0
            SET index TO 0
            WHILE index LESS THAN LENGTH OF nums
                SET element TO nums AT index
                IF accumulator IS EQUAL TO -1
                    SET accumulator TO element
                ELSE
                    SET accumulator TO accumulator BITWISE_AND element
                END IF
                IF (accumulator BITWISE_AND expectation) IS EQUAL TO 0
                    SET accumulator TO -1
                ELSE
                    SET count TO count + 1
                    IF count GREATER THAN k
                        RETURN false
                    END IF
                END IF
                SET index TO index + 1
            END WHILE
            RETURN true
        END FUNCTION

        SET limit TO (1 LEFT_SHIFT 30) - 1
        SET candidate TO limit
        SET position TO 0
        WHILE position LESS THAN 30
            SET mask TO 1 LEFT_SHIFT position
            IF (candidate BITWISE_AND mask) EQUALS 0
                SET position TO position + 1
                CONTINUE
            END IF
            IF canAchieve((NOT candidate) XOR mask k)
                SET candidate TO candidate BITWISE_AND (NOT mask)
            END IF
            SET position TO position + 1
        END WHILE
        RETURN candidate
    END FUNCTION
END CLASS
