CLASS Solution
    FUNCTION minimumCost(m n horizontalCut verticalCut)
        DECLARE totalCost AS 0
        SET idxH TO 0
        SET idxV TO 0
        SET heightParts TO 1
        SET widthParts TO 1
        SORT horizontalCut USING a comparator that orders from greatest to least
        SORT verticalCut USING a comparator that orders from greatest to least
        WHILE NOT (idxH >= LENGTH(horizontalCut) AND idxV >= LENGTH(verticalCut))
            IF idxV = LENGTH(verticalCut) OR (idxH < LENGTH(horizontalCut) AND horizontalCut[idxH] > verticalCut[idxV])
                SET increment TO horizontalCut[idxH] * widthParts
                SET totalCost TO totalCost + increment
                heightParts = heightParts + 1
                idxH = idxH + 1
            ELSE
                SET increment TO verticalCut[idxV] * heightParts
                totalCost += increment
                widthParts = widthParts + 1
                idxV = idxV + 1
            END IF
        END WHILE
        RETURN totalCost
    END FUNCTION
END CLASS
