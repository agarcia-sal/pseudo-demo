CLASS Solution  
    FUNCTION largestSquareArea(blPoints, trPoints)  
        FUNCTION intersecting_square_area(blA, trA, blB, trB)  
            SET x_left TO blA[0]  
            IF blB[0] GREATER THAN x_left  
                SET x_left TO blB[0]  
            END IF  
            SET x_right TO trA[0]  
            IF trB[0] LESS THAN x_right  
                SET x_right TO trB[0]  
            END IF  
            SET y_bottom TO blA[1]  
            IF blB[1] GREATER THAN y_bottom  
                SET y_bottom TO blB[1]  
            END IF  
            SET y_top TO trA[1]  
            IF trB[1] LESS THAN y_top  
                SET y_top TO trB[1]  
            END IF  

            IF x_left GREATER THAN OR EQUAL TO x_right  
                RETURN 0  
            END IF  
            IF y_bottom GREATER THAN OR EQUAL TO y_top  
                RETURN 0  
            END IF  

            SET width_x TO x_right  
            SET width_x SUBTRACT x_left  
            SET height_y TO y_top  
            SET height_y SUBTRACT y_bottom  

            SET edge_length TO width_x  
            IF height_y LESS THAN edge_length  
                SET edge_length TO height_y  
            END IF  

            RETURN edge_length TIMES edge_length  
        END FUNCTION  

        SET length_points TO bottomLeft.LENGTH()  
        SET greatest_area TO 0  
        SET outer_index TO 0  
        WHILE outer_index LESS THAN length_points MINUS 1  
            SET inner_index TO outer_index PLUS 1  
            WHILE inner_index LESS THAN length_points  
                SET current_area TO intersecting_square_area(bottomLeft[outer_index], topRight[outer_index], bottomLeft[inner_index], topRight[inner_index])  
                IF current_area GREATER THAN greatest_area  
                    SET greatest_area TO current_area  
                END IF  
                SET inner_index TO inner_index PLUS 1  
            END WHILE  
            SET outer_index TO outer_index PLUS 1  
        END WHILE  
        RETURN greatest_area  
    END FUNCTION  
END CLASS
