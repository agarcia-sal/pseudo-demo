CLASS Solution
    FUNCTION maxTotalReward(rewardValues)
        SET uniqueValues TO rewardValues CONVERTED TO A COLLECTION WITHOUT DUPLICATES AND SORTED
        SET flagMask TO 0b1
        SET index TO 0
        WHILE index LESS THAN LENGTH OF uniqueValues DO
            SET value TO uniqueValues[index]
            SET shiftedOne TO (1 COMBINED WITH 0) LEFT SHIFTED BY value
            SET tempMask TO flagMask BITWISE-AND (shiftedOne MINUS 1 LEFT SHIFTED BY value)
            SET flagMask TO flagMask BITWISE-OR tempMask
            SET index TO index PLUS 1
        END WHILE
        RETURN BIT COUNT OF flagMask MINUS 1
    END FUNCTION
END CLASS
