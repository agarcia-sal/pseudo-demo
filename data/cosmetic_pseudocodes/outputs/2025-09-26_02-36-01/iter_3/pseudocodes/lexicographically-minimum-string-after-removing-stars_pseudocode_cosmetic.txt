CLASS Solution
    FUNCTION clearStars(s)
        SET accumulator TO empty list
        SET index TO 0
        WHILE index < LENGTH(s)
            SET current_char TO s[index]
            IF NOT(current_char != '*') THEN
                IF LENGTH(accumulator) <> 0 THEN
                    REMOVE accumulator[LENGTH(accumulator) - 1]
                END IF
            ELSE
                SET length_before TO LENGTH(accumulator)
                accumulator += [current_char]
                SET length_after TO LENGTH(accumulator)
            END IF
            SET index TO index + 1
        END WHILE
        SET output TO ""
        SET pos TO 0
        WHILE pos < LENGTH(accumulator)
            SET output TO output + accumulator[pos]
            SET pos TO pos + 1
        END WHILE
        RETURN output
    END FUNCTION
END CLASS
