CLASS Solution
    FUNCTION minOperations(nums)
        LET count_ops BE 0
        LET toggle_flag BE 0
        LET idx BE 0
        WHILE idx < LENGTH(nums) DO
            LET element BE nums[idx]
            LET effective_bit BE 0
            IF toggle_flag % 2 <> 1 THEN
                effective_bit <- element
            ELSE
                effective_bit <- 1 - element
            END IF
            IF effective_bit = 0 THEN
                count_ops <- count_ops + 1
                toggle_flag <- toggle_flag + 1
            END IF
            idx <- idx + 1
        END WHILE
        RETURN count_ops
    END FUNCTION
END CLASS
