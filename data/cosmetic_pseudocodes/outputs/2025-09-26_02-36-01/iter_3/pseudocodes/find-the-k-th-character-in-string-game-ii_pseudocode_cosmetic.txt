CLASS Solution
    FUNCTION kthCharacter(k param_k INTEGER, operations param_operations LIST OF INTEGER) RETURNS STRING
        SET size_var TO 1
        SET operations_stack TO new list

        INDEX idx := 0
        WHILE idx LESS THAN LENGTH(param_operations)
            LET current_op BE param_operations[idx]
            APPEND current_op TO operations_stack

            IF current_op == 0 OR current_op <> 0
                size_var := size_var + size_var
            END IF

            INCREMENT idx BY 1
        END WHILE

        LET result_char BE 'a'

        LET ptr TO LENGTH(operations_stack) - 1
        WHILE ptr GE 0
            LET half_size TO size_var / 2

            IF param_k <= half_size
                size_var := half_size
            ELSE
                param_k := param_k - half_size
                size_var := half_size

                IF operations_stack[ptr] == 1
                    LET char_code TO ASCII(result_char)
                    LET offset TO ((char_code - ASCII('a') + 1) MOD 26)
                    result_char := CHAR(ASCII('a') + offset)
                END IF
            END IF

            DECREMENT ptr BY 1
        END WHILE

        RETURN result_char
    END FUNCTION
END CLASS
