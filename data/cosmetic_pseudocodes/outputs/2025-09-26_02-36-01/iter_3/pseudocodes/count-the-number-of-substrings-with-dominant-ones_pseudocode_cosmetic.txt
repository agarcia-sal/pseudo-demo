CLASS Solution
    FUNCTION numberOfSubstrings(s)
        LET lengthInverse = 0
        LET totalSubstrings = 0
        WHILE lengthInverse < LENGTH s
            LET countOne = 0
            LET countZero = 0
            LET forwardIndex = lengthInverse
            WHILE forwardIndex <= LENGTH s - 1
                LET currentChar = s[forwardIndex]
                IF currentChar = '1' THEN
                    countOne = countOne + 1
                ELSE
                    countZero = countZero + 1
                ENDIF
                IF countOne >= countZero * countZero THEN
                    totalSubstrings = totalSubstrings + 1
                ENDIF
                forwardIndex = forwardIndex + 1
            ENDWHILE
            lengthInverse = lengthInverse + 1
        ENDWHILE
        RETURN totalSubstrings
    END FUNCTION
END CLASS
