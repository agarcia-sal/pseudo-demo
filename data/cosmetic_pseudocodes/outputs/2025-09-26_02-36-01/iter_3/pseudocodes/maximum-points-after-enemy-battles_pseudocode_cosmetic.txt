CLASS Solution  
    FUNCTION maximumPoints(enemyEnergies totalEnergy)  
        CALL enemyEnergies.SORT()  
        IF NOT (totalEnergy >= enemyEnergies[0]) THEN  
            RETURN 0  
        END IF  
        SET score = 0  
        SET idx = enemyEnergies.LENGTH - 1  
        WHILE idx >= 0 DO  
            SET quotient = totalEnergy DIV enemyEnergies[0]  
            SET score = score + quotient  
            SET totalEnergy = totalEnergy - quotient * enemyEnergies[0]  
            SET totalEnergy = totalEnergy + enemyEnergies[idx]  
            SET idx = idx - 1  
        END WHILE  
        RETURN score  
    END FUNCTION  
END CLASS
