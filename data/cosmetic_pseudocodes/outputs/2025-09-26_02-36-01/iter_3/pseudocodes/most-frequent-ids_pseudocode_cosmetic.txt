CLASS Solution
    FUNCTION mostFrequentIDs(nums PARAMETER, freq PARAMETER)
        SET tally TO an empty map defaulting to 0
        SET heap_container TO an empty list
        SET output TO an empty list
        SET idx TO 0
        WHILE idx < LENGTH(nums)
            SET element TO nums[idx]
            SET increment TO freq[idx]
            SET tally[element] TO tally[element] + increment
            ADD (-1 * tally[element], element) TO heap_container
            WHILE LENGTH(heap_container) > 0 AND (-1 * GET_FIRST(heap_container)[0]) != tally[GET_FIRST(heap_container)[1]]
                REMOVE_FIRST(heap_container)
            END WHILE
            IF LENGTH(heap_container) > 0 THEN
                SET top_pair TO GET_FIRST(heap_container)
                SET current_max TO -top_pair[0]
                APPEND current_max TO output
            ELSE
                APPEND 0 TO output
            END IF
            SET idx TO idx + 1
        END WHILE
        RETURN output
    END FUNCTION
END CLASS
