CLASS Solution
    FUNCTION numberOfPairs(nums1, nums2, k)
        DEFINE freq_map = empty dictionary
        INDEX i = 0
        WHILE i < length of nums2
            temp_val = nums2 at position i
            IF temp_val in freq_map
                freq_map[temp_val] = freq_map[temp_val] plus 1
            ELSE 
                freq_map[temp_val] = 1
            END IF
            i = i + 1
        END WHILE

        SET total_pairs TO 0
        INDEX j = 0
        WHILE j < size of nums1
            current_val = nums1 at index j
            FOR keyVal, times IN freq_map as entries
                product_val = keyVal multiplied by k
                div_result = current_val modulo product_val
                BOOL is_divisible = (div_result EQ zero)
                IF is_divisible
                    total_pairs = total_pairs plus times
                END IF
            END FOR
            j = j plus 1
        END WHILE

        RETURN total_pairs
    END FUNCTION
END CLASS
