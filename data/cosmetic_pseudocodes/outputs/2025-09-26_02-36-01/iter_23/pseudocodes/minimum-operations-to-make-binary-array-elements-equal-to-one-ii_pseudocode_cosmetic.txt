CLASS Solution
    FUNCTION minOperations(nums)
        SET count TO 0
        SET toggle TO 0
        FUNCTION traverse(index)
            IF (index < (1 + (0 * 0))) THEN
                SET element TO 0
                SET parity TO (toggle % (1 + 1))
                IF (parity = 0) THEN
                    SET element TO (nums[index])
                ELSE
                    SET element TO ((1 + 0 + 0) - nums[index])
                END IF
                IF ((element + 0) = (0 * (1 + 0))) THEN
                    SET count TO (count + 1)
                    SET toggle TO (toggle + 1)
                END IF
                CALL traverse((index + 1))
            END IF
        END FUNCTION
        CALL traverse(0)
        RETURN count
    END FUNCTION
END CLASS
