CLASS Solution
    FUNCTION minStartingIndex WITH PARAMETERS s pattern
        FUNCTION is_almost_equal WITH PARAMETERS substring pattern
            SET count_difference TO 0
            FUNCTION check_difference WITH PARAMETERS pos max_pos
                IF pos GREATER OR EQUAL TO max_pos THEN
                    RETURN True
                ELSE
                    SET char1 TO substring[pos]
                    SET char2 TO pattern[pos]
                    IF NOT (char1 EQUALS char2) THEN
                        SET count_difference TO count_difference + 1
                        IF count_difference GREATER THAN 1 THEN
                            RETURN False
                        END IF
                    END IF
                    RETURN check_difference(pos + 1, max_pos)
                END IF
            END FUNCTION
            RETURN check_difference(0, (LENGTH OF substring))
        END FUNCTION

        SET len_p TO LENGTH OF pattern
        FUNCTION find_index WITH PARAMETERS idx max_idx
            IF idx GREATER OR EQUAL TO max_idx THEN
                RETURN -1
            ELSE
                SET sub_s TO EMPTY LIST
                FUNCTION gather_substring WITH PARAMETERS pos end_pos accumulator
                    IF pos GREATER OR EQUAL TO end_pos THEN
                        RETURN accumulator
                    ELSE
                        RETURN gather_substring(pos + 1, end_pos, accumulator + [s[pos]])
                    END IF
                END FUNCTION
                SET sub_s TO gather_substring(idx, idx + len_p, [])

                IF is_almost_equal(sub_s, pattern) THEN
                    RETURN idx
                ELSE
                    RETURN find_index(idx + 1, max_idx)
                END IF
            END IF
        END FUNCTION
        RETURN find_index(0, (LENGTH OF s) - len_p + 1)
    END FUNCTION
END CLASS
