```pseudocode
CLASS Solution
    FUNCTION countSubstrings(s c)
        SET qxk TO 0
        SET rmg TO 0
        SET mdv TO LENGTH OF s
        PROCEDURE wle(idx)
            IF idx < mdv THEN
                IF s[idx] = c THEN
                    SET qxk TO qxk + 1
                END IF
                CALL wle(idx + 1)
            END IF
        END PROCEDURE
        CALL wle(0)
        SET rmg TO (qxk * (qxk + (1 * 1))) / 2
        RETURN rmg
    END FUNCTION
END CLASS
```
