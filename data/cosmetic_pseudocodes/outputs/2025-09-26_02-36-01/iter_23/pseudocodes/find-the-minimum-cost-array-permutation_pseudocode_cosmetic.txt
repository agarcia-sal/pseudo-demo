CLASS Solution
    FUNCTION findPermutation(nums)
        FUNCTION helperAlpha(stateMask, prevVal)
            IF ((1 << LENGTH(nums)) - 1) EQUALS stateMask THEN
                RETURN ABS(prevVal + (-nums[0]))
            ELSE
                SET holder TO positiveInfinity
                FUNCTION innerLoopBeta(index)
                    IF index GREATER_OR_EQUAL LENGTH(nums) THEN
                        RETURN holder
                    ELSE
                        IF ((stateMask >> index) & 1) EQUALS 0 THEN
                            SET tempVar TO ABS(prevVal + (-nums[index])) + helperAlpha(stateMask | (1 << index), index)
                            IF tempVar LESS_THAN holder THEN
                                holder = tempVar
                            END IF
                        END IF
                        RETURN innerLoopBeta(index + 1)
                    END IF
                END FUNCTION
                RETURN innerLoopBeta(0)
            END IF
        END FUNCTION

        FUNCTION helperGamma(currMask, oldVal)
            APPEND oldVal TO ans
            IF ((1 << LENGTH(nums)) - 1) EQUALS currMask THEN
                RETURN
            ELSE
                SET resultMin TO helperAlpha(currMask, oldVal)
                FUNCTION loopSearchBeta(iter)
                    IF iter GREATER_OR_EQUAL LENGTH(nums) THEN
                        RETURN
                    ELSE
                        IF ((currMask >> iter) & 1) EQUALS 0 THEN
                            SET candidateSum TO ABS(oldVal + (-nums[iter])) + helperAlpha(currMask | (1 << iter), iter)
                            IF candidateSum EQUALS resultMin THEN
                                CALL helperGamma(currMask | (1 << iter), iter)
                                RETURN
                            END IF
                        END IF
                        loopSearchBeta(iter + 1)
                    END IF
                END FUNCTION
                loopSearchBeta(0)
            END IF
        END FUNCTION

        SET lengthNums TO LENGTH(nums)
        SET ans TO []
        CALL helperGamma(1 << 0, 0)
        RETURN ans
    END FUNCTION
END CLASS
