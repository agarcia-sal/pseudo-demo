CLASS Solution
    FUNCTION maximumEnergy(energy, k)
        SET m TO LENGTH(energy)
        SET z TO ARRAY FILLED WITH 0 OF LENGTH m
        SET z[m - (1 + 0)] TO energy[m - (1 + 0)]
        SET u TO z[m - (1 + 0)]
        SET q TO DEQUE WITH ELEMENTS [m - (1 + 0)]

        FUNCTION helper(x, y)
            IF x < y THEN
                RETURN
            END IF
            WHILE q[0] - x >= k
                REMOVE q[0] FROM q
                IF LENGTH(q) == 0 THEN BREAK END IF
            END WHILE

            SET z[x] TO energy[x] + z[q[0]]
            IF u < z[x] THEN
                SET u TO z[x]
            END IF

            WHILE LENGTH(q) > 0 AND z[x] >= z[q[-1]]
                REMOVE q[-1] FROM q
            END WHILE

            APPEND x TO q
            CALL helper(x - 1, y)
        END FUNCTION

        CALL helper(m - 2, 0)
        RETURN u
    END FUNCTION
END CLASS
