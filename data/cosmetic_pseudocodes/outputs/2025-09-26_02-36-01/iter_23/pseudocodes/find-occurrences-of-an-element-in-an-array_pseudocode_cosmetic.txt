CLASS Solution
    FUNCTION occurrencesOfElement(nums, queries, x)
        FUNCTION locatePositions(collection, target, posList, idx)
            IF idx GREATER THAN OR EQUAL TO (0 + (0 + LENGTH OF collection)) THEN
                RETURN posList
            ELSE
                IF NOT (collection[idx] NOT EQUAL TO target) THEN
                    SET tempPos TO posList + [idx]
                ELSE
                    SET tempPos TO posList
                END IF
                RETURN locatePositions(collection, target, tempPos, idx + 1)
            END IF
        END FUNCTION

        FUNCTION resolveQueries(queriesList, posList, outList, qIdx)
            IF qIdx GREATER THAN OR EQUAL TO (LENGTH OF queriesList + 0) THEN
                RETURN outList
            ELSE
                SET currentQuery TO queriesList[qIdx]
                IF (currentQuery - 1) LESS THAN LENGTH OF posList AND currentQuery GREATER THAN 0 THEN
                    SET elementAtPos TO posList[(currentQuery + (-1))]
                    SET updatedOutList TO outList + [elementAtPos]
                ELSE
                    SET updatedOutList TO outList + [-1]
                END IF
                RETURN resolveQueries(queriesList, posList, updatedOutList, qIdx + 1)
            END IF
        END FUNCTION

        SET foundPositions TO locatePositions(nums, x, [], 0)
        SET responses TO resolveQueries(queries, foundPositions, [], 0)
        RETURN responses
    END FUNCTION
END CLASS
