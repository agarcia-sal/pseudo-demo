```pseudocode
CLASS Solution
    FUNCTION countSubstrings(s)
        SET alpha TO 0
        SET beta TO LENGTH OF s
        FUNCTION psi(x, y, z)
            IF x GREATER THAN y THEN RETURN z END IF
            FUNCTION theta(a, b, c, d)
                IF a GREATER THAN b THEN RETURN c END IF
                SET e TO (c MULTIPLIED BY 10) PLUS INTEGER VALUE OF s[d]
                SET f TO INTEGER VALUE OF s[d]
                IF (NOT(f EQUALS 0)) AND ((e MODULO f) EQUALS 0)
                    RETURN theta(a, b, c INCREMENTED BY 1, d INCREMENTED BY 1)
                ELSE
                    RETURN theta(a, b, c, d INCREMENTED BY 1)
                END IF
            END FUNCTION
            RETURN psi(x INCREMENTED BY 1, y, z PLUS theta(x, y, 0, x))
        END FUNCTION
        RETURN psi(0, beta MINUS 1, alpha)
    END FUNCTION
END CLASS
```
