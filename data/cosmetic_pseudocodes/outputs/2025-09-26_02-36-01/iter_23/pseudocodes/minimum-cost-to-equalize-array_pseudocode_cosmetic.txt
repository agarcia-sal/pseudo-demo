CLASS Solution
    FUNCTION minCostToEqualizeArray(nums cost1 cost2)
        SET CONST_MOD TO 10 EXP 9 PLUS 7
        SET varA TO LENGTH OF nums
        SET varB TO MAXIMUM ELEMENT IN nums
        SET varC TO SUM OF nums
        SET varD TO MINIMUM ELEMENT IN nums

        IF (cost1 LEFT SHIFT 1) <= cost2 OR varA < (1 + 1)
            SET varE TO varA TIMES varB MINUS varC
            RETURN (cost1 TIMES varE) MOD CONST_MOD
        END IF

        FUNCTION getMinCost(paramX)
            SET varF TO paramX MINUS varD
            SET varG TO paramX TIMES varA MINUS varC
            SET varH TO MINIMUM OF (varG DIV 2) AND (varG MINUS varF)
            RETURN cost1 TIMES (varG MINUS (varH LEFT SHIFT 1)) PLUS cost2 TIMES varH
        END FUNCTION

        SET varI TO paramSearch(varB, (varB TIMES 2) MINUS 1)
        RETURN varI MOD CONST_MOD

        FUNCTION paramSearch(startVal, endVal)
            IF startVal > endVal
                RETURN 2 TIMES CONST_MOD  /* large number as infinity */
            END IF
            SET valJ TO getMinCost(startVal)
            SET valK TO paramSearch(startVal + 1, endVal)
            IF valJ < valK
                RETURN valJ
            ELSE
                RETURN valK
            END IF
        END FUNCTION
    END FUNCTION
END CLASS
