```pseudocode
CLASS Solution
    FUNCTION findWinningPlayer(skills, k)
        SET size TO LENGTH OF skills
        DEFINE FUNCTION dequeueFirst(L)
            SET firstElem TO L[0]
            REMOVE ELEMENT AT INDEX 0 FROM L
            RETURN firstElem
        END FUNCTION
        SET idxQueue TO EMPTY LIST
        SET iterIndex TO 0
        WHILE iterIndex LESS THAN size
            APPEND iterIndex TO idxQueue
            SET iterIndex TO iterIndex PLUS 1
        END WHILE
        SET countWins TO 0
        SET champ TO dequeueFirst(idxQueue)
        FUNCTION conditionCheck(winsVal, qList)
            IF winsVal GREATER OR EQUAL TO k OR LENGTH OF qList EQUALS 0
                RETURN true
            ELSE
                RETURN false
            END IF
        END FUNCTION
        WHILE NOT conditionCheck(countWins, idxQueue)
            SET challenger TO dequeueFirst(idxQueue)
            IF (skills[champ] MINUS skills[challenger]) GREATER THAN 0
                SET countWins TO countWins PLUS 1
                APPEND challenger TO idxQueue
            ELSE
                SET countWins TO 1
                APPEND champ TO idxQueue
                SET champ TO challenger
            END IF
        END WHILE
        RETURN champ
    END FUNCTION
END CLASS
```
