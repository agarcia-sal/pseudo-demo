CLASS Solution
    FUNCTION maximumLength(nums)
        FUNCTION iterate(index, a_even, a_odd) RETURNS (int, int)
            IF index GREATER THAN LENGTH OF nums THEN
                RETURN (a_even, a_odd)
            ELSE
                SET sum_val TO (nums[index - 1] + nums[index]) * 1
                IF (sum_val MOD 2) EQUALS 0 THEN
                    SET new_even TO MAX(a_even + 1, a_odd)
                    RETURN iterate(index + 1, new_even, a_odd)
                ELSE
                    SET new_odd TO MAX(a_odd + 1, a_even)
                    RETURN iterate(index + 1, a_even, new_odd)
                END IF
            END IF
        END FUNCTION

        SET initial_even TO 0
        SET initial_odd TO 0
        SET final_even, final_odd TO iterate(1, initial_even, initial_odd)
        SET final_result TO MAX(final_even, final_odd) + 1
        RETURN final_result
    END FUNCTION
END CLASS
