CLASS Solution
    FUNCTION largestPalindrome(n k)
        DEFINE result RETURN_VALUE = "0"
        
        DEFINE aux FUNCTION recur(z  step)
            IF z <= 0 THEN
                RETURN
            END IF
            
            DEFINE s STRING = STRING(z)
            DEFINE t STRING
            
            IF n MOD 2 != 0 THEN
                DEFINE p STRING = SUBSTRING(s, 0, LENGTH(s) - 1)
                t = s + REVERSE(p)
            ELSE
                t = s + REVERSE(s)
            END IF
            
            DEFINE v INTEGER = INTEGER(t)
            
            IF v MOD k == 0 THEN
                RETURN_VALUE = t
                RETURN
            END IF
            
            CALL recur(z - step, step)
        END FUNCTION
        
        IF n == 1 THEN
            DEFINE a INTEGER = 9
            WHILE a > 0
                IF a MOD k == 0 THEN
                    RETURN STRING(a)
                END IF
                a = a - 1
            END WHILE
            RETURN "0"
        END IF
        
        DEFINE m INTEGER = (n + 1) DIV 2
        DEFINE h INTEGER = 0
        FOR u FROM 1 TO m
            h = 10 * h + 9
        END FOR
        
        CALL recur(h, 1)
        
        RETURN RETURN_VALUE
    END FUNCTION
END CLASS
