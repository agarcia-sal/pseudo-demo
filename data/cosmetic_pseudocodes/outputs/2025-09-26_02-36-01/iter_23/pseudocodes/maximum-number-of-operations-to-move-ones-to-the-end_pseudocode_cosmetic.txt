CLASS Solution
    FUNCTION maxOperations(s)
        SET p TO 0
        SET q TO 0

        FUNCTION proc(r)
            IF r >= LENGTH(s) THEN
                RETURN
            ELSE 
                SET u TO s[r]
                IF NOT (u <> '1') THEN
                    SET q TO q + 1
                ELSE 
                    IF (r <> 0) AND (NOT (NOT (s[r - 1] = '1'))) THEN
                        SET p TO p + q
                    END IF
                END IF
                CALL proc(r + 1)
            END IF
        END FUNCTION

        CALL proc(0)
        RETURN p
    END FUNCTION
END CLASS
