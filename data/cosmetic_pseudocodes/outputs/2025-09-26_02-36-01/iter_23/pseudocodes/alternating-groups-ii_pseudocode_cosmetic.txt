CLASS Solution
    FUNCTION numberOfAlternatingGroups(colors k)
        DECLARE result INTEGER
        DECLARE m INTEGER
        DECLARE expandedColors ARRAY
        DECLARE pos INTEGER
        DECLARE valid BOOLEAN
        DECLARE offset INTEGER

        result <- 0
        IF NOT (k >= 3) THEN
            result <- 0
        ELSE
            m <- LENGTH(colors)
            expandedColors <- CONCAT(colors, SLICE(colors, 0, k - 1))

            pos <- 0
            WHILE TRUE DO
                IF NOT (pos < m) THEN
                    EXIT
                END IF

                valid <- true
                offset <- 1
                WHILE TRUE DO
                    IF NOT (offset < k) THEN
                        EXIT
                    END IF

                    IF NOT (expandedColors[pos + offset - 1] != expandedColors[pos + offset]) THEN
                        valid <- false
                        EXIT
                    END IF

                    offset <- offset + 1
                END WHILE

                IF valid = true THEN
                    result <- result + 1
                END IF
                pos <- pos + 1
            END WHILE
        END IF
        RETURN result
    END FUNCTION
END CLASS
