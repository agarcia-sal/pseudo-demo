CLASS Solution
    FUNCTION countAlternatingSubarrays(nums)
        SET totalElements TO LENGTH OF nums
        IF totalElements EQUALS 0 THEN
            SET result TO 0
        ELSE
            SET result TO totalElements
            SET sequenceLength TO 1

            FUNCTION recur(j)
                IF j GREATER THAN OR EQUAL TO totalElements THEN
                    RETURN
                END IF
                IF NOT (nums[j] EQUALS nums[j - 1]) THEN
                    SET sequenceLength TO sequenceLength + 1
                ELSE
                    SET sequenceLength TO 1
                END IF
                SET result TO result + (sequenceLength - 1)
                CALL recur(j + 1)
            END FUNCTION

            CALL recur(1)
        END IF
        RETURN result
    END FUNCTION
END CLASS
