```pseudocode
CLASS Solution
    FUNCTION maxScore(nums)
        SET uvkqitp TO LENGTH OF nums
        SET xfalrme TO list containing uvkqitp elements each set to 0

        FUNCTION zbyrna(a)
            IF a LESS THAN 0 THEN
                RETURN
            END IF
            SET mqxrh TO 0
            FUNCTION ojtzfw(b)
                IF b GREATER OR EQUAL TO uvkqitp THEN
                    RETURN
                END IF
                SET yqwzolh TO (b - a) * nums[b]
                SET sthxqv TO yqwzolh + xfalrme[b]
                IF sthxqv GREATER THAN mqxrh THEN
                    SET mqxrh TO sthxqv
                END IF
                ojtzfw(b + 1)
            END FUNCTION
            ojtzfw(a + 1)
            SET xfalrme[a] TO mqxrh
            zbyrna(a - 1)
        END FUNCTION

        zbyrna(uvkqitp - 2)

        RETURN xfalrme[0]
    END FUNCTION
END CLASS
```
