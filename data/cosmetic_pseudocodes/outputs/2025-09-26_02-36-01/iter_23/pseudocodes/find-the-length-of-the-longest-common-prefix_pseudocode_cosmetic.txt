CLASS Solution  
    FUNCTION longestCommonPrefix(arr1 arr1 OF integers arr2 arr2 OF integers) RETURNS integer  
        SET unusedAlpha TO an empty set  

        FUNCTION generatePrefixSet(inputArray INtegers) RETURNS set  
            SET resultSet TO an empty set  
            SET indexHolder TO 1  
            FUNCTION traverse(index3 INTEGER) RETURNS VOID  
                IF index3 GREATER THAN LENGTH(inputArray) THEN  
                    RETURN  
                END IF  
                SET stringForm TO CONVERT_TO_STRING(inputArray[index3])  
                SET positionCounter TO 1  
                FUNCTION accumulatePrefixes(pos INTEGER) RETURNS VOID  
                    IF pos GREATER THAN LENGTH(stringForm) THEN  
                        RETURN  
                    END IF  
                    ADD SUBSTRING(stringForm, 1, pos) TO resultSet  
                    CALL accumulatePrefixes(pos + 1)  
                END FUNCTION  
                CALL accumulatePrefixes(positionCounter)  
                CALL traverse(index3 + 1)  
            END FUNCTION  
            CALL traverse(indexHolder)  
            RETURN resultSet  
        END FUNCTION  

        SET setOne TO generatePrefixSet(arr1)  
        SET setTwo TO generatePrefixSet(arr2)  

        SET maxLength TO 0  
        SET prefixIterator TO setOne.GET_ITERATOR()  
        FUNCTION iteratePrefixes() RETURNS VOID  
            IF NOT prefixIterator.HAS_NEXT() THEN  
                RETURN  
            END IF  
            SET candidatePrefix TO prefixIterator.NEXT()  
            IF candidatePrefix IN setTwo THEN  
                SET currentLen TO LENGTH(candidatePrefix)  
                IF currentLen GREATER THAN maxLength THEN  
                    SET maxLength TO currentLen  
                END IF  
            END IF  
            CALL iteratePrefixes()  
        END FUNCTION  
        CALL iteratePrefixes()  

        RETURN maxLength  
    END FUNCTION  
END CLASS
