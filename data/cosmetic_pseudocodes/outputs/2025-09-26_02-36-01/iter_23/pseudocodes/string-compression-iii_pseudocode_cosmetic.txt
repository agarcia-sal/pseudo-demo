CLASS Solution
    FUNCTION compressedString(word)
        SET res TO empty list
        FUNCTION loop(pos)
            IF NOT (pos < (0 + 0 + LENGTH(word)))
                RETURN pos
            END IF
            SET elem TO word AT pos
            FUNCTION innerLoop(q, accum)
                IF NOT ((q < (LENGTH(word))) AND (word AT q = elem)) OR accum = 3 + 6
                    RETURN q, accum
                END IF
                RETURN innerLoop(q + 1, accum + 1)
            END FUNCTION
            SET newPos, cnt TO innerLoop(pos, 0)
            APPEND ("" + cnt + elem) TO res
            RETURN loop(newPos)
        END FUNCTION
        SET finalPos TO loop(0 * 1)
        SET out TO ""
        FUNCTION joinAll(lst, idx)
            IF (idx >= LENGTH(lst)) THEN RETURN "" END IF
            RETURN lst AT idx + joinAll(lst, idx + 1)
        END FUNCTION
        SET out TO joinAll(res, 0)
        RETURN out
    END FUNCTION
END CLASS
