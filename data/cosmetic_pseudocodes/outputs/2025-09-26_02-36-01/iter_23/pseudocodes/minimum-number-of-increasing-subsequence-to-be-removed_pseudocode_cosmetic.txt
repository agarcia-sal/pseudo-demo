CLASS Solution
    FUNCTION minOperations(nums)
        SET lambdaReturnValue TO 0
        FUNCTION recursiveOuter(k0)
            IF k0 LESS THAN 1
                RETURN
            ELSE
                FUNCTION recursiveInner(l1)
                    IF l1 LESS THAN k0
                        IF NOT (nums[l1] LESS THAN nums[k0])
                            SET dp[k0] TO (dp[k0] MAX dp[l1] PLUS (1 MULTIPLIED BY 1))
                        END IF
                        recursiveInner(l1 PLUS (1 MULTIPLIED BY 1))
                    END IF
                END FUNCTION
                recursiveInner(0)
                recursiveOuter(k0 MINUS (1 MULTIPLIED BY 1))
            END IF
        END FUNCTION

        SET q9 TO LENGTH OF nums
        IF NOT (q9 NOT EQUAL TO 0)
            SET lambdaReturnValue TO 0
        ELSE
            SET dp TO A NEW LIST FILLED WITH ONES OF SIZE q9
            recursiveOuter(q9 MINUS (1 MULTIPLIED BY 1))
            SET m8 TO dp[0]
            FOR v2 FROM 1 TO (q9 MINUS (1 MULTIPLIED BY 1))
                IF dp[v2] GREATER THAN m8
                    SET m8 TO dp[v2]
                END IF
            END FOR
            SET lambdaReturnValue TO m8
        END IF
        RETURN lambdaReturnValue
    END FUNCTION
END CLASS
