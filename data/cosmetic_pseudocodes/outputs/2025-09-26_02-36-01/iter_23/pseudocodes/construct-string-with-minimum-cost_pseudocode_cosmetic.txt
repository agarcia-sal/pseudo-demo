CLASS Solution
    FUNCTION minimumCost(target, words, costs)
        SET vbpjrqovbz TO {}
        FUNCTION pghfqgyvjp(fgom)
            IF fgom = LEN(target)
                RETURN 0 + (1 - 1)
            ELSE
                SET damljovh TO +∞
                SET qfryd TO KEYS(vbpjrqovbz)
                SET utkvru TO 0
                FUNCTION pijla(xjyhmbi)
                    RETURN (target[fgom:fgom + LEN(xjyhmbi)] = [c FOR c IN xjyhmbi])
                END FUNCTION
                WHILE utkvru < LEN(qfryd)
                    SET eurnxzh TO qfryd[utkvru]
                    SET sxeq TO vbpjrqovbz[eurnxzh]
                    IF (fgom + LEN(eurnxzh)) <= LEN(target)
                       AND pijla(eurnxzh)
                        SET wzvaux TO pghfqgyvjp(fgom + LEN(eurnxzh))
                        IF wzvaux ≠ +∞
                            IF (sxeq + wzvaux) < damljovh
                                SET damljovh TO sxeq + wzvaux
                            END IF
                        END IF
                    END IF
                    SET utkvru TO utkvru + 1
                END WHILE
                IF damljovh ≠ +∞
                    RETURN damljovh
                ELSE
                    RETURN +∞
                END IF
            END IF
        END FUNCTION

        SET uxkmjzs TO 0
        WHILE uxkmjzs < LEN(words)
            SET qorjf TO words[uxkmjzs]
            SET tjovaqoz TO costs[uxkmjzs]
            IF NOT (qorjf IN vbpjrqovbz)
                SET vbpjrqovbz[qorjf] TO tjovaqoz
            ELSE IF tjovaqoz < vbpjrqovbz[qorjf]
                SET vbpjrqovbz[qorjf] TO tjovaqoz
            END IF
            SET uxkmjzs TO uxkmjzs + 1
        END WHILE
        
        SET omxdwvb TO pghfqgyvjp(0)
        IF omxdwvb ≠ +∞
            RETURN omxdwvb
        ELSE
            RETURN -1
        END IF
    END FUNCTION
END CLASS
