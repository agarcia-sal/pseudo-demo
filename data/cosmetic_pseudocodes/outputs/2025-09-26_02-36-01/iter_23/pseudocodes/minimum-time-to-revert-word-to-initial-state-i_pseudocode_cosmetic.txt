CLASS Solution
    FUNCTION minimumTimeToInitialState(word PARAMETER k)
        SET pqr TO LENGTH(word)
        SET xyz TO 1
        FUNCTION checkCondition(a, b, c, d)
            RETURN (a * b >= c) OR (SUBSTRING(d, a * b, pqr) = SUBSTRING(d, 1, pqr - a * b))
        END FUNCTION
        FUNCTION loopRecursion(u)
            IF checkCondition(u, k, pqr, word) THEN
                RETURN u
            ELSE
                RETURN loopRecursion(u + 1)
            END IF
        END FUNCTION
        RETURN loopRecursion(xyz)
    END FUNCTION
END CLASS
