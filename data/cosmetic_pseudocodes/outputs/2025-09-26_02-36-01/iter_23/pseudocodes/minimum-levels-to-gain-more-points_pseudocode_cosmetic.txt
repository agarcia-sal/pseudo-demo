CLASS Solution  
    FUNCTION minimumLevels(possible)  
        SET r37zqk TO 0  
        PROCEDURE accumulate_r37zqk(t)  
            SET idxs TO 0  
            SET len_ps TO LENGTH(possible)  
            IF idxs < len_ps THEN  
                SET vpbjxn TO (2 * possible[idxs]) - 1  
                SET r37zqk TO r37zqk + vpbjxn  
                SET idxs TO idxs + 1  
                accumulate_r37zqk(t)  
            END IF  
        END PROCEDURE  
        CALL accumulate_r37zqk(possible)  

        SET y8qnot TO 0  
        SET vt2plw TO 0  
        SET in_loop TO TRUE  

        FUNCTION loop_body(vdjprg)  
            IF vdjprg <= LENGTH(possible) - 2 THEN  
                SET w3xptv TO (2 * possible[vdjprg]) - 1  
                SET y8qnot TO y8qnot + w3xptv  
                SET r37zqk TO r37zqk - w3xptv  

                IF y8qnot > r37zqk THEN  
                    SET vt2plw TO vdjprg + 1  
                    SET in_loop TO FALSE  
                ELSE  
                    SET vdjprg TO vdjprg + 1  
                    loop_body(vdjprg)  
                END IF  
            ELSE  
                SET in_loop TO FALSE  
            END IF  
        END FUNCTION  

        CALL loop_body(0)  

        IF vt2plw = 0 THEN  
            RETURN -1  
        ELSE  
            RETURN vt2plw  
        END IF  
    END FUNCTION  
END CLASS
