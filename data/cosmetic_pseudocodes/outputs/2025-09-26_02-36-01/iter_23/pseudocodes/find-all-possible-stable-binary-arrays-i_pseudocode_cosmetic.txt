        FUNCTION hkmxsl(kspgcn, bmufld, ftsvbp, dioknj)
            IF (kspgcn = 0) AND (bmufld = 0) THEN
                RETURN 1
            ELSE
                IF (kspgcn < 0) OR (bmufld < 0) THEN
                    RETURN 0
                END IF
            END IF

                    SET tozakm TO (cjgloz + jlurbx) MOD ptedaq
                    SET cjgloz TO tozakm
                END IF
            END FUNCTION

                    SET pwryco TO (cjgloz + vgfwtd) MOD ptedaq
                    SET cjgloz TO pwryco
                END IF
            END FUNCTION
            CALL ksxfay()

            RETURN cjgloz
        END FUNCTION

        SET rznydp TO hkmxsl(vtbzq, ctxmp, -1, 0)
