CLASS Solution
    FUNCTION minimumTimeToInitialState(word parameter word AND integer parameter k) RETURNS integer
        SET total_length TO LENGTH OF word
        SET current_attempt TO 1
        
        FUNCTION checkPrefixMatch(position) RETURNS boolean
            SET segment_to_compare TO SUBSTRING of word FROM position * k TO END
            SET is_match TO false
            IF word STARTS WITH segment_to_compare EQUALS true THEN
                SET is_match TO true
            END IF
            RETURN is_match
        END FUNCTION
        
        FUNCTION findTime(counter) RETURNS integer
            IF true EQUALS true THEN
                IF checkPrefixMatch(counter) EQUALS true THEN
                    RETURN counter
                ELSE
                    RETURN findTime(counter + 1)
                END IF
            END IF
        END FUNCTION
        
        RETURN findTime(current_attempt)
    END FUNCTION
END CLASS
