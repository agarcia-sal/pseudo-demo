CLASS Solution
    FUNCTION minMovesToCaptureTheQueen(a b c d e f)
        VARIABLE uvwxyz INTEGER
        uvwxyz = 2
        FUNCTION checkBetween(m n o)
            RETURN (m < n AND n < o) OR (o < n AND n < m)
        END FUNCTION

        IF a = e THEN
            IF c <> a THEN
                uvwxyz = 1
            ELSE
                IF checkBetween(b d f) THEN
                    uvwxyz = 2
                ELSE
                    uvwxyz = 1
                END IF
            END IF
        ELSEIF b = f THEN
            IF d <> b THEN
                uvwxyz = 1
            ELSE
                IF checkBetween(a c e) THEN
                    uvwxyz = 2
                ELSE
                    uvwxyz = 1
                END IF
            END IF
        ELSEIF (c + d) = (e + f) THEN
            IF (a + b) <> (c + d) THEN
                uvwxyz = 1
            ELSE
                IF ( (c < a AND a < e AND d < b AND b < f) OR (e < a AND a < c AND f < b AND b < d) ) THEN
                    uvwxyz = 2
                ELSE
                    uvwxyz = 1
                END IF
            END IF
        ELSEIF (c - d) = (e - f) THEN
            IF (a - b) <> (c - d) THEN
                uvwxyz = 1
            ELSE
                IF ( (c < a AND a < e AND d > b AND b > f) OR (e < a AND a < c AND f > b AND b > d) ) THEN
                    uvwxyz = 2
                ELSE
                    uvwxyz = 1
                END IF
            END IF
        END IF
        RETURN uvwxyz
    END FUNCTION
END CLASS
