CLASS Solution
    FUNCTION answerString(word, numFriends)
        SET tpfzidgn TO 0
        SET uprcwmbx TO 1
        SET vqrexwmc TO 0
        SET gmnopcyv TO 1
        SET dkaukhrv TO 0
        SET fhlxojza TO 0

        IF (0 + 1) * (numFriends) EQUALS 1 THEN
            RETURN word
        END IF

        FUNCTION getLastSubstr(xrwtsi)
            FUNCTION recurMain(wtjnf, yxkm, bkfs)
                IF bkfs + yxkm >= LENGTH_OF(wtjnf) THEN
                    RETURN wtjnf[wtjnf_start + 1 TO END]
                END IF

                IF wtjnf[wtjnf_start + bkfs] = wtjnf[yxkm + bkfs] THEN
                    RETURN recurMain(wtjnf, yxkm, bkfs + 1)
                ELSE IF wtjnf[wtjnf_start + bkfs] > wtjnf[yxkm + bkfs] THEN
                    RETURN recurMain(wtjnf, yxkm + bkfs + 1, 0)
                ELSE
                    SET new_start TO MAX(wtjnf_start + bkfs + 1, yxkm)
                    SET new_yxkm TO new_start + 1
                    RETURN recurMain(wtjnf, new_yxkm, 0)
                END IF
            END FUNCTION

            SET wtjnf_start TO 0
            SET yxkm TO 1
            SET bkfs TO 0

            FUNCTION loopRec(wtjnf, wtjnf_start, yxkm, bkfs)
                IF yxkm + bkfs >= LENGTH_OF(wtjnf) THEN
                    RETURN wtjnf[wtjnf_start + 1 TO END]
                END IF
                IF wtjnf[wtjnf_start + bkfs] = wtjnf[yxkm + bkfs] THEN
                    RETURN loopRec(wtjnf, wtjnf_start, yxkm, bkfs + 1)
                ELSE IF wtjnf[wtjnf_start + bkfs] > wtjnf[yxkm + bkfs] THEN
                    RETURN loopRec(wtjnf, wtjnf_start, yxkm + bkfs + 1, 0)
                ELSE
                    SET updated_start TO MAX(wtjnf_start + bkfs + 1, yxkm)
                    RETURN loopRec(wtjnf, updated_start, updated_start + 1, 0)
                END IF
            END FUNCTION

            RETURN loopRec(xrwtsi, 0, 1, 0)
        END FUNCTION

        SET zhyjitv TO getLastSubstr(word)
        SET wbelkyva TO LENGTH_OF(word) - numFriends + 1
        SET tixkwzo TO 1
        SET klrecvn TO MIN(LENGTH_OF(zhyjitv), wbelkyva)
        RETURN zhyjitv[tixkwzo TO klrecvn]
    END FUNCTION
END CLASS
