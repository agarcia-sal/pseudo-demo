CLASS Solution
    FUNCTION maxScore(nums)
        FUNCTION helper(a, b, c)
            RETURN a + (b - c) * nums[b]
        END FUNCTION

        SET x1 TO LENGTH OF nums
        SET y9 TO CREATE LIST OF x1 ELEMENTS FILLED WITH 0
        SET y9[0] TO 0

        FUNCTION recur(p7)
            IF p7 > x1 - 1 THEN
                RETURN
            END IF
            SET idx_z TO 0
            FUNCTION inner_loop(q4)
                IF q4 >= p7 THEN
                    RETURN
                END IF
                IF NOT (y9[p7] >= helper(y9[q4], p7, q4)) THEN
                    SET y9[p7] TO y9[q4] + (p7 - q4) * nums[p7]
                END IF
                inner_loop(q4 + 1)
            END FUNCTION
            inner_loop(0)
            recur(p7 + 1)
        END FUNCTION

        recur(1)
        RETURN y9[x1 - 1]
    END FUNCTION
END CLASS
