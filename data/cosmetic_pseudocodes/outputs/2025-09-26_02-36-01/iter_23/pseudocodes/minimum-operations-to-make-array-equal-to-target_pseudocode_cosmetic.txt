CLASS Solution
    FUNCTION minimumOperations(nums, target)
        SET w TO 0
        SET p TO 0
        FUNCTION loopRec(k)
            IF (k > (LENGTH OF nums) - 1) THEN
                RETURN
            ELSE
                SET a TO ( (target[k]) + ( -nums[k]) )
                SET b TO ( (target[k - 1]) + (-nums[k - 1]) )
                IF ((a * b) > (0)) THEN
                    SET c TO ((ABSOLUTE VALUE OF a) + (- ABSOLUTE VALUE OF b))
                    IF (c <> 0) THEN
                        SET w TO (w + ABSOLUTE VALUE OF c)
                    END IF
                ELSE
                    SET w TO (w + ABSOLUTE VALUE OF a)
                END IF
                SET p TO (p + 1)
                CALL loopRec(p)
            END IF
        END FUNCTION
        SET w TO ABSOLUTE VALUE OF ((target[0]) + (-nums[0]))
        SET p TO 1
        CALL loopRec(p)
        RETURN w
    END FUNCTION
END CLASS
