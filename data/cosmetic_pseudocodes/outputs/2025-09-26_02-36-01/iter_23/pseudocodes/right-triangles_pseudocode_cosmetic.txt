CLASS Solution  
    FUNCTION numberOfRightTriangles(grid)  
        SET lengthX TO 0  
        SET lengthY TO 0  
        SET accumulatorZ TO 0  
        SET indexA TO 0  
        SET indexB TO 0  
        SET indexC TO 0  
        SET tempD TO 0  
        SET tempE TO 0  

        SET lengthX TO LENGTH OF grid  
        SET lengthY TO LENGTH OF element at position zero of grid  

        FUNCTION loopOverRows(currentI)  
            IF currentI GREATER THAN OR EQUAL TO lengthX THEN  
                RETURN  
            END IF  
            FUNCTION loopOverCols(currentJ)  
                IF currentJ GREATER THAN OR EQUAL TO lengthY THEN  
                    RETURN  
                END IF  
                IF (element at position currentI of grid) at position currentJ EQUALS 1 THEN  
                    SET tempD TO 0  
                    FUNCTION scanRow(currentK)  
                        IF currentK GREATER THAN OR EQUAL TO lengthY THEN  
                            RETURN  
                        END IF  
                        IF currentK NOT EQUAL currentJ THEN  
                            SET tempD TO tempD PLUS (element at position currentI of grid) at position currentK  
                        END IF  
                        CALL scanRow(currentK PLUS 1)  
                    END FUNCTION  
                    CALL scanRow(0)  

                    SET tempE TO 0  
                    FUNCTION scanCol(currentL)  
                        IF currentL GREATER THAN OR EQUAL TO lengthX THEN  
                            RETURN  
                        END IF  
                        IF currentL NOT EQUAL currentI THEN  
                            SET tempE TO tempE PLUS (element at position currentL of grid) at position currentJ  
                        END IF  
                        CALL scanCol(currentL PLUS 1)  
                    END FUNCTION  
                    CALL scanCol(0)  

                    SET accumulatorZ TO accumulatorZ PLUS (tempD MULTIPLIED BY tempE)  
                END IF  
                CALL loopOverCols(currentJ PLUS 1)  
            END FUNCTION  
            CALL loopOverCols(0)  
            CALL loopOverRows(currentI PLUS 1)  
        END FUNCTION  
        CALL loopOverRows(0)  

        RETURN accumulatorZ  
    END FUNCTION  
END CLASS
