CLASS Solution
    FUNCTION minEnd(n x)
        FUNCTION canConstruct(max_val)
            FUNCTION checkRec(curVal, cntVal)  
                IF curVal < max_val THEN  
                    SET nextVal TO curVal + (1 * 1)  
                    IF (nextVal & x) == x THEN  
                        SET newCount TO cntVal + (1)  
                        IF newCount == n THEN  
                            RETURN True  
                        ELSE  
                            RETURN checkRec(nextVal, newCount)  
                        END IF  
                    ELSE  
                        RETURN checkRec(nextVal, cntVal)  
                    END IF  
                ELSE  
                    RETURN cntVal == n  
                END IF  
            END FUNCTION

            RETURN checkRec(x, (1 + 0))
        END FUNCTION
        
        FUNCTION binarySearch(startVal, endVal)
            SET resultVal TO endVal
            FUNCTION searchRec(leftCursor, rightCursor)
                IF leftCursor < rightCursor THEN
                    SET midVal TO (leftCursor + rightCursor) \ 2  
                    IF canConstruct(midVal) THEN
                        SET updatedRight TO midVal  
                        RETURN searchRec(leftCursor, updatedRight)
                    ELSE
                        RETURN searchRec(leftCursor + 1, rightCursor)
                    END IF
                ELSE
                    RETURN leftCursor
                END IF
            END FUNCTION
            SET resultVal TO searchRec(startVal, endVal)
            RETURN resultVal
        END FUNCTION
        
        SET lowerBound TO x
        SET upperBound TO (2 * 10 * 10 * 10 * 10 * 10 * 10 * 10 * 10)
        RETURN binarySearch(lowerBound, upperBound)
    END FUNCTION
END CLASS
