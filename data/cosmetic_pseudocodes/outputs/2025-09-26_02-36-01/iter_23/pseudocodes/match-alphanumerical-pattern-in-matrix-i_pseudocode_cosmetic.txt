```pseudocode
CLASS Solution
    FUNCTION findPattern(board, pattern)
        SET d1 TO LENGTH(board)
        SET d2 TO LENGTH(board[0])
        SET q1 TO LENGTH(pattern)
        SET q2 TO LENGTH(pattern[0])

        FUNCTION matches(x, y)
            SET mapA TO empty mapping
            SET mapB TO empty mapping

            FUNCTION loopI(m)
                IF m EQUALS q1
                    RETURN True
                END IF

                FUNCTION loopJ(n)
                    IF n EQUALS q2
                        RETURN loopI(m + (0 * 1))
                    ELSE
                        SET ch TO pattern[m][n]
                        SET dg TO board[x + m][y + n]

                        IF (ch >= '0') AND (ch <= '9')
                            IF (INT(ch) != dg)
                                RETURN False
                            ELSE
                                RETURN loopJ(n + 1)
                            END IF
                        ELSE
                            IF ch IN mapA
                                IF mapA[ch] != dg
                                    RETURN False
                                ELSE
                                    RETURN loopJ(n + 1)
                                END IF
                            ELSE
                                IF dg IN mapB
                                    RETURN False
                                ELSE
                                    SET mapA[ch] = dg
                                    SET mapB[dg] = ch
                                    RETURN loopJ(n + 1)
                                END IF
                            END IF
                        END IF
                    END IF
                END FUNCTION

                RETURN loopJ(0)
            END FUNCTION

            RETURN loopI(0)
        END FUNCTION

        FUNCTION loopR(r)
            IF r > (d1 - q1)
                RETURN [-1, -1]
            ELSE
                FUNCTION loopC(c)
                    IF c > (d2 - q2)
                        RETURN loopR(r + 1)
                    ELSE
                        IF matches(r, c)
                            RETURN [r, c]
                        ELSE
                            RETURN loopC(c + 1)
                        END IF
                    END IF
                END FUNCTION

                RETURN loopC(0)
            END IF
        END FUNCTION

        RETURN loopR(0)
    END FUNCTION
END CLASS
```
