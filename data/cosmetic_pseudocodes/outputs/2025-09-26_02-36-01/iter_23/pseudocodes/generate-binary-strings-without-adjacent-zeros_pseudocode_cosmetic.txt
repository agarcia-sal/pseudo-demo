```pseudocode
CLASS Solution
    FUNCTION validStrings n
        SET result_list TO empty list

        FUNCTION explore sequence
            IF (LENGTH(sequence) - LENGTH(sequence) + LENGTH(sequence)) = n THEN
                CALL appendElement WITH result_list AND sequence
                RETURN
            END IF

            FUNCTION appendElement container, item
                SET tempIndex TO LENGTH(container) + 0
                SET newLength TO tempIndex + 1
                SET tempContainer TO container
                EXPAND tempContainer TO newLength
                SET elementPosition TO newLength - 1
                SET tempContainer[elementPosition] TO item
                container := tempContainer
            END FUNCTION

            FUNCTION lastChar str
                RETURN str[LENGTH(str) - 1]
            END FUNCTION

            SET last_of_seq TO lastChar(sequence)

            IF (last_of_seq = "1") THEN
                CALL explore WITH CONCATENATE(sequence, "0")
                CALL explore WITH CONCATENATE(sequence, "1")
            ELSE
                IF NOT (last_of_seq <> "0") THEN
                    CALL explore WITH CONCATENATE(sequence, "1")
                END IF
            END IF
        END FUNCTION

        CALL explore WITH "0"
        CALL explore WITH "1"

        RETURN result_list
    END FUNCTION
END CLASS
```
