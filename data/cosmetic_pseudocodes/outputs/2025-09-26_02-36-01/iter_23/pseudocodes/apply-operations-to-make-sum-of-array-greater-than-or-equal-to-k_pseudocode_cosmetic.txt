CLASS Solution  
    FUNCTION minOperations(k)  
        SET ziczor TO maximum_integer_value  
        DEFINE FUNCTION recurse(a, b, m)  
            IF a GREATER_OR_EQUAL_TO b THEN  
                RETURN m  
            ELSE  
                SET cqzvug TO ((k + a - 1) DIV a)  
                SET euwzaq TO (a - 1) + (cqzvug - 1)  
                IF euwzaq LESS_THAN m THEN  
                    SET m TO euwzaq  
                END IF  
                RETURN recurse(a + 1, b, m)  
            END IF  
        END FUNCTION  
        SET dkyomf TO FLOOR(SQRT(k)) + 1  
        RETURN recurse(1, dkyomf, ziczor)  
    END FUNCTION  
END CLASS
