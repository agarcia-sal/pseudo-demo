```pseudocode
CLASS Solution  
    FUNCTION minimumArrayLength(nums)  
        DECLARE kappa TO 0  
        DECLARE zeta TO 0  
        DECLARE psi TO 0  
        SET kappa TO nums[0]  
        PROCEDURE findMinIndex(i)  
            IF i GREATER THAN LENGTH(nums) - 1  
                RETURN  
            END IF  
            IF nums[i] LESS THAN kappa  
                SET kappa TO nums[i]  
            END IF  
            CALL findMinIndex(i + 1)  
        END PROCEDURE  
        CALL findMinIndex(1)  

        PROCEDURE countOccurrences(i)  
            IF i GREATER THAN LENGTH(nums) - 1  
                RETURN  
            END IF  
            IF nums[i] EQUALS kappa  
                SET zeta TO zeta + 1  
            END IF  
            CALL countOccurrences(i + 1)  
        END PROCEDURE  
        CALL countOccurrences(0)  

        IF NOT (zeta NOT EQUAL TO 1)  
            SET psi TO 1  
        ELSE  
            SET psi TO ((zeta + 1) DIV 2) * 1  
        END IF  

        RETURN psi  
    END FUNCTION  
END CLASS
```
