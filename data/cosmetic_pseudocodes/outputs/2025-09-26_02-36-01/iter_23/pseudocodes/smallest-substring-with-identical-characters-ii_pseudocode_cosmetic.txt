```pseudocode
CLASS Solution
    FUNCTION minLength(s parameter representing a string, numOps parameter representing an integer) RETURNS integer

        FUNCTION check(m parameter representing an integer) RETURNS boolean
            SET resCounter TO 0
            SET intervalLen TO 0

            FUNCTION iterate(index parameter representing integer) RETURNS boolean
                IF index GREATER THAN OR EQUAL TO LENGTH(s)
                    RETURN (resCounter <= numOps)
                END IF

                SET intervalLen TO intervalLen + 1

                IF (index EQUALS LENGTH(s) - 1) OR (s[index] NOT EQUAL TO s[index + 1])
                    SET resCounter TO resCounter + ((intervalLen - 1) DIV m) + 1
                    IF resCounter > numOps
                        RETURN false
                    END IF
                    SET intervalLen TO 0
                END IF

                RETURN iterate(index + 1)
            END FUNCTION

            RETURN iterate(0)
        END FUNCTION

        SET strLen TO LENGTH(s)
        SET lowBound TO 1
        SET upBound TO strLen

        FUNCTION binarySearch(low parameter representing integer, high parameter representing integer) RETURNS integer
            IF low >= high
                RETURN low
            END IF

            SET middle TO (low + high) DIV 2

            IF check(middle)
                RETURN binarySearch(low, middle)
            ELSE
                RETURN binarySearch(middle + 1, high)
            END IF
        END FUNCTION

        RETURN binarySearch(lowBound, upBound)
    END FUNCTION
END CLASS
```
