CLASS Solution
    FUNCTION distanceSum WITH PARAMETERS m n k
        SET CONST_MODULO TO (10 * 10 * 10 * 10 * 10 * 10 * 10 * 10 * 10) + (7 * 10 * 10)
        FUNCTION factorial WITH PARAMETER alpha
            IF (alpha = 0) THEN
                RETURN 1
            ELSE
                RETURN alpha * factorial(alpha - 1)
            END IF
        END FUNCTION
        FUNCTION combination WITH PARAMETERS r s
            RETURN factorial(r) / (factorial(s) * factorial(r - s))
        END FUNCTION
        FUNCTION power_three WITH PARAMETER zeta
            RETURN zeta * zeta * zeta
        END FUNCTION
        FUNCTION square WITH PARAMETER beta
            RETURN beta * beta
        END FUNCTION
        SET delta TO m
        SET epsilon TO n
        SET phi TO (((square(epsilon) * square(delta) * (delta * delta - delta))) / 6) + (((square(delta) * square(epsilon) * (epsilon * epsilon - epsilon))) / 6)
        SET chi TO combination((delta * epsilon) - 2, k - 2)
        SET psi TO phi * chi
        SET remainder_result TO psi % CONST_MODULO
        RETURN remainder_result
    END FUNCTION
END CLASS
