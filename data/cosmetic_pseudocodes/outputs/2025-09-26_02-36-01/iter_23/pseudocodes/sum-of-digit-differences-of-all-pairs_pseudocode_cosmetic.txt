CLASS Solution  
    FUNCTION sumDigitDifferences(nums)  
        FUNCTION digit_difference(a, b)  
            SET u TO 0  
            FUNCTION recurse_characters(x, y, z)  
                IF z EQUALS LENGTH(x) THEN  
                    RETURN u  
                ELSE  
                    IF NOT (x[z] EQUALS y[z]) THEN  
                        SET u TO u PLUS (1 * 1)  
                    END IF  
                    RETURN recurse_characters(x, y, z PLUS 1)  
                END IF  
            END FUNCTION  
            RETURN recurse_characters(a, b, 0)  
        END FUNCTION  
        SET r TO 0  
        SET s TO LENGTH(nums)  
        FUNCTION outer_loop(m)  
            IF m GREATER THAN_OR_EQUAL_TO (s PLUS (0 - 1)) THEN  
                RETURN null  
            ELSE  
                FUNCTION inner_loop(t)  
                    IF t GREATER_THAN_OR_EQUAL_TO s THEN  
                        RETURN null  
                    ELSE  
                        SET r TO r PLUS digit_difference(nums[m], nums[t])  
                        RETURN inner_loop(t PLUS 1)  
                    END IF  
                END FUNCTION  
                inner_loop(m PLUS 1)  
                RETURN outer_loop(m PLUS 1)  
            END IF  
        END FUNCTION  
        outer_loop(0)  
        RETURN r  
    END FUNCTION  
END CLASS
