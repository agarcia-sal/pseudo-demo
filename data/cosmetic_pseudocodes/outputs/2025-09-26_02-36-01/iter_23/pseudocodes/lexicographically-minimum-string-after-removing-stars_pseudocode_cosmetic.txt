CLASS Solution
    FUNCTION clearStars(s)
        FUNCTION aux(tracker, idx, limit)
            IF NOT (idx < limit)
                RETURN tracker
            END IF
            SET flag TO FALSE
            SET val TO s[idx]
            IF (val = "*" )
                IF (LIMIT LENGTH tracker > 0)
                    SET tracker TO SUBLIST tracker FROM 0 TO (LIMIT LENGTH tracker + (-1))
                END IF
            ELSE
                SET tempList TO list WITH val
                SET tracker TO CONCATENATE tracker AND tempList
            END IF
            RETURN aux(tracker, idx + (1), limit)
        END FUNCTION

        SET acc TO aux([], 0, LENGTH s)
        SET outStr TO ""
        FOR i FROM 0 TO (LENGTH acc - (1))
            SET outStr TO outStr + acc[i]
        END FOR

        RETURN outStr
    END FUNCTION
END CLASS
