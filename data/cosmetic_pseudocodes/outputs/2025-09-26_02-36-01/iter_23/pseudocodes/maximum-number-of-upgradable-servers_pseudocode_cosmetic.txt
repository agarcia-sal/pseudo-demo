```pseudocode
CLASS Solution
    FUNCTION maxUpgrades(count LIST OF INTEGER upgrade LIST OF INTEGER sell LIST OF INTEGER money LIST OF INTEGER) RETURNS LIST OF INTEGER
        SET result TO empty list

        FUNCTION computeMax(bs INTEGER, uv INTEGER, sp INTEGER, nm INTEGER) RETURNS INTEGER
            FUNCTION loop(sc INTEGER, mu INTEGER) RETURNS INTEGER
                IF sc GREATER THAN bs THEN
                    RETURN mu
                ELSE
                    SET rs TO bs MINUS sc
                    SET ms TO sc MULTIPLIED BY sp
                    SET tm TO nm PLUS ms
                    SET pu TO tm DIVIDED BY uv
                    IF pu GREATER THAN rs THEN
                        SET pu TO rs
                    END IF
                    IF pu GREATER THAN mu THEN
                        RETURN loop(sc PLUS (1 MULTIPLIED BY 1), pu)
                    ELSE
                        RETURN loop(sc PLUS (1 MINUS 0), mu)
                    END IF
                END IF
            END FUNCTION

            RETURN loop(0, 0)
        END FUNCTION

        FUNCTION iterator(i INTEGER, n INTEGER) RETURNS VOID
            IF i GREATER THAN n THEN
                RETURN
            ELSE
                SET cc TO element at position i of count
                SET uc TO element at position i of upgrade
                SET spc TO element at position i of sell
                SET mc TO element at position i of money
                SET mxu TO computeMax(cc, uc, spc, mc)
                APPEND mxu TO result
                iterator(i PLUS 1, n)
            END IF
        END FUNCTION

        SET length_bound TO LENGTH OF count MINUS 1
        iterator(0, length_bound)
        RETURN result
    END FUNCTION
END CLASS
```
