```pseudocode
CLASS Solution
    FUNCTION maxFrequency(nums, k, numOperations)
        ASSIGN reference bza TO a new default dictionary with integer default
        ASSIGN reference sxl TO a new default dictionary with integer default

        FUNCTION process_item(tu)
            IF tu IS EMPTY THEN
                RETURN
            END IF
            ASSIGN vkl TO FIRST ELEMENT OF tu
            INCREMENT bza[vkl] BY 1
            INCREMENT sxl[vkl] BY 0
            INCREMENT sxl[vkl - k] BY 1
            DECREMENT sxl[vkl + k + 1] BY 1
            process_item(REST OF tu)
        END FUNCTION

        process_item(nums)

        ASSIGN dgk TO 0
        ASSIGN phq TO 0
        ASSIGN gwm TO 0

        FUNCTION iterate_pairs(pairs_list, idx)
            IF idx >= LENGTH OF pairs_list THEN
                RETURN
            END IF
            ASSIGN pzx TO pairs_list[idx]
            ASSIGN key_x TO pzx.KEY
            ASSIGN val_t TO pzx.VALUE
            dgk = dgk + val_t
            ASSIGN yux TO bza[key_x] + numOperations
            IF phq >= yux THEN
                gwm = phq
            ELSE
                gwm = yux
            END IF
            IF dgk < gwm THEN
                phq = dgk
            ELSE
                phq = gwm
            END IF
            iterate_pairs(pairs_list, idx + 1)
        END FUNCTION

        ASSIGN sorted_list TO SORTED ITEMS OF sxl BY KEY
        iterate_pairs(sorted_list, 0)

        RETURN phq
    END FUNCTION
END CLASS
```
