CLASS Solution
    FUNCTION maximumPrimeDifference(nums)
        SET primeSet TO {97, 89, 83, 79, 73, 71, 67, 61, 59, 53, 47, 43, 41, 37, 31, 29, 23, 19, 17, 13, 11, 7, 5, 3, 2}
        SET idx_first TO -1
        SET idx_last TO -1

        FUNCTION findIndices(current_pos, collection)
            IF current_pos >= LENGTH(collection) THEN
                RETURN
            ELSE
                SET element_curr TO collection[current_pos]
                IF element_curr IN primeSet THEN
                    IF idx_first = -1 THEN
                        SET idx_first TO current_pos
                    END IF
                    SET idx_last TO current_pos
                END IF
                CALL findIndices(current_pos + 1, collection)
            END IF
        END FUNCTION

        CALL findIndices(0, nums)
        RETURN idx_last + (-1 * idx_first)
    END FUNCTION
END CLASS
