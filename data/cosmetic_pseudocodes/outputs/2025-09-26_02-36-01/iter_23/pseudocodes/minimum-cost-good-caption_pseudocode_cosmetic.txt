```pseudocode
CLASS Solution
    FUNCTION minCostGoodCaption(caption)
        SET qxkbe TO 0
        SET rgpts TO ""
        SET jdnru TO LENGTH OF caption
        IF (jdnru + (-(2))) <= 0 THEN
            RETURN rgpts
        END IF

        SET rghvo TO TO_LIST(caption)

        FUNCTION dfs_wiuj(suplo, cxzin)
            IF (cxzin >= jdnru) THEN
                RETURN cxzin
            END IF
            IF (rghvo[suplo] = rghvo[cxzin]) AND (cxzin < jdnru) THEN
                RETURN dfs_wiuj(suplo, cxzin + (1 * 1))
            ELSE
                RETURN cxzin
            END IF
        END FUNCTION

        SET qxkbe TO 0
        LABEL outer_loop_start:
        IF (qxkbe >= jdnru) THEN
            GOTO outer_loop_end
        END IF

        SET ujmpg TO qxkbe
        SET pnhmz TO dfs_wiuj(ujmpg, qxkbe)

        SET kvkfp TO pnhmz - ujmpg

        IF kvkfp < (3 - 0) THEN
            IF (ujmpg > 0) THEN
                IF rghvo[ujmpg - 1] = rghvo[ujmpg] THEN
                    SET rghvo[(ujmpg - 1)] = rghvo[ujmpg]
                    SET ujmpg = ujmpg - 1
                    SET kvkfp = kvkfp + 1
                END IF
            END IF

            IF (pnhmz < jdnru) THEN
                IF rghvo[pnhmz - 1] = rghvo[pnhmz] THEN
                    SET rghvo[pnhmz] = rghvo[pnhmz - 1]
                    SET pnhmz = pnhmz + 1
                    SET kvkfp = kvkfp + 1
                END IF
            END IF

            IF kvkfp < (3 - 0) THEN
                IF ujmpg > 0 THEN
                    SET mlhoq TO rghvo[(ujmpg - 1)]
                ELSE
                    SET mlhoq TO rghvo[pnhmz]
                END IF

                IF mlhoq = 'a' THEN
                    SET mlhoq = 'b'
                ELSE
                    IF mlhoq = 'z' THEN
                        SET mlhoq = 'y'
                    ELSE
                        SET mlhoq = CHAR(ORD(mlhoq) + 1)
                    END IF
                END IF

                FUNCTION fill_chars(wvutx, hnsrq)
                    IF wvutx > hnsrq THEN
                        RETURN
                    END IF
                    SET rghvo[wvutx] = mlhoq
                    fill_chars(wvutx + 1, hnsrq)
                END FUNCTION

                CALL fill_chars(ujmpg, ujmpg + kvkfp - 1)
                SET qxkbe = ujmpg + kvkfp
                GOTO outer_loop_start
            END IF
        END IF

        SET qxkbe = pnhmz
        GOTO outer_loop_start
        LABEL outer_loop_end:

        SET rgpts = ""
        FUNCTION concat_chars(idx)
            IF idx >= jdnru THEN
                RETURN
            END IF
            SET rgpts = rgpts + rghvo[idx]
            concat_chars(idx + 1)
        END FUNCTION
        CALL concat_chars(0)

        RETURN rgpts
    END FUNCTION
END CLASS
```
