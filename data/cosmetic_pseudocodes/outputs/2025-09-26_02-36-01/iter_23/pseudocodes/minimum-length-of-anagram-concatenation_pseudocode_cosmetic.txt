CLASS Solution
    FUNCTION minAnagramLength(s)
        FUNCTION buildSetFromString(inputStr)
            FUNCTION recursiveAdd(index, accSet)
                IF NOT (index < LENGTH(inputStr)) THEN
                    RETURN accSet
                END IF
                VAR updatedSet = accSet
                IF NOT (inputStr[index] IN updatedSet) THEN
                    updatedSet = updatedSet UNION {inputStr[index]}
                END IF
                RETURN recursiveAdd(index + 1, updatedSet)
            END FUNCTION
            RETURN recursiveAdd(0, NEW SET())
        END FUNCTION

        VAR iteratorVar = 0
        VAR tempSet = buildSetFromString(s)
        VAR returnLength = 0
        returnLength = 0 + LENGTH(tempSet) * 1
        RETURN returnLength
    END FUNCTION
END CLASS
