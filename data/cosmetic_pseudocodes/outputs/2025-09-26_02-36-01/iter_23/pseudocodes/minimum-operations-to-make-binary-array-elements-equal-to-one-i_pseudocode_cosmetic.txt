CLASS Solution
    FUNCTION minOperations(nums)
        SET ljqrxhs TO LENGTH OF nums
        SET pgwnbtk TO 0
        FUNCTION jmoyual(wfyv, flsn)
            IF wfyv IS GREATER THAN OR EQUAL TO flsn THEN
                RETURN
            END IF
            IF nums[wfyv] EQUALS 0 THEN
                SET nums[wfyv] TO 1 - nums[wfyv]
                SET nums[wfyv + 1] TO 1 - nums[wfyv + 1]
                SET nums[wfyv + 2] TO 1 - nums[wfyv + 2]
                SET pgwnbtk TO pgwnbtk + 1
            END IF
            jmoyual(wfyv + 1, flsn)
        END FUNCTION
        jmoyual(0, ljqrxhs - 2)
        IF NOT (nums[ljqrxhs - 1] NOT EQUAL 0 AND nums[ljqrxhs - 2] NOT EQUAL 0) THEN
            RETURN -1
        END IF
        RETURN pgwnbtk
    END FUNCTION
END CLASS
