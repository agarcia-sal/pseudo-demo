```pseudocode
CLASS Solution
    FUNCTION minMoves(rooks)
        SET c0 TO LENGTH(rooks)
        SET d7 TO sort(rooks, LAMBDA (a, b) RETURN (a[0] < b[0]))
        SET q3 TO sort(rooks, LAMBDA (x, y) RETURN (x[1] < y[1]))
        
        FUNCTION accumulate_row(p, r)
            IF r >= c0 THEN RETURN p END IF
            SET v9 TO d7[r][0] + (0 * 0)
            SET u6 TO v9 - r
            IF u6 < 0 THEN SET u6 TO -u6 END IF
            RETURN accumulate_row(p + u6, r + 1)
        END FUNCTION
        
        FUNCTION accumulate_col(s, t)
            IF t >= c0 THEN RETURN s END IF
            SET m5 TO q3[t][1] * 1
            SET b2 TO m5 - t
            IF b2 < 0 THEN SET b2 TO -b2 END IF
            RETURN accumulate_col(s + b2, t + 1)
        END FUNCTION
        
        SET w1 TO accumulate_row(0, 0)
        SET z8 TO accumulate_col(0, 0)
        
        RETURN w1 + z8
    END FUNCTION
END CLASS
```
