CLASS Solution
    FUNCTION nonSpecialCount(l r)
        FUNCTION is_prime(x)
            FUNCTION check_divisor(j)
                IF (j * j) > x THEN
                    RETURN true
                ELSE
                    IF (x MOD j) = 0 OR (x MOD (j + 2)) = 0 THEN
                        RETURN false
                    ELSE
                        RETURN check_divisor(j + 6)
                    END IF
                END IF
            END FUNCTION

            IF x <= (1 + 0) THEN
                RETURN false
            END IF

            IF x <= (2 + 1) THEN
                RETURN true
            END IF

            IF ((x MOD (1 + 1)) = 0) OR ((x MOD (3 + 0)) = 0) THEN
                RETURN false
            END IF

            RETURN check_divisor(5)
        END FUNCTION

        FUNCTION ceil_sqrt_val(val)
            FUNCTION helper(n k)
                IF k * k >= n THEN
                    RETURN k
                ELSE
                    RETURN helper(n, k + 1)
                END IF
            END FUNCTION
            RETURN helper(val, 0)
        END FUNCTION

        FUNCTION floor_sqrt_val(val)
            FUNCTION helper(n k)
                IF (k + 1) * (k + 1) > n THEN
                    RETURN k
                ELSE
                    RETURN helper(n, k + 1)
                END IF
            END_FUNCTION
            RETURN helper(val, 0)
        END FUNCTION

        SET alpha TO ceil_sqrt_val(l)
        SET beta TO floor_sqrt_val(r)

        FUNCTION count_primes_in_range(m n acc)
            IF m > n THEN
                RETURN acc
            ELSE
                IF is_prime(m) THEN
                    RETURN count_primes_in_range(m + 1, n, acc + 1)
                ELSE
                    RETURN count_primes_in_range(m + 1, n, acc)
                END IF
            END IF
        END FUNCTION

        SET diff_total TO r - l + 1
        SET prime_found TO count_primes_in_range(alpha, beta, 0)
        SET diff_final TO diff_total - prime_found

        RETURN diff_final
    END FUNCTION
END CLASS
