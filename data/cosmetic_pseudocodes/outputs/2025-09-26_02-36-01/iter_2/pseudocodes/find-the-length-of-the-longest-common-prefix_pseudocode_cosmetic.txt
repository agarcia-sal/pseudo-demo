CLASS Solution  
    FUNCTION longestCommonPrefix(arr1 arr1 OF integers arr2 arr2 OF integers) RETURNS integer  
        DECLARE setA AS empty set  
        DECLARE index1 AS integer  
        DECLARE strNum AS string  
        DECLARE lenStr AS integer  
        DECLARE setB AS empty set  
        DECLARE index2 AS integer  
        DECLARE longestPrefixLen AS integer  
        longestPrefixLen := 0  
  
        DECLARE iterator1 AS integer  
        iterator1 := 0  
        WHILE iterator1 < LENGTH(arr1) DO  
            strNum := TO_STRING(arr1[iterator1])  
            lenStr := LENGTH(strNum)  
            index1 := 0  
            WHILE index1 < lenStr DO  
                setA.ADD(SUBSTRING(strNum, 0, index1 + 1))  
                index1 := index1 + 1  
            END WHILE  
            iterator1 := iterator1 + 1  
        END WHILE  
  
        DECLARE iterator2 AS integer  
        iterator2 := 0  
        WHILE iterator2 < LENGTH(arr2) DO  
            strNum := TO_STRING(arr2[iterator2])  
            lenStr := LENGTH(strNum)  
            index2 := 0  
            WHILE index2 < lenStr DO  
                setB.ADD(SUBSTRING(strNum, 0, index2 + 1))  
                index2 := index2 + 1  
            END WHILE  
            iterator2 := iterator2 + 1  
        END WHILE  
  
        FOR EACH candidatePrefix IN setA DO  
            IF setB.CONTAINS(candidatePrefix) THEN  
                DECLARE prefixLen AS integer  
                prefixLen := LENGTH(candidatePrefix)  
                IF prefixLen > longestPrefixLen THEN  
                    longestPrefixLen := prefixLen  
                END IF  
            END IF  
        END FOR  
  
        RETURN longestPrefixLen  
    END FUNCTION  
END CLASS
