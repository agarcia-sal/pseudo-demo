CLASS Solution  
    FUNCTION countSubarrays(nums LIST OF INTEGER, k INTEGER) RETURNS INTEGER  
        LET length ASSIGN LENGTH(nums)  
        LET totalCount ASSIGN 0  
        LET i ASSIGN 0  
        WHILE i < length DO  
            LET andValue ASSIGN nums[i]  
            LET j ASSIGN i  
            WHILE j < length DO  
                andValue ASSIGN andValue & nums[j]  
                IF andValue == k THEN  
                    totalCount ASSIGN totalCount + 1  
                END IF  
                IF andValue < k THEN  
                    EXIT WHILE  
                END IF  
                j ASSIGN j + 1  
            END WHILE  
            i ASSIGN i + 1  
        END WHILE  
        RETURN totalCount  
    END FUNCTION  
END CLASS
