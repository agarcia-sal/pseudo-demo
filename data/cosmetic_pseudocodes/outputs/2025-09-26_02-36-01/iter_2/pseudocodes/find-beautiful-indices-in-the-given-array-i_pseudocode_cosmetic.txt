CLASS Solution  
    FUNCTION beautifulIndices(s parameter, a parameter, b parameter, k parameter)  
        LET positionsA BE []  
        LET sLen BE LENGTH(s)  
        LET aLen BE LENGTH(a)  
        LET posAIndex BE 0  
        WHILE posAIndex <= sLen - aLen  
            LET segmentA BE SUBSTRING(s, posAIndex, posAIndex + aLen - 1)  
            IF segmentA == a THEN  
                positionsA APPEND posAIndex  
            END IF  
            posAIndex = posAIndex + 1  
        END WHILE  

        LET positionsB BE []  
        LET bLen BE LENGTH(b)  
        LET posBIndex BE 0  
        WHILE posBIndex <= sLen - bLen  
            LET segmentB BE SUBSTRING(s, posBIndex, posBIndex + bLen - 1)  
            IF segmentB == b THEN  
                positionsB APPEND posBIndex  
            END IF  
            posBIndex = posBIndex + 1  
        END WHILE  

        LET results BE []  
        FOR each startA IN positionsA  
            LET innerIndex BE 0  
            WHILE innerIndex < LENGTH(positionsB)  
                LET startB BE positionsB[innerIndex]  
                LET distance BE startA - startB  
                LET distPositive BE distance  
                IF distance < 0 THEN  
                    distPositive = (-1) * distance  
                END IF  
                IF distPositive <= k THEN  
                    results APPEND startA  
                    BREAK  
                END IF  
                innerIndex = innerIndex + 1  
            END WHILE  
        END FOR  
        RETURN results  
    END FUNCTION  
END CLASS
