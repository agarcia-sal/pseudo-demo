CLASS Solution  
    FUNCTION validStrings length  
        IF length IS 1  
            RETURN [ "0", "1" ]  
        END IF  

        FUNCTION backtrack sequence  
            IF sequence.LENGTH() = length  
                validSequences.ADD(sequence)  
                RETURN  
            END IF  

            lastChar = sequence[sequence.LENGTH() - 1]  
            IF lastChar = "1"  
                backtrack(sequence + "0")  
                backtrack(sequence + "1")  
            ELSEIF lastChar = "0"  
                backtrack(sequence + "1")  
            END IF  
        END FUNCTION  

        validSequences = []  
        backtrack("0")  
        backtrack("1")  

        RETURN validSequences  
    END FUNCTION  
END CLASS
