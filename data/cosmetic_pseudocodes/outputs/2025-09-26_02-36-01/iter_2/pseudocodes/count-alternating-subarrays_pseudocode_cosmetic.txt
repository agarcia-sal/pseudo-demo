CLASS Solution  
    FUNCTION countAlternatingSubarrays(nums)  
        LET totalElements BE LENGTH(nums)  
        IF totalElements IS 0 THEN  
            RETURN 0  
        END IF  
        LET totalCount BE totalElements  
        LET seqLength BE 1  
        LET index BE 1  
        WHILE index < totalElements DO  
            LET currentElement BE nums[index]  
            LET previousElement BE nums[index - 1]  
            IF currentElement != previousElement THEN  
                seqLength = seqLength + 1  
            ELSE  
                seqLength = 1  
            END IF  
            totalCount = totalCount + (seqLength - 1)  
            index = index + 1  
        END WHILE  
        RETURN totalCount  
    END FUNCTION  
END CLASS
