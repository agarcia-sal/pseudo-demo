CLASS Solution
    FUNCTION valueAfterKSeconds(n, k)
        LET MODULO be 10^9 + 7
        LET arr INITIATED AS array WITH n ELEMENTS EACH SET TO 1
        LET idx BE 1
        WHILE idx < n
            SET idx TO idx + 1
        END WHILE
        LET step BE 0
        WHILE step < k
            LET pos BE 1
            WHILE pos < n
                LET prevVal BE arr[pos - 1]
                LET currVal BE arr[pos]
                LET summation BE (currVal ADD prevVal) MODULO MODULO
                SET arr[pos] TO summation
                SET pos TO pos + 1
            END WHILE
            SET step TO step + 1
        END WHILE
        RETURN arr[n - 1]
    END FUNCTION
END CLASS
