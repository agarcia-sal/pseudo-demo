CLASS Solution
    FUNCTION maxScore(numbers)
        LET lengthVar ASSIGN LENGTH(numbers)
        LET dpArray ASSIGN ARRAY OF lengthVar ELEMENTS INITIALIZED TO 0
        dpArray[0] ASSIGN 0
        LET outerCounter ASSIGN 1
        WHILE outerCounter < lengthVar
            LET innerCounter ASSIGN 0
            WHILE innerCounter < outerCounter
                LET candidateValue ASSIGN dpArray[innerCounter] + ((outerCounter - innerCounter) TIMES numbers[outerCounter])
                IF dpArray[outerCounter] < candidateValue THEN
                    dpArray[outerCounter] = candidateValue
                END IF
                innerCounter = innerCounter + 1
            END WHILE
            outerCounter = outerCounter + 1
        END WHILE
        RETURN dpArray[lengthVar - 1]
    END FUNCTION
END CLASS
