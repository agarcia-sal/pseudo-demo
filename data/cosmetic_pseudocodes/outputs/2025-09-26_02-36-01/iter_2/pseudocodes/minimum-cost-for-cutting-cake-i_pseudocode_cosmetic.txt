CLASS Solution  
    FUNCTION minimumCost(m n horizontalCut verticalCut)  
        SORT horizontalCut FROM LARGEST TO SMALLEST  
        SORT verticalCut FROM LARGEST TO SMALLEST  
        SET totalCost TO 0  
        ASSIGN idxH TO 0  
        ASSIGN idxV TO 0  
        ASSIGN countH TO 1  
        ASSIGN countV TO 1  
        WHILE (idxH < (m - 1)) OR (idxV < (n - 1))  
            IF (idxV = (n - 1)) OR ((idxH < (m - 1)) AND (horizontalCut[idxH] > verticalCut[idxV])) THEN  
                SET costIncrement TO horizontalCut[idxH] * countV  
                totalCost = totalCost + costIncrement  
                countH = countH + 1  
                idxH = idxH + 1  
            ELSE  
                SET costIncrement TO verticalCut[idxV] * countH  
                totalCost = totalCost + costIncrement  
                countV = countV + 1  
                idxV = idxV + 1  
            END IF  
        END WHILE  
        RETURN totalCost  
    END FUNCTION  
END CLASS
