CLASS Solution
    FUNCTION numberOfPairs(points)
        LET totalPairs BE 0
        LET lengthPoints BE SIZE OF points
        LET outerIndex BE 0
        WHILE outerIndex < lengthPoints
            LET innerIndex BE 0
            WHILE innerIndex < lengthPoints
                IF outerIndex != innerIndex THEN
                    LET xStart BE points[outerIndex][0]
                    LET yStart BE points[outerIndex][1]
                    LET xEnd BE points[innerIndex][0]
                    LET yEnd BE points[innerIndex][1]
                    IF (xStart <= xEnd) AND (yStart >= yEnd) THEN
                        LET isValidPair BE TRUE
                        LET checkIndex BE 0
                        LOOP
                            EXIT WHEN checkIndex >= lengthPoints
                            IF (checkIndex != outerIndex) AND (checkIndex != innerIndex) THEN
                                LET xMid BE points[checkIndex][0]
                                LET yMid BE points[checkIndex][1]
                                IF (xStart <= xMid) AND (xMid <= xEnd) AND (yStart >= yMid) AND (yMid >= yEnd) THEN
                                    isValidPair <- FALSE
                                    LEAVE
                                END IF
                            END IF
                            checkIndex <- checkIndex + 1
                        END LOOP
                        IF isValidPair = TRUE THEN
                            totalPairs <- totalPairs + 1
                        END IF
                    END IF
                END IF
                innerIndex <- innerIndex + 1
            END WHILE
            outerIndex <- outerIndex + 1
        END WHILE
        RETURN totalPairs
    END FUNCTION
END CLASS
