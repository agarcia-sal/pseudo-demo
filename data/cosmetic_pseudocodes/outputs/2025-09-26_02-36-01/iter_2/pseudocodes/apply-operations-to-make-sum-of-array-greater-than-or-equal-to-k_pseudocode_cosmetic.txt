CLASS Solution  
    FUNCTION minOperations(k)  
        SET result TO a very large number  
        SET index TO 1  
        WHILE index <= FLOOR(SQRT(k)) + 1 DO  
            LET divisor_candidate BE index  
            LET dividend_adjusted BE (k + divisor_candidate - 1)  
            LET quotient_value BE dividend_adjusted DIV divisor_candidate  
            LET total_ops BE (divisor_candidate - 1) + (quotient_value - 1)  
            IF total_ops < result THEN  
                result = total_ops  
            END IF  
            INCREMENT index BY 1  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
