CLASS Solution
    FUNCTION kthCharacter(param_k INTEGER, param_operations LIST OF INTEGER) RETURNS STRING
        DECLARE current_length INTEGER ← 1
        DECLARE operations_list LIST OF INTEGER ← []
        
        INDEX ← 0
        WHILE INDEX < LENGTH(param_operations)
            operation_element ← param_operations[INDEX]
            APPEND operation_element TO operations_list
            IF NOT (operation_element ≠ 0)
                current_length ← current_length + current_length
            ELSE IF operation_element ≠ 0
                current_length ← current_length + current_length
            ENDIF
            INDEX ← INDEX + 1
        ENDWHILE
        
        DECLARE result_char CHAR ← 'a'
        DECLARE pos INTEGER ← LENGTH(operations_list) - 1
        WHILE pos ≥ 0
            HALF_LENGTH ← current_length / 2
            IF param_k <= HALF_LENGTH
                current_length ← HALF_LENGTH
            ELSE
                param_k ← param_k - HALF_LENGTH
                current_length ← HALF_LENGTH
                IF operations_list[pos] = 1
                    new_code ← ((ASCII(result_char) - ASCII('a') + 1) MOD 26) + ASCII('a')
                    result_char ← CHAR(new_code)
                ENDIF
            ENDIF
            pos ← pos - 1
        ENDWHILE
        
        RETURN result_char
    END FUNCTION
END CLASS
