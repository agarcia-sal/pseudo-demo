CLASS Solution  
    FUNCTION minimumOperationsToMakeKPeriodic(word, k)  
        LET total_length BE SIZE OF word  
        LET partition_list BE []  
        LET pointer BE 0  
        WHILE pointer < total_length DO  
            LET segment_end BE pointer + k  
            LET partition_list APPENDED WITH word SUBSTRING FROM pointer TO (segment_end - 1)  
            pointer = pointer + k  
        END WHILE  
        LET frequency_map BE DICTIONARY COUNTING OCCURRENCES OF EACH ELEMENT IN partition_list  
        LET highest_frequency BE MAXIMUM VALUE IN frequency_map VALUES  
        LET difference BE LENGTH OF partition_list - highest_frequency  
        RETURN difference  
    END FUNCTION  
END CLASS
