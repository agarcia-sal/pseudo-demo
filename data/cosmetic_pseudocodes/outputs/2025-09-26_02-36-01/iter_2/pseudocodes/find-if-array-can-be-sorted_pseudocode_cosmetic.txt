CLASS Solution
    FUNCTION canSortArray(array)
        FUNCTION countSetBits(value)
            SET bitCount TO 0
            SET tempVal TO value
            WHILE tempVal IS NOT 0
                SET bitCount TO bitCount PLUS (tempVal MOD 2)
                SET tempVal TO tempVal DIVIDED BY 2
            END WHILE
            RETURN bitCount
        END FUNCTION
        
        SET lengthArray TO SIZE OF array
        SET target TO COPY OF array
        CALL target SORTED IN NON-DECREASING ORDER
        
        SET indexOuter TO 0
        WHILE indexOuter < lengthArray MINUS 1
            SET indexInner TO 0
            WHILE indexInner < lengthArray MINUS 1
                SET leftVal TO array AT indexInner
                SET rightVal TO array AT (indexInner PLUS 1)
                SET leftBits TO countSetBits(leftVal)
                SET rightBits TO countSetBits(rightVal)
                IF leftBits EQUALS rightBits AND leftVal GREATER THAN rightVal THEN
                    SET hold TO leftVal
                    SET array AT indexInner TO rightVal
                    SET array AT (indexInner PLUS 1) TO hold
                END IF
                SET indexInner TO indexInner PLUS 1
            END WHILE
            INCREMENT indexOuter BY 1
        END WHILE
        
        RETURN array EQUALS target
    END FUNCTION
END CLASS
