CLASS Solution  
    FUNCTION minimumArrayLength(nums)  
        LET smallest = nums[0]  
        LET i = 1  
        WHILE i < LENGTH(nums)  
            IF nums[i] < smallest THEN  
                smallest = nums[i]  
            END IF  
            i = i + 1  
        END WHILE  

        LET repetition = 0  
        LET j = 0  
        WHILE j < LENGTH(nums)  
            IF nums[j] = smallest THEN  
                repetition = repetition + 1  
            END IF  
            j = j + 1  
        END WHILE  

        IF NOT (repetition - 1) = 0 THEN  
            RETURN (repetition + 1) DIV 2  
        ELSE  
            RETURN 1  
        END IF  
    END FUNCTION  
END CLASS
