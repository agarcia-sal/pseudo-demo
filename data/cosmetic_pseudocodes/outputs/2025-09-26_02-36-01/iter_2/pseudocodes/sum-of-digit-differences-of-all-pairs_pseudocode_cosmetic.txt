CLASS Solution
    FUNCTION sumDigitDifferences(numbers)
        FUNCTION digit_difference(x, y)
            SET mismatchCount = 0
            SET idx = 0
            WHILE idx < LENGTH(x)
                IF NOT (x[idx] = y[idx]) THEN
                    mismatchCount = mismatchCount + 1
                END IF
                idx = idx + 1
            END WHILE
            RETURN mismatchCount
        END FUNCTION

        SET aggregateSum := 0
        SET lengthNums := LENGTH(numbers)
        SET outerIdx := 0
        WHILE outerIdx < lengthNums - 1
            SET innerIdx := outerIdx + 1
            WHILE innerIdx < lengthNums
                aggregateSum := aggregateSum + digit_difference(numbers[outerIdx], numbers[innerIdx])
                innerIdx := innerIdx + 1
            END WHILE
            outerIdx := outerIdx + 1
        END WHILE
        RETURN aggregateSum
    END FUNCTION
END CLASS
