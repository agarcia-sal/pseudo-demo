CLASS Solution  
    FUNCTION numberOfPairs(nums1, nums2, k)  
        DECLARE frequency_map AS Dictionary = empty  
        FOR index FROM 0 TO LENGTH(nums2) - 1  
            LET val = nums2[index]  
            IF val IN frequency_map  
                SET frequency_map[val] = frequency_map[val] PLUS 1  
            ELSE  
                SET frequency_map[val] = 1  
            END IF  
        END FOR  

        SET total_pairs TO 0  
        LET i = 0  
        WHILE i < LENGTH(nums1)  
            LET current = nums1[i]  
            FOR each key IN frequency_map KEYS  
                LET product_term = key TIMES k  
                LET remainder_check = current MOD product_term  
                IF remainder_check == 0  
                    SET total_pairs = total_pairs + frequency_map[key]  
                END IF  
            END FOR  
            INCREMENT i BY 1  
        END WHILE  
        RETURN total_pairs  
    END FUNCTION  
END CLASS
