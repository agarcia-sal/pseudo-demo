CLASS Solution  
    FUNCTION maxTotalReward(rewardValues)  
        DECLARE uniqueNumbers AS A SORTED LIST OF DISTINCT ELEMENTS FROM rewardValues  
        DECLARE flag AS INTEGER = 1  
        DECLARE index AS INTEGER = 0  
        WHILE index < LENGTH(uniqueNumbers)  
            DECLARE currentValue AS INTEGER = uniqueNumbers[index]  
            DECLARE bitMask AS INTEGER = ((1 << currentValue) - 1) << currentValue  
            flag = flag OR (flag AND bitMask)  
            index = index + 1  
        END WHILE  
        RETURN BIT_LENGTH(flag) - 1  
    END FUNCTION  
END CLASS
