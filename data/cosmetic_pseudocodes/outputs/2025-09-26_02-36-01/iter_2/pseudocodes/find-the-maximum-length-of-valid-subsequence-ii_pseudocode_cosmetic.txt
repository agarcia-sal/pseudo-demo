CLASS Solution
    FUNCTION maximumLength(nums k)
        LET length_of_nums BE THE COUNT OF nums
        IF length_of_nums == 1 THEN
            RETURN 1
        END IF

        LET dp BE A NEW ARRAY OF length_of_nums WITH EACH ELEMENT INITIALIZED TO AN EMPTY DICTIONARY

        LET longest_subsequence BE 1

        LET outer_index BE 0
        WHILE outer_index < length_of_nums
            LET inner_index BE 0
            WHILE inner_index < outer_index
                LET sum_val BE nums[outer_index] + nums[inner_index]
                LET mod_key BE sum_val - k * ((sum_val) DIV k)
                
                IF mod_key IN dp[inner_index] THEN
                    LET current_length BE dp[inner_index][mod_key] + 1
                ELSE
                    LET current_length BE 2
                END IF

                dp[outer_index][mod_key] = current_length

                IF current_length > longest_subsequence THEN
                    longest_subsequence = current_length
                END IF

                inner_index = inner_index + 1
            END WHILE
            outer_index = outer_index + 1
        END WHILE

        RETURN longest_subsequence
    END FUNCTION
END CLASS
