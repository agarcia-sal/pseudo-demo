CLASS Solution  
    FUNCTION minEnd(n alpha)  
        FUNCTION canConstruct(limit)  
            LET temp = alpha  
            LET tally = 1  
            WHILE temp < limit DO  
                temp = temp + 1  
                IF (temp BITWISEAND alpha) == alpha THEN  
                    tally = tally + 1  
                    IF tally == alpha THEN  
                        RETURN True  
                    END IF  
                END IF  
            END WHILE  
            RETURN tally == alpha  
        END FUNCTION  
        
        LET startVal = alpha  
        LET upperBound = 2 * 10 * 10 * 10 * 10 * 10 * 10 * 10 * 10  
        WHILE startVal < upperBound DO  
            LET midpoint = (startVal + upperBound) DIV 2  
            IF canConstruct(midpoint) THEN  
                upperBound = midpoint  
            ELSE  
                startVal = startVal + 1  
            END IF  
        END WHILE
        
        RETURN startVal  
    END FUNCTION  
END CLASS
