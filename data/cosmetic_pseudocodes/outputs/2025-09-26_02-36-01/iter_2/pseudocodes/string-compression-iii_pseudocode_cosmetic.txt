CLASS Solution  
    FUNCTION compressedString(word)  
        DECLARE encoded AS empty array  
        DECLARE position AS 0  
        WHILE position < LENGTH(word)  
            DECLARE currentChar AS word[position]  
            DECLARE tally AS 0  
            WHILE position < LENGTH(word) AND word[position] == currentChar  
                tally = tally + 1  
                position = position + 1  
                IF tally == 9 THEN  
                    BREAK  
                END IF  
            END WHILE  
            APPEND (STRINGIFY(tally) + currentChar) TO encoded  
        END WHILE  
        DECLARE output AS ""  
        FOR EACH segment IN encoded DO  
            output = output + segment  
        END FOR  
        RETURN output  
    END FUNCTION  
END CLASS
