CLASS Solution
    FUNCTION maximumPoints(enemyEnergies activeEnergy)
        CALL enemyEnergies SORTED IN ASCENDING ORDER AND STORE IN orderedEnergy
        IF NOT (activeEnergy >= orderedEnergy[0])
            RETURN 0
        END IF

        SET totalPoints TO 0
        SET index TO LENGTH OF orderedEnergy MINUS 1

        WHILE index >= 0
            SET divisionResult TO activeEnergy DIVIDED BY orderedEnergy[0]
            SET totalPoints TO totalPoints + divisionResult

            SET activeEnergy TO activeEnergy - (divisionResult * orderedEnergy[0])

            SET activeEnergy TO activeEnergy + orderedEnergy[index]

            SET index TO index - 1
        END WHILE

        RETURN totalPoints
    END FUNCTION
END CLASS
