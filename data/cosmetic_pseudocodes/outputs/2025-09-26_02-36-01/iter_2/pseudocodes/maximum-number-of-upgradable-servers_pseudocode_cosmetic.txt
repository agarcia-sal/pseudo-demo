CLASS Solution  
    FUNCTION maxUpgrades(count LIST OF INTEGER upgrade LIST OF INTEGER sell LIST OF INTEGER money LIST OF INTEGER) RETURNS LIST OF INTEGER  
        DECLARE result AS empty list  
        SET position TO 0  
        WHILE position LESS THAN LENGTH OF count DO  
            LET total_servers = count[position]  
            LET cost_per_upgrade = upgrade[position]  
            LET price_per_sell = sell[position]  
            LET available_funds = money[position]  
            LET best_upgrade_count = 0  
            LET sale_counter = 0  
            WHILE sale_counter <= total_servers DO  
                LET servers_left = total_servers - sale_counter  
                LET funds_after_sale = available_funds + (sale_counter * price_per_sell)  
                LET upgrades_possible = (funds_after_sale) DIV cost_per_upgrade  
                IF upgrades_possible > servers_left THEN  
                    SET upgrades_possible TO servers_left  
                END IF  
                IF upgrades_possible > best_upgrade_count THEN  
                    SET best_upgrade_count TO upgrades_possible  
                END IF  
                INCREMENT sale_counter BY 1  
            END WHILE  
            APPEND best_upgrade_count TO result  
            INCREMENT position BY 1  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
