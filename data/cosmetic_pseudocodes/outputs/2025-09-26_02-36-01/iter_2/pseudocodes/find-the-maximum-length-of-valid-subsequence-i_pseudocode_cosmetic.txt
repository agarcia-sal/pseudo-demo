CLASS Solution  
    FUNCTION maximumLength(nums)  
        LET count_even = 0  
        LET count_odd = 0  
        LET index = 1  
        WHILE index < LENGTH(nums)  
            LET pair_sum = nums[index - 1] + nums[index]  
            IF (pair_sum & 1) = 0 THEN  
                LET count_even = MAX(count_even + 1, count_odd)  
            ELSE  
                LET count_odd = MAX(count_odd + 1, count_even)  
            END IF  
            index = index + 1  
        END WHILE  
        LET maximum_length = MAX(count_even, count_odd) + 1  
        RETURN maximum_length  
    END FUNCTION  
END CLASS
