CLASS Solution  
    FUNCTION countSubstrings(s)  
        LET length_of_s BE LENGTH(s)  
        LET substrings_count BE 0  
        LET start_index BE 0  
        WHILE start_index < length_of_s  
            LET number_accumulator BE 0  
            LET end_index BE start_index  
            WHILE end_index <= length_of_s - 1  
                LET digit_char BE s[end_index]  
                LET digit_value BE digit_char - '0'  
                number_accumulator = (number_accumulator * 10) + digit_value  
                IF digit_value <> 0 AND (number_accumulator MOD digit_value) == 0 THEN  
                    substrings_count = substrings_count + 1  
                END IF  
                end_index = end_index + 1  
            END WHILE  
            start_index = start_index + 1  
        END WHILE  
        RETURN substrings_count  
    END FUNCTION  
END CLASS
