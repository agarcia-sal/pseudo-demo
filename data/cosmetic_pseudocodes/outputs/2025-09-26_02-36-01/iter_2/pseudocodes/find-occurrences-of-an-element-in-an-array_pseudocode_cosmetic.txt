CLASS Solution
    FUNCTION occurrencesOfElement(nums, queries, x)
        INITIALIZE positions AS an empty list
        INITIALIZE idx AS 0
        WHILE idx < LENGTH(nums)
            INITIALIZE currentValue AS nums[idx]
            IF currentValue = x THEN
                SET positions = positions + [idx]
            END IF
            SET idx = idx + 1
        END WHILE
        INITIALIZE results AS an empty collection
        SET qIndex = 0
        WHILE qIndex < SIZE(queries)
            LET target = queries[qIndex]
            IF NOT(target > SIZE(positions)) THEN
                LET resultIndex = target - 1
                SET results = results + [positions[resultIndex]]
            ELSE
                SET results = results + [-1]
            END IF
            SET qIndex = qIndex + 1
        END WHILE
        RETURN results
    END FUNCTION
END CLASS
