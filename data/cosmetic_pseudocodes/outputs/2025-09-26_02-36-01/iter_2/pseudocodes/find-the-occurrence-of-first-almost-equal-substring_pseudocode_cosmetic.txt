CLASS Solution
    FUNCTION minStartingIndex(s, pattern)
        FUNCTION is_almost_equal(subseq, pat)
            SET mismatch_accumulator TO 0
            SET pos TO 0
            WHILE pos < LENGTH(subseq) AND mismatch_accumulator <= 1
                SET char_left TO subseq[pos]
                SET char_right TO pat[pos]
                IF char_left != char_right THEN
                    mismatch_accumulator = mismatch_accumulator + 1
                END IF
                SET pos TO pos + 1
            END WHILE
            IF mismatch_accumulator <= 1 THEN
                RETURN True
            ELSE
                RETURN False
            END IF
        END FUNCTION

        SET limit TO LENGTH(s) - LENGTH(pattern)
        SET start_idx TO 0
        WHILE start_idx <= limit
            SET segment TO s[start_idx : start_idx + LENGTH(pattern)]
            IF is_almost_equal(segment, pattern) THEN
                RETURN start_idx
            END IF
            start_idx = start_idx + 1
        END WHILE
        RETURN -1
    END FUNCTION
END CLASS
