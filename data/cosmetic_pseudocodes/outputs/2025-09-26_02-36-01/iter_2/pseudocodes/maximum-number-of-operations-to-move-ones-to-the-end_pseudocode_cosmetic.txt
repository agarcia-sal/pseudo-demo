CLASS Solution  
    FUNCTION maxOperations(s)  
        LET total BE 0  
        LET countOnes BE 0  
        LET position BE 0  
        WHILE position < LENGTH(s)  
            LET currentChar BE s[position]  
            IF currentChar == '1' THEN  
                countOnes = countOnes + 1  
            ELSE IF position > 0 THEN  
                LET prevChar BE s[position - 1]  
                IF prevChar == '1' THEN  
                    total = total + countOnes  
                END IF  
            END IF  
            position = position + 1  
        END WHILE  
        RETURN total  
    END FUNCTION  
END CLASS
