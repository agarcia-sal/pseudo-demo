CLASS Solution
    FUNCTION findWinningPlayer(skills, k)
        LET total_players BE LENGTH(skills)
        LET lineup BE EMPTY LIST
        FOR index FROM 0 TO total_players - 1
            APPEND index TO lineup
        END FOR

        LET score_streak BE 0
        LET champion BE POP FRONT OF lineup

        WHILE score_streak < k AND LENGTH(lineup) > 0
            LET challenger BE POP FRONT OF lineup

            IF skills[champion] - skills[challenger] > 0
                SET score_streak TO score_streak + 1
                APPEND challenger TO lineup
            ELSE
                SET score_streak TO 1
                APPEND champion TO lineup
                SET champion TO challenger
            END IF
        END WHILE

        RETURN champion
    END FUNCTION
END CLASS
