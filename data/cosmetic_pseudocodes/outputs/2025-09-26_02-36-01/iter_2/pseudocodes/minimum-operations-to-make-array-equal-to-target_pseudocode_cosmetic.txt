CLASS Solution  
    FUNCTION minimumOperations(nums, target)  
        LET length BE LENGTH(nums)  
        LET result BE ABS(target[0] - nums[0])  
        LET index BE 1  
        WHILE index < length  
            LET diffCurrent BE target[index] - nums[index]  
            LET diffPrev BE target[index - 1] - nums[index - 1]  
            IF diffCurrent * diffPrev > 0 THEN  
                LET diffAbsCurrent BE ABS(diffCurrent)  
                LET diffAbsPrev BE ABS(diffPrev)  
                LET delta BE diffAbsCurrent - diffAbsPrev  
                IF delta > 0 THEN  
                    result = result + delta  
                END IF  
            ELSE  
                result = result + ABS(diffCurrent)  
            END IF  
            index = index + 1  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
