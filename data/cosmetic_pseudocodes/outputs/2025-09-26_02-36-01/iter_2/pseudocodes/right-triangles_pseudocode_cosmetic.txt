CLASS Solution  
    FUNCTION numberOfRightTriangles(grid)  
        LET rowTotal = LENGTH(grid)  
        LET colTotal = LENGTH(grid[0])  
        LET triangleSum = 0  
        
        LET rowIndex = 0  
        WHILE rowIndex < rowTotal  
            LET colIndex = 0  
            WHILE colIndex < colTotal  
                LET cellValue = grid[rowIndex][colIndex]  
                IF cellValue EQUALS 1 THEN  
                    LET rowAccumulator = 0  
                    LET colAccumulator = 0  
                    
                    LET colChecker = 0  
                    WHILE colChecker < colTotal  
                        IF colChecker NOT EQUAL colIndex THEN  
                            rowAccumulator = rowAccumulator ADD grid[rowIndex][colChecker]  
                        END IF  
                        colChecker = colChecker ADD 1  
                    END WHILE  
                    
                    LET rowChecker = 0  
                    WHILE rowChecker < rowTotal  
                        IF rowChecker IS NOT EQUAL TO rowIndex THEN  
                            colAccumulator = colAccumulator ADD grid[rowChecker][colIndex]  
                        END IF  
                        rowChecker = rowChecker PLUS 1  
                    END WHILE  
                    
                    triangleSum = triangleSum ADD (rowAccumulator TIMES colAccumulator)  
                END IF  
                colIndex = colIndex PLUS 1  
            END WHILE  
            rowIndex = rowIndex PLUS 1  
        END WHILE  
        
        RETURN triangleSum  
    END FUNCTION  
END CLASS
