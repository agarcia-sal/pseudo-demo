CLASS Solution
    FUNCTION numberOfSubstrings(s)
        LET lengthVal BE LENGTH(s)
        LET resultCounter BE 0
        LET idxStart BE 0
        WHILE idxStart <= lengthVal - 1
            LET tallyOnes BE 0
            LET tallyZeros BE 0
            LET idxEnd BE idxStart
            WHILE idxEnd <= lengthVal - 1
                LET currentChar BE s[idxEnd]
                IF currentChar == '1' THEN
                    tallyOnes = tallyOnes + 1
                ELSE
                    tallyZeros = tallyZeros + 1
                END IF
                IF tallyOnes >= (tallyZeros * tallyZeros) THEN
                    resultCounter = resultCounter + 1
                END IF
                idxEnd = idxEnd + 1
            END WHILE
            idxStart = idxStart + 1
        END WHILE
        RETURN resultCounter
    END FUNCTION
END CLASS
