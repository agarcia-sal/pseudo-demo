CLASS Solution
    FUNCTION maxPointsInsideSquare(points s)
        LET total_points BE LENGTH s
        LET highest_count BE 0
        LET index_i BE 0
        WHILE index_i < total_points DO
            LET coordinate_x BE s[index_i][0]
            LET coordinate_y BE s[index_i][1]
            
            LET absolute_x BE coordinate_x
            IF absolute_x < 0 THEN
                absolute_x = 0 - absolute_x
            END IF

            LET absolute_y BE coordinate_y
            IF absolute_y < 0 THEN
                absolute_y = 0 - absolute_y
            END IF

            LET square_edge BE absolute_x
            IF absolute_y > square_edge THEN
                square_edge = absolute_y
            END IF

            LET observed_tags DICTIONARY = { }
            LET is_valid_square BE true

            LET index_j BE 0
            WHILE index_j < total_points DO
                LET point_x BE s[index_j][0]
                LET point_y BE s[index_j][1]

                LET abs_x_check BE point_x
                IF abs_x_check < 0 THEN
                    abs_x_check = 0 - abs_x_check
                END IF

                LET abs_y_check BE point_y
                IF abs_y_check < 0 THEN
                    abs_y_check = 0 - abs_y_check
                END IF
                
                IF abs_x_check <= square_edge AND abs_y_check <= square_edge THEN
                    LET current_tag BE s[index_j]
                    IF current_tag IN observed_tags THEN
                        is_valid_square = false
                        BREAK
                    ELSE
                        observed_tags[current_tag] = true
                    END IF
                END IF
                index_j = index_j + 1
            END WHILE
            
            IF is_valid_square THEN
                LET count_tags BE LENGTH observed_tags
                IF count_tags > highest_count THEN
                    highest_count = count_tags
                END IF
            END IF

            index_i = index_i + 1
        END WHILE

        RETURN highest_count
    END FUNCTION
END CLASS
