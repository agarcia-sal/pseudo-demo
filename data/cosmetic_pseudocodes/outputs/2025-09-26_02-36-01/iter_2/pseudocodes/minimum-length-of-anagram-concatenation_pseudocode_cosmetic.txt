CLASS Solution  
    FUNCTION minAnagramLength(s)  
        DECLARE distinctSymbols AS SET  
        INITIALIZE distinctSymbols WITH EMPTY SET  
        DECLARE indexCounter AS INTEGER  
        indexCounter = 0  
        WHILE indexCounter < LENGTH(s)  
            DECLARE currentElement AS CHARACTER  
            currentElement = s[indexCounter]  
            ADD currentElement TO distinctSymbols  
            indexCounter = indexCounter + 1  
        END WHILE  
        DECLARE resultValue AS INTEGER  
        resultValue = CARDINALITY(distinctSymbols)  
        RETURN resultValue  
    END FUNCTION  
END CLASS
