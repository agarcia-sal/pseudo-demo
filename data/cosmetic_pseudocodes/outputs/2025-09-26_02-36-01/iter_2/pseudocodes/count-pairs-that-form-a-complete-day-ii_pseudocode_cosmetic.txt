CLASS Solution  
    FUNCTION countCompleteDayPairs(hours)  
        let modulo_freqs be a map defaulting to zero  
        let total_pairs be 0  
        let idx be 0  
        WHILE idx < length of hours  
            let current_hour be hours[idx]  
            let mod_val be current_hour - (current_hour DIV 24) * 24  
            let needed_complement be (24 - mod_val) MOD 24  
            total_pairs = total_pairs + modulo_freqs[needed_complement]  
            modulo_freqs[mod_val] = modulo_freqs[mod_val] + 1  
            idx = idx + 1  
        END WHILE  
        RETURN total_pairs  
    END FUNCTION  
END CLASS
