CLASS Solution  
    FUNCTION minimumArea(grid)  
        IF grid IS EMPTY OR grid[0] IS EMPTY THEN  
            RETURN 0  
        END IF  

        LET total_rows BE LENGTH(grid)  
        LET total_columns BE LENGTH(grid[0])  

        LET top_boundary BE +∞  
        LET bottom_boundary BE -∞  
        LET left_boundary BE +∞  
        LET right_boundary BE -∞  

        LET row_index BE 0  
        WHILE row_index < total_rows DO  
            LET col_index BE 0  
            WHILE col_index < total_columns DO  
                IF grid[row_index][col_index] == 1 THEN  
                    IF top_boundary > row_index THEN  
                        top_boundary = row_index  
                    END IF  
                    IF bottom_boundary < row_index THEN  
                        bottom_boundary = row_index  
                    END IF  
                    IF left_boundary > col_index THEN  
                        left_boundary = col_index  
                    END IF  
                    IF right_boundary < col_index THEN  
                        right_boundary = col_index  
                    END IF  
                END IF  
                col_index = col_index + 1  
            END WHILE  
            row_index = row_index + 1  
        END WHILE  

        LET vertical_span BE (bottom_boundary - top_boundary) + 1  
        LET horizontal_span BE (right_boundary - left_boundary) + 1  
        LET resultArea = vertical_span * horizontal_span  

        RETURN resultArea  
    END FUNCTION  
END CLASS
