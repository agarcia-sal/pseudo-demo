CLASS Solution  
    FUNCTION minimumLevels(possible)  
        LET sum_points BE 0  
        LET i BE 0  
        WHILE i < LENGTH(possible)  
            LET val BE possible[i]  
            LET increment BE (val TIMES 2) MINUS 1  
            sum_points = sum_points + increment  
            i = i + 1  
        END WHILE  
  
        LET alice_sum BE 0  
        LET j BE 0  
        LET limit BE LENGTH(possible) - 2  
        WHILE j <= limit  
            LET curr_val BE possible[j]  
            LET add_val BE (curr_val TIMES 2) - 1  
            alice_sum = alice_sum + add_val  
            sum_points = sum_points - add_val  
            IF alice_sum > sum_points  
                RETURN j + 1  
            END IF  
            j = j + 1  
        END WHILE  
  
        RETURN NEGATIVE 1  
    END FUNCTION  
END CLASS
