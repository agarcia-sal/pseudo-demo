CLASS Solution  
    FUNCTION numberOfWays(n)  
        CONSTANT modulus ← 7 + 10 ^ 9  
        CREATE table AS an array filled with 0 of length (n + 1)  
        ASSIGN table[0] ← 1  
        FOR coinValue IN [1, 2, 6] DO  
            SET idx ← coinValue  
            WHILE idx ≤ n DO  
                tempSum ← table[idx] + table[idx - coinValue]  
                table[idx] ← tempSum MOD modulus  
                INCREMENT idx BY 1  
            END WHILE  
        END FOR  
        SET totalWays ← 0  
        SET counter ← 0  
        WHILE counter ≤ 2 DO  
            IF (counter × 4) ≤ n THEN  
                offset ← n - (counter * 4)  
                totalWays ← (totalWays + table[offset]) MOD modulus  
            END IF  
            counter ← counter + 1  
        END WHILE  
        RETURN totalWays  
    END FUNCTION  
END CLASS
