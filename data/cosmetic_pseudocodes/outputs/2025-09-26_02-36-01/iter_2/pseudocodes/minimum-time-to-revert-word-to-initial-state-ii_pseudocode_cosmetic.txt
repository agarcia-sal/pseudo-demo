CLASS Solution
    FUNCTION minimumTimeToInitialState(word STRING, k INTEGER) RETURNS INTEGER
        LET length_of_word BE LENGTH(word)
        LET counter BE 1
        LOOP
            LET segment BE word[(counter * k) : length_of_word]
            IF word BEGINS_WITH segment THEN
                RETURN counter
            END IF
            counter = counter + 1
        END LOOP
    END FUNCTION
END CLASS
