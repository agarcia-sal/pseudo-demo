CLASS Solution
    FUNCTION countSubstrings(s inputString, c targetChar)
        LET totalOccurrences BE 0
        LET lengthOfString BE LENGTH(inputString)
        LET index BE 0
        WHILE index < lengthOfString
            IF inputString[index] = targetChar THEN
                totalOccurrences := totalOccurrences + 1
            END IF
            index := index + 1
        END WHILE
        LET intermediateSum BE totalOccurrences + 1
        LET product BE totalOccurrences * intermediateSum
        LET finalResult BE product / 2
        RETURN finalResult
    END FUNCTION
END CLASS
