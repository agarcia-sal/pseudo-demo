CLASS Solution  
    FUNCTION minOperations(nums parameter k parameter)  
        heapify(nums)  
        operationsCount := 0  
        WHILE (nums[0] < k) AND (LENGTH(nums) > 1) DO  
            firstElement := heappop(nums)  
            secondElement := heappop(nums)  
            smallerVal := MIN(firstElement, secondElement)  
            largerVal := MAX(firstElement, secondElement)  
            combined := (smallerVal * 2) + largerVal  
            heappush(nums, combined)  
            operationsCount := operationsCount + 1  
        END WHILE  
        RETURN operationsCount  
    END FUNCTION  
END CLASS
