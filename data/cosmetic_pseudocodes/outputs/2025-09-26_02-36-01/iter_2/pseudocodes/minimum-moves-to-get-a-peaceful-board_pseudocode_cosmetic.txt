CLASS Solution
    FUNCTION minMoves(rooks)
        LET total_rooks BE LENGTH(rooks)
        LET sorted_by_row BE SORT(rooks) USING (element) -> element[0]
        LET sorted_by_col BE SORT(rooks) USING (element) -> element[1]

        LET moves_rows BE 0
        LET idx_row BE 0
        WHILE idx_row < total_rooks
            LET current_row_pos BE sorted_by_row[idx_row][0]
            LET displacement_row BE current_row_pos
            IF displacement_row < idx_row THEN
                LET displacement_row TO idx_row MINUS displacement_row
            ELSE
                LET displacement_row TO displacement_row MINUS idx_row
            END IF
            moves_rows = moves_rows + displacement_row
            idx_row = idx_row + 1
        END WHILE

        LET moves_columns BE 0
        LET idx_col BE 0
        WHILE idx_col < total_rooks
            LET current_col_pos BE sorted_by_col[idx_col][1]
            LET displacement_col BE current_col_pos
            IF displacement_col < idx_col THEN
                LET displacement_col TO idx_col MINUS displacement_col
            ELSE
                LET displacement_col TO displacement_col MINUS idx_col
            END IF
            moves_columns = moves_columns + displacement_col
            idx_col = idx_col + 1
        END WHILE

        LET final_moves BE moves_rows + moves_columns
        RETURN final_moves
    END FUNCTION
END CLASS
