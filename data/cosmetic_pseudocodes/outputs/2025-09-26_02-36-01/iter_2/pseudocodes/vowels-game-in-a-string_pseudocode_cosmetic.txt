CLASS Solution  
    FUNCTION doesAliceWin(s)  
        DEFINE vowelSet AS {'a', 'e', 'i', 'o', 'u'}  
        LET oddCountSegments = 0  
        LET vowelRunLength = 0  
        
        LET index = 0  
        WHILE index < LENGTH(s)  
            LET ch = s[index]  
            IF vowelSet CONTAINS ch THEN  
                vowelRunLength = vowelRunLength + 1  
            END IF  
            index = index + 1  
        END WHILE  
        
        LET oddSegmentFlag = FALSE  
        LET pointer = 0  
        WHILE pointer < LENGTH(s)  
            LET currentChar = s[pointer]  
            IF vowelSet CONTAINS currentChar THEN  
                oddSegmentFlag = NOT oddSegmentFlag  
            END IF  
            IF (oddSegmentFlag = FALSE) AND ((vowelRunLength MOD 2) = 1) THEN  
                oddCountSegments = oddCountSegments + 1  
                vowelRunLength = 0  
            ELSE IF oddSegmentFlag = TRUE THEN  
                vowelRunLength = vowelRunLength + 1  
            END IF  
            pointer = pointer + 1  
        END WHILE  
        
        IF (oddSegmentFlag = TRUE) AND ((vowelRunLength MOD 2) = 1) THEN  
            oddCountSegments = oddCountSegments + 1  
        END IF  
        
        LET result = (oddCountSegments MOD 2) = 1  
        RETURN result  
    END FUNCTION  
END CLASS
