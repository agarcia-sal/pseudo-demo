CLASS Solution
    FUNCTION largestPalindrome(n, k)
        IF n IS 1 THEN
            SET digitVar TO 9
            WHILE digitVar IS GREATER OR EQUAL TO 1 DO
                IF (digitVar - (digitVar DIV k) * k) IS 0 THEN
                    RETURN TO_STRING(digitVar)
                END IF
                SET digitVar TO digitVar - 1
            END WHILE
            RETURN "0"
        END IF

        SET halfLen TO (n + 1) DIV 2
        SET nineStr TO "" 
        SET counter TO 0
        WHILE counter < halfLen DO
            nineStr <- nineStr + "9"
            SET counter TO counter + 1
        END WHILE

        SET halfVal TO TO_INT(nineStr)

        WHILE halfVal > 0 DO
            SET halfStr TO TO_STRING(halfVal)
            IF (n MOD 2) IS 0 THEN
                SET fullStr TO halfStr + REVERSE(halfStr)
            ELSE
                SET fullStr TO halfStr + REVERSE(SUBSTRING(halfStr, 0, LENGTH(halfStr) - 1))
            END IF

            SET fullNum TO TO_INT(fullStr)

            IF (fullNum - (fullNum DIV k) * k) IS 0 THEN
                RETURN fullStr
            END IF

            SET halfVal TO halfVal - 1
        END WHILE

        RETURN "0"
    END FUNCTION
END CLASS
