CLASS Solution  
    FUNCTION makeAntiPalindrome(s)  
        LET charArray BE array formed by sorting characters of s in non-decreasing order  
        LET lengthVal BE length of charArray  
        LET halfIndex BE lengthVal DIVIDED BY 2  
        
        IF charArray[halfIndex] == charArray[halfIndex - 1] THEN  
            LET forwardIdx BE halfIndex  
            WHILE forwardIdx < lengthVal AND charArray[forwardIdx] == charArray[forwardIdx - 1] DO  
                forwardIdx = forwardIdx + 1  
            END WHILE  
            
            LET backwardIdx BE halfIndex  
            WHILE backwardIdx < lengthVal AND charArray[backwardIdx] == charArray[lengthVal - backwardIdx - 1] DO  
                IF forwardIdx >= lengthVal THEN  
                    RETURN "-1"  
                END IF  
                TEMP_VAR = charArray[forwardIdx]  
                charArray[forwardIdx] = charArray[backwardIdx]  
                charArray[backwardIdx] = TEMP_VAR  
                forwardIdx = forwardIdx + 1  
                backwardIdx = backwardIdx + 1  
            END WHILE  
        END IF  
        
        LET outerIndex BE 0  
        WHILE outerIndex <= halfIndex - 1 DO  
            IF charArray[outerIndex] == charArray[lengthVal - outerIndex - 1] THEN  
                LET replaced BE false  
                LET innerIndex BE halfIndex  
                WHILE innerIndex < lengthVal DO  
                    IF charArray[innerIndex] != charArray[outerIndex] AND charArray[innerIndex] != charArray[lengthVal - outerIndex - 1] THEN  
                        TEMP_STORE = charArray[innerIndex]  
                        charArray[innerIndex] = charArray[outerIndex]  
                        charArray[outerIndex] = TEMP_STORE  
                        replaced = true  
                        EXIT WHILE  
                    END IF  
                    innerIndex = innerIndex + 1  
                END WHILE  
                IF replaced == false THEN  
                    RETURN "-1"  
                END IF  
            END IF  
            outerIndex = outerIndex + 1  
        END WHILE  
        
        RETURN concatenation of elements in charArray into string  
    END FUNCTION  
END CLASS
