CLASS Solution  
    FUNCTION queryResults(limit parameter, queries parameter)  
        DECLARE color_map AS empty dictionary  
        DECLARE distinct_colors AS empty set  
        DECLARE output AS empty list  
        
        SET idx TO 0  
        WHILE idx < LENGTH(queries parameter)  
            LET current_pair = queries parameter[idx]  
            LET key_elem = current_pair[0]  
            LET color_elem = current_pair[1]  
            
            IF color_map CONTAINS key_elem THEN  
                LET previous_color = color_map[key_elem]  
                IF previous_color IN distinct_colors THEN  
                    CALL distinct_colors.REMOVE(previous_color)  
                END IF  
            END IF  
            
            SET color_map[key_elem] = color_elem  
            CALL distinct_colors.ADD(color_elem)  
            
            LET unique_count = SIZE(distinct_colors)  
            APPEND unique_count TO output  
            
            SET idx = idx + 1  
        END WHILE  
        
        RETURN output  
    END FUNCTION  
END CLASS
