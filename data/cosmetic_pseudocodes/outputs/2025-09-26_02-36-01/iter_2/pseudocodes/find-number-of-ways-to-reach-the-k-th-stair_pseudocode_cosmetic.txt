CLASS Solution
    FUNCTION waysToReachStair(k)
        FUNCTION dfs(currentStep, toggle, jumpCount)
            IF currentStep > (k + 1)
                RETURN 0
            END IF

            tempResult ← (currentStep = k) * 1

            IF (currentStep > 0) AND (toggle = 0)
                tempResult ← tempResult + dfs(currentStep - 1, 1, jumpCount)
            END IF

            nextStep ← currentStep + 2 * jumpCount
            tempResult ← tempResult + dfs(nextStep, 0, jumpCount + 1)

            RETURN tempResult
        END FUNCTION

        RETURN dfs(1, 0, 0)
    END FUNCTION
END CLASS
