CLASS Solution  
    FUNCTION minimumDifference(nums k)  
        FUNCTION subarray_or(left right)  
            LET accumulator BE 0  
            LET index BE left  
            WHILE index <= right DO  
                LET accumulator BE accumulator BITWISE_OR nums[index]  
                SET index TO index PLUS 1  
            END WHILE  
            RETURN accumulator  
        END FUNCTION  

        LET length BE LENGTH(nums)  
        LET minimal_difference BE +âˆž  

        LET start_idx BE 0  
        WHILE start_idx < length DO  
            LET or_result BE 0  
            LET finish_idx BE start_idx  
            WHILE finish_idx < length DO  
                LET or_result BE or_result BITWISE_OR nums[finish_idx]  
                LET difference BE ABS(k MINUS or_result)  
                IF difference < minimal_difference THEN  
                    SET minimal_difference TO difference  
                END IF  
                IF difference == 0 THEN  
                    RETURN 0  
                END IF  
                finish_idx = finish_idx PLUS 1  
            END WHILE  
            start_idx = start_idx PLUS 1  
        END WHILE  
        
        RETURN minimal_difference  
    END FUNCTION  
END CLASS
