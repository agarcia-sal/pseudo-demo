CLASS Solution  
    FUNCTION minRectanglesToCoverPoints(points, w)  
        DECLARE sortedPoints AS points SORTED BY first element IN ascending order  
        SET countRectangles TO 0  
        SET limitX TO -1  
        DECLARE idx TO 0  
        WHILE idx < LENGTH(sortedPoints) DO  
            LET xCoord, yCoord BE sortedPoints[idx]  
            IF NOT (xCoord <= limitX) THEN  
                SET limitX TO xCoord PLUS w  
                SET countRectangles TO countRectangles + 1  
            END IF  
            SET idx TO idx + 1  
        END WHILE  
        RETURN countRectangles  
    END FUNCTION  
END CLASS
