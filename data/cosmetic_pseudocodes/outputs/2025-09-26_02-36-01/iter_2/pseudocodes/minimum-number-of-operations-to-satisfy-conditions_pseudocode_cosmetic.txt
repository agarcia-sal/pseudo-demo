CLASS Solution
    FUNCTION minimumOperations(grid)
        LET rowCount BE LENGTH(grid)
        LET colCount BE LENGTH(grid[0])
        LET totalOps BE 0
        
        LET colIndex BE 0
        WHILE colIndex < colCount
            LET rowIndex BE 0
            WHILE rowIndex < rowCount - 1
                IF NOT (grid[rowIndex][colIndex] = grid[rowIndex + 1][colIndex])
                    totalOps = totalOps + 1
                    grid[rowIndex + 1][colIndex] = grid[rowIndex][colIndex]
                END IF
                rowIndex = rowIndex + 1
            END WHILE
            
            LET idx BE 0
            WHILE idx < rowCount
                IF colIndex < colCount - 1 AND (grid[idx][colIndex] = grid[idx][colIndex + 1])
                    totalOps += 1
                    LET replacementFound BE FALSE
                    LET candidate BE 0
                    WHILE candidate <= 9 AND NOT replacementFound
                        IF candidate != grid[idx][colIndex]
                            grid[idx][colIndex + 1] = candidate
                            replacementFound = TRUE
                        END IF
                        candidate = candidate + 1
                    END WHILE
                END IF
                idx = idx + 1
            END WHILE
            
            colIndex = colIndex + 1
        END WHILE
        
        RETURN totalOps
    END FUNCTION
END CLASS
