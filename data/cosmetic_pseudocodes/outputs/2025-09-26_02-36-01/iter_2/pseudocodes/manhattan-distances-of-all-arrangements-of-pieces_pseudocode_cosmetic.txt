CLASS Solution
    FUNCTION distanceSum WITH PARAMETERS m n k
        ASSIGN modulus <- 1000000007
        ASSIGN rows_squared <- m MULTIPLY m
        ASSIGN cols_squared <- n MULTIPLY n
        ASSIGN rows_cubed <- rows_squared MULTIPLY m
        ASSIGN cols_cubed <- cols_squared MULTIPLY n
        
        ASSIGN numerator_rows <- rows_cubed MINUS m
        ASSIGN numerator_cols <- cols_cubed MINUS n
        
        ASSIGN part_rows <- cols_squared MULTIPLY numerator_rows DIVIDE 6
        ASSIGN part_cols <- rows_squared MULTIPLY numerator_cols DIVIDE 6
        
        ASSIGN total_points <- m MULTIPLY n
        ASSIGN combination_factor <- COMBIN(total_points MINUS 2, k MINUS 2)
        
        ASSIGN sum_distance <- (part_rows PLUS part_cols) MULTIPLY combination_factor
        
        ASSIGN remainder <- sum_distance MOD modulus
        
        RETURN remainder
    END FUNCTION
END CLASS
