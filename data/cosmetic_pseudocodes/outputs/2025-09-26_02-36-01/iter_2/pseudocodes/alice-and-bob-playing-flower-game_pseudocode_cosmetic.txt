CLASS Solution  
    FUNCTION flowerGame(n, m)  
        LET half_ceil_n BE (n + 1) / 2  
        LET half_floor_n BE n DIV 2  
        LET half_ceil_m BE (m + 1) / 2  
        LET half_floor_m BE m DIV 2  
        LET mixed_pairs1 BE half_ceil_n * half_floor_m  
        LET mixed_pairs2 BE half_floor_n * half_ceil_m  
        LET combined_pairs BE mixed_pairs1 + mixed_pairs2  
        RETURN combined_pairs  
    END FUNCTION  
END CLASS
