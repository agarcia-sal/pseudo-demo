CLASS Solution
    FUNCTION minimumLength(s)
        LET frequency_map BE Counter(s)
        LET odd_count BE 0
        LET even_sum BE 0
        LET iterator BE 0
        LET value_list BE list of values in frequency_map
        WHILE iterator < length of value_list DO
            LET current_val BE value_list[iterator]
            IF NOT (current_val % 2 == 0) THEN
                odd_count <- odd_count + 1
            ELSE IF (current_val % 2 == 0) AND (current_val != 0) THEN
                even_sum <- even_sum + 1 + 1
            END IF
            iterator <- iterator + 1
        END WHILE
        LET total_length BE odd_count + even_sum
        RETURN total_length
    END FUNCTION
END CLASS
