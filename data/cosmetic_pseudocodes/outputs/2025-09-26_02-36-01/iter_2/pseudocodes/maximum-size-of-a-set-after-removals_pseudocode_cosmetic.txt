CLASS Solution
    FUNCTION maximumSetSize(nums1 PARAMETER List OF integers nums2 PARAMETER List OF integers) RETURNS integer
        LET length_nums1 BE LENGTH(nums1)
        LET half_length BE length_nums1 INTEGER_DIVIDE 2

        LET unique_nums1 BE CONVERT_TO_SET(nums1)
        LET unique_nums2 BE CONVERT_TO_SET(nums2)

        LET intersect_sets BE unique_nums1 INTERSECT unique_nums2

        LET only_in_nums1 BE unique_nums1 MINUS intersect_sets
        LET only_in_nums2 BE unique_nums2 MINUS intersect_sets

        LET part1 TAKE MINIMUM(half_length, LENGTH(only_in_nums1))
        LET part2 TAKE MINIMUM(half_length, LENGTH(only_in_nums2))

        LET remaining1 BE MAXIMUM(0, half_length - part1)
        LET remaining2 BE MAXIMUM(0, half_length - part2)
        LET remaining_total BE remaining1 + remaining2

        LET count_common TAKE MINIMUM(remaining_total, LENGTH(intersect_sets))

        LET result BE part1 + part2 + count_common

        RETURN result
    END FUNCTION
END CLASS
