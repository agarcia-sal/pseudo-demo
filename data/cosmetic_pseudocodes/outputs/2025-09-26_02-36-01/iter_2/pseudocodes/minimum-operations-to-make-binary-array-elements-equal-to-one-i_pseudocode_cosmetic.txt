CLASS Solution
    FUNCTION minOperations(arr)
        LET lengthVar BE SIZE OF arr
        LET countOps BE 0
        LET indexVar BE 0
        WHILE indexVar <= lengthVar - 3
            IF arr[indexVar] == 0 THEN
                LET temp0 BE arr[indexVar]
                LET temp1 BE arr[indexVar + 1]
                LET temp2 BE arr[indexVar + 2]
                
                arr[indexVar] = 1 - temp0
                arr[indexVar + 1] = 1 - temp1
                arr[indexVar + 2] = 1 - temp2
                
                countOps = countOps + 1
            END IF
            indexVar = indexVar + 1
        END WHILE
        
        IF (arr[lengthVar - 1] == 0) OR NOT (arr[lengthVar - 2] != 0) THEN
            RETURN -1
        END IF
        
        RETURN countOps
    END FUNCTION
END CLASS
