CLASS Solution
    FUNCTION minLength(s STRING, numOps INTEGER) RETURNS INTEGER
        FUNCTION check(m INTEGER) RETURNS BOOLEAN
            DECLARE groupCount INTEGER = 0
            DECLARE runLength INTEGER = 0
            DECLARE pos INTEGER = 0
            DECLARE totalLength INTEGER = LENGTH(s)
            WHILE pos < totalLength
                INCREMENT runLength BY 1
                IF pos = totalLength - 1 OR s[pos] != s[pos + 1]
                    DECLARE groupsInRun INTEGER = runLength DIV m + 1
                    groupCount = groupCount + groupsInRun
                    IF groupCount > numOps
                        RETURN FALSE
                    END IF
                    runLength = 0
                END IF
                pos = pos + 1
            END WHILE
            RETURN groupCount <= numOps
        END FUNCTION

        DECLARE lengthOfS INTEGER = LENGTH(s)
        DECLARE low INTEGER = 1
        DECLARE high INTEGER = lengthOfS
        WHILE low < high
            DECLARE median INTEGER = low + (high - low) DIV 2
            IF check(median)
                high = median
            ELSE
                low = median + 1
            END IF
        END WHILE
        RETURN low
    END FUNCTION
END CLASS
