CLASS Solution  
    FUNCTION minOperations(nums)  
        LET length BE SIZE(nums)  
        IF NOT (length > 0) THEN  
            RETURN 0  
        END IF  
        LET sequenceLengths BE ARRAY FILLED WITH 1 OF SIZE length  
        LET currentIndex BE 1  
        WHILE currentIndex < length DO  
            LET previousIndex BE 0  
            WHILE previousIndex < currentIndex DO  
                IF nums[currentIndex] <= nums[previousIndex] THEN  
                    LET candidateLength BE sequenceLengths[previousIndex] + 1  
                    IF candidateLength > sequenceLengths[currentIndex] THEN  
                        SET sequenceLengths[currentIndex] TO candidateLength  
                    END IF  
                END IF  
                SET previousIndex TO previousIndex + 1  
            END WHILE  
            SET currentIndex TO currentIndex + 1  
        END WHILE  
        LET maxLength BE sequenceLengths[0]  
        LET pos BE 1  
        WHILE pos < length DO  
            IF sequenceLengths[pos] > maxLength THEN  
                SET maxLength TO sequenceLengths[pos]  
            END IF  
            SET pos TO pos + 1  
        END WHILE  
        RETURN maxLength  
    END FUNCTION  
END CLASS
