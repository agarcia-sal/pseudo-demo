CLASS Solution  
    FUNCTION maxArea(height arr dirs)  
        LET lengthVars BE SIZE OF arr  
        LET resMax BE SUM OVER arr  
        LET counter BE 1  
        WHILE counter <= height TIMES 2  
            LET idx BE 0  
            WHILE idx < lengthVars  
                LET posVal BE arr[idx]  
                LET dirVal BE dirs[idx]  
                IF (posVal == 0) AND (dirVal == 'D') THEN  
                    LET prefixStr BE SUBSTRING(dirs, 0, idx)  
                    LET suffixStr BE SUBSTRING(dirs, idx + 1, LENGTH(dirs))  
                    LET dirs BE prefixStr CONCAT 'U' CONCAT suffixStr  
                ELSE IF (posVal == height) AND (dirVal == 'U') THEN  
                    LET prefixStrAlt BE SUBSTRING(dirs, 0, idx)  
                    LET suffixStrAlt BE SUBSTRING(dirs, idx + 1, LENGTH(dirs))  
                    LET dirs BE prefixStrAlt CONCAT 'D' CONCAT suffixStrAlt  
                END IF  
                IF dirs[idx] == 'U' THEN  
                    LET arr[idx] BE arr[idx] PLUS 1  
                ELSE  
                    LET arr[idx] BE arr[idx] MINUS 1  
                END IF  
                SET idx TO idx + 1  
            END WHILE  
            LET currSum BE SUM OVER arr  
            IF resMax < currSum THEN  
                LET resMax BE currSum  
            END IF  
            SET counter TO counter + 1  
        END WHILE  
        RETURN resMax  
    END FUNCTION  
END CLASS
