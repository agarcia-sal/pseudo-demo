CLASS Solution  
    FUNCTION mostFrequentIDs(nums PARAMETER, freq PARAMETER)  
        DECLARE tally AS dictionary defaulting to 0  
        DECLARE heap_container AS empty array  
        DECLARE output_list AS empty array  
        SET index TO 0  
        WHILE index < LENGTH(nums)  
            LET current_num BE nums[index]  
            LET current_freq BE freq[index]  
            LET updated_count BE tally[current_num] + current_freq  
            ASSIGN updated_count TO tally[current_num]  
            INSERT ( -updated_count , current_num ) INTO heap_container  
            WHILE LENGTH(heap_container) > 0 AND (-heap_container[0][0]) ≠ tally[heap_container[0][1]]  
                REMOVE heap_container[0]  
            END WHILE  
            IF LENGTH(heap_container) = 0 THEN  
                APPEND 0 TO output_list  
            ELSE  
                APPEND (-heap_container[0][0]) TO output_list  
            ENDIF  
            index ← index + 1  
        END WHILE  
        RETURN output_list  
    END FUNCTION  
END CLASS
