CLASS Solution  
    FUNCTION clearStars(input_string)  
        INITIALIZE container AS empty array  
        SET idx TO 0  
        WHILE idx < LENGTH(input_string)  
            LET current_char BE input_string[idx]  
            IF current_char == "*" THEN  
                IF SIZE(container) > 0 THEN  
                    REMOVE container[SIZE(container) - 1]  
                END IF  
            ELSE  
                INSERT current_char AT END OF container  
            END IF  
            idx = idx + 1  
        END WHILE  
        DECLARE output_string AS EMPTY STRING  
        FOR element IN container  
            output_string = output_string + element  
        END FOR  
        RETURN output_string  
    END FUNCTION  
END CLASS
