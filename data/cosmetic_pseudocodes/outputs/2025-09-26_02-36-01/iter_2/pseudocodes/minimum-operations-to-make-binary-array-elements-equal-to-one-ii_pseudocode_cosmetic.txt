CLASS Solution  
    FUNCTION minOperations(nums)  
        LET count = 0  
        LET toggles = 0  
        LET index = 0  
        WHILE index LESS THAN LENGTH OF nums  
            LET element  
            IF (toggles MOD 2) IS NOT 1  
                element = nums[index]  
            ELSE  
                element = 1 - nums[index]  
            END IF  
            IF element == 0  
                count = count + 1  
                toggles = toggles + 1  
            END IF  
            index = index + 1  
        END WHILE  
        RETURN count  
    END FUNCTION  
END CLASS
