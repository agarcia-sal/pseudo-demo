CLASS Solution
    FUNCTION maxScore(nums)
        SET length TO SIZE OF nums
        SET results TO ARRAY OF length ELEMENTS INITIALIZED WITH 0
        SET index TO length - 2
        WHILE index >= 0
            SET best_value TO 0
            SET inner_index TO index + 1
            WHILE inner_index < length
                SET gap TO inner_index - index
                SET current_score TO gap TIMES nums{inner_index}
                SET candidate TO current_score PLUS results{inner_index}
                IF best_value < candidate
                    best_value = candidate
                END IF
                SET inner_index TO inner_index + 1
            END WHILE
            results{index} = best_value
            SET index TO index - 1
        END WHILE
        RETURN results{0}
    END FUNCTION
END CLASS
