CLASS Solution
    FUNCTION maxNumber(n)
        IF NOT (n NOT EQUALS 1) THEN
            RETURN 0
        END IF

        LET leadingBit BE 1
        WHILE n >= leadingBit DO
            LET leadingBit BE leadingBit + leadingBit
        END WHILE

        LET leadingBit BE leadingBit / 2

        LET resultNum BE leadingBit - 1
        RETURN resultNum
    END FUNCTION
END CLASS
