CLASS Solution
  FUNCTION maxScore(nums)
      SET p TO LENGTH OF nums
      SET q TO a LIST OF ZEROES WITH SIZE p
      SET ELEMENT AT POSITION 1 OF q TO 0
      SET r TO p MINUS 1
      WHILE r GREATER THAN 0
        SET s TO 1
        WHILE s LESS THAN OR EQUAL TO r
          SET t TO q AT INDEX s
          SET u TO r MINUS s
          SET v TO nums AT INDEX r
          SET w TO t PLUS u MULTIPLIED BY v
          IF q AT INDEX r LESS THAN w
            SET q AT INDEX r TO w
          END IF
          INCREMENT s BY 1
        END WHILE
        DECREMENT r BY 1
      END WHILE
      RETURN q AT INDEX p MINUS 1
  END FUNCTION
END CLASS
