CLASS Solution
  FUNCTION distanceSum WITH PARAMETERS m n k
    SET a TO 1000000000 + 7
    SET b TO ((n * n) * ((m * m) * (m * m - m))) / 6
    SET c TO ((m * m) * ((n * n) * (n * n - n))) / 6

    FUNCTION helperCombination WITH PARAMETERS p q
        SET x TO 1
        SET y TO 1
        SET z TO 1
        SET r TO q
        WHILE r > 0 DO
          SET x TO (x * (p - r + 1))
          SET y TO y * r
          SET z TO x / y
          SET r TO r - 1
        END WHILE
        RETURN z
    END FUNCTION

    SET d TO helperCombination((m * n) - 2, k - 2)
    SET e TO (b + c) * d
    RETURN e % a
  END FUNCTION
END CLASS
