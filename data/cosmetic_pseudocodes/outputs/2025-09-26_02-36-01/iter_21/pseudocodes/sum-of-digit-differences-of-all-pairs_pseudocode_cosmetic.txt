CLASS Solution
	FUNCTION sumDigitDifferences(nums)
		FUNCTION digit_difference(a, b)
			SET r TO 0
			SET c TO 0
			WHILE c < LENGTH(a) DO
				IF NOT (a[c] = b[c]) THEN
					SET r TO r + 1
				END IF
				SET c TO c + 1
			END WHILE
			RETURN r
		END FUNCTION

		SET u TO 0
		SET v TO LENGTH(nums)
		SET w TO 0
	
		WHILE w < v DO
			SET x TO w + 1
			WHILE x < v DO
				SET u TO (u) + (digit_difference(nums[w], nums[x]))
				SET x TO x + 1
			END WHILE
			SET w TO w + 1
		END WHILE

		RETURN u
	END FUNCTION
END CLASS
