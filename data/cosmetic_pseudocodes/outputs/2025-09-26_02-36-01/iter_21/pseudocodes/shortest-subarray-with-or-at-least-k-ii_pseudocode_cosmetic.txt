CLASS Solution
    FUNCTION minimumSubarrayLength(nums LIST OF integers k integer) RETURNS integer
        FUNCTION f_1(a LIST OF integers b integer c integer)
            SET d TO integer 1
            SET e TO integer 0
            REPEAT
                IF (b BITWISE_AND d) NOT_EQUAL 0 THEN
                    SET a[e] TO a[e] PLUS c
                END IF
                SET d TO d PLUS d
                SET e TO e PLUS 1
            UNTIL e GREATER_THAN 31
        END FUNCTION

        FUNCTION f_2(g LIST OF integers) RETURNS integer
            SET h TO integer 0
            SET j TO integer 0
            WHILE j LESS_THAN_EQUAL 31 DO
                IF g[j] GREATER_THAN 0 THEN
                    SET h TO (h BITWISE_OR (integer 1 BITWISE_SHIFT_LEFT j))
                END IF
                SET j TO j PLUS 1
            END WHILE
            RETURN h
        END FUNCTION

        SET l TO LENGTH OF nums
        SET m TO a new LIST OF integers initially zero with size 32
        SET n TO integer 0
        SET o TO integer 0
        SET p TO positive infinity

        FUNCTION q()
            WHILE o LESS_THAN l DO
                CALL f_1 WITH m nums[o] integer 1

                SET n TO n BITWISE_OR nums[o]

                WHILE (n GREATER_THAN_OR_EQUAL k) AND (o GREATER_THAN_EQUAL p) DO
                    IF p GREATER_THAN (o MINUS p PLUS integer 1) THEN
                        SET p TO o MINUS p PLUS integer 1
                    END IF
                    CALL f_1 WITH m nums[p] -integer 1
                    SET n TO CALL f_2 WITH m
                    SET p TO p PLUS integer 1
                END WHILE

                SET o TO o PLUS integer 1
            END WHILE
        END FUNCTION

        SET r TO 0
        SET s TO 0
        SET t TO p
        SET u TO n
        SET v TO o

        SET w TO 0
        SET x TO positive infinity
        SET y TO integer 0

        SET z TO 0
        SET aa TO 0
        SET ab TO 0

        SET ac TO 0

        SET ad TO integer 0
        SET ae TO integer 0

        SET af TO integer 0
        SET ag TO integer 0

        CALL q()

        IF p EQUALS positive infinity THEN
            RETURN -integer 1
        ELSE
            RETURN p
        END IF
    END FUNCTION
END CLASS
