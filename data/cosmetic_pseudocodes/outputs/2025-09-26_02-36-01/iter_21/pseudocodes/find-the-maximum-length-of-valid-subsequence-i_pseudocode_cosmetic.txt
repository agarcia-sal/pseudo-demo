CLASS Solution
    FUNCTION maximumLength(nums)
        SET X1 TO 0
        SET Y2 TO 0
        SET Z3 TO 1
        WHILE (Z3 < LENGTH(nums))
            SET A4 TO (nums[Z3 - 1] + nums[Z3])
            SET B5 TO (A4 % 2)
            IF NOT (B5 = 1)
                SET X1 TO ((X1 + 1) > Y2 ? (X1 + 1) : Y2)
            ELSE
                SET Y2 TO ((Y2 + 1) > X1 ? (Y2 + 1) : X1)
            END IF
            SET Z3 TO Z3 + 1
        END WHILE
        SET C6 TO (X1 > Y2 ? X1 : Y2)
        RETURN (C6 + 1)
    END FUNCTION
END CLASS
