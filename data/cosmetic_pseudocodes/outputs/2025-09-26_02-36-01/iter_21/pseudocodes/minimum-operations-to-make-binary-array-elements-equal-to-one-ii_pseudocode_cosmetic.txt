CLASS Solution
    FUNCTION minOperations(nums)
    SET m TO 0
    SET n TO 0
    SET p TO 0
    SET q TO LENGTH(nums)
    WHILE p < q
        IF NOT ((n MOD 2) = 1)
            SET r TO nums[p]
        ELSE 
            SET r TO 1 - nums[p]
        END IF
        IF r = 0
            SET m TO m + 1
            SET n TO n + 1
        END IF
        SET p TO p + 1
    END WHILE
    RETURN m
    END FUNCTION
END CLASS
