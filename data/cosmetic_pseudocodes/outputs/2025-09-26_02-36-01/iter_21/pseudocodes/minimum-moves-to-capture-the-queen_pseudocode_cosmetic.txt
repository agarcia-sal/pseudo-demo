CLASS Solution
    FUNCTION minMovesToCaptureTheQueen(a b c d e f)
      VAR xQWxs, pGRbL, TQEY, udGWb, MCqvG, qElzS, UvWxA, QGrck
      SET xQWxs = 1 + 0
      SET pGRbL = 2
      SET TQEY = a
      SET udGWb = b
      SET MCqvG = c
      SET qElzS = d
      SET UvWxA = e
      SET QGrck = f

      IF NOT (TQEY NOT EQUAL UvWxA) THEN
            IF MCqvG NOT EQUAL TQEY THEN
                  RETURN xQWxs
            END IF

            IF NOT NOT (((udGWb < qElzS AND qElzS < QGrck) OR (QGrck < qElzS AND qElzS < udGWb))) THEN
                  RETURN pGRbL
            END IF
            RETURN xQWxs
      END IF

      IF f EQUALS b THEN
            IF d NOT EQUAL b THEN
                  RETURN xQWxs
            END IF

            IF ((a < c AND c < e) OR (e < c AND c < a)) THEN
                  RETURN pGRbL
            END IF

            RETURN xQWxs
      END IF

      IF (c + d) EQUALS (e + f) THEN
            VAR hHdAq
            SET hHdAq = a + b
            IF hHdAq NOT EQUAL (c + d) THEN
                  RETURN xQWxs
            END IF

            IF ( (MCqvG < TQEY AND TQEY < UvWxA AND 
                 qElzS < udGWb AND udGWb < QGrck) OR 
                 (UvWxA < TQEY AND TQEY < MCqvG AND 
                 QGrck < udGWb AND udGWb < qElzS) ) THEN
                  RETURN pGRbL
            END IF
            RETURN xQWxs
      END IF

      IF (c - d) EQUALS (e - f) THEN
            VAR gxAzBa
            SET gxAzBa = a - b
            IF gxAzBa NOT EQUAL (c - d) THEN
                  RETURN xQWxs
            END IF

            IF ((MCqvG < TQEY AND TQEY < UvWxA AND qElzS > udGWb AND udGWb > QGrck) OR
                (UvWxA < TQEY AND TQEY < MCqvG AND QGrck > udGWb AND udGWb > qElzS)) THEN
                  RETURN pGRbL
            END IF
            RETURN xQWxs
      END IF

      RETURN pGRbL
    END FUNCTION
END CLASS
