CLASS Solution
	 FUNCTION countCompleteDayPairs WITH parameter bQvcgds
    SET CtUm TO 0
    SET QFsaqwabh BY mapping returning 0 on nonexisting keys
    SET Ljrm TO 0
    LABEL XztvYm
        IF Ljrm GREATER OR EQUAL TO length of bQvcgds THEN
            GOTO UdQHi
        END IF
        SET yZhwrtvn TO bQvcgds AT index Ljrm
        SET rMvX TO yZhwrtvn MINUS ( ( yZhwrtvn MINUS 24 ) DIVIDE 24 ) TIMES 24
        SET Juyvmk TO 24 MINUS rMvX
        SET passo TO Juyvmk MINUS ( ( Juyvmk MINUS 24 ) DIVIDE 24 ) TIMES 24
        SET CtUm TO CtUm PLUS QFsaqwabh AT key passo
        SET QFsaqwabh AT key rMvX TO QFsaqwabh AT key rMvX PLUS 1
        SET Ljrm TO Ljrm PLUS 1
        GOTO XztvYm
    LABEL UdQHi
    RETURN CtUm
END FUNCTION
END CLASS
