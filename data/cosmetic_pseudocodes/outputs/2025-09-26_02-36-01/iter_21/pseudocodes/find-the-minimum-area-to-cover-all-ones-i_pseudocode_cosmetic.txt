CLASS Solution
    FUNCTION minimumArea(grid)
        CALL FUNCTION check_empty(grid)
            SET res TO 0
            RETURN res
        END FUNCTION

        SET a TO 0
        SET b TO 0
        SET c TO +∞
        SET d TO -∞
        SET e TO +∞
        SET f TO -∞

        IF NOT (grid != [] AND grid[0] != [])
            RETURN 0
        END IF

        SET g TO LENGTH(grid)
        SET h TO LENGTH(grid[0])

        CALL FUNCTION loop_i(k)
            IF k ≥ g THEN RETURN 0 END IF

            CALL FUNCTION loop_j(l)
                IF l ≥ h THEN RETURN 0 END IF

                SET tmp1 TO grid[k][l]
                IF tmp1 = 1 THEN
                    IF c > k THEN
                        SET c TO k
                    END IF
                    IF d < k THEN
                        SET d TO k
                    END IF
                    IF e > l THEN
                        SET e TO l
                    END IF
                    IF f < l THEN
                        SET f TO l
                    END IF
                END IF

                CALL loop_j(l + 1)
            END FUNCTION
            CALL loop_j(0)

            CALL loop_i(k + 1)
        END FUNCTION
        CALL loop_i(0)

        SET H TO d − c + (1 × 1)
        SET W TO f − e + (1 + 0)
        RETURN H * W
    END FUNCTION
END CLASS
