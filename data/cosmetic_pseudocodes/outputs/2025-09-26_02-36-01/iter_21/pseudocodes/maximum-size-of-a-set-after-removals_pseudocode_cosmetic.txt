CLASS Solution  
    FUNCTION maximumSetSize(nums1 PARAMETER List OF integers nums2 PARAMETER List OF integers) RETURNS integer  
      SET zxy TO LENGTH OF nums1  
      SET klt TO zxy DIVIDED BY 2  

      FUNCTION vqw (npk PARAMETER List OF integers) RETURNS Set  
        IF LENGTH OF npk EQUALS 0 THEN  
          RETURN EMPTY SET  
        ELSE  
          FUNCTION wmh (ixb PARAMETER List OF integers index PARAMETER integer) RETURNS Set  
            IF index GREATER THAN LENGTH OF ixb MINUS 1 THEN  
              RETURN EMPTY SET  
            ELSE  
              SET zrv TO wmh(ixb, index + 1)  
              IF ixb[index] IN zrv THEN  
                RETURN zrv  
              ELSE  
                RETURN zrv UNION {ixb[index]}  
              END IF  
            END IF  
          END FUNCTION  
          RETURN wmh(npk, 0)  
        END IF  
      END FUNCTION  

      SET swz TO vqw(nums1)  
      SET ytq TO vqw(nums2)  

      FUNCTION xrf (a PARAMETER Set b PARAMETER Set) RETURNS Set  
        IF LENGTH OF a EQUALS 0 OR LENGTH OF b EQUALS 0 THEN RETURN EMPTY SET ELSE  
          SET puc TO a  
          SET omk TO EMPTY SET  
          FUNCTION eos (lst PARAMETER Set iter PARAMETER integer) RETURNS Set  
            IF iter GREATER THAN LENGTH OF lst MINUS 1 THEN RETURN EMPTY SET ELSE  
              SET rjx TO eos(lst, iter + 1)  
              IF lst[iter] IN b THEN  
                SET omk TO omk UNION {lst[iter]}  
              END IF  
              RETURN rjx UNION (IF lst[iter] IN b THEN {lst[iter]} ELSE EMPTY SET)  
            END IF  
          END FUNCTION  
          RETURN eos(CONVERT_SET_TO_LIST(puc), 0)  
        END IF  
      END FUNCTION  

      SET zkp TO xrf(swz, ytq)  

      FUNCTION qrb (full PARAMETER Set part PARAMETER Set) RETURNS Set  
        FUNCTION dsa (fr PARAMETER Set to_find PARAMETER Set) RETURNS Set  
          SET edt TO EMPTY SET  
          FUNCTION gyi (ac PARAMETER Set idx PARAMETER integer) RETURNS Set  
            IF idx GREATER THAN LENGTH OF ac MINUS 1 THEN RETURN EMPTY SET ELSE  
              SET kme TO gyi(ac, idx + 1)  
              IF ac[idx] NOT IN to_find THEN  
                SET edt TO edt UNION {ac[idx]}  
              END IF  
              RETURN kme UNION (IF ac[idx] NOT IN to_find THEN {ac[idx]} ELSE EMPTY SET)  
            END IF  
          END FUNCTION  
          RETURN gyi(CONVERT_SET_TO_LIST(fr), 0)  
        END FUNCTION  
        RETURN dsa(full, part)  
      END FUNCTION  

      SET ofs TO qrb(swz, zkp)  
      SET bin TO qrb(ytq, zkp)  

      FUNCTION minx (a PARAMETER integer b PARAMETER integer) RETURNS integer  
        IF a LESS THAN b THEN RETURN a ELSE RETURN b END IF  
      END FUNCTION  

      FUNCTION maxy (a PARAMETER integer b PARAMETER integer) RETURNS integer  
        IF a GREATER THAN b THEN RETURN a ELSE RETURN b END IF  
      END FUNCTION  

      SET tlf TO minx(klt, LENGTH OF ofs)  
      SET cwe TO minx(klt, LENGTH OF bin)  

      SET wdl TO maxy(0, klt - tlf) + maxy(0, klt - cwe)  

      SET yev TO tlf + cwe + minx(wdl, LENGTH OF zkp)  

      RETURN yev  
    END FUNCTION  
END CLASS
