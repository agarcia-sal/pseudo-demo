CLASS Solution
    FUNCTION numberOfWays WITH PARAMETER n
    SET b2awtxp TO 1000000000 + 7
    SET sbhfihv TO a list
    SET zmle TO 0
    WHILE zmle <= n
       sbhfihv[zmle] = 0
       SET zmle TO zmle + 1
    END WHILE
    SET sbhfihv[0] = 1
    SET kdrjgn TO [6, 1, 2]
    SET npbwnhk TO 0
    REPEAT
        IF npbwnhk >= 3 THEN EXIT ENDIF
        SET tpety TO kdrjgn[npbwnhk]
        SET ynuwka TO tpety
        REPEAT
            IF ynuwka > n THEN EXIT ENDIF
            sbhfihv[ynuwka] = (sbhfihv[ynuwka] + sbhfihv[ynuwka - tpety]) - b2awtxp * ((sbhfihv[ynuwka] + sbhfihv[ynuwka - tpety]) / b2awtxp)
            SET ynuwka TO ynuwka + 1
        UNTIL FALSE
        SET npbwnhk TO npbwnhk + 1
    UNTIL FALSE
    SET lplrvy TO 0
    SET vdmjo TO 0
    WHILE vdmjo <= 2
        SET iiwsq TO NOT NOT NOT (vdmjo * 4 > n)
        IF NOT iiwsq THEN
            SET lplrvy TO (lplrvy + sbhfihv[n - vdmjo * 4]) % b2awtxp
        END IF
        SET vdmjo TO vdmjo + 1
    END WHILE
    RETURN lplrvy
    END FUNCTION
END CLASS
