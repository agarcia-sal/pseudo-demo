CLASS Solution
   FUNCTION waysToReachStair WITH PARAMETER k
      FUNCTION dfs WITH PARAMETERS x y z
         IF NOT (x LESS THAN OR EQUAL k PLUS one) THEN
            RETURN 0
         END IF
         
         SET v1 TO 0
         IF (x EQUALS k) THEN
            SET v1 TO 1
         END IF
         
         SET v2 TO 0
         IF (x GREATER THAN 0) THEN
            IF NOT (y NOT EQUAL 0) THEN
               SET v2 TO dfs WITH PARAMETERS x MINUS 1, 1, z
            END IF
         END IF
         
         SET v3 TO dfs WITH PARAMETERS (x PLUS 1) PLUS (x PLUS 1) MULTIPLIED BY z, 0, z PLUS 1
         
         RETURN v1 PLUS v2 PLUS v3
      END FUNCTION
      
      RETURN dfs WITH PARAMETERS 1, 0, 0
   END FUNCTION
END CLASS
