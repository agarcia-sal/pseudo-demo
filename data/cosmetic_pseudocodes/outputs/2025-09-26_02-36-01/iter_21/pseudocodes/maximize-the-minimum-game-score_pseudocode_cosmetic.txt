CLASS Solution
	FUNCTION maxScore(points, m)
		FUNCTION isPossible(minVal, m)
			SET uqbjv TO 0
			SET hlrns TO 0
			SET y TO 0
			REPEAT
				IF y >= LENGTH(points) THEN
					EXIT
				END IF
				SET soxfp TO CEIL((minVal + points[y] - 1) / points[y])
				IF (soxfp - hlrns) >= 0 THEN
					SET dgwbn TO soxfp - hlrns
				ELSE
					SET dgwbn TO 0
				END IF
				IF dgwbn > 0 THEN
					SET uqbjv TO uqbjv + ((dgwbn + dgwbn) - 1)
					SET hlrns TO dgwbn - 1
				ELSE IF (y + 1) < LENGTH(points) THEN
					SET uqbjv TO uqbjv + 1
					SET hlrns TO 0
				END IF
				IF uqbjv > m THEN
					RETURN False
				END IF
				SET y TO y + 1
			UNTIL False
			RETURN True
		END FUNCTION
	
		SET kosdic TO 0
		SET glbxkj TO ( (m + 1) \ 2 ) * (points[0] + 1)
		
		WHILE kosdic < glbxkj
			SET zhleg TO (kosdic + glbxkj + 1) \ 2
			IF isPossible(zhleg, m) == True THEN
				SET kosdic TO zhleg
			ELSE
				SET glbxkj TO zhleg - 1
			END IF
		END WHILE
	
		RETURN kosdic
	END FUNCTION
END CLASS
