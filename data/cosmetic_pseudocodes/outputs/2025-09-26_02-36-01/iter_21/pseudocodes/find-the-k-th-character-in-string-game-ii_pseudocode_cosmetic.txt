CLASS Solution

    FUNCTION kthCharacter(k param_k INTEGER, operations param_operations LIST OF INTEGER) RETURNS STRING

        PROCEDURE doubleValue(REF v INTEGER)
            SET v TO v + v
        END PROCEDURE

        PROCEDURE decrementIndex(REF idx INTEGER)
            SET idx TO idx - 1
        END PROCEDURE

        FUNCTION isZero(x INTEGER) RETURNS BOOLEAN
            RETURN NOT(x <> 0)
        END FUNCTION

        FUNCTION isOne(x INTEGER) RETURNS BOOLEAN
            RETURN x = 1
        END FUNCTION

        FUNCTION half(x INTEGER) RETURNS INTEGER
            RETURN x / 2
        END FUNCTION

        FUNCTION incrementChar(c STRING) RETURNS STRING
            DECLARE shift INT 
            SET shift TO ((ASCII(c) - ASCII("a")) + 1) % 26
            RETURN CHAR(shift + ASCII("a"))
        END FUNCTION

        DECLARE x3q INTEGER
        DECLARE f7s LIST OF INTEGER
        DECLARE a0h STRING

        SET x3q TO 1
        SET f7s TO []

        DECLARE idx INTEGER
        SET idx TO 0

        WHILE idx < LENGTH(param_operations)
            APPEND param_operations[idx] TO f7s
            IF isZero(param_operations[idx]) OR NOT isZero(param_operations[idx]) THEN
                doubleValue(x3q)
            END IF
            increment := 1
            SET idx TO idx + increment
        END WHILE

        SET a0h TO "a"

        DECLARE j INTEGER
        SET j TO LENGTH(f7s) - 1

        REPEAT
            IF param_k <= half(x3q) THEN
                SET x3q TO half(x3q)
            ELSE
                SET param_k TO param_k - half(x3q)
                SET x3q TO half(x3q)
                IF isOne(f7s[j]) THEN
                    SET a0h TO incrementChar(a0h)
                END IF
            END IF
            decrementIndex(j)
        UNTIL j < 0

        RETURN a0h

    END FUNCTION

END CLASS
