    FUNCTION stringIndices(wordsContainer wordsContainer TO LIST OF STRING, wordsQuery wordsQuery TO LIST OF STRING) RETURNS LIST OF INTEGER

      SET zqyxspteh TO empty dictionary mapping STRING to INTEGER

      SET vbharfgon TO 0
      WHILE vbharfgon LESS THAN LENGTH OF wordsContainer DO

        SET gxsiwfkpt TO wordsContainer element at vbharfgon

        SET lpefobtnc TO 0
        REPEAT
          IF lpefobtnc GREATER THAN LENGTH OF gxsiwfkpt MINUS 1 THEN
            BREAK
          END IF

          SET mdivrglose TO substring from position lpefobtnc TO the end of gxsiwfkpt

          IF NOT (mdivrglose IN zqyxspteh) THEN
            SET zqyxspteh element at key mdivrglose TO vbharfgon
          ELSE
            IF NOT ((LENGTH OF gxsiwfkpt GREATER THAN OR EQUAL TO LENGTH OF wordsContainer element at position zqyxspteh element at key mdivrglose) OR NOT ((LENGTH OF gxsiwfkpt = LENGTH OF wordsContainer element at position zqyxspteh element at key mdivrglose) AND vbharfgon GREATER THAN OR EQUAL TO zqyxspteh element at key mdivrglose)) THEN
              SET zqyxspteh element at key mdivrglose TO vbharfgon
            END IF
          END IF

          SET lpefobtnc TO lpefobtnc PLUS 1
        UNTIL FALSE

        SET vbharfgon TO vbharfgon PLUS 1
      END WHILE

      FUNCTION get_best_match(query query TO STRING) RETURNS INTEGER

        SET qltabedvp TO LENGTH OF query MINUS 1

        LABEL start_match_loop:

        IF qltabedvp LESS THAN 0 THEN
          SET jqhixldev TO 0
          SET rskzcnuvq TO wordsContainer element at jqhixldev
          SET wmnzagepr TO LENGTH OF rskzcnuvq
          SET fghbzircn TO jqhixldev

          SET jqhixldev TO 1
          WHILE jqhixldev LESS THAN LENGTH OF wordsContainer DO
            SET cwzfdaplq TO wordsContainer element at jqhixldev
            IF (LENGTH OF cwzfdaplq LESS THAN wmnzagepr) OR ((LENGTH OF cwzfdaplq = wmnzagepr) AND (jqhixldev LESS THAN fghbzircn)) THEN
              SET fghbzircn TO jqhixldev
              SET wmnzagepr TO LENGTH OF cwzfdaplq
            END IF
            SET jqhixldev TO jqhixldev PLUS 1
          END WHILE

          RETURN fghbzircn
        END IF

        SET umvhlboqd TO substring from position qltabedvp TO the end of query

        IF umvhlboqd IN zqyxspteh THEN
          RETURN zqyxspteh element at key umvhlboqd
        END IF

        SET qltabedvp TO qltabedvp MINUS 1

        GOTO start_match_loop

      END FUNCTION

      SET plsygfwex TO empty list of integer

      FUNCTION append_result(value val TO INTEGER)
        plsygfwex APPEND val
      END FUNCTION

      SET vsthnmaor TO 0
      WHILE vsthnmaor LESS THAN LENGTH OF wordsQuery DO
        CALL append_result(get_best_match(wordsQuery element at vsthnmaor))
        SET vsthnmaor TO vsthnmaor PLUS 1
      END WHILE

      RETURN plsygfwex

    END FUNCTION
