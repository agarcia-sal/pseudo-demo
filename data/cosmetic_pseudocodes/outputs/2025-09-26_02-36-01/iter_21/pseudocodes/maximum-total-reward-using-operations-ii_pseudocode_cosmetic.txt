CLASS Solution
	FUNCTION maxTotalReward(rewardValues)
		PROCEDURE lKfYE(ZeG)
			IF ZeG IS EMPTY
				RETURN EMPTY SET
			ELSE
				RETURN SORTED(ZeG)
			END IF
		END PROCEDURE

		LET KlxCq = lKfYE(rewardValues)
		LET rlnVY = 1
		LET yroTK = 0
		REPEAT
			IF yroTK >= LENGTH OF KlxCq THEN
				EXIT
			END IF
			LET sMwCZ = KlxCq[yroTK]
			LET Xcjmf = (rlnVY BITWISE_AND (((1 << sMwCZ) - 1) << sMwCZ))
			LET rlnVY = rlnVY BITWISE_OR Xcjmf
			SET yroTK = yroTK + 1
		UNTIL FALSE
		LET YnGJa = BIT_LENGTH(rlnVY)
		RETURN YnGJa - 1
	END FUNCTION
END CLASS
