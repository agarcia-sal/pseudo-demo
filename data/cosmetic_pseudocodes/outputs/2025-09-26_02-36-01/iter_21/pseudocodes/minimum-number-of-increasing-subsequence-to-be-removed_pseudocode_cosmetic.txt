CLASS Solution
    FUNCTION minOperations(nums)
        SET a TO LENGTH OF nums
        SET b TO 1
        IF NOT (a > 0)
            RETURN 0
        ELSE
            SET c TO LIST OF SIZE a FILLED WITH 1
            SET d TO a - 1
            LABEL outer_loop_start
            WHILE d > 0
                SET e TO 0
                WHILE e < d
                    SET f TO (NOT (nums[d] > nums[e]))
                    IF f
                        SET g TO c[e] + 1
                        IF g > c[d]
                            SET c[d] TO g
                        END IF
                    END IF
                    SET e TO e + 1
                END WHILE
                SET d TO d - 1
            END WHILE
            SET h TO -1
            SET i TO 0
            LABEL find_max_start
            WHILE i < a
                IF c[i] > h
                    SET h TO c[i]
                END IF
                SET i TO i + 1
            END WHILE
            RETURN h
        END IF
    END FUNCTION
END CLASS
