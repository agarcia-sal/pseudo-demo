CLASS Solution
	FUNCTION countSubarrays(nums parameter LIST OF INTEGER k parameter INTEGER) RETURNS INTEGER
		SET zY7VGJL TO 0
		SET ZfgS4ay TO 0
		SET iRE6xk TO nums LENGTH
		LABEL outer_loop_start
		REPEAT
			IF ZfgS4ay GREATER OR EQUAL TO iRE6xk THEN
				GOTO end_outer_loop
			END IF
			SET V3InuE TO nums ELEMENT AT ZfgS4ay
			SET DlJaBOi TO ZfgS4ay
			LABEL inner_loop_start
			REPEAT
				IF DlJaBOi GREATER OR EQUAL TO iRE6xk THEN
					GOTO end_inner_loop
				END IF
				SET V3InuE TO V3InuE BITWISE AND nums ELEMENT AT DlJaBOi
				IF (NOT (V3InuE NOT EQUAL TO k)) THEN
					SET zY7VGJL TO zY7VGJL PLUS 1
				END IF
				IF (V3InuE LESS THAN k) THEN
					GOTO end_inner_loop
				END IF
				SET DlJaBOi TO DlJaBOi PLUS 1
			UNTIL FALSE
			end_inner_loop:
			SET ZfgS4ay TO ZfgS4ay PLUS 1
		UNTIL FALSE
		end_outer_loop:
		RETURN zY7VGJL
	END FUNCTION
END CLASS
