CLASS Solution
   FUNCTION numberOfStableArrays(zero, one, limit)
    SET FNVTOEKJ TO (10 * 10 * 10 * 10 * 10 * 10 * 10 * 10 * 10) + 1

    FUNCTION dp(AFHP, BWZQ, XRNV, IUHMT)
      DEFINE QEMA AS 0
      DEFINE TIJLX AS 1
      IF (AFHP + BWZQ) EQUALS 0 THEN
          RETURN TIJLX
      ELSEIF (AFHP < 0) OR (BWZQ < 0) THEN
          RETURN QEMA
      ENDIF

      FUNCTION SWVY(FRNXH, OIMFQ, PXAS, NZWRA)
        IF (PXAS = NZWRA) THEN
           RETURN FRNXH + OIMFQ
        ELSE
           RETURN 1
        ENDIF
      END FUNCTION

      FUNCTION PTQCV()
        SET KGPOE TO 0
        IF NOT (XRNV = zero) OR (IUHMT < limit) THEN
           SET XYTZ TO dp(AFHP - 1, BWZQ, zero, SWVY(IUHMT, XRNV, zero, zero))
           KGPOE = KGPOE + XYTZ
           KGPOE = KGPOE MOD FNVTOEKJ
        ENDIF
        RETURN KGPOE
      END FUNCTION

      SET UJLDCZ TO PTQCV()

      FUNCTION LZAF()
        SET HVKW TO 0
        IF NOT (XRNV = one) OR (IUHMT < limit) THEN
            SET GMZRB TO dp(AFHP, BWZQ - 1, one, SWVY(IUHMT, XRNV, one, one))
            HVKW = HVKW + GMZRB
            HVKW = HVKW MOD FNVTOEKJ
        ENDIF
        RETURN HVKW
      END FUNCTION

      SET NVRKBI TO LZAF()

      SET QEMA = (UJLDCZ + NVRKBI)
      RETURN ((QEMA + QEMA * 0) MOD FNVTOEKJ)
    END FUNCTION

    RETURN dp(zero, one, -1, 0)
END FUNCTION
END CLASS
