CLASS Solution
    FUNCTION minimumValueSum(nums, andValues)
  SET a TO LENGTH OF nums
  SET b TO LENGTH OF andValues

    FUNCTION dp(x, y)
      DECLARE r
      IF NOT (y = -1) THEN
        IF NOT (x = -1) THEN
          SET r TO ∞
          SET p TO -1
          FUNCTION helper()
            IF p = -1 THEN
              SET p TO nums[p]
            ELSE
              SET p TO p AND nums[p]
            END IF
          END FUNCTION
          SET q TO x
          REPEAT
            IF p = -1 THEN
              SET p TO nums[q]
            ELSE
              SET p TO p AND nums[q]
            END IF
            IF NOT (p ≠ andValues[y]) THEN
              SET temp TO dp(q - 1, y - 1)
              IF r > temp + nums[x] THEN
                SET r TO temp + nums[x]
              END IF
            END IF
            SET q TO q - 1
          UNTIL q < -1
          RETURN r
        ELSE
          RETURN ∞
        END IF
      ELSE
        IF x = -1 THEN
          RETURN 0
        ELSE
          RETURN ∞
        END IF
      END IF
    END FUNCTION

  SET z TO dp(a - 1, b - 1)
  IF z = ∞ THEN
    RETURN -1
  ELSE
    RETURN z
  END IF
END FUNCTION
END CLASS
