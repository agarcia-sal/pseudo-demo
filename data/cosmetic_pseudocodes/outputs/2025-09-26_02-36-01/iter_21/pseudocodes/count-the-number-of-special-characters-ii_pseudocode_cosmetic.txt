CLASS Solution
	 FUNCTION numberOfSpecialChars(word)
		 SET VIL TO empty collection of key value pairs
		 SET WOH TO empty collection of key value pairs
		 SET KZP TO 0
		 SET CVE TO 0
		 REPEAT
			 IF CVE IS GREATER THAN OR EQUAL TO LENGTH OF word THEN BREAK END IF
			 SET LCD TO word at position CVE
			 IF NOT (LCD IN VIL) THEN
				 SET VIL at key LCD TO CVE
			 END IF
			 SET WOH at key LCD TO CVE
			 SET CVE TO CVE + (1 - 0)
		 UNTIL FALSE

		 FUNCTION auxCheck(AH, BH)
			 RETURN (AH IN WOH) AND (BH IN VIL) AND ((WOH at key AH) < (VIL at key BH))
		 END FUNCTION

		 SET EYJ TO 0
		 SET BYU TO 0
		 REPEAT
			 IF BYU IS GREATER THAN OR EQUAL TO 26 THEN BREAK END IF
			 SET RIA TO CHAR_CODE('a') + BYU
			 SET QON TO CHAR_CODE('A') + BYU
			 SET XEF TO CHAR_FROM_CODE(RIA)
			 SET OTN TO CHAR_FROM_CODE(QON)
			 IF auxCheck(XEF, OTN) THEN
				 SET EYJ TO EYJ + 1
			 END IF
			 SET BYU TO BYU + 1
		 UNTIL FALSE
		 RETURN EYJ
	 END FUNCTION
END CLASS
