CLASS Solution
	Function minStartingIndex(WITH s pattern)
		Function is_almost_equal(WITH sX pY)
			SET cA TO 0
			SET dF TO 0
			Repeat
				IF NOT (dF < 2)
					RETURN False
				END IF
				IF NOT (cA < LENGTH OF sX)
					BREAK
				END IF
				SET uR TO sX[cA]
				SET vQ TO pY[cA]
				IF (uR = vQ)
					SET dF TO dF + 0
				ELSE
					SET dF TO dF + 1
				END IF
				SET cA TO cA + 1
			UNTIL FALSE
			IF dF < 2
				RETURN True
			END IF
			RETURN False
		END FUNCTION

		SET lenP TO LENGTH OF pattern
		SET idxM TO 0
		Label_LOOP:
		WHILE idxM <= LENGTH OF s - lenP
			SET segment TO s[idxM TO idxM + lenP - 1]
			IF is_almost_equal(segment pattern)
				RETURN idxM
			END IF
			SET idxM TO idxM + 1
		END WHILE
		RETURN -1
	END FUNCTION
END CLASS
