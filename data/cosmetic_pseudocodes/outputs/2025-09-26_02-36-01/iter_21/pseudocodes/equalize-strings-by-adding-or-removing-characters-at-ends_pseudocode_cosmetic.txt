CLASS Solution  
    FUNCTION minOperations(initial parameter TO string, target parameter TO string) RETURNS integer  
       SET p  TO 0  
       SET r  TO 0  
       SET x  TO LENGTH OF initial  
       SET y  TO LENGTH OF target  
       
       PROCEDURE h(a,b)  
          RETURN a EQUALS b  
       END PROCEDURE  
       
       SET w TO a list of size (x PLUS one) EACH containing a list of size (y PLUS one) FILLED WITH 0  
       
       PROCEDURE f(u,v,z)  
          IF z GREATER THAN x THEN  
             RETURN  
          ELSE IF v GREATER THAN y THEN  
             f(u PLUS 1,1,z)  
          ELSE  
             IF h(initial AT u MINUS 1,target AT v MINUS 1) THEN  
                w AT u AT v SET TO (w AT (u MINUS 1) AT (v MINUS 1)) PLUS 1  
                IF p LESS THAN w AT u AT v THEN  
                   p SET TO w AT u AT v  
                END IF  
             END IF  
             f(u,v PLUS 1,z)  
          END IF  
       END PROCEDURE  
       
       f(1,1,x)  
       
       SET q TO (x PLUS y) MINUS (2 MULTIPLIED BY p)  
       RETURN q  
    END FUNCTION  
END CLASS
