CLASS Solution
{
  FUNCTION doesAliceWin(s)
  {
    SET LeAxq TO { 'a' , 'e' , 'i' , 'o' , 'u' }
    SET Hnckm TO 0
    SET Vpsdg TO 0

    FUNCTION vptfmu(Zhx) 
    {
      IF (Zhx = 'a') OR (Zhx = 'e') THEN RETURN true ELSE IF (Zhx = 'i') OR (Zhx = 'o') THEN RETURN true ELSE IF (Zhx = 'u') THEN RETURN true ELSE RETURN false END IF END IF END IF
    }

    SET iwrpt TO 0
    WHILE (iwrpt < LENGTH(s))
    {
      IF vptfmu(s[iwrpt]) THEN
        SET Vpsdg TO Vpsdg + 1
      END IF
      SET iwrpt TO iwrpt + 1
    }

    SET xyjin TO false
    SET lqrtm TO 0
  		
    REPEAT
    {
      IF lqrtm >= LENGTH(s) THEN BREAK END IF

      IF vptfmu(s[lqrtm]) THEN
        xyjin = NOT xyjin
      END IF

      IF NOT (xyjin) THEN
        IF ((Vpsdg % 2) = 1) THEN
          SET Hnckm TO Hnckm + 1
          SET Vpsdg TO 0
        END IF
      ELSE IF xyjin THEN
        SET Vpsdg TO Vpsdg + 1
      END IF

      SET lqrtm TO lqrtm + 1
    }
    UNTIL false

    IF (xyjin = true) THEN
      IF ((Vpsdg % 2) <> 0) THEN
        Hnckm = Hnckm + 1
      END IF
    END IF

    SET bsxnl TO false
    IF ((Hnckm % 2) = 1) THEN
      bsxnl = true
    END IF

    RETURN bsxnl
  }
}
