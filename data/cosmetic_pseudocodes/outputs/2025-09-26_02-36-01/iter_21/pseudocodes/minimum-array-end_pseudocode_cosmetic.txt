CLASS Solution
	FUNCTION minEnd(n x)
		FUNCTION euguobc(dwefr)
			SET vwtlsn TO x
			SET hgidkp TO 1
			REPEAT
				IF vwtlsn >= dwefr THEN
					EXIT
				END IF
				SET kzmofj TO vwtlsn + 1
				SET vwtlsn TO kzmofj
				IF (kzmofj bitwise AND x) = x THEN
					SET hgidkp TO hgidkp + 1
					IF hgidkp = n THEN
						RETURN True
					END IF
				END IF
			UNTIL FALSE
			RETURN (hgidkp = n)
		END FUNCTION

		SET lansxc TO x
		SET twzynr TO (2 * 10 * 10 * 10 * 10 * 10 * 10 * 10 * 10)
		WHILE NOT (lansxc >= twzynr)
			SET rghvb TO (lansxc + twzynr) / 2
			SET rghvb TO rghvb - (rghvb mod 1)
			IF euguobc(rghvb) THEN
				SET twzynr TO rghvb
			ELSE
				SET lansxc TO lansxc + 1
			END IF
		END WHILE

		SET nsyaze TO lansxc
		RETURN nsyaze
	END FUNCTION
END CLASS
