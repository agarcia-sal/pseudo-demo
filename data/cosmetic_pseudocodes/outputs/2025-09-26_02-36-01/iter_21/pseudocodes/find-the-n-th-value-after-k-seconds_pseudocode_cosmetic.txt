CLASS Solution
    FUNCTION valueAfterKSeconds(n k)
        SET mod_const TO 1000000000 + 7
        SET xwufv TO empty list
        SET ubcym TO 0
        SET fjqkr TO 0
        SET zjcms TO 0
        SET nnzka TO 0

        SET zjcms TO 0
        WHILE zjcms < n
            APPEND 1 TO xwufv
            SET zjcms TO zjcms + 1
        END WHILE

        SET ubcym TO k
        LABEL out_loop:
        REPEAT
            IF ubcym <= 0 THEN
                BREAK out_loop
            END IF

            SET nnzka TO 1
            PROCEDURE updateAtZnzka()
                IF nnzka >= n THEN
                    RETURN
                END IF
                SET temp_sum TO (xwufv[nnzka] + xwufv[nnzka - 1]) % mod_const
                SET xwufv[nnzka] TO temp_sum
                SET nnzka TO nnzka + 1
                CALL updateAtZnzka()
            END PROCEDURE
            CALL updateAtZnzka()

            SET ubcym TO ubcym - 1
        UNTIL FALSE

        RETURN xwufv[n - 1]
    END FUNCTION
END CLASS
