CLASS Solution  
    FUNCTION findKthSmallest(coins k)  
        
        FUNCTION helper_count(x)  
            SET alpha TO zero  
            SET beta TO one  
            SET gamma TO zero  
            SET delta TO LENGTH OF coins  
            FUNCTION gcd_helper(a b)  
                IF b EQUALS zero THEN RETURN a END IF  
                RETURN gcd_helper(b a MODULO b)  
            END FUNCTION  
            SET epsilon TO (2 RAISED TO delta) MINUS one  
            SET zeta TO zero  
            REPEAT  
                SET bf TO one  
                SET ct TO zero  
                SET eta TO zero  
                WHILE eta LESS THAN delta  
                    IF (zeta BITWISE AND (1 LEFT SHIFTED BY eta)) NOT EQUAL TO zero THEN  
                        SET bf TO (bf MULTIPLIED BY coins AT eta) DIVIDED BY gcd_helper(bf coins AT eta)  
                        SET ct TO ct PLUS one  
                    END IF  
                    SET eta TO eta PLUS one  
                END WHILE  
                IF (ct MODULO two) EQUALS one THEN  
                    SET alpha TO alpha PLUS (x DIVIDED BY bf)  
                ELSE  
                    SET alpha TO alpha MINUS (x DIVIDED BY bf)  
                END IF  
                SET zeta TO zeta PLUS one  
            UNTIL zeta GREATER THAN epsilon  
            RETURN alpha  
        END FUNCTION  
        
        SET omicron TO one  
        SET pi TO k MULTIPLIED BY MIN(coins)  
        
        FUNCTION mid_calc(a b)  
            RETURN (a PLUS b) DIVIDED BY 2  
        END FUNCTION  
        
        LOOP  
            IF NOT (omicron LESS THAN pi) THEN EXIT END IF  
            SET sigma TO mid_calc(omicron pi)  
            IF helper_count(sigma) LESS THAN k THEN  
                SET omicron TO sigma PLUS one  
            ELSE  
                SET pi TO sigma  
            END IF  
        END LOOP  
        RETURN omicron  
    END FUNCTION  
END CLASS
