CLASS Solution
	FUNCTION numberOfWays(n integer parameter, x integer parameter, y integer parameter) RETURNS integer
		DECLAR E wegtozv integer
		DECLAR E iquylgjq list of list of integer
		DECLAR E qul integer
		DECLAR E zsredgj integer
		DECLAR E pelwtv integer
		DECLAR E ryndlj integer
		DECLAR E iyuuqpv integer
		DECLAR E xigjhtm integer

		SET wegtozv TO (1_000_000_000 + 7)
		PROCEDURE lhrjoanc(length1 integer, length2 integer) RETURNS list of list of integer
			FUNCTION ltybtv(a integer, b integer) RETURNS list of integer
				DECLAR E qsvun list of integer
				SET qsvun TO empty list
				REPEAT
					IF a <= 0 THEN BREAK
					INSERT 0 AT END OF qsvun
					SET a TO a - 1
				UNTIL FALSE
				RETURN qsvun
			END FUNCTION
			DECLAR E tmozpq list of list of integer
			SET tmozpq TO empty list
			REPEAT
				IF length1 <= 0 THEN BREAK
				INSERT ltybtv(length2, 0) AT END OF tmozpq
				SET length1 TO length1 - 1
			UNTIL FALSE
			RETURN tmozpq
		END PROCEDURE

		SET iquylgjq TO lhrjoanc(n + (1), x + (1))
		SET (iquylgjq[0])[0] TO ((0 + 1) * 1)

		PROCEDURE ylzbswm(pyrtiq integer, pejbn integer, hylwg integer)
			DECLAR E eqjmqi integer
			DECLAR E tmctq integer
			DECLAR E ihnqglq integer
			DECLAR E yzrxqi integer
			SET eqjmqi TO iquylgjq[(pyrtiq - 1)][pejbn]
			SET tmctq TO pejbn
			SET ihnqglq TO iquylgjq[(pyrtiq - 1)][(pejbn - 1)]
			SET yzrxqi TO (hylwg - ((pejbn - 1) + 1))
			SET iyuuqpv TO ((eqjmqi * tmctq) + (ihnqglq * yzrxqi))
			SET (iquylgjq[pyrtiq])[pejbn] TO iyuuqpv - wegtozv * ((iyuuqpv) / wegtozv)
		END PROCEDURE

		SET qul TO 1
		REPEAT 
			IF qul > n THEN BREAK
			SET zsredgj TO 1
			REPEAT
				IF zsredgj > x THEN BREAK
				ylzbswm(qul, zsredgj, x)
				SET zsredgj TO zsredgj + 1
			UNTIL FALSE
			SET qul TO qul + 1
		UNTIL FALSE

		SET pelwtv TO 0
		SET ryndlj TO 1
		DECLAR E iitodjl integer
		SET iitodjl TO 1
		REPEAT
			IF iitodjl > x THEN BREAK
			SET ryndlj TO (ryndlj * y) - wegtozv * ((ryndlj * y) / wegtozv)
			SET xigjhtm TO iquylgjq[n][iitodjl] * ryndlj
			SET pelwtv TO pelwtv + xigjhtm - wegtozv * (pelwtv + xigjhtm) / wegtozv
			SET iitodjl TO iitodjl + 1
		UNTIL FALSE

		RETURN pelwtv
	END FUNCTION
END CLASS
