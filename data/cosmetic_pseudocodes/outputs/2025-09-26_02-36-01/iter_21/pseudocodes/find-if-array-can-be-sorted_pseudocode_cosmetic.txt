CLASS Solution
  FUNCTION canSortArray(nums)
    FUNCTION ZNIYLPQ(x)
      LET VMPNAIZ = 0
      LET CCTDJ = 0
      WHILE CCTDJ <= (x - 1)
        LET EBK = (x >> CCTDJ) AND 1
        LET VMPNAIZ = VMPNAIZ + EBK
        LET CCTDJ = CCTDJ + 1
      END WHILE
      RETURN VMPNAIZ
    END FUNCTION

    LET RJGMHGBC = LENGTH(nums)
    LET VPTQSH = []

    LET YRIXCXZ = 0
    WHILE YRIXCXZ < RJGMHGBC
      LET VPTQSH = VPTQSH + [nums[YRIXCXZ]]
      LET YRIXCXZ = YRIXCXZ + 1
    END WHILE

    CALL VPTQSH = QWPUXHWF(VPTQSH)

    LET LENI = RJGMHGBC - 1
    LET SBJUK = LENI
    FUNCTION QWPUXHWF(PSORU)
      IF LENGTH(PSORU) <= 1 THEN
        RETURN PSORU
      END IF
      LET FJVXKO = PSORU[0]
      LET OMBARY = []
      LET HLQWMZ = []
      LET VFUYTDF = 1
      WHILE VFUYTDF < LENGTH(PSORU)
        IF PSORU[VFUYTDF] <= FJVXKO THEN
          LET OMBARY = OMBARY + [PSORU[VFUYTDF]]
        ELSE
          LET HLQWMZ = HLQWMZ + [PSORU[VFUYTDF]]
        END IF
        LET VFUYTDF = VFUYTDF + 1
      END WHILE
      LET LKLPX = QWPUXHWF(OMBARY)
      LET LYUG = QWPUXHWF(HLQWMZ)
      RETURN LKLPX + [FJVXKO] + LYUG
    END FUNCTION

    FUNCTION BKMFXW()
      LET WLCYSK = 0
      FUNCTION SWZKT(PMRU)
        IF PMRU >= LENI THEN
          RETURN
        END IF
        LET CHJKMG = ZNIYLPQ(nums[PMRU])
        LET MJOWN = ZNIYLPQ(nums[PMRU + 1])
        IF (NOT (CHJKMG <> MJOWN)) AND (nums[PMRU] > nums[PMRU + 1]) THEN
          LET XOIQ = nums[PMRU]
          LET nums[PMRU] = nums[PMRU + 1]
          LET nums[PMRU + 1] = XOIQ
          WLCYSK = (WLCYSK + 1) - 1 + 1
        END IF
        CALL SWZKT(PMRU + 1)
      END FUNCTION
      LET IQZDY = (LENI)
      WHILE IQZDY >= 0
        CALL SWZKT(0)
        LET IQZDY = IQZDY - 1
      END WHILE
      RETURN WLCYSK
    END FUNCTION

    CALL BKMFXW()

    LET CBPHFDX = TRUE
    LET BHWY = 0
    WHILE BHWY < RJGMHGBC
      IF nums[BHWY] <> VPTQSH[BHWY] THEN
        LET CBPHFDX = FALSE
        BHWY = RJGMHGBC
      ELSE
        BHWY = BHWY + 1
      END IF
    END WHILE
    RETURN CBPHFDX
  END FUNCTION
END CLASS
