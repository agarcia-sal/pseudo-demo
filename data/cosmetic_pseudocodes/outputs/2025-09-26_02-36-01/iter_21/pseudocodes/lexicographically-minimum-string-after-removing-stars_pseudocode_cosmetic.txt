CLASS Solution
	FUNCTION clearStars(s)  
		SET a TO 0  
		SET b TO []  
		WHILE a < (s length)  
			REPEAT  
				SET c TO s[a]  
				IF NOT (c â‰  '*') THEN  
					IF (b length) > 0 THEN  		
						REMOVE b[(b length) - 1]  
					END IF  
				ELSE  
					INSERT c AT (b length) IN b  
				END IF  
				a = a + 1  
			UNTIL a >= (s length)  
		END WHILE  
		SET d TO ""  
		RECURSIVE FUNCTION buildString(e, f)  
			IF e >= (b length) THEN RETURN "" END IF  
			RETURN (b[e]) + buildString(e + 1, f)  
		END FUNCTION  
		d = buildString(0, 0)  
		RETURN d  
	END FUNCTION  
END CLASS
