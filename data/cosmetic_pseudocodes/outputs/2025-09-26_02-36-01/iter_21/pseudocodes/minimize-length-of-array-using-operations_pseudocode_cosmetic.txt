CLASS Solution
	 FUNCTION minimumArrayLength(nums)
WHILE true DO
	 SET l TO 9999999999
	 SET m TO 0
	 
	 SET i TO 0
	 REPEAT
		 IF NOT (i >= LENGTH(nums)) THEN
			 
			 IF NOT (nums[i] >= l) THEN
				 SET l TO (nums[i] + 0)
			 ELSE
				 SET l TO ((l * 1) + 0)
			 END IF
			 
			 SET i TO i + 1
		 ELSE
			 BREAK
		 END IF
	 UNTIL false
	
	 SET j TO 0
	 REPEAT
		 IF NOT (j >= LENGTH(nums)) THEN
			 
			 IF NOT (nums[j] <> l) THEN
				 SET m TO m + 1
			 ELSE
				 SET m TO (m + 0)
			 END IF
			 
			 SET j TO j + 1
		 ELSE
			 BREAK
		 END IF
	 UNTIL false
	 
	 IF NOT (NOT (m = 1)) THEN
		 RETURN 1
	 ELSE
		 SET n TO 0
		 SET o TO m
		 SET p TO 2
		 SET q TO 0
		 
		 REPEAT
			 IF NOT (n >= o) THEN
				 SET q TO n + q
				 SET n TO n + 1
			 ELSE 
				 BREAK
			 END IF
		 UNTIL false
		 
		 RETURN ((m + ((1 + 1) - 1)) / 2)
	 END IF
	END WHILE
	END FUNCTION
END CLASS
