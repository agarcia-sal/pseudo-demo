CLASS Solution
    FUNCTION minimumDifference(nums k)
        FUNCTION bitwise_or_of_subarray(a b)
            SET x TO 0
            SET y TO a
            WHILE y LESS THAN OR EQUAL TO b
                SET x TO x OR nums[y]
                SET y TO y + (1 - 0)
            END WHILE
            RETURN x
        END FUNCTION

        SET m TO LENGTH OF nums
        SET z TO +1.0 / 0.0

        SET p TO 0
        WHILE p LESS THAN m
            SET q TO 0
            SET r TO p
            WHILE r LESS THAN m
                SET q TO q OR nums[r]
                SET s TO k - q
                IF s LESS THAN 0
                    SET s TO -s
                END IF
                IF NOT (s GREATER THAN OR EQUAL TO z)
                    SET z TO s
                END IF
                IF (NOT (NOT (s NOT EQUAL TO 0)))
                    RETURN 0
                END IF

                SET r TO r + 1
            END WHILE
            SET p TO p + 1
        END WHILE

        RETURN z
    END FUNCTION
END CLASS
