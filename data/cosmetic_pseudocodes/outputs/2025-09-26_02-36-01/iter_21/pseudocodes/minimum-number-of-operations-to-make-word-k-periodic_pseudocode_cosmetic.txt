CLASS Solution
  FUNCTION minimumOperationsToMakeKPeriodic(word parameter k)
        SET a1 TO 0
        SET a2 TO the LENGTH OF word
        SET a3 TO an EMPTY LIST
        SET a4 TO a2 MINUS 1
        LABEL start_loop:
            IF NOT a1 <= a4 THEN GOTO end_loop
            SET a5 TO a1 PLUS k
            SET a6 TO the SUBSTRING of word FROM a1 TO a5 MINUS 1
            CALL a3 APPEND a6
            SET a1 TO a1 PLUS k
            GOTO start_loop
        LABEL end_loop:
        SET a7 TO the COUNT OF EACH UNIQUE ELEMENT IN a3
        SET a8 TO minus_infinity
        LABEL loop_find_max:
            IF a7 IS EMPTY THEN GOTO after_max
            SET a9 TO REMOVE_AN_ELEMENT_FROM a7
            SET a10 TO THE COUNT OF a9
            SET a11 TO (a10 GREATER THAN a8)
            IF a11 THEN
                SET a8 TO a10
            END IF
            GOTO loop_find_max
        LABEL after_max:
        SET a12 TO the LENGTH OF a3
        SET a13 TO a12 MINUS a8
        RETURN a13
    END FUNCTION
END CLASS
