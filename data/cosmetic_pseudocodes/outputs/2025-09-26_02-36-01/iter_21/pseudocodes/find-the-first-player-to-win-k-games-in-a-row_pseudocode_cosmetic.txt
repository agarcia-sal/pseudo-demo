CLASS Solution
    FUNCTION findWinningPlayer(a,b)
        SET c TO 0
        SET d TO 0
        SET e TO 0
        SET f TO 0
        SET g TO 0
        SET h TO 0

        PROCEDURE helper_initializeQueue(x)
            SET y TO 0
            SET z TO EMPTY LIST
            REPEAT
                APPEND y TO z
                SET y TO y + 1
                UNTIL y GREATER OR EQUAL TO x
            RETURN z
        END PROCEDURE

        SET d TO LENGTH OF a
        SET e TO helper_initializeQueue(d)
        SET f TO e[0]
        REMOVE e[0]
        SET g TO 0

        PROCEDURE isLoopConditionTrue(p,q,r,s)
            IF q LESS THAN r THEN
                IF LENGTH OF s GREATER THAN 0 THEN
                    RETURN TRUE
                ELSE
                    RETURN FALSE
                END IF
            ELSE
                RETURN FALSE
            END IF
        END PROCEDURE

        PROCEDURE getAndRemoveFirstElement(t)
            SET u TO t[0]
            REMOVE t[0]
            RETURN u
        END PROCEDURE

        label_loopStart:
        IF NOT isLoopConditionTrue(g,b,e,a) THEN GOTO label_loopEnd

        SET h TO getAndRemoveFirstElement(e)

        PROCEDURE compareSkills(i,j,k)
            IF i GREATER THAN j THEN RETURN TRUE ELSE RETURN FALSE END IF
        END PROCEDURE

        SET x1 TO a[f]
        SET x2 TO a[h]
        IF compareSkills(x1,x2,a,b) THEN
            SET g TO g + 1
            APPEND h TO e
        ELSE
            SET g TO 1
            APPEND f TO e
            SET f TO h
        END IF
        GOTO label_loopStart

        label_loopEnd:
        RETURN f
    END FUNCTION
END CLASS
