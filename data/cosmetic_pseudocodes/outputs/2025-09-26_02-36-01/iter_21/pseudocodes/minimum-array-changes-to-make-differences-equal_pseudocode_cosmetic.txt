CLASS Solution
    FUNCTION minChanges(nums k)
        SET e TO 0
        FUNCTION aux()
            IF e < ( ( LENGTH OF nums ) // 2 ) 
                SET m TO 0
                SET s TO LENGTH OF nums
                SET y TO 0
                SET z TO 0
                SET f TO 0
                SET a TO 0
                SET b TO 0
                SET p TO 0
                SET q TO 0

                SET m TO nums[e]
                SET s TO nums[ ( ( 0 - e ) - 1 ) ]

                IF NOT ( m <= s )
                    SET y TO m
                    SET z TO s
                    SET m TO z
                    SET s TO y
                ELSE
                    SET m TO m
                    SET s TO s
                END IF

                SET f TO 0 + 1
                d[0] = d[0] + f

                a = s - m
                d[a] = d[a] + ( -1 )

                b = a + 1
                d[b] = d[b] + ( 0 + 1 )

                p = MAX(s, (k - m)) + 1
                d[p] = d[p] + ( -1 )

                q = p + 1
                d[q] = d[q] + ( 0 + 1 )

                SET e TO e + 1
                CALL aux()
            END IF
        END FUNCTION

        SET d TO list of zeros of length ( k + 2 )
        CALL aux()
        SET c TO d[0]
        SET g TO c
        SET h TO 0
        WHILE h < ( LENGTH OF d )
            IF d[h] < g
                SET g TO d[h]
            END IF
            SET h TO h + 1
        END WHILE
        RETURN g
    END FUNCTION
END CLASS
