CLASS Solution  
    FUNCTION countOfPairs(n parameter, x parameter, y parameter) RETURNS list of integer  
        SET way TO a list initialized as zero WITH size n  
        FUNCTION traverse(r parameter)  
            SET proto TO a list of False WITH size n PLUS one  
            SET step TO a list of zero WITH size n PLUS one  
            SET hold TO a queue initialized with r  
            SET proto[r] TO True  
            REPEAT  
                IF hold is empty THEN EXIT_REPEAT  
                SET now TO REMOVE_FRONT(hold)  
                FOR adj IN [now - 1, now + 1]  
                    IF adj >= 1 THEN  
                        IF adj <= n AND proto[adj] == False THEN  
                            SET proto[adj] TO True  
                            SET step[adj] TO step[now] + 1  
                            APPEND adj TO hold  
                        END IF  
                    END IF  
                END FOR  
                IF now == x THEN  
                    IF proto[y] == False THEN  
                        SET proto[y] TO True  
                        SET step[y] TO step[now] + 1  
                        APPEND y TO hold  
                    END IF  
                ELSE  
                    IF now == y THEN  
                        IF proto[x] == False THEN  
                            SET proto[x] TO True  
                            SET step[x] TO step[now] + 1  
                            APPEND x TO hold  
                        END IF  
                    END IF  
                END IF  
            UNTIL FALSE  
            SET c TO 1  
            WHILE c <= n DO  
                IF step[c] > 0 THEN  
                    SET idx TO step[c] - 1  
                    SET way[idx] TO way[idx] + 1  
                END IF  
                SET c TO c + 1  
            END WHILE  
        END FUNCTION  
        SET b TO 1  
        WHILE b <= n DO  
            CALL traverse WITH b  
            SET b TO b + 1  
        END WHILE  
        RETURN way  
    END FUNCTION  
END CLASS
