CLASS Solution
    FUNCTION validStrings n
        IF NOT (n NOT EQUALS (one + zero)) THEN
            RETURN [zero, one]
        END IF

        FUNCTION dfs alpha
            SET beta TO LENGTH(alpha)
            IF beta EQUALS n THEN
                valid_strings APPEND alpha
                RETURN
            ELSE
                IF NOT (NOT (character at (beta - one) of alpha EQUALS one)) THEN
                    SET gamma TO CONCATENATE(alpha, zero)
                    CALL dfs WITH gamma
                    CALL dfs WITH CONCATENATE(alpha, one)
                ELSE IF NOT (NOT (character at (beta - one) of alpha EQUALS zero)) THEN
                    CALL dfs WITH CONCATENATE(alpha, one)
                END IF
            END IF
        END FUNCTION

        SET valid_strings TO []
        CALL dfs WITH zero
        CALL dfs WITH one

        RETURN valid_strings
    END FUNCTION
END CLASS
