CLASS Solution
    FUNCTION minimumTimeToInitialState(word PARAMETER k)
        SET a TO LENGTH OF word
        SET b TO 1
        DEFINE FUNCTION checkCondition(x)
            RETURN (x * k >= a) OR (SUBSTRING(word, (x * k) + 1, a) = SUBSTRING(word, 1, a - (x * k)))
        END FUNCTION

        PROCEDURE loop(c)
            IF NOT checkCondition(c) THEN
                CALL loop(c + 1)
            ELSE
                RETURN c
            END IF
        END PROCEDURE

        RETURN loop(b)
    END FUNCTION
END CLASS
