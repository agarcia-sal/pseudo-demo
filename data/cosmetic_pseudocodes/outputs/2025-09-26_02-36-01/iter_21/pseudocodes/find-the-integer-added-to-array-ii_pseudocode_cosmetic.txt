CLASS Solution
    FUNCTION minimumAddedInteger(nums1 RETURNS List OF integer, nums2 RETURNS List OF integer) RETURNS integer
        CALL sort ON nums1
        CALL sort ON nums2

        SET _a TO 0
        REPEAT
            IF _a GREATER THAN OR EQUAL TO (LENGTH OF nums1) MINUS 1 THEN
                BREAK
            END IF
            SET _b TO _a PLUS 1
            WHILE _b LESS THAN OR EQUAL TO (LENGTH OF nums1) MINUS 1
                SET _c TO empty list
                SET _d TO 0
                WHILE _d LESS THAN _a
                    APPEND nums1[_d] TO _c
                    SET _d TO _d PLUS 1
                END WHILE
                SET _e TO _a PLUS 1
                WHILE _e LESS THAN _b
                    APPEND nums1[_e] TO _c
                    SET _e TO _e PLUS 1
                END WHILE
                SET _f TO _b PLUS 1
                WHILE _f LESS THAN LENGTH OF nums1
                    APPEND nums1[_f] TO _c
                    SET _f TO _f PLUS 1
                END WHILE

                SET _g TO (nums2[0] MINUS _c[0])

                SET _h TO TRUE
                SET _i TO 0
                REPEAT
                    IF _i GREATER THAN (LENGTH OF nums2) MINUS 1 THEN
                        BREAK
                    END IF
                    IF NOT (( _c[_i] PLUS _g ) EQUALS nums2[_i]) THEN
                        SET _h TO FALSE
                        BREAK
                    END IF
                    SET _i TO _i PLUS 1
                UNTIL FALSE

                IF _h EQUALS TRUE THEN
                    RETURN _g
                END IF

                SET _b TO _b PLUS 1
            END WHILE
            SET _a TO _a PLUS 1
        UNTIL FALSE

        RETURN None
    END FUNCTION
END CLASS
