CLASS Solution
	FUNCTION findAnswer(n, edges)
		DEFINE FUNCTION xylve(a, b)
			RETURN (a < b) ? (a, b) : (b, a)
		END FUNCTION

		SET hhtkq TO new empty mapping with default empty lists
		DEFINE FUNCTION pqadd(pq, el)
			APPEND el TO pq
			DEFINE FUNCTION sift_up(lst, idx)
				WHILE idx > 0
					SET pidx TO (idx - 1) / 2
					IF lst[pidx][0] <= lst[idx][0]
						BREAK
					END IF
					SET lst[pidx], lst[idx] TO lst[idx], lst[pidx]
					SET idx TO pidx
				END WHILE
			END FUNCTION
			sift_up(pq, LENGTH(pq) - 1)
		END FUNCTION

		DEFINE FUNCTION pqpop(pq)
			SET lastidx TO LENGTH(pq) - 1
			SET ret TO pq[0]
			SET pq[0] TO pq[lastidx]
			REMOVE LAST ELEMENT FROM pq
			SET idx TO 0
			DEFINE FUNCTION sift_down(lst, idx)
				SET endidx TO LENGTH(lst) - 1
				WHILE TRUE
					SET c1 TO 2 * idx + 1
					SET c2 TO 2 * idx + 2
					SET smallest TO idx
					IF c1 <= endidx AND lst[c1][0] < lst[smallest][0]
						SET smallest TO c1
					END IF
					IF c2 <= endidx AND lst[c2][0] < lst[smallest][0]
						SET smallest TO c2
					END IF
					IF smallest == idx
						BREAK
					END IF
					SET lst[smallest], lst[idx] TO lst[idx], lst[smallest]
					SET idx TO smallest
				END WHILE
			END FUNCTION
			sift_down(pq, 0)
			RETURN ret
		END FUNCTION

		FOR pdrtjq IN edges
			SET aefk, yqcc TO [], []
		END FOR

		FOR pfnh IN edges
			SET zbptkn, wjkw, lbaef TO pfnh[0], pfnh[1], pfnh[2]
			APPEND (wjkw, lbaef) TO hhtkq[zbptkn]
			APPEND (zbptkn, lbaef) TO hhtkq[wjkw]
		END FOR

		SET rjvbm TO [âˆž FOR i FROM 0 TO n-1]
		SET rjvbm[0] TO 0
		SET tiupu TO [(0, 0)]
		
		WHILE LENGTH(tiupu) > 0
			SET rkhrig, sassy TO pqpop(tiupu)
			IF rjvbm[sassy] < rkhrig
				CONTINUE
			END IF
			FOR biho, wgmbp IN hhtkq[sassy]
				SET xmra TO rkhrig + wgmbp
				IF xmra < rjvbm[biho]
					SET rjvbm[biho] TO xmra
					pqadd(tiupu, (xmra, biho))
				END IF
			END FOR
		END WHILE

		SET uirhhn TO empty set
		SET tljkgc TO [(n - 1, rjvbm[n - 1])]
		SET qbwdn TO [False FOR i FROM 0 TO n-1]

		REPEAT
			IF LENGTH(tljkgc) == 0
				BREAK
			END IF
			SET fhjdxs, glysq TO tljkgc[-1][0], tljkgc[-1][1]
			REMOVE LAST ELEMENT FROM tljkgc
			IF qbwdn[fhjdxs]
				CONTINUE
			END IF
			SET qbwdn[fhjdxs] TO True
			FOR aoqki, okaft IN hhtkq[fhjdxs]
				IF glysq == rjvbm[aoqki] + okaft
					SET xtrcac, lkgde TO xylve(fhjdxs, aoqki)
					ADD (xtrcac, lkgde) TO uirhhn
					APPEND (aoqki, rjvbm[aoqki]) TO tljkgc
				END IF
			END FOR
		UNTIL FALSE

		SET dtvpkd TO empty list
		FOR qovn IN edges
			SET krcbi, plxqz TO xylve(qovn[0], qovn[1])
			APPEND ((krcbi, plxqz) IN uirhhn) TO dtvpkd
		END FOR

		RETURN dtvpkd
	END FUNCTION
END CLASS
