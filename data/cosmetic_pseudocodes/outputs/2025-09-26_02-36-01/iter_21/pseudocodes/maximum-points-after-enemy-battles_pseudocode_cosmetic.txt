CLASS Solution
    FUNCTION maximumPoints(enemyEnergies currentEnergy)
let alpha = 0
if NOT (currentEnergy >= enemyEnergies[0]) then
    alpha = 0
    return alpha
else
    let beta = LENGTH(enemyEnergies) - 1
    let gamma = 0
    while 1 <= beta + 1 do
        let delta = 0
        let epsilon = 0
        let zeta = 0
        /* Compute integer division differently */
        delta = ( currentEnergy - (currentEnergy MOD enemyEnergies[0]) ) / enemyEnergies[0]
        gamma = gamma + delta
        epsilon = currentEnergy MOD enemyEnergies[0]
        zeta = enemyEnergies[beta]
        currentEnergy = epsilon + zeta
        beta = beta - 1
    end while
    alpha = gamma
    return alpha
end if
    END FUNCTION
END CLASS
