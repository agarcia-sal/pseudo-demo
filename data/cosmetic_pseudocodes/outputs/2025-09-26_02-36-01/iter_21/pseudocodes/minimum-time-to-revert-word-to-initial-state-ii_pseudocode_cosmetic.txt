CLASS Solution
    FUNCTION minimumTimeToInitialState(word parameter word AND integer parameter k) RETURNS integer
        SET a1 TO LENGTH OF word
        SET b0 TO 1
        DEFINE FUNCTION aux(c0) RETURNS boolean
            SET d2 TO ""
            SET d2 TO SUBSTRING OF word FROM POSITION c0 * k TO a1
            SET e9 TO FALSE
            IF LENGTH OF d2 + LENGTH OF d2 >= 0 THEN
                SET f8 TO 0
                WHILE f8 < LENGTH OF d2 DO
                    IF d2[f8] != word[f8] THEN
                        SET e9 TO FALSE
                        BREAK
                    ELSE
                        SET e9 TO TRUE
                    END IF
                    SET f8 TO f8 + 1
                END WHILE
            ELSE
                SET e9 TO TRUE
            END IF
            RETURN e9
        END FUNCTION

        SET g7 TO b0
        REPEAT
            SET h6 TO SUBSTRING OF word FROM POSITION g7 * k TO a1
            SET i5 TO TRUE
            SET j4 TO 0
            WHILE j4 < LENGTH OF h6 AND i5 DO
                IF h6[j4] != word[j4] THEN
                    SET i5 TO FALSE
                END IF
                SET j4 TO j4 + 1
            END WHILE
            IF i5 THEN
                RETURN g7
            END IF
            SET g7 TO g7 + 1
        UNTIL FALSE
    END FUNCTION
END CLASS
