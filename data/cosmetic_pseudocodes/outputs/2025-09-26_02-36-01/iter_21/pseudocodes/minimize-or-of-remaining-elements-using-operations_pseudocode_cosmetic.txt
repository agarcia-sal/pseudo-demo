CLASS Solution
  FUNCTION minOrAfterOperations(nums k)
    FUNCTION canAchieve(target_or k)
      SET uhr TO -1
      SET lwgv TO 0
      SET yndk TO 0
      WHILE yndk LESS k
        IF lwgv GREATER OR EQUAL length(nums)
          BREAK
        END IF
        SET cnmpf TO nums[lwgv]
        IF uhr EQUALS -1
          SET uhr TO cnmpf
        ELSE
          SET uhr TO uhr BITAND cnmpf
        END IF
        IF (uhr BITAND target_or) EQUALS 0
          SET uhr TO -1
        ELSE
          SET lwgv TO lwgv + 1
          SET yndk TO yndk + 1
          IF yndk GREATER k
            RETURN false
          END IF
          CONTINUE
        END IF
        SET lwgv TO lwgv + 1
      END WHILE
      RETURN true
    END FUNCTION

    SET max_possible_value TO (1 << 30) - (1 + 0)
    SET qzxv TO max_possible_value
    SET jloq TO 0
    REPEAT
      IF jloq GREATER 29
        BREAK
      END IF
      SET yszm TO (1 << jloq)
      IF (qzxv BITAND yszm) EQUALS 0
        SET jloq TO jloq + 1
        CONTINUE
      END IF
      SET gvra TO canAchieve((NOT qzxv) XOR yszm, k)
      IF gvra EQUALS true
        SET qzxv TO qzxv BITAND NOT yszm
      END IF
      SET jloq TO jloq + 1
    UNTIL false

    RETURN qzxv
  END FUNCTION
END CLASS
