CLASS Solution
    FUNCTION occurrencesOfElement(nums, queries, x)
        DECLARE aList AS list ← []
        DECLARE bCounter AS integer ← 0
        DECLARE cIndex AS integer
        DECLARE idHolder AS integer
        DECLARE resultShield AS list ← []
        DECLARE positionSave AS integer

        FUNCTION recordOccurrence(position)
            aList <- aList + [position]
        END FUNCTION

        FUNCTION fetchOccurrence(pos) RETURNS integer
            RETURN aList[pos - 1]
        END FUNCTION

        cIndex <- 0
        WHILE TRUE
            IF cIndex >= LENGTH(nums) THEN
                BREAK
            END IF

            idHolder <- nums[cIndex]
            IF NOT (idHolder != x) THEN
                recordOccurrence(cIndex)
            END IF

            cIndex <- cIndex + 1
        END WHILE

        DECLARE queryPosition AS integer ← 0
        DECLARE queryLimit AS integer ← LENGTH(aList)

        FUNCTION appendNegativeOne()
            resultShield <- resultShield + [-(0 + 1)]
        END FUNCTION

        REPEAT
            IF queryPosition >= LENGTH(queries) THEN
                EXIT REPEAT
            END IF

            positionSave <- queries[queryPosition]
            IF positionSave <= queryLimit THEN
                resultShield <- resultShield + [fetchOccurrence(positionSave)]
            ELSE
                appendNegativeOne()
            END IF

            queryPosition <- queryPosition + 1
        UNTIL FALSE

        RETURN resultShield
    END FUNCTION
END CLASS
