CLASS Solution
    FUNCTION minOperations(k)
        SET R0 TO some very large number
        SET R1 TO 1
        LABEL_START:
        IF NOT (R1 * R1 <= k) THEN GOTO LABEL_END
        SET R2 TO ((k + R1 - 1) DIV R1)
        SET R3 TO (R1 - 1) + (R2 - 1)
        IF NOT (R3 >= R0) THEN
            SET R0 TO R3
        END IF
        SET R1 TO R1 + 1
        GOTO LABEL_START
        LABEL_END:
        RETURN R0
    END FUNCTION
END CLASS
