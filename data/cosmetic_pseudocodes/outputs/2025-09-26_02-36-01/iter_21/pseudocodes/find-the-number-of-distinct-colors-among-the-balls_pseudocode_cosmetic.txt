CLASS Solution
	FUNCTION queryResults(limit parameter, queries parameter)
		LET pdoa TO an empty list for storing counts of unique colors
		LET frsb TO an empty map from identifiers to colors
		LET vneo TO an empty collection for distinct colors

		DEFINE FUNCTION dplr(xxsb, mvrq)
			LET qcik TO (xxsb, mvrq)
			IF qcik[0] IN frsb
				LET cfod TO frsb[qcik[0]]
				WHILE (cfod IN vneo) AND (vneo REMOVE cfod IS TRUE)
					BREAK
				END WHILE
			END IF

			frsb[qcik[0]] = qcik[1]
			vneo ADD qcik[1]

		END FUNCTION

		LET voxa TO 0
		WHILE voxa < limit parameter
			CALL dplr(queries parameter[voxa][0], queries parameter[voxa][1])
			pdoa APPEND LENGTH(vneo)
			voxa = voxa + 1
		END WHILE

		RETURN pdoa
	END FUNCTION
END CLASS
