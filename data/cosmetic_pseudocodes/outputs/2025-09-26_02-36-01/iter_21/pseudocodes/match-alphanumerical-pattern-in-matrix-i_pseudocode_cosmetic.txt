CLASS Solution
    FUNCTION findPattern(board, pattern)
let a, b, c, d, e, f, g, h, i, j, k, l, m

        SET a TO LENGTH OF board
        SET b TO LENGTH OF element AT POSITION 0 OF board
        SET c TO LENGTH OF pattern
        SET d TO LENGTH OF element AT POSITION 0 OF pattern

        FUNCTION matches(x, y)
let u, v, w, x0, y0, z0, aa, bb, cc

            SET u TO empty mapping
            SET v TO empty mapping
            SET w TO 0
            WHILE w < c DO
                SET x0 TO 0
                REPEAT
                    SET y0 TO element AT POSITION w OF pattern THEN element AT POSITION x0
                    SET z0 TO element AT POSITION x PLUS w OF board THEN element AT POSITION y PLUS x0
                    IF y0 IS A digit THEN
                        SET aa TO NOT (INTEGER VALUE OF y0 = z0)
                        IF aa THEN
                            RETURN false
                        END IF
                    ELSE
                        IF y0 IN u THEN
                            SET bb TO NOT (u[y0] = z0)
                            IF bb THEN
                                RETURN false
                            END IF
                        ELSE
                            IF z0 IN v THEN
                                RETURN false
                            END IF
                            ASSIGN z0 TO u[y0]
                            ASSIGN y0 TO v[z0]
                        END IF
                    END IF
                    SET x0 TO x0 + 1
                UNTIL x0 = d
                SET w TO w + 1
            END WHILE
            RETURN true
        END FUNCTION

        LET r_temp TO 0
        WHILE r_temp <= a - c
            LET c_temp TO 0
            WHILE c_temp <= b - d
                IF matches(r_temp, c_temp) THEN
                    RETURN [r_temp, c_temp]
                END IF
                SET c_temp TO c_temp + 1
            END WHILE
            SET r_temp TO r_temp + 1
        END WHILE

        LET retList TO [-1 + 0, -1 * 1]
        RETURN retList
    END FUNCTION
END CLASS
