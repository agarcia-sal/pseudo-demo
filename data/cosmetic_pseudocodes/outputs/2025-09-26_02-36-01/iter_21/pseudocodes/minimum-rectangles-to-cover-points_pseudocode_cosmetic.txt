CLASS Solution
  FUNCTION minRectanglesToCoverPoints(points, w)
    SET vX TO 0
    SET vY TO 0
    SET resCount TO 0
    CALL quickSortByX(points, 0, LENGTH(points) - 1)
    SET vU TO -1
    DEFINE recurseIndex(idx)
      IF NOT (idx < LENGTH(points)) THEN
        RETURN
      END IF
      SET vX TO points[idx][0]
      SET vY TO points[idx][1]
      IF vX > vU THEN
        SET vU TO vX + w
        SET resCount TO resCount + 1
      END IF
      CALL recurseIndex(idx + 1)
    END DEFINE
    CALL recurseIndex(0)
    RETURN resCount
  END FUNCTION

  FUNCTION quickSortByX(arr, start, end)
    IF NOT (start < end) THEN
      RETURN
    END IF
    SET pivotVal TO arr[end][0]
    SET pIndex TO start
    SET i TO start
    WHILE i < end
      SET vA TO arr[i][0]
      IF vA <= pivotVal
        CALL swap(arr, i, pIndex)
        SET pIndex TO pIndex + 1
      END IF
      SET i TO i + 1
    END WHILE
    CALL swap(arr, pIndex, end)
    CALL quickSortByX(arr, start, pIndex - 1)
    CALL quickSortByX(arr, pIndex + 1, end)
  END FUNCTION

  FUNCTION swap(arr, idx1, idx2)
    SET temp TO arr[idx1]
    SET arr[idx1] TO arr[idx2]
    SET arr[idx2] TO temp
  END FUNCTION
END CLASS
