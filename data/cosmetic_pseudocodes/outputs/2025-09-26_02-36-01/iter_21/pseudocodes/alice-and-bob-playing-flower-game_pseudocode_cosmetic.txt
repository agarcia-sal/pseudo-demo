CLASS Solution
    FUNCTION flowerGame(a, b)
        PROCEDURE accumulatePairs(x, y, resultRef)
            SET i TO 1
            WHILE i <= x
                SET j TO 1
                WHILE j <= y
                    IF ((i + j) MOD 2) == 1 THEN
                        SET resultRef[0] TO resultRef[0] + 1
                    END IF
                    SET j TO j + 1
                END WHILE
                SET i TO i + 1
            END WHILE
        END PROCEDURE

        SET pairCount TO 0
        DECLARE accumulator AS LIST = [pairCount]

        CALL accumulatePairs((a + 1) - ((a + 1) DIV 2) * 2, b DIV 2, accumulator)
        CALL accumulatePairs(a DIV 2, (b + 1) - ((b + 1) DIV 2)*2, accumulator)

        RETURN accumulator[0]
    END FUNCTION
END CLASS
