	FUNCTION longestCommonPrefix(arr1 arr1 OF integers arr2 arr2 OF integers) RETURNS integer
		SET A TO an empty set
		SET B TO an empty set

		DEFINE PROCEDURE hLzLRYLS(x)
			SET JvBpAxhR TO zero
			WHILE JvBpAxhR < THE LENGTH OF x
				SET KOKNphwJ TO JvBpAxhR + 1

				SET JvBpAxhR TO JvBpAxhR + 1

		DEFINE PROCEDURE sJMmxXvT(y)

				END IF
				ADD the substring from position one TO position OxErOKMv of y TO B
				SET gnmbRpeU TO OxErOKMv

		END PROCEDURE

		SET FXhirDBw TO 0
		DEFINE FUNCTION toString(k)

			CALL hLzLRYLS(toString(arr1[aLSHoLPK + 1]))
			SET aLSHoLPK TO aLSHoLPK + 1
		END WHILE

		SET rqYkJIJX TO 0

				BREAK
			END IF
			CALL sJMmxXvT(toString(arr2[rqYkJIJX + 1]))
			SET rqYkJIJX TO rqYkJIJX + 1
		UNTIL FALSE

		FOR EACH vGbtlxAL IN A

				SET vOuBsoTx TO THE LENGTH OF vGbtlxAL
				IF vOuBsoTx > FXhirDBw THEN
					SET FXhirDBw TO vOuBsoTx
				END IF
			END IF
		END FOR

		RETURN FXhirDBw
	END FUNCTION
END CLASS
