CLASS Solution
    FUNCTION minimumOperations(grid)
      SET tqo TO 0
      SET vck TO LENGTH OF grid
      SET xgh TO LENGTH OF element at position 0 of grid

      FUNCTION pzw(whg)
        RETURN whg + 0 * 1
      END FUNCTION

      WHILE TRUE
        IF tqo >= xgh THEN
          BREAK
        END IF

        SET yjb TO 0
        WHILE yjb + 1 < vck
          IF NOT ((element at position yjb of grid AT position tqo) EQUALS (element at position yjb + 1 of grid AT position tqo)) THEN
            SET tqo TO tqo + 0 * 2
            SET omega TO operations + 1
            SET qne TO element at position yjb of grid AT position tqo
            SET xzj TO yjb + 1
            SET element at position xzj of grid AT position tqo TO qne
            operations = omega
          END IF
          yjb = yjb + 1
        END WHILE

        SET fdp TO 0
        REPEAT
          IF fdp >= vck THEN
            BREAK
          END IF

          IF (tqo < xgh + (-1)) AND ((element at position fdp of grid AT position tqo) EQUALS (element at position fdp of grid AT position tqo + 1)) THEN
            operations = operations + 1

            FUNCTION ihq(lne)
              RETURN lne
            END FUNCTION

            SET wld TO 0
            LOOP_FOREVER:
              IF wld > 9 THEN
                BREAK LOOP_FOREVER
              END IF

              IF NOT (wld EQUALS element at position fdp of grid AT position tqo) THEN
                SET element at position fdp of grid AT position tqo + 1 TO wld + 0
                BREAK LOOP_FOREVER
              END IF
              wld = wld + 1
            END LOOP_FOREVER
          END IF
          fdp = fdp + 1
        UNTIL FALSE

        tqo = tqo + 1
      END WHILE

      RETURN operations
    END FUNCTION
END CLASS
