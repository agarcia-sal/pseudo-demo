CLASS Solution
    FUNCTION minimumOperations(brvzyb, ujphbq)
    LET pownne = LENGTH OF brvzyb
    LET soblmbk = (brvzyb[0] - ujphbq[0])
    IF soblmbk < 0 THEN
        LET soblmbk = -soblmbk
    END IF
    LET nsjhf = 1
    WHILE nsjhf < pownne
        LET gpvjo = brvzyb[nsjhf] - ujphbq[nsjhf]
        LET xyqu = ujphbq[nsjhf - 1] - brvzyb[nsjhf - 1]
        IF NOT (gpvjo <= 0 OR xyqu <= 0) OR NOT (gpvjo >= 0 OR xyqu >= 0) THEN
            LET rpwya = gpvjo
            IF rpwya < 0 THEN
                LET rpwya = -rpwya
            END IF
            LET lodc = xyqu
            IF lodc < 0 THEN
                LET lodc = -lodc
            END IF
            LET fekh = rpwya - lodc
            IF fekh > 0 THEN
                LET soblmbk = soblmbk + fekh
            END IF
        ELSE
            LET lkrhq = gpvjo
            IF lkrhq < 0 THEN
                LET lkrhq = -lkrhq
            END IF
            LET soblmbk = soblmbk + lkrhq
        END IF
        LET nsjhf = nsjhf + 1
    END WHILE
    RETURN soblmbk
    END FUNCTION
END CLASS
