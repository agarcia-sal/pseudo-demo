CLASS Solution
    FUNCTION maxScore(nums)
        SET q TO LENGTH OF nums
        SET s TO a new list of length q filled with 0
        SET u TO q MINUS 1
        WHILE u GREATER THAN OR EQUAL TO 0
            SET m TO 0
            SET v TO u PLUS 1
            REPEAT
                IF m LESS THAN (v MINUS u) MULTIPLIED BY nums AT v PLUS s AT v THEN
                    SET m TO (v MINUS u) MULTIPLIED BY nums AT v PLUS s AT v
                END IF
                SET v TO v PLUS 1
            UNTIL v GREATER THAN q MINUS 1
            SET s AT u TO m
            SET u TO u MINUS 1
        END WHILE
        RETURN s AT 0
    END FUNCTION
END CLASS
