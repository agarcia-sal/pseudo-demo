CLASS Solution
	FUNCTION minCostToEqualizeArray(dvszbdf oxifre xesoen)
		SET trlxcjvn TO 1000000000 ADD 7
		SET rjxpmti TO 0
		WHILE rjxpmti LESS THAN THE LENGTH OF dvszbdf
			INCREMENT rjxpmti BY 1
		END WHILE
		SET thlotl TO dvszbdf[0]
		SET qrekyge TO dvszbdf[0]
		SET vrmuoujx TO 0
		SET ttplfi TO 0
		SET eyrqvkp TO 0
		WHILE eyrqvkp LESS THAN rjxpmti
			IF dvszbdf[eyrqvkp] LESS THAN thlotl
				SET thlotl TO dvszbdf[eyrqvkp]
			END IF
			IF dvszbdf[eyrqvkp] GREATER THAN qrekyge
				SET qrekyge TO dvszbdf[eyrqvkp]
			END IF
			SET vrmuoujx TO vrmuoujx ADD dvszbdf[eyrqvkp]
			INCREMENT eyrqvkp BY 1
		END WHILE

		IF oxifre ADD oxifre LESS THAN OR EQUAL TO xesoen OR rjxpmti LESS THAN 3
			SET iufhely TO qrekyge MULTIPLIED BY rjxpmti SUBTRACT vrmuoujx
			RETURN (oxifre MULTIPLIED BY iufhely) MOD trlxcjvn
		END IF

		FUNCTION getMinCost(stmxqk)
			SET qevphtz TO stmxqk SUBTRACT thlotl
			SET vuqxcs TO stmxqk MULTIPLIED BY rjxpmti SUBTRACT vrmuoujx
			SET jqryco TO vuqxcs DIVIDED BY 2
			IF jqryco GREATER THAN vuqxcs SUBTRACT qevphtz
				SET jqryco TO vuqxcs SUBTRACT qevphtz
			END IF
			RETURN (oxifre MULTIPLIED BY vuqxcs) SUBTRACT (2 MULTIPLIED BY jqryco MULTIPLIED BY oxifre) ADD (jqryco MULTIPLIED BY xesoen)
		END FUNCTION

		SET bnwjk TO xesoen
		SET bvlsf TO qrekyge
		SET uatyxnz TO (2 MULTIPLIED BY qrekyge) SUBTRACT 1
		WHILE bvlsf LESS THAN OR EQUAL TO uatyxnz
			SET qkdptw TO getMinCost(bvlsf)
			IF qkdptw LESS THAN bnwjk
				SET bnwjk TO qkdptw
			END IF
			ADD 1 TO bvlsf
		END WHILE
		RETURN bnwjk MOD trlxcjvn
	END FUNCTION
END CLASS
