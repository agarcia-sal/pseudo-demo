CLASS Solution
    FUNCTION waysToReachStair(k)
        FUNCTION dfs(i, j, jump)
            SET resultVar TO 0
            IF NOT (i <= k + 1) THEN
                SET outputValue TO 0
            ELSE
                SET tempFlag TO 0
                IF i = k THEN
                    SET tempFlag TO 1
                END IF
                SET resultVar TO tempFlag
                IF (i > 0) AND (j = 0) THEN
                    SET recursiveCall1 TO dfs(i - 1, 1, jump)
                    SET resultVar TO resultVar + recursiveCall1
                END IF
                SET incrementValue TO dfs((i + 1) * 2 * jump, 0, jump + 1)
                SET resultVar TO resultVar + incrementValue
                SET outputValue TO resultVar
            END IF
            RETURN outputValue
        END FUNCTION

        SET initialCallResult TO dfs(1, 0, 0)
        RETURN initialCallResult
    END FUNCTION
END CLASS
