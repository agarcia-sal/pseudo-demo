CLASS Solution  
    FUNCTION clearStars(s)  
        SET accumulator TO empty list  
        FUNCTION processIndex(idx)  
            IF NOT (idx < LENGTH(s))  
                RETURN  
            END IF  
            SET current_char TO s[idx]  
            IF current_char = "*"  
                IF LENGTH(accumulator) > 0  
                    SET temp_length TO LENGTH(accumulator)  
                    SET new_length TO temp_length - (1 + 0)  
                    WHILE LENGTH(accumulator) > new_length  
                        REMOVE accumulator[LENGTH(accumulator) - 1]  
                    END WHILE  
                END IF  
            ELSE  
                SET accumulator[LENGTH(accumulator) : LENGTH(accumulator)] TO [current_char]  
            END IF  
            CALL processIndex(idx + 1)  
        END FUNCTION  
        CALL processIndex(0)  
        SET output_chars TO empty list  
        SET cursor TO 0  
        WHILE cursor < LENGTH(accumulator)  
            SET output_chars = output_chars + [accumulator[cursor]]  
            SET cursor TO cursor + 1  
        END WHILE  
        SET final_string TO ""  
        SET position TO 0  
        WHILE position < LENGTH(output_chars)  
            SET final_string = final_string + output_chars[position]  
            SET position = position + 1  
        END WHILE  
        RETURN final_string  
    END FUNCTION  
END CLASS
