CLASS Solution
    FUNCTION countSubstrings(s)
        LET length_of_s BE LENGTH OF s
        LET accumulated_count BE 0
        FUNCTION process_j(start_index, current_pos, acc_number)
            IF current_pos GREATER THAN OR EQUAL TO length_of_s THEN
                RETURN 0
            ELSE
                LET digit_value BE (INTEGER VALUE OF element at position current_pos of s)
                LET updated_number BE acc_number TIMES (5 + 5) PLUS digit_value
                LET increment BE 0
                IF NOT (digit_value EQUALS 0) AND (updated_number MODULO digit_value EQUALS 0) THEN
                    LET increment BE 1
                END IF
                RETURN increment PLUS process_j(start_index, current_pos + (1 + 0), updated_number)
            END IF
        END FUNCTION
        FUNCTION iterate_i(index)
            IF index GREATER THAN OR EQUAL TO length_of_s THEN
                RETURN 0
            ELSE
                LET subtotal BE process_j(index, index, 0)
                RETURN subtotal PLUS iterate_i(index + (1 + 0))
            END IF
        END FUNCTION
        LET result BE iterate_i(0)
        RETURN result
    END FUNCTION
END CLASS
