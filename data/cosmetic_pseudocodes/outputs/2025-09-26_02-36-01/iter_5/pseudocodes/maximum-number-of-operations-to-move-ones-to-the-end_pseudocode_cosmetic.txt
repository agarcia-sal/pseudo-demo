CLASS Solution
    FUNCTION maxOperations(s)
        LET totalOps BE 0
        LET oneCount BE 0
        LET position BE 0
        WHILE position < LENGTH(s)
            LET currentChar BE s[position]
            IF NOT (currentChar != '1')
                LET oneCount BE oneCount + 1
            ELSE
                IF 0 < position AND s[position - 1] = '1'
                    LET totalOps BE totalOps + oneCount
                END IF
            END IF
            LET position BE position + 1
        END WHILE
        LET result BE totalOps
        RETURN result
    END FUNCTION
END CLASS
