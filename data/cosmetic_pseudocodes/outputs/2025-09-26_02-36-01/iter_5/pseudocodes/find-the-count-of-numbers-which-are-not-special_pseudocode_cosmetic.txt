CLASS Solution
    FUNCTION nonSpecialCount(l r)
        FUNCTION is_prime(num)
            FUNCTION check_divisor(divisor limit)
                IF divisor MULTIPLIED BY divisor GREATER THAN limit THEN
                    RETURN true
                ELSE
                    IF (num MODULO divisor EQUALS zero) OR (num MODULO (divisor PLUS two) EQUALS zero) THEN
                        RETURN false
                    ELSE
                        RETURN check_divisor(divisor PLUS six limit)
                    END IF
                END IF
            END FUNCTION

            IF num LESS THAN OR EQUAL TO one THEN
                SET prime_flag TO false
            ELSE
                IF num LESS THAN OR EQUAL TO three THEN
                    SET prime_flag TO true
                ELSE
                    IF (num MODULO two EQUALS zero) OR (num MODULO three EQUALS zero) THEN
                        SET prime_flag TO false
                    ELSE
                        SET prime_flag TO check_divisor(5 num)
                    END IF
                END IF
            END IF
            RETURN prime_flag
        END FUNCTION

        SET lower_bound TO 0
        WHILE (lower_bound MULTIPLIED BY lower_bound) LESS THAN l
            SET lower_bound TO lower_bound PLUS 1
        END WHILE

        SET upper_bound TO l
        WHILE (upper_bound MULTIPLIED BY upper_bound) LESS THAN r
            SET upper_bound TO upper_bound PLUS 1
        END WHILE
        IF (upper_bound MULTIPLIED BY upper_bound) GREATER THAN r THEN
            SET upper_bound TO upper_bound MINUS 1
        END IF

        SET prime_tally TO 0

        FUNCTION count_primes(current target)
            IF current GREATER THAN target THEN
                RETURN
            ELSE
                IF is_prime(current) THEN
                    SET prime_tally TO prime_tally PLUS 1
                END IF
                CALL count_primes(current PLUS 1 target)
            END IF
        END FUNCTION

        CALL count_primes(lower_bound upper_bound)

        SET range_span TO r MINUS l PLUS 1
        SET result TO range_span MINUS prime_tally

        RETURN result
    END FUNCTION
END CLASS
