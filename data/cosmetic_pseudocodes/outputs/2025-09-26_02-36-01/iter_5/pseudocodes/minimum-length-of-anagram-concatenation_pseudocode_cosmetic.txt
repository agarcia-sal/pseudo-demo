CLASS Solution
    FUNCTION minAnagramLength(s)
        FUNCTION countUniqueChars(index, charsSet) 
            IF index >= LENGTH(s) THEN
                SET distinctCount TO LENGTH(charsSet)
                RETURN distinctCount
            ELSE
                SET currentChar TO s[index]
                IF NOT (currentChar IN charsSet) THEN
                    SET updatedSet TO charsSet UNION {currentChar}
                ELSE
                    SET updatedSet TO charsSet
                END IF
                RETURN countUniqueChars(index + (1), updatedSet)
            END IF
        END FUNCTION

        SET initialSet TO {}
        SET resultValue TO countUniqueChars(0, initialSet)
        RETURN resultValue
    END FUNCTION
END CLASS
