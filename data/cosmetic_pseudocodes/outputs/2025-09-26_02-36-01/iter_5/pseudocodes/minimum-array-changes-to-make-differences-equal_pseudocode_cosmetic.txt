CLASS Solution  
    FUNCTION minChanges(array count)  
        SET table TO list of zeros WITH LENGTH count PLUS two  
        SET lengthVal TO LENGTH OF array  
        
        FUNCTION processIndex(idx curr, idx limit)  
            IF curr GREATER THAN limit  
                RETURN  
            END IF  
            
            SET firstVal TO element at position curr of array  
            SET secondVal TO element at position negative curr MINUS one of array  
            
            IF NOT (firstVal LESS THAN OR EQUAL TO secondVal)  
                SET tempVal TO firstVal  
                SET firstVal TO secondVal  
                SET secondVal TO tempVal  
            END IF
            
            SET temp1 TO element at position zero of table PLUS one  
            SET element at position zero of table TO temp1  
            
            SET diff1 TO secondVal MINUS firstVal  
            SET temp2 TO element at position diff1 of table MINUS one  
            SET element at position diff1 of table TO temp2  
            
            SET diff2 TO diff1 PLUS one  
            SET temp3 TO element at position diff2 of table PLUS one  
            SET element at position diff2 of table TO temp3
            
            SET maxVal TO firstVal  
            IF secondVal GREATER THAN maxVal  
                SET maxVal TO secondVal  
            END IF
            
            SET index1 TO maxVal PLUS (count MINUS firstVal) PLUS one
            SET temp4 TO element at position index1 of table MINUS one
            SET element at position index1 of table TO temp4
            
            SET index2 TO maxVal PLUS (count MINUS firstVal) PLUS two
            SET temp5 TO element at position index2 of table PLUS one
            SET element at position index2 of table TO temp5
            
            CALL processIndex(curr PLUS one, limit)
        END FUNCTION
        
        CALL processIndex(zero, (lengthVal INTEGER DIVIDED BY two) MINUS one)
        
        SET runningSum TO zero
        SET minResult TO none
        SET idxCounter TO zero
        
        WHILE idxCounter LESS THAN LENGTH OF table  
            SET runningSum TO runningSum PLUS element at position idxCounter of table  
            IF minResult IS none OR runningSum LESS THAN minResult  
                SET minResult TO runningSum  
            END IF  
            SET idxCounter TO idxCounter PLUS one  
        END WHILE
        
        RETURN minResult  
    END FUNCTION  
END CLASS
