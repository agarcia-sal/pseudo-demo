CLASS Solution
    FUNCTION occurrencesOfElement(nums, queries, x)
        FUNCTION collectPositions(arr, target, idx, acc)
            IF idx GREATER THAN LENGTH OF arr MINUS one
                RETURN acc
            ELSE
                SET currentVal TO arr[idx]
                SET newAcc TO acc
                IF currentVal EQUALS target
                    SET tempList TO [idx]
                    SET newAcc TO newAcc + tempList
                END IF
                RETURN collectPositions(arr, target, idx + 1, newAcc)
            END IF
        END FUNCTION

        SET positions TO collectPositions(nums, x, zero, [])

        FUNCTION processQueries(qList, posList, pos, res)
            IF pos GREATER THAN LENGTH OF qList MINUS one
                RETURN res
            ELSE
                SET currentQuery TO qList[pos]
                SET resultValue TO -1
                IF NOT (currentQuery GREATER THAN LENGTH OF posList)
                    SET resultValue TO posList[currentQuery - 1]
                END IF
                RETURN processQueries(qList, posList, pos + 1, res + [resultValue])
            END IF
        END FUNCTION

        SET finalAnswer TO processQueries(queries, positions, zero, [])
        RETURN finalAnswer
    END FUNCTION
END CLASS
