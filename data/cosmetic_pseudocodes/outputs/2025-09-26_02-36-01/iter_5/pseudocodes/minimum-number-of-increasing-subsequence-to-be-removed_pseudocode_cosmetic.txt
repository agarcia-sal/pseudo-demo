CLASS Solution
    FUNCTION minOperations(inputList)
        SET sizeCounter TO LENGTH OF inputList
        IF NOT (sizeCounter GREATER THAN zero)
            SET resultValue TO zero
        ELSE
            DEFINE FUNCTION helper(index, dpArray)
                IF index GREATER OR EQUAL TO sizeCounter
                    RETURN dpArray
                ELSE
                    DEFINE FUNCTION innerLoop(pointer, currentDp)
                        IF pointer GREATER OR EQUAL TO index
                            RETURN currentDp
                        ELSE
                            SET conditionCheck TO NOT (inputList[index] GREATER THAN inputList[pointer])
                            IF conditionCheck
                                SET updatedValue TO MAXIMUM OF currentDp[index] AND (currentDp[pointer] PLUS 1)
                                SET currentDp[index] TO updatedValue
                            END IF
                            RETURN innerLoop(pointer PLUS 1, currentDp)
                        END IF
                    END FUNCTION
                    SET updatedDp TO innerLoop(zero, dpArray)
                    RETURN helper(index PLUS 1, updatedDp)
                END IF
            END FUNCTION
            SET initialDp TO A LIST filled with ones, length sizeCounter
            SET finalDp TO helper(one, initialDp)
            SET resultValue TO A NEGATIVE INFINITY REPRESENTATION
            DEFINE FUNCTION findMax(valueList, pos, currentMax)
                IF pos GREATER OR EQUAL TO LENGTH OF valueList
                    RETURN currentMax
                ELSE
                    SET updatedMax TO currentMax
                    IF valueList AT pos GREATER THAN currentMax
                        SET updatedMax TO valueList AT pos
                    END IF
                    RETURN findMax(valueList, pos PLUS 1, updatedMax)
                END IF
            END FUNCTION
            SET resultValue TO findMax(finalDp, zero, NEGATIVE_INFINITY)
        END IF
        RETURN resultValue
    END FUNCTION
END CLASS
