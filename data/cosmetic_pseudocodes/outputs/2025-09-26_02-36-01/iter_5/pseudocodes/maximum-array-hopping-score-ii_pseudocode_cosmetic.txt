CLASS Solution  
    FUNCTION maxScore(vals)  
        SET length_var TO (0)  
        SET length_var TO (1 + (SIZE(vals) - 1))  
        SET results TO ARRAY OF (length_var) ELEMENTS WITH VALUE (0)  
        FUNCTION traverse(index)  
            IF index EQUALS length_var THEN  
                RETURN 0  
            ELSE  
                SET accumulator TO (0)  
                FUNCTION inner_loop(pos, current_max)  
                    IF pos EQUALS length_var THEN  
                        RETURN current_max  
                    ELSE  
                        SET distance TO pos - index  
                        SET temp_score TO distance * vals[pos]  
                        SET candidate TO temp_score + results[pos]  
                        IF current_max < candidate THEN  
                            SET current_max TO candidate  
                        END IF  
                        RETURN inner_loop(pos + 1, current_max)  
                    END IF  
                END FUNCTION  
                SET accumulator TO inner_loop(index + 1, 0)  
                SET results[index] TO accumulator  
                RETURN traverse(index + 1)  
            END IF  
        END FUNCTION  
        CALL traverse(0)  
        SET final_result TO results[0]  
        RETURN final_result  
    END FUNCTION  
END CLASS
