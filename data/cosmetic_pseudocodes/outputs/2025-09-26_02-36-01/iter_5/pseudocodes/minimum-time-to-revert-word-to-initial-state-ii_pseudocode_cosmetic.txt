CLASS Solution
    FUNCTION minimumTimeToInitialState(word parameter input_word AND integer parameter key) RETURNS integer
        SET total_length TO LENGTH OF input_word
        SET elapsed TO 1
        FUNCTION checkPrefix(index)
            IF index IS GREATER THAN total_length THEN
                RETURN -1
            END IF
            SET segment TO SUBSTRING(input_word, index * key, total_length - index * key)
            IF input_word BEGINS_WITH segment THEN
                RETURN index
            ELSE
                RETURN checkPrefix(index + 1)
            END IF
        END FUNCTION
        SET result TO checkPrefix(elapsed)
        RETURN result
    END FUNCTION
END CLASS
