CLASS Solution
    FUNCTION validStrings n
        SET results TO empty list
        FUNCTION explore sequence
            IF (LENGTH sequence) IS_EQUAL_TO n THEN
                APPEND sequence TO results
                RETURN
            ELSE
                SET lastCharIndex TO (LENGTH sequence) MINUS 1
                SET lastChar TO sequence AT lastCharIndex
                IF lastChar IS_EQUAL_TO "1" THEN
                    CALL explore WITH (sequence CONCATENATED WITH "0")
                    CALL explore WITH (sequence CONCATENATED WITH "1")
                ELSE
                    IF lastChar IS_EQUAL_TO "0" THEN
                        CALL explore WITH (sequence CONCATENATED WITH "1")
                    END IF
                END IF
            END IF
        END FUNCTION

        IF n IS_EQUAL_TO 1 THEN
            SET tempResult TO ["0", "1"]
        ELSE
            CALL explore WITH "0"
            CALL explore WITH "1"
            SET tempResult TO results
        END IF

        RETURN tempResult
    END FUNCTION
END CLASS
