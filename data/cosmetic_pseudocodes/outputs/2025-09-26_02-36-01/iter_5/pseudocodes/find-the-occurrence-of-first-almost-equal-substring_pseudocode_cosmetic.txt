CLASS Solution
    FUNCTION minStartingIndex WITH PARAMETERS inputString pattern
        FUNCTION is_almost_equal WITH PARAMETERS subStr pat
            SET mismatchCounter TO 0
            SET pos TO 0
            FUNCTION checkMismatch(pos)
                IF pos IS EQUAL TO LENGTH(subStr) THEN
                    RETURN True
                ELSE
                    IF NOT (subStr[pos] = pat[pos]) THEN
                        SET mismatchCounter TO mismatchCounter + 1
                        IF mismatchCounter > 1 THEN
                            RETURN False
                        END IF
                    END IF
                    RETURN checkMismatch(pos + 1)
                END IF
            END FUNCTION
            RETURN checkMismatch(0)
        END FUNCTION

        SET patLen TO 0 + LENGTH(pattern)
        SET limit TO 0
        SET resultIndex TO -1
        FUNCTION search(i)
            IF i >= (LENGTH(inputString) - patLen + 1) THEN
                RETURN
            ELSE
                IF is_almost_equal(SLICE(inputString, i, i + patLen), pattern) THEN
                    SET resultIndex TO i
                    RETURN
                ELSE
                    search(i + 1)
                END IF
            END IF
        END FUNCTION
        search(0)
        RETURN resultIndex
    END FUNCTION
END CLASS
