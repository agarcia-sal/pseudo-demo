CLASS Solution
    FUNCTION minOperations(nums)
        SET count_ops TO 0
        SET toggle TO 0

        FUNCTION process(index)
            IF index >= LENGTH(nums) THEN
                RETURN count_ops
            END IF

            SET val_to_check TO 0
            IF (toggle % 2) == 0 THEN
                SET val_to_check TO nums[index]
            ELSE
                SET val_to_check TO 1 - nums[index]
            END IF

            IF val_to_check == 0 THEN
                SET count_ops TO count_ops + 1
                SET toggle TO toggle + 1
            END IF

            RETURN process(index + 1)
        END FUNCTION

        SET result TO process(0)
        RETURN result
    END FUNCTION
END CLASS
