CLASS Solution  
    FUNCTION minimumDifference(nums k)  
        FUNCTION bitwise_or_of_subarray(s e)  
            SET acc TO 0  
            SET idx TO s  
            WHILE idx LESS THAN OR EQUAL TO e DO  
                SET acc TO acc BITWISE_OR nums AT idx  
                SET idx TO idx PLUS 1  
            END WHILE  
            SET output TO acc  
            RETURN output  
        END FUNCTION  

        SET length_nums TO LENGTH OF nums  
        SET best_diff TO (1 SHIFT_LEFT 31) TIMES 2  // large number instead of infinity  

        SET start_index TO 0  
        WHILE start_index LESS THAN length_nums DO  
            SET aggregate_or TO 0  
            SET end_index TO start_index  
            WHILE end_index LESS THAN length_nums DO  
                SET temp_or TO aggregate_or BITWISE_OR nums AT end_index  
                SET aggregate_or TO temp_or  
                SET difference TO k MINUS temp_or  
                IF difference LESS THAN 0 THEN  
                    SET difference TO difference TIMES -1  
                END IF  

                IF difference LESS THAN best_diff THEN  
                    SET best_diff TO difference  
                END IF  

                IF best_diff EQUALS 0 THEN  
                    SET final_result TO 0  
                    RETURN final_result  
                END IF  

                SET end_index TO end_index PLUS 1  
            END WHILE  
            SET start_index TO start_index PLUS 1  
        END WHILE  

        SET final_result TO best_diff  
        RETURN final_result  
    END FUNCTION  
END CLASS
