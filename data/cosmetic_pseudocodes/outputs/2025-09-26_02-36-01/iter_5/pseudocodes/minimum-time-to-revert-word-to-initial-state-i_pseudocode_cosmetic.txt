CLASS Solution
    FUNCTION minimumTimeToInitialState(word PARAMETER alpha)
        SET beta TO LENGTH(word)
        SET gamma TO 1
        FUNCTION checkCondition(delta)
            RETURN (delta * alpha >= beta) OR (SUBSTRING(word, delta * alpha, beta) == SUBSTRING(word, 1, beta - delta * alpha))
        END FUNCTION
        FUNCTION recursiveSearch(epsilon)
            IF checkCondition(epsilon) THEN
                RETURN epsilon
            ELSE
                RETURN recursiveSearch(epsilon + 1)
            END IF
        END FUNCTION
        SET result TO recursiveSearch(gamma)
        RETURN result
    END FUNCTION
END CLASS
