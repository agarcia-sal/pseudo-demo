CLASS Solution
    FUNCTION distanceSum WITH PARAMETERS m n k
        LET divisor = (1000000007)
        LET rowsSquared = m * m
        LET colsSquared = n * n
        LET rowFactorNumerator = rowsSquared * (m - 1) * (2 * m - 1)
        LET rowFactorDenominator = 6
        LET rowContribution = (rowFactorNumerator) / (rowFactorDenominator) * (colsSquared)
        
        LET colFactorNumerator = colsSquared * (n - 1) * (2 * n - 1)
        LET colFactorDenominator = 6
        LET colContribution = (colFactorNumerator) / (colFactorDenominator) * (rowsSquared)
        
        LET totalPoints = m * n
        LET selectionCount = k - 2
        LET arrangementContribution = 1
        IF (selectionCount < 0) THEN
            LET arrangementContribution = 0
        ELSE
            LET numerator = 1
            LET denominator = 1
            LET i = 0
            WHILE (i < selectionCount)
                LET numerator = numerator * (totalPoints - 2 - i)
                LET denominator = denominator * (selectionCount - i)
                SET i = i + 1
            END WHILE
            LET arrangementContribution = numerator / denominator
        END IF
        
        LET combinedDistance = (rowContribution + colContribution) * arrangementContribution
        LET result = combinedDistance % divisor
        RETURN result
    END FUNCTION
END CLASS
