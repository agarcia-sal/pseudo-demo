CLASS Solution
	FUNCTION maximumPrimeDifference(nums)
		DEFINE FUNCTION helper_find_indices(current_pos, current_first, current_last)
			IF current_pos GREATER THAN OR EQUAL TO LENGTH(nums) THEN
				RETURN current_last - current_first
			ELSE
				DEFINE current_element AS nums[current_pos]
				DEFINE prime_values AS SET {two, three, five, seven, eleven, thirteen, seventeen, nineteen, twenty three, twenty nine, thirty one, thirty seven, forty one, forty three, forty seven, fifty three, fifty nine, sixty one, sixty seven, seventy one, seventy three, seventy nine, eighty three, eighty nine, ninety seven}
				IF (prime_values CONTAINS current_element) THEN
					IF current_first EQUALS -1 THEN
						DEFINE updated_first AS current_pos
					ELSE
						DEFINE updated_first AS current_first
					END IF
					DEFINE updated_last AS current_pos
				ELSE
					DEFINE updated_first AS current_first
					DEFINE updated_last AS current_last
				END IF
				RETURN helper_find_indices(current_pos + 1, updated_first, updated_last)
			END IF
		END DEFINE
		DEFINE res AS helper_find_indices(0, -1, -1)
		RETURN res
	END FUNCTION
END CLASS
