CLASS Solution
    FUNCTION flowerGame(n, m)
        DEFINE function halfCeil(x)
            RETURN (x + 1) / 2
        END FUNCTION

        DEFINE function halfFloor(x)
            RETURN x / 2
        END FUNCTION

        LET alpha = halfCeil(n)
        LET beta = halfFloor(n)
        LET gamma = halfCeil(m)
        LET delta = halfFloor(m)

        LET part1 = alpha * delta
        LET part2 = beta * gamma

        LET result_sum = 0
        LET temp_sum = part1

        LET accumulator = temp_sum + part2
        result_sum = accumulator

        LET output_value = result_sum
        RETURN output_value
    END FUNCTION
END CLASS
