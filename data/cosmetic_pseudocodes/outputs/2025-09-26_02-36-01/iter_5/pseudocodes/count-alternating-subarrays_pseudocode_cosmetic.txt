CLASS Solution
    FUNCTION countAlternatingSubarrays(nums)
        SET total_elements TO LENGTH OF nums
        SET final_result TO zero
        IF NOT (total_elements GREATER THAN zero) THEN
            SET final_result TO zero
        ELSE
            SET cumulative_count TO total_elements
            SET streak_length TO 1
            PROCEDURE recursive_increment(index)
                IF index GREATER OR EQUAL TO total_elements THEN
                    RETURN
                ELSE
                    SET current_value TO nums[index]
                    SET previous_value TO nums[index - 1]
                    IF NOT (current_value EQUALS previous_value) THEN
                        SET extended_length TO streak_length + 1
                    ELSE
                        SET extended_length TO 1
                    END IF
                    SET streak_length TO extended_length
                    SET cumulative_count TO cumulative_count + (streak_length - 1)
                    CALL recursive_increment(index + 1)
                END IF
            END PROCEDURE
            CALL recursive_increment(1)
            SET final_result TO cumulative_count
        END IF
        RETURN final_result
    END FUNCTION
END CLASS
