CLASS Solution
    FUNCTION numberOfRightTriangles(grid)
        LET rowLength BE LENGTH grid
        LET colLength BE LENGTH grid[0]
        LET accumulator BE 0
        LET xIndex BE 0
        WHILE xIndex < rowLength
            LET yIndex BE 0
            WHILE yIndex < colLength
                IF (NOT (grid[xIndex][yIndex] = 0)) THEN
                    LET horizontalSum BE 0
                    LET colIter BE 0
                    WHILE colIter < colLength
                        IF colIter <> yIndex THEN
                            LET horizontalSum BE horizontalSum + grid[xIndex][colIter]
                        END IF
                        LET colIter BE colIter + 1
                    END WHILE

                    LET verticalSum BE 0
                    LET rowIter BE 0
                    WHILE rowIter < rowLength
                        IF rowIter <> xIndex THEN
                            LET verticalSum BE verticalSum + grid[rowIter][yIndex]
                        END IF
                        LET rowIter BE rowIter + 1
                    END WHILE

                    LET incrementValue BE horizontalSum * verticalSum
                    LET accumulator BE accumulator + incrementValue
                END IF
                LET yIndex BE yIndex + 1
            END WHILE
            LET xIndex BE xIndex + 1
        END WHILE
        LET result TO accumulator
        RETURN result
    END FUNCTION
END CLASS
