CLASS Solution
	FUNCTION minMovesToCaptureTheQueen(p q r s t u)
		SET result TO 2
		IF p = t THEN
			IF r ≠ p THEN
				SET result TO 1
			ELSE
				IF ((q < s AND s < u) OR (u < s AND s < q)) THEN
					SET result TO 2
				ELSE
					SET result TO 1
				END IF
			END IF
		ELSE
			IF q = u THEN
				IF s ≠ q THEN
					SET result TO 1
				ELSE
					IF ((p < r AND r < t) OR (t < r AND r < p)) THEN
						SET result TO 2
					ELSE
						SET result TO 1
					END IF
				END IF
			ELSE
				SET sumCR TO r + s
				SET sumTU TO t + u
				IF sumCR = sumTU THEN
					IF (p + q) ≠ sumCR THEN
						SET result TO 1
					ELSE
						SET exprLeft1 TO (r < p AND p < t AND s < q AND q < u)
						SET exprLeft2 TO (t < p AND p < r AND u < q AND q < s)
						IF (exprLeft1 OR exprLeft2) THEN
							SET result TO 2
						ELSE
							SET result TO 1
						END IF
					END IF
				ELSE
					SET diffCR TO r - s
					SET diffTU TO t - u
					IF diffCR = diffTU THEN
						IF (p - q) ≠ diffCR THEN
							SET result TO 1
						ELSE
							SET condA TO (r < p AND p < t AND s > q AND q > u)
							SET condB TO (t < p AND p < r AND u > q AND q > s)
							IF (condA OR condB) THEN
								SET result TO 2
							ELSE
								SET result TO 1
							END IF
						END IF
					END IF
				END IF
			END IF
		END IF
		RETURN result
	END FUNCTION
END CLASS
