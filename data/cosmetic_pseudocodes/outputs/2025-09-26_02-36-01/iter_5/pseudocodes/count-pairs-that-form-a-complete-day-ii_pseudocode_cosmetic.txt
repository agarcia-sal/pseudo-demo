CLASS Solution
    FUNCTION countCompleteDayPairs WITH parameter hours
        SET frequency_map TO empty map with default zero
        SET total_pairs TO 0
        DEFINE FUNCTION processIndex WITH parameter idx
            IF NOT (idx < LENGTH OF hours) THEN
                RETURN
            END IF
            SET current_value TO hours[idx]
            SET mod_val TO (current_value - (current_value / 24) * 24)
            SET complement_val TO (24 - mod_val) - ((24 - mod_val) / 24) * 24
            SET total_pairs TO total_pairs + frequency_map[complement_val]
            SET frequency_map[mod_val] TO frequency_map[mod_val] + 1
            CALL processIndex WITH idx + 1
        END FUNCTION
        CALL processIndex WITH 0
        SET final_result TO total_pairs
        RETURN final_result
    END FUNCTION
END CLASS
