CLASS Solution
    FUNCTION maxScore(nums)
        LET lengthVal BE LENGTH(nums)
        LET dpArray BE [0 FOR _ IN RANGE(lengthVal)]

        FUNCTION innerLoop(targetIndex, currentIndex, currentMax)
            IF currentIndex >= targetIndex THEN
                RETURN currentMax
            ELSE
                LET candidateScore BE dpArray[currentIndex] + (targetIndex - currentIndex) * nums[targetIndex]
                LET updatedMax BE currentMax
                IF candidateScore > currentMax THEN
                    updatedMax = candidateScore
                END IF
                RETURN innerLoop(targetIndex, currentIndex + 1, updatedMax)
            END IF
        END FUNCTION

        FUNCTION outerLoop(i)
            IF i >= lengthVal THEN
                RETURN
            ELSE
                dpArray[i] = innerLoop(i, 0, 0)
                outerLoop(i + 1)
            END IF
        END FUNCTION

        outerLoop(1)
        LET resultValue BE dpArray[lengthVal - 1]
        RETURN resultValue
    END FUNCTION
END CLASS
