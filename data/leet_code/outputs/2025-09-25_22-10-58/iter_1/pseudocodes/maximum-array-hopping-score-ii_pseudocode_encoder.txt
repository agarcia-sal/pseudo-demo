CLASS Solution  
    FUNCTION maxScore(nums)  
        SET n TO the LENGTH OF nums  
        SET dp TO a list of zeros with LENGTH n  
        FOR i FROM n MINUS two TO zero STEP MINUS one  
            SET max_score TO zero  
            FOR j FROM i PLUS one TO n MINUS one  
                SET score TO (j MINUS i) MULTIPLIED BY element at position j of nums  
                IF max_score LESS THAN score PLUS element at position j of dp  
                    SET max_score TO score PLUS element at position j of dp  
                END IF  
            END FOR  
            SET element at position i of dp TO max_score  
        END FOR  
        RETURN element at position zero of dp  
    END FUNCTION  
END CLASS
