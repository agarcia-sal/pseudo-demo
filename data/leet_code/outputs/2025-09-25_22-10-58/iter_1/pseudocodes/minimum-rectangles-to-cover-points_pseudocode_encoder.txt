CLASS Solution  
    FUNCTION minRectanglesToCoverPoints(points, w)  
        SORT points IN ascending order BASED ON the first element of each point  
        SET rectangles TO zero  
        SET current_x TO negative one  
        FOR each point x and y IN points  
            IF x IS GREATER THAN current_x THEN  
                SET current_x TO x PLUS w  
                INCREMENT rectangles BY one  
            END IF  
        END FOR  
        RETURN rectangles  
    END FUNCTION  
END CLASS
