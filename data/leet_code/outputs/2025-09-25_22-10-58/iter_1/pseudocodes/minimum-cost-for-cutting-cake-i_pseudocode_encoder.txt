CLASS Solution  
    FUNCTION minimumCost(m n horizontalCut verticalCut)  
        SORT horizontalCut IN DESCENDING ORDER  
        SORT verticalCut IN DESCENDING ORDER  
        SET ans TO zero  
        SET i TO zero  
        SET j TO zero  
        SET h TO one  
        SET v TO one  
        WHILE i LESS THAN m MINUS one OR j LESS THAN n MINUS one  
            IF j EQUALS n MINUS one OR (i LESS THAN m MINUS one AND element at position i of horizontalCut GREATER THAN element at position j of verticalCut)  
                INCREMENT ans BY element at position i of horizontalCut MULTIPLIED BY v  
                INCREMENT h BY one  
                INCREMENT i BY one  
            ELSE  
                INCREMENT ans BY element at position j of verticalCut MULTIPLIED BY h  
                INCREMENT v BY one  
                INCREMENT j BY one  
            END IF  
        END WHILE  
        RETURN ans  
    END FUNCTION  
END CLASS
