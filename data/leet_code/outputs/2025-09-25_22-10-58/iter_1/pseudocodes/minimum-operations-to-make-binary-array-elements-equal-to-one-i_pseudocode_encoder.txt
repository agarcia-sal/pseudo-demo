CLASS Solution  
    FUNCTION minOperations(nums)  
        SET n TO the LENGTH OF nums  
        SET operations TO zero  
        FOR i FROM zero TO n MINUS two MINUS one  
            IF element at position i of nums EQUALS zero  
                SET element at position i of nums TO one MINUS element at position i of nums  
                SET element at position i PLUS one of nums TO one MINUS element at position i PLUS one of nums  
                SET element at position i PLUS two of nums TO one MINUS element at position i PLUS two of nums  
                INCREMENT operations BY one  
            END IF  
        END FOR  
        IF element at position n MINUS one of nums EQUALS zero OR element at position n MINUS two of nums EQUALS zero  
            RETURN minus one  
        END IF  
        RETURN operations  
    END FUNCTION  
END CLASS
