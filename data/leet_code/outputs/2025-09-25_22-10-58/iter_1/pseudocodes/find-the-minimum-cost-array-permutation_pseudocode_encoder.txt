CLASS Solution  
    FUNCTION findPermutation(nums)  
        FUNCTION dfs(mask, pre)  
            IF mask EQUALS the quantity one LEFT SHIFTED BY the LENGTH OF nums MINUS one THEN  
                RETURN the absolute value OF pre MINUS element at position zero OF nums  
            END IF  
            SET res TO infinity  
            FOR cur FROM zero TO the LENGTH OF nums MINUS one  
                IF the quantity mask RIGHT SHIFTED BY cur AND one EQUALS zero THEN  
                    SET candidate_value TO the absolute value OF pre MINUS element at position cur OF nums PLUS dfs the quantity mask OR the quantity one LEFT SHIFTED BY cur AND cur  
                    IF candidate_value LESS THAN res THEN  
                        SET res TO candidate_value  
                    END IF  
                END IF  
            END FOR  
            RETURN res  
        END FUNCTION  

        FUNCTION g(mask, pre)  
            APPEND pre TO ans  
            IF mask EQUALS the quantity one LEFT SHIFTED BY the LENGTH OF nums MINUS one THEN  
                RETURN  
            END IF  
            SET res TO dfs WITH PARAMETERS mask AND pre  
            FOR cur FROM zero TO the LENGTH OF nums MINUS one  
                IF the quantity mask RIGHT SHIFTED BY cur AND one EQUALS zero THEN  
                    SET candidate_value TO the absolute value OF pre MINUS element at position cur OF nums PLUS dfs the quantity mask OR the quantity one LEFT SHIFTED BY cur AND cur  
                    IF candidate_value EQUALS res THEN  
                        CALL g WITH PARAMETERS the quantity mask OR the quantity one LEFT SHIFTED BY cur AND cur  
                        BREAK  
                    END IF  
                END IF  
            END FOR  
        END FUNCTION  

        SET n TO the LENGTH OF nums  
        SET ans TO an empty list  
        CALL g WITH PARAMETERS the quantity one LEFT SHIFTED BY zero AND zero  
        RETURN ans  
    END FUNCTION  
END CLASS
