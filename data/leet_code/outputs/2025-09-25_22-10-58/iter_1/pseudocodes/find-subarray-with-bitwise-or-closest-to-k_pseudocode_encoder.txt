CLASS Solution  
    FUNCTION minimumDifference(nums k)  
        FUNCTION bitwise_or_of_subarray(start end)  
            SET result TO zero  
            FOR i FROM start TO end  
                SET result TO result OR element at position i of nums  
            END FOR  
            RETURN result  
        END FUNCTION  

        SET n TO the LENGTH OF nums  
        SET min_diff TO infinity  

        FOR start FROM zero TO n MINUS one  
            SET current_or TO zero  
            FOR end FROM start TO n MINUS one  
                SET current_or TO current_or OR element at position end of nums  
                SET diff TO the ABSOLUTE VALUE OF k MINUS current_or  
                IF diff LESS THAN min_diff  
                    SET min_diff TO diff  
                END IF  
                IF diff EQUALS zero  
                    RETURN zero  
                END IF  
            END FOR  
        END FOR  

        RETURN min_diff  
    END FUNCTION  
END CLASS
