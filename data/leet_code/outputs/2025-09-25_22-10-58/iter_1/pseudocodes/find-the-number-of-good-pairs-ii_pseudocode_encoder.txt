CLASS Solution  
    FUNCTION numberOfPairs(nums1, nums2, k)  
        SET count_nums2 TO result of Counter FUNCTION called with nums2  
        SET good_pairs TO zero  
        FOR each num1 IN nums1  
            FOR each element num2 and associated count IN count_nums2 items  
                IF remainder of num1 DIVIDED BY the product of num2 MULTIPLIED BY k EQUALS zero  
                    INCREMENT good_pairs BY count  
                END IF  
            END FOR  
        END FOR  
        RETURN good_pairs  
    END FUNCTION  
END CLASS
