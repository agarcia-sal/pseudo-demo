CLASS Solution  
    FUNCTION beautifulIndices(s parameter, a parameter, b parameter, k parameter)  
        SET indices_a TO an empty list  
        SET length_of_s TO the LENGTH OF s  
        SET length_of_a TO the LENGTH OF a  
        FOR index FROM zero TO length_of_s MINUS length_of_a PLUS one  
            IF the SUBSTRING of s FROM position index TO position index PLUS length_of_a MINUS one EQUALS a  
                APPEND index TO indices_a  
            END IF  
        END FOR  
        SET indices_b TO an empty list  
        SET length_of_b TO the LENGTH OF b  
        FOR index FROM zero TO length_of_s MINUS length_of_b PLUS one  
            IF the SUBSTRING of s FROM position index TO position index PLUS length_of_b MINUS one EQUALS b  
                APPEND index TO indices_b  
            END IF  
        END FOR  
        SET beautiful_indices TO an empty list  
        FOR each element i IN indices_a  
            FOR each element j IN indices_b  
                IF the ABSOLUTE VALUE OF i MINUS j LESS THAN OR EQUAL TO k  
                    APPEND i TO beautiful_indices  
                    BREAK the inner loop  
                END IF  
            END FOR  
        END FOR  
        RETURN beautiful_indices  
    END FUNCTION  
END CLASS
