CLASS Solution  
    FUNCTION numberOfAlternatingGroups(colors k)  
        IF k LESS THAN three  
            RETURN zero  
        END IF  
        
        SET n TO the LENGTH OF colors  
        SET extended_colors TO the CONCATENATION OF colors AND the SUBSEQUENCE OF colors FROM position zero TO position k MINUS one  
        
        SET count TO zero  
        
        FOR i FROM zero TO n MINUS one  
            SET is_alternating TO true  
            FOR j FROM one TO k MINUS one  
                IF the element at position i PLUS j OF extended_colors EQUALS the element at position i PLUS j MINUS one OF extended_colors  
                    SET is_alternating TO false  
                    BREAK the inner loop  
                END IF  
            END FOR  
            IF is_alternating EQUALS true  
                INCREMENT count BY one  
            END IF  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
