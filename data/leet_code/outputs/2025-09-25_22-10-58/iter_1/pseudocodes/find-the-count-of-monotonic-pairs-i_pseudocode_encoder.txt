CLASS Solution  
    FUNCTION countOfPairs(nums)  
        SET MOD TO the value one billion PLUS the value seven  
        SET n TO the LENGTH OF nums  
        SET max_val TO the MAXIMUM value IN nums  
        
        SET dp TO a list of length n WHERE each element is a list of length max_val PLUS one WHERE each element is a list of length max_val PLUS one FILLED WITH zero  
        
        FOR each integer j FROM zero TO the value at position zero of nums INCLUSIVE  
            SET k TO the value at position zero of nums MINUS j  
            SET the element at position zero of dp AT position j AT position k TO one  
        END FOR  
        
        FOR each integer i FROM one TO n MINUS one INCLUSIVE  
            FOR each integer j FROM zero TO the value at position i of nums INCLUSIVE  
                SET k TO the value at position i of nums MINUS j  
                FOR each integer j_prev FROM zero TO j INCLUSIVE  
                    FOR each integer k_prev FROM k TO max_val INCLUSIVE  
                        SET the element at position i of dp AT position j AT position k TO the element at position i of dp AT position j AT position k PLUS the element at position i MINUS one of dp AT position j_prev AT position k_prev  
                        SET the element at position i of dp AT position j AT position k TO the element at position i of dp AT position j AT position k MODULO MOD  
                    END FOR  
                END FOR  
            END FOR  
        END FOR  
        
        SET result TO zero  
        FOR each integer j FROM zero TO max_val INCLUSIVE  
            FOR each integer k FROM zero TO max_val INCLUSIVE  
                IF j PLUS k EQUALS the element at the last position of nums THEN  
                    SET result TO result PLUS the element at the last position of dp AT position j AT position k  
                    SET result TO result MODULO MOD  
                END IF  
            END FOR  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
