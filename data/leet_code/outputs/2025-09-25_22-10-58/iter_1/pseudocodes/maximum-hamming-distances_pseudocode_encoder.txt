CLASS Solution  
    FUNCTION maxHammingDistances(nums LIST OF INTEGER m INTEGER) RETURNS LIST OF INTEGER  
        SET binary_nums TO an empty list  
        FOR each num IN nums  
            SET binary_string TO the binary representation of num WITH leading zeros TO make length equal TO m  
            APPEND binary_string TO binary_nums  
        END FOR  
        
        SET answer TO an empty list  
        
        FUNCTION hamming_distance(bin1 STRING bin2 STRING) RETURNS INTEGER  
            SET distance TO zero  
            FOR each position FROM the first TO the length OF bin1  
                IF the element at position OF bin1 NOT EQUALS the element at position OF bin2  
                    INCREMENT distance BY one  
                END IF  
            END FOR  
            RETURN distance  
        END FUNCTION  
        
        FOR index i FROM zero TO the length OF nums MINUS one  
            SET max_dist TO zero  
            FOR index j FROM zero TO the length OF nums MINUS one  
                IF i NOT EQUALS j  
                    SET dist TO the result OF hamming_distance WITH binary_nums at position i AND binary_nums at position j  
                    IF dist GREATER THAN max_dist  
                        SET max_dist TO dist  
                    END IF  
                END IF  
            END FOR  
            APPEND max_dist TO answer  
        END FOR  
        
        RETURN answer  
    END FUNCTION  
END CLASS
