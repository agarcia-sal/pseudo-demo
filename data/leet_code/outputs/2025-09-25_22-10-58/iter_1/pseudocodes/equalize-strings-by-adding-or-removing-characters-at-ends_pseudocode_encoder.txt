CLASS Solution  
    FUNCTION minOperations(initial parameter TO string, target parameter TO string) RETURNS integer  
        SET m TO the LENGTH OF initial  
        SET n TO the LENGTH OF target  
        SET dp TO a two dimensional list with dimensions m PLUS one by n PLUS one filled with zeros  
        SET max_lcs TO zero  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the element at position i MINUS one of initial EQUALS the element at position j MINUS one of target THEN  
                    SET the element at position i and position j of dp TO the element at position i MINUS one and position j MINUS one of dp PLUS one  
                    IF max_lcs LESS THAN the element at position i and position j of dp THEN  
                        SET max_lcs TO the element at position i and position j of dp  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN m PLUS n MINUS two MULTIPLIED BY max_lcs  
    END FUNCTION  
END CLASS
