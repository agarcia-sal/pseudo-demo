CLASS Solution  
    FUNCTION maximumTotalCost(nums)  
        SET n TO the LENGTH OF nums  
        IF n EQUALS one  
            RETURN element at position one of nums  
        END IF  
        
        SET dp TO a list of zeros with the LENGTH OF nums  
        
        SET element at position n MINUS one of dp TO element at position n MINUS one of nums  
        
        FOR i FROM n MINUS two DOWN TO zero  
            SET current_cost TO element at position i of nums  
            IF current_cost GREATER THAN element at position i PLUS one of dp  
                SET element at position i of dp TO current_cost  
            ELSE  
                SET element at position i of dp TO element at position i PLUS one of dp PLUS current_cost  
            END IF  
            
            FOR j FROM i PLUS one TO n MINUS one  
                SET alternating_sign TO NEGATIVE one RAISED TO the POWER OF j MINUS i  
                INCREMENT current_cost BY element at position j of nums MULTIPLIED BY alternating_sign  
                
                IF j PLUS one LESS THAN n  
                    IF element at position i of dp LESS THAN current_cost PLUS element at position j PLUS one of dp  
                        SET element at position i of dp TO current_cost PLUS element at position j PLUS one of dp  
                    END IF  
                ELSE  
                    IF element at position i of dp LESS THAN current_cost  
                        SET element at position i of dp TO current_cost  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position zero of dp  
    END FUNCTION  
END CLASS
