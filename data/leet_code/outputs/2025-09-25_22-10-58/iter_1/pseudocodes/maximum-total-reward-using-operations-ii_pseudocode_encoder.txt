CLASS Solution  
    FUNCTION maxTotalReward(rewardValues)  
        SET nums TO the SORTED SET OF elements IN rewardValues  
        SET f TO one  
        FOR each v IN nums  
            SET f TO f OR the BITWISE AND of f AND the DIFFERENCE of the VALUE one SHIFTED LEFT BY v and one, SHIFTED LEFT BY v  
        END FOR  
        RETURN the BIT LENGTH OF f MINUS one  
    END FUNCTION  
END CLASS
