CLASS Solution  
    FUNCTION minEnd(n x)  
        FUNCTION canConstruct(max_val)  
            SET current TO x  
            SET count TO one  
            WHILE current LESS THAN max_val  
                SET current TO current PLUS one  
                IF the binary AND of current AND x EQUALS x  
                    INCREMENT count BY one  
                    IF count EQUALS n  
                        RETURN True  
                    END IF  
                END IF  
            END WHILE  
            RETURN count EQUALS n  
        END FUNCTION  
        
        SET left TO x  
        SET right TO the product of two AND ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten  
        WHILE left LESS THAN right  
            SET mid TO the integer division of the sum of left AND right BY two  
            IF canConstruct(mid)  
                SET right TO mid  
            ELSE  
                SET left TO left PLUS one  
            END IF  
        END WHILE  
        
        RETURN left  
    END FUNCTION  
END CLASS
