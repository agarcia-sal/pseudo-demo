CLASS Solution  
    FUNCTION minOperations(k)  
        SET min_operations TO a value representing infinity  
        FOR x FROM one TO the integer part of the square root of k PLUS one  
            SET y TO the quotient of the sum of k PLUS x MINUS one DIVIDED BY x using integer division  
            SET operations TO the sum of x MINUS one PLUS y MINUS one  
            IF operations LESS THAN min_operations THEN  
                SET min_operations TO operations  
            END IF  
        END FOR  
        RETURN min_operations  
    END FUNCTION  
END CLASS
