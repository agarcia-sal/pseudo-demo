CLASS Solution  
    FUNCTION nonSpecialCount(l r)  
        FUNCTION is_prime(num)  
            IF num LESS THAN OR EQUAL TO one  
                RETURN false  
            END IF  
            IF num LESS THAN OR EQUAL TO three  
                RETURN true  
            END IF  
            IF num MODULO two EQUALS zero OR num MODULO three EQUALS zero  
                RETURN false  
            END IF  
            SET i TO five  
            WHILE i MULTIPLIED BY i LESS THAN OR EQUAL TO num  
                IF num MODULO i EQUALS zero OR num MODULO i PLUS two EQUALS zero  
                    RETURN false  
                END IF  
                INCREMENT i BY six  
            END WHILE  
            RETURN true  
        END FUNCTION  
  
        SET start_prime TO the smallest integer greater than or equal to the square root of l  
        SET end_prime TO the greatest integer less than or equal to the square root of r  
  
        SET special_count TO zero  
  
        FOR num FROM start_prime TO end_prime  
            IF is_prime(num) EQUALS true  
                INCREMENT special_count BY one  
            END IF  
        END FOR  
  
        SET total_count TO r MINUS l PLUS one  
  
        RETURN total_count MINUS special_count  
    END FUNCTION  
END CLASS
