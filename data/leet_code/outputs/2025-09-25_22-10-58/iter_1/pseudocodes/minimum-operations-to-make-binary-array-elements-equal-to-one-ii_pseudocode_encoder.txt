CLASS Solution  
    FUNCTION minOperations(nums)  
        SET operations TO zero  
        SET flip TO zero  
        FOR each num IN nums  
            IF the remainder of flip DIVIDED BY two EQUALS zero  
                SET current_num TO num  
            ELSE  
                SET current_num TO one MINUS num  
            END IF  
            IF current_num EQUALS zero  
                INCREMENT operations BY one  
                INCREMENT flip BY one  
            END IF  
        END FOR  
        RETURN operations  
    END FUNCTION  
END CLASS
