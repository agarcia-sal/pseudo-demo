CLASS Solution  
    FUNCTION countAlternatingSubarrays(nums)  
        SET n TO the LENGTH OF nums  
        IF n EQUALS zero  
            RETURN zero  
        END IF  
        SET count TO n  
        SET current_length TO one  
        FOR i FROM one TO n MINUS one  
            IF element at position i of nums NOT EQUALS element at position i MINUS one of nums  
                INCREMENT current_length BY one  
            ELSE  
                SET current_length TO one  
            END IF  
            INCREMENT count BY current_length MINUS one  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
