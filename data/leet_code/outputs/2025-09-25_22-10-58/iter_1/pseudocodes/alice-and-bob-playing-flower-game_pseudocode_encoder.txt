CLASS Solution  
    FUNCTION flowerGame(n, m)  
        SET odd_n TO n PLUS one DIVIDED BY two  
        SET even_n TO n DIVIDED BY two  
        SET odd_m TO m PLUS one DIVIDED BY two  
        SET even_m TO m DIVIDED BY two  
        SET total_pairs TO odd_n MULTIPLIED BY even_m PLUS even_n MULTIPLIED BY odd_m  
        RETURN total_pairs  
    END FUNCTION  
END CLASS
