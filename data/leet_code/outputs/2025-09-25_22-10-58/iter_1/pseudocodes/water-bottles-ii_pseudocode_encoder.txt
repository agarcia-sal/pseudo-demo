CLASS Solution  
    FUNCTION maxBottlesDrunk(numBottles numBottles numExchange numExchange)  
        SET total_drunk TO zero  
        SET empty_bottles TO zero  
        WHILE numBottles GREATER THAN zero  
            INCREMENT total_drunk BY numBottles  
            INCREMENT empty_bottles BY numBottles  
            SET numBottles TO zero  
            WHILE empty_bottles GREATER THAN OR EQUAL TO numExchange  
                DECREMENT empty_bottles BY numExchange  
                INCREMENT numBottles BY one  
                INCREMENT numExchange BY one  
            END WHILE  
        END WHILE  
        RETURN total_drunk  
    END FUNCTION  
END CLASS
