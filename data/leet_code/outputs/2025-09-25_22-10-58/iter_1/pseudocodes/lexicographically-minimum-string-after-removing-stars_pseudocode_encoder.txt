CLASS Solution  
    FUNCTION clearStars(s)  
        SET stack TO an empty list  
        FOR each char IN s  
            IF char EQUALS an asterisk character  
                IF the LENGTH OF stack IS GREATER THAN zero  
                    REMOVE the last element FROM stack  
                END IF  
            ELSE  
                APPEND char TO stack  
            END IF  
        END FOR  
        SET result_string TO concatenation of all elements IN stack  
        RETURN result_string  
    END FUNCTION  
END CLASS
