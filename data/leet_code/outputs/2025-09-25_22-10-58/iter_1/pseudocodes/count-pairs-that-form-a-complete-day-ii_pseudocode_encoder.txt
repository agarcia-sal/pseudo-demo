CLASS Solution  
    FUNCTION countCompleteDayPairs WITH parameter hours  
        SET remainder_count TO an empty mapping with default value zero for any key  
        SET pairs TO zero  
        FOR each element hour IN hours  
            SET remainder TO element at position hour of hours MODULO twenty-four  
            SET complement TO twenty-four MINUS remainder MODULO twenty-four  
            INCREMENT pairs BY the value associated with complement in remainder_count  
            INCREMENT the value associated with remainder in remainder_count BY one  
        END FOR  
        RETURN pairs  
    END FUNCTION  
END CLASS
