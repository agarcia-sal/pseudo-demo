CLASS Solution  
    FUNCTION countDays(days, meetings)  
        SORT meetings BY the first element of each meeting sublist in ascending order  
        SET current_day TO one  
        SET available_days TO zero  
        FOR each meeting IN meetings  
            SET start TO the first element of meeting  
            SET end TO the second element of meeting  
            IF current_day LESS THAN start THEN  
                INCREMENT available_days BY start MINUS current_day  
            END IF  
            SET current_day TO the greater value BETWEEN current_day AND end PLUS one  
        END FOR  
        IF current_day LESS THAN OR EQUAL TO days THEN  
            INCREMENT available_days BY days MINUS current_day PLUS one  
        END IF  
        RETURN available_days  
    END FUNCTION  
END CLASS
