CLASS Solution  
    FUNCTION sumOfPower(nums parameter, k parameter)  
        SET MOD TO one followed by nine zeros PLUS seven  
        SET n TO the LENGTH OF nums  
        SET dp TO a list of lists each containing zero repeated k PLUS one times repeated n PLUS one times  
        SET element at position zero of element at position zero of dp TO one  
        
        FOR i FROM one TO n  
            FOR j FROM zero TO k  
                SET element at position j of element at position i of dp TO element at position j of element at position i MINUS one of dp  
                IF j GREATER THAN OR EQUAL TO element at position i MINUS one of nums  
                    INCREMENT element at position j of element at position i of dp BY element at position j MINUS element at position i MINUS one of element at position i MINUS one of dp  
                END IF  
                SET element at position j of element at position i of dp TO element at position j of element at position i of dp MODULO MOD  
            END FOR  
        END FOR  
        
        SET total_power TO zero  
        FOR i FROM one TO two RAISED TO THE POWER OF n MINUS one  
            SET current_sum TO zero  
            SET count TO zero  
            FOR j FROM zero TO n MINUS one  
                IF element at position j of the binary representation of i EQUALS one  
                    INCREMENT current_sum BY element at position j of nums  
                    INCREMENT count BY one  
                END IF  
            END FOR  
            IF current_sum EQUALS k  
                INCREMENT total_power BY two RAISED TO THE POWER OF n MINUS count  
                SET total_power TO total_power MODULO MOD  
            END IF  
        END FOR  
        
        RETURN total_power  
    END FUNCTION  
END CLASS
