CLASS Solution  
    FUNCTION findPattern(board, pattern)  
        SET rows TO the LENGTH OF board  
        SET cols TO the LENGTH OF element at position zero of board  
        SET p_rows TO the LENGTH OF pattern  
        SET p_cols TO the LENGTH OF element at position zero of pattern  
        
        FUNCTION matches(r, c)  
            SET char_to_digit TO an empty mapping  
            SET digit_to_char TO an empty mapping  
            
            FOR i FROM zero TO p_rows MINUS one  
                FOR j FROM zero TO p_cols MINUS one  
                    SET p_char TO element at position i of pattern THEN element at position j  
                    SET b_digit TO element at position r PLUS i of board THEN element at position c PLUS j  
                    
                    IF p_char IS a digit  
                        IF the integer value of p_char NOT EQUALS b_digit  
                            RETURN False  
                        END IF  
                    ELSE  
                        IF p_char IS IN char_to_digit  
                            IF the value associated with p_char in char_to_digit NOT EQUALS b_digit  
                                RETURN False  
                            END IF  
                        ELSE  
                            IF b_digit IS IN digit_to_char  
                                RETURN False  
                            END IF  
                            ASSIGN the value b_digit TO char_to_digit at key p_char  
                            ASSIGN the value p_char TO digit_to_char at key b_digit  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
            RETURN True  
        END FUNCTION  
        
        FOR r FROM zero TO rows MINUS p_rows  
            FOR c FROM zero TO cols MINUS p_cols  
                IF matches(r, c)  
                    RETURN a list containing r and c  
                END IF  
            END FOR  
        END FOR  
        
        RETURN a list containing minus one and minus one  
    END FUNCTION  
END CLASS
