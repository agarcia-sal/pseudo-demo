CLASS Solution  
    FUNCTION minimumArrayLength(nums)  
        SET min_val TO the minimum element IN nums  
        SET count_min_val TO the number of occurrences OF min_val IN nums  
        IF count_min_val EQUALS one  
            RETURN one  
        END IF  
        RETURN the quotient of count_min_val PLUS one DIVIDED BY two  
    END FUNCTION  
END CLASS
