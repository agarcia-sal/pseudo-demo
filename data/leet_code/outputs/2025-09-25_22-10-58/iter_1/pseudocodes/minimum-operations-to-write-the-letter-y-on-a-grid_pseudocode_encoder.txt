CLASS Solution  
    FUNCTION minimumOperationsToWriteY(grid)  
        SET n TO the LENGTH OF grid  
        SET center TO the INTEGER DIVISION OF n BY two  
        SET y_cells TO an EMPTY SET  
        
        FOR i FROM zero TO center INCLUSIVE  
            ADD the TUPLE of i AND i TO y_cells  
        END FOR  
        
        FOR i FROM zero TO center INCLUSIVE  
            ADD the TUPLE of i AND the RESULT OF n MINUS i MINUS one TO y_cells  
        END FOR  
        
        FOR i FROM center TO n MINUS one INCLUSIVE  
            ADD the TUPLE of i AND center TO y_cells  
        END FOR  
        
        SET y_count TO a COUNTER OF values at positions specified by y_cells in grid  
        SET non_y_count TO a COUNTER OF values at positions not in y_cells in grid  
        
        SET min_operations TO positive infinity  
        
        FOR y_val FROM zero TO two INCLUSIVE  
            FOR non_y_val FROM zero TO two INCLUSIVE  
                IF y_val NOT EQUALS non_y_val  
                    SET operations TO the SUM of all y_count values MINUS the value in y_count at y_val PLUS the SUM of all non_y_count values MINUS the value in non_y_count at non_y_val  
                    SET min_operations TO the SMALLER OF min_operations AND operations  
                END IF  
            END FOR  
        END FOR  
        
        RETURN min_operations  
    END FUNCTION  
END CLASS
