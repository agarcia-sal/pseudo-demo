CLASS Solution  
    FUNCTION maximumLength(nums k)  
        SET n TO the LENGTH OF nums  
        IF n EQUALS one  
            RETURN one  
        END IF  
        
        SET dp TO a new list with n elements where each element is an empty mapping  
        
        SET max_length TO one  
        
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO i MINUS one  
                SET remainder TO the sum of element at position i of nums PLUS element at position j of nums MODULO k  
                IF remainder IS a key IN the mapping at position j of dp  
                    SET the value at key remainder in the mapping at position i of dp TO the value at key remainder in the mapping at position j of dp PLUS one  
                ELSE  
                    SET the value at key remainder in the mapping at position i of dp TO two  
                END IF  
                IF the value at key remainder in the mapping at position i of dp GREATER THAN max_length  
                    SET max_length TO the value at key remainder in the mapping at position i of dp  
                END IF  
            END FOR  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
