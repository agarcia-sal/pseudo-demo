CLASS Solution  
    FUNCTION canSortArray(nums)  
        FUNCTION count_set_bits(n)  
            RETURN the count of one bits in the binary representation of n  
        END FUNCTION  
        
        SET n TO the LENGTH OF nums  
        SET sorted_nums TO nums sorted in ascending order  
        
        FOR each i FROM zero TO n MINUS one  
            FOR each j FROM zero TO n MINUS two  
                IF the count_set_bits of element at position j of nums EQUALS the count_set_bits of element at position j PLUS one of nums AND element at position j of nums GREATER THAN element at position j PLUS one of nums  
                    SET temp TO element at position j of nums  
                    SET element at position j of nums TO element at position j PLUS one of nums  
                    SET element at position j PLUS one of nums TO temp  
                END IF  
            END FOR  
        END FOR  
        
        RETURN whether nums EQUALS sorted_nums  
    END FUNCTION  
END CLASS
