CLASS Solution  
    FUNCTION maximumPoints(enemyEnergies currentEnergy)  
        SORT enemyEnergies IN ASCENDING ORDER  
        IF currentEnergy LESS THAN element at position zero of enemyEnergies  
            RETURN zero  
        END IF  
        SET ans TO zero  
        FOR i FROM the LENGTH OF enemyEnergies MINUS one TO zero STEP minus one  
            INCREMENT ans BY the INTEGER DIVISION OF currentEnergy BY element at position zero of enemyEnergies  
            SET currentEnergy TO the REMAINDER OF currentEnergy DIVIDED BY element at position zero of enemyEnergies  
            INCREMENT currentEnergy BY element at position i of enemyEnergies  
        END FOR  
        RETURN ans  
    END FUNCTION  
END CLASS
