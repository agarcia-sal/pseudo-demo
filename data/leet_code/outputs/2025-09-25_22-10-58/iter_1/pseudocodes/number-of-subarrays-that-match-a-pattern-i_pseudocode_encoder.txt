CLASS Solution  
    FUNCTION countMatchingSubarrays(nums parameter, pattern parameter)  
        SET n TO the LENGTH OF nums parameter  
        SET m TO the LENGTH OF pattern parameter  
        SET count TO zero  
        FOR i FROM zero TO n MINUS m MINUS one  
            SET match TO True  
            FOR k FROM zero TO m MINUS one  
                IF the element at position k of pattern parameter EQUALS one AND the element at position i PLUS k PLUS one of nums parameter LESS THAN OR EQUAL TO the element at position i PLUS k of nums parameter  
                    SET match TO False  
                    BREAK out of the FOR loop  
                ELSE IF the element at position k of pattern parameter EQUALS zero AND the element at position i PLUS k PLUS one of nums parameter NOT EQUALS the element at position i PLUS k of nums parameter  
                    SET match TO False  
                    BREAK out of the FOR loop  
                ELSE IF the element at position k of pattern parameter EQUALS minus one AND the element at position i PLUS k PLUS one of nums parameter GREATER THAN OR EQUAL TO the element at position i PLUS k of nums parameter  
                    SET match TO False  
                    BREAK out of the FOR loop  
                END IF  
            END FOR  
            IF match EQUALS True  
                INCREMENT count BY one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
