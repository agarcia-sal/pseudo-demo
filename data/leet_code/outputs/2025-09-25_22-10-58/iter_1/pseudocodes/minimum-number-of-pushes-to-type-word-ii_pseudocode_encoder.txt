CLASS Solution  
    FUNCTION minimumPushes(word)  
        SET freq TO the result of counting the frequency of each letter in word  
        SET sorted_freq TO the list of the values of freq sorted in the order from largest to smallest  
        SET total_pushes TO zero  
        SET key_presses TO one  
        SET keys_assigned TO zero  
        FOR each count IN sorted_freq  
            INCREMENT total_pushes BY count MULTIPLIED BY key_presses  
            INCREMENT keys_assigned BY one  
            IF keys_assigned EQUALS eight  
                SET keys_assigned TO zero  
                INCREMENT key_presses BY one  
            END IF  
        END FOR  
        RETURN total_pushes  
    END FUNCTION  
END CLASS
