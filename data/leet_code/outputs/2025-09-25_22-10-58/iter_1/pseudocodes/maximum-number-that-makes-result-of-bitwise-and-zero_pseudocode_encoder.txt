CLASS Solution  
    FUNCTION maxNumber(n)  
        IF n EQUALS one  
            RETURN zero  
        END IF  
        SET highest_bit TO one  
        WHILE highest_bit LESS THAN OR EQUAL TO n  
            SET highest_bit TO highest_bit MULTIPLIED BY two  
        END WHILE  
        SET highest_bit TO highest_bit DIVIDED BY two  
        SET x TO highest_bit MINUS one  
        RETURN x  
    END FUNCTION  
END CLASS
