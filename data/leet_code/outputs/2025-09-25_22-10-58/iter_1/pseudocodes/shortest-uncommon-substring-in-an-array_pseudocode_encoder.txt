CLASS Solution  
    FUNCTION shortestSubstrings(arr)  
        SET substr_count TO a new dictionary with default value zero for each key  
        FOR each string s IN arr  
            SET length TO the length of string s  
            FOR index i FROM zero TO length MINUS one  
                FOR index j FROM i PLUS one TO length  
                    SET substr TO the substring of s from position i TO position j MINUS one  
                    INCREMENT the value associated with key substr IN substr_count BY one  
                END FOR  
            END FOR  
        END FOR  
        SET answer TO an empty list  
        FOR each string s IN arr  
            SET length TO the length of string s  
            SET shortest_unique_substr TO the empty string  
            FOR index i FROM zero TO length MINUS one  
                FOR index j FROM i PLUS one TO length  
                    SET substr TO the substring of s from position i TO position j MINUS one  
                    IF the value associated with key substr IN substr_count EQUALS one THEN  
                        IF shortest_unique_substr EQUALS the empty string OR  
                           the length of substr LESS THAN the length of shortest_unique_substr OR  
                           (the length of substr EQUALS the length of shortest_unique_substr AND substr LESS THAN shortest_unique_substr) THEN  
                            SET shortest_unique_substr TO substr  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
            APPEND shortest_unique_substr TO answer  
        END FOR  
        RETURN answer  
    END FUNCTION  
END CLASS
