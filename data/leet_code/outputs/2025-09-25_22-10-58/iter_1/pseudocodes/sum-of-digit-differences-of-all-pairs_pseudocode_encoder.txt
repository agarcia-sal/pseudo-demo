CLASS Solution  
    FUNCTION sumDigitDifferences(nums)  
        FUNCTION digit_difference(num1, num2)  
            SET count TO zero  
            FOR each pair of character d1 and character d2 FROM the characters of num1 AND the characters of num2 IN PARALLEL  
                IF character d1 NOT EQUALS character d2 THEN  
                    INCREMENT count BY one  
                END IF  
            END FOR  
            RETURN count  
        END FUNCTION  
        SET total_sum TO zero  
        SET n TO the LENGTH OF nums  
        FOR index i FROM zero TO n MINUS one  
            FOR index j FROM i PLUS one TO n MINUS one  
                SET total_sum TO total_sum PLUS digit_difference(element at position i of nums, element at position j of nums)  
            END FOR  
        END FOR  
        RETURN total_sum  
    END FUNCTION  
END CLASS
