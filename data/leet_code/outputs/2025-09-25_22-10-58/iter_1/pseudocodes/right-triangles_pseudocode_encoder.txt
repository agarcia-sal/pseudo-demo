CLASS Solution  
    FUNCTION numberOfRightTriangles(grid)  
        SET rows TO the LENGTH OF grid  
        SET cols TO the LENGTH OF element at position zero of grid  
        SET total_triangles TO zero  
        
        FOR i FROM zero TO rows MINUS one  
            FOR j FROM zero TO cols MINUS one  
                IF element at position i of grid THEN element at position j of element at position i of grid EQUALS one  
                    SET row_count TO zero  
                    FOR k FROM zero TO cols MINUS one  
                        IF k NOT EQUALS j THEN  
                            INCREMENT row_count BY element at position k of element at position i of grid  
                        END IF  
                    END FOR  
                    SET col_count TO zero  
                    FOR k FROM zero TO rows MINUS one  
                        IF k NOT EQUALS i THEN  
                            INCREMENT col_count BY element at position j of element at position k of grid  
                        END IF  
                    END FOR  
                    INCREMENT total_triangles BY row_count MULTIPLIED BY col_count  
                END IF  
            END FOR  
        END FOR  
        RETURN total_triangles  
    END FUNCTION  
END CLASS
