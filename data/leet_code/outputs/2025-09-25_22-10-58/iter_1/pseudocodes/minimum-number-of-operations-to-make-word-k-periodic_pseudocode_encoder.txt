CLASS Solution  
    FUNCTION minimumOperationsToMakeKPeriodic(word parameter k)  
        SET n TO the LENGTH OF word  
        SET segments TO an EMPTY LIST  
        FOR index FROM zero TO n MINUS one IN STEPS OF k  
            APPEND the substring from position index TO position index PLUS k MINUS one of word TO segments  
        END FOR  
        SET segment_count TO the COUNT OF EACH UNIQUE ELEMENT IN segments  
        SET max_count TO the HIGHEST OCCURRENCE COUNT AMONG THE UNIQUE ELEMENTS IN segment_count  
        SET result TO the LENGTH OF segments MINUS max_count  
        RETURN result  
    END FUNCTION  
END CLASS
