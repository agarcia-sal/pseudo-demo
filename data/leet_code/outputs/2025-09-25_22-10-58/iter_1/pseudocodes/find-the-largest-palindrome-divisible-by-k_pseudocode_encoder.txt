CLASS Solution  
    FUNCTION largestPalindrome(n k)  
        IF n EQUALS one  
            FOR x FROM nine DOWN TO one  
                IF x MODULO k EQUALS zero  
                    RETURN the string representation of x  
                END IF  
            END FOR  
            RETURN zero as a string  
        END IF  
        
        SET half TO the integer value of the concatenation of the digit nine repeated the integer division of the sum of n and one by two times  
        
        WHILE half GREATER THAN zero  
            IF n MODULO two EQUALS zero  
                SET full TO the integer value of the concatenation of the string representation of half and the reverse of the string representation of half  
            ELSE  
                SET full TO the integer value of the concatenation of the string representation of half and the reverse of the substring of the string representation of half from the first position to the position before the last position  
            END IF  
            
            IF full MODULO k EQUALS zero  
                RETURN the string representation of full  
            END IF  
            
            DECREMENT half BY one  
        END WHILE  
        
        RETURN zero as a string  
    END FUNCTION  
END CLASS
