CLASS Solution  
    FUNCTION valueAfterKSeconds(n k)  
        SET MOD TO one billion plus seven  
        SET a TO list of size n FILLED WITH one  
        FOR each index FROM one TO n MINUS one  
            CONTINUE  
        END FOR  
        FOR each iteration FROM zero TO k MINUS one  
            FOR each index FROM one TO n MINUS one  
                SET value at position index of a TO the value at position index of a PLUS the value at position index MINUS one of a MODULO MOD  
            END FOR  
        END FOR  
        RETURN the value at position n MINUS one of a  
    END FUNCTION  
END CLASS
