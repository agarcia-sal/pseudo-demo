CLASS Solution  
    FUNCTION minOrAfterOperations(nums k)  
        FUNCTION canAchieve(target_or k)  
            SET current_and TO minus one  
            SET operations TO zero  
            FOR each num IN nums  
                IF current_and EQUALS minus one  
                    SET current_and TO num  
                ELSE  
                    SET current_and TO current_and AND num  
                END IF  
                IF current_and AND target_or EQUALS zero  
                    SET current_and TO minus one  
                ELSE  
                    INCREMENT operations BY one  
                    IF operations GREATER THAN k  
                        RETURN false  
                    END IF  
                END IF  
            END FOR  
            RETURN true  
        END FUNCTION  
        SET max_possible_value TO two RAISED TO THE POWER OF thirty MINUS one  
        SET result TO max_possible_value  
        FOR bit FROM zero TO twenty nine  
            SET bit_mask TO two RAISED TO THE POWER OF bit  
            IF result AND bit_mask EQUALS zero  
                CONTINUE TO next iteration of loop  
            END IF  
            IF canAchieve(LOGICAL NOT OF result XOR bit_mask k)  
                SET result TO result AND LOGICAL NOT OF bit_mask  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
