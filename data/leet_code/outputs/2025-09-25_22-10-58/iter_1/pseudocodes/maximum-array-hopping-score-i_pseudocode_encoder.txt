CLASS Solution  
    FUNCTION maxScore(nums)  
        SET n TO the LENGTH OF nums  
        SET dp TO a list of zeroes of SIZE n  
        SET element at position one of dp TO zero  
        FOR i FROM two TO n  
            FOR j FROM one TO i MINUS one  
                IF the value of element at position i of dp LESS THAN the value of element at position j of dp PLUS the difference of i MINUS j MULTIPLIED BY the element at position i of nums  
                    SET element at position i of dp TO the value of element at position j of dp PLUS the difference of i MINUS j MULTIPLIED BY the element at position i of nums  
                END IF  
            END FOR  
        END FOR  
        RETURN the value of element at position n MINUS one of dp  
    END FUNCTION  
END CLASS
