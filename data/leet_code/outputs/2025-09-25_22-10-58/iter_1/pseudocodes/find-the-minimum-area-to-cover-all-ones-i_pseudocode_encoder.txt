CLASS Solution  
    FUNCTION minimumArea(grid)  
        IF grid EQUALS empty OR element at position zero of grid EQUALS empty  
            RETURN zero  
        END IF  
        SET rows TO the LENGTH OF grid  
        SET cols TO the LENGTH OF element at position zero of grid  
        SET min_row TO positive infinity  
        SET max_row TO negative infinity  
        SET min_col TO positive infinity  
        SET max_col TO negative infinity  
        FOR i FROM zero TO rows MINUS one  
            FOR j FROM zero TO cols MINUS one  
                IF element at position i of grid AT position j EQUALS one  
                    IF min_row GREATER THAN i  
                        SET min_row TO i  
                    END IF  
                    IF max_row LESS THAN i  
                        SET max_row TO i  
                    END IF  
                    IF min_col GREATER THAN j  
                        SET min_col TO j  
                    END IF  
                    IF max_col LESS THAN j  
                        SET max_col TO j  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        SET height TO max_row MINUS min_row PLUS one  
        SET width TO max_col MINUS min_col PLUS one  
        RETURN height MULTIPLIED BY width  
    END FUNCTION  
END CLASS
