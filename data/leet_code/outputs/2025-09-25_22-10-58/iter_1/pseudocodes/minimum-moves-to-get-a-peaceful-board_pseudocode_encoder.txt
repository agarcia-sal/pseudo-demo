CLASS Solution  
    FUNCTION minMoves(rooks)  
        SET n TO the LENGTH OF rooks  
        SET rooks_by_row TO rooks SORTED BY the first element of each element  
        SET rooks_by_col TO rooks SORTED BY the second element of each element  
        SET row_moves TO zero  
        FOR index FROM zero TO n MINUS one  
            SET difference TO the absolute value of the first element of the element at position index of rooks_by_row MINUS index  
            INCREMENT row_moves BY difference  
        END FOR  
        SET col_moves TO zero  
        FOR index FROM zero TO n MINUS one  
            SET difference TO the absolute value of the second element of the element at position index of rooks_by_col MINUS index  
            INCREMENT col_moves BY difference  
        END FOR  
        RETURN row_moves PLUS col_moves  
    END FUNCTION  
END CLASS
