CLASS Solution  
    FUNCTION countSubstrings(s)  
        SET n TO the LENGTH OF s  
        SET total_substrings TO zero  
        FOR i FROM zero TO the VALUE OF n MINUS one  
            SET current_number TO zero  
            FOR j FROM i TO the VALUE OF n MINUS one  
                SET current_number TO current_number MULTIPLIED BY ten PLUS the INTEGER VALUE OF element at position j of s  
                SET last_digit TO the INTEGER VALUE OF element at position j of s  
                IF last_digit NOT EQUALS zero AND current_number MODULO last_digit EQUALS zero  
                    INCREMENT total_substrings BY one  
                END IF  
            END FOR  
        END FOR  
        RETURN total_substrings  
    END FUNCTION  
END CLASS
