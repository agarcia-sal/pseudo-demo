CLASS Solution  
    FUNCTION validStrings n  
        IF n EQUALS one  
            RETURN list containing string zero and string one  
        END IF  
        
        FUNCTION backtrack current  
            IF the LENGTH OF current EQUALS n  
                APPEND current TO valid_strings  
                RETURN  
            END IF  
            IF element at position from the END one of current EQUALS string one  
                CALL backtrack WITH argument current PLUS string zero  
                CALL backtrack WITH argument current PLUS string one  
            ELSE IF element at position from the END one of current EQUALS string zero  
                CALL backtrack WITH argument current PLUS string one  
            END IF  
        END FUNCTION  
        
        SET valid_strings TO empty list  
        CALL backtrack WITH argument string zero  
        CALL backtrack WITH argument string one  
        
        RETURN valid_strings  
    END FUNCTION  
END CLASS
