CLASS Solution  
    FUNCTION queryResults(limit parameter, queries parameter)  
        SET ball_colors TO an empty dictionary mapping identifiers to colors  
        SET unique_colors TO an empty set containing distinct colors  
        SET result TO an empty list for storing counts of unique colors  
        
        FOR each pair of values x parameter and y parameter IN queries parameter  
            IF x parameter EXISTS as a key IN ball_colors dictionary  
                SET old_color TO the value associated with the key x parameter in ball_colors dictionary  
                IF old_color EXISTS IN unique_colors set  
                    REMOVE old_color FROM unique_colors set  
                END IF  
            END IF  
            
            SET the value associated with the key x parameter IN ball_colors dictionary TO y parameter  
            ADD y parameter TO unique_colors set  
            APPEND the number of elements IN unique_colors set TO result list  
        END FOR  
        
        RETURN result list  
    END FUNCTION  
END CLASS
