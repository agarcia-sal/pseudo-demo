CLASS Solution  
    FUNCTION waysToReachStair WITH PARAMETER k  
        FUNCTION dfs WITH PARAMETERS i j jump  
            IF i GREATER THAN k PLUS one  
                RETURN zero  
            END IF  
            SET ans TO the value of the boolean expression i EQUALS k converted to an integer  
            IF i GREATER THAN zero AND j EQUALS zero  
                INCREMENT ans BY the result of dfs WITH PARAMETERS i MINUS one j SET TO one jump  
            END IF  
            INCREMENT ans BY the result of dfs WITH PARAMETERS i PLUS one MULTIPLIED BY two MULTIPLIED BY jump j SET TO zero jump PLUS one  
            RETURN ans  
        END FUNCTION  
        RETURN the result of dfs WITH PARAMETERS one zero zero  
    END FUNCTION  
END CLASS
