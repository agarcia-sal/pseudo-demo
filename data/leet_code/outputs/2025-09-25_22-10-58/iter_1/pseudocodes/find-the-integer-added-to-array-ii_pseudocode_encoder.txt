CLASS Solution  
    FUNCTION minimumAddedInteger(nums1 RETURNS List OF integer, nums2 RETURNS List OF integer) RETURNS integer  
        CALL sort ON nums1  
        CALL sort ON nums2  
        
        FOR i FROM zero TO the LENGTH OF nums1 MINUS one  
            FOR j FROM i PLUS one TO the LENGTH OF nums1 MINUS one  
                SET new_nums1 TO an empty list  
                FOR index FROM zero TO i MINUS one  
                    APPEND element at position index of nums1 TO new_nums1  
                END FOR  
                FOR index FROM i PLUS one TO j MINUS one  
                    APPEND element at position index of nums1 TO new_nums1  
                END FOR  
                FOR index FROM j PLUS one TO the LENGTH OF nums1 MINUS one  
                    APPEND element at position index of nums1 TO new_nums1  
                END FOR  
                
                SET x TO element at position zero of nums2 MINUS element at position zero of new_nums1  
                
                SET condition_holds TO TRUE  
                FOR k FROM zero TO the LENGTH OF nums2 MINUS one  
                    IF element at position k of new_nums1 PLUS x NOT EQUALS element at position k of nums2  
                        SET condition_holds TO FALSE  
                        BREAK out of the FOR loop  
                    END IF  
                END FOR  
                
                IF condition_holds EQUALS TRUE  
                    RETURN x  
                END IF  
            END FOR  
        END FOR  
        
        RETURN None  
    END FUNCTION  
END CLASS
