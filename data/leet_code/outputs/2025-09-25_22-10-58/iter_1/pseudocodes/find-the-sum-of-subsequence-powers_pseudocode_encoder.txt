CLASS Solution  
    FUNCTION sumOfPowers(nums parameter LIST OF INTEGERS k parameter INTEGER) RETURNS INTEGER  
        SET MOD TO ten RAISED TO THE power OF nine PLUS seven  
        SET total_sum TO zero  
        FOR each combo IN all combinations OF nums TAKING k elements  
            SET minimum_difference TO a very large number  
            FOR index i FROM zero TO k MINUS one  
                FOR index j FROM i PLUS one TO k MINUS one  
                    SET current_difference TO the absolute value OF element at position i OF combo MINUS element at position j OF combo  
                    IF current_difference LESS THAN minimum_difference THEN  
                        SET minimum_difference TO current_difference  
                    END IF  
                END FOR  
            END FOR  
            SET total_sum TO total_sum PLUS minimum_difference  
            SET total_sum TO total_sum MODULO MOD  
        END FOR  
        RETURN total_sum  
    END FUNCTION  
END CLASS
