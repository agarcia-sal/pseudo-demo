CLASS Solution  
    FUNCTION maxTotalReward(rewardValues)  
        FUNCTION dfs(x)  
            SET i TO position where value x would be inserted to maintain sorted order in rewardValues using bisect_right  
            SET ans TO zero  
            FOR each element v IN rewardValues starting from position i to the end  
                IF the sum of x and v is GREATER THAN x  
                    SET ans TO the maximum of ans and the sum of v and the result of calling dfs with the argument as the sum of x and v  
                END IF  
            END FOR  
            RETURN ans  
        END FUNCTION  

        SORT rewardValues in ascending order  
        RETURN the result of calling dfs with argument zero  
    END FUNCTION  
END CLASS
