CLASS Solution  
    FUNCTION minOperations(nums)  
        SET n TO the LENGTH OF nums  
        IF n EQUALS zero  
            RETURN zero  
        END IF  
        SET dp TO a list of size n where each element is one  
        FOR i FROM one TO n MINUS one  
            FOR j FROM zero TO i MINUS one  
                IF element at position i of nums LESS THAN OR EQUAL TO element at position j of nums  
                    SET dp at position i TO the GREATER OF dp at position i AND dp at position j PLUS one  
                END IF  
            END FOR  
        END FOR  
        RETURN the MAXIMUM value in dp  
    END FUNCTION  
END CLASS
