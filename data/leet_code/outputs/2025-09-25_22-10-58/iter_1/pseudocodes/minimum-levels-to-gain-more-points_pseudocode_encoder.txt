CLASS Solution  
    FUNCTION minimumLevels(possible)  
        SET total_points TO zero  
        FOR each number IN possible  
            INCREMENT total_points BY two MULTIPLIED BY number MINUS one  
        END FOR  
        SET alice_points TO zero  
        FOR index FROM zero TO the LENGTH OF possible MINUS one MINUS one  
            INCREMENT alice_points BY two MULTIPLIED BY element at position index of possible MINUS one  
            DECREMENT total_points BY two MULTIPLIED BY element at position index of possible MINUS one  
            IF alice_points GREATER THAN total_points  
                RETURN index PLUS one  
            END IF  
        END FOR  
        RETURN minus one  
    END FUNCTION  
END CLASS
