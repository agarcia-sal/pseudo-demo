CLASS Solution  
    FUNCTION numberOfSubstrings(s)  
        SET n TO the LENGTH OF s  
        SET count TO zero  
        FOR start FROM zero TO n MINUS one  
            SET ones TO zero  
            SET zeros TO zero  
            FOR end FROM start TO n MINUS one  
                IF element at position end of s EQUALS the character one  
                    INCREMENT ones BY one  
                ELSE  
                    INCREMENT zeros BY one  
                END IF  
                IF ones GREATER THAN OR EQUAL TO zeros MULTIPLIED BY zeros  
                    INCREMENT count BY one  
                END IF  
            END FOR  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
