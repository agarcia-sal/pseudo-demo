CLASS Solution  
    FUNCTION minimumTimeToInitialState(word PARAMETER k)  
        SET n TO the LENGTH OF word  
        SET time TO one  
        WHILE true holds  
            IF time MULTIPLIED BY k GREATER THAN OR EQUAL TO n OR the SUBSTRING of word from POSITION time MULTIPLIED BY k TO the END EQUALS the SUBSTRING of word from POSITION one TO POSITION n MINUS time MULTIPLIED BY k THEN  
                RETURN time  
            END IF  
            INCREMENT time BY one  
        END WHILE  
    END FUNCTION  
END CLASS
