CLASS Solution  
    FUNCTION longestCommonPrefix(arr1 arr1 OF integers arr2 arr2 OF integers) RETURNS integer  
        SET prefixes1 TO an empty set  
        FOR each num IN arr1  
            SET num_str TO the string representation of num  
            FOR i FROM one TO the LENGTH OF num_str  
                ADD the substring from position one TO position i of num_str TO prefixes1  
            END FOR  
        END FOR  
        
        SET prefixes2 TO an empty set  
        FOR each num IN arr2  
            SET num_str TO the string representation of num  
            FOR i FROM one TO the LENGTH OF num_str  
                ADD the substring from position one TO position i of num_str TO prefixes2  
            END FOR  
        END FOR  
        
        SET longest_common_length TO zero  
        FOR each prefix IN prefixes1  
            IF prefix IS A MEMBER OF prefixes2 THEN  
                IF the LENGTH OF prefix GREATER THAN longest_common_length THEN  
                    SET longest_common_length TO the LENGTH OF prefix  
                END IF  
            END IF  
        END FOR  
        
        RETURN longest_common_length  
    END FUNCTION  
END CLASS
