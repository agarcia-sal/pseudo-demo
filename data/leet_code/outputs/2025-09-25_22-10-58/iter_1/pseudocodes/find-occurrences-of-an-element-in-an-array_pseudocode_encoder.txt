CLASS Solution  
    FUNCTION occurrencesOfElement(nums, queries, x)  
        SET occurrences TO empty list  
        FOR each index AND num IN enumeration of nums  
            IF num EQUALS x  
                APPEND index TO occurrences  
            END IF  
        END FOR  
        SET answer TO empty list  
        FOR each query IN queries  
            IF query LESS THAN OR EQUAL TO the LENGTH OF occurrences  
                APPEND element at position query MINUS one OF occurrences TO answer  
            ELSE  
                APPEND negative one TO answer  
            END IF  
        END FOR  
        RETURN answer  
    END FUNCTION  
END CLASS
