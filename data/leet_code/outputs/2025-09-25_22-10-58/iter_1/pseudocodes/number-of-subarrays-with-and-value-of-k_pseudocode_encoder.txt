CLASS Solution  
    FUNCTION countSubarrays(nums parameter LIST OF INTEGER k parameter INTEGER) RETURNS INTEGER  
        SET n TO the LENGTH OF nums  
        SET count TO zero  
        FOR start FROM zero TO n MINUS one  
            SET current_and TO the element at position start of nums  
            FOR end FROM start TO n MINUS one  
                SET current_and TO current_and BITWISE AND element at position end of nums  
                IF current_and EQUALS k THEN  
                    INCREMENT count BY one  
                END IF  
                IF current_and LESS THAN k THEN  
                    BREAK THE INNER LOOP  
                END IF  
            END FOR  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
