CLASS Solution  
    FUNCTION maximumHappinessSum(happiness, k)  
        SORT the collection happiness in order from greatest TO least  
        SET max_sum TO zero  
        SET decrement TO zero  
        FOR index i FROM zero TO the value of k MINUS one  
            SET current_value TO element at position i of the collection happiness MINUS decrement  
            IF current_value LESS THAN zero  
                SET current_value TO zero  
            END IF  
            INCREMENT max_sum BY current_value  
            INCREMENT decrement BY one  
        END FOR  
        RETURN max_sum  
    END FUNCTION  
END CLASS
