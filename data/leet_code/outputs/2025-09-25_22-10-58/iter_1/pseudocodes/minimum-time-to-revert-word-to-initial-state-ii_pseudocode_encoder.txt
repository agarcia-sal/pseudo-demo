CLASS Solution  
    FUNCTION minimumTimeToInitialState(word parameter word AND integer parameter k) RETURNS integer  
        SET n TO the LENGTH OF word  
        SET time TO one  
        WHILE true holds  
            SET prefix_to_match TO the SUBSTRING of word STARTING AT the position time MULTIPLIED BY k TO THE END  
            IF the word STARTS WITH prefix_to_match THEN  
                RETURN time  
            END IF  
            INCREMENT time BY one  
        END WHILE  
    END FUNCTION  
END CLASS
