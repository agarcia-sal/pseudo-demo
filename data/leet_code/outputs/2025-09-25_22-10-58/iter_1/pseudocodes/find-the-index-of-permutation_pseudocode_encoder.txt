CLASS Solution  
    FUNCTION getPermutationIndex(perm)  
        SET n TO the LENGTH OF perm  
        SET MOD TO ten RAISED TO THE POWER OF nine PLUS one  
        
        SET fact TO a LIST of LENGTH n FILLED WITH one  
        FOR i FROM two TO n MINUS one  
            SET element at position i of fact TO element at position i MINUS one of fact MULTIPLIED BY i  
        END FOR  
        
        SET numbers TO a LIST of INTEGERS from one TO n  
        
        SET index TO zero  
        FOR i FROM zero TO n MINUS one  
            SET pos TO the POSITION OF element at position i of perm IN numbers  
            
            INCREMENT index BY pos MULTIPLIED BY element at position n MINUS i MINUS one of fact  
            
            REMOVE element at position pos FROM numbers  
        END FOR  
        
        RETURN index MODULO MOD  
    END FUNCTION  
END CLASS
