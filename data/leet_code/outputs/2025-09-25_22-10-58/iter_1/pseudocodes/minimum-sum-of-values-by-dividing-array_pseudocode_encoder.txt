CLASS Solution  
    FUNCTION minimumValueSum(nums, andValues)  
        SET n TO the LENGTH OF nums  
        SET m TO the LENGTH OF andValues  
        
        FUNCTION dp(i, j)  
            IF j EQUALS negative one THEN  
                IF i EQUALS negative one THEN  
                    RETURN zero  
                ELSE  
                    RETURN infinity  
                END IF  
            END IF  
            
            IF i EQUALS negative one THEN  
                RETURN infinity  
            END IF  
            
            SET min_sum TO infinity  
            SET current_and TO negative one  
            FOR k FROM i DOWN TO negative one INCLUSIVE  
                IF current_and EQUALS negative one THEN  
                    SET current_and TO element at position k of nums  
                ELSE  
                    SET current_and TO current_and AND element at position k of nums  
                END IF  
                
                IF current_and EQUALS element at position j of andValues THEN  
                    SET min_sum TO the MINIMUM OF min_sum AND dp(k MINUS one, j MINUS one) PLUS element at position i of nums  
                END IF  
            END FOR  
            
            RETURN min_sum  
        END FUNCTION  
        
        SET result TO dp(n MINUS one, m MINUS one)  
        IF result NOT EQUALS infinity THEN  
            RETURN result  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
