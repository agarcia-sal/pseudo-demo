CLASS Solution  
    FUNCTION minCostGoodCaption(caption)  
        SET n TO the LENGTH OF caption  
        IF n LESS THAN three  
            RETURN empty string  
        END IF  
        
        SET caption TO the LIST OF characters IN caption  
        
        SET i TO zero  
        WHILE i LESS THAN n  
            SET start TO i  
            WHILE i LESS THAN n AND the element at position i of caption EQUALS the element at position start of caption  
                INCREMENT i BY one  
            END WHILE  
            
            SET length TO i MINUS start  
            
            IF length LESS THAN three  
                IF start GREATER THAN zero AND the element at position start MINUS one of caption EQUALS the element at position start of caption  
                    SET the element at position start MINUS one of caption TO the element at position start of caption  
                    DECREMENT start BY one  
                    INCREMENT length BY one  
                END IF  
                IF i LESS THAN n AND the element at position i MINUS one of caption EQUALS the element at position i of caption  
                    SET the element at position i of caption TO the element at position i MINUS one of caption  
                    INCREMENT i BY one  
                    INCREMENT length BY one  
                END IF  
                
                IF length LESS THAN three  
                    IF start GREATER THAN zero  
                        SET target_char TO the element at position start MINUS one of caption  
                    ELSE  
                        SET target_char TO the element at position i of caption  
                    END IF  
                    
                    IF target_char EQUALS the character a  
                        SET target_char TO the character b  
                    ELSE IF target_char EQUALS the character z  
                        SET target_char TO the character y  
                    ELSE  
                        SET target_char TO the character that follows target_char in the alphabetical order  
                    END IF  
                    
                    FOR j FROM start TO start PLUS length MINUS one  
                        SET the element at position j of caption TO target_char  
                    END FOR  
                    
                    SET i TO start PLUS length  
                END IF  
            END IF  
        END WHILE  
        
        RETURN the concatenation of elements of caption  
    END FUNCTION  
END CLASS
