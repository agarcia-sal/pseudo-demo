CLASS Solution  
    FUNCTION maxOperations(s)  
        SET ans TO zero  
        SET cnt TO zero  
        FOR each index i AND character c IN the enumeration of s  
            IF character c EQUALS the character one  
                INCREMENT cnt BY one  
            ELSE IF index i IS NOT zero AND the character at position index i MINUS one of s EQUALS the character one  
                INCREMENT ans BY cnt  
            END IF  
        END FOR  
        RETURN ans  
    END FUNCTION  
END CLASS
