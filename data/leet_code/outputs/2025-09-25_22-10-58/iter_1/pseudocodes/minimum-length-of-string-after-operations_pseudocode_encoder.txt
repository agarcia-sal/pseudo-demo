CLASS Solution  
    FUNCTION minimumLength(s)  
        SET cnt TO result of calling Counter on s  
        SET sum_for_odd TO zero  
        SET sum_for_even TO zero  
        FOR each x IN the collection of values in cnt  
            IF the result of x MODULO two EQUALS one  
                INCREMENT sum_for_odd BY one  
            ELSE IF the result of x MODULO two EQUALS zero AND x NOT EQUALS zero  
                INCREMENT sum_for_even BY two  
            END IF  
        END FOR  
        SET result TO sum_for_odd PLUS sum_for_even  
        RETURN result  
    END FUNCTION  
END CLASS
