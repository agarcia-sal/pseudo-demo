CLASS Solution  
    FUNCTION maximumLength(nums)  
        SET even_length TO zero  
        SET odd_length TO zero  
        FOR i FROM one TO the LENGTH OF nums MINUS one  
            SET current_sum TO element at position i MINUS one of nums PLUS element at position i of nums  
            IF current_sum MODULO two EQUALS zero  
                SET even_length TO the GREATER OF even_length PLUS one AND odd_length  
            ELSE  
                SET odd_length TO the GREATER OF odd_length PLUS one AND even_length  
            END IF  
        END FOR  
        SET result TO the GREATER OF even_length AND odd_length PLUS one  
        RETURN result  
    END FUNCTION  
END CLASS
