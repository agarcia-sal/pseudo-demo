CLASS Solution  
    FUNCTION compressedString(word)  
        SET comp TO empty list  
        SET i TO zero  
        WHILE i LESS THAN the LENGTH OF word  
            SET c TO element at position i of word  
            SET count TO zero  
            WHILE i LESS THAN the LENGTH OF word AND element at position i of word EQUALS c  
                INCREMENT count BY one  
                INCREMENT i BY one  
                IF count EQUALS nine  
                    BREAK  
                END IF  
            END WHILE  
            APPEND concatenation of count AS string AND c TO comp  
        END WHILE  
        SET result TO concatenation of all elements in comp SEQUENTIALLY  
        RETURN result  
    END FUNCTION  
END CLASS
