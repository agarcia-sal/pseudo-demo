CLASS Solution  
    FUNCTION countOfPairs(n parameter, x parameter, y parameter) RETURNS list of integer  
        SET result TO a list of zeros with the length equal to n  

        FUNCTION bfs(start parameter)  
            SET visited TO a list of boolean values False with the length equal to n PLUS one  
            SET distance TO a list of zeros with the length equal to n PLUS one  
            SET queue TO a double ended queue containing the start parameter only  
            SET the element at position start parameter of visited TO True  
            
            WHILE the queue is not empty  
                SET current TO the element removed from the front of queue  
                
                FOR each neighbor IN the list containing current MINUS one and current PLUS one  
                    IF the neighbor is greater than or equal to one AND less than or equal to n AND the element at position neighbor of visited EQUALS False  
                        SET the element at position neighbor of visited TO True  
                        SET the element at position neighbor of distance TO the element at position current of distance PLUS one  
                        APPEND neighbor TO the end of queue  
                    END IF  
                END FOR  
                
                IF current EQUALS x AND the element at position y of visited EQUALS False  
                    SET the element at position y of visited TO True  
                    SET the element at position y of distance TO the element at position current of distance PLUS one  
                    APPEND y TO the end of queue  
                END IF  
                
                IF current EQUALS y AND the element at position x of visited EQUALS False  
                    SET the element at position x of visited TO True  
                    SET the element at position x of distance TO the element at position current of distance PLUS one  
                    APPEND x TO the end of queue  
                END IF  
            END WHILE  
            
            FOR dist FROM one TO n  
                IF the element at position dist of distance GREATER THAN zero  
                    INCREMENT the element at position distance at position dist MINUS one of result BY one  
                END IF  
            END FOR  
        END FUNCTION  

        FOR i FROM one TO n  
            CALL bfs WITH i  
        END FOR  

        RETURN result  
    END FUNCTION  
END CLASS
